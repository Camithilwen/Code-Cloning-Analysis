131def test_estimator():
132 omp = OrthogonalMatchingPursuit(n_nonzero_coefs=n_nonzero_coefs)
133 omp.fit(X, y[:, 0])
134 assert omp.coef_.shape == (n_features,)
135 assert omp.intercept_.shape == ()
136 assert np.count_nonzero(omp.coef_) <= n_nonzero_coefs

139 assert omp.coef_.shape == (n_targets, n_features)
140 assert omp.intercept_.shape == (n_targets,)
141 assert np.count_nonzero(omp.coef_) <= n_targets * n_nonzero_coefs
142
143 coef_normalized = omp.coef_[0].copy()
144 omp.set_params(fit_intercept=True)
145 omp.fit(X, y[:, 0])
146 assert_array_almost_equal(coef_normalized, omp.coef_)
147
148 omp.set_params(fit_intercept=False)
149 omp.fit(X, y[:, 0])
150 assert np.count_nonzero(omp.coef_) <= n_nonzero_coefs
151 assert omp.coef_.shape == (n_features,)
152 assert omp.intercept_ == 0
153
154 omp.fit(X, y)
155 assert omp.coef_.shape == (n_targets, n_features)
156 assert omp.intercept_ == 0
157 assert np.count_nonzero(omp.coef_) <= n_targets * n_nonzero_coefs

186def test_swapped_regressors():
187 gamma = np.zeros(n_features)
188 # X[:, 21] should be selected first, then X[:, 0] selected second,
189 # which will take X[:, 21]'s place in case the algorithm does
190 # column swapping for optimization (which is the case at the moment)
191 gamma[21] = 1.0
192 gamma[0] = 0.5
193 new_y = np.dot(X, gamma)
194 new_Xy = np.dot(X.T, new_y)
195 gamma_hat = orthogonal_mp(X, new_y, n_nonzero_coefs=2)
196 gamma_hat_gram = orthogonal_mp_gram(G, new_Xy, n_nonzero_coefs=2)
197 assert_array_equal(np.flatnonzero(gamma_hat), [0, 21])
198 assert_array_equal(np.flatnonzero(gamma_hat_gram), [0, 21])
