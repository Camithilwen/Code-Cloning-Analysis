16def _move_from_center(coord, centers, deltas, axmask=(True, True, True)):
17 """
18 For each coordinate where *axmask* is True, move *coord* away from
19 *centers* by *deltas*.
20 """
21 coord = np.asarray(coord)
22 return coord + axmask * np.copysign(1, coord - centers) * deltas

269 def _get_coord_info(self):
270 mins, maxs = np.array([
271 self.axes.get_xbound(),
272 self.axes.get_ybound(),
273 self.axes.get_zbound(),
274 ]).T
275
276 # Project the bounds along the current position of the cube:
277 bounds = mins[0], maxs[0], mins[1], maxs[1], mins[2], maxs[2]
278 bounds_proj = self.axes._transformed_cube(bounds)
279
280 # Determine which one of the parallel planes are higher up:
281 means_z0 = np.zeros(3)
282 means_z1 = np.zeros(3)
283 for i in range(3):
284 means_z0[i] = np.mean(bounds_proj[self._PLANES[2 * i], 2])
285 means_z1[i] = np.mean(bounds_proj[self._PLANES[2 * i + 1], 2])
286 highs = means_z0 < means_z1
287
288 # Special handling for edge-on views
289 equals = np.abs(means_z0 - means_z1) <= np.finfo(float).eps
290 if np.sum(equals) == 2:
291 vertical = np.where(~equals)[0][0]
292 if vertical == 2: # looking at XY plane
293 highs = np.array([True, True, highs[2]])
294 elif vertical == 1: # looking at XZ plane
295 highs = np.array([True, highs[1], False])
296 elif vertical == 0: # looking at YZ plane
297 highs = np.array([highs[0], False, False])
298
299 return mins, maxs, bounds_proj, highs

301 def _calc_centers_deltas(self, maxs, mins):
302 centers = 0.5 * (maxs + mins)
303 # In mpl3.8, the scale factor was 1/12. mpl3.9 changes this to
304 # 1/12 * 24/25 = 0.08 to compensate for the change in automargin
305 # behavior and keep appearance the same. The 24/25 factor is from the
306 # 1/48 padding added to each side of the axis in mpl3.8.
307 scale = 0.08
308 deltas = (maxs - mins) * scale
309 return centers, deltas

311 def _get_axis_line_edge_points(self, minmax, maxmin, position=None):
312 """Get the edge points for the black bolded axis line."""
313 # When changing vertical axis some of the axes has to be
314 # moved to the other plane so it looks the same as if the z-axis
315 # was the vertical axis.
316 mb = [minmax, maxmin] # line from origin to nearest corner to camera
317 mb_rev = mb[::-1]
318 mm = [[mb, mb_rev, mb_rev], [mb_rev, mb_rev, mb], [mb, mb, mb]]
319 mm = mm[self.axes._vertical_axis][self._axinfo["i"]]
320
321 juggled = self._axinfo["juggled"]
322 edge_point_0 = mm[0].copy() # origin point
323
324 if ((position == 'lower' and mm[1][juggled[-1]] < mm[0][juggled[-1]]) or
325 (position == 'upper' and mm[1][juggled[-1]] > mm[0][juggled[-1]])):
326 edge_point_0[juggled[-1]] = mm[1][juggled[-1]]
327 else:
328 edge_point_0[juggled[0]] = mm[1][juggled[0]]
329
330 edge_point_1 = edge_point_0.copy()
331 edge_point_1[juggled[1]] = mm[1][juggled[1]]
332
333 return edge_point_0, edge_point_1

360 def _get_tickdir(self, position):
361 """
362 Get the direction of the tick.
363
364 Parameters
365 ----------
366 position : str, optional : {'upper', 'lower', 'default'}
367 The position of the axis.
368
369 Returns
370 -------
371 tickdir : int
372 Index which indicates which coordinate the tick line will
373 align with.
374 """
375 _api.check_in_list(('upper', 'lower', 'default'), position=position)
376
377 # TODO: Move somewhere else where it's triggered less:
378 tickdirs_base = [v["tickdir"] for v in self._AXINFO.values()] # default
379 elev_mod = np.mod(self.axes.elev + 180, 360) - 180
380 azim_mod = np.mod(self.axes.azim, 360)
381 if position == 'upper':
382 if elev_mod >= 0:
383 tickdirs_base = [2, 2, 0]
384 else:
385 tickdirs_base = [1, 0, 0]
386 if 0 <= azim_mod < 180:
387 tickdirs_base[2] = 1
388 elif position == 'lower':
389 if elev_mod >= 0:
390 tickdirs_base = [1, 0, 1]
391 else:
392 tickdirs_base = [2, 2, 1]
393 if 0 <= azim_mod < 180:
394 tickdirs_base[2] = 0
395 info_i = [v["i"] for v in self._AXINFO.values()]
396
397 i = self._axinfo["i"]
398 vert_ax = self.axes._vertical_axis
399 j = vert_ax - 2
400 # default: tickdir = [[1, 2, 1], [2, 2, 0], [1, 0, 0]][vert_ax][i]
401 tickdir = np.roll(info_i, -j)[np.roll(tickdirs_base, j)][i]
402 return tickdir

404 def active_pane(self):
405 mins, maxs, tc, highs = self._get_coord_info()

407 index = info['i']

436 def _draw_ticks(self, renderer, edgep1, centers, deltas, highs,
437 deltas_per_point, pos):
438 ticks = self._update_ticks()
439 info = self._axinfo
440 index = info["i"]
441 juggled = info["juggled"]
442
443 mins, maxs, tc, highs = self._get_coord_info()
444 centers, deltas = self._calc_centers_deltas(maxs, mins)
445
446 # Draw ticks:
447 tickdir = self._get_tickdir(pos)
448 tickdelta = deltas[tickdir] if highs[tickdir] else -deltas[tickdir]
449
450 tick_info = info['tick']
451 tick_out = tick_info['outward_factor'] * tickdelta
452 tick_in = tick_info['inward_factor'] * tickdelta
453 tick_lw = tick_info['linewidth']
454 edgep1_tickdir = edgep1[tickdir]
455 out_tickdir = edgep1_tickdir + tick_out
456 in_tickdir = edgep1_tickdir - tick_in
457
458 default_label_offset = 8. # A rough estimate
459 points = deltas_per_point * deltas
460 for tick in ticks:
461 # Get tick line positions
462 pos = edgep1.copy()
463 pos[index] = tick.get_loc()
464 pos[tickdir] = out_tickdir
465 x1, y1, z1 = proj3d.proj_transform(*pos, self.axes.M)
466 pos[tickdir] = in_tickdir
467 x2, y2, z2 = proj3d.proj_transform(*pos, self.axes.M)
468
469 # Get position of label
470 labeldeltas = (tick.get_pad() + default_label_offset) * points
471
472 pos[tickdir] = edgep1_tickdir
473 pos = _move_from_center(pos, centers, labeldeltas, self._axmask())
474 lx, ly, lz = proj3d.proj_transform(*pos, self.axes.M)
475
476 _tick_update_position(tick, (x1, x2), (y1, y2), (lx, ly))
477 tick.tick1line.set_linewidth(tick_lw[tick._major])
478 tick.draw(renderer)

480 def _draw_offset_text(self, renderer, edgep1, edgep2, labeldeltas, centers,
481 highs, pep, dx, dy):
482 # Get general axis information:
483 info = self._axinfo
484 index = info["i"]
485 juggled = info["juggled"]
486 tickdir = info["tickdir"]
487
488 # Which of the two edge points do we want to
489 # use for locating the offset text?
490 if juggled[2] == 2:
491 outeredgep = edgep1
492 outerindex = 0
493 else:
494 outeredgep = edgep2
495 outerindex = 1
496
497 pos = _move_from_center(outeredgep, centers, labeldeltas,
498 self._axmask())
499 olx, oly, olz = proj3d.proj_transform(*pos, self.axes.M)
500 self.offsetText.set_text(self.major.formatter.get_offset())
501 self.offsetText.set_position((olx, oly))
502 angle = art3d._norm_text_angle(np.rad2deg(np.arctan2(dy, dx)))
503 self.offsetText.set_rotation(angle)
504 # Must set rotation mode to "anchor" so that
505 # the alignment point is used as the "fulcrum" for rotation.
506 self.offsetText.set_rotation_mode('anchor')
507
508 # ----------------------------------------------------------------------
509 # Note: the following statement for determining the proper alignment of
510 # the offset text. This was determined entirely by trial-and-error
511 # and should not be in any way considered as "the way". There are
512 # still some edge cases where alignment is not quite right, but this
513 # seems to be more of a geometry issue (in other words, I might be
514 # using the wrong reference points).
515 #
516 # (TT, FF, TF, FT) are the shorthand for the tuple of
517 # (centpt[tickdir] <= pep[tickdir, outerindex],
518 # centpt[index] <= pep[index, outerindex])
519 #
520 # Three-letters (e.g., TFT, FTT) are short-hand for the array of bools
521 # from the variable 'highs'.
522 # ---------------------------------------------------------------------
523 centpt = proj3d.proj_transform(*centers, self.axes.M)
524 if centpt[tickdir] > pep[tickdir, outerindex]:
525 # if FT and if highs has an even number of Trues
526 if (centpt[index] <= pep[index, outerindex]
527 and np.count_nonzero(highs) % 2 == 0):
528 # Usually, this means align right, except for the FTT case,
529 # in which offset for axis 1 and 2 are aligned left.
530 if highs.tolist() == [False, True, True] and index in (1, 2):
531 align = 'left'
532 else:
533 align = 'right'
534 else:
535 # The FF case
536 align = 'left'
537 else:
538 # if TF and if highs has an even number of Trues
539 if (centpt[index] > pep[index, outerindex]
540 and np.count_nonzero(highs) % 2 == 0):
541 # Usually mean align left, except if it is axis 2
542 align = 'right' if index == 2 else 'left'
543 else:
544 # The TT case
545 align = 'right'
546
547 self.offsetText.set_va('center')
548 self.offsetText.set_ha(align)
549 self.offsetText.draw(renderer)

567 @artist.allow_rasterization
568 def draw(self, renderer):
569 self.label._transform = self.axes.transData
570 self.offsetText._transform = self.axes.transData
571 renderer.open_group("axis3d", gid=self.get_gid())
572
573 # Get general axis information:
574 mins, maxs, tc, highs = self._get_coord_info()
575 centers, deltas = self._calc_centers_deltas(maxs, mins)
576
577 # Calculate offset distances
578 # A rough estimate; points are ambiguous since 3D plots rotate
579 reltoinches = self.get_figure(root=False).dpi_scale_trans.inverted()
580 ax_inches = reltoinches.transform(self.axes.bbox.size)
581 ax_points_estimate = sum(72. * ax_inches)
582 deltas_per_point = 48 / ax_points_estimate
583 default_offset = 21.
584 labeldeltas = (self.labelpad + default_offset) * deltas_per_point * deltas
585
586 # Determine edge points for the axis lines
587 minmax = np.where(highs, maxs, mins) # "origin" point
588 maxmin = np.where(~highs, maxs, mins) # "opposite" corner near camera
589
590 for edgep1, edgep2, pos in zip(*self._get_all_axis_line_edge_points(
591 minmax, maxmin, self._tick_position)):
592 # Project the edge points along the current position
593 pep = proj3d._proj_trans_points([edgep1, edgep2], self.axes.M)
594 pep = np.asarray(pep)
595
596 # The transAxes transform is used because the Text object
597 # rotates the text relative to the display coordinate system.
598 # Therefore, if we want the labels to remain parallel to the
599 # axis regardless of the aspect ratio, we need to convert the
600 # edge points of the plane to display coordinates and calculate
601 # an angle from that.
602 # TODO: Maybe Text objects should handle this themselves?
603 dx, dy = (self.axes.transAxes.transform([pep[0:2, 1]]) -
604 self.axes.transAxes.transform([pep[0:2, 0]]))[0]
605
606 # Draw the lines
607 self.line.set_data(pep[0], pep[1])
608 self.line.draw(renderer)
609
610 # Draw ticks
611 self._draw_ticks(renderer, edgep1, centers, deltas, highs,
612 deltas_per_point, pos)
613
614 # Draw Offset text
615 self._draw_offset_text(renderer, edgep1, edgep2, labeldeltas,
616 centers, highs, pep, dx, dy)
617
618 for edgep1, edgep2, pos in zip(*self._get_all_axis_line_edge_points(
619 minmax, maxmin, self._label_position)):
620 # See comments above
621 pep = proj3d._proj_trans_points([edgep1, edgep2], self.axes.M)
