163 """
164 Copy the array if its base points to a parent array.
165 """

167 return a.copy()
168 elif np.issubsctype(a, np.float32):

170 else:

184def _randdm(pnts):
185 """
186 Generate a random distance matrix stored in condensed form.
187
188 Parameters
189 ----------
190 pnts : int
191 The number of points in the distance matrix. Has to be at least 2.
192
193 Returns
194 -------
195 D : ndarray
196 A ``pnts * (pnts - 1) / 2`` sized vector is returned.

836def linkage(y, method='single', metric='euclidean', optimal_ordering=False):
837 """
838 Perform hierarchical/agglomerative clustering.
839
840 The input y may be either a 1-D condensed distance matrix
841 or a 2-D array of observation vectors.
842
843 If y is a 1-D condensed distance matrix,
844 then y must be a :math:`\\binom{n}{2}` sized
845 vector, where n is the number of original observations paired
846 in the distance matrix. The behavior of this function is very
847 similar to the MATLAB linkage function.
848
849 A :math:`(n-1)` by 4 matrix ``Z`` is returned. At the
850 :math:`i`-th iteration, clusters with indices ``Z[i, 0]`` and
851 ``Z[i, 1]`` are combined to form cluster :math:`n + i`. A
852 cluster with an index less than :math:`n` corresponds to one of
853 the :math:`n` original observations. The distance between
854 clusters ``Z[i, 0]`` and ``Z[i, 1]`` is given by ``Z[i, 2]``. The
855 fourth value ``Z[i, 3]`` represents the number of original
856 observations in the newly formed cluster.
857
858 The following linkage methods are used to compute the distance
859 :math:`d(s, t)` between two clusters :math:`s` and
860 :math:`t`. The algorithm begins with a forest of clusters that
861 have yet to be used in the hierarchy being formed. When two
862 clusters :math:`s` and :math:`t` from this forest are combined
863 into a single cluster :math:`u`, :math:`s` and :math:`t` are
864 removed from the forest, and :math:`u` is added to the
865 forest. When only one cluster remains in the forest, the algorithm
866 stops, and this cluster becomes the root.
867
868 A distance matrix is maintained at each iteration. The ``d[i,j]``
869 entry corresponds to the distance between cluster :math:`i` and
870 :math:`j` in the original forest.
871
872 At each iteration, the algorithm must update the distance matrix
873 to reflect the distance of the newly formed cluster u with the
874 remaining clusters in the forest.
875
876 Suppose there are :math:`|u|` original observations
877 :math:`u[0], \\ldots, u[|u|-1]` in cluster :math:`u` and
878 :math:`|v|` original objects :math:`v[0], \\ldots, v[|v|-1]` in
879 cluster :math:`v`. Recall, :math:`s` and :math:`t` are
880 combined to form cluster :math:`u`. Let :math:`v` be any
881 remaining cluster in the forest that is not :math:`u`.
882
883 The following are methods for calculating the distance between the
884 newly formed cluster :math:`u` and each :math:`v`.
885
886 * method='single' assigns
887
888 .. math::
889 d(u,v) = \\min(dist(u[i],v[j]))
890
891 for all points :math:`i` in cluster :math:`u` and
892 :math:`j` in cluster :math:`v`. This is also known as the
893 Nearest Point Algorithm.
894
895 * method='complete' assigns
896
897 .. math::
898 d(u, v) = \\max(dist(u[i],v[j]))
899
900 for all points :math:`i` in cluster u and :math:`j` in
901 cluster :math:`v`. This is also known by the Farthest Point
902 Algorithm or Voor Hees Algorithm.
903
904 * method='average' assigns
905
906 .. math::
907 d(u,v) = \\sum_{ij} \\frac{d(u[i], v[j])}
908 {(|u|*|v|)}
909
910 for all points :math:`i` and :math:`j` where :math:`|u|`
911 and :math:`|v|` are the cardinalities of clusters :math:`u`
912 and :math:`v`, respectively. This is also called the UPGMA
913 algorithm.
914
915 * method='weighted' assigns
916
917 .. math::
918 d(u,v) = (dist(s,v) + dist(t,v))/2
919
920 where cluster u was formed with cluster s and t and v
921 is a remaining cluster in the forest (also called WPGMA).
922
923 * method='centroid' assigns
924
925 .. math::
926 dist(s,t) = ||c_s-c_t||_2
927
928 where :math:`c_s` and :math:`c_t` are the centroids of
929 clusters :math:`s` and :math:`t`, respectively. When two
930 clusters :math:`s` and :math:`t` are combined into a new
931 cluster :math:`u`, the new centroid is computed over all the
932 original objects in clusters :math:`s` and :math:`t`. The
933 distance then becomes the Euclidean distance between the
934 centroid of :math:`u` and the centroid of a remaining cluster
935 :math:`v` in the forest. This is also known as the UPGMC
936 algorithm.
937
938 * method='median' assigns :math:`d(s,t)` like the ``centroid``
939 method. When two clusters :math:`s` and :math:`t` are combined
940 into a new cluster :math:`u`, the average of centroids s and t
941 give the new centroid :math:`u`. This is also known as the
942 WPGMC algorithm.
943
944 * method='ward' uses the Ward variance minimization algorithm.
945 The new entry :math:`d(u,v)` is computed as follows,
946
947 .. math::
948
949 d(u,v) = \\sqrt{\\frac{|v|+|s|}
950 {T}d(v,s)^2
951 + \\frac{|v|+|t|}
952 {T}d(v,t)^2
953 - \\frac{|v|}
954 {T}d(s,t)^2}
955
956 where :math:`u` is the newly joined cluster consisting of
957 clusters :math:`s` and :math:`t`, :math:`v` is an unused
958 cluster in the forest, :math:`T=|v|+|s|+|t|`, and
959 :math:`|*|` is the cardinality of its argument. This is also
960 known as the incremental algorithm.
961
962 Warning: When the minimum distance pair in the forest is chosen, there
963 may be two or more pairs with the same minimum distance. This
964 implementation may choose a different minimum than the MATLAB
965 version.
966
967 Parameters
968 ----------
969 y : ndarray
970 A condensed distance matrix. A condensed distance matrix
971 is a flat array containing the upper triangular of the distance matrix.
972 This is the form that ``pdist`` returns. Alternatively, a collection of
973 :math:`m` observation vectors in :math:`n` dimensions may be passed as
974 an :math:`m` by :math:`n` array. All elements of the condensed distance
975 matrix must be finite, i.e., no NaNs or infs.
976 method : str, optional
977 The linkage algorithm to use. See the ``Linkage Methods`` section below
978 for full descriptions.
979 metric : str or function, optional
980 The distance metric to use in the case that y is a collection of
981 observation vectors; ignored otherwise. See the ``pdist``
982 function for a list of valid distance metrics. A custom distance
983 function can also be used.
984 optimal_ordering : bool, optional
985 If True, the linkage matrix will be reordered so that the distance
986 between successive leaves is minimal. This results in a more intuitive
987 tree structure when the data are visualized. defaults to False, because
988 this algorithm can be slow, particularly on large datasets [2]_. See
989 also the `optimal_leaf_ordering` function.
990
991 .. versionadded:: 1.0.0
992
993 Returns
994 -------
995 Z : ndarray
996 The hierarchical clustering encoded as a linkage matrix.
997
998 Notes
999 -----
1000 1. For method 'single', an optimized algorithm based on minimum spanning
1001 tree is implemented. It has time complexity :math:`O(n^2)`.
1002 For methods 'complete', 'average', 'weighted' and 'ward', an algorithm
1003 called nearest-neighbors chain is implemented. It also has time
1004 complexity :math:`O(n^2)`.
1005 For other methods, a naive algorithm is implemented with :math:`O(n^3)`
1006 time complexity.
1007 All algorithms use :math:`O(n^2)` memory.
1008 Refer to [1]_ for details about the algorithms.
1009 2. Methods 'centroid', 'median', and 'ward' are correctly defined only if
1010 Euclidean pairwise metric is used. If `y` is passed as precomputed
1011 pairwise distances, then it is the user's responsibility to assure that
1012 these distances are in fact Euclidean, otherwise the produced result
1013 will be incorrect.
1014
1015 See Also
1016 --------
1017 scipy.spatial.distance.pdist : pairwise distance metrics
1018
1019 References
1020 ----------
1021 .. [1] Daniel Mullner, "Modern hierarchical, agglomerative clustering
1022 algorithms", :arXiv:`1109.2378v1`.
1023 .. [2] Ziv Bar-Joseph, David K. Gifford, Tommi S. Jaakkola, "Fast optimal
1024 leaf ordering for hierarchical clustering", 2001. Bioinformatics
1025 :doi:`10.1093/bioinformatics/17.suppl_1.S22`
1026
1027 Examples
1028 --------
1029 >>> from scipy.cluster.hierarchy import dendrogram, linkage
1030 >>> from matplotlib import pyplot as plt
1031 >>> X = [[i] for i in [2, 8, 0, 4, 1, 9, 9, 0]]
1032
1033 >>> Z = linkage(X, 'ward')
1034 >>> fig = plt.figure(figsize=(25, 10))
1035 >>> dn = dendrogram(Z)
1036
1037 >>> Z = linkage(X, 'single')
1038 >>> fig = plt.figure(figsize=(25, 10))
1039 >>> dn = dendrogram(Z)
1040 >>> plt.show()
1041 """
1042 if method not in _LINKAGE_METHODS:
1043 raise ValueError(f"Invalid method: {method}")
1044
1045 y = _convert_to_double(np.asarray(y, order='c'))
1046
1047 if y.ndim == 1:
1048 distance.is_valid_y(y, throw=True, name='y')
1049 [y] = _copy_arrays_if_base_present([y])
1050 elif y.ndim == 2:
1051 if method in _EUCLIDEAN_METHODS and metric != 'euclidean':
1052 raise ValueError("Method '{}' requires the distance metric "
1053 "to be Euclidean".format(method))
1054 if y.shape[0] == y.shape[1] and np.allclose(np.diag(y), 0):
1055 if np.all(y >= 0) and np.allclose(y, y.T):
1056 _warning('The symmetric non-negative hollow observation '
1057 'matrix looks suspiciously like an uncondensed '
1058 'distance matrix')
1059 y = distance.pdist(y, metric)
1060 else:
1061 raise ValueError("`y` must be 1 or 2 dimensional.")
1062
1063 if not np.all(np.isfinite(y)):
1064 raise ValueError("The condensed distance matrix must contain only "
1065 "finite values.")
1066
1067 n = int(distance.num_obs_y(y))
1068 method_code = _LINKAGE_METHODS[method]
1069
1070 if method == 'single':
1071 result = _hierarchy.mst_single_linkage(y, n)
1072 elif method in ['complete', 'average', 'weighted', 'ward']:
1073 result = _hierarchy.nn_chain(y, n, method_code)
1074 else:
1075 result = _hierarchy.fast_linkage(y, n, method_code)
1076
1077 if optimal_ordering:
1078 return optimal_leaf_ordering(result, y)
1079 else:
1080 return result

1221 def pre_order(self, func=(lambda x: x.id)):
1222 """
1223 Perform pre-order traversal without recursive function calls.
1224
1225 When a leaf node is first encountered, ``func`` is called with
1226 the leaf node as its argument, and its result is appended to
1227 the list.
1228
1229 For example, the statement::
1230
1231 ids = root.pre_order(lambda x: x.id)
1232
1233 returns a list of the node ids corresponding to the leaf nodes
1234 of the tree as they appear from left to right.
1235
1236 Parameters
1237 ----------
1238 func : function
1239 Applied to each leaf ClusterNode object in the pre-order traversal.
1240 Given the ``i``-th leaf node in the pre-order traversal ``n[i]``,
1241 the result of ``func(n[i])`` is stored in ``L[i]``. If not
1242 provided, the index of the original observation to which the node
1243 corresponds is used.
1244
1245 Returns
1246 -------
1247 L : list
1248 The pre-order traversal.
1249
1250 """
1251 # Do a preorder traversal, caching the result. To avoid having to do
1252 # recursion, we'll store the previous index we've visited in a vector.
1253 n = self.count
1254
1255 curNode = [None] * (2 * n)
1256 lvisited = set()
1257 rvisited = set()
1258 curNode[0] = self
1259 k = 0
1260 preorder = []
1261 while k >= 0:
1262 nd = curNode[k]
1263 ndid = nd.id
1264 if nd.is_leaf():
1265 preorder.append(func(nd))
1266 k = k - 1
1267 else:
1268 if ndid not in lvisited:
1269 curNode[k + 1] = nd.left
1270 lvisited.add(ndid)
1271 k = k + 1
1272 elif ndid not in rvisited:
1273 curNode[k + 1] = nd.right
1274 rvisited.add(ndid)
1275 k = k + 1
1276 # If we've visited the left and right of this non-leaf
1277 # node already, go up in the tree.
1278 else:
1279 k = k - 1
1280
1281 return preorder

1316def cut_tree(Z, n_clusters=None, height=None):
1317 """
1318 Given a linkage matrix Z, return the cut tree.
1319
1320 Parameters
1321 ----------
1322 Z : scipy.cluster.linkage array
1323 The linkage matrix.
1324 n_clusters : array_like, optional
1325 Number of clusters in the tree at the cut point.
1326 height : array_like, optional
1327 The height at which to cut the tree. Only possible for ultrametric
1328 trees.
1329
1330 Returns
1331 -------
1332 cutree : array
1333 An array indicating group membership at each agglomeration step. I.e.,
1334 for a full cut tree, in the first column each data point is in its own
1335 cluster. At the next step, two nodes are merged. Finally, all
1336 singleton and non-singleton clusters are in one group. If `n_clusters`
1337 or `height` are given, the columns correspond to the columns of
1338 `n_clusters` or `height`.
1339
1340 Examples
1341 --------
1342 >>> from scipy import cluster
1343 >>> import numpy as np
1344 >>> from numpy.random import default_rng
1345 >>> rng = default_rng()
1346 >>> X = rng.random((50, 4))
1347 >>> Z = cluster.hierarchy.ward(X)
1348 >>> cutree = cluster.hierarchy.cut_tree(Z, n_clusters=[5, 10])
1349 >>> cutree[:10]
1350 array([[0, 0],
1351 [1, 1],
1352 [2, 2],
1353 [3, 3],
1354 [3, 4],
1355 [2, 2],
1356 [0, 0],
1357 [1, 5],
1358 [3, 6],
1359 [4, 7]]) # random
1360
1361 """
1362 nobs = num_obs_linkage(Z)
1363 nodes = _order_cluster_tree(Z)
1364
1365 if height is not None and n_clusters is not None:
1366 raise ValueError("At least one of either height or n_clusters "
1367 "must be None")
1368 elif height is None and n_clusters is None: # return the full cut tree
1369 cols_idx = np.arange(nobs)
1370 elif height is not None:
1371 heights = np.array([x.dist for x in nodes])
1372 cols_idx = np.searchsorted(heights, height)
1373 else:
1374 cols_idx = nobs - np.searchsorted(np.arange(nobs), n_clusters)
1375
1376 try:
1377 n_cols = len(cols_idx)
1378 except TypeError: # scalar
1379 n_cols = 1
1380 cols_idx = np.array([cols_idx])
1381
1382 groups = np.zeros((n_cols, nobs), dtype=int)
1383 last_group = np.arange(nobs)
1384 if 0 in cols_idx:
1385 groups[0] = last_group
1386
1387 for i, node in enumerate(nodes):
1388 idx = node.pre_order()
1389 this_group = last_group.copy()
1390 this_group[idx] = last_group[idx].min()
1391 this_group[this_group > last_group[idx].max()] -= 1
1392 if i + 1 in cols_idx:
1393 groups[np.nonzero(i + 1 == cols_idx)[0]] = this_group
1394 last_group = this_group
1395
1396 return groups.T

1401 Convert a linkage matrix into an easy-to-use tree object.
1402
1403 The reference to the root `ClusterNode` object is returned (by default).
1404
1405 Each `ClusterNode` object has a ``left``, ``right``, ``dist``, ``id``,
1406 and ``count`` attribute. The left and right attributes point to
1407 ClusterNode objects that were combined to generate the cluster.

1409 must be 1, and its distance is meaningless but set to 0.
1410

1412 user. ClusterNodes are not used as input to any of the functions in this
1413 library.*
1414
1415 Parameters
1416 ----------
1417 Z : ndarray
1418 The linkage matrix in proper form (see the `linkage`
1419 function documentation).
1420 rd : bool, optional
1421 When False (default), a reference to the root `ClusterNode` object is
1422 returned. Otherwise, a tuple ``(r, d)`` is returned. ``r`` is a
1423 reference to the root node while ``d`` is a list of `ClusterNode`
1424 objects - one per original entry in the linkage matrix plus entries
1425 for all clustering steps. If a cluster id is
1426 less than the number of samples ``n`` in the data that the linkage
1427 matrix describes, then it corresponds to a singleton cluster (leaf
1428 node).
1429 See `linkage` for more information on the assignment of cluster ids
1430 to clusters.
1431
1432 Returns
1433 -------
1434 tree : ClusterNode or tuple (ClusterNode, list of ClusterNode)
1435 If ``rd`` is False, a `ClusterNode`.
1436 If ``rd`` is True, a list of length ``2*n - 1``, with ``n`` the number
1437 of samples. See the description of `rd` above for more details.
1438
1439 See Also
1440 --------
1441 linkage, is_valid_linkage, ClusterNode
1442
1443 Examples
1444 --------
1445 >>> import numpy as np
1446 >>> from scipy.cluster import hierarchy

1455 >>> len(nodelist)
1456 9

1463 n = Z.shape[0] + 1
1464
1465 # Create a list full of None's to store the node objects
1466 d = [None] * (n * 2 - 1)

1469 for i in range(0, n):
1470 d[i] = ClusterNode(i)
1471

1476 fj = int(row[1])
1477 if fi > i + n:

1490 'incorrect.') % i)
1491 d[n + i] = nd
1492
1493 if rd:
1494 return (nd, d)
1495 else:
1496 return nd

2427def fcluster(Z, t, criterion='inconsistent', depth=2, R=None, monocrit=None):
2428 """

2430 the given linkage matrix.
2431
2432 Parameters
2433 ----------
2434 Z : ndarray
2435 The hierarchical clustering encoded with the matrix returned
2436 by the `linkage` function.
2437 t : scalar
2438 For criteria 'inconsistent', 'distance' or 'monocrit',
2439 this is the threshold to apply when forming flat clusters.
2440 For 'maxclust' or 'maxclust_monocrit' criteria,
2441 this would be max number of clusters requested.
2442 criterion : str, optional
2443 The criterion to use in forming flat clusters. This can
2444 be any of the following values:
2445
2446 ``inconsistent`` :
2447 If a cluster node and all its
2448 descendants have an inconsistent value less than or equal
2449 to `t`, then all its leaf descendants belong to the
2450 same flat cluster. When no non-singleton cluster meets
2451 this criterion, every node is assigned to its own
2452 cluster. (Default)

2454 ``distance`` :
2455 Forms flat clusters so that the original
2456 observations in each flat cluster have no greater a

2461 the cophenetic distance between any two original
2462 observations in the same flat cluster is no more than

2467 with index i when ``monocrit[j] <= t``.
2468

2470 as computed in the inconsistency matrix R with a
2471 threshold of 0.8 do::
2472

2479 r`` for all cluster indices ``i`` below and including
2480 ``c``. ``r`` is minimized such that no more than ``t``
2481 flat clusters are formed. monocrit must be
2482 monotonic. For example, to minimize the threshold t on
2483 maximum inconsistency values so that no more than 3 flat

2485

3694def is_isomorphic(T1, T2):
3695 """
3696 Determine if two different cluster assignments are equivalent.
3697
3698 Parameters
3699 ----------
3700 T1 : array_like
3701 An assignment of singleton cluster ids to flat cluster ids.
3702 T2 : array_like
3703 An assignment of singleton cluster ids to flat cluster ids.
3704
3705 Returns
3706 -------
3707 b : bool
3708 Whether the flat cluster assignments `T1` and `T2` are
3709 equivalent.
3710
3711 See Also
3712 --------
3713 linkage : for a description of what a linkage matrix is.
3714 fcluster : for the creation of flat cluster assignments.
3715
3716 Examples
3717 --------
3718 >>> from scipy.cluster.hierarchy import fcluster, is_isomorphic
3719 >>> from scipy.cluster.hierarchy import single, complete
3720 >>> from scipy.spatial.distance import pdist
3721
3722 Two flat cluster assignments can be isomorphic if they represent the same
3723 cluster assignment, with different labels.
3724
3725 For example, we can use the `scipy.cluster.hierarchy.single`: method
3726 and flatten the output to four clusters:
3727
3728 >>> X = [[0, 0], [0, 1], [1, 0],
3729 ... [0, 4], [0, 3], [1, 4],
3730 ... [4, 0], [3, 0], [4, 1],
3731 ... [4, 4], [3, 4], [4, 3]]
3732
3733 >>> Z = single(pdist(X))
3734 >>> T = fcluster(Z, 1, criterion='distance')
3735 >>> T
3736 array([3, 3, 3, 4, 4, 4, 2, 2, 2, 1, 1, 1], dtype=int32)
3737
3738 We can then do the same using the
3739 `scipy.cluster.hierarchy.complete`: method:
3740
3741 >>> Z = complete(pdist(X))
3742 >>> T_ = fcluster(Z, 1.5, criterion='distance')
3743 >>> T_
3744 array([1, 1, 1, 2, 2, 2, 3, 3, 3, 4, 4, 4], dtype=int32)
3745
3746 As we can see, in both cases we obtain four clusters and all the data
3747 points are distributed in the same way - the only thing that changes
3748 are the flat cluster labels (3 => 1, 4 =>2, 2 =>3 and 4 =>1), so both
3749 cluster assignments are isomorphic:
3750
3751 >>> is_isomorphic(T, T_)
3752 True
3753
3754 """
3755 T1 = np.asarray(T1, order='c')
3756 T2 = np.asarray(T2, order='c')
3757
3758 if type(T1) != np.ndarray:
3759 raise TypeError('T1 must be a numpy array.')
3760 if type(T2) != np.ndarray:
3761 raise TypeError('T2 must be a numpy array.')
3762
3763 T1S = T1.shape
3764 T2S = T2.shape
3765
3766 if len(T1S) != 1:
3767 raise ValueError('T1 must be one-dimensional.')
3768 if len(T2S) != 1:
3769 raise ValueError('T2 must be one-dimensional.')
3770 if T1S[0] != T2S[0]:
3771 raise ValueError('T1 and T2 must have the same number of elements.')
3772 n = T1S[0]
3773 d1 = {}
3774 d2 = {}
3775 for i in range(0, n):
3776 if T1[i] in d1:
3777 if T2[i] not in d2:
3778 return False
3779 if d1[T1[i]] != T2[i] or d2[T2[i]] != T1[i]:
3780 return False
3781 elif T2[i] in d2:
3782 return False
3783 else:
3784 d1[T1[i]] = T2[i]
3785 d2[T2[i]] = T1[i]
3786 return True
