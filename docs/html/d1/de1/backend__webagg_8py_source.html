143 def on_message(self, message):
144 message = json.loads(message)
145 # The 'supports_binary' message is on a client-by-client
146 # basis. The others affect the (shared) canvas as a
147 # whole.

151 manager = Gcf.get_fig_manager(self.fignum)

153 # but a stale figure UI is still sending messages

219 # This port selection algorithm is borrowed, more or less
220 # verbatim, from IPython.
221 def random_ports(port, n):
222 """
223 Generate a list of n random ports near the given port.
224
225 The first 5 ports will be sequential, and the remaining n-5 will be
226 randomly selected in the range [port-2*n, port+2*n].
227 """
228 for i in range(min(5, n)):
229 yield port + i
230 for i in range(n - 5):
231 yield port + random.randint(-2 * n, 2 * n)
232
233 if address is None:
234 cls.address = mpl.rcParams['webagg.address']
235 else:
236 cls.address = address
237 cls.port = mpl.rcParams['webagg.port']
238 for port in random_ports(cls.port,
239 mpl.rcParams['webagg.port_retries']):
240 try:
241 app.listen(port, cls.address)
242 except OSError as e:
243 if e.errno != errno.EADDRINUSE:
244 raise
245 else:
246 cls.port = port
247 break
248 else:
249 raise SystemExit(
250 "The webagg server could not be started because an available "

268 """
269 IOLoop.running() was removed as of Tornado 2.4; see for example
270 https://groups.google.com/forum/#!topic/python-tornado/QLMzkpQBGOY
271 Thus there is no correct way to check if the loop has already been
272 launched. We may end up with two concurrently running loops in that
273 unlucky case with all the expected consequences.
274 """
