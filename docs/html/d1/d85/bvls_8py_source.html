9def compute_kkt_optimality(g, on_bound):
10 """Compute the maximum violation of KKT conditions."""
11 g_kkt = g * on_bound
12 free_set = on_bound == 0
13 g_kkt[free_set] = np.abs(g[free_set])
14 return np.max(g_kkt)
