46 def _str_map(
47 self,
48 f,
49 na_value=lib.no_default,
50 dtype: NpDtype | None = None,
51 convert: bool = True,
52 ):
53 """
54 Map a callable over valid elements of the array.
55
56 Parameters
57 ----------
58 f : Callable
59 A function to call on each non-NA element.
60 na_value : Scalar, optional
61 The value to set for NA values. Might also be used for the
62 fill value if the callable `f` raises an exception.
63 This defaults to ``self.dtype.na_value`` which is ``np.nan``
64 for object-dtype and Categorical and ``pd.NA`` for StringArray.
65 dtype : Dtype, optional
66 The dtype of the result array.
67 convert : bool, default True
68 Whether to call `maybe_convert_objects` on the resulting ndarray
69 """
70 if dtype is None:
71 dtype = np.dtype("object")
72 if na_value is lib.no_default:
73 na_value = self.dtype.na_value # type: ignore[attr-defined]
74
75 if not len(self):
76 return np.array([], dtype=dtype)
77
78 arr = np.asarray(self, dtype=object)
79 mask = isna(arr)
80 map_convert = convert and not np.all(mask)
81 try:
82 result = lib.map_infer_mask(arr, f, mask.view(np.uint8), map_convert)
83 except (TypeError, AttributeError) as err:
84 # Reraise the exception if callable `f` got wrong number of args.
85 # The user may want to be warned by this, instead of getting NaN
86 p_err = (
87 r"((takes)|(missing)) (?(2)from \d+ to )?\d+ "
88 r"(?(3)required )positional arguments?"
89 )
90
91 if len(err.args) >= 1 and re.search(p_err, err.args[0]):
92 # FIXME: this should be totally avoidable
93 raise err
94
95 def g(x):
96 # This type of fallback behavior can be removed once
97 # we remove object-dtype .str accessor.
98 try:
99 return f(x)
100 except (TypeError, AttributeError):
101 return na_value
102
103 return self._str_map(g, na_value=na_value, dtype=dtype)
104 if not isinstance(result, np.ndarray):
105 return result
106 if na_value is not np.nan:
107 np.putmask(result, mask, na_value)
108 if convert and result.dtype == object:
109 result = lib.maybe_convert_objects(result)
110 return result
