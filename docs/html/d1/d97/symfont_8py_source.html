44def green(f, curveXY):
45 f = -sp.integrate(sp.sympify(f), y)
46 f = f.subs({x: curveXY[0], y: curveXY[1]})
47 f = sp.integrate(f * sp.diff(curveXY[0], t), (t, 0, 1))
