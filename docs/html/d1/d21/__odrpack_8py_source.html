1"""
2Python wrappers for Orthogonal Distance Regression (ODRPACK).
3
4Notes
5=====
6
7* Array formats -- FORTRAN stores its arrays in memory column first, i.e., an
8 array element A(i, j, k) will be next to A(i+1, j, k). In C and, consequently,
9 NumPy, arrays are stored row first: A[i, j, k] is next to A[i, j, k+1]. For
10 efficiency and convenience, the input and output arrays of the fitting
11 function (and its Jacobians) are passed to FORTRAN without transposition.
12 Therefore, where the ODRPACK documentation says that the X array is of shape
13 (N, M), it will be passed to the Python function as an array of shape (M, N).
14 If M==1, the 1-D case, then nothing matters; if M>1, then your
15 Python functions will be dealing with arrays that are indexed in reverse of
16 the ODRPACK documentation. No real issue, but watch out for your indexing of
17 the Jacobians: the i,jth elements (@f_i/@x_j) evaluated at the nth
18 observation will be returned as jacd[j, i, n]. Except for the Jacobians, it
19 really is easier to deal with x[0] and x[1] than x[:,0] and x[:,1]. Of course,
20 you can always use the transpose() function from SciPy explicitly.
21
22* Examples -- See the accompanying file test/test.py for examples of how to set
23 up fits of your own. Some are taken from the User's Guide; some are from
24 other sources.
25
26* Models -- Some common models are instantiated in the accompanying module
27 models.py . Contributions are welcome.
28
29Credits

31
32* Thanks to Arnold Moene and Gerard Vermeulen for fixing some killer bugs.
33
34Robert Kern
35robert.kern@gmail.com

86def _conv(obj, dtype=None):
87 """ Convert an object to the preferred form for input to the odr routine.
88 """
89
90 if obj is None:
91 return obj
92 else:
93 if dtype is None:
94 obj = numpy.asarray(obj)
95 else:
96 obj = numpy.asarray(obj, dtype)
97 if obj.shape == ():
98 # Scalar.
99 return obj.dtype.type(obj)
100 else:
101 return obj

104def _report_error(info):
105 """ Interprets the return code of the odr routine.
106
107 Parameters
108 ----------
109 info : int
110 The return code of the odr routine.
111
112 Returns
113 -------
114 problems : list(str)
115 A list of messages about why the odr() routine stopped.
116 """
117
118 stopreason = ('Blank',
119 'Sum of squares convergence',
120 'Parameter convergence',
121 'Both sum of squares and parameter convergence',
122 'Iteration limit reached')[info % 5]
123
124 if info >= 5:
125 # questionable results or fatal error
126
127 I = (info//10000 % 10,
128 info//1000 % 10,
129 info//100 % 10,
130 info//10 % 10,
131 info % 10)
132 problems = []
133
134 if I[0] == 0:
135 if I[1] != 0:
136 problems.append('Derivatives possibly not correct')
137 if I[2] != 0:
138 problems.append('Error occurred in callback')
139 if I[3] != 0:
140 problems.append('Problem is not full rank at solution')
141 problems.append(stopreason)
142 elif I[0] == 1:
143 if I[1] != 0:
144 problems.append('N < 1')
145 if I[2] != 0:
146 problems.append('M < 1')
147 if I[3] != 0:
148 problems.append('NP < 1 or NP > N')
149 if I[4] != 0:
150 problems.append('NQ < 1')
151 elif I[0] == 2:
152 if I[1] != 0:
153 problems.append('LDY and/or LDX incorrect')
154 if I[2] != 0:
155 problems.append('LDWE, LD2WE, LDWD, and/or LD2WD incorrect')
156 if I[3] != 0:
157 problems.append('LDIFX, LDSTPD, and/or LDSCLD incorrect')
158 if I[4] != 0:
159 problems.append('LWORK and/or LIWORK too small')
160 elif I[0] == 3:
161 if I[1] != 0:
162 problems.append('STPB and/or STPD incorrect')
163 if I[2] != 0:
164 problems.append('SCLB and/or SCLD incorrect')
165 if I[3] != 0:
166 problems.append('WE incorrect')
167 if I[4] != 0:
168 problems.append('WD incorrect')
169 elif I[0] == 4:
170 problems.append('Error in derivatives')
171 elif I[0] == 5:
172 problems.append('Error occurred in callback')
173 elif I[0] == 6:
174 problems.append('Numerical error detected')
175
176 return problems
177
178 else:
179 return [stopreason]

182class Data:
183 """
184 The data to fit.
185
186 Parameters
187 ----------
188 x : array_like
189 Observed data for the independent variable of the regression
190 y : array_like, optional
191 If array-like, observed data for the dependent variable of the
192 regression. A scalar input implies that the model to be used on
193 the data is implicit.
194 we : array_like, optional
195 If `we` is a scalar, then that value is used for all data points (and
196 all dimensions of the response variable).
197 If `we` is a rank-1 array of length q (the dimensionality of the
198 response variable), then this vector is the diagonal of the covariant
199 weighting matrix for all data points.
200 If `we` is a rank-1 array of length n (the number of data points), then
201 the i'th element is the weight for the i'th response variable
202 observation (single-dimensional only).
203 If `we` is a rank-2 array of shape (q, q), then this is the full
204 covariant weighting matrix broadcast to each observation.
205 If `we` is a rank-2 array of shape (q, n), then `we[:,i]` is the
206 diagonal of the covariant weighting matrix for the i'th observation.
207 If `we` is a rank-3 array of shape (q, q, n), then `we[:,:,i]` is the
208 full specification of the covariant weighting matrix for each
209 observation.
210 If the fit is implicit, then only a positive scalar value is used.
211 wd : array_like, optional
212 If `wd` is a scalar, then that value is used for all data points
213 (and all dimensions of the input variable). If `wd` = 0, then the
214 covariant weighting matrix for each observation is set to the identity
215 matrix (so each dimension of each observation has the same weight).
216 If `wd` is a rank-1 array of length m (the dimensionality of the input
217 variable), then this vector is the diagonal of the covariant weighting
218 matrix for all data points.
219 If `wd` is a rank-1 array of length n (the number of data points), then
220 the i'th element is the weight for the ith input variable observation
221 (single-dimensional only).
222 If `wd` is a rank-2 array of shape (m, m), then this is the full
223 covariant weighting matrix broadcast to each observation.
224 If `wd` is a rank-2 array of shape (m, n), then `wd[:,i]` is the
225 diagonal of the covariant weighting matrix for the ith observation.
226 If `wd` is a rank-3 array of shape (m, m, n), then `wd[:,:,i]` is the
227 full specification of the covariant weighting matrix for each
228 observation.
229 fix : array_like of ints, optional
230 The `fix` argument is the same as ifixx in the class ODR. It is an
231 array of integers with the same shape as data.x that determines which
232 input observations are treated as fixed. One can use a sequence of
233 length m (the dimensionality of the input observations) to fix some
234 dimensions for all observations. A value of 0 fixes the observation,
235 a value > 0 makes it free.
236 meta : dict, optional
237 Free-form dictionary for metadata.
238
239 Notes
240 -----
241 Each argument is attached to the member of the instance of the same name.
242 The structures of `x` and `y` are described in the Model class docstring.
243 If `y` is an integer, then the Data instance can only be used to fit with
244 implicit models where the dimensionality of the response is equal to the
245 specified value of `y`.
246
247 The `we` argument weights the effect a deviation in the response variable
248 has on the fit. The `wd` argument weights the effect a deviation in the
249 input variable has on the fit. To handle multidimensional inputs and
250 responses easily, the structure of these arguments has the n'th
251 dimensional axis first. These arguments heavily use the structured
252 arguments feature of ODRPACK to conveniently and flexibly support all
253 options. See the ODRPACK User's Guide for a full explanation of how these
254 weights are used in the algorithm. Basically, a higher value of the weight
255 for a particular data point makes a deviation at that point more
256 detrimental to the fit.
257
258 """
259
260 def __init__(self, x, y=None, we=None, wd=None, fix=None, meta=None):
261 self.x = _conv(x)
262
263 if not isinstance(self.x, numpy.ndarray):
264 raise ValueError(("Expected an 'ndarray' of data for 'x', "
265 "but instead got data of type '{name}'").format(
266 name=type(self.x).__name__))
267
268 self.y = _conv(y)
269 self.we = _conv(we)
270 self.wd = _conv(wd)
271 self.fix = _conv(fix)
272 self.meta = {} if meta is None else meta
273
274 def set_meta(self, **kwds):
275 """ Update the metadata dictionary with the keywords and data provided
276 by keywords.
277
278 Examples
279 --------
280 ::
281
282 data.set_meta(lab="Ph 7; Lab 26", title="Ag110 + Ag108 Decay")
283 """
284
285 self.meta.update(kwds)
286
287 def __getattr__(self, attr):
288 """ Dispatch attribute access to the metadata dictionary.
289 """
290 if attr in self.meta:
291 return self.meta[attr]
292 else:
293 raise AttributeError("'%s' not in metadata" % attr)

296class RealData(Data):
297 """
298 The data, with weightings as actual standard deviations and/or
299 covariances.
300
301 Parameters
302 ----------
303 x : array_like
304 Observed data for the independent variable of the regression
305 y : array_like, optional
306 If array-like, observed data for the dependent variable of the
307 regression. A scalar input implies that the model to be used on
308 the data is implicit.
309 sx : array_like, optional
310 Standard deviations of `x`.
311 `sx` are standard deviations of `x` and are converted to weights by
312 dividing 1.0 by their squares.
313 sy : array_like, optional
314 Standard deviations of `y`.
315 `sy` are standard deviations of `y` and are converted to weights by
316 dividing 1.0 by their squares.
317 covx : array_like, optional
318 Covariance of `x`
319 `covx` is an array of covariance matrices of `x` and are converted to
320 weights by performing a matrix inversion on each observation's
321 covariance matrix.
322 covy : array_like, optional
323 Covariance of `y`
324 `covy` is an array of covariance matrices and are converted to
325 weights by performing a matrix inversion on each observation's
326 covariance matrix.
327 fix : array_like, optional
328 The argument and member fix is the same as Data.fix and ODR.ifixx:
329 It is an array of integers with the same shape as `x` that
330 determines which input observations are treated as fixed. One can
331 use a sequence of length m (the dimensionality of the input
332 observations) to fix some dimensions for all observations. A value
333 of 0 fixes the observation, a value > 0 makes it free.
334 meta : dict, optional
335 Free-form dictionary for metadata.
336
337 Notes
338 -----
339 The weights `wd` and `we` are computed from provided values as follows:
340
341 `sx` and `sy` are converted to weights by dividing 1.0 by their squares.
342 For example, ``wd = 1./numpy.power(`sx`, 2)``.
343
344 `covx` and `covy` are arrays of covariance matrices and are converted to
345 weights by performing a matrix inversion on each observation's covariance
346 matrix. For example, ``we[i] = numpy.linalg.inv(covy[i])``.
347
348 These arguments follow the same structured argument conventions as wd and
349 we only restricted by their natures: `sx` and `sy` can't be rank-3, but
350 `covx` and `covy` can be.
351
352 Only set *either* `sx` or `covx` (not both). Setting both will raise an
353 exception. Same with `sy` and `covy`.
354
355 """
356
357 def __init__(self, x, y=None, sx=None, sy=None, covx=None, covy=None,
358 fix=None, meta=None):
359 if (sx is not None) and (covx is not None):
360 raise ValueError("cannot set both sx and covx")
361 if (sy is not None) and (covy is not None):
362 raise ValueError("cannot set both sy and covy")
363
364 # Set flags for __getattr__
365 self._ga_flags = {}
366 if sx is not None:
367 self._ga_flags['wd'] = 'sx'
368 else:
369 self._ga_flags['wd'] = 'covx'
370 if sy is not None:
371 self._ga_flags['we'] = 'sy'
372 else:
373 self._ga_flags['we'] = 'covy'
374
375 self.x = _conv(x)
376
377 if not isinstance(self.x, numpy.ndarray):
378 raise ValueError(("Expected an 'ndarray' of data for 'x', "
379 "but instead got data of type '{name}'").format(
380 name=type(self.x).__name__))
381
382 self.y = _conv(y)
383 self.sx = _conv(sx)
384 self.sy = _conv(sy)
385 self.covx = _conv(covx)
386 self.covy = _conv(covy)
387 self.fix = _conv(fix)
388 self.meta = {} if meta is None else meta
389
390 def _sd2wt(self, sd):
391 """ Convert standard deviation to weights.
392 """
393
394 return 1./numpy.power(sd, 2)
395
396 def _cov2wt(self, cov):
397 """ Convert covariance matrix(-ices) to weights.
398 """
399
400 from scipy.linalg import inv
401
402 if len(cov.shape) == 2:
403 return inv(cov)
404 else:
405 weights = numpy.zeros(cov.shape, float)
406
407 for i in range(cov.shape[-1]): # n
408 weights[:,:,i] = inv(cov[:,:,i])
409
410 return weights
411
412 def __getattr__(self, attr):
413 lookup_tbl = {('wd', 'sx'): (self._sd2wt, self.sx),
414 ('wd', 'covx'): (self._cov2wt, self.covx),
415 ('we', 'sy'): (self._sd2wt, self.sy),
416 ('we', 'covy'): (self._cov2wt, self.covy)}
417
418 if attr not in ('wd', 'we'):
419 if attr in self.meta:
420 return self.meta[attr]
421 else:
422 raise AttributeError("'%s' not in metadata" % attr)
423 else:
424 func, arg = lookup_tbl[(attr, self._ga_flags[attr])]
425
426 if arg is not None:
427 return func(*(arg,))
428 else:
429 return None

432class Model:
433 """
434 The Model class stores information about the function you wish to fit.
435
436 It stores the function itself, at the least, and optionally stores
437 functions which compute the Jacobians used during fitting. Also, one
438 can provide a function that will provide reasonable starting values
439 for the fit parameters possibly given the set of data.
440
441 Parameters
442 ----------
443 fcn : function
444 fcn(beta, x) --> y
445 fjacb : function
446 Jacobian of fcn wrt the fit parameters beta.
447
448 fjacb(beta, x) --> @f_i(x,B)/@B_j
449 fjacd : function
450 Jacobian of fcn wrt the (possibly multidimensional) input
451 variable.
452
453 fjacd(beta, x) --> @f_i(x,B)/@x_j
454 extra_args : tuple, optional
455 If specified, `extra_args` should be a tuple of extra
456 arguments to pass to `fcn`, `fjacb`, and `fjacd`. Each will be called
457 by `apply(fcn, (beta, x) + extra_args)`
458 estimate : array_like of rank-1
459 Provides estimates of the fit parameters from the data
460
461 estimate(data) --> estbeta
462 implicit : boolean
463 If TRUE, specifies that the model
464 is implicit; i.e `fcn(beta, x)` ~= 0 and there is no y data to fit
465 against
466 meta : dict, optional
467 freeform dictionary of metadata for the model
468
469 Notes
470 -----
471 Note that the `fcn`, `fjacb`, and `fjacd` operate on NumPy arrays and
472 return a NumPy array. The `estimate` object takes an instance of the
473 Data class.
474
475 Here are the rules for the shapes of the argument and return
476 arrays of the callback functions:
477
478 `x`
479 if the input data is single-dimensional, then `x` is rank-1
480 array; i.e., ``x = array([1, 2, 3, ...]); x.shape = (n,)``
481 If the input data is multi-dimensional, then `x` is a rank-2 array;
482 i.e., ``x = array([[1, 2, ...], [2, 4, ...]]); x.shape = (m, n)``.
483 In all cases, it has the same shape as the input data array passed to
484 `~scipy.odr.odr`. `m` is the dimensionality of the input data,
485 `n` is the number of observations.
486 `y`
487 if the response variable is single-dimensional, then `y` is a
488 rank-1 array, i.e., ``y = array([2, 4, ...]); y.shape = (n,)``.
489 If the response variable is multi-dimensional, then `y` is a rank-2
490 array, i.e., ``y = array([[2, 4, ...], [3, 6, ...]]); y.shape =
491 (q, n)`` where `q` is the dimensionality of the response variable.
492 `beta`
493 rank-1 array of length `p` where `p` is the number of parameters;
494 i.e. ``beta = array([B_1, B_2, ..., B_p])``
495 `fjacb`
496 if the response variable is multi-dimensional, then the
497 return array's shape is `(q, p, n)` such that ``fjacb(x,beta)[l,k,i] =
498 d f_l(X,B)/d B_k`` evaluated at the ith data point. If `q == 1`, then
499 the return array is only rank-2 and with shape `(p, n)`.
500 `fjacd`
501 as with fjacb, only the return array's shape is `(q, m, n)`
502 such that ``fjacd(x,beta)[l,j,i] = d f_l(X,B)/d X_j`` at the ith data
503 point. If `q == 1`, then the return array's shape is `(m, n)`. If
504 `m == 1`, the shape is (q, n). If `m == q == 1`, the shape is `(n,)`.
505
506 """
507
508 def __init__(self, fcn, fjacb=None, fjacd=None,
509 extra_args=None, estimate=None, implicit=0, meta=None):
510
511 self.fcn = fcn
512 self.fjacb = fjacb
513 self.fjacd = fjacd
514
515 if extra_args is not None:
516 extra_args = tuple(extra_args)
517
518 self.extra_args = extra_args
519 self.estimate = estimate
520 self.implicit = implicit
521 self.meta = meta if meta is not None else {}
522
523 def set_meta(self, **kwds):
524 """ Update the metadata dictionary with the keywords and data provided
525 here.
526
527 Examples
528 --------
529 set_meta(name="Exponential", equation="y = a exp(b x) + c")
530 """
531
532 self.meta.update(kwds)
533
534 def __getattr__(self, attr):
535 """ Dispatch attribute access to the metadata.
536 """
537
538 if attr in self.meta:
539 return self.meta[attr]
540 else:
541 raise AttributeError("'%s' not in metadata" % attr)

544class Output:
545 """
546 The Output class stores the output of an ODR run.
547
548 Attributes
549 ----------
550 beta : ndarray
551 Estimated parameter values, of shape (q,).
552 sd_beta : ndarray
553 Standard deviations of the estimated parameters, of shape (p,).
554 cov_beta : ndarray
555 Covariance matrix of the estimated parameters, of shape (p,p).
556 Note that this `cov_beta` is not scaled by the residual variance
557 `res_var`, whereas `sd_beta` is. This means
558 ``np.sqrt(np.diag(output.cov_beta * output.res_var))`` is the same
559 result as `output.sd_beta`.
560 delta : ndarray, optional
561 Array of estimated errors in input variables, of same shape as `x`.
562 eps : ndarray, optional
563 Array of estimated errors in response variables, of same shape as `y`.
564 xplus : ndarray, optional
565 Array of ``x + delta``.
566 y : ndarray, optional
567 Array ``y = fcn(x + delta)``.
568 res_var : float, optional
569 Residual variance.
570 sum_square : float, optional
571 Sum of squares error.
572 sum_square_delta : float, optional
573 Sum of squares of delta error.
574 sum_square_eps : float, optional
575 Sum of squares of eps error.
576 inv_condnum : float, optional
577 Inverse condition number (cf. ODRPACK UG p. 77).
578 rel_error : float, optional
579 Relative error in function values computed within fcn.
580 work : ndarray, optional
581 Final work array.
582 work_ind : dict, optional
583 Indices into work for drawing out values (cf. ODRPACK UG p. 83).
584 info : int, optional
585 Reason for returning, as output by ODRPACK (cf. ODRPACK UG p. 38).
586 stopreason : list of str, optional
587 `info` interpreted into English.
588
589 Notes
590 -----
591 Takes one argument for initialization, the return value from the
592 function `~scipy.odr.odr`. The attributes listed as "optional" above are
593 only present if `~scipy.odr.odr` was run with ``full_output=1``.
594
595 """
596
597 def __init__(self, output):
598 self.beta = output[0]
599 self.sd_beta = output[1]
600 self.cov_beta = output[2]
601
602 if len(output) == 4:
603 # full output
604 self.__dict__.update(output[3])
605 self.stopreason = _report_error(self.info)
606
607 def pprint(self):
608 """ Pretty-print important results.
609 """
610
611 print('Beta:', self.beta)
612 print('Beta Std Error:', self.sd_beta)
613 print('Beta Covariance:', self.cov_beta)
614 if hasattr(self, 'info'):
615 print('Residual Variance:',self.res_var)
616 print('Inverse Condition #:', self.inv_condnum)
617 print('Reason(s) for Halting:')
618 for r in self.stopreason:
619 print(' %s' % r)

622class ODR:
623 """
624 The ODR class gathers all information and coordinates the running of the
625 main fitting routine.
626
627 Members of instances of the ODR class have the same names as the arguments
628 to the initialization routine.
629
630 Parameters
631 ----------
632 data : Data class instance
633 instance of the Data class
634 model : Model class instance
635 instance of the Model class
636
637 Other Parameters
638 ----------------
639 beta0 : array_like of rank-1
640 a rank-1 sequence of initial parameter values. Optional if
641 model provides an "estimate" function to estimate these values.
642 delta0 : array_like of floats of rank-1, optional
643 a (double-precision) float array to hold the initial values of
644 the errors in the input variables. Must be same shape as data.x
645 ifixb : array_like of ints of rank-1, optional
646 sequence of integers with the same length as beta0 that determines
647 which parameters are held fixed. A value of 0 fixes the parameter,
648 a value > 0 makes the parameter free.
649 ifixx : array_like of ints with same shape as data.x, optional
650 an array of integers with the same shape as data.x that determines
651 which input observations are treated as fixed. One can use a sequence
652 of length m (the dimensionality of the input observations) to fix some
653 dimensions for all observations. A value of 0 fixes the observation,
654 a value > 0 makes it free.
655 job : int, optional
656 an integer telling ODRPACK what tasks to perform. See p. 31 of the
657 ODRPACK User's Guide if you absolutely must set the value here. Use the
658 method set_job post-initialization for a more readable interface.
659 iprint : int, optional
660 an integer telling ODRPACK what to print. See pp. 33-34 of the
661 ODRPACK User's Guide if you absolutely must set the value here. Use the
662 method set_iprint post-initialization for a more readable interface.
663 errfile : str, optional
664 string with the filename to print ODRPACK errors to. If the file already
665 exists, an error will be thrown. The `overwrite` argument can be used to
666 prevent this. *Do Not Open This File Yourself!*
667 rptfile : str, optional
668 string with the filename to print ODRPACK summaries to. If the file
669 already exists, an error will be thrown. The `overwrite` argument can be
670 used to prevent this. *Do Not Open This File Yourself!*
671 ndigit : int, optional
672 integer specifying the number of reliable digits in the computation
673 of the function.
674 taufac : float, optional
675 float specifying the initial trust region. The default value is 1.
676 The initial trust region is equal to taufac times the length of the
677 first computed Gauss-Newton step. taufac must be less than 1.
678 sstol : float, optional
679 float specifying the tolerance for convergence based on the relative
680 change in the sum-of-squares. The default value is eps**(1/2) where eps
681 is the smallest value such that 1 + eps > 1 for double precision
682 computation on the machine. sstol must be less than 1.
683 partol : float, optional
684 float specifying the tolerance for convergence based on the relative
685 change in the estimated parameters. The default value is eps**(2/3) for
686 explicit models and ``eps**(1/3)`` for implicit models. partol must be less
687 than 1.
688 maxit : int, optional
689 integer specifying the maximum number of iterations to perform. For
690 first runs, maxit is the total number of iterations performed and
691 defaults to 50. For restarts, maxit is the number of additional
692 iterations to perform and defaults to 10.
693 stpb : array_like, optional
694 sequence (``len(stpb) == len(beta0)``) of relative step sizes to compute

790 def _check(self):
791 """ Check the inputs for consistency, but don't bother checking things
792 that the builtin function odr will check.
793 """
794
795 x_s = list(self.data.x.shape)
796
797 if isinstance(self.data.y, numpy.ndarray):
798 y_s = list(self.data.y.shape)
799 if self.model.implicit:
800 raise OdrError("an implicit model cannot use response data")
801 else:
802 # implicit model with q == self.data.y
803 y_s = [self.data.y, x_s[-1]]
804 if not self.model.implicit:
805 raise OdrError("an explicit model needs response data")
806 self.set_job(fit_type=1)
807
808 if x_s[-1] != y_s[-1]:
809 raise OdrError("number of observations do not match")
810
811 n = x_s[-1]
812
813 if len(x_s) == 2:
814 m = x_s[0]
815 else:
816 m = 1
817 if len(y_s) == 2:
818 q = y_s[0]
819 else:
820 q = 1
821
822 p = len(self.beta0)
823
824 # permissible output array shapes
825
826 fcn_perms = [(q, n)]
827 fjacd_perms = [(q, m, n)]
828 fjacb_perms = [(q, p, n)]
829
830 if q == 1:
831 fcn_perms.append((n,))
832 fjacd_perms.append((m, n))
833 fjacb_perms.append((p, n))
834 if m == 1:
835 fjacd_perms.append((q, n))
836 if p == 1:
837 fjacb_perms.append((q, n))
838 if m == q == 1:
839 fjacd_perms.append((n,))
840 if p == q == 1:
841 fjacb_perms.append((n,))
842
843 # try evaluating the supplied functions to make sure they provide
844 # sensible outputs
845
846 arglist = (self.beta0, self.data.x)
847 if self.model.extra_args is not None:
848 arglist = arglist + self.model.extra_args
849 res = self.model.fcn(*arglist)
850
851 if res.shape not in fcn_perms:
852 print(res.shape)
853 print(fcn_perms)
854 raise OdrError("fcn does not output %s-shaped array" % y_s)
855
856 if self.model.fjacd is not None:
857 res = self.model.fjacd(*arglist)
858 if res.shape not in fjacd_perms:
859 raise OdrError(
860 "fjacd does not output %s-shaped array" % repr((q, m, n)))
861 if self.model.fjacb is not None:
862 res = self.model.fjacb(*arglist)
863 if res.shape not in fjacb_perms:
864 raise OdrError(
865 "fjacb does not output %s-shaped array" % repr((q, p, n)))
866
867 # check shape of delta0
868
869 if self.delta0 is not None and self.delta0.shape != self.data.x.shape:
870 raise OdrError(
871 "delta0 is not a %s-shaped array" % repr(self.data.x.shape))
872
873 if self.data.x.size == 0:
874 warn(("Empty data detected for ODR instance. "
875 "Do not expect any fitting to occur"),
876 OdrWarning)

878 def _gen_work(self):
879 """ Generate a suitable work array if one does not already exist.
880 """
881
882 n = self.data.x.shape[-1]
883 p = self.beta0.shape[0]
884
885 if len(self.data.x.shape) == 2:
886 m = self.data.x.shape[0]
887 else:
888 m = 1
889
890 if self.model.implicit:
891 q = self.data.y
892 elif len(self.data.y.shape) == 2:
893 q = self.data.y.shape[0]
894 else:
895 q = 1
896
897 if self.data.we is None:
898 ldwe = ld2we = 1
899 elif len(self.data.we.shape) == 3:
900 ld2we, ldwe = self.data.we.shape[1:]
901 else:
902 we = self.data.we
903 ldwe = 1
904 ld2we = 1
905 if we.ndim == 1 and q == 1:
906 ldwe = n
907 elif we.ndim == 2:
908 if we.shape == (q, q):
909 ld2we = q
910 elif we.shape == (q, n):
911 ldwe = n
912
913 if self.job % 10 < 2:
914 # ODR not OLS
915 lwork = (18 + 11*p + p*p + m + m*m + 4*n*q + 6*n*m + 2*n*q*p +
916 2*n*q*m + q*q + 5*q + q*(p+m) + ldwe*ld2we*q)
917 else:
918 # OLS not ODR
919 lwork = (18 + 11*p + p*p + m + m*m + 4*n*q + 2*n*m + 2*n*q*p +
920 5*q + q*(p+m) + ldwe*ld2we*q)
921
922 if isinstance(self.work, numpy.ndarray) and self.work.shape == (lwork,)\
923 and self.work.dtype.str.endswith('f8'):
924 # the existing array is fine
925 return
926 else:
927 self.work = numpy.zeros((lwork,), float)

929 def set_job(self, fit_type=None, deriv=None, var_calc=None,
930 del_init=None, restart=None):
931 """
932 Sets the "job" parameter is a hopefully comprehensible way.
933
934 If an argument is not specified, then the value is left as is. The
935 default value from class initialization is for all of these options set
936 to 0.
937
938 Parameters
939 ----------
940 fit_type : {0, 1, 2} int
941 0 -> explicit ODR
942
943 1 -> implicit ODR
944
945 2 -> ordinary least-squares
946 deriv : {0, 1, 2, 3} int
947 0 -> forward finite differences
948
949 1 -> central finite differences
950
951 2 -> user-supplied derivatives (Jacobians) with results
952 checked by ODRPACK
953
954 3 -> user-supplied derivatives, no checking
955 var_calc : {0, 1, 2} int
956 0 -> calculate asymptotic covariance matrix and fit
957 parameter uncertainties (V_B, s_B) using derivatives
958 recomputed at the final solution
959
960 1 -> calculate V_B and s_B using derivatives from last iteration
961
962 2 -> do not calculate V_B and s_B
963 del_init : {0, 1} int
964 0 -> initial input variable offsets set to 0
965
966 1 -> initial offsets provided by user in variable "work"
967 restart : {0, 1} int
968 0 -> fit is not a restart
969
970 1 -> fit is a restart
971
972 Notes
973 -----
974 The permissible values are different from those given on pg. 31 of the
975 ODRPACK User's Guide only in that one cannot specify numbers greater than
976 the last value for each variable.
977
978 If one does not supply functions to compute the Jacobians, the fitting
979 procedure will change deriv to 0, finite differences, as a default. To
980 initialize the input variable offsets by yourself, set del_init to 1 and
981 put the offsets into the "work" variable correctly.
982
983 """
984
985 if self.job is None:
986 job_l = [0, 0, 0, 0, 0]
987 else:
988 job_l = [self.job // 10000 % 10,
989 self.job // 1000 % 10,
990 self.job // 100 % 10,
991 self.job // 10 % 10,
992 self.job % 10]
993
994 if fit_type in (0, 1, 2):
995 job_l[4] = fit_type
996 if deriv in (0, 1, 2, 3):
997 job_l[3] = deriv
998 if var_calc in (0, 1, 2):
999 job_l[2] = var_calc
1000 if del_init in (0, 1):
1001 job_l[1] = del_init
1002 if restart in (0, 1):
1003 job_l[0] = restart
1004
1005 self.job = (job_l[0]*10000 + job_l[1]*1000 +
1006 job_l[2]*100 + job_l[3]*10 + job_l[4])

1008 def set_iprint(self, init=None, so_init=None,
1009 iter=None, so_iter=None, iter_step=None, final=None, so_final=None):
1010 """ Set the iprint parameter for the printing of computation reports.
1011
1012 If any of the arguments are specified here, then they are set in the
1013 iprint member. If iprint is not set manually or with this method, then
1014 ODRPACK defaults to no printing. If no filename is specified with the
1015 member rptfile, then ODRPACK prints to stdout. One can tell ODRPACK to
1016 print to stdout in addition to the specified filename by setting the
1017 so_* arguments to this function, but one cannot specify to print to
1018 stdout but not a file since one can do that by not specifying a rptfile
1019 filename.
1020
1021 There are three reports: initialization, iteration, and final reports.
1022 They are represented by the arguments init, iter, and final
1023 respectively. The permissible values are 0, 1, and 2 representing "no
1024 report", "short report", and "long report" respectively.
1025
1026 The argument iter_step (0 <= iter_step <= 9) specifies how often to make
1027 the iteration report; the report will be made for every iter_step'th
1028 iteration starting with iteration one. If iter_step == 0, then no
1029 iteration report is made, regardless of the other arguments.
1030
1031 If the rptfile is None, then any so_* arguments supplied will raise an
1032 exception.
1033 """
1034 if self.iprint is None:
1035 self.iprint = 0
1036
1037 ip = [self.iprint // 1000 % 10,
1038 self.iprint // 100 % 10,
1039 self.iprint // 10 % 10,
1040 self.iprint % 10]
1041
1042 # make a list to convert iprint digits to/from argument inputs
1043 # rptfile, stdout
1044 ip2arg = [[0, 0], # none, none
1045 [1, 0], # short, none
1046 [2, 0], # long, none
1047 [1, 1], # short, short
1048 [2, 1], # long, short
1049 [1, 2], # short, long
1050 [2, 2]] # long, long
1051
1052 if (self.rptfile is None and
1053 (so_init is not None or
1054 so_iter is not None or
1055 so_final is not None)):
1056 raise OdrError(
1057 "no rptfile specified, cannot output to stdout twice")
1058
1059 iprint_l = ip2arg[ip[0]] + ip2arg[ip[1]] + ip2arg[ip[3]]
1060
1061 if init is not None:
1062 iprint_l[0] = init
1063 if so_init is not None:
1064 iprint_l[1] = so_init
1065 if iter is not None:
1066 iprint_l[2] = iter
1067 if so_iter is not None:
1068 iprint_l[3] = so_iter
1069 if final is not None:
1070 iprint_l[4] = final
1071 if so_final is not None:
1072 iprint_l[5] = so_final
1073
1074 if iter_step in range(10):
1075 # 0..9
1076 ip[2] = iter_step
1077
1078 ip[0] = ip2arg.index(iprint_l[0:2])
1079 ip[1] = ip2arg.index(iprint_l[2:4])
1080 ip[3] = ip2arg.index(iprint_l[4:6])
1081
1082 self.iprint = ip[0]*1000 + ip[1]*100 + ip[2]*10 + ip[3]
