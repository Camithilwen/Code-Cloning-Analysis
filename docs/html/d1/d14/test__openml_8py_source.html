107 def _mock_urlopen_shared(url, has_gzip_header, expected_prefix, suffix):

140 def _mock_urlopen_download_data(url, has_gzip_header):
141 # For simplicity the mock filenames don't contain the filename, i.e.
142 # the last part of the data description url after the last /.
143 # For example for id_1, data description download url is:
144 # gunzip -c sklearn/datasets/tests/data/openml/id_1/api-v1-jd-1.json.gz | grep '"url" # noqa: E501
145 # "https:\/\/www.openml.org\/data\/v1\/download\/1\/anneal.arff"
146 # but the mock filename does not contain anneal.arff and is:
147 # sklearn/datasets/tests/data/openml/id_1/data-v1-dl-1.arff.gz.
148 # We only keep the part of the url before the last /
149 url_without_filename = url.rsplit("/", 1)[0]

151 return _mock_urlopen_shared(
152 url=url_without_filename,
153 has_gzip_header=has_gzip_header,
154 expected_prefix=url_prefix_download_data,
155 suffix=".arff",

158 def _mock_urlopen_data_list(url, has_gzip_header):
159 assert url.startswith(url_prefix_data_list), (
160 f"{url_prefix_data_list!r} does not match {url!r}"

163 data_file_name = _file_name(url, ".json")

170 json_data = json.loads(decoded_s)

173 url=None, code=412, msg="Simulated mock error", hdrs=None, fp=BytesIO()

177 if has_gzip_header:
178 fp = BytesIO(f.read())
179 return _MockHTTPResponse(fp, True)

181 decompressed_f = read_fn(f, "rb")
182 fp = BytesIO(decompressed_f.read())

185 def _mock_urlopen(request, *args, **kwargs):
186 url = request.get_full_url()
187 has_gzip_header = request.get_header("Accept-encoding") == "gzip"
188 if url.startswith(url_prefix_data_list):
189 return _mock_urlopen_data_list(url, has_gzip_header)
190 elif url.startswith(url_prefix_data_features):
191 return _mock_urlopen_data_features(url, has_gzip_header)
192 elif url.startswith(url_prefix_download_data):
193 return _mock_urlopen_download_data(url, has_gzip_header)
194 elif url.startswith(url_prefix_data_description):
195 return _mock_urlopen_data_description(url, has_gzip_header)
196 else:
197 raise ValueError("Unknown mocking URL pattern: %s" % url)

339@pytest.mark.parametrize("data_id", [61, 1119, 40945])
340def test_fetch_openml_consistency_parser(monkeypatch, data_id):
341 """Check the consistency of the LIAC-ARFF and pandas parsers."""
342 pd = pytest.importorskip("pandas")
343
344 _monkey_patch_webbased_functions(monkeypatch, data_id, gzip_response=True)
345 bunch_liac = fetch_openml(
346 data_id=data_id,
347 as_frame=True,
348 cache=False,
349 parser="liac-arff",
350 )
351 bunch_pandas = fetch_openml(
352 data_id=data_id,
353 as_frame=True,
354 cache=False,
355 parser="pandas",
356 )
357
358 # The data frames for the input features should match up to some numerical
359 # dtype conversions (e.g. float64 <=> Int64) due to limitations of the
360 # LIAC-ARFF parser.
361 data_liac, data_pandas = bunch_liac.data, bunch_pandas.data
362
363 def convert_numerical_dtypes(series):
364 pandas_series = data_pandas[series.name]
365 if pd.api.types.is_numeric_dtype(pandas_series):
366 return series.astype(pandas_series.dtype)
367 else:
368 return series
369
370 data_liac_with_fixed_dtypes = data_liac.apply(convert_numerical_dtypes)
371 pd.testing.assert_frame_equal(data_liac_with_fixed_dtypes, data_pandas)
372
373 # Let's also check that the .frame attributes also match
374 frame_liac, frame_pandas = bunch_liac.frame, bunch_pandas.frame
375
376 # Note that the .frame attribute is a superset of the .data attribute:
377 pd.testing.assert_frame_equal(frame_pandas[bunch_pandas.feature_names], data_pandas)
378
379 # However the remaining columns, typically the target(s), are not necessarily
380 # dtyped similarly by both parsers due to limitations of the LIAC-ARFF parser.
381 # Therefore, extra dtype conversions are required for those columns:
382
383 def convert_numerical_and_categorical_dtypes(series):
384 pandas_series = frame_pandas[series.name]
385 if pd.api.types.is_numeric_dtype(pandas_series):
386 return series.astype(pandas_series.dtype)
387 elif isinstance(pandas_series.dtype, pd.CategoricalDtype):
388 # Compare categorical features by converting categorical liac uses
389 # strings to denote the categories, we rename the categories to make
390 # them comparable to the pandas parser. Fixing this behavior in
391 # LIAC-ARFF would allow to check the consistency in the future but
392 # we do not plan to maintain the LIAC-ARFF on the long term.
393 return series.cat.rename_categories(pandas_series.cat.categories)
394 else:
395 return series
396
397 frame_liac_with_fixed_dtypes = frame_liac.apply(
398 convert_numerical_and_categorical_dtypes
399 )
400 pd.testing.assert_frame_equal(frame_liac_with_fixed_dtypes, frame_pandas)
