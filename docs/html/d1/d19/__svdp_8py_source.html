51class _AProd:
52 """
53 Wrapper class for linear operator
54
55 The call signature of the __call__ method matches the callback of
56 the PROPACK routines.
57 """
58 def __init__(self, A):
59 try:
60 self.A = aslinearoperator(A)
61 except TypeError:
62 self.A = aslinearoperator(np.asarray(A))
63
64 def __call__(self, transa, m, n, x, y, sparm, iparm):
65 if transa == 'n':
66 y[:] = self.A.matvec(x)
67 else:
68 y[:] = self.A.rmatvec(x)
69
70 @property
71 def shape(self):
72 return self.A.shape
73
74 @property
75 def dtype(self):
76 try:
77 return self.A.dtype
78 except AttributeError:
79 return self.A.matvec(np.zeros(self.A.shape[1])).dtype

82def _svdp(A, k, which='LM', irl_mode=True, kmax=None,
83 compute_u=True, compute_v=True, v0=None, full_output=False, tol=0,
84 delta=None, eta=None, anorm=0, cgs=False, elr=True,
85 min_relgap=0.002, shifts=None, maxiter=None, random_state=None):
86 """
87 Compute the singular value decomposition of a linear operator using PROPACK
88
89 Parameters
90 ----------
91 A : array_like, sparse matrix, or LinearOperator
92 Operator for which SVD will be computed. If `A` is a LinearOperator
93 object, it must define both ``matvec`` and ``rmatvec`` methods.
94 k : int
95 Number of singular values/vectors to compute
96 which : {"LM", "SM"}
97 Which singluar triplets to compute:
98 - 'LM': compute triplets corresponding to the `k` largest singular
99 values
100 - 'SM': compute triplets corresponding to the `k` smallest singular
101 values
102 `which='SM'` requires `irl_mode=True`. Computes largest singular
103 values by default.
104 irl_mode : bool, optional
105 If `True`, then compute SVD using IRL (implicitly restarted Lanczos)
106 mode. Default is `True`.
107 kmax : int, optional
108 Maximal number of iterations / maximal dimension of the Krylov
109 subspace. Default is ``10 * k``.
110 compute_u : bool, optional
111 If `True` (default) then compute left singular vectors, `u`.
112 compute_v : bool, optional
113 If `True` (default) then compute right singular vectors, `v`.
114 tol : float, optional
115 The desired relative accuracy for computed singular values.
116 If not specified, it will be set based on machine precision.
117 v0 : array_like, optional
118 Starting vector for iterations: must be of length ``A.shape[0]``.
119 If not specified, PROPACK will generate a starting vector.
120 full_output : bool, optional
121 If `True`, then return sigma_bound. Default is `False`.
122 delta : float, optional
123 Level of orthogonality to maintain between Lanczos vectors.
124 Default is set based on machine precision.
125 eta : float, optional
126 Orthogonality cutoff. During reorthogonalization, vectors with
127 component larger than `eta` along the Lanczos vector will be purged.
128 Default is set based on machine precision.
129 anorm : float, optional
130 Estimate of ``||A||``. Default is `0`.
131 cgs : bool, optional
132 If `True`, reorthogonalization is done using classical Gram-Schmidt.
133 If `False` (default), it is done using modified Gram-Schmidt.
134 elr : bool, optional
135 If `True` (default), then extended local orthogonality is enforced
136 when obtaining singular vectors.
137 min_relgap : float, optional
138 The smallest relative gap allowed between any shift in IRL mode.
139 Default is `0.001`. Accessed only if ``irl_mode=True``.
140 shifts : int, optional
141 Number of shifts per restart in IRL mode. Default is determined
142 to satisfy ``k <= min(kmax-shifts, m, n)``. Must be
143 >= 0, but choosing 0 might lead to performance degredation.
144 Accessed only if ``irl_mode=True``.
145 maxiter : int, optional
146 Maximum number of restarts in IRL mode. Default is `1000`.
147 Accessed only if ``irl_mode=True``.
148 random_state : {None, int, `numpy.random.Generator`,
149 `numpy.random.RandomState`}, optional
150
151 Pseudorandom number generator state used to generate resamples.
152
153 If `random_state` is ``None`` (or `np.random`), the
154 `numpy.random.RandomState` singleton is used.
155 If `random_state` is an int, a new ``RandomState`` instance is used,
156 seeded with `random_state`.
157 If `random_state` is already a ``Generator`` or ``RandomState``
158 instance then that instance is used.
159
160 Returns
161 -------
162 u : ndarray
163 The `k` largest (``which="LM"``) or smallest (``which="SM"``) left
164 singular vectors, ``shape == (A.shape[0], 3)``, returned only if
165 ``compute_u=True``.
166 sigma : ndarray
167 The top `k` singular values, ``shape == (k,)``
168 vt : ndarray
169 The `k` largest (``which="LM"``) or smallest (``which="SM"``) right
170 singular vectors, ``shape == (3, A.shape[1])``, returned only if
171 ``compute_v=True``.
172 sigma_bound : ndarray
173 the error bounds on the singular values sigma, returned only if
174 ``full_output=True``.
175
176 """
177 # 32-bit complex PROPACK functions have Fortran LAPACK ABI
178 # incompatibility issues
179 if np.iscomplexobj(A) and (np.intp(0).itemsize < 8):
180 raise TypeError('PROPACK complex-valued SVD methods not available '
181 'for 32-bit builds')
182
183 random_state = check_random_state(random_state)
184
185 which = which.upper()
186 if which not in {'LM', 'SM'}:
187 raise ValueError("`which` must be either 'LM' or 'SM'")
188 if not irl_mode and which == 'SM':
189 raise ValueError("`which`='SM' requires irl_mode=True")
190
191 aprod = _AProd(A)
192 typ = aprod.dtype.char
193
194 try:
195 lansvd_irl = _lansvd_irl_dict[typ]
196 lansvd = _lansvd_dict[typ]
197 except KeyError:
198 # work with non-supported types using native system precision
199 if np.iscomplexobj(np.empty(0, dtype=typ)):
200 typ = np.dtype(complex).char
201 else:
202 typ = np.dtype(float).char
203 lansvd_irl = _lansvd_irl_dict[typ]
204 lansvd = _lansvd_dict[typ]

206 m, n = aprod.shape
207 if (k < 1) or (k > min(m, n)):
208 raise ValueError("k must be positive and not greater than m or n")
209
210 if kmax is None:
211 kmax = 10*k
212 if maxiter is None:
213 maxiter = 1000
214
215 # guard against unnecessarily large kmax
216 kmax = min(m + 1, n + 1, kmax)
217 if kmax < k:
218 raise ValueError(
219 "kmax must be greater than or equal to k, "
220 f"but kmax ({kmax}) < k ({k})")
221
222 # convert python args to fortran args
223 jobu = 'y' if compute_u else 'n'
224 jobv = 'y' if compute_v else 'n'
225
226 # these will be the output arrays
227 u = np.zeros((m, kmax + 1), order='F', dtype=typ)
228 v = np.zeros((n, kmax), order='F', dtype=typ)

230 # Specify the starting vector. if v0 is all zero, PROPACK will generate
231 # a random starting vector: the random seed cannot be controlled in that
232 # case, so we'll instead use numpy to generate a random vector
233 if v0 is None:
234 u[:, 0] = random_state.uniform(size=m)
235 if np.iscomplexobj(np.empty(0, dtype=typ)): # complex type
236 u[:, 0] += 1j * random_state.uniform(size=m)
237 else:

239 u[:, 0] = v0
240 except ValueError:
241 raise ValueError(f"v0 must be of length {m}")

243 # process options for the fit
244 if delta is None:
245 delta = np.sqrt(np.finfo(typ).eps)
246 if eta is None:
247 eta = np.finfo(typ).eps ** 0.75
248

251
252 # validate or find default shifts
253 if shifts is None:
254 shifts = kmax - k
255 if k > min(kmax - shifts, m, n):
256 raise ValueError('shifts must satisfy '
257 'k <= min(kmax-shifts, m, n)!')
258 elif shifts < 0:
259 raise ValueError('shifts must be >= 0!')
260
261 else:
262 doption = np.array((delta, eta, anorm), dtype=typ.lower())
263
264 ioption = np.array((int(bool(cgs)), int(bool(elr))), dtype='i')

266 # If computing `u` or `v` (left and right singular vectors,
267 # respectively), `blocksize` controls how large a fraction of the
268 # work is done via fast BLAS level 3 operations. A larger blocksize
269 # may lead to faster computation at the expense of greater memory
270 # consumption. `blocksize` must be ``>= 1``. Choosing blocksize
271 # of 16, but docs don't specify; it's almost surely a
272 # power of 2.
273 blocksize = 16
274
275 # Determine lwork & liwork:
276 # the required lengths are specified in the PROPACK documentation
277 if compute_u or compute_v:
278 lwork = m + n + 9*kmax + 5*kmax*kmax + 4 + max(
279 3*kmax*kmax + 4*kmax + 4,
280 blocksize*max(m, n))
281 liwork = 8*kmax
282 else:
283 lwork = m + n + 9*kmax + 2*kmax*kmax + 4 + max(m + n, 4*kmax + 4)
284 liwork = 2*kmax + 1
285 work = np.empty(lwork, dtype=typ.lower())
286 iwork = np.empty(liwork, dtype=np.int32)
287
288 # dummy arguments: these are passed to aprod, and not used in this wrapper
289 dparm = np.empty(1, dtype=typ.lower())
290 iparm = np.empty(1, dtype=np.int32)
291
292 if typ.isupper():
293 # PROPACK documentation is unclear on the required length of zwork.
294 # Use the same length Julia's wrapper uses
295 # see https://github.com/JuliaSmoothOptimizers/PROPACK.jl/
296 zwork = np.empty(m + n + 32*m, dtype=typ)
297 works = work, zwork, iwork
298 else:
299 works = work, iwork
300
301 if irl_mode:
302 u, sigma, bnd, v, info = lansvd_irl(_which_converter[which], jobu,
303 jobv, m, n, shifts, k, maxiter,
304 aprod, u, v, tol, *works, doption,
305 ioption, dparm, iparm)
306 else:
307 u, sigma, bnd, v, info = lansvd(jobu, jobv, m, n, k, aprod, u, v, tol,
308 *works, doption, ioption, dparm, iparm)
309
310 if info > 0:
311 raise LinAlgError(
312 f"An invariant subspace of dimension {info} was found.")
313 elif info < 0:
314 raise LinAlgError(
315 f"k={k} singular triplets did not converge within "
316 f"kmax={kmax} iterations")
317
318 # info == 0: The K largest (or smallest) singular triplets were computed
319 # succesfully!
320
321 return u[:, :k], sigma, v[:, :k].conj().T, bnd
