27GVAR_HEADER_FORMAT_HEAD = """
28 > # big endian
29 version: H
30 reserved: H
31 axisCount: H
32 sharedTupleCount: H
33 offsetToSharedTuples: I
34"""
35# In between the HEAD and TAIL lies the glyphCount, which is
36# of different size: 2 bytes for gvar, and 3 bytes for GVAR.
37GVAR_HEADER_FORMAT_TAIL = """
38 > # big endian
39 flags: H
40 offsetToGlyphVariationData: I
41"""
42
43GVAR_HEADER_SIZE_HEAD = sstruct.calcsize(GVAR_HEADER_FORMAT_HEAD)
44GVAR_HEADER_SIZE_TAIL = sstruct.calcsize(GVAR_HEADER_FORMAT_TAIL)

193 def decompileOffsets_(data, tableFormat, glyphCount):
194 if tableFormat == 0:
195 # Short format: array of UInt16
196 offsets = array.array("H")
197 offsetsSize = (glyphCount + 1) * 2
198 else:
199 # Long format: array of UInt32
200 offsets = array.array("I")
201 offsetsSize = (glyphCount + 1) * 4
202 offsets.frombytes(data[0:offsetsSize])
203 if sys.byteorder != "big":
204 offsets.byteswap()
205
206 # In the short format, offsets need to be multiplied by 2.
207 # This is not documented in Apple's TrueType specification,
208 # but can be inferred from the FreeType implementation, and
209 # we could verify it with two sample GX fonts.
210 if tableFormat == 0:
211 offsets = [off * 2 for off in offsets]
212
213 return offsets

216 def compileOffsets_(offsets):
217 """Packs a list of offsets into a 'gvar' offset table.
218
219 Returns a pair (bytestring, tableFormat). Bytestring is the
220 packed offset table. Format indicates whether the table
221 uses short (tableFormat=0) or long (tableFormat=1) integers.
222 The returned tableFormat should get packed into the flags field
223 of the 'gvar' header.
224 """
225 assert len(offsets) >= 2
226 for i in range(1, len(offsets)):
227 assert offsets[i - 1] <= offsets[i]
228 if max(offsets) <= 0xFFFF * 2:
229 packed = array.array("H", [n >> 1 for n in offsets])
230 tableFormat = 0
231 else:
232 packed = array.array("I", offsets)
233 tableFormat = 1
234 if sys.byteorder != "big":
235 packed.byteswap()
236 return (packed.tobytes(), tableFormat)
