39def test_same_predictions_regression(
40 seed, loss, min_samples_leaf, n_samples, max_leaf_nodes
41):
42 # Make sure sklearn has the same predictions as lightgbm for easy targets.
43 #
44 # In particular when the size of the trees are bound and the number of
45 # samples is large enough, the structure of the prediction trees found by
46 # LightGBM and sklearn should be exactly identical.
47 #
48 # Notes:
49 # - Several candidate splits may have equal gains when the number of
50 # samples in a node is low (and because of float errors). Therefore the
51 # predictions on the test set might differ if the structure of the tree
52 # is not exactly the same. To avoid this issue we only compare the
53 # predictions on the test set when the number of samples is large enough
54 # and max_leaf_nodes is low enough.
55 # - To ignore discrepancies caused by small differences in the binning
56 # strategy, data is pre-binned if n_samples > 255.
57 # - We don't check the absolute_error loss here. This is because
58 # LightGBM's computation of the median (used for the initial value of
59 # raw_prediction) is a bit off (they'll e.g. return midpoints when there
60 # is no need to.). Since these tests only run 1 iteration, the
61 # discrepancy between the initial values leads to biggish differences in
62 # the predictions. These differences are much smaller with more
63 # iterations.
64 pytest.importorskip("lightgbm")
65
66 rng = np.random.RandomState(seed=seed)
67 max_iter = 1
68 max_bins = 255
69
70 X, y = make_regression(
71 n_samples=n_samples, n_features=5, n_informative=5, random_state=0
72 )
73
74 if loss in ("gamma", "poisson"):
75 # make the target positive
76 y = np.abs(y) + np.mean(np.abs(y))
77
78 if n_samples > 255:
79 # bin data and convert it to float32 so that the estimator doesn't
80 # treat it as pre-binned
81 X = _BinMapper(n_bins=max_bins + 1).fit_transform(X).astype(np.float32)
82
83 X_train, X_test, y_train, y_test = train_test_split(X, y, random_state=rng)
84
85 est_sklearn = HistGradientBoostingRegressor(
86 loss=loss,
87 max_iter=max_iter,
88 max_bins=max_bins,
89 learning_rate=1,
90 early_stopping=False,
91 min_samples_leaf=min_samples_leaf,
92 max_leaf_nodes=max_leaf_nodes,
93 )
94 est_lightgbm = get_equivalent_estimator(est_sklearn, lib="lightgbm")
95 est_lightgbm.set_params(min_sum_hessian_in_leaf=0)
96
97 est_lightgbm.fit(X_train, y_train)
98 est_sklearn.fit(X_train, y_train)
99
100 # We need X to be treated an numerical data, not pre-binned data.
101 X_train, X_test = X_train.astype(np.float32), X_test.astype(np.float32)
102
103 pred_lightgbm = est_lightgbm.predict(X_train)
104 pred_sklearn = est_sklearn.predict(X_train)
105 if loss in ("gamma", "poisson"):
106 # More than 65% of the predictions must be close up to the 2nd decimal.
107 # TODO: We are not entirely satisfied with this lax comparison, but the root
108 # cause is not clear, maybe algorithmic differences. One such example is the
109 # poisson_max_delta_step parameter of LightGBM which does not exist in HGBT.
110 assert (
111 np.mean(np.isclose(pred_lightgbm, pred_sklearn, rtol=1e-2, atol=1e-2))
112 > 0.65
113 )
114 else:
115 # Less than 1% of the predictions may deviate more than 1e-3 in relative terms.
116 assert np.mean(np.isclose(pred_lightgbm, pred_sklearn, rtol=1e-3)) > 1 - 0.01
117
118 if max_leaf_nodes < 10 and n_samples >= 1000 and loss in ("squared_error",):
119 pred_lightgbm = est_lightgbm.predict(X_test)
120 pred_sklearn = est_sklearn.predict(X_test)
121 # Less than 1% of the predictions may deviate more than 1e-4 in relative terms.
122 assert np.mean(np.isclose(pred_lightgbm, pred_sklearn, rtol=1e-4)) > 1 - 0.01
