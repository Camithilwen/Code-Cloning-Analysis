226 def test_sparse_ill_conditioned(self):
227 # Sparse matrix with condition number of ~4 million
228 data = np.array([1., 1., 1., 1. + 1e-6, 1.])
229 row = np.array([0, 0, 1, 2, 2])
230 col = np.array([0, 2, 1, 0, 2])
231 A = coo_matrix((data, (row, col)), shape=(3, 3))
232
233 # Get the exact solution
234 exact_sol = lsq_linear(A.toarray(), b, lsq_solver='exact')
235
236 # Default lsmr arguments should not fully converge the solution
237 default_lsmr_sol = lsq_linear(A, b, lsq_solver='lsmr')
238 with pytest.raises(AssertionError, match=""):
239 assert_allclose(exact_sol.x, default_lsmr_sol.x)
240
241 # By increasing the maximum lsmr iters, it will converge
242 conv_lsmr = lsq_linear(A, b, lsq_solver='lsmr', lsmr_maxiter=10)
