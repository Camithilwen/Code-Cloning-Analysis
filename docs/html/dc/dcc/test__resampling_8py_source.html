223def test_multisample_BCa_against_R():
224 # Because bootstrap is stochastic, it's tricky to test against reference
225 # behavior. Here, we show that SciPy's BCa CI matches R wboot's BCa CI
226 # much more closely than the other SciPy CIs do.
227
228 # arbitrary skewed data
229 x = [0.75859206, 0.5910282, -0.4419409, -0.36654601,
230 0.34955357, -1.38835871, 0.76735821]
231 y = [1.41186073, 0.49775975, 0.08275588, 0.24086388,
232 0.03567057, 0.52024419, 0.31966611, 1.32067634]
233
234 # a multi-sample statistic for which the BCa CI tends to be different
235 # from the other CIs
236 def statistic(x, y, axis):
237 s1 = stats.skew(x, axis=axis)
238 s2 = stats.skew(y, axis=axis)
239 return s1 - s2
240
241 # compute confidence intervals using each method
242 rng = np.random.default_rng(468865032284792692)
243
244 res_basic = stats.bootstrap((x, y), statistic, method='basic',
245 batch=100, random_state=rng)
246 res_percent = stats.bootstrap((x, y), statistic, method='percentile',
247 batch=100, random_state=rng)
248 res_bca = stats.bootstrap((x, y), statistic, method='bca',
249 batch=100, random_state=rng)
250
251 # compute midpoints so we can compare just one number for each
252 mid_basic = np.mean(res_basic.confidence_interval)
253 mid_percent = np.mean(res_percent.confidence_interval)
254 mid_bca = np.mean(res_bca.confidence_interval)
255
256 # reference for BCA CI computed using R wboot package:
257 # library(wBoot)
258 # library(moments)
259
260 # x = c(0.75859206, 0.5910282, -0.4419409, -0.36654601,
261 # 0.34955357, -1.38835871, 0.76735821)
262 # y = c(1.41186073, 0.49775975, 0.08275588, 0.24086388,
263 # 0.03567057, 0.52024419, 0.31966611, 1.32067634)
264
265 # twoskew <- function(x1, y1) {skewness(x1) - skewness(y1)}
266 # boot.two.bca(x, y, skewness, conf.level = 0.95,
267 # R = 9999, stacked = FALSE)
268 mid_wboot = -1.5519
269
270 # compute percent difference relative to wboot BCA method
271 diff_basic = (mid_basic - mid_wboot)/abs(mid_wboot)
272 diff_percent = (mid_percent - mid_wboot)/abs(mid_wboot)
273 diff_bca = (mid_bca - mid_wboot)/abs(mid_wboot)
274
275 # SciPy's BCa CI midpoint is much closer than that of the other methods
276 assert diff_basic < -0.15
277 assert diff_percent > 0.15
278 assert abs(diff_bca) < 0.03
