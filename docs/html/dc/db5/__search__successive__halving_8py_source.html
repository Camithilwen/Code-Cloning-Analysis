32 def split(self, X, y, **kwargs):
33 for train_idx, test_idx in self.base_cv.split(X, y, **kwargs):
34 train_idx = resample(
35 train_idx,
36 replace=False,
37 random_state=self.random_state,
38 n_samples=int(self.fraction * len(train_idx)),
39 )
40 if self.subsample_test:
41 test_idx = resample(
42 test_idx,
43 replace=False,
44 random_state=self.random_state,
45 n_samples=int(self.fraction * len(test_idx)),
46 )
47 yield train_idx, test_idx

50def _top_k(results, k, itr):
51 # Return the best candidates of a given iteration
52 iteration, mean_test_score, params = (
53 np.asarray(a)
54 for a in (results["iter"], results["mean_test_score"], results["params"])
55 )
56 iter_indices = np.flatnonzero(iteration == itr)
57 scores = mean_test_score[iter_indices]
58 # argsort() places NaNs at the end of the array so we move NaNs to the
59 # front of the array so the last `k` items are the those with the
60 # highest scores.
61 sorted_indices = np.roll(np.argsort(scores), np.count_nonzero(np.isnan(scores)))
62 return np.array(params[iter_indices][sorted_indices[-k:]])

128 def _check_input_parameters(self, X, y, split_params):
129 # We need to enforce that successive calls to cv.split() yield the same
130 # splits: see https://github.com/scikit-learn/scikit-learn/issues/15149
131 if not _yields_constant_splits(self._checked_cv_orig):
132 raise ValueError(
133 "The cv parameter must yield consistent folds across "
134 "calls to split(). Set its random_state to an int, or set "
135 "shuffle=False."
136 )
137
138 if (
139 self.resource != "n_samples"
140 and self.resource not in self.estimator.get_params()
141 ):
142 raise ValueError(
143 f"Cannot use resource={self.resource} which is not supported "
144 f"by estimator {self.estimator.__class__.__name__}"
145 )
146
147 if isinstance(self, HalvingRandomSearchCV):
148 if self.min_resources == self.n_candidates == "exhaust":
149 # for n_candidates=exhaust to work, we need to know what
150 # min_resources is. Similarly min_resources=exhaust needs to
151 # know the actual number of candidates.
152 raise ValueError(
153 "n_candidates and min_resources cannot be both set to 'exhaust'."
154 )
155
156 self.min_resources_ = self.min_resources
157 if self.min_resources_ in ("smallest", "exhaust"):
158 if self.resource == "n_samples":
159 n_splits = self._checked_cv_orig.get_n_splits(X, y, **split_params)
160 # please see https://gph.is/1KjihQe for a justification
161 magic_factor = 2
162 self.min_resources_ = n_splits * magic_factor
163 if is_classifier(self.estimator):
164 y = validate_data(self, X="no_validation", y=y)
165 check_classification_targets(y)
166 n_classes = np.unique(y).shape[0]
167 self.min_resources_ *= n_classes
168 else:
169 self.min_resources_ = 1
170 # if 'exhaust', min_resources_ might be set to a higher value later
171 # in _run_search
172
173 self.max_resources_ = self.max_resources
174 if self.max_resources_ == "auto":
175 if not self.resource == "n_samples":
176 raise ValueError(
177 "resource can only be 'n_samples' when max_resources='auto'"
178 )
179 self.max_resources_ = _num_samples(X)
180
181 if self.min_resources_ > self.max_resources_:
182 raise ValueError(
183 f"min_resources_={self.min_resources_} is greater "
184 f"than max_resources_={self.max_resources_}."
185 )
186
187 if self.min_resources_ == 0:
188 raise ValueError(
189 f"min_resources_={self.min_resources_}: you might have passed "
190 "an empty dataset X."
191 )

193 @staticmethod
194 def _select_best_index(refit, refit_metric, results):
195 """Custom refit callable to return the index of the best candidate.
196
197 We want the best candidate out of the last iteration. By default
198 BaseSearchCV would return the best candidate out of all iterations.
199
200 Currently, we only support for a single metric thus `refit` and
201 `refit_metric` are not required.
202 """
203 last_iter = np.max(results["iter"])
204 last_iter_indices = np.flatnonzero(results["iter"] == last_iter)
205
206 test_scores = results["mean_test_score"][last_iter_indices]
207 # If all scores are NaNs there is no way to pick between them,
208 # so we (arbitrarily) declare the zero'th entry the best one
209 if np.isnan(test_scores).all():
210 best_idx = 0
211 else:
212 best_idx = np.nanargmax(test_scores)
213
214 return last_iter_indices[best_idx]

220 def fit(self, X, y=None, **params):
221 """Run fit with all sets of parameters.
222
223 Parameters
224 ----------
225
226 X : array-like, shape (n_samples, n_features)
227 Training vector, where `n_samples` is the number of samples and
228 `n_features` is the number of features.
229
230 y : array-like, shape (n_samples,) or (n_samples, n_output), optional
231 Target relative to X for classification or regression;
232 None for unsupervised learning.
233
234 **params : dict of string -> object
235 Parameters passed to the ``fit`` method of the estimator.
236
237 Returns
238 -------
239 self : object
240 Instance of fitted estimator.
241 """
242 self._checked_cv_orig = check_cv(
243 self.cv, y, classifier=is_classifier(self.estimator)
244 )
245
246 routed_params = self._get_routed_params_for_fit(params)
247 self._check_input_parameters(
248 X=X, y=y, split_params=routed_params.splitter.split
249 )
250
251 self._n_samples_orig = _num_samples(X)
252
253 super().fit(X, y=y, **params)
254
255 # Set best_score_: BaseSearchCV does not set it, as refit is a callable
256 self.best_score_ = self.cv_results_["mean_test_score"][self.best_index_]
257
258 return self

260 def _run_search(self, evaluate_candidates):
261 candidate_params = self._generate_candidate_params()
262
263 if self.resource != "n_samples" and any(
264 self.resource in candidate for candidate in candidate_params
265 ):
266 # Can only check this now since we need the candidates list
267 raise ValueError(
268 f"Cannot use parameter {self.resource} as the resource since "
269 "it is part of the searched parameters."
270 )
271
272 # n_required_iterations is the number of iterations needed so that the
273 # last iterations evaluates less than `factor` candidates.
274 n_required_iterations = 1 + floor(log(len(candidate_params), self.factor))
275
276 if self.min_resources == "exhaust":
277 # To exhaust the resources, we want to start with the biggest
278 # min_resources possible so that the last (required) iteration
279 # uses as many resources as possible
280 last_iteration = n_required_iterations - 1
281 self.min_resources_ = max(
282 self.min_resources_,
283 self.max_resources_ // self.factor**last_iteration,
284 )
285
286 # n_possible_iterations is the number of iterations that we can
287 # actually do starting from min_resources and without exceeding
288 # max_resources. Depending on max_resources and the number of
289 # candidates, this may be higher or smaller than
290 # n_required_iterations.
291 n_possible_iterations = 1 + floor(
292 log(self.max_resources_ // self.min_resources_, self.factor)
293 )
294
295 if self.aggressive_elimination:
296 n_iterations = n_required_iterations
297 else:
298 n_iterations = min(n_possible_iterations, n_required_iterations)
299
300 if self.verbose:
301 print(f"n_iterations: {n_iterations}")
302 print(f"n_required_iterations: {n_required_iterations}")
303 print(f"n_possible_iterations: {n_possible_iterations}")
304 print(f"min_resources_: {self.min_resources_}")
305 print(f"max_resources_: {self.max_resources_}")
306 print(f"aggressive_elimination: {self.aggressive_elimination}")
307 print(f"factor: {self.factor}")
308
309 self.n_resources_ = []
310 self.n_candidates_ = []
311
312 for itr in range(n_iterations):
313 power = itr # default
314 if self.aggressive_elimination:
315 # this will set n_resources to the initial value (i.e. the
316 # value of n_resources at the first iteration) for as many
317 # iterations as needed (while candidates are being
318 # eliminated), and then go on as usual.
319 power = max(0, itr - n_required_iterations + n_possible_iterations)
320
321 n_resources = int(self.factor**power * self.min_resources_)
322 # guard, probably not needed
323 n_resources = min(n_resources, self.max_resources_)
324 self.n_resources_.append(n_resources)
325
326 n_candidates = len(candidate_params)
327 self.n_candidates_.append(n_candidates)
328
329 if self.verbose:
330 print("-" * 10)
331 print(f"iter: {itr}")
332 print(f"n_candidates: {n_candidates}")
333 print(f"n_resources: {n_resources}")
334
335 if self.resource == "n_samples":
336 # subsampling will be done in cv.split()
337 cv = _SubsampleMetaSplitter(
338 base_cv=self._checked_cv_orig,
339 fraction=n_resources / self._n_samples_orig,
340 subsample_test=True,
341 random_state=self.random_state,
342 )
343
344 else:
345 # Need copy so that the n_resources of next iteration does
346 # not overwrite
347 candidate_params = [c.copy() for c in candidate_params]
348 for candidate in candidate_params:
349 candidate[self.resource] = n_resources
350 cv = self._checked_cv_orig
351
352 more_results = {
353 "iter": [itr] * n_candidates,
354 "n_resources": [n_resources] * n_candidates,
355 }
356
357 results = evaluate_candidates(
358 candidate_params, cv, more_results=more_results
359 )
360
361 n_candidates_to_keep = ceil(n_candidates / self.factor)
362 candidate_params = _top_k(results, n_candidates_to_keep, itr)
363
364 self.n_remaining_candidates_ = len(candidate_params)
365 self.n_required_iterations_ = n_required_iterations
366 self.n_possible_iterations_ = n_possible_iterations
367 self.n_iterations_ = n_iterations

374class HalvingGridSearchCV(BaseSuccessiveHalving):
375 """Search over specified parameter values with successive halving.
376
377 The search strategy starts evaluating all the candidates with a small
378 amount of resources and iteratively selects the best candidates, using
379 more and more resources.
380
381 Read more in the :ref:`User guide <successive_halving_user_guide>`.
382
383 .. note::
384
385 This estimator is still **experimental** for now: the predictions
386 and the API might change without any deprecation cycle. To use it,
387 you need to explicitly import ``enable_halving_search_cv``::
388
389 >>> # explicitly require this experimental feature
390 >>> from sklearn.experimental import enable_halving_search_cv # noqa
391 >>> # now you can import normally from model_selection
392 >>> from sklearn.model_selection import HalvingGridSearchCV
393
394 Parameters
395 ----------
396 estimator : estimator object
397 This is assumed to implement the scikit-learn estimator interface.
398 Either estimator needs to provide a ``score`` function,
399 or ``scoring`` must be passed.
400
401 param_grid : dict or list of dictionaries
402 Dictionary with parameters names (string) as keys and lists of
403 parameter settings to try as values, or a list of such
404 dictionaries, in which case the grids spanned by each dictionary
405 in the list are explored. This enables searching over any sequence
406 of parameter settings.
407
408 factor : int or float, default=3
409 The 'halving' parameter, which determines the proportion of candidates
410 that are selected for each subsequent iteration. For example,
411 ``factor=3`` means that only one third of the candidates are selected.
412
413 resource : ``'n_samples'`` or str, default='n_samples'
414 Defines the resource that increases with each iteration. By default,
415 the resource is the number of samples. It can also be set to any
416 parameter of the base estimator that accepts positive integer
417 values, e.g. 'n_iterations' or 'n_estimators' for a gradient
418 boosting estimator. In this case ``max_resources`` cannot be 'auto'
419 and must be set explicitly.
420
421 max_resources : int, default='auto'
422 The maximum amount of resource that any candidate is allowed to use
423 for a given iteration. By default, this is set to ``n_samples`` when
424 ``resource='n_samples'`` (default), else an error is raised.
425
426 min_resources : {'exhaust', 'smallest'} or int, default='exhaust'
427 The minimum amount of resource that any candidate is allowed to use
428 for a given iteration. Equivalently, this defines the amount of
429 resources `r0` that are allocated for each candidate at the first
430 iteration.
431
432 - 'smallest' is a heuristic that sets `r0` to a small value:
433
434 - ``n_splits * 2`` when ``resource='n_samples'`` for a regression problem
435 - ``n_classes * n_splits * 2`` when ``resource='n_samples'`` for a
436 classification problem
437 - ``1`` when ``resource != 'n_samples'``
438
439 - 'exhaust' will set `r0` such that the **last** iteration uses as
440 much resources as possible. Namely, the last iteration will use the
441 highest value smaller than ``max_resources`` that is a multiple of
442 both ``min_resources`` and ``factor``. In general, using 'exhaust'
443 leads to a more accurate estimator, but is slightly more time
444 consuming.
445
446 Note that the amount of resources used at each iteration is always a
447 multiple of ``min_resources``.
448
449 aggressive_elimination : bool, default=False
450 This is only relevant in cases where there isn't enough resources to
451 reduce the remaining candidates to at most `factor` after the last
452 iteration. If ``True``, then the search process will 'replay' the
453 first iteration for as long as needed until the number of candidates
454 is small enough. This is ``False`` by default, which means that the
455 last iteration may evaluate more than ``factor`` candidates. See
456 :ref:`aggressive_elimination` for more details.
457
458 cv : int, cross-validation generator or iterable, default=5
459 Determines the cross-validation splitting strategy.
460 Possible inputs for cv are:
461
462 - integer, to specify the number of folds in a `(Stratified)KFold`,
463 - :term:`CV splitter`,
464 - An iterable yielding (train, test) splits as arrays of indices.
465
466 For integer/None inputs, if the estimator is a classifier and ``y`` is
467 either binary or multiclass, :class:`StratifiedKFold` is used. In all
468 other cases, :class:`KFold` is used. These splitters are instantiated
469 with `shuffle=False` so the splits will be the same across calls.
470
471 Refer :ref:`User Guide <cross_validation>` for the various
472 cross-validation strategies that can be used here.
473
474 .. note::
475 Due to implementation details, the folds produced by `cv` must be
476 the same across multiple calls to `cv.split()`. For
477 built-in `scikit-learn` iterators, this can be achieved by
478 deactivating shuffling (`shuffle=False`), or by setting the
479 `cv`'s `random_state` parameter to an integer.
480
481 scoring : str or callable, default=None
482 Scoring method to use to evaluate the predictions on the test set.
483
484 - str: see :ref:`scoring_string_names` for options.
485 - callable: a scorer callable object (e.g., function) with signature
486 ``scorer(estimator, X, y)``. See :ref:`scoring_callable` for details.
487 - `None`: the `estimator`'s
488 :ref:`default evaluation criterion <scoring_api_overview>` is used.
489
490 refit : bool or callable, default=True
491 Refit an estimator using the best found parameters on the whole
492 dataset.
493
494 Where there are considerations other than maximum score in
495 choosing a best estimator, ``refit`` can be set to a function which
496 returns the selected ``best_index_`` given ``cv_results_``. In that
497 case, the ``best_estimator_`` and ``best_params_`` will be set
498 according to the returned ``best_index_`` while the ``best_score_``
499 attribute will not be available.
500
501 The refitted estimator is made available at the ``best_estimator_``
502 attribute and permits using ``predict`` directly on this
503 ``HalvingGridSearchCV`` instance.
504
505 See :ref:`this example
506 <sphx_glr_auto_examples_model_selection_plot_grid_search_refit_callable.py>`
507 for an example of how to use ``refit=callable`` to balance model
508 complexity and cross-validated score.
509
510 error_score : 'raise' or numeric
511 Value to assign to the score if an error occurs in estimator fitting.
512 If set to 'raise', the error is raised. If a numeric value is given,
513 FitFailedWarning is raised. This parameter does not affect the refit
514 step, which will always raise the error. Default is ``np.nan``.
515
516 return_train_score : bool, default=False
517 If ``False``, the ``cv_results_`` attribute will not include training
518 scores.
519 Computing training scores is used to get insights on how different
520 parameter settings impact the overfitting/underfitting trade-off.
521 However computing the scores on the training set can be computationally
522 expensive and is not strictly required to select the parameters that
523 yield the best generalization performance.
524
525 random_state : int, RandomState instance or None, default=None
526 Pseudo random number generator state used for subsampling the dataset
527 when `resources != 'n_samples'`. Ignored otherwise.
528 Pass an int for reproducible output across multiple function calls.
529 See :term:`Glossary <random_state>`.
530
531 n_jobs : int or None, default=None
532 Number of jobs to run in parallel.
533 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
534 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
535 for more details.
536
537 verbose : int
538 Controls the verbosity: the higher, the more messages.
539
540 Attributes
541 ----------
542 n_resources_ : list of int
543 The amount of resources used at each iteration.
544
545 n_candidates_ : list of int
546 The number of candidate parameters that were evaluated at each
547 iteration.
548
549 n_remaining_candidates_ : int
550 The number of candidate parameters that are left after the last
551 iteration. It corresponds to `ceil(n_candidates[-1] / factor)`
552
553 max_resources_ : int
554 The maximum number of resources that any candidate is allowed to use
555 for a given iteration. Note that since the number of resources used
556 at each iteration must be a multiple of ``min_resources_``, the
557 actual number of resources used at the last iteration may be smaller
558 than ``max_resources_``.
559
560 min_resources_ : int
561 The amount of resources that are allocated for each candidate at the
562 first iteration.
563
564 n_iterations_ : int
565 The actual number of iterations that were run. This is equal to
566 ``n_required_iterations_`` if ``aggressive_elimination`` is ``True``.
567 Else, this is equal to ``min(n_possible_iterations_,
568 n_required_iterations_)``.
569
570 n_possible_iterations_ : int
571 The number of iterations that are possible starting with
572 ``min_resources_`` resources and without exceeding
573 ``max_resources_``.
574
575 n_required_iterations_ : int
576 The number of iterations that are required to end up with less than
577 ``factor`` candidates at the last iteration, starting with
578 ``min_resources_`` resources. This will be smaller than
579 ``n_possible_iterations_`` when there isn't enough resources.
580
581 cv_results_ : dict of numpy (masked) ndarrays
582 A dict with keys as column headers and values as columns, that can be
583 imported into a pandas ``DataFrame``. It contains lots of information
584 for analysing the results of a search.
585 Please refer to the :ref:`User guide<successive_halving_cv_results>`
586 for details.
587
588 best_estimator_ : estimator or dict
589 Estimator that was chosen by the search, i.e. estimator
590 which gave highest score (or smallest loss if specified)
591 on the left out data. Not available if ``refit=False``.
592
593 best_score_ : float
594 Mean cross-validated score of the best_estimator.
595
596 best_params_ : dict
597 Parameter setting that gave the best results on the hold out data.
598
599 best_index_ : int
600 The index (of the ``cv_results_`` arrays) which corresponds to the best
601 candidate parameter setting.
602
603 The dict at ``search.cv_results_['params'][search.best_index_]`` gives
604 the parameter setting for the best model, that gives the highest
605 mean score (``search.best_score_``).
606
607 scorer_ : function or a dict
608 Scorer function used on the held out data to choose the best
609 parameters for the model.
610
611 n_splits_ : int
612 The number of cross-validation splits (folds/iterations).
613
614 refit_time_ : float
615 Seconds used for refitting the best model on the whole dataset.
616
617 This is present only if ``refit`` is not False.
618
619 multimetric_ : bool
620 Whether or not the scorers compute several metrics.
621
622 classes_ : ndarray of shape (n_classes,)
623 The classes labels. This is present only if ``refit`` is specified and
624 the underlying estimator is a classifier.
625
626 n_features_in_ : int
627 Number of features seen during :term:`fit`. Only defined if
628 `best_estimator_` is defined (see the documentation for the `refit`
629 parameter for more details) and that `best_estimator_` exposes
630 `n_features_in_` when fit.
631
632 .. versionadded:: 0.24
633
634 feature_names_in_ : ndarray of shape (`n_features_in_`,)
635 Names of features seen during :term:`fit`. Only defined if
636 `best_estimator_` is defined (see the documentation for the `refit`
637 parameter for more details) and that `best_estimator_` exposes
638 `feature_names_in_` when fit.
639
640 .. versionadded:: 1.0
641
642 See Also
643 --------
644 :class:`HalvingRandomSearchCV`:
645 Random search over a set of parameters using successive halving.
646
647 Notes
648 -----
649 The parameters selected are those that maximize the score of the held-out
650 data, according to the scoring parameter.
651
652 All parameter combinations scored with a NaN will share the lowest rank.
653
654 Examples
655 --------
656
657 >>> from sklearn.datasets import load_iris
658 >>> from sklearn.ensemble import RandomForestClassifier
659 >>> from sklearn.experimental import enable_halving_search_cv # noqa
660 >>> from sklearn.model_selection import HalvingGridSearchCV
661 ...
662 >>> X, y = load_iris(return_X_y=True)
663 >>> clf = RandomForestClassifier(random_state=0)
664 ...
665 >>> param_grid = {"max_depth": [3, None],
666 ... "min_samples_split": [5, 10]}
667 >>> search = HalvingGridSearchCV(clf, param_grid, resource='n_estimators',
668 ... max_resources=10,
669 ... random_state=0).fit(X, y)
670 >>> search.best_params_ # doctest: +SKIP
671 {'max_depth': None, 'min_samples_split': 10, 'n_estimators': 9}
672 """

720class HalvingRandomSearchCV(BaseSuccessiveHalving):
721 """Randomized search on hyper parameters.
722
723 The search strategy starts evaluating all the candidates with a small
724 amount of resources and iteratively selects the best candidates, using more
725 and more resources.
726
727 The candidates are sampled at random from the parameter space and the
728 number of sampled candidates is determined by ``n_candidates``.
729
730 Read more in the :ref:`User guide<successive_halving_user_guide>`.
731
732 .. note::
733
734 This estimator is still **experimental** for now: the predictions
735 and the API might change without any deprecation cycle. To use it,
736 you need to explicitly import ``enable_halving_search_cv``::
737
738 >>> # explicitly require this experimental feature
739 >>> from sklearn.experimental import enable_halving_search_cv # noqa
740 >>> # now you can import normally from model_selection
741 >>> from sklearn.model_selection import HalvingRandomSearchCV
742
743 Parameters
744 ----------
745 estimator : estimator object
746 This is assumed to implement the scikit-learn estimator interface.
747 Either estimator needs to provide a ``score`` function,
748 or ``scoring`` must be passed.
749
750 param_distributions : dict or list of dicts
751 Dictionary with parameters names (`str`) as keys and distributions
752 or lists of parameters to try. Distributions must provide a ``rvs``
753 method for sampling (such as those from scipy.stats.distributions).
754 If a list is given, it is sampled uniformly.
755 If a list of dicts is given, first a dict is sampled uniformly, and
756 then a parameter is sampled using that dict as above.
757
758 n_candidates : "exhaust" or int, default="exhaust"
759 The number of candidate parameters to sample, at the first
760 iteration. Using 'exhaust' will sample enough candidates so that the
761 last iteration uses as many resources as possible, based on
762 `min_resources`, `max_resources` and `factor`. In this case,
763 `min_resources` cannot be 'exhaust'.
764
765 factor : int or float, default=3
766 The 'halving' parameter, which determines the proportion of candidates
767 that are selected for each subsequent iteration. For example,
768 ``factor=3`` means that only one third of the candidates are selected.
769
770 resource : ``'n_samples'`` or str, default='n_samples'
771 Defines the resource that increases with each iteration. By default,
772 the resource is the number of samples. It can also be set to any
773 parameter of the base estimator that accepts positive integer
774 values, e.g. 'n_iterations' or 'n_estimators' for a gradient
775 boosting estimator. In this case ``max_resources`` cannot be 'auto'
776 and must be set explicitly.
777
778 max_resources : int, default='auto'
779 The maximum number of resources that any candidate is allowed to use
780 for a given iteration. By default, this is set ``n_samples`` when
781 ``resource='n_samples'`` (default), else an error is raised.
782
783 min_resources : {'exhaust', 'smallest'} or int, default='smallest'
784 The minimum amount of resource that any candidate is allowed to use
785 for a given iteration. Equivalently, this defines the amount of
786 resources `r0` that are allocated for each candidate at the first
787 iteration.
788
789 - 'smallest' is a heuristic that sets `r0` to a small value:
790
791 - ``n_splits * 2`` when ``resource='n_samples'`` for a regression problem
792 - ``n_classes * n_splits * 2`` when ``resource='n_samples'`` for a
793 classification problem
794 - ``1`` when ``resource != 'n_samples'``
795
796 - 'exhaust' will set `r0` such that the **last** iteration uses as
797 much resources as possible. Namely, the last iteration will use the
798 highest value smaller than ``max_resources`` that is a multiple of
799 both ``min_resources`` and ``factor``. In general, using 'exhaust'
800 leads to a more accurate estimator, but is slightly more time
801 consuming. 'exhaust' isn't available when `n_candidates='exhaust'`.
802
803 Note that the amount of resources used at each iteration is always a
804 multiple of ``min_resources``.
805
806 aggressive_elimination : bool, default=False
807 This is only relevant in cases where there isn't enough resources to
808 reduce the remaining candidates to at most `factor` after the last
809 iteration. If ``True``, then the search process will 'replay' the
810 first iteration for as long as needed until the number of candidates
811 is small enough. This is ``False`` by default, which means that the
812 last iteration may evaluate more than ``factor`` candidates. See
813 :ref:`aggressive_elimination` for more details.
814
815 cv : int, cross-validation generator or an iterable, default=5
816 Determines the cross-validation splitting strategy.
817 Possible inputs for cv are:
818
819 - integer, to specify the number of folds in a `(Stratified)KFold`,
820 - :term:`CV splitter`,
821 - An iterable yielding (train, test) splits as arrays of indices.
822
823 For integer/None inputs, if the estimator is a classifier and ``y`` is
824 either binary or multiclass, :class:`StratifiedKFold` is used. In all
825 other cases, :class:`KFold` is used. These splitters are instantiated
826 with `shuffle=False` so the splits will be the same across calls.
827
828 Refer :ref:`User Guide <cross_validation>` for the various
829 cross-validation strategies that can be used here.
830
831 .. note::
832 Due to implementation details, the folds produced by `cv` must be
833 the same across multiple calls to `cv.split()`. For
834 built-in `scikit-learn` iterators, this can be achieved by
835 deactivating shuffling (`shuffle=False`), or by setting the
836 `cv`'s `random_state` parameter to an integer.
837
838 scoring : str or callable, default=None
839 Scoring method to use to evaluate the predictions on the test set.
840
841 - str: see :ref:`scoring_string_names` for options.
842 - callable: a scorer callable object (e.g., function) with signature
843 ``scorer(estimator, X, y)``. See :ref:`scoring_callable` for details.
844 - `None`: the `estimator`'s
845 :ref:`default evaluation criterion <scoring_api_overview>` is used.
846
847 refit : bool or callable, default=True
848 Refit an estimator using the best found parameters on the whole
849 dataset.
850
851 Where there are considerations other than maximum score in
852 choosing a best estimator, ``refit`` can be set to a function which
853 returns the selected ``best_index_`` given ``cv_results_``. In that
854 case, the ``best_estimator_`` and ``best_params_`` will be set
855 according to the returned ``best_index_`` while the ``best_score_``
856 attribute will not be available.
857
858 The refitted estimator is made available at the ``best_estimator_``
859 attribute and permits using ``predict`` directly on this
860 ``HalvingRandomSearchCV`` instance.
861
862 See :ref:`this example
863 <sphx_glr_auto_examples_model_selection_plot_grid_search_refit_callable.py>`
864 for an example of how to use ``refit=callable`` to balance model
865 complexity and cross-validated score.
866
867 error_score : 'raise' or numeric
868 Value to assign to the score if an error occurs in estimator fitting.
869 If set to 'raise', the error is raised. If a numeric value is given,
870 FitFailedWarning is raised. This parameter does not affect the refit
871 step, which will always raise the error. Default is ``np.nan``.
872
873 return_train_score : bool, default=False
874 If ``False``, the ``cv_results_`` attribute will not include training
875 scores.
876 Computing training scores is used to get insights on how different
877 parameter settings impact the overfitting/underfitting trade-off.
878 However computing the scores on the training set can be computationally
879 expensive and is not strictly required to select the parameters that
880 yield the best generalization performance.
881
882 random_state : int, RandomState instance or None, default=None
883 Pseudo random number generator state used for subsampling the dataset
884 when `resources != 'n_samples'`. Also used for random uniform
885 sampling from lists of possible values instead of scipy.stats
886 distributions.
887 Pass an int for reproducible output across multiple function calls.
888 See :term:`Glossary <random_state>`.
889
890 n_jobs : int or None, default=None
891 Number of jobs to run in parallel.
892 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
893 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
894 for more details.
895
896 verbose : int
897 Controls the verbosity: the higher, the more messages.
898
899 Attributes
900 ----------
901 n_resources_ : list of int
902 The amount of resources used at each iteration.
903
904 n_candidates_ : list of int
905 The number of candidate parameters that were evaluated at each
906 iteration.
907
908 n_remaining_candidates_ : int
909 The number of candidate parameters that are left after the last
910 iteration. It corresponds to `ceil(n_candidates[-1] / factor)`
911
912 max_resources_ : int
913 The maximum number of resources that any candidate is allowed to use
914 for a given iteration. Note that since the number of resources used at
915 each iteration must be a multiple of ``min_resources_``, the actual
916 number of resources used at the last iteration may be smaller than
917 ``max_resources_``.
918
919 min_resources_ : int
920 The amount of resources that are allocated for each candidate at the
921 first iteration.
922
923 n_iterations_ : int
924 The actual number of iterations that were run. This is equal to
925 ``n_required_iterations_`` if ``aggressive_elimination`` is ``True``.
926 Else, this is equal to ``min(n_possible_iterations_,
927 n_required_iterations_)``.
928
929 n_possible_iterations_ : int
930 The number of iterations that are possible starting with
931 ``min_resources_`` resources and without exceeding
932 ``max_resources_``.
933
934 n_required_iterations_ : int
935 The number of iterations that are required to end up with less than
936 ``factor`` candidates at the last iteration, starting with
937 ``min_resources_`` resources. This will be smaller than
938 ``n_possible_iterations_`` when there isn't enough resources.
939
940 cv_results_ : dict of numpy (masked) ndarrays
941 A dict with keys as column headers and values as columns, that can be
942 imported into a pandas ``DataFrame``. It contains lots of information
943 for analysing the results of a search.
944 Please refer to the :ref:`User guide<successive_halving_cv_results>`
945 for details.
946
947 best_estimator_ : estimator or dict
948 Estimator that was chosen by the search, i.e. estimator
949 which gave highest score (or smallest loss if specified)
950 on the left out data. Not available if ``refit=False``.
951
952 best_score_ : float
953 Mean cross-validated score of the best_estimator.
954
955 best_params_ : dict
956 Parameter setting that gave the best results on the hold out data.
957
958 best_index_ : int
959 The index (of the ``cv_results_`` arrays) which corresponds to the best
960 candidate parameter setting.
961
962 The dict at ``search.cv_results_['params'][search.best_index_]`` gives
963 the parameter setting for the best model, that gives the highest
964 mean score (``search.best_score_``).
965
966 scorer_ : function or a dict
967 Scorer function used on the held out data to choose the best
968 parameters for the model.
969
970 n_splits_ : int
971 The number of cross-validation splits (folds/iterations).
972
973 refit_time_ : float
974 Seconds used for refitting the best model on the whole dataset.
975
976 This is present only if ``refit`` is not False.
977
978 multimetric_ : bool
979 Whether or not the scorers compute several metrics.
980
981 classes_ : ndarray of shape (n_classes,)
982 The classes labels. This is present only if ``refit`` is specified and
983 the underlying estimator is a classifier.
984
985 n_features_in_ : int
986 Number of features seen during :term:`fit`. Only defined if
987 `best_estimator_` is defined (see the documentation for the `refit`
988 parameter for more details) and that `best_estimator_` exposes
989 `n_features_in_` when fit.
990
991 .. versionadded:: 0.24
992
993 feature_names_in_ : ndarray of shape (`n_features_in_`,)
994 Names of features seen during :term:`fit`. Only defined if
995 `best_estimator_` is defined (see the documentation for the `refit`
996 parameter for more details) and that `best_estimator_` exposes
997 `feature_names_in_` when fit.
