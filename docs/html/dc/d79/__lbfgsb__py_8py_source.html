102 ``iprint = 99`` print details of every iteration except n-vectors;

104 ``iprint > 100`` print details of every iteration including x and g.

217 """
218 Minimize a scalar function of one or more variables using the L-BFGS-B
219 algorithm.
220
221 Options
222 -------
223 disp : None or int
224 If `disp is None` (the default), then the supplied version of `iprint`
225 is used. If `disp is not None`, then it overrides the supplied version
226 of `iprint` with the behaviour you outlined.
227 maxcor : int
228 The maximum number of variable metric corrections used to
229 define the limited memory matrix. (The limited memory BFGS
230 method does not store the full hessian but uses this many terms
231 in an approximation to it.)
232 ftol : float
233 The iteration stops when ``(f^k -
234 f^{k+1})/max{|f^k|,|f^{k+1}|,1} <= ftol``.
235 gtol : float
236 The iteration will stop when ``max{|proj g_i | i = 1, ..., n}
237 <= gtol`` where ``pg_i`` is the i-th component of the
238 projected gradient.
239 eps : float or ndarray
240 If `jac is None` the absolute step size used for numerical
241 approximation of the jacobian via forward differences.
242 maxfun : int
243 Maximum number of function evaluations. Note that this function
244 may violate the limit because of evaluating gradients by numerical
245 differentiation.
246 maxiter : int
247 Maximum number of iterations.
248 iprint : int, optional
249 Controls the frequency of output. ``iprint < 0`` means no output;
250 ``iprint = 0`` print only one line at the last iteration;
251 ``0 < iprint < 99`` print also f and ``|proj g|`` every iprint iterations;
252 ``iprint = 99`` print details of every iteration except n-vectors;
253 ``iprint = 100`` print also the changes of active set and final x;
254 ``iprint > 100`` print details of every iteration including x and g.
255 maxls : int, optional
256 Maximum number of line search steps (per iteration). Default is 20.
257 finite_diff_rel_step : None or array_like, optional
258 If `jac in ['2-point', '3-point', 'cs']` the relative step size to
259 use for numerical approximation of the jacobian. The absolute step
260 size is computed as ``h = rel_step * sign(x) * max(1, abs(x))``,
261 possibly adjusted to fit into the bounds. For ``method='3-point'``
262 the sign of `h` is ignored. If None (default) then step is selected
263 automatically.
264
265 Notes
266 -----
267 The option `ftol` is exposed via the `scipy.optimize.minimize` interface,
268 but calling `scipy.optimize.fmin_l_bfgs_b` directly exposes `factr`. The
269 relationship between the two is ``ftol = factr * numpy.finfo(float).eps``.
270 I.e., `factr` multiplies the default machine floating-point precision to
271 arrive at `ftol`.
272
273 """
274 _check_unknown_options(unknown_options)
275 m = maxcor
276 pgtol = gtol
277 factr = ftol / np.finfo(float).eps

279 x0 = asarray(x0).ravel()
280 n, = x0.shape

282 if bounds is None:

284 if len(bounds) != n:
285 raise ValueError('length of x0 != length of bounds')

287 # unbounded variables must use None, not +-inf, for optimizer to work properly
288 bounds = [(None if l == -np.inf else l, None if u == np.inf else u) for l, u in bounds]
289 # LBFGSB is sent 'old-style' bounds, 'new-style' bounds are required by

301 if disp is not None:
302 if disp == 0:
303 iprint = -1
304 else:
305 iprint = disp
306
307 sf = _prepare_scalar_function(fun, x0, jac=jac, args=args, epsilon=eps,
308 bounds=new_bounds,
309 finite_diff_rel_step=finite_diff_rel_step)

311 func_and_grad = sf.fun_and_grad

313 fortran_int = _lbfgsb.types.intvar.dtype
314
315 nbd = zeros(n, fortran_int)
316 low_bnd = zeros(n, float64)
317 upper_bnd = zeros(n, float64)
318 bounds_map = {(None, None): 0,
319 (1, None): 1,
320 (1, 1): 2,
321 (None, 1): 3}
322 for i in range(0, n):
323 l, u = bounds[i]
324 if l is not None:
325 low_bnd[i] = l
326 l = 1
327 if u is not None:
328 upper_bnd[i] = u
329 u = 1

332 if not maxls > 0:
333 raise ValueError('maxls must be positive.')

336 f = array(0.0, float64)
337 g = zeros((n,), float64)
338 wa = zeros(2*m*n + 5*n + 11*m*m + 8*m, float64)
339 iwa = zeros(3*n, fortran_int)
340 task = zeros(1, 'S60')
341 csave = zeros(1, 'S60')
342 lsave = zeros(4, fortran_int)
343 isave = zeros(44, fortran_int)
344 dsave = zeros(29, float64)

356 _lbfgsb.setulb(m, x, low_bnd, upper_bnd, nbd, f, g, factr,

362 # Note that interruptions due to maxfun are postponed
363 # until the completion of the current minimization iteration.
364 # Overwrite f and g:
365 f, g = func_and_grad(x)

367 # new iteration

369
370 intermediate_result = OptimizeResult(x=x, fun=f)
371 if _call_callback_maybe_halt(callback, intermediate_result):
372 task[:] = 'STOP: CALLBACK REQUESTED HALT'
373 if n_iterations >= maxiter:
374 task[:] = 'STOP: TOTAL NO. of ITERATIONS REACHED LIMIT'
375 elif sf.nfev > maxfun:
376 task[:] = ('STOP: TOTAL NO. of f AND g EVALUATIONS '
377 'EXCEEDS LIMIT')
378 else:
379 break
380
381 task_str = task.tobytes().strip(b'\x00').strip()
382 if task_str.startswith(b'CONV'):

384 elif sf.nfev > maxfun or n_iterations >= maxiter:
385 warnflag = 1

390 # subroutine in lbfgsb.f. See line 363.
391 s = wa[0: m*n].reshape(m, n)
392 y = wa[m*n: 2*m*n].reshape(m, n)
393

395 # isave(31) = the total number of BFGS updates prior the current iteration;
396 n_bfgs_updates = isave[30]
397
398 n_corrs = min(n_bfgs_updates, maxcor)
399 hess_inv = LbfgsInvHessProduct(s[:n_corrs], y[:n_corrs])
400
401 task_str = task_str.decode()

403 njev=sf.ngev,
404 nit=n_iterations, status=warnflag, message=task_str,
405 x=x, success=(warnflag == 0), hess_inv=hess_inv)

408class LbfgsInvHessProduct(LinearOperator):
409 """Linear operator for the L-BFGS approximate inverse Hessian.
410
411 This operator computes the product of a vector with the approximate inverse
412 of the Hessian of the objective function, using the L-BFGS limited
413 memory approximation to the inverse Hessian, accumulated during the
414 optimization.
415
416 Objects of this class implement the ``scipy.sparse.linalg.LinearOperator``
417 interface.
418
419 Parameters
420 ----------
421 sk : array_like, shape=(n_corr, n)
422 Array of `n_corr` most recent updates to the solution vector.
423 (See [1]).
424 yk : array_like, shape=(n_corr, n)
425 Array of `n_corr` most recent updates to the gradient. (See [1]).
426
427 References
428 ----------
429 .. [1] Nocedal, Jorge. "Updating quasi-Newton matrices with limited
430 storage." Mathematics of computation 35.151 (1980): 773-782.
431
432 """
433
434 def __init__(self, sk, yk):
435 """Construct the operator."""
436 if sk.shape != yk.shape or sk.ndim != 2:
437 raise ValueError('sk and yk must have matching shape, (n_corrs, n)')
438 n_corrs, n = sk.shape
439
440 super().__init__(dtype=np.float64, shape=(n, n))
441
442 self.sk = sk
443 self.yk = yk
444 self.n_corrs = n_corrs
445 self.rho = 1 / np.einsum('ij,ij->i', sk, yk)
446
447 def _matvec(self, x):
448 """Efficient matrix-vector multiply with the BFGS matrices.
449
450 This calculation is described in Section (4) of [1].
451
452 Parameters
453 ----------
454 x : ndarray
455 An array with shape (n,) or (n,1).
456
457 Returns
458 -------
459 y : ndarray
460 The matrix-vector product
461
462 """
463 s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
464 q = np.array(x, dtype=self.dtype, copy=True)
465 if q.ndim == 2 and q.shape[1] == 1:
466 q = q.reshape(-1)
467
468 alpha = np.empty(n_corrs)
469
470 for i in range(n_corrs-1, -1, -1):
471 alpha[i] = rho[i] * np.dot(s[i], q)
472 q = q - alpha[i]*y[i]
473
474 r = q
475 for i in range(n_corrs):
476 beta = rho[i] * np.dot(y[i], r)
477 r = r + s[i] * (alpha[i] - beta)
478
479 return r

481 def todense(self):
482 """Return a dense array representation of this operator.
483
484 Returns
485 -------
486 arr : ndarray, shape=(n, n)
487 An array with the same shape and containing
488 the same data represented by this `LinearOperator`.
489
490 """
491 s, y, n_corrs, rho = self.sk, self.yk, self.n_corrs, self.rho
492 I = np.eye(*self.shape, dtype=self.dtype)
493 Hk = I
494
495 for i in range(n_corrs):
496 A1 = I - s[i][:, np.newaxis] * y[i][np.newaxis, :] * rho[i]
497 A2 = I - y[i][:, np.newaxis] * s[i][np.newaxis, :] * rho[i]
498
499 Hk = np.dot(A1, np.dot(Hk, A2)) + (rho[i] * s[i][:, np.newaxis] *
500 s[i][np.newaxis, :])
501 return Hk
