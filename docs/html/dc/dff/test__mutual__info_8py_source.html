28def test_compute_mi_cc(global_dtype):
29 # For two continuous variables a good approach is to test on bivariate
30 # normal distribution, where mutual information is known.
31
32 # Mean of the distribution, irrelevant for mutual information.
33 mean = np.zeros(2)
34
35 # Setup covariance matrix with correlation coeff. equal 0.5.
36 sigma_1 = 1
37 sigma_2 = 10
38 corr = 0.5
39 cov = np.array(
40 [
41 [sigma_1**2, corr * sigma_1 * sigma_2],
42 [corr * sigma_1 * sigma_2, sigma_2**2],
43 ]
44 )
45
46 # True theoretical mutual information.
47 I_theory = np.log(sigma_1) + np.log(sigma_2) - 0.5 * np.log(np.linalg.det(cov))
48
49 rng = check_random_state(0)
50 Z = rng.multivariate_normal(mean, cov, size=1000).astype(global_dtype, copy=False)
51
52 x, y = Z[:, 0], Z[:, 1]
53
54 # Theory and computed values won't be very close
55 # We here check with a large relative tolerance
56 for n_neighbors in [3, 5, 7]:
57 I_computed = _compute_mi(
58 x, y, x_discrete=False, y_discrete=False, n_neighbors=n_neighbors
59 )
60 assert_allclose(I_computed, I_theory, rtol=1e-1)

63def test_compute_mi_cd(global_dtype):
64 # To test define a joint distribution as follows:
65 # p(x, y) = p(x) p(y | x)
66 # X ~ Bernoulli(p)
67 # (Y | x = 0) ~ Uniform(-1, 1)
68 # (Y | x = 1) ~ Uniform(0, 2)
69
70 # Use the following formula for mutual information:
71 # I(X; Y) = H(Y) - H(Y | X)
72 # Two entropies can be computed by hand:
73 # H(Y) = -(1-p)/2 * ln((1-p)/2) - p/2*log(p/2) - 1/2*log(1/2)
74 # H(Y | X) = ln(2)
75
76 # Now we need to implement sampling from out distribution, which is
77 # done easily using conditional distribution logic.
78
79 n_samples = 1000
80 rng = check_random_state(0)
81
82 for p in [0.3, 0.5, 0.7]:
83 x = rng.uniform(size=n_samples) > p
84
85 y = np.empty(n_samples, global_dtype)
86 mask = x == 0
87 y[mask] = rng.uniform(-1, 1, size=np.sum(mask))
88 y[~mask] = rng.uniform(0, 2, size=np.sum(~mask))
89
90 I_theory = -0.5 * (
91 (1 - p) * np.log(0.5 * (1 - p)) + p * np.log(0.5 * p) + np.log(0.5)
92 ) - np.log(2)
93
94 # Assert the same tolerance.
95 for n_neighbors in [3, 5, 7]:
96 I_computed = _compute_mi(
97 x, y, x_discrete=True, y_discrete=False, n_neighbors=n_neighbors
98 )
99 assert_allclose(I_computed, I_theory, rtol=1e-1)

134def test_mutual_info_regression(global_dtype):
135 # We generate sample from multivariate normal distribution, using
136 # transformation from initially uncorrelated variables. The zero
137 # variables after transformation is selected as the target vector,
138 # it has the strongest correlation with the variable 2, and
139 # the weakest correlation with the variable 1.
140 T = np.array([[1, 0.5, 2, 1], [0, 1, 0.1, 0.0], [0, 0.1, 1, 0.1], [0, 0.1, 0.1, 1]])
141 cov = T.dot(T.T)
142 mean = np.zeros(4)
143
144 rng = check_random_state(0)
145 Z = rng.multivariate_normal(mean, cov, size=1000).astype(global_dtype, copy=False)
146 X = Z[:, 1:]
147 y = Z[:, 0]
148
149 mi = mutual_info_regression(X, y, random_state=0)
150 assert_array_equal(np.argsort(-mi), np.array([1, 2, 0]))
151 # XXX: should mutual_info_regression be fixed to avoid
152 # up-casting float32 inputs to float64?

156def test_mutual_info_classif_mixed(global_dtype):
157 # Here the target is discrete and there are two continuous and one
158 # discrete feature. The idea of this test is clear from the code.
159 rng = check_random_state(0)
160 X = rng.rand(1000, 3).astype(global_dtype, copy=False)
161 X[:, 1] += X[:, 0]
162 y = ((0.5 * X[:, 0] + X[:, 2]) > 0.5).astype(int)
163 X[:, 2] = X[:, 2] > 0.5
164
165 mi = mutual_info_classif(X, y, discrete_features=[2], n_neighbors=3, random_state=0)
166 assert_array_equal(np.argsort(-mi), [2, 0, 1])
167 for n_neighbors in [5, 7, 9]:
168 mi_nn = mutual_info_classif(
169 X, y, discrete_features=[2], n_neighbors=n_neighbors, random_state=0
170 )
171 # Check that the continuous values have an higher MI with greater
172 # n_neighbors
173 assert mi_nn[0] > mi[0]
174 assert mi_nn[1] > mi[1]
175 # The n_neighbors should not have any effect on the discrete value
176 # The MI should be the same
177 assert mi_nn[2] == mi[2]
