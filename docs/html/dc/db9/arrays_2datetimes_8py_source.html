849 def tz_convert(self, tz) -> Self:
850 """
851 Convert tz-aware Datetime Array/Index from one time zone to another.
852
853 Parameters
854 ----------
855 tz : str, pytz.timezone, dateutil.tz.tzfile, datetime.tzinfo or None

861 -------
862 Array or Index

866 TypeError

872 DatetimeIndex.tz_localize : Localize tz-naive DatetimeIndex to a
873 given time zone, or remove timezone from a tz-aware DatetimeIndex.

876 --------
877 With the `tz` parameter, we can change the DatetimeIndex
878 to other time zones:

891 '2014-08-01 03:00:00-05:00',
892 '2014-08-01 04:00:00-05:00'],

894

896 to UTC if necessary):

899 ... periods=3, tz='Europe/Berlin')
900
901 >>> dti
902 DatetimeIndex(['2014-08-01 09:00:00+02:00',

919 )
920

927 self,

930 nonexistent: TimeNonexistent = "raise",

932 """
933 Localize tz-naive Datetime Array/Index to tz-aware Datetime Array/Index.
934
935 This method takes a time zone (tz) naive Datetime Array/Index object
936 and makes this time zone aware. It does not move the time to another
937 time zone.
938
939 This method can also be used to do the inverse -- to create a time
940 zone unaware object from an aware object. To that end, pass `tz=None`.
941
942 Parameters
943 ----------
944 tz : str, pytz.timezone, dateutil.tz.tzfile, datetime.tzinfo or None
945 Time zone to convert timestamps to. Passing ``None`` will
946 remove the time zone information preserving local time.
947 ambiguous : 'infer', 'NaT', bool array, default 'raise'
948 When clocks moved backward due to DST, ambiguous times may arise.
949 For example in Central European Time (UTC+01), when going from
950 03:00 DST to 02:00 non-DST, 02:30:00 local time occurs both at
951 00:30:00 UTC and at 01:30:00 UTC. In such a situation, the
952 `ambiguous` parameter dictates how ambiguous times should be
953 handled.
954
955 - 'infer' will attempt to infer fall dst-transition hours based on
956 order
957 - bool-ndarray where True signifies a DST time, False signifies a
958 non-DST time (note that this flag is only applicable for
959 ambiguous times)
960 - 'NaT' will return NaT where there are ambiguous times
961 - 'raise' will raise an AmbiguousTimeError if there are ambiguous
962 times.
963
964 nonexistent : 'shift_forward', 'shift_backward, 'NaT', timedelta, \
965default 'raise'
966 A nonexistent time does not exist in a particular timezone
967 where clocks moved forward due to DST.
968
969 - 'shift_forward' will shift the nonexistent time forward to the
970 closest existing time
971 - 'shift_backward' will shift the nonexistent time backward to the
972 closest existing time
973 - 'NaT' will return NaT where there are nonexistent times
974 - timedelta objects will shift nonexistent times by the timedelta
975 - 'raise' will raise an NonExistentTimeError if there are
976 nonexistent times.
977
978 Returns
979 -------
980 Same type as self
981 Array/Index converted to the specified time zone.
982
983 Raises
984 ------
985 TypeError
986 If the Datetime Array/Index is tz-aware and tz is not None.
987

989 --------
990 DatetimeIndex.tz_convert : Convert tz-aware DatetimeIndex from
991 one time zone to another.
992

994 --------
995 >>> tz_naive = pd.date_range('2018-03-01 09:00', periods=3)

998 '2018-03-03 09:00:00'],
999 dtype='datetime64[ns]', freq='D')

1004 >>> tz_aware
1005 DatetimeIndex(['2018-03-01 09:00:00-05:00',
1006 '2018-03-02 09:00:00-05:00',
1007 '2018-03-03 09:00:00-05:00'],
1008 dtype='datetime64[ns, US/Eastern]', freq=None)
1009
1010 With the ``tz=None``, we can remove the time zone information
1011 while keeping the local time (not converted to UTC):
1012
1013 >>> tz_aware.tz_localize(None)
1014 DatetimeIndex(['2018-03-01 09:00:00', '2018-03-02 09:00:00',

1027 ... '2018-10-28 03:30:00']))

1030 1 2018-10-28 02:00:00+02:00
1031 2 2018-10-28 02:30:00+02:00
1032 3 2018-10-28 02:00:00+01:00

1067 0 2015-03-29 03:30:00+02:00

1069 dtype: datetime64[ns, Europe/Warsaw]

1074 ):

1079 )

1082 if tz is None:
1083 new_dates = tz_convert_from_utc(self.asi8, self.tz, reso=self._creso)
1084 else:
1085 raise TypeError("Already tz-aware, use tz_convert to convert.")

1091 self.asi8,

1095 creso=self._creso,
1096 )

1098 dtype = tz_to_dtype(tz, unit=self.unit)

1100 freq = None
1101 if timezones.is_utc(tz) or (len(self) == 1 and not isna(new_dates_dt64[0])):

2102 """

2104 0 Julian date is noon January 1, 4713 BC.
2105 https://en.wikipedia.org/wiki/Julian_day
2106 """

2109 year = np.asarray(self.year)
2110 month = np.asarray(self.month)
2111 day = np.asarray(self.day)
2112 testarr = month < 3
2113 year[testarr] -= 1
2114 month[testarr] += 12

2116 day
2117 + np.fix((153 * month - 457) / 5)
2118 + 365 * year
2119 + np.floor(year / 4)
2120 - np.floor(year / 100)
2121 + np.floor(year / 400)

2124 self.hour
2125 + self.minute / 60

2127 + self.microsecond / 3600 / 10**6
