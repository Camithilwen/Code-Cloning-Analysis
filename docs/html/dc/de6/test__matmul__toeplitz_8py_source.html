117 # For toeplitz matrices, matmul_toeplitz() should be equivalent to @.
118 def do(self, x, c, r=None, check_finite=False, workers=None):

120 actual = matmul_toeplitz(c, x, check_finite, workers)

122 actual = matmul_toeplitz((c, r), x, check_finite)
