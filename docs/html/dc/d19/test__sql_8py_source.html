4123def format_query(sql, *args):
4124 _formatters = {
4125 datetime: "'{}'".format,
4126 str: "'{}'".format,
4127 np.str_: "'{}'".format,
4128 bytes: "'{}'".format,
4129 float: "{:.8f}".format,
4130 int: "{:d}".format,
4131 type(None): lambda x: "NULL",
4132 np.float64: "{:.10f}".format,
4133 bool: "'{!s}'".format,
4134 }
4135 processed_args = []
4136 for arg in args:
4137 if isinstance(arg, float) and isna(arg):
4138 arg = None
4139
4140 formatter = _formatters[type(arg)]
4141 processed_args.append(formatter(arg))
4142
4143 return sql % tuple(processed_args)
