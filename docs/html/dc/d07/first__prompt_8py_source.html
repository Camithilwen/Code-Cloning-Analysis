107def ensemble_assessment(code1, code2, model_name, prompt, n=3):
108 '''! Performs multiple assessments and aggregates predictions using voting.
109
110 @param code1 First code string.
111 @param code2 Second code string.
112 @param model_name The name of the LLM model to use.
113 @param thr Threshold for similarity.
114 @param n Number of repeated assessments for ensemble.
115 @return Tuple of (average type scores, majority predicted type, majority binary similarity).
116 '''
117 predictions = []
118 for _ in range(n):
119 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name, prompt)
120 predictions.append((results, predicted_type, predicted_sim))
121 # Majority voting on predicted_sim
122 sim_votes = [pred[2] for pred in predictions]
123 final_sim = max(set(sim_votes), key=sim_votes.count)
124 print("sim votes: ", sim_votes)
125 print("final_sim: ", final_sim)
126 # Most frequent predicted_type
127 type_votes = [pred[1] for pred in predictions]
128 final_type = max(set(type_votes), key=type_votes.count)
129 print("type_votes: ", type_votes)
130 print("final_type: ", final_type)
131 # Average results for reporting
132 average_res = dict()
133 for t in ["Type-1", "Type-2", "Type-3", "Type-4"]:
134 sum_yes = 0
135 for pred in predictions:
136 n = pred[0][t]
137 # print(f'n: {n}')
138 if n == 'yes' or n=='Yes' or 'yes' in n or 'Yes' in n:
139 n = 1
140 else:
141 n = 0
142 sum_yes += n
143 try:
144 average_res[t] = sum_yes / n
145 except ZeroDivisionError:
146 average_res[t] = 0
147 print("average res", average_res)
148 for key, value in average_res.items():
149 if value == 0:
150 average_res[key] = "no"
151 else:
152 average_res[key] = "yes"
153 print("average res", average_res)
154 return average_res, final_type, final_sim
