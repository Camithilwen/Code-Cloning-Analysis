40def test_asarray_copy():
41 a = asarray([1])
42 b = asarray(a, copy=True)
43 a[0] = 0
44 assert all(b[0] == 1)
45 assert all(a[0] == 0)
46 a = asarray([1])
47 b = asarray(a, copy=np._CopyMode.ALWAYS)
48 a[0] = 0
49 assert all(b[0] == 1)
50 assert all(a[0] == 0)
51 a = asarray([1])
52 b = asarray(a, copy=np._CopyMode.NEVER)
53 a[0] = 0
54 assert all(b[0] == 0)
