876 def _accumulate(self, name: str, *, skipna: bool = True, **kwargs) -> StringArray:
877 """
878 Return an ExtensionArray performing an accumulation operation.
879
880 The underlying data type might change.
881
882 Parameters
883 ----------
884 name : str
885 Name of the function, supported values are:
886 - cummin
887 - cummax
888 - cumsum
889 - cumprod
890 skipna : bool, default True
891 If True, skip NA values.
892 **kwargs
893 Additional keyword arguments passed to the accumulation function.
894 Currently, there is no supported kwarg.
895
896 Returns
897 -------
898 array
899
900 Raises
901 ------
902 NotImplementedError : subclass does not define accumulations
903 """
904 if name == "cumprod":
905 msg = f"operation '{name}' not supported for dtype '{self.dtype}'"
906 raise TypeError(msg)
907
908 # We may need to strip out trailing NA values
909 tail: np.ndarray | None = None
910 na_mask: np.ndarray | None = None
911 ndarray = self._ndarray
912 np_func = {
913 "cumsum": np.cumsum,
914 "cummin": np.minimum.accumulate,
915 "cummax": np.maximum.accumulate,
916 }[name]
917
918 if self._hasna:
919 na_mask = cast("npt.NDArray[np.bool_]", isna(ndarray))
920 if np.all(na_mask):
921 return type(self)(ndarray)
922 if skipna:
923 if name == "cumsum":
924 ndarray = np.where(na_mask, "", ndarray)
925 else:
926 # We can retain the running min/max by forward/backward filling.
927 ndarray = ndarray.copy()
928 missing.pad_or_backfill_inplace(
929 ndarray,
930 method="pad",
931 axis=0,
932 )
933 missing.pad_or_backfill_inplace(

935 method="backfill",
936 axis=0,
937 )

940 # the first NA value onward.
941 idx = np.argmax(na_mask)
942 tail = np.empty(len(ndarray) - idx, dtype="object")
943 tail[:] = self.dtype.na_value
944 ndarray = ndarray[:idx]
945
946 # mypy: Cannot call function of unknown type
947 np_result = np_func(ndarray) # type: ignore[operator]

950 np_result = np.hstack((np_result, tail))

953 np_result = np.where(na_mask, self.dtype.na_value, np_result) # type: ignore[arg-type]

956 return result
