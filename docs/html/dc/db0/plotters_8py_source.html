40@deprecated('This will be removed in v0.4.0. Please use '
41 'scikitplot.metrics.plot_confusion_matrix instead.')
42def plot_confusion_matrix(y_true, y_pred, labels=None, true_labels=None,
43 pred_labels=None, title=None, normalize=False,
44 hide_zeros=False, x_tick_rotation=0, ax=None,
45 figsize=None, cmap='Blues', title_fontsize="large",
46 text_fontsize="medium"):
47 """Generates confusion matrix plot from predictions and true labels
48
49 Args:
50 y_true (array-like, shape (n_samples)):
51 Ground truth (correct) target values.
52
53 y_pred (array-like, shape (n_samples)):
54 Estimated targets as returned by a classifier.
55
56 labels (array-like, shape (n_classes), optional): List of labels to
57 index the matrix. This may be used to reorder or select a subset
58 of labels. If none is given, those that appear at least once in
59 ``y_true`` or ``y_pred`` are used in sorted order. (new in v0.2.5)
60
61 true_labels (array-like, optional): The true labels to display.
62 If none is given, then all of the labels are used.
63
64 pred_labels (array-like, optional): The predicted labels to display.
65 If none is given, then all of the labels are used.
66
67 title (string, optional): Title of the generated plot. Defaults to
68 "Confusion Matrix" if `normalize` is True. Else, defaults to
69 "Normalized Confusion Matrix.
70
71 normalize (bool, optional): If True, normalizes the confusion matrix
72 before plotting. Defaults to False.
73
74 hide_zeros (bool, optional): If True, does not plot cells containing a
75 value of zero. Defaults to False.
76
77 x_tick_rotation (int, optional): Rotates x-axis tick labels by the
78 specified angle. This is useful in cases where there are numerous
79 categories and the labels overlap each other.
80
81 ax (:class:`matplotlib.axes.Axes`, optional): The axes upon which to
82 plot the curve. If None, the plot is drawn on a new set of axes.
83
84 figsize (2-tuple, optional): Tuple denoting figure size of the plot
85 e.g. (6, 6). Defaults to ``None``.
86
87 cmap (string or :class:`matplotlib.colors.Colormap` instance, optional):
88 Colormap used for plotting the projection. View Matplotlib Colormap
89 documentation for available options.
90 https://matplotlib.org/users/colormaps.html
91
92 title_fontsize (string or int, optional): Matplotlib-style fontsizes.
93 Use e.g. "small", "medium", "large" or integer-values. Defaults to
94 "large".
95
96 text_fontsize (string or int, optional): Matplotlib-style fontsizes.
97 Use e.g. "small", "medium", "large" or integer-values. Defaults to
98 "medium".
99
100 Returns:
101 ax (:class:`matplotlib.axes.Axes`): The axes on which the plot was
102 drawn.
103
104 Example:
105 >>> import scikitplot.plotters as skplt
106 >>> rf = RandomForestClassifier()
107 >>> rf = rf.fit(X_train, y_train)
108 >>> y_pred = rf.predict(X_test)
109 >>> skplt.plot_confusion_matrix(y_test, y_pred, normalize=True)
110 <matplotlib.axes._subplots.AxesSubplot object at 0x7fe967d64490>
111 >>> plt.show()
112
113 .. image:: _static/examples/plot_confusion_matrix.png
114 :align: center
115 :alt: Confusion matrix
116 """
117 if ax is None:
118 fig, ax = plt.subplots(1, 1, figsize=figsize)
119
120 cm = confusion_matrix(y_true, y_pred, labels=labels)
121 if labels is None:
122 classes = unique_labels(y_true, y_pred)
123 else:
124 classes = np.asarray(labels)
125
126 if normalize:
127 cm = cm.astype('float') / cm.sum(axis=1)[:, np.newaxis]
128 cm = np.around(cm, decimals=2)
129 cm[np.isnan(cm)] = 0.0
130
131 if true_labels is None:
132 true_classes = classes
133 else:
134 validate_labels(classes, true_labels, "true_labels")
135
136 true_label_indexes = np.in1d(classes, true_labels)
137
138 true_classes = classes[true_label_indexes]
139 cm = cm[true_label_indexes]
140
141 if pred_labels is None:
142 pred_classes = classes
143 else:
144 validate_labels(classes, pred_labels, "pred_labels")
145
146 pred_label_indexes = np.in1d(classes, pred_labels)
147
148 pred_classes = classes[pred_label_indexes]
149 cm = cm[:, pred_label_indexes]
150
151 if title:
152 ax.set_title(title, fontsize=title_fontsize)
153 elif normalize:
154 ax.set_title('Normalized Confusion Matrix', fontsize=title_fontsize)
155 else:
156 ax.set_title('Confusion Matrix', fontsize=title_fontsize)
157
158 image = ax.imshow(cm, interpolation='nearest', cmap=plt.cm.get_cmap(cmap))
159 plt.colorbar(mappable=image)
160 x_tick_marks = np.arange(len(pred_classes))
161 y_tick_marks = np.arange(len(true_classes))
162 ax.set_xticks(x_tick_marks)
163 ax.set_xticklabels(pred_classes, fontsize=text_fontsize,
164 rotation=x_tick_rotation)
165 ax.set_yticks(y_tick_marks)
166 ax.set_yticklabels(true_classes, fontsize=text_fontsize)
167
168 thresh = cm.max() / 2.
169 for i, j in itertools.product(range(cm.shape[0]), range(cm.shape[1])):
170 if not (hide_zeros and cm[i, j] == 0):
171 ax.text(j, i, cm[i, j],
172 horizontalalignment="center",
173 verticalalignment="center",
174 fontsize=text_fontsize,
175 color="white" if cm[i, j] > thresh else "black")
176
177 ax.set_ylabel('True label', fontsize=text_fontsize)
178 ax.set_xlabel('Predicted label', fontsize=text_fontsize)
179 ax.grid('off')
180
181 return ax

184@deprecated('This will be removed in v0.4.0. Please use '
185 'scikitplot.metrics.plot_roc_curve instead.')
186def plot_roc_curve(y_true, y_probas, title='ROC Curves',
187 curves=('micro', 'macro', 'each_class'),
188 ax=None, figsize=None, cmap='nipy_spectral',
189 title_fontsize="large", text_fontsize="medium"):
190 """Generates the ROC curves from labels and predicted scores/probabilities
191
192 Args:
193 y_true (array-like, shape (n_samples)):
194 Ground truth (correct) target values.
195
196 y_probas (array-like, shape (n_samples, n_classes)):
197 Prediction probabilities for each class returned by a classifier.
198
199 title (string, optional): Title of the generated plot. Defaults to
200 "ROC Curves".
201
202 curves (array-like): A listing of which curves should be plotted on the
203 resulting plot. Defaults to `("micro", "macro", "each_class")`
204 i.e. "micro" for micro-averaged curve, "macro" for macro-averaged
205 curve
206
207 ax (:class:`matplotlib.axes.Axes`, optional): The axes upon which to
208 plot the curve. If None, the plot is drawn on a new set of axes.
209
210 figsize (2-tuple, optional): Tuple denoting figure size of the plot
211 e.g. (6, 6). Defaults to ``None``.
212
213 cmap (string or :class:`matplotlib.colors.Colormap` instance, optional):
214 Colormap used for plotting the projection. View Matplotlib Colormap
215 documentation for available options.
216 https://matplotlib.org/users/colormaps.html
217
218 title_fontsize (string or int, optional): Matplotlib-style fontsizes.
219 Use e.g. "small", "medium", "large" or integer-values. Defaults to
220 "large".
221
222 text_fontsize (string or int, optional): Matplotlib-style fontsizes.
223 Use e.g. "small", "medium", "large" or integer-values. Defaults to
224 "medium".
225
226 Returns:
227 ax (:class:`matplotlib.axes.Axes`): The axes on which the plot was
228 drawn.
229
230 Example:
231 >>> import scikitplot.plotters as skplt
232 >>> nb = GaussianNB()
233 >>> nb = nb.fit(X_train, y_train)
234 >>> y_probas = nb.predict_proba(X_test)
235 >>> skplt.plot_roc_curve(y_test, y_probas)
236 <matplotlib.axes._subplots.AxesSubplot object at 0x7fe967d64490>
237 >>> plt.show()
238
239 .. image:: _static/examples/plot_roc_curve.png
240 :align: center
241 :alt: ROC Curves
242 """
243 y_true = np.array(y_true)
244 y_probas = np.array(y_probas)
245
246 if 'micro' not in curves and 'macro' not in curves and \
247 'each_class' not in curves:
248 raise ValueError('Invalid argument for curves as it '
249 'only takes "micro", "macro", or "each_class"')
250
251 classes = np.unique(y_true)
252 probas = y_probas
253
254 fpr = dict()
255 tpr = dict()
256 roc_auc = dict()
257 for i in range(len(classes)):
258 fpr[i], tpr[i], _ = roc_curve(y_true, probas[:, i],
259 pos_label=classes[i])
260 roc_auc[i] = auc(fpr[i], tpr[i])
261
262 # Compute micro-average ROC curve and ROC area
263 micro_key = 'micro'
264 i = 0
265 while micro_key in fpr:
266 i += 1
267 micro_key += str(i)
268
269 y_true = label_binarize(y_true, classes=classes)
270 if len(classes) == 2:
271 y_true = np.hstack((1 - y_true, y_true))
272
273 fpr[micro_key], tpr[micro_key], _ = roc_curve(y_true.ravel(),
274 probas.ravel())
275 roc_auc[micro_key] = auc(fpr[micro_key], tpr[micro_key])
276
277 # Compute macro-average ROC curve and ROC area
278
279 # First aggregate all false positive rates
280 all_fpr = np.unique(np.concatenate([fpr[x] for x in range(len(classes))]))
281
282 # Then interpolate all ROC curves at this points
283 mean_tpr = np.zeros_like(all_fpr)
284 for i in range(len(classes)):
285 mean_tpr += interp(all_fpr, fpr[i], tpr[i])
286
287 # Finally average it and compute AUC
288 mean_tpr /= len(classes)
289
290 macro_key = 'macro'
291 i = 0
292 while macro_key in fpr:
293 i += 1
294 macro_key += str(i)
295 fpr[macro_key] = all_fpr
296 tpr[macro_key] = mean_tpr
297 roc_auc[macro_key] = auc(fpr[macro_key], tpr[macro_key])
298
299 if ax is None:
300 fig, ax = plt.subplots(1, 1, figsize=figsize)
301
302 ax.set_title(title, fontsize=title_fontsize)
303
304 if 'each_class' in curves:
305 for i in range(len(classes)):
306 color = plt.cm.get_cmap(cmap)(float(i) / len(classes))
307 ax.plot(fpr[i], tpr[i], lw=2, color=color,
308 label='ROC curve of class {0} (area = {1:0.2f})'
309 ''.format(classes[i], roc_auc[i]))
310
311 if 'micro' in curves:
312 ax.plot(fpr[micro_key], tpr[micro_key],
313 label='micro-average ROC curve '
314 '(area = {0:0.2f})'.format(roc_auc[micro_key]),
315 color='deeppink', linestyle=':', linewidth=4)
316
317 if 'macro' in curves:
318 ax.plot(fpr[macro_key], tpr[macro_key],
319 label='macro-average ROC curve '
320 '(area = {0:0.2f})'.format(roc_auc[macro_key]),
321 color='navy', linestyle=':', linewidth=4)
322
323 ax.plot([0, 1], [0, 1], 'k--', lw=2)
324 ax.set_xlim([0.0, 1.0])
325 ax.set_ylim([0.0, 1.05])
326 ax.set_xlabel('False Positive Rate', fontsize=text_fontsize)
327 ax.set_ylabel('True Positive Rate', fontsize=text_fontsize)
328 ax.tick_params(labelsize=text_fontsize)
329 ax.legend(loc='lower right', fontsize=text_fontsize)
330 return ax

335def plot_ks_statistic(y_true, y_probas, title='KS Statistic Plot',
336 ax=None, figsize=None, title_fontsize="large",
337 text_fontsize="medium"):
338 """Generates the KS Statistic plot from labels and scores/probabilities
339
340 Args:
341 y_true (array-like, shape (n_samples)):
342 Ground truth (correct) target values.
343
344 y_probas (array-like, shape (n_samples, n_classes)):
345 Prediction probabilities for each class returned by a classifier.
346
347 title (string, optional): Title of the generated plot. Defaults to
348 "KS Statistic Plot".
349
350 ax (:class:`matplotlib.axes.Axes`, optional): The axes upon which to
351 plot the learning curve. If None, the plot is drawn on a new set of
352 axes.
353
354 figsize (2-tuple, optional): Tuple denoting figure size of the plot
355 e.g. (6, 6). Defaults to ``None``.
356
357 title_fontsize (string or int, optional): Matplotlib-style fontsizes.
358 Use e.g. "small", "medium", "large" or integer-values. Defaults to
359 "large".
360
361 text_fontsize (string or int, optional): Matplotlib-style fontsizes.
362 Use e.g. "small", "medium", "large" or integer-values. Defaults to
363 "medium".
364
365 Returns:
366 ax (:class:`matplotlib.axes.Axes`): The axes on which the plot was
367 drawn.
368
369 Example:
370 >>> import scikitplot.plotters as skplt
371 >>> lr = LogisticRegression()
372 >>> lr = lr.fit(X_train, y_train)
373 >>> y_probas = lr.predict_proba(X_test)
374 >>> skplt.plot_ks_statistic(y_test, y_probas)
375 <matplotlib.axes._subplots.AxesSubplot object at 0x7fe967d64490>
376 >>> plt.show()
377
378 .. image:: _static/examples/plot_ks_statistic.png
379 :align: center
380 :alt: KS Statistic
381 """
382 y_true = np.array(y_true)
383 y_probas = np.array(y_probas)
384
385 classes = np.unique(y_true)
386 if len(classes) != 2:
387 raise ValueError('Cannot calculate KS statistic for data with '
388 '{} category/ies'.format(len(classes)))
389 probas = y_probas
390
391 # Compute KS Statistic curves
392 thresholds, pct1, pct2, ks_statistic, \
393 max_distance_at, classes = binary_ks_curve(y_true,
394 probas[:, 1].ravel())
395
396 if ax is None:
397 fig, ax = plt.subplots(1, 1, figsize=figsize)
398
399 ax.set_title(title, fontsize=title_fontsize)
400
401 ax.plot(thresholds, pct1, lw=3, label='Class {}'.format(classes[0]))
402 ax.plot(thresholds, pct2, lw=3, label='Class {}'.format(classes[1]))
403 idx = np.where(thresholds == max_distance_at)[0][0]
404 ax.axvline(max_distance_at, *sorted([pct1[idx], pct2[idx]]),
405 label='KS Statistic: {:.3f} at {:.3f}'.format(ks_statistic,
406 max_distance_at),
407 linestyle=':', lw=3, color='black')
408
409 ax.set_xlim([0.0, 1.0])
410 ax.set_ylim([0.0, 1.0])
411
412 ax.set_xlabel('Threshold', fontsize=text_fontsize)
413 ax.set_ylabel('Percentage below threshold', fontsize=text_fontsize)
414 ax.tick_params(labelsize=text_fontsize)
415 ax.legend(loc='lower right', fontsize=text_fontsize)
416
417 return ax

552
553 Args:
554 clf: Classifier instance that implements ``fit`` and ``predict_proba``
555 methods. The classifier must also have a ``feature_importances_``

559 "Feature importances".
560
561 feature_names (None, :obj:`list` of string, optional): Determines the
562 feature names used to plot the feature importances. If None,
563 feature names will be numbered.
564
565 max_num_features (int): Determines the maximum number of features to

568 order ('ascending', 'descending', or None, optional): Determines the
569 order in which the feature importances are plotted. Defaults to
570 'descending'.
571
572 x_tick_rotation (int, optional): Rotates x-axis tick labels by the
573 specified angle. This is useful in cases where there are numerous
574 categories and the labels overlap each other.
575
576 ax (:class:`matplotlib.axes.Axes`, optional): The axes upon which to
577 plot the curve. If None, the plot is drawn on a new set of axes.
578
579 figsize (2-tuple, optional): Tuple denoting figure size of the plot
580 e.g. (6, 6). Defaults to ``None``.
581
582 title_fontsize (string or int, optional): Matplotlib-style fontsizes.
583 Use e.g. "small", "medium", "large" or integer-values. Defaults to
584 "large".
585
586 text_fontsize (string or int, optional): Matplotlib-style fontsizes.
587 Use e.g. "small", "medium", "large" or integer-values. Defaults to
588 "medium".
589
590 Returns:
591 ax (:class:`matplotlib.axes.Axes`): The axes on which the plot was
592 drawn.
593
594 Example:
595 >>> import scikitplot.plotters as skplt
596 >>> rf = RandomForestClassifier()
597 >>> rf.fit(X, y)
598 >>> skplt.plot_feature_importances(
599 ... rf, feature_names=['petal length', 'petal width',
600 ... 'sepal length', 'sepal width'])
601 <matplotlib.axes._subplots.AxesSubplot object at 0x7fe967d64490>
602 >>> plt.show()
603
604 .. image:: _static/examples/plot_feature_importances.png
605 :align: center
606 :alt: Feature Importances
607 """
608 if not hasattr(clf, 'feature_importances_'):
609 raise TypeError('"feature_importances_" attribute not in classifier. '
610 'Cannot plot feature importances.')
611
612 importances = clf.feature_importances_
613
614 if hasattr(clf, 'estimators_')\
615 and isinstance(clf.estimators_, list)\
616 and hasattr(clf.estimators_[0], 'feature_importances_'):
617 std = np.std([tree.feature_importances_ for tree in clf.estimators_],
618 axis=0)
619
620 else:
621 std = None
622
623 if order == 'descending':
624 indices = np.argsort(importances)[::-1]
625
626 elif order == 'ascending':
627 indices = np.argsort(importances)
628
629 elif order is None:
630 indices = np.array(range(len(importances)))
