81def _get_weights(dist, weights):
82 """Get the weights from an array of distances and a parameter ``weights``.
83
84 Assume weights have already been validated.
85
86 Parameters
87 ----------
88 dist : ndarray
89 The input distances.
90
91 weights : {'uniform', 'distance'}, callable or None
92 The kind of weighting used.
93
94 Returns
95 -------
96 weights_arr : array of the same shape as ``dist``
97 If ``weights == 'uniform'``, then returns None.
98 """
99 if weights in (None, "uniform"):
100 return None
101
102 if weights == "distance":
103 # if user attempts to classify a point that was zero distance from one
104 # or more training points, those training points are weighted as 1.0
105 # and the other points as 0.0
106 if dist.dtype is np.dtype(object):
107 for point_dist_i, point_dist in enumerate(dist):
108 # check if point_dist is iterable
109 # (ex: RadiusNeighborClassifier.predict may set an element of
110 # dist to 1e-6 to represent an 'outlier')
111 if hasattr(point_dist, "__contains__") and 0.0 in point_dist:
112 dist[point_dist_i] = point_dist == 0.0
113 else:
114 dist[point_dist_i] = 1.0 / point_dist
115 else:
116 with np.errstate(divide="ignore"):
117 dist = 1.0 / dist
118 inf_mask = np.isinf(dist)
119 inf_row = np.any(inf_mask, axis=1)
120 dist[inf_row] = inf_mask[inf_row]
121 return dist
122
123 if callable(weights):
124 return weights(dist)

127def _is_sorted_by_data(graph):
128 """Return whether the graph's non-zero entries are sorted by data.
129
130 The non-zero entries are stored in graph.data and graph.indices.
131 For each row (or sample), the non-zero entries can be either:
132 - sorted by indices, as after graph.sort_indices();
133 - sorted by data, as after _check_precomputed(graph);
134 - not sorted.
135
136 Parameters
137 ----------
138 graph : sparse matrix of shape (n_samples, n_samples)
139 Neighbors graph as given by `kneighbors_graph` or
140 `radius_neighbors_graph`. Matrix should be of format CSR format.
141
142 Returns
143 -------
144 res : bool
145 Whether input graph is sorted by data.
146 """
147 assert graph.format == "csr"
148 out_of_order = graph.data[:-1] > graph.data[1:]
149 line_change = np.unique(graph.indptr[1:-1] - 1)
150 line_change = line_change[line_change < out_of_order.shape[0]]
151 return out_of_order.sum() == out_of_order[line_change].sum()

154def _check_precomputed(X):
155 """Check precomputed distance matrix.
156
157 If the precomputed distance matrix is sparse, it checks that the non-zero
158 entries are sorted by distances. If not, the matrix is copied and sorted.
159
160 Parameters
161 ----------
162 X : {sparse matrix, array-like}, (n_samples, n_samples)
163 Distance matrix to other samples. X may be a sparse matrix, in which
164 case only non-zero elements may be considered neighbors.
165
166 Returns
167 -------
168 X : {sparse matrix, array-like}, (n_samples, n_samples)
169 Distance matrix to other samples. X may be a sparse matrix, in which
170 case only non-zero elements may be considered neighbors.
171 """
172 if not issparse(X):
173 X = check_array(X, ensure_non_negative=True, input_name="X")
174 return X
175 else:
176 graph = X
177
178 if graph.format not in ("csr", "csc", "coo", "lil"):
179 raise TypeError(
180 "Sparse matrix in {!r} format is not supported due to "
181 "its handling of explicit zeros".format(graph.format)
182 )
183 copied = graph.format != "csr"
184 graph = check_array(
185 graph,
186 accept_sparse="csr",
187 ensure_non_negative=True,
188 input_name="precomputed distance matrix",
189 )
190 graph = sort_graph_by_row_values(graph, copy=not copied, warn_when_not_sorted=True)
191
192 return graph

197 "graph": ["sparse matrix"],
198 "copy": ["boolean"],

203def sort_graph_by_row_values(graph, copy=False, warn_when_not_sorted=True):
204 """Sort a sparse graph such that each row is stored with increasing values.
205
206 .. versionadded:: 1.2
207
208 Parameters
209 ----------
210 graph : sparse matrix of shape (n_samples, n_samples)
211 Distance matrix to other samples, where only non-zero elements are
212 considered neighbors. Matrix is converted to CSR format if not already.
213
214 copy : bool, default=False
215 If True, the graph is copied before sorting. If False, the sorting is
216 performed inplace. If the graph is not of CSR format, `copy` must be
217 True to allow the conversion to CSR format, otherwise an error is
218 raised.
219
220 warn_when_not_sorted : bool, default=True
221 If True, a :class:`~sklearn.exceptions.EfficiencyWarning` is raised
222 when the input graph is not sorted by row values.
223
224 Returns
225 -------
226 graph : sparse matrix of shape (n_samples, n_samples)
227 Distance matrix to other samples, where only non-zero elements are
228 considered neighbors. Matrix is in CSR format.
229
230 Examples
231 --------
232 >>> from scipy.sparse import csr_matrix
233 >>> from sklearn.neighbors import sort_graph_by_row_values
234 >>> X = csr_matrix(
235 ... [[0., 3., 1.],
236 ... [3., 0., 2.],
237 ... [1., 2., 0.]])
238 >>> X.data
239 array([3., 1., 3., 2., 1., 2.])
240 >>> X_ = sort_graph_by_row_values(X)
241 >>> X_.data
242 array([1., 3., 2., 3., 1., 2.])
243 """
244 if graph.format == "csr" and _is_sorted_by_data(graph):
245 return graph
246
247 if warn_when_not_sorted:
248 warnings.warn(
249 (
250 "Precomputed sparse input was not sorted by row values. Use the"
251 " function sklearn.neighbors.sort_graph_by_row_values to sort the input"
252 " by row values, with warn_when_not_sorted=False to remove this"
253 " warning."
254 ),
255 EfficiencyWarning,
256 )
257
258 if graph.format not in ("csr", "csc", "coo", "lil"):
259 raise TypeError(
260 f"Sparse matrix in {graph.format!r} format is not supported due to "
261 "its handling of explicit zeros"
262 )
263 elif graph.format != "csr":
264 if not copy:
265 raise ValueError(
266 "The input graph is not in CSR format. Use copy=True to allow "
267 "the conversion to CSR format."
268 )
269 graph = graph.asformat("csr")
270 elif copy: # csr format with copy=True
271 graph = graph.copy()
272
273 row_nnz = np.diff(graph.indptr)
274 if row_nnz.max() == row_nnz.min():
275 # if each sample has the same number of provided neighbors
276 n_samples = graph.shape[0]
277 distances = graph.data.reshape(n_samples, -1)
278
279 order = np.argsort(distances, kind="mergesort")
280 order += np.arange(n_samples)[:, None] * row_nnz[0]
281 order = order.ravel()
282 graph.data = graph.data[order]
283 graph.indices = graph.indices[order]
284
285 else:
286 for start, stop in zip(graph.indptr, graph.indptr[1:]):
287 order = np.argsort(graph.data[start:stop], kind="mergesort")
288 graph.data[start:stop] = graph.data[start:stop][order]
289 graph.indices[start:stop] = graph.indices[start:stop][order]
290
291 return graph

294def _kneighbors_from_graph(graph, n_neighbors, return_distance):
295 """Decompose a nearest neighbors sparse graph into distances and indices.
296
297 Parameters
298 ----------
299 graph : sparse matrix of shape (n_samples, n_samples)
300 Neighbors graph as given by `kneighbors_graph` or
301 `radius_neighbors_graph`. Matrix should be of format CSR format.
302
303 n_neighbors : int
304 Number of neighbors required for each sample.
305
306 return_distance : bool
307 Whether or not to return the distances.
308
309 Returns
310 -------
311 neigh_dist : ndarray of shape (n_samples, n_neighbors)
312 Distances to nearest neighbors. Only present if `return_distance=True`.
313
314 neigh_ind : ndarray of shape (n_samples, n_neighbors)
315 Indices of nearest neighbors.
316 """
317 n_samples = graph.shape[0]
318 assert graph.format == "csr"
319
320 # number of neighbors by samples
321 row_nnz = np.diff(graph.indptr)
322 row_nnz_min = row_nnz.min()
323 if n_neighbors is not None and row_nnz_min < n_neighbors:
324 raise ValueError(
325 "%d neighbors per samples are required, but some samples have only"
326 " %d neighbors in precomputed graph matrix. Decrease number of "
327 "neighbors used or recompute the graph with more neighbors."
328 % (n_neighbors, row_nnz_min)
329 )
330
331 def extract(a):
332 # if each sample has the same number of provided neighbors
333 if row_nnz.max() == row_nnz_min:
334 return a.reshape(n_samples, -1)[:, :n_neighbors]
335 else:
336 idx = np.tile(np.arange(n_neighbors), (n_samples, 1))
337 idx += graph.indptr[:-1, None]
338 return a.take(idx, mode="clip").reshape(n_samples, n_neighbors)
339
340 if return_distance:
341 return extract(graph.data), extract(graph.indices)
342 else:
343 return extract(graph.indices)

346def _radius_neighbors_from_graph(graph, radius, return_distance):
347 """Decompose a nearest neighbors sparse graph into distances and indices.
348
349 Parameters
350 ----------
351 graph : sparse matrix of shape (n_samples, n_samples)
352 Neighbors graph as given by `kneighbors_graph` or
353 `radius_neighbors_graph`. Matrix should be of format CSR format.
354
355 radius : float
356 Radius of neighborhoods which should be strictly positive.
357
358 return_distance : bool
359 Whether or not to return the distances.
360
361 Returns
362 -------
363 neigh_dist : ndarray of shape (n_samples,) of arrays
364 Distances to nearest neighbors. Only present if `return_distance=True`.
365
366 neigh_ind : ndarray of shape (n_samples,) of arrays
367 Indices of nearest neighbors.
368 """
369 assert graph.format == "csr"
370
371 no_filter_needed = bool(graph.data.max() <= radius)
372
373 if no_filter_needed:
374 data, indices, indptr = graph.data, graph.indices, graph.indptr
375 else:
376 mask = graph.data <= radius
377 if return_distance:
378 data = np.compress(mask, graph.data)
379 indices = np.compress(mask, graph.indices)
380 indptr = np.concatenate(([0], np.cumsum(mask)))[graph.indptr]
381
382 indices = indices.astype(np.intp, copy=no_filter_needed)
383
384 if return_distance:
385 neigh_dist = _to_object_array(np.split(data, indptr[1:-1]))
386 neigh_ind = _to_object_array(np.split(indices, indptr[1:-1]))
387
388 if return_distance:
389 return neigh_dist, neigh_ind
390 else:
391 return neigh_ind

474 def _fit(self, X, y=None):
475 ensure_all_finite = "allow-nan" if get_tags(self).input_tags.allow_nan else True
476 if self.__sklearn_tags__().target_tags.required:
477 if not isinstance(X, (KDTree, BallTree, NeighborsBase)):
478 X, y = validate_data(
479 self,
480 X,
481 y,
482 accept_sparse="csr",
483 multi_output=True,
484 order="C",
485 ensure_all_finite=ensure_all_finite,
486 )
487
488 if is_classifier(self):
489 # Classification targets require a specific format
490 if y.ndim == 1 or (y.ndim == 2 and y.shape[1] == 1):
491 if y.ndim != 1:
492 warnings.warn(
493 (
494 "A column-vector y was passed when a "
495 "1d array was expected. Please change "
496 "the shape of y to (n_samples,), for "
497 "example using ravel()."
498 ),
499 DataConversionWarning,
500 stacklevel=2,
501 )
502
503 self.outputs_2d_ = False
504 y = y.reshape((-1, 1))
505 else:
506 self.outputs_2d_ = True
507
508 check_classification_targets(y)
509 self.classes_ = []
510 # Using `dtype=np.intp` is necessary since `np.bincount`
511 # (called in _classification.py) fails when dealing
512 # with a float64 array on 32bit systems.
513 self._y = np.empty(y.shape, dtype=np.intp)
514 for k in range(self._y.shape[1]):
515 classes, self._y[:, k] = np.unique(y[:, k], return_inverse=True)
516 self.classes_.append(classes)
517
518 if not self.outputs_2d_:
519 self.classes_ = self.classes_[0]
520 self._y = self._y.ravel()
521 else:
522 self._y = y
523
524 else:
525 if not isinstance(X, (KDTree, BallTree, NeighborsBase)):
526 X = validate_data(
527 self,
528 X,
529 ensure_all_finite=ensure_all_finite,
530 accept_sparse="csr",
531 order="C",
532 )
533
534 self._check_algorithm_metric()
535 if self.metric_params is None:
536 self.effective_metric_params_ = {}
537 else:
538 self.effective_metric_params_ = self.metric_params.copy()
539
540 effective_p = self.effective_metric_params_.get("p", self.p)
541 if self.metric == "minkowski":
542 self.effective_metric_params_["p"] = effective_p
543
544 self.effective_metric_ = self.metric
545 # For minkowski distance, use more efficient methods where available
546 if self.metric == "minkowski":
547 p = self.effective_metric_params_.pop("p", 2)
548 w = self.effective_metric_params_.pop("w", None)
549
550 if p == 1 and w is None:
551 self.effective_metric_ = "manhattan"
552 elif p == 2 and w is None:
553 self.effective_metric_ = "euclidean"
554 elif p == np.inf and w is None:
555 self.effective_metric_ = "chebyshev"
556 else:
557 # Use the generic minkowski metric, possibly weighted.
558 self.effective_metric_params_["p"] = p
559 self.effective_metric_params_["w"] = w
560
561 if isinstance(X, NeighborsBase):
562 self._fit_X = X._fit_X
563 self._tree = X._tree
564 self._fit_method = X._fit_method
565 self.n_samples_fit_ = X.n_samples_fit_
566 return self
567
568 elif isinstance(X, BallTree):
569 self._fit_X = X.data
570 self._tree = X
571 self._fit_method = "ball_tree"
572 self.n_samples_fit_ = X.data.shape[0]
573 return self
574
575 elif isinstance(X, KDTree):
576 self._fit_X = X.data
577 self._tree = X
578 self._fit_method = "kd_tree"
579 self.n_samples_fit_ = X.data.shape[0]
580 return self
581
582 if self.metric == "precomputed":
583 X = _check_precomputed(X)
584 # Precomputed matrix X must be squared
585 if X.shape[0] != X.shape[1]:
586 raise ValueError(
587 "Precomputed matrix must be square."
588 " Input is a {}x{} matrix.".format(X.shape[0], X.shape[1])
589 )
590 self.n_features_in_ = X.shape[1]
591
592 n_samples = X.shape[0]
593 if n_samples == 0:
594 raise ValueError("n_samples must be greater than 0")
595
596 if issparse(X):
597 if self.algorithm not in ("auto", "brute"):
598 warnings.warn("cannot use tree with sparse input: using brute force")
599
600 if (
601 self.effective_metric_ not in VALID_METRICS_SPARSE["brute"]
602 and not callable(self.effective_metric_)
603 and not isinstance(self.effective_metric_, DistanceMetric)
604 ):
605 raise ValueError(
606 "Metric '%s' not valid for sparse input. "
607 "Use sorted(sklearn.neighbors."
608 "VALID_METRICS_SPARSE['brute']) "
609 "to get valid options. "
610 "Metric can also be a callable function." % (self.effective_metric_)
611 )
612 self._fit_X = X.copy()
613 self._tree = None
614 self._fit_method = "brute"
615 self.n_samples_fit_ = X.shape[0]
616 return self
617
618 self._fit_method = self.algorithm
619 self._fit_X = X
620 self.n_samples_fit_ = X.shape[0]
621
622 if self._fit_method == "auto":
623 # A tree approach is better for small number of neighbors or small
624 # number of features, with KDTree generally faster when available
625 if (
626 self.metric == "precomputed"
627 or self._fit_X.shape[1] > 15
628 or (
629 self.n_neighbors is not None
630 and self.n_neighbors >= self._fit_X.shape[0] // 2
631 )
632 ):
633 self._fit_method = "brute"
634 else:
635 if (
636 self.effective_metric_ == "minkowski"
637 and self.effective_metric_params_["p"] < 1
638 ):
639 self._fit_method = "brute"
640 elif (
641 self.effective_metric_ == "minkowski"
642 and self.effective_metric_params_.get("w") is not None
643 ):
644 # 'minkowski' with weights is not supported by KDTree but is
645 # supported byBallTree.
646 self._fit_method = "ball_tree"
647 elif self.effective_metric_ in VALID_METRICS["kd_tree"]:
648 self._fit_method = "kd_tree"
649 elif (
650 callable(self.effective_metric_)
651 or self.effective_metric_ in VALID_METRICS["ball_tree"]
652 ):
653 self._fit_method = "ball_tree"
654 else:
655 self._fit_method = "brute"
656
657 if (
658 self.effective_metric_ == "minkowski"
659 and self.effective_metric_params_["p"] < 1
660 ):
661 # For 0 < p < 1 Minkowski distances aren't valid distance
662 # metric as they do not satisfy triangular inequality:
663 # they are semi-metrics.
664 # algorithm="kd_tree" and algorithm="ball_tree" can't be used because
665 # KDTree and BallTree require a proper distance metric to work properly.
666 # However, the brute-force algorithm supports semi-metrics.
667 if self._fit_method == "brute":
668 warnings.warn(
669 "Mind that for 0 < p < 1, Minkowski metrics are not distance"
670 " metrics. Continuing the execution with `algorithm='brute'`."
671 )
672 else: # self._fit_method in ("kd_tree", "ball_tree")
673 raise ValueError(
674 f'algorithm="{self._fit_method}" does not support 0 < p < 1 for '
675 "the Minkowski metric. To resolve this problem either "
676 'set p >= 1 or algorithm="brute".'
677 )
678
679 if self._fit_method == "ball_tree":
680 self._tree = BallTree(
681 X,
682 self.leaf_size,
683 metric=self.effective_metric_,
684 **self.effective_metric_params_,
685 )
686 elif self._fit_method == "kd_tree":
687 if (
688 self.effective_metric_ == "minkowski"
689 and self.effective_metric_params_.get("w") is not None
690 ):
691 raise ValueError(
692 "algorithm='kd_tree' is not valid for "
693 "metric='minkowski' with a weight parameter 'w': "
694 "try algorithm='ball_tree' "
695 "or algorithm='brute' instead."
696 )
697 self._tree = KDTree(
698 X,
699 self.leaf_size,
700 metric=self.effective_metric_,
701 **self.effective_metric_params_,
702 )
703 elif self._fit_method == "brute":
704 self._tree = None
705
706 return self

722 def _kneighbors_reduce_func(self, dist, start, n_neighbors, return_distance):
723 """Reduce a chunk of distances to the nearest neighbors.
724
725 Callback to :func:`sklearn.metrics.pairwise.pairwise_distances_chunked`
726
727 Parameters
728 ----------
729 dist : ndarray of shape (n_samples_chunk, n_samples)
730 The distance matrix.
731
732 start : int
733 The index in X which the first row of dist corresponds to.
734
735 n_neighbors : int
736 Number of neighbors required for each sample.
737
738 return_distance : bool
739 Whether or not to return the distances.
740
741 Returns
742 -------
743 dist : array of shape (n_samples_chunk, n_neighbors)
744 Returned only if `return_distance=True`.
745
746 neigh : array of shape (n_samples_chunk, n_neighbors)
747 The neighbors indices.
748 """
749 sample_range = np.arange(dist.shape[0])[:, None]
750 neigh_ind = np.argpartition(dist, n_neighbors - 1, axis=1)
751 neigh_ind = neigh_ind[:, :n_neighbors]
752 # argpartition doesn't guarantee sorted order, so we sort again
753 neigh_ind = neigh_ind[sample_range, np.argsort(dist[sample_range, neigh_ind])]
754 if return_distance:
755 if self.effective_metric_ == "euclidean":
756 result = np.sqrt(dist[sample_range, neigh_ind]), neigh_ind
757 else:
758 result = dist[sample_range, neigh_ind], neigh_ind
759 else:
760 result = neigh_ind
761 return result

763 def kneighbors(self, X=None, n_neighbors=None, return_distance=True):
764 """Find the K-neighbors of a point.
765
766 Returns indices of and distances to the neighbors of each point.
767
768 Parameters
769 ----------
770 X : {array-like, sparse matrix}, shape (n_queries, n_features), \
771 or (n_queries, n_indexed) if metric == 'precomputed', default=None
772 The query point or points.
773 If not provided, neighbors of each indexed point are returned.
774 In this case, the query point is not considered its own neighbor.
775
776 n_neighbors : int, default=None
777 Number of neighbors required for each sample. The default is the
778 value passed to the constructor.
779
780 return_distance : bool, default=True
781 Whether or not to return the distances.
782
783 Returns
784 -------
785 neigh_dist : ndarray of shape (n_queries, n_neighbors)
786 Array representing the lengths to points, only present if
787 return_distance=True.
788
789 neigh_ind : ndarray of shape (n_queries, n_neighbors)
790 Indices of the nearest points in the population matrix.
791
792 Examples
793 --------
794 In the following example, we construct a NearestNeighbors
795 class from an array representing our data set and ask who's
796 the closest point to [1,1,1]
797
798 >>> samples = [[0., 0., 0.], [0., .5, 0.], [1., 1., .5]]
799 >>> from sklearn.neighbors import NearestNeighbors
800 >>> neigh = NearestNeighbors(n_neighbors=1)
801 >>> neigh.fit(samples)
802 NearestNeighbors(n_neighbors=1)
803 >>> print(neigh.kneighbors([[1., 1., 1.]]))
804 (array([[0.5]]), array([[2]]))
805
806 As you can see, it returns [[0.5]], and [[2]], which means that the
807 element is at distance 0.5 and is the third element of samples
808 (indexes start at 0). You can also query for multiple points:
809
810 >>> X = [[0., 1., 0.], [1., 0., 1.]]
811 >>> neigh.kneighbors(X, return_distance=False)
812 array([[1],
813 [2]]...)
814 """
815 check_is_fitted(self)
816
817 if n_neighbors is None:
818 n_neighbors = self.n_neighbors
819 elif n_neighbors <= 0:
820 raise ValueError("Expected n_neighbors > 0. Got %d" % n_neighbors)
821 elif not isinstance(n_neighbors, numbers.Integral):
822 raise TypeError(
823 "n_neighbors does not take %s value, enter integer value"
824 % type(n_neighbors)
825 )
826
827 ensure_all_finite = "allow-nan" if get_tags(self).input_tags.allow_nan else True
828 query_is_train = X is None
829 if query_is_train:
830 X = self._fit_X
831 # Include an extra neighbor to account for the sample itself being
832 # returned, which is removed later
833 n_neighbors += 1
834 else:
835 if self.metric == "precomputed":
836 X = _check_precomputed(X)
837 else:
838 X = validate_data(
839 self,
840 X,
841 ensure_all_finite=ensure_all_finite,
842 accept_sparse="csr",
843 reset=False,
844 order="C",
845 )
846
847 n_samples_fit = self.n_samples_fit_
848 if n_neighbors > n_samples_fit:
849 if query_is_train:
850 n_neighbors -= 1 # ok to modify inplace because an error is raised
851 inequality_str = "n_neighbors < n_samples_fit"
852 else:
853 inequality_str = "n_neighbors <= n_samples_fit"
854 raise ValueError(
855 f"Expected {inequality_str}, but "
856 f"n_neighbors = {n_neighbors}, n_samples_fit = {n_samples_fit}, "
857 f"n_samples = {X.shape[0]}" # include n_samples for common tests
858 )
859
860 n_jobs = effective_n_jobs(self.n_jobs)
861 chunked_results = None
862 use_pairwise_distances_reductions = (
863 self._fit_method == "brute"
864 and ArgKmin.is_usable_for(
865 X if X is not None else self._fit_X, self._fit_X, self.effective_metric_
866 )
867 )
868 if use_pairwise_distances_reductions:
869 results = ArgKmin.compute(
870 X=X,
871 Y=self._fit_X,
872 k=n_neighbors,
873 metric=self.effective_metric_,
874 metric_kwargs=self.effective_metric_params_,
875 strategy="auto",
876 return_distance=return_distance,
877 )
878
879 elif (
880 self._fit_method == "brute" and self.metric == "precomputed" and issparse(X)
881 ):
882 results = _kneighbors_from_graph(
883 X, n_neighbors=n_neighbors, return_distance=return_distance
884 )
885
886 elif self._fit_method == "brute":
887 # Joblib-based backend, which is used when user-defined callable
888 # are passed for metric.
889
890 # This won't be used in the future once PairwiseDistancesReductions
891 # support:
892 # - DistanceMetrics which work on supposedly binary data
893 # - CSR-dense and dense-CSR case if 'euclidean' in metric.
894 reduce_func = partial(
895 self._kneighbors_reduce_func,
896 n_neighbors=n_neighbors,
897 return_distance=return_distance,
898 )
899
900 # for efficiency, use squared euclidean distances
901 if self.effective_metric_ == "euclidean":
902 kwds = {"squared": True}
903 else:
904 kwds = self.effective_metric_params_
905
906 chunked_results = list(
907 pairwise_distances_chunked(
908 X,
909 self._fit_X,
910 reduce_func=reduce_func,
911 metric=self.effective_metric_,
912 n_jobs=n_jobs,
913 **kwds,
914 )
915 )
916
917 elif self._fit_method in ["ball_tree", "kd_tree"]:
918 if issparse(X):
919 raise ValueError(
920 "%s does not work with sparse matrices. Densify the data, "
921 "or set algorithm='brute'" % self._fit_method
922 )
923 chunked_results = Parallel(n_jobs, prefer="threads")(
924 delayed(self._tree.query)(X[s], n_neighbors, return_distance)
925 for s in gen_even_slices(X.shape[0], n_jobs)
926 )
927 else:
928 raise ValueError("internal: _fit_method not recognized")
929
930 if chunked_results is not None:
931 if return_distance:
932 neigh_dist, neigh_ind = zip(*chunked_results)
933 results = np.vstack(neigh_dist), np.vstack(neigh_ind)
934 else:
935 results = np.vstack(chunked_results)
936
937 if not query_is_train:
938 return results
939 else:
940 # If the query data is the same as the indexed data, we would like
941 # to ignore the first nearest neighbor of every sample, i.e
942 # the sample itself.
943 if return_distance:
944 neigh_dist, neigh_ind = results
945 else:
946 neigh_ind = results
947
948 n_queries, _ = X.shape
949 sample_range = np.arange(n_queries)[:, None]
950 sample_mask = neigh_ind != sample_range
951
952 # Corner case: When the number of duplicates are more
953 # than the number of neighbors, the first NN will not
954 # be the sample, but a duplicate.
955 # In that case mask the first duplicate.
956 dup_gr_nbrs = np.all(sample_mask, axis=1)
957 sample_mask[:, 0][dup_gr_nbrs] = False
958 neigh_ind = np.reshape(neigh_ind[sample_mask], (n_queries, n_neighbors - 1))
959
960 if return_distance:
961 neigh_dist = np.reshape(
962 neigh_dist[sample_mask], (n_queries, n_neighbors - 1)
963 )
964 return neigh_dist, neigh_ind
965 return neigh_ind

967 def kneighbors_graph(self, X=None, n_neighbors=None, mode="connectivity"):
968 """Compute the (weighted) graph of k-Neighbors for points in X.
969
970 Parameters
971 ----------
972 X : {array-like, sparse matrix} of shape (n_queries, n_features), \
973 or (n_queries, n_indexed) if metric == 'precomputed', default=None
974 The query point or points.
975 If not provided, neighbors of each indexed point are returned.
976 In this case, the query point is not considered its own neighbor.
977 For ``metric='precomputed'`` the shape should be
978 (n_queries, n_indexed). Otherwise the shape should be
979 (n_queries, n_features).
980
981 n_neighbors : int, default=None
982 Number of neighbors for each sample. The default is the value
983 passed to the constructor.
984
985 mode : {'connectivity', 'distance'}, default='connectivity'
986 Type of returned matrix: 'connectivity' will return the
987 connectivity matrix with ones and zeros, in 'distance' the
988 edges are distances between points, type of distance
989 depends on the selected metric parameter in
990 NearestNeighbors class.
991
992 Returns
993 -------
994 A : sparse-matrix of shape (n_queries, n_samples_fit)
995 `n_samples_fit` is the number of samples in the fitted data.
996 `A[i, j]` gives the weight of the edge connecting `i` to `j`.
997 The matrix is of CSR format.
998
999 See Also
1000 --------
1001 NearestNeighbors.radius_neighbors_graph : Compute the (weighted) graph
1002 of Neighbors for points in X.
1003
1004 Examples
1005 --------
1006 >>> X = [[0], [3], [1]]
1007 >>> from sklearn.neighbors import NearestNeighbors
1008 >>> neigh = NearestNeighbors(n_neighbors=2)
1009 >>> neigh.fit(X)
1010 NearestNeighbors(n_neighbors=2)
1011 >>> A = neigh.kneighbors_graph(X)
1012 >>> A.toarray()
1013 array([[1., 0., 1.],
1014 [0., 1., 1.],
1015 [1., 0., 1.]])
1016 """
1017 check_is_fitted(self)
1018 if n_neighbors is None:
1019 n_neighbors = self.n_neighbors
1020
1021 # check the input only in self.kneighbors
1022
1023 # construct CSR matrix representation of the k-NN graph
1024 if mode == "connectivity":
1025 A_ind = self.kneighbors(X, n_neighbors, return_distance=False)
1026 n_queries = A_ind.shape[0]
1027 A_data = np.ones(n_queries * n_neighbors)
1028
1029 elif mode == "distance":
1030 A_data, A_ind = self.kneighbors(X, n_neighbors, return_distance=True)
1031 A_data = np.ravel(A_data)
1032
1033 else:
1034 raise ValueError(
1035 'Unsupported mode, must be one of "connectivity", '
1036 f'or "distance" but got "{mode}" instead'
1037 )
1038
1039 n_queries = A_ind.shape[0]
1040 n_samples_fit = self.n_samples_fit_
1041 n_nonzero = n_queries * n_neighbors
1042 A_indptr = np.arange(0, n_nonzero + 1, n_neighbors)
1043
1044 kneighbors_graph = csr_matrix(
1045 (A_data, A_ind.ravel(), A_indptr), shape=(n_queries, n_samples_fit)
1046 )
1047
1048 return kneighbors_graph

1055 """Reduce a chunk of distances to the nearest neighbors.
1056
1057 Callback to :func:`sklearn.metrics.pairwise.pairwise_distances_chunked`
1058
1059 Parameters
1060 ----------
1061 dist : ndarray of shape (n_samples_chunk, n_samples)
1062 The distance matrix.
1063
1064 start : int
1065 The index in X which the first row of dist corresponds to.

1068 The radius considered when making the nearest neighbors search.

1070 return_distance : bool
1071 Whether or not to return the distances.
1072

1076 Returned only if `return_distance=True`.
1077

1088 results = dist, neigh_ind
1089 else:
1090 results = neigh_ind
1091 return results

1093 def radius_neighbors(
1094 self, X=None, radius=None, return_distance=True, sort_results=False
1095 ):
1096 """Find the neighbors within a given radius of a point or points.
1097
1098 Return the indices and distances of each point from the dataset
1099 lying in a ball with size ``radius`` around the points of the query
1100 array. Points lying on the boundary are included in the results.
1101
1102 The result points are *not* necessarily sorted by distance to their
1103 query point.
1104
1105 Parameters
1106 ----------
1107 X : {array-like, sparse matrix} of (n_samples, n_features), default=None
1108 The query point or points.
1109 If not provided, neighbors of each indexed point are returned.
1110 In this case, the query point is not considered its own neighbor.
1111
1112 radius : float, default=None
1113 Limiting distance of neighbors to return. The default is the value
1114 passed to the constructor.
1115
1116 return_distance : bool, default=True
1117 Whether or not to return the distances.
1118
1119 sort_results : bool, default=False
1120 If True, the distances and indices will be sorted by increasing
1121 distances before being returned. If False, the results may not
1122 be sorted. If `return_distance=False`, setting `sort_results=True`
1123 will result in an error.
1124
1125 .. versionadded:: 0.22
1126
1127 Returns
1128 -------
1129 neigh_dist : ndarray of shape (n_samples,) of arrays
1130 Array representing the distances to each point, only present if
1131 `return_distance=True`. The distance values are computed according
1132 to the ``metric`` constructor parameter.
1133
1134 neigh_ind : ndarray of shape (n_samples,) of arrays
1135 An array of arrays of indices of the approximate nearest points
1136 from the population matrix that lie within a ball of size
1137 ``radius`` around the query points.
1138
1139 Notes
1140 -----
1141 Because the number of neighbors of each point is not necessarily
1142 equal, the results for multiple query points cannot be fit in a
1143 standard data array.
1144 For efficiency, `radius_neighbors` returns arrays of objects, where
1145 each object is a 1D array of indices or distances.
1146
1147 Examples
1148 --------
1149 In the following example, we construct a NeighborsClassifier
1150 class from an array representing our data set and ask who's
1151 the closest point to [1, 1, 1]:
1152
1153 >>> import numpy as np
1154 >>> samples = [[0., 0., 0.], [0., .5, 0.], [1., 1., .5]]
1155 >>> from sklearn.neighbors import NearestNeighbors
1156 >>> neigh = NearestNeighbors(radius=1.6)
1157 >>> neigh.fit(samples)
1158 NearestNeighbors(radius=1.6)
1159 >>> rng = neigh.radius_neighbors([[1., 1., 1.]])
1160 >>> print(np.asarray(rng[0][0]))
1161 [1.5 0.5]
1162 >>> print(np.asarray(rng[1][0]))
1163 [1 2]
1164
1165 The first array returned contains the distances to all points which
1166 are closer than 1.6, while the second array returned contains their
1167 indices. In general, multiple points can be queried at the same time.
1168 """
1169 check_is_fitted(self)
1170
1171 if sort_results and not return_distance:
1172 raise ValueError("return_distance must be True if sort_results is True.")
1173

1177 X = self._fit_X
1178 else:

1180 X = _check_precomputed(X)
1181 else:
1182 X = validate_data(
1183 self,
1184 X,
1185 ensure_all_finite=ensure_all_finite,
1186 accept_sparse="csr",
1187 reset=False,
1188 order="C",
1189 )
1190
1191 if radius is None:
1192 radius = self.radius
1193
1194 use_pairwise_distances_reductions = (
1195 self._fit_method == "brute"
1196 and RadiusNeighbors.is_usable_for(

1198 )
1199 )

1279 for s in gen_even_slices(X.shape[0], n_jobs)
1280 )
1281 if return_distance:
1282 neigh_ind, neigh_dist = tuple(zip(*chunked_results))
1283 results = np.hstack(neigh_dist), np.hstack(neigh_ind)
1284 else:
1285 results = np.hstack(chunked_results)
1286 else:
1287 raise ValueError("internal: _fit_method not recognized")

1291 else:

1294 # the sample itself.

1297 else:
1298 neigh_ind = results
1299

1301 mask = ind_neighbor != ind
1302
1303 neigh_ind[ind] = ind_neighbor[mask]
1304 if return_distance:
1305 neigh_dist[ind] = neigh_dist[ind][mask]
1306
1307 if return_distance:

1311 def radius_neighbors_graph(
1312 self, X=None, radius=None, mode="connectivity", sort_results=False
1313 ):
1314 """Compute the (weighted) graph of Neighbors for points in X.
1315
1316 Neighborhoods are restricted the points at a distance lower than
1317 radius.
1318
1319 Parameters
1320 ----------
1321 X : {array-like, sparse matrix} of shape (n_samples, n_features), default=None
1322 The query point or points.
1323 If not provided, neighbors of each indexed point are returned.
1324 In this case, the query point is not considered its own neighbor.
1325
1326 radius : float, default=None
1327 Radius of neighborhoods. The default is the value passed to the
1328 constructor.
1329
1330 mode : {'connectivity', 'distance'}, default='connectivity'
1331 Type of returned matrix: 'connectivity' will return the
1332 connectivity matrix with ones and zeros, in 'distance' the
1333 edges are distances between points, type of distance
1334 depends on the selected metric parameter in
1335 NearestNeighbors class.
1336
1337 sort_results : bool, default=False
1338 If True, in each row of the result, the non-zero entries will be
1339 sorted by increasing distances. If False, the non-zero entries may
1340 not be sorted. Only used with mode='distance'.
1341
1342 .. versionadded:: 0.22
1343
1344 Returns
1345 -------
1346 A : sparse-matrix of shape (n_queries, n_samples_fit)
1347 `n_samples_fit` is the number of samples in the fitted data.
1348 `A[i, j]` gives the weight of the edge connecting `i` to `j`.
1349 The matrix is of CSR format.
1350
1351 See Also
1352 --------
1353 kneighbors_graph : Compute the (weighted) graph of k-Neighbors for
1354 points in X.
1355
1356 Examples
1357 --------
1358 >>> X = [[0], [3], [1]]
1359 >>> from sklearn.neighbors import NearestNeighbors
1360 >>> neigh = NearestNeighbors(radius=1.5)
1361 >>> neigh.fit(X)
1362 NearestNeighbors(radius=1.5)
1363 >>> A = neigh.radius_neighbors_graph(X)
1364 >>> A.toarray()
1365 array([[1., 0., 1.],
1366 [0., 1., 0.],
1367 [1., 0., 1.]])
1368 """
1369 check_is_fitted(self)
1370
1371 # check the input only in self.radius_neighbors
1372
1373 if radius is None:
1374 radius = self.radius
1375
1376 # construct CSR matrix representation of the NN graph
1377 if mode == "connectivity":
1378 A_ind = self.radius_neighbors(X, radius, return_distance=False)
1379 A_data = None
1380 elif mode == "distance":
1381 dist, A_ind = self.radius_neighbors(
1382 X, radius, return_distance=True, sort_results=sort_results
1383 )
1384 A_data = np.concatenate(list(dist))
1385 else:
1386 raise ValueError(
1387 'Unsupported mode, must be one of "connectivity", '
1388 f'or "distance" but got "{mode}" instead'
