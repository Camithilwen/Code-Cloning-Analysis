71# --- SIMILARITY SEARCH ---
72def find_top_k_similar(target_embedding, all_embeddings, target_file_id, target_collection, top_k):
73 similarities = []
74 for file_id, code, emb, collection_name in all_embeddings:
75 if file_id == target_file_id and collection_name == target_collection:
76 continue
77 sim = float(np.dot(target_embedding, emb) / (np.linalg.norm(target_embedding) * np.linalg.norm(emb)))
78 similarities.append((sim, file_id, code, collection_name))
79 similarities.sort(reverse=True)
80 return similarities[:top_k]

87def rag_similarity_assessment(target_code, similar_codes, model_name, target_fid, target_col, codenet_clone_labels, results_writer):
88 model = lms.llm(model_name)
89 context = "\n\n".join([f"Similar Code {i+1} (from {col}):\n{code}" for i, (_, _, code, col) in enumerate(similar_codes)])

91 prompt = f'''Given the target code and its most similar code files, assess the degree of similarity and provide a score (0-1) with explanation for each Type (Type-1, Type-2, Type-3, and Type-4),
92 given the following definitions:
93 Type-1 or identical code fragments represent the same code except for white
94 space, comments,
95 and layout. Type-2 or lexical code snippets represent identical clone pairs
96 except for differences in variables or function names with Type-1 clone
97 differences. Type-3 or syntactically represent similar code fragments
98 that differ at the statement level. The code fragments differ in some
99 lines with removed or added of some lines in addition to type-2 clone
100 differences. Type-4 or semantic code clone represents code snippets that
101 perform the same functionality but the implementation is different. In
102 global they are syntactically dissimilar.
103 Target Code:\n{target_code}\n\n{context}\n'''
104
105 schema = {
106 "type": "object",
107 "properties": {
108 "Type-1": {"type": "number", "minimum": 0, "maximum": 1},
109 "Type-2": {"type": "number", "minimum": 0, "maximum": 1},
110 "Type-3": {"type": "number", "minimum": 0, "maximum": 1},
111 "Type-4": {"type": "number", "minimum": 0, "maximum": 1},
112 },
113 "required": ["Type-1", "Type-2", "Type-3", "Type-4"]
114 }
115 response = model.respond(prompt, response_format=schema)
116 results = response.parsed
117
118 for i, (sim, file_id, code, col) in enumerate(similar_codes):
119 key = (target_fid, file_id)
120 ground_truth = codenet_clone_labels.get(key, "Unknown")
121 predicted_type = max(results, key=lambda t: results[t])
122 predicted_sim = type_to_binary(predicted_type)

124 results_writer.writerow([
