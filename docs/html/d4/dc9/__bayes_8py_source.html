26class BayesianRidge(RegressorMixin, LinearModel):
27 """Bayesian ridge regression.
28
29 Fit a Bayesian ridge model. See the Notes section for details on this
30 implementation and the optimization of the regularization parameters
31 lambda (precision of the weights) and alpha (precision of the noise).
32
33 Read more in the :ref:`User Guide <bayesian_regression>`.
34 For an intuitive visualization of how the sinusoid is approximated by
35 a polynomial using different pairs of initial values, see
36 :ref:`sphx_glr_auto_examples_linear_model_plot_bayesian_ridge_curvefit.py`.
37
38 Parameters
39 ----------
40 max_iter : int, default=300
41 Maximum number of iterations over the complete dataset before
42 stopping independently of any early stopping criterion.
43
44 .. versionchanged:: 1.3
45
46 tol : float, default=1e-3
47 Stop the algorithm if w has converged.
48
49 alpha_1 : float, default=1e-6
50 Hyper-parameter : shape parameter for the Gamma distribution prior
51 over the alpha parameter.
52
53 alpha_2 : float, default=1e-6
54 Hyper-parameter : inverse scale parameter (rate parameter) for the
55 Gamma distribution prior over the alpha parameter.
56
57 lambda_1 : float, default=1e-6
58 Hyper-parameter : shape parameter for the Gamma distribution prior
59 over the lambda parameter.
60
61 lambda_2 : float, default=1e-6
62 Hyper-parameter : inverse scale parameter (rate parameter) for the
63 Gamma distribution prior over the lambda parameter.
64
65 alpha_init : float, default=None
66 Initial value for alpha (precision of the noise).
67 If not set, alpha_init is 1/Var(y).
68
69 .. versionadded:: 0.22
70
71 lambda_init : float, default=None
72 Initial value for lambda (precision of the weights).
73 If not set, lambda_init is 1.
74
75 .. versionadded:: 0.22
76
77 compute_score : bool, default=False
78 If True, compute the log marginal likelihood at each iteration of the
79 optimization.
80
81 fit_intercept : bool, default=True
82 Whether to calculate the intercept for this model.
83 The intercept is not treated as a probabilistic parameter
84 and thus has no associated variance. If set
85 to False, no intercept will be used in calculations
86 (i.e. data is expected to be centered).
87
88 copy_X : bool, default=True
89 If True, X will be copied; else, it may be overwritten.
90
91 verbose : bool, default=False
92 Verbose mode when fitting the model.
93
94 Attributes
95 ----------
96 coef_ : array-like of shape (n_features,)
97 Coefficients of the regression model (mean of distribution)
98
99 intercept_ : float
100 Independent term in decision function. Set to 0.0 if
101 `fit_intercept = False`.
102
103 alpha_ : float
104 Estimated precision of the noise.
105
106 lambda_ : float
107 Estimated precision of the weights.
108
109 sigma_ : array-like of shape (n_features, n_features)
110 Estimated variance-covariance matrix of the weights
111
112 scores_ : array-like of shape (n_iter_+1,)
113 If computed_score is True, value of the log marginal likelihood (to be
114 maximized) at each iteration of the optimization. The array starts
115 with the value of the log marginal likelihood obtained for the initial
116 values of alpha and lambda and ends with the value obtained for the
117 estimated alpha and lambda.
118
119 n_iter_ : int
120 The actual number of iterations to reach the stopping criterion.
121
122 X_offset_ : ndarray of shape (n_features,)
123 If `fit_intercept=True`, offset subtracted for centering data to a
124 zero mean. Set to np.zeros(n_features) otherwise.
125
126 X_scale_ : ndarray of shape (n_features,)
127 Set to np.ones(n_features).
128
129 n_features_in_ : int
130 Number of features seen during :term:`fit`.
131
132 .. versionadded:: 0.24
133
134 feature_names_in_ : ndarray of shape (`n_features_in_`,)
135 Names of features seen during :term:`fit`. Defined only when `X`
136 has feature names that are all strings.
137
138 .. versionadded:: 1.0
139
140 See Also
141 --------
142 ARDRegression : Bayesian ARD regression.
143
144 Notes
145 -----
146 There exist several strategies to perform Bayesian ridge regression. This
147 implementation is based on the algorithm described in Appendix A of
148 (Tipping, 2001) where updates of the regularization parameters are done as
149 suggested in (MacKay, 1992). Note that according to A New
150 View of Automatic Relevance Determination (Wipf and Nagarajan, 2008) these
151 update rules do not guarantee that the marginal likelihood is increasing
152 between two consecutive iterations of the optimization.
153
154 References
155 ----------
156 D. J. C. MacKay, Bayesian Interpolation, Computation and Neural Systems,
157 Vol. 4, No. 3, 1992.
158
159 M. E. Tipping, Sparse Bayesian Learning and the Relevance Vector Machine,
160 Journal of Machine Learning Research, Vol. 1, 2001.
161
162 Examples
163 --------
164 >>> from sklearn import linear_model
165 >>> clf = linear_model.BayesianRidge()
166 >>> clf.fit([[0,0], [1, 1], [2, 2]], [0, 1, 2])
167 BayesianRidge()
168 >>> clf.predict([[1, 1]])
169 array([1.])
170 """

216 @_fit_context(prefer_skip_nested_validation=True)
217 def fit(self, X, y, sample_weight=None):
218 """Fit the model.
219
220 Parameters
221 ----------
222 X : ndarray of shape (n_samples, n_features)
223 Training data.
224 y : ndarray of shape (n_samples,)
225 Target values. Will be cast to X's dtype if necessary.
226
227 sample_weight : ndarray of shape (n_samples,), default=None
228 Individual weights for each sample.
229
230 .. versionadded:: 0.20
231 parameter *sample_weight* support to BayesianRidge.
232
233 Returns
234 -------
235 self : object
236 Returns the instance itself.
237 """
238 X, y = validate_data(
239 self,
240 X,
241 y,
242 dtype=[np.float64, np.float32],
243 force_writeable=True,
244 y_numeric=True,
245 )
246 dtype = X.dtype
247 n_samples, n_features = X.shape
248
249 sw_sum = n_samples
250 y_var = y.var()
251 if sample_weight is not None:
252 sample_weight = _check_sample_weight(sample_weight, X, dtype=dtype)
253 sw_sum = sample_weight.sum()
254 y_mean = np.average(y, weights=sample_weight)
255 y_var = np.average((y - y_mean) ** 2, weights=sample_weight)
256
257 X, y, X_offset_, y_offset_, X_scale_ = _preprocess_data(
258 X,
259 y,
260 fit_intercept=self.fit_intercept,
261 copy=self.copy_X,
262 sample_weight=sample_weight,
263 )
264
265 if sample_weight is not None:
266 # Sample weight can be implemented via a simple rescaling.
267 X, y, _ = _rescale_data(X, y, sample_weight)
268
269 self.X_offset_ = X_offset_
270 self.X_scale_ = X_scale_
271
272 # Initialization of the values of the parameters
273 eps = np.finfo(np.float64).eps
274 # Add `eps` in the denominator to omit division by zero
275 alpha_ = self.alpha_init
276 lambda_ = self.lambda_init
277 if alpha_ is None:
278 alpha_ = 1.0 / (y_var + eps)
279 if lambda_ is None:
280 lambda_ = 1.0
281
282 # Avoid unintended type promotion to float64 with numpy 2
283 alpha_ = np.asarray(alpha_, dtype=dtype)
284 lambda_ = np.asarray(lambda_, dtype=dtype)
285
286 verbose = self.verbose
287 lambda_1 = self.lambda_1
288 lambda_2 = self.lambda_2
289 alpha_1 = self.alpha_1
290 alpha_2 = self.alpha_2
291
292 self.scores_ = list()
293 coef_old_ = None
294
295 XT_y = np.dot(X.T, y)
296 # Let M, N = n_samples, n_features and K = min(M, N).
297 # The posterior covariance matrix needs Vh_full: (N, N).
298 # The full SVD is only required when n_samples < n_features.
299 # When n_samples < n_features, K=M and full_matrices=True
300 # U: (M, M), S: M, Vh_full: (N, N), Vh: (M, N)
301 # When n_samples > n_features, K=N and full_matrices=False
302 # U: (M, N), S: N, Vh_full: (N, N), Vh: (N, N)
303 U, S, Vh_full = linalg.svd(X, full_matrices=(n_samples < n_features))
304 K = len(S)
305 eigen_vals_ = S**2
306 eigen_vals_full = np.zeros(n_features, dtype=dtype)
307 eigen_vals_full[0:K] = eigen_vals_
308 Vh = Vh_full[0:K, :]
309
310 # Convergence loop of the bayesian ridge regression
311 for iter_ in range(self.max_iter):
312 # update posterior mean coef_ based on alpha_ and lambda_ and
313 # compute corresponding sse (sum of squared errors)
314 coef_, sse_ = self._update_coef_(
315 X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
316 )
317 if self.compute_score:
318 # compute the log marginal likelihood
319 s = self._log_marginal_likelihood(
320 n_samples,
321 n_features,
322 sw_sum,
323 eigen_vals_,
324 alpha_,
325 lambda_,
326 coef_,
327 sse_,
328 )
329 self.scores_.append(s)
330
331 # Update alpha and lambda according to (MacKay, 1992)
332 gamma_ = np.sum((alpha_ * eigen_vals_) / (lambda_ + alpha_ * eigen_vals_))
333 lambda_ = (gamma_ + 2 * lambda_1) / (np.sum(coef_**2) + 2 * lambda_2)
334 alpha_ = (sw_sum - gamma_ + 2 * alpha_1) / (sse_ + 2 * alpha_2)
335
336 # Check for convergence
337 if iter_ != 0 and np.sum(np.abs(coef_old_ - coef_)) < self.tol:
338 if verbose:
339 print("Convergence after ", str(iter_), " iterations")
340 break
341 coef_old_ = np.copy(coef_)
342
343 self.n_iter_ = iter_ + 1
344
345 # return regularization parameters and corresponding posterior mean,
346 # log marginal likelihood and posterior covariance
347 self.alpha_ = alpha_
348 self.lambda_ = lambda_
349 self.coef_, sse_ = self._update_coef_(
350 X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
351 )
352 if self.compute_score:
353 # compute the log marginal likelihood
354 s = self._log_marginal_likelihood(
355 n_samples,
356 n_features,
357 sw_sum,
358 eigen_vals_,
359 alpha_,
360 lambda_,
361 coef_,
362 sse_,
363 )
364 self.scores_.append(s)
365 self.scores_ = np.array(self.scores_)
366
367 # posterior covariance
368 self.sigma_ = np.dot(
369 Vh_full.T, Vh_full / (alpha_ * eigen_vals_full + lambda_)[:, np.newaxis]
370 )
371
372 self._set_intercept(X_offset_, y_offset_, X_scale_)
373
374 return self

376 def predict(self, X, return_std=False):
377 """Predict using the linear model.
378
379 In addition to the mean of the predictive distribution, also its
380 standard deviation can be returned.
381
382 Parameters
383 ----------
384 X : {array-like, sparse matrix} of shape (n_samples, n_features)
385 Samples.
386
387 return_std : bool, default=False
388 Whether to return the standard deviation of posterior prediction.
389
390 Returns
391 -------
392 y_mean : array-like of shape (n_samples,)
393 Mean of predictive distribution of query points.
394
395 y_std : array-like of shape (n_samples,)
396 Standard deviation of predictive distribution of query points.
397 """
398 y_mean = self._decision_function(X)
399 if not return_std:
400 return y_mean
401 else:
402 sigmas_squared_data = (np.dot(X, self.sigma_) * X).sum(axis=1)
403 y_std = np.sqrt(sigmas_squared_data + (1.0 / self.alpha_))
404 return y_mean, y_std
405
406 def _update_coef_(
407 self, X, y, n_samples, n_features, XT_y, U, Vh, eigen_vals_, alpha_, lambda_
408 ):
409 """Update posterior mean and compute corresponding sse (sum of squared errors).
410
411 Posterior mean is given by coef_ = scaled_sigma_ * X.T * y where
412 scaled_sigma_ = (lambda_/alpha_ * np.eye(n_features)
413 + np.dot(X.T, X))^-1
414 """
415
416 if n_samples > n_features:
417 coef_ = np.linalg.multi_dot(
418 [Vh.T, Vh / (eigen_vals_ + lambda_ / alpha_)[:, np.newaxis], XT_y]
419 )
420 else:
421 coef_ = np.linalg.multi_dot(
422 [X.T, U / (eigen_vals_ + lambda_ / alpha_)[None, :], U.T, y]
423 )
424
425 # Note: we do not need to explicitly use the weights in this sum because
426 # y and X were preprocessed by _rescale_data to handle the weights.
427 sse_ = np.sum((y - np.dot(X, coef_)) ** 2)
428
429 return coef_, sse_
430
431 def _log_marginal_likelihood(
432 self, n_samples, n_features, sw_sum, eigen_vals, alpha_, lambda_, coef, sse
433 ):
434 """Log marginal likelihood."""
435 alpha_1 = self.alpha_1
436 alpha_2 = self.alpha_2
437 lambda_1 = self.lambda_1
438 lambda_2 = self.lambda_2
439
440 # compute the log of the determinant of the posterior covariance.
441 # posterior covariance is given by
442 # sigma = (lambda_ * np.eye(n_features) + alpha_ * np.dot(X.T, X))^-1
443 if n_samples > n_features:
444 logdet_sigma = -np.sum(np.log(lambda_ + alpha_ * eigen_vals))
445 else:
446 logdet_sigma = np.full(n_features, lambda_, dtype=np.array(lambda_).dtype)
447 logdet_sigma[:n_samples] += alpha_ * eigen_vals
448 logdet_sigma = -np.sum(np.log(logdet_sigma))
449
450 score = lambda_1 * log(lambda_) - lambda_2 * lambda_
451 score += alpha_1 * log(alpha_) - alpha_2 * alpha_
452 score += 0.5 * (
453 n_features * log(lambda_)
454 + sw_sum * log(alpha_)
455 - alpha_ * sse
456 - lambda_ * np.sum(coef**2)
457 + logdet_sigma
458 - sw_sum * log(2 * np.pi)
459 )
460
461 return score

468class ARDRegression(RegressorMixin, LinearModel):
469 """Bayesian ARD regression.
470
471 Fit the weights of a regression model, using an ARD prior. The weights of
472 the regression model are assumed to be in Gaussian distributions.
473 Also estimate the parameters lambda (precisions of the distributions of the
474 weights) and alpha (precision of the distribution of the noise).
475 The estimation is done by an iterative procedures (Evidence Maximization)
476
477 Read more in the :ref:`User Guide <bayesian_regression>`.
478
479 Parameters
480 ----------
481 max_iter : int, default=300
482 Maximum number of iterations.
483
484 .. versionchanged:: 1.3
485
486 tol : float, default=1e-3
487 Stop the algorithm if w has converged.
488
489 alpha_1 : float, default=1e-6
490 Hyper-parameter : shape parameter for the Gamma distribution prior
491 over the alpha parameter.
492
493 alpha_2 : float, default=1e-6
494 Hyper-parameter : inverse scale parameter (rate parameter) for the
495 Gamma distribution prior over the alpha parameter.
496
497 lambda_1 : float, default=1e-6
498 Hyper-parameter : shape parameter for the Gamma distribution prior
499 over the lambda parameter.
500
501 lambda_2 : float, default=1e-6
502 Hyper-parameter : inverse scale parameter (rate parameter) for the
503 Gamma distribution prior over the lambda parameter.
504
505 compute_score : bool, default=False
506 If True, compute the objective function at each step of the model.
507
508 threshold_lambda : float, default=10 000
509 Threshold for removing (pruning) weights with high precision from
510 the computation.
511
512 fit_intercept : bool, default=True
513 Whether to calculate the intercept for this model. If set
514 to false, no intercept will be used in calculations
515 (i.e. data is expected to be centered).
516
517 copy_X : bool, default=True
518 If True, X will be copied; else, it may be overwritten.
519
520 verbose : bool, default=False
521 Verbose mode when fitting the model.
522
523 Attributes
524 ----------
525 coef_ : array-like of shape (n_features,)
526 Coefficients of the regression model (mean of distribution)
527
528 alpha_ : float
529 estimated precision of the noise.
530
531 lambda_ : array-like of shape (n_features,)
532 estimated precisions of the weights.
533
534 sigma_ : array-like of shape (n_features, n_features)
535 estimated variance-covariance matrix of the weights
536
537 scores_ : float
538 if computed, value of the objective function (to be maximized)
539
540 n_iter_ : int
541 The actual number of iterations to reach the stopping criterion.
542
543 .. versionadded:: 1.3
544
545 intercept_ : float
546 Independent term in decision function. Set to 0.0 if
547 ``fit_intercept = False``.
548
549 X_offset_ : float
550 If `fit_intercept=True`, offset subtracted for centering data to a
551 zero mean. Set to np.zeros(n_features) otherwise.
552
553 X_scale_ : float
554 Set to np.ones(n_features).
555
556 n_features_in_ : int
557 Number of features seen during :term:`fit`.
558
559 .. versionadded:: 0.24
560
561 feature_names_in_ : ndarray of shape (`n_features_in_`,)
562 Names of features seen during :term:`fit`. Defined only when `X`
563 has feature names that are all strings.
564
565 .. versionadded:: 1.0
566
567 See Also
568 --------
569 BayesianRidge : Bayesian ridge regression.
570
571 References
572 ----------
573 D. J. C. MacKay, Bayesian nonlinear modeling for the prediction
574 competition, ASHRAE Transactions, 1994.
575
576 R. Salakhutdinov, Lecture notes on Statistical Machine Learning,
577 http://www.utstat.toronto.edu/~rsalakhu/sta4273/notes/Lecture2.pdf#page=15
578 Their beta is our ``self.alpha_``
579 Their alpha is our ``self.lambda_``
580 ARD is a little different than the slide: only dimensions/features for
581 which ``self.lambda_ < self.threshold_lambda`` are kept and the rest are
582 discarded.
583
584 Examples
585 --------
586 >>> from sklearn import linear_model
587 >>> clf = linear_model.ARDRegression()
588 >>> clf.fit([[0,0], [1, 1], [2, 2]], [0, 1, 2])
589 ARDRegression()
590 >>> clf.predict([[1, 1]])
591 array([1.])
592
593 - :ref:`sphx_glr_auto_examples_linear_model_plot_ard.py` demonstrates ARD
594 Regression.
595 - :ref:`sphx_glr_auto_examples_linear_model_plot_lasso_and_elasticnet.py`
596 showcases ARD Regression alongside Lasso and Elastic-Net for sparse,
597 correlated signals, in the presence of noise.
598 """

641 @_fit_context(prefer_skip_nested_validation=True)
642 def fit(self, X, y):
643 """Fit the model according to the given training data and parameters.
644
645 Iterative procedure to maximize the evidence
646
647 Parameters
648 ----------
649 X : array-like of shape (n_samples, n_features)
650 Training vector, where `n_samples` is the number of samples and
651 `n_features` is the number of features.
652 y : array-like of shape (n_samples,)
653 Target values (integers). Will be cast to X's dtype if necessary.
654
655 Returns
656 -------
657 self : object
658 Fitted estimator.
659 """
660 X, y = validate_data(
661 self,
662 X,
663 y,
664 dtype=[np.float64, np.float32],
665 force_writeable=True,
666 y_numeric=True,
667 ensure_min_samples=2,
668 )
669 dtype = X.dtype
670
671 n_samples, n_features = X.shape
672 coef_ = np.zeros(n_features, dtype=dtype)
673
674 X, y, X_offset_, y_offset_, X_scale_ = _preprocess_data(
675 X, y, fit_intercept=self.fit_intercept, copy=self.copy_X
676 )
677
678 self.X_offset_ = X_offset_
679 self.X_scale_ = X_scale_
680
681 # Launch the convergence loop
682 keep_lambda = np.ones(n_features, dtype=bool)
683
684 lambda_1 = self.lambda_1
685 lambda_2 = self.lambda_2
686 alpha_1 = self.alpha_1
687 alpha_2 = self.alpha_2
688 verbose = self.verbose
689
690 # Initialization of the values of the parameters
691 eps = np.finfo(np.float64).eps
692 # Add `eps` in the denominator to omit division by zero if `np.var(y)`
693 # is zero.
694 # Explicitly set dtype to avoid unintended type promotion with numpy 2.
695 alpha_ = np.asarray(1.0 / (np.var(y) + eps), dtype=dtype)
696 lambda_ = np.ones(n_features, dtype=dtype)
697
698 self.scores_ = list()
699 coef_old_ = None
700
701 def update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_):
702 coef_[keep_lambda] = alpha_ * np.linalg.multi_dot(
703 [sigma_, X[:, keep_lambda].T, y]
704 )
705 return coef_
706
707 update_sigma = (
708 self._update_sigma
709 if n_samples >= n_features
710 else self._update_sigma_woodbury
711 )
712 # Iterative procedure of ARDRegression
713 for iter_ in range(self.max_iter):
714 sigma_ = update_sigma(X, alpha_, lambda_, keep_lambda)
715 coef_ = update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_)
716
717 # Update alpha and lambda
718 sse_ = np.sum((y - np.dot(X, coef_)) ** 2)
719 gamma_ = 1.0 - lambda_[keep_lambda] * np.diag(sigma_)
720 lambda_[keep_lambda] = (gamma_ + 2.0 * lambda_1) / (
721 (coef_[keep_lambda]) ** 2 + 2.0 * lambda_2
722 )
723 alpha_ = (n_samples - gamma_.sum() + 2.0 * alpha_1) / (sse_ + 2.0 * alpha_2)
724
725 # Prune the weights with a precision over a threshold
726 keep_lambda = lambda_ < self.threshold_lambda
727 coef_[~keep_lambda] = 0
728
729 # Compute the objective function
730 if self.compute_score:
731 s = (lambda_1 * np.log(lambda_) - lambda_2 * lambda_).sum()
732 s += alpha_1 * log(alpha_) - alpha_2 * alpha_
733 s += 0.5 * (
734 fast_logdet(sigma_)
735 + n_samples * log(alpha_)
736 + np.sum(np.log(lambda_))
737 )
738 s -= 0.5 * (alpha_ * sse_ + (lambda_ * coef_**2).sum())
739 self.scores_.append(s)
740
741 # Check for convergence
742 if iter_ > 0 and np.sum(np.abs(coef_old_ - coef_)) < self.tol:
743 if verbose:
744 print("Converged after %s iterations" % iter_)
745 break
746 coef_old_ = np.copy(coef_)
747
748 if not keep_lambda.any():
749 break
750
751 self.n_iter_ = iter_ + 1
752
753 if keep_lambda.any():
754 # update sigma and mu using updated params from the last iteration
755 sigma_ = update_sigma(X, alpha_, lambda_, keep_lambda)
756 coef_ = update_coeff(X, y, coef_, alpha_, keep_lambda, sigma_)
757 else:
758 sigma_ = np.array([]).reshape(0, 0)
759
760 self.coef_ = coef_
761 self.alpha_ = alpha_
762 self.sigma_ = sigma_
763 self.lambda_ = lambda_
764 self._set_intercept(X_offset_, y_offset_, X_scale_)
765 return self

767 def _update_sigma_woodbury(self, X, alpha_, lambda_, keep_lambda):
768 # See slides as referenced in the docstring note
769 # this function is used when n_samples < n_features and will invert
770 # a matrix of shape (n_samples, n_samples) making use of the
771 # woodbury formula:
772 # https://en.wikipedia.org/wiki/Woodbury_matrix_identity
773 n_samples = X.shape[0]
774 X_keep = X[:, keep_lambda]
775 inv_lambda = 1 / lambda_[keep_lambda].reshape(1, -1)
776 sigma_ = pinvh(
777 np.eye(n_samples, dtype=X.dtype) / alpha_
778 + np.dot(X_keep * inv_lambda, X_keep.T)
779 )
780 sigma_ = np.dot(sigma_, X_keep * inv_lambda)
781 sigma_ = -np.dot(inv_lambda.reshape(-1, 1) * X_keep.T, sigma_)
782 sigma_[np.diag_indices(sigma_.shape[1])] += 1.0 / lambda_[keep_lambda]
783 return sigma_

785 def _update_sigma(self, X, alpha_, lambda_, keep_lambda):
786 # See slides as referenced in the docstring note
787 # this function is used when n_samples >= n_features and will
788 # invert a matrix of shape (n_features, n_features)
789 X_keep = X[:, keep_lambda]
790 gram = np.dot(X_keep.T, X_keep)
791 eye = np.eye(gram.shape[0], dtype=X.dtype)
792 sigma_inv = lambda_[keep_lambda] * eye + alpha_ * gram
793 sigma_ = pinvh(sigma_inv)
794 return sigma_

796 def predict(self, X, return_std=False):
797 """Predict using the linear model.
798
799 In addition to the mean of the predictive distribution, also its
800 standard deviation can be returned.
801
802 Parameters
803 ----------
804 X : {array-like, sparse matrix} of shape (n_samples, n_features)
805 Samples.
806
807 return_std : bool, default=False
808 Whether to return the standard deviation of posterior prediction.
809
810 Returns
811 -------
812 y_mean : array-like of shape (n_samples,)
813 Mean of predictive distribution of query points.
814
815 y_std : array-like of shape (n_samples,)
816 Standard deviation of predictive distribution of query points.
817 """
818 y_mean = self._decision_function(X)
819 if return_std is False:
820 return y_mean
821 else:
822 col_index = self.lambda_ < self.threshold_lambda
823 X = _safe_indexing(X, indices=col_index, axis=1)
824 sigmas_squared_data = (np.dot(X, self.sigma_) * X).sum(axis=1)
825 y_std = np.sqrt(sigmas_squared_data + (1.0 / self.alpha_))
826 return y_mean, y_std
