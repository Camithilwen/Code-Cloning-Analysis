32def assert_csr_equal_values(current, expected):
33 current.eliminate_zeros()
34 expected.eliminate_zeros()
35 expected = expected.astype(current.dtype)
36 assert current.shape[0] == expected.shape[0]
37 assert current.shape[1] == expected.shape[1]
38 assert_array_equal(current.data, expected.data)
39 assert_array_equal(current.indices, expected.indices)
40 assert_array_equal(current.indptr, expected.indptr)

43def _make_dense_dataset(float_dtype):

49def _make_sparse_dataset(csr_container, float_dtype):
50 if float_dtype == np.float32:
51 X, y, sample_weight, csr_dataset = X32, y32, sample_weight32, CSRDataset32
52 else:
53 X, y, sample_weight, csr_dataset = X64, y64, sample_weight64, CSRDataset64
54 X = csr_container(X)
55 return csr_dataset(X.data, X.indptr, X.indices, y, sample_weight, seed=42)

58def _make_dense_datasets():
59 return [_make_dense_dataset(float_dtype) for float_dtype in floating]

62def _make_sparse_datasets():

69def _make_fused_types_datasets():

77def test_seq_dataset_basic_iteration(dataset, csr_container):
78 NUMBER_OF_RUNS = 5
79 X_csr64 = csr_container(X64)
80 for _ in range(NUMBER_OF_RUNS):
