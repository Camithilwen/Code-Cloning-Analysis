28def _iterate_sparse_X(X):
29 """This little hack returns a densified row when iterating over a sparse
30 matrix, instead of constructing a sparse matrix for every row that is
31 expensive.
32 """
33 n_samples = X.shape[0]
34 X_indices = X.indices
35 X_data = X.data
36 X_indptr = X.indptr
37
38 for i in range(n_samples):
39 row = np.zeros(X.shape[1])
40 startptr, endptr = X_indptr[i], X_indptr[i + 1]
41 nonzero_indices = X_indices[startptr:endptr]
42 row[nonzero_indices] = X_data[startptr:endptr]
43 yield row

46def _split_node(node, threshold, branching_factor):
47 """The node has to be split if there is no place for a new subcluster
48 in the node.
49 1. Two empty nodes and two empty subclusters are initialized.
50 2. The pair of distant subclusters are found.
51 3. The properties of the empty subclusters and nodes are updated
52 according to the nearest distance between the subclusters to the
53 pair of distant subclusters.
54 4. The two nodes are set as children to the two subclusters.
55 """
56 new_subcluster1 = _CFSubcluster()
57 new_subcluster2 = _CFSubcluster()
58 new_node1 = _CFNode(
59 threshold=threshold,
60 branching_factor=branching_factor,
61 is_leaf=node.is_leaf,
62 n_features=node.n_features,
63 dtype=node.init_centroids_.dtype,
64 )
65 new_node2 = _CFNode(
66 threshold=threshold,
67 branching_factor=branching_factor,
68 is_leaf=node.is_leaf,
69 n_features=node.n_features,
70 dtype=node.init_centroids_.dtype,
71 )
72 new_subcluster1.child_ = new_node1
73 new_subcluster2.child_ = new_node2
74
75 if node.is_leaf:
76 if node.prev_leaf_ is not None:
77 node.prev_leaf_.next_leaf_ = new_node1
78 new_node1.prev_leaf_ = node.prev_leaf_
79 new_node1.next_leaf_ = new_node2
80 new_node2.prev_leaf_ = new_node1
81 new_node2.next_leaf_ = node.next_leaf_
82 if node.next_leaf_ is not None:
83 node.next_leaf_.prev_leaf_ = new_node2
84
85 dist = euclidean_distances(
86 node.centroids_, Y_norm_squared=node.squared_norm_, squared=True
87 )
88 n_clusters = dist.shape[0]
89
90 farthest_idx = np.unravel_index(dist.argmax(), (n_clusters, n_clusters))
91 node1_dist, node2_dist = dist[(farthest_idx,)]
92
93 node1_closer = node1_dist < node2_dist
94 # make sure node1 is closest to itself even if all distances are equal.
95 # This can only happen when all node.centroids_ are duplicates leading to all
96 # distances between centroids being zero.
97 node1_closer[farthest_idx[0]] = True
98
99 for idx, subcluster in enumerate(node.subclusters_):
100 if node1_closer[idx]:
101 new_node1.append_subcluster(subcluster)
102 new_subcluster1.update(subcluster)
103 else:
104 new_node2.append_subcluster(subcluster)
105 new_subcluster2.update(subcluster)
106 return new_subcluster1, new_subcluster2

109class _CFNode:
110 """Each node in a CFTree is called a CFNode.
111
112 The CFNode can have a maximum of branching_factor
113 number of CFSubclusters.
114
115 Parameters
116 ----------
117 threshold : float
118 Threshold needed for a new subcluster to enter a CFSubcluster.
119
120 branching_factor : int
121 Maximum number of CF subclusters in each node.
122
123 is_leaf : bool
124 We need to know if the CFNode is a leaf or not, in order to
125 retrieve the final subclusters.
126
127 n_features : int
128 The number of features.
129
130 Attributes
131 ----------
132 subclusters_ : list
133 List of subclusters for a particular CFNode.
134
135 prev_leaf_ : _CFNode
136 Useful only if is_leaf is True.
137
138 next_leaf_ : _CFNode
139 next_leaf. Useful only if is_leaf is True.
140 the final subclusters.
141
142 init_centroids_ : ndarray of shape (branching_factor + 1, n_features)
143 Manipulate ``init_centroids_`` throughout rather than centroids_ since
144 the centroids are just a view of the ``init_centroids_`` .
145
146 init_sq_norm_ : ndarray of shape (branching_factor + 1,)
147 manipulate init_sq_norm_ throughout. similar to ``init_centroids_``.
148
149 centroids_ : ndarray of shape (branching_factor + 1, n_features)
150 View of ``init_centroids_``.
151
152 squared_norm_ : ndarray of shape (branching_factor + 1,)
153 View of ``init_sq_norm_``.
154
155 """
156
157 def __init__(self, *, threshold, branching_factor, is_leaf, n_features, dtype):
158 self.threshold = threshold
159 self.branching_factor = branching_factor
160 self.is_leaf = is_leaf
161 self.n_features = n_features
162
163 # The list of subclusters, centroids and squared norms
164 # to manipulate throughout.
165 self.subclusters_ = []
166 self.init_centroids_ = np.zeros((branching_factor + 1, n_features), dtype=dtype)
167 self.init_sq_norm_ = np.zeros((branching_factor + 1), dtype)
168 self.squared_norm_ = []
169 self.prev_leaf_ = None
170 self.next_leaf_ = None
171
172 def append_subcluster(self, subcluster):
173 n_samples = len(self.subclusters_)
174 self.subclusters_.append(subcluster)
175 self.init_centroids_[n_samples] = subcluster.centroid_
176 self.init_sq_norm_[n_samples] = subcluster.sq_norm_
177
178 # Keep centroids and squared norm as views. In this way
179 # if we change init_centroids and init_sq_norm_, it is
180 # sufficient,
181 self.centroids_ = self.init_centroids_[: n_samples + 1, :]
182 self.squared_norm_ = self.init_sq_norm_[: n_samples + 1]
183
184 def update_split_subclusters(self, subcluster, new_subcluster1, new_subcluster2):
185 """Remove a subcluster from a node and update it with the
186 split subclusters.
187 """
188 ind = self.subclusters_.index(subcluster)
189 self.subclusters_[ind] = new_subcluster1
190 self.init_centroids_[ind] = new_subcluster1.centroid_
191 self.init_sq_norm_[ind] = new_subcluster1.sq_norm_
192 self.append_subcluster(new_subcluster2)
193
194 def insert_cf_subcluster(self, subcluster):
195 """Insert a new subcluster into the node."""
196 if not self.subclusters_:
197 self.append_subcluster(subcluster)
198 return False
199
200 threshold = self.threshold
201 branching_factor = self.branching_factor
202 # We need to find the closest subcluster among all the
203 # subclusters so that we can insert our new subcluster.
204 dist_matrix = np.dot(self.centroids_, subcluster.centroid_)
205 dist_matrix *= -2.0
206 dist_matrix += self.squared_norm_
207 closest_index = np.argmin(dist_matrix)
208 closest_subcluster = self.subclusters_[closest_index]
209
210 # If the subcluster has a child, we need a recursive strategy.
211 if closest_subcluster.child_ is not None:
212 split_child = closest_subcluster.child_.insert_cf_subcluster(subcluster)
213
214 if not split_child:
215 # If it is determined that the child need not be split, we
216 # can just update the closest_subcluster
217 closest_subcluster.update(subcluster)
218 self.init_centroids_[closest_index] = self.subclusters_[
219 closest_index
220 ].centroid_
221 self.init_sq_norm_[closest_index] = self.subclusters_[
222 closest_index
223 ].sq_norm_
224 return False
225
226 # things not too good. we need to redistribute the subclusters in
227 # our child node, and add a new subcluster in the parent
228 # subcluster to accommodate the new child.
229 else:
230 new_subcluster1, new_subcluster2 = _split_node(
231 closest_subcluster.child_,
232 threshold,
233 branching_factor,
234 )
235 self.update_split_subclusters(
236 closest_subcluster, new_subcluster1, new_subcluster2
237 )
238
239 if len(self.subclusters_) > self.branching_factor:
240 return True
241 return False
242
243 # good to go!
244 else:
245 merged = closest_subcluster.merge_subcluster(subcluster, self.threshold)
246 if merged:
247 self.init_centroids_[closest_index] = closest_subcluster.centroid_
248 self.init_sq_norm_[closest_index] = closest_subcluster.sq_norm_
249 return False
250
251 # not close to any other subclusters, and we still
252 # have space, so add.
253 elif len(self.subclusters_) < self.branching_factor:
254 self.append_subcluster(subcluster)
255 return False
256
257 # We do not have enough space nor is it closer to an
258 # other subcluster. We need to split.
259 else:
260 self.append_subcluster(subcluster)
261 return True

264class _CFSubcluster:
265 """Each subcluster in a CFNode is called a CFSubcluster.
266
267 A CFSubcluster can have a CFNode has its child.
268
269 Parameters
270 ----------
271 linear_sum : ndarray of shape (n_features,), default=None
272 Sample. This is kept optional to allow initialization of empty
273 subclusters.
274
275 Attributes
276 ----------
277 n_samples_ : int
278 Number of samples that belong to each subcluster.
279
280 linear_sum_ : ndarray
281 Linear sum of all the samples in a subcluster. Prevents holding
282 all sample data in memory.
283
284 squared_sum_ : float
285 Sum of the squared l2 norms of all samples belonging to a subcluster.
286
287 centroid_ : ndarray of shape (branching_factor + 1, n_features)
288 Centroid of the subcluster. Prevent recomputing of centroids when
289 ``CFNode.centroids_`` is called.
290
291 child_ : _CFNode
292 Child Node of the subcluster. Once a given _CFNode is set as the child
293 of the _CFNode, it is set to ``self.child_``.
294
295 sq_norm_ : ndarray of shape (branching_factor + 1,)
296 Squared norm of the subcluster. Used to prevent recomputing when
297 pairwise minimum distances are computed.
298 """
299
300 def __init__(self, *, linear_sum=None):
301 if linear_sum is None:
302 self.n_samples_ = 0
303 self.squared_sum_ = 0.0
304 self.centroid_ = self.linear_sum_ = 0
305 else:
306 self.n_samples_ = 1
307 self.centroid_ = self.linear_sum_ = linear_sum
308 self.squared_sum_ = self.sq_norm_ = np.dot(
309 self.linear_sum_, self.linear_sum_
310 )
311 self.child_ = None
312
313 def update(self, subcluster):
314 self.n_samples_ += subcluster.n_samples_
315 self.linear_sum_ += subcluster.linear_sum_
316 self.squared_sum_ += subcluster.squared_sum_
317 self.centroid_ = self.linear_sum_ / self.n_samples_
318 self.sq_norm_ = np.dot(self.centroid_, self.centroid_)
319
320 def merge_subcluster(self, nominee_cluster, threshold):
321 """Check if a cluster is worthy enough to be merged. If
322 yes then merge.
323 """
324 new_ss = self.squared_sum_ + nominee_cluster.squared_sum_
325 new_ls = self.linear_sum_ + nominee_cluster.linear_sum_
326 new_n = self.n_samples_ + nominee_cluster.n_samples_
327 new_centroid = (1 / new_n) * new_ls
328 new_sq_norm = np.dot(new_centroid, new_centroid)
329
330 # The squared radius of the cluster is defined:
331 # r^2 = sum_i ||x_i - c||^2 / n
332 # with x_i the n points assigned to the cluster and c its centroid:
333 # c = sum_i x_i / n
334 # This can be expanded to:
335 # r^2 = sum_i ||x_i||^2 / n - 2 < sum_i x_i / n, c> + n ||c||^2 / n
336 # and therefore simplifies to:
337 # r^2 = sum_i ||x_i||^2 / n - ||c||^2
338 sq_radius = new_ss / new_n - new_sq_norm
339
340 if sq_radius <= threshold**2:
341 (
342 self.n_samples_,
343 self.linear_sum_,
344 self.squared_sum_,
345 self.centroid_,
346 self.sq_norm_,
347 ) = (new_n, new_ls, new_ss, new_centroid, new_sq_norm)
348 return True
349 return False
350
351 @property
352 def radius(self):
353 """Return radius of the subcluster"""
354 # Because of numerical issues, this could become negative
355 sq_radius = self.squared_sum_ / self.n_samples_ - self.sq_norm_
356 return sqrt(max(0, sq_radius))

359class Birch(
360 ClassNamePrefixFeaturesOutMixin, ClusterMixin, TransformerMixin, BaseEstimator
361):
362 """Implements the BIRCH clustering algorithm.
363
364 It is a memory-efficient, online-learning algorithm provided as an
365 alternative to :class:`MiniBatchKMeans`. It constructs a tree
366 data structure with the cluster centroids being read off the leaf.
367 These can be either the final cluster centroids or can be provided as input
368 to another clustering algorithm such as :class:`AgglomerativeClustering`.
369
370 Read more in the :ref:`User Guide <birch>`.
371
372 .. versionadded:: 0.16
373
374 Parameters
375 ----------
376 threshold : float, default=0.5
377 The radius of the subcluster obtained by merging a new sample and the
378 closest subcluster should be lesser than the threshold. Otherwise a new
379 subcluster is started. Setting this value to be very low promotes
380 splitting and vice-versa.
381
382 branching_factor : int, default=50
383 Maximum number of CF subclusters in each node. If a new samples enters
384 such that the number of subclusters exceed the branching_factor then
385 that node is split into two nodes with the subclusters redistributed
386 in each. The parent subcluster of that node is removed and two new
387 subclusters are added as parents of the 2 split nodes.
388
389 n_clusters : int, instance of sklearn.cluster model or None, default=3
390 Number of clusters after the final clustering step, which treats the
391 subclusters from the leaves as new samples.
392
393 - `None` : the final clustering step is not performed and the
394 subclusters are returned as they are.
395
396 - :mod:`sklearn.cluster` Estimator : If a model is provided, the model
397 is fit treating the subclusters as new samples and the initial data
398 is mapped to the label of the closest subcluster.
399
400 - `int` : the model fit is :class:`AgglomerativeClustering` with
401 `n_clusters` set to be equal to the int.
402
403 compute_labels : bool, default=True
404 Whether or not to compute labels for each fit.
405
406 copy : bool, default=True
407 Whether or not to make a copy of the given data. If set to False,
408 the initial data will be overwritten.
409
410 .. deprecated:: 1.6
411 `copy` was deprecated in 1.6 and will be removed in 1.8. It has no effect
412 as the estimator does not perform in-place operations on the input data.
413
414 Attributes
415 ----------
416 root_ : _CFNode
417 Root of the CFTree.
418
419 dummy_leaf_ : _CFNode
420 Start pointer to all the leaves.
421
422 subcluster_centers_ : ndarray
423 Centroids of all subclusters read directly from the leaves.
424
425 subcluster_labels_ : ndarray
426 Labels assigned to the centroids of the subclusters after
427 they are clustered globally.
428
429 labels_ : ndarray of shape (n_samples,)
430 Array of labels assigned to the input data.
431 if partial_fit is used instead of fit, they are assigned to the
432 last batch of data.
433
434 n_features_in_ : int
435 Number of features seen during :term:`fit`.
436
437 .. versionadded:: 0.24
438
439 feature_names_in_ : ndarray of shape (`n_features_in_`,)
440 Names of features seen during :term:`fit`. Defined only when `X`
441 has feature names that are all strings.
442
443 .. versionadded:: 1.0
444
445 See Also
446 --------
447 MiniBatchKMeans : Alternative implementation that does incremental updates
448 of the centers' positions using mini-batches.
449
450 Notes
451 -----
452 The tree data structure consists of nodes with each node consisting of
453 a number of subclusters. The maximum number of subclusters in a node
454 is determined by the branching factor. Each subcluster maintains a
455 linear sum, squared sum and the number of samples in that subcluster.
456 In addition, each subcluster can also have a node as its child, if the
457 subcluster is not a member of a leaf node.
458
459 For a new point entering the root, it is merged with the subcluster closest
460 to it and the linear sum, squared sum and the number of samples of that
461 subcluster are updated. This is done recursively till the properties of
462 the leaf node are updated.
463
464 See :ref:`sphx_glr_auto_examples_cluster_plot_birch_vs_minibatchkmeans.py` for a
465 comparison with :class:`~sklearn.cluster.MiniBatchKMeans`.
466
467 References
468 ----------
469 * Tian Zhang, Raghu Ramakrishnan, Maron Livny
470 BIRCH: An efficient data clustering method for large databases.
471 https://www.cs.sfu.ca/CourseCentral/459/han/papers/zhang96.pdf
472
473 * Roberto Perdisci
474 JBirch - Java implementation of BIRCH clustering algorithm
475 https://code.google.com/archive/p/jbirch
476
477 Examples
478 --------
479 >>> from sklearn.cluster import Birch
480 >>> X = [[0, 1], [0.3, 1], [-0.3, 1], [0, -1], [0.3, -1], [-0.3, -1]]
481 >>> brc = Birch(n_clusters=None)
482 >>> brc.fit(X)
483 Birch(n_clusters=None)
484 >>> brc.predict(X)
485 array([0, 0, 0, 1, 1, 1])
486
487 For a comparison of the BIRCH clustering algorithm with other clustering algorithms,
488 see :ref:`sphx_glr_auto_examples_cluster_plot_cluster_comparison.py`
489 """
490
491 _parameter_constraints: dict = {
492 "threshold": [Interval(Real, 0.0, None, closed="neither")],
493 "branching_factor": [Interval(Integral, 1, None, closed="neither")],
494 "n_clusters": [None, ClusterMixin, Interval(Integral, 1, None, closed="left")],
495 "compute_labels": ["boolean"],
496 "copy": ["boolean", Hidden(StrOptions({"deprecated"}))],
497 }
498

503 branching_factor=50,

514 @_fit_context(prefer_skip_nested_validation=True)
515 def fit(self, X, y=None):
516 """
517 Build a CF Tree for the input data.
518
519 Parameters
520 ----------
521 X : {array-like, sparse matrix} of shape (n_samples, n_features)
522 Input data.
523
524 y : Ignored
525 Not used, present here for API consistency by convention.
526
527 Returns
528 -------
529 self
530 Fitted estimator.
531 """
532 return self._fit(X, partial=False)

534 def _fit(self, X, partial):
535 has_root = getattr(self, "root_", None)
536 first_call = not (partial and has_root)
537
538 if self.copy != "deprecated" and first_call:
539 warnings.warn(
540 "`copy` was deprecated in 1.6 and will be removed in 1.8 since it "
541 "has no effect internally. Simply leave this parameter to its default "
542 "value to avoid this warning.",
543 FutureWarning,
544 )
545
546 X = validate_data(
547 self,
548 X,
549 accept_sparse="csr",
550 reset=first_call,
551 dtype=[np.float64, np.float32],
552 )
553 threshold = self.threshold
554 branching_factor = self.branching_factor
555
556 n_samples, n_features = X.shape
557
558 # If partial_fit is called for the first time or fit is called, we
559 # start a new tree.
560 if first_call:
561 # The first root is the leaf. Manipulate this object throughout.
562 self.root_ = _CFNode(
563 threshold=threshold,
564 branching_factor=branching_factor,
565 is_leaf=True,
566 n_features=n_features,
567 dtype=X.dtype,
568 )
569
570 # To enable getting back subclusters.
571 self.dummy_leaf_ = _CFNode(
572 threshold=threshold,
573 branching_factor=branching_factor,
574 is_leaf=True,
575 n_features=n_features,
576 dtype=X.dtype,
577 )
578 self.dummy_leaf_.next_leaf_ = self.root_
579 self.root_.prev_leaf_ = self.dummy_leaf_
580
581 # Cannot vectorize. Enough to convince to use cython.
582 if not sparse.issparse(X):
583 iter_func = iter
584 else:
585 iter_func = _iterate_sparse_X
586
587 for sample in iter_func(X):
588 subcluster = _CFSubcluster(linear_sum=sample)
589 split = self.root_.insert_cf_subcluster(subcluster)
590
591 if split:
592 new_subcluster1, new_subcluster2 = _split_node(
593 self.root_, threshold, branching_factor
594 )
595 del self.root_
596 self.root_ = _CFNode(
597 threshold=threshold,
598 branching_factor=branching_factor,
599 is_leaf=False,
600 n_features=n_features,
601 dtype=X.dtype,
602 )
603 self.root_.append_subcluster(new_subcluster1)
604 self.root_.append_subcluster(new_subcluster2)
605
606 centroids = np.concatenate([leaf.centroids_ for leaf in self._get_leaves()])
607 self.subcluster_centers_ = centroids
608 self._n_features_out = self.subcluster_centers_.shape[0]
609
610 self._global_clustering(X)
611 return self

613 def _get_leaves(self):
614 """
615 Retrieve the leaves of the CF Node.
616
617 Returns
618 -------
619 leaves : list of shape (n_leaves,)
620 List of the leaf nodes.
621 """
622 leaf_ptr = self.dummy_leaf_.next_leaf_
623 leaves = []
624 while leaf_ptr is not None:
625 leaves.append(leaf_ptr)
626 leaf_ptr = leaf_ptr.next_leaf_
627 return leaves

629 @_fit_context(prefer_skip_nested_validation=True)
630 def partial_fit(self, X=None, y=None):
631 """
632 Online learning. Prevents rebuilding of CFTree from scratch.
633
634 Parameters
635 ----------
636 X : {array-like, sparse matrix} of shape (n_samples, n_features), \
637 default=None
638 Input data. If X is not provided, only the global clustering
639 step is done.
640
641 y : Ignored
642 Not used, present here for API consistency by convention.
643
644 Returns
645 -------
646 self
647 Fitted estimator.
648 """
649 if X is None:
650 # Perform just the final global clustering step.
651 self._global_clustering()
652 return self
653 else:
654 return self._fit(X, partial=True)

676 def _predict(self, X):
677 """Predict data using the ``centroids_`` of subclusters."""
678 kwargs = {"Y_norm_squared": self._subcluster_norms}
679
680 with config_context(assume_finite=True):
681 argmin = pairwise_distances_argmin(
682 X, self.subcluster_centers_, metric_kwargs=kwargs
683 )
684 return self.subcluster_labels_[argmin]

686 def transform(self, X):
687 """
688 Transform X into subcluster centroids dimension.
689
690 Each dimension represents the distance from the sample point to each
691 cluster centroid.
692
693 Parameters
694 ----------
695 X : {array-like, sparse matrix} of shape (n_samples, n_features)
696 Input data.
697
698 Returns
699 -------
700 X_trans : {array-like, sparse matrix} of shape (n_samples, n_clusters)
701 Transformed data.
702 """
703 check_is_fitted(self)
704 X = validate_data(self, X, accept_sparse="csr", reset=False)
705 with config_context(assume_finite=True):
706 return euclidean_distances(X, self.subcluster_centers_)

708 def _global_clustering(self, X=None):
709 """
710 Global clustering for the subclusters obtained after fitting
711 """
712 clusterer = self.n_clusters
713 centroids = self.subcluster_centers_
714 compute_labels = (X is not None) and self.compute_labels
715
716 # Preprocessing for the global clustering.
717 not_enough_centroids = False
718 if isinstance(clusterer, Integral):
719 clusterer = AgglomerativeClustering(n_clusters=self.n_clusters)
720 # There is no need to perform the global clustering step.
721 if len(centroids) < self.n_clusters:
722 not_enough_centroids = True
