40def _grid_from_X(X, percentiles, is_categorical, grid_resolution, custom_values):
41 """Generate a grid of points based on the percentiles of X.
42
43 The grid is a cartesian product between the columns of ``values``. The
44 ith column of ``values`` consists in ``grid_resolution`` equally-spaced
45 points between the percentiles of the jth column of X.
46
47 If ``grid_resolution`` is bigger than the number of unique values in the
48 j-th column of X or if the feature is a categorical feature (by inspecting
49 `is_categorical`) , then those unique values will be used instead.
50
51 Parameters
52 ----------
53 X : array-like of shape (n_samples, n_target_features)
54 The data.
55
56 percentiles : tuple of float
57 The percentiles which are used to construct the extreme values of
58 the grid. Must be in [0, 1].
59
60 is_categorical : list of bool
61 For each feature, tells whether it is categorical or not. If a feature
62 is categorical, then the values used will be the unique ones
63 (i.e. categories) instead of the percentiles.
64
65 grid_resolution : int
66 The number of equally spaced points to be placed on the grid for each
67 feature.
68
69 custom_values: dict
70 Mapping from column index of X to an array-like of values where
71 the partial dependence should be calculated for that feature
72
73 Returns
74 -------
75 grid : ndarray of shape (n_points, n_target_features)
76 A value for each feature at each point in the grid. ``n_points`` is
77 always ``<= grid_resolution ** X.shape[1]``.
78
79 values : list of 1d ndarrays
80 The values with which the grid has been created. The size of each
81 array ``values[j]`` is either ``grid_resolution``, the number of
82 unique values in ``X[:, j]``, if j is not in ``custom_range``.
83 If j is in ``custom_range``, then it is the length of ``custom_range[j]``.
84 """
85 if not isinstance(percentiles, Iterable) or len(percentiles) != 2:
86 raise ValueError("'percentiles' must be a sequence of 2 elements.")
87 if not all(0 <= x <= 1 for x in percentiles):
88 raise ValueError("'percentiles' values must be in [0, 1].")
89 if percentiles[0] >= percentiles[1]:
90 raise ValueError("percentiles[0] must be strictly less than percentiles[1].")
91
92 if grid_resolution <= 1:
93 raise ValueError("'grid_resolution' must be strictly greater than 1.")
94
95 def _convert_custom_values(values):
96 # Convert custom types such that object types are always used for string arrays
97 dtype = object if any(isinstance(v, str) for v in values) else None
98 return np.asarray(values, dtype=dtype)
99
100 custom_values = {k: _convert_custom_values(v) for k, v in custom_values.items()}
101 if any(v.ndim != 1 for v in custom_values.values()):
102 error_string = ", ".join(
103 f"Feature {k}: {v.ndim} dimensions"
104 for k, v in custom_values.items()
105 if v.ndim != 1
106 )
107
108 raise ValueError(
109 "The custom grid for some features is not a one-dimensional array. "
110 f"{error_string}"
111 )
112
113 values = []
114 # TODO: we should handle missing values (i.e. `np.nan`) specifically and store them
115 # in a different Bunch attribute.
116 for feature, is_cat in enumerate(is_categorical):
117 if feature in custom_values:
118 # Use values in the custom range
119 axis = custom_values[feature]
120 else:
121 try:
122 uniques = np.unique(_safe_indexing(X, feature, axis=1))
123 except TypeError as exc:
124 # `np.unique` will fail in the presence of `np.nan` and `str` categories
125 # due to sorting. Temporary, we reraise an error explaining the problem.
126 raise ValueError(
127 f"The column #{feature} contains mixed data types. Finding unique "
128 "categories fail due to sorting. It usually means that the column "
129 "contains `np.nan` values together with `str` categories. Such use "
130 "case is not yet supported in scikit-learn."
131 ) from exc
132
133 if is_cat or uniques.shape[0] < grid_resolution:
134 # Use the unique values either because:
135 # - feature has low resolution use unique values
136 # - feature is categorical
137 axis = uniques
138 else:
139 # create axis based on percentiles and grid resolution
140 emp_percentiles = mquantiles(
141 _safe_indexing(X, feature, axis=1), prob=percentiles, axis=0
142 )
143 if np.allclose(emp_percentiles[0], emp_percentiles[1]):
144 raise ValueError(
145 "percentiles are too close to each other, "
146 "unable to build the grid. Please choose percentiles "
147 "that are further apart."
148 )
149 axis = np.linspace(
150 emp_percentiles[0],
151 emp_percentiles[1],
152 num=grid_resolution,
153 endpoint=True,
154 )
155 values.append(axis)
156
157 return cartesian(values), values

160def _partial_dependence_recursion(est, grid, features):
161 """Calculate partial dependence via the recursion method.
162
163 The recursion method is in particular enabled for tree-based estimators.
164
165 For each `grid` value, a weighted tree traversal is performed: if a split node
166 involves an input feature of interest, the corresponding left or right branch
167 is followed; otherwise both branches are followed, each branch being weighted
168 by the fraction of training samples that entered that branch. Finally, the
169 partial dependence is given by a weighted average of all the visited leaves
170 values.
171
172 This method is more efficient in terms of speed than the `'brute'` method
173 (:func:`~sklearn.inspection._partial_dependence._partial_dependence_brute`).
174 However, here, the partial dependence computation is done explicitly with the
175 `X` used during training of `est`.
176
177 Parameters
178 ----------
179 est : BaseEstimator
180 A fitted estimator object implementing :term:`predict` or
181 :term:`decision_function`. Multioutput-multiclass classifiers are not
182 supported. Note that `'recursion'` is only supported for some tree-based
183 estimators (namely
184 :class:`~sklearn.ensemble.GradientBoostingClassifier`,
185 :class:`~sklearn.ensemble.GradientBoostingRegressor`,
186 :class:`~sklearn.ensemble.HistGradientBoostingClassifier`,
187 :class:`~sklearn.ensemble.HistGradientBoostingRegressor`,
188 :class:`~sklearn.tree.DecisionTreeRegressor`,
189 :class:`~sklearn.ensemble.RandomForestRegressor`,
190 ).
191
192 grid : array-like of shape (n_points, n_target_features)
193 The grid of feature values for which the partial dependence is calculated.
194 Note that `n_points` is the number of points in the grid and `n_target_features`
195 is the number of features you are doing partial dependence at.
196
197 features : array-like of {int, str}
198 The feature (e.g. `[0]`) or pair of interacting features
199 (e.g. `[(0, 1)]`) for which the partial dependency should be computed.
200
201 Returns
202 -------
203 averaged_predictions : array-like of shape (n_targets, n_points)
204 The averaged predictions for the given `grid` of features values.
205 Note that `n_targets` is the number of targets (e.g. 1 for binary
206 classification, `n_tasks` for multi-output regression, and `n_classes` for
207 multiclass classification) and `n_points` is the number of points in the `grid`.
208 """
209 averaged_predictions = est._compute_partial_dependence_recursion(grid, features)
210 if averaged_predictions.ndim == 1:
211 # reshape to (1, n_points) for consistency with
212 # _partial_dependence_brute
213 averaged_predictions = averaged_predictions.reshape(1, -1)
214
215 return averaged_predictions

218def _partial_dependence_brute(
219 est, grid, features, X, response_method, sample_weight=None
220):
221 """Calculate partial dependence via the brute force method.
222
223 The brute method explicitly averages the predictions of an estimator over a
224 grid of feature values.
225
226 For each `grid` value, all the samples from `X` have their variables of
227 interest replaced by that specific `grid` value. The predictions are then made
228 and averaged across the samples.
229
230 This method is slower than the `'recursion'`
231 (:func:`~sklearn.inspection._partial_dependence._partial_dependence_recursion`)
232 version for estimators with this second option. However, with the `'brute'`
233 force method, the average will be done with the given `X` and not the `X`
234 used during training, as it is done in the `'recursion'` version. Therefore
235 the average can always accept `sample_weight` (even when the estimator was
236 fitted without).
237
238 Parameters
239 ----------
240 est : BaseEstimator
241 A fitted estimator object implementing :term:`predict`,
242 :term:`predict_proba`, or :term:`decision_function`.
243 Multioutput-multiclass classifiers are not supported.
244
245 grid : array-like of shape (n_points, n_target_features)
246 The grid of feature values for which the partial dependence is calculated.
247 Note that `n_points` is the number of points in the grid and `n_target_features`
248 is the number of features you are doing partial dependence at.
249
250 features : array-like of {int, str}
251 The feature (e.g. `[0]`) or pair of interacting features
252 (e.g. `[(0, 1)]`) for which the partial dependency should be computed.
253
254 X : array-like of shape (n_samples, n_features)
255 `X` is used to generate values for the complement features. That is, for
256 each value in `grid`, the method will average the prediction of each
257 sample from `X` having that grid value for `features`.
258
259 response_method : {'auto', 'predict_proba', 'decision_function'}, \
260 default='auto'
261 Specifies whether to use :term:`predict_proba` or
262 :term:`decision_function` as the target response. For regressors
263 this parameter is ignored and the response is always the output of
264 :term:`predict`. By default, :term:`predict_proba` is tried first
265 and we revert to :term:`decision_function` if it doesn't exist.
266
267 sample_weight : array-like of shape (n_samples,), default=None
268 Sample weights are used to calculate weighted means when averaging the
269 model output. If `None`, then samples are equally weighted. Note that
270 `sample_weight` does not change the individual predictions.
271
272 Returns
273 -------
274 averaged_predictions : array-like of shape (n_targets, n_points)
275 The averaged predictions for the given `grid` of features values.
276 Note that `n_targets` is the number of targets (e.g. 1 for binary
277 classification, `n_tasks` for multi-output regression, and `n_classes` for
278 multiclass classification) and `n_points` is the number of points in the `grid`.
279
280 predictions : array-like
281 The predictions for the given `grid` of features values over the samples
282 from `X`. For non-multioutput regression and binary classification the
283 shape is `(n_instances, n_points)` and for multi-output regression and
284 multiclass classification the shape is `(n_targets, n_instances, n_points)`,
285 where `n_targets` is the number of targets (`n_tasks` for multi-output
286 regression, and `n_classes` for multiclass classification), `n_instances`
287 is the number of instances in `X`, and `n_points` is the number of points
288 in the `grid`.
289 """
290 predictions = []
291 averaged_predictions = []
292
293 if response_method == "auto":
294 response_method = (
295 "predict" if is_regressor(est) else ["predict_proba", "decision_function"]
296 )
297
298 X_eval = X.copy()
299 for new_values in grid:
300 for i, variable in enumerate(features):
301 _safe_assign(X_eval, new_values[i], column_indexer=variable)
302
303 # Note: predictions is of shape
304 # (n_points,) for non-multioutput regressors
305 # (n_points, n_tasks) for multioutput regressors
306 # (n_points, 1) for the regressors in cross_decomposition (I think)
307 # (n_points, 1) for binary classification (positive class already selected)
308 # (n_points, n_classes) for multiclass classification
309 pred, _ = _get_response_values(est, X_eval, response_method=response_method)
310
311 predictions.append(pred)
312 # average over samples
313 averaged_predictions.append(np.average(pred, axis=0, weights=sample_weight))
314
315 n_samples = X.shape[0]
316
317 # reshape to (n_targets, n_instances, n_points) where n_targets is:
318 # - 1 for non-multioutput regression and binary classification (shape is
319 # already correct in those cases)
320 # - n_tasks for multi-output regression
321 # - n_classes for multiclass classification.
322 predictions = np.array(predictions).T
323 if is_regressor(est) and predictions.ndim == 2:
324 # non-multioutput regression, shape is (n_instances, n_points,)
325 predictions = predictions.reshape(n_samples, -1)
326 elif is_classifier(est) and predictions.shape[0] == 2:
327 # Binary classification, shape is (2, n_instances, n_points).
328 # we output the effect of **positive** class
329 predictions = predictions[1]
330 predictions = predictions.reshape(n_samples, -1)
331
332 # reshape averaged_predictions to (n_targets, n_points) where n_targets is:
333 # - 1 for non-multioutput regression and binary classification (shape is
334 # already correct in those cases)
335 # - n_tasks for multi-output regression
336 # - n_classes for multiclass classification.
337 averaged_predictions = np.array(averaged_predictions).T
338 if averaged_predictions.ndim == 1:
339 # reshape to (1, n_points) for consistency with
340 # _partial_dependence_recursion
341 averaged_predictions = averaged_predictions.reshape(1, -1)
342
343 return averaged_predictions, predictions

367def partial_dependence(
368 estimator,
369 X,
370 features,
371 *,
372 sample_weight=None,
373 categorical_features=None,
374 feature_names=None,
375 response_method="auto",
376 percentiles=(0.05, 0.95),
377 grid_resolution=100,
378 custom_values=None,
379 method="auto",
380 kind="average",
381):
382 """Partial dependence of ``features``.
383
384 Partial dependence of a feature (or a set of features) corresponds to
385 the average response of an estimator for each possible value of the
386 feature.
387
388 Read more in
389 :ref:`sphx_glr_auto_examples_inspection_plot_partial_dependence.py`
390 and the :ref:`User Guide <partial_dependence>`.
391
392 .. warning::
393
394 For :class:`~sklearn.ensemble.GradientBoostingClassifier` and
395 :class:`~sklearn.ensemble.GradientBoostingRegressor`, the
396 `'recursion'` method (used by default) will not account for the `init`
397 predictor of the boosting process. In practice, this will produce
398 the same values as `'brute'` up to a constant offset in the target
399 response, provided that `init` is a constant estimator (which is the
400 default). However, if `init` is not a constant estimator, the
401 partial dependence values are incorrect for `'recursion'` because the
402 offset will be sample-dependent. It is preferable to use the `'brute'`
403 method. Note that this only applies to
404 :class:`~sklearn.ensemble.GradientBoostingClassifier` and
405 :class:`~sklearn.ensemble.GradientBoostingRegressor`, not to
406 :class:`~sklearn.ensemble.HistGradientBoostingClassifier` and
407 :class:`~sklearn.ensemble.HistGradientBoostingRegressor`.
408
409 Parameters
410 ----------
411 estimator : BaseEstimator
412 A fitted estimator object implementing :term:`predict`,
413 :term:`predict_proba`, or :term:`decision_function`.
414 Multioutput-multiclass classifiers are not supported.
415
416 X : {array-like, sparse matrix or dataframe} of shape (n_samples, n_features)
417 ``X`` is used to generate a grid of values for the target
418 ``features`` (where the partial dependence will be evaluated), and
419 also to generate values for the complement features when the
420 `method` is 'brute'.
421
422 features : array-like of {int, str, bool} or int or str
423 The feature (e.g. `[0]`) or pair of interacting features
424 (e.g. `[(0, 1)]`) for which the partial dependency should be computed.
425
426 sample_weight : array-like of shape (n_samples,), default=None
427 Sample weights are used to calculate weighted means when averaging the
428 model output. If `None`, then samples are equally weighted. If
429 `sample_weight` is not `None`, then `method` will be set to `'brute'`.
430 Note that `sample_weight` is ignored for `kind='individual'`.
431
432 .. versionadded:: 1.3
433
434 categorical_features : array-like of shape (n_features,) or shape \
435 (n_categorical_features,), dtype={bool, int, str}, default=None
436 Indicates the categorical features.
437
438 - `None`: no feature will be considered categorical;
439 - boolean array-like: boolean mask of shape `(n_features,)`
440 indicating which features are categorical. Thus, this array has
441 the same shape has `X.shape[1]`;
442 - integer or string array-like: integer indices or strings
443 indicating categorical features.
444
445 .. versionadded:: 1.2
446
447 feature_names : array-like of shape (n_features,), dtype=str, default=None
448 Name of each feature; `feature_names[i]` holds the name of the feature
449 with index `i`.
450 By default, the name of the feature corresponds to their numerical
451 index for NumPy array and their column name for pandas dataframe.
452
453 .. versionadded:: 1.2
454
455 response_method : {'auto', 'predict_proba', 'decision_function'}, \
456 default='auto'
457 Specifies whether to use :term:`predict_proba` or
458 :term:`decision_function` as the target response. For regressors
459 this parameter is ignored and the response is always the output of
460 :term:`predict`. By default, :term:`predict_proba` is tried first
461 and we revert to :term:`decision_function` if it doesn't exist. If
462 ``method`` is 'recursion', the response is always the output of
463 :term:`decision_function`.
464
465 percentiles : tuple of float, default=(0.05, 0.95)
466 The lower and upper percentile used to create the extreme values
467 for the grid. Must be in [0, 1].
468 This parameter is overridden by `custom_values` if that parameter is set.
469
470 grid_resolution : int, default=100
471 The number of equally spaced points on the grid, for each target
472 feature.
473 This parameter is overridden by `custom_values` if that parameter is set.
474
475 custom_values : dict
476 A dictionary mapping the index of an element of `features` to an array
477 of values where the partial dependence should be calculated
478 for that feature. Setting a range of values for a feature overrides
479 `grid_resolution` and `percentiles`.
480
481 See :ref:`how to use partial_dependence
482 <plt_partial_dependence_custom_values>` for an example of how this parameter can
483 be used.
484
485 .. versionadded:: 1.7
486
487 method : {'auto', 'recursion', 'brute'}, default='auto'
488 The method used to calculate the averaged predictions:
489
490 - `'recursion'` is only supported for some tree-based estimators
491 (namely
492 :class:`~sklearn.ensemble.GradientBoostingClassifier`,
493 :class:`~sklearn.ensemble.GradientBoostingRegressor`,
494 :class:`~sklearn.ensemble.HistGradientBoostingClassifier`,
495 :class:`~sklearn.ensemble.HistGradientBoostingRegressor`,
496 :class:`~sklearn.tree.DecisionTreeRegressor`,
497 :class:`~sklearn.ensemble.RandomForestRegressor`,
498 ) when `kind='average'`.
499 This is more efficient in terms of speed.
500 With this method, the target response of a
501 classifier is always the decision function, not the predicted
502 probabilities. Since the `'recursion'` method implicitly computes
503 the average of the Individual Conditional Expectation (ICE) by
504 design, it is not compatible with ICE and thus `kind` must be
505 `'average'`.
506
507 - `'brute'` is supported for any estimator, but is more
508 computationally intensive.
509
510 - `'auto'`: the `'recursion'` is used for estimators that support it,
511 and `'brute'` is used otherwise. If `sample_weight` is not `None`,
512 then `'brute'` is used regardless of the estimator.
513
514 Please see :ref:`this note <pdp_method_differences>` for
515 differences between the `'brute'` and `'recursion'` method.
516
517 kind : {'average', 'individual', 'both'}, default='average'
518 Whether to return the partial dependence averaged across all the
519 samples in the dataset or one value per sample or both.
520 See Returns below.
521
522 Note that the fast `method='recursion'` option is only available for
523 `kind='average'` and `sample_weights=None`. Computing individual
524 dependencies and doing weighted averages requires using the slower
525 `method='brute'`.
526
527 .. versionadded:: 0.24
528
529 Returns
530 -------
531 predictions : :class:`~sklearn.utils.Bunch`
532 Dictionary-like object, with the following attributes.
533
534 individual : ndarray of shape (n_outputs, n_instances, \
535 len(values[0]), len(values[1]), ...)
536 The predictions for all the points in the grid for all
537 samples in X. This is also known as Individual
538 Conditional Expectation (ICE).
539 Only available when `kind='individual'` or `kind='both'`.
540
541 average : ndarray of shape (n_outputs, len(values[0]), \
542 len(values[1]), ...)
543 The predictions for all the points in the grid, averaged
544 over all samples in X (or over the training data if
545 `method` is 'recursion').
546 Only available when `kind='average'` or `kind='both'`.
547
548 grid_values : seq of 1d ndarrays
549 The values with which the grid has been created. The generated
550 grid is a cartesian product of the arrays in `grid_values` where
551 `len(grid_values) == len(features)`. The size of each array
552 `grid_values[j]` is either `grid_resolution`, or the number of
553 unique values in `X[:, j]`, whichever is smaller.
554
555 .. versionadded:: 1.3
556
557 `n_outputs` corresponds to the number of classes in a multi-class
558 setting, or to the number of tasks for multi-output regression.
559 For classical regression and binary classification `n_outputs==1`.
560 `n_values_feature_j` corresponds to the size `grid_values[j]`.
561
562 See Also
563 --------
564 PartialDependenceDisplay.from_estimator : Plot Partial Dependence.
565 PartialDependenceDisplay : Partial Dependence visualization.
566
567 Examples
568 --------
569 >>> X = [[0, 0, 2], [1, 0, 0]]
570 >>> y = [0, 1]
571 >>> from sklearn.ensemble import GradientBoostingClassifier
572 >>> gb = GradientBoostingClassifier(random_state=0).fit(X, y)
573 >>> partial_dependence(gb, features=[0], X=X, percentiles=(0, 1),
574 ... grid_resolution=2) # doctest: +SKIP
575 (array([[-4.52, 4.52]]), [array([ 0., 1.])])
576 """
577 check_is_fitted(estimator)
578
579 if not (is_classifier(estimator) or is_regressor(estimator)):
580 raise ValueError("'estimator' must be a fitted regressor or classifier.")
581
582 if is_classifier(estimator) and isinstance(estimator.classes_[0], np.ndarray):
583 raise ValueError("Multiclass-multioutput estimators are not supported")
584
585 # Use check_array only on lists and other non-array-likes / sparse. Do not
586 # convert DataFrame into a NumPy array.
587 if not (hasattr(X, "__array__") or sparse.issparse(X)):
588 X = check_array(X, ensure_all_finite="allow-nan", dtype=object)
589
590 if is_regressor(estimator) and response_method != "auto":
591 raise ValueError(
592 "The response_method parameter is ignored for regressors and "
593 "must be 'auto'."
594 )
595
596 if kind != "average":
597 if method == "recursion":
598 raise ValueError(
599 "The 'recursion' method only applies when 'kind' is set to 'average'"
600 )
601 method = "brute"
602
603 if method == "recursion" and sample_weight is not None:
604 raise ValueError(
605 "The 'recursion' method can only be applied when sample_weight is None."
606 )
607
608 if method == "auto":
609 if sample_weight is not None:
610 method = "brute"
611 elif isinstance(estimator, BaseGradientBoosting) and estimator.init is None:
612 method = "recursion"
613 elif isinstance(
614 estimator,
615 (BaseHistGradientBoosting, DecisionTreeRegressor, RandomForestRegressor),
616 ):
617 method = "recursion"
618 else:
619 method = "brute"
620
621 if method == "recursion":
622 if not isinstance(
623 estimator,
624 (
625 BaseGradientBoosting,
626 BaseHistGradientBoosting,
627 DecisionTreeRegressor,
628 RandomForestRegressor,
629 ),
630 ):
631 supported_classes_recursion = (
632 "GradientBoostingClassifier",
633 "GradientBoostingRegressor",
634 "HistGradientBoostingClassifier",
635 "HistGradientBoostingRegressor",
636 "HistGradientBoostingRegressor",
637 "DecisionTreeRegressor",
638 "RandomForestRegressor",
