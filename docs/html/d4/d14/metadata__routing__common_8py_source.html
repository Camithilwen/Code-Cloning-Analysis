30def record_metadata(obj, record_default=True, **kwargs):
31 """Utility function to store passed metadata to a method of obj.
32
33 If record_default is False, kwargs whose values are "default" are skipped.
34 This is so that checks on keyword arguments whose default was not changed
35 are skipped.
36
37 """
38 stack = inspect.stack()
39 callee = stack[1].function
40 caller = stack[2].function
41 if not hasattr(obj, "_records"):
42 obj._records = defaultdict(lambda: defaultdict(list))
43 if not record_default:
44 kwargs = {
45 key: val
46 for key, val in kwargs.items()
47 if not isinstance(val, str) or (val != "default")
48 }
49 obj._records[callee][caller].append(kwargs)

52def check_recorded_metadata(obj, method, parent, split_params=tuple(), **kwargs):
53 """Check whether the expected metadata is passed to the object's method.
54
55 Parameters
56 ----------
57 obj : estimator object
58 sub-estimator to check routed params for
59 method : str
60 sub-estimator's method where metadata is routed to, or otherwise in
61 the context of metadata routing referred to as 'callee'
62 parent : str
63 the parent method which should have called `method`, or otherwise in
64 the context of metadata routing referred to as 'caller'
65 split_params : tuple, default=empty
66 specifies any parameters which are to be checked as being a subset
67 of the original values
68 **kwargs : dict
69 passed metadata
70 """
71 all_records = (
72 getattr(obj, "_records", dict()).get(method, dict()).get(parent, list())
73 )
74 for record in all_records:
75 # first check that the names of the metadata passed are the same as
76 # expected. The names are stored as keys in `record`.
77 assert set(kwargs.keys()) == set(record.keys()), (
78 f"Expected {kwargs.keys()} vs {record.keys()}"
79 )
80 for key, value in kwargs.items():
81 recorded_value = record[key]
82 # The following condition is used to check for any specified parameters
83 # being a subset of the original values
84 if key in split_params and recorded_value is not None:
85 assert np.isin(recorded_value, value).all()
86 else:
87 if isinstance(recorded_value, np.ndarray):
88 assert_array_equal(recorded_value, value)
89 else:
90 assert recorded_value is value, (
91 f"Expected {recorded_value} vs {value}. Method: {method}"
92 )

98def assert_request_is_empty(metadata_request, exclude=None):
99 """Check if a metadata request dict is empty.
100
101 One can exclude a method or a list of methods from the check using the
102 ``exclude`` parameter. If metadata_request is a MetadataRouter, then
103 ``exclude`` can be of the form ``{"object" : [method, ...]}``.
104 """
105 if isinstance(metadata_request, MetadataRouter):
106 for name, route_mapping in metadata_request:
107 if exclude is not None and name in exclude:
108 _exclude = exclude[name]
109 else:
110 _exclude = None
111 assert_request_is_empty(route_mapping.router, exclude=_exclude)
112 return
113
114 exclude = [] if exclude is None else exclude
115 for method in SIMPLE_METHODS:
116 if method in exclude:
117 continue
118 mmr = getattr(metadata_request, method)
119 props = [
120 prop
121 for prop, alias in mmr.requests.items()
122 if isinstance(alias, str) or alias is not None
123 ]
124 assert not props

127def assert_request_equal(request, dictionary):
128 for method, requests in dictionary.items():
129 mmr = getattr(request, method)
130 assert mmr.requests == requests
131
132 empty_methods = [method for method in SIMPLE_METHODS if method not in dictionary]
133 for method in empty_methods:
134 assert not len(getattr(request, method).requests)
