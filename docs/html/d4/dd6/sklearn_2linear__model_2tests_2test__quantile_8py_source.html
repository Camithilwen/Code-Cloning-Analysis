96def test_quantile_sample_weight():
97 # test that with unequal sample weights we still estimate weighted fraction
98 n = 1000
99 X, y = make_regression(n_samples=n, n_features=5, random_state=0, noise=10.0)
100 weight = np.ones(n)
101 # when we increase weight of upper observations,
102 # estimate of quantile should go up
103 weight[y > y.mean()] = 100
104 quant = QuantileRegressor(quantile=0.5, alpha=1e-8)
105 quant.fit(X, y, sample_weight=weight)
106 fraction_below = np.mean(y < quant.predict(X))
107 assert fraction_below > 0.5
108 weighted_fraction_below = np.average(y < quant.predict(X), weights=weight)
109 assert weighted_fraction_below == approx(0.5, abs=3e-2)

112@pytest.mark.parametrize("quantile", [0.2, 0.5, 0.8])
113def test_asymmetric_error(quantile):
114 """Test quantile regression for asymmetric distributed targets."""
115 n_samples = 1000
116 rng = np.random.RandomState(42)
117 X = np.concatenate(
118 (
119 np.abs(rng.randn(n_samples)[:, None]),
120 -rng.randint(2, size=(n_samples, 1)),
121 ),
122 axis=1,
123 )
124 intercept = 1.23
125 coef = np.array([0.5, -2])
126 # Take care that X @ coef + intercept > 0
127 assert np.min(X @ coef + intercept) > 0
128 # For an exponential distribution with rate lambda, e.g. exp(-lambda * x),
129 # the quantile at level q is:
130 # quantile(q) = - log(1 - q) / lambda
131 # scale = 1/lambda = -quantile(q) / log(1 - q)
132 y = rng.exponential(
133 scale=-(X @ coef + intercept) / np.log(1 - quantile), size=n_samples
134 )
135 model = QuantileRegressor(
136 quantile=quantile,
137 alpha=0,
138 ).fit(X, y)
139 # This test can be made to pass with any solver but in the interest
140 # of sparing continuous integration resources, the test is performed
141 # with the fastest solver only.
142
143 assert model.intercept_ == approx(intercept, rel=0.2)
144 assert_allclose(model.coef_, coef, rtol=0.6)
145 assert_allclose(np.mean(model.predict(X) > y), quantile, atol=1e-2)
146
147 # Now compare to Nelder-Mead optimization with L1 penalty
148 alpha = 0.01
149 model.set_params(alpha=alpha).fit(X, y)
150 model_coef = np.r_[model.intercept_, model.coef_]
151
152 def func(coef):
153 loss = mean_pinball_loss(y, X @ coef[1:] + coef[0], alpha=quantile)
154 L1 = np.sum(np.abs(coef[1:]))
155 return loss + alpha * L1
156
157 res = minimize(
158 fun=func,
159 x0=[1, 0, -1],
160 method="Nelder-Mead",
161 tol=1e-12,
162 options={"maxiter": 2000},
163 )
164
165 assert func(model_coef) == approx(func(res.x))
166 assert_allclose(model.intercept_, res.x[0])
167 assert_allclose(model.coef_, res.x[1:])
168 assert_allclose(np.mean(model.predict(X) > y), quantile, atol=1e-2)

171@pytest.mark.parametrize("quantile", [0.2, 0.5, 0.8])
172def test_equivariance(quantile):
173 """Test equivariace of quantile regression.
174
175 See Koenker (2005) Quantile Regression, Chapter 2.2.3.
176 """
177 rng = np.random.RandomState(42)
178 n_samples, n_features = 100, 5
179 X, y = make_regression(
180 n_samples=n_samples,
181 n_features=n_features,
182 n_informative=n_features,
183 noise=0,
184 random_state=rng,
185 shuffle=False,
186 )
187 # make y asymmetric
188 y += rng.exponential(scale=100, size=y.shape)
189 params = dict(alpha=0)
190 model1 = QuantileRegressor(quantile=quantile, **params).fit(X, y)
191
192 # coef(q; a*y, X) = a * coef(q; y, X)
193 a = 2.5
194 model2 = QuantileRegressor(quantile=quantile, **params).fit(X, a * y)
195 assert model2.intercept_ == approx(a * model1.intercept_, rel=1e-5)
196 assert_allclose(model2.coef_, a * model1.coef_, rtol=1e-5)
197
198 # coef(1-q; -a*y, X) = -a * coef(q; y, X)
199 model2 = QuantileRegressor(quantile=1 - quantile, **params).fit(X, -a * y)
200 assert model2.intercept_ == approx(-a * model1.intercept_, rel=1e-5)
201 assert_allclose(model2.coef_, -a * model1.coef_, rtol=1e-5)
202
203 # coef(q; y + X @ g, X) = coef(q; y, X) + g
204 g_intercept, g_coef = rng.randn(), rng.randn(n_features)
205 model2 = QuantileRegressor(quantile=quantile, **params)
206 model2.fit(X, y + X @ g_coef + g_intercept)
207 assert model2.intercept_ == approx(model1.intercept_ + g_intercept)
208 assert_allclose(model2.coef_, model1.coef_ + g_coef, rtol=1e-6)
209
210 # coef(q; y, X @ A) = A^-1 @ coef(q; y, X)
211 A = rng.randn(n_features, n_features)
212 model2 = QuantileRegressor(quantile=quantile, **params)
213 model2.fit(X @ A, y)
214 assert model2.intercept_ == approx(model1.intercept_, rel=1e-5)
215 assert_allclose(model2.coef_, np.linalg.solve(A, model1.coef_), rtol=1e-5)

236@pytest.mark.parametrize(
237 "sparse_container", CSC_CONTAINERS + CSR_CONTAINERS + COO_CONTAINERS

239@pytest.mark.parametrize("solver", ["highs", "highs-ds", "highs-ipm"])
240@pytest.mark.parametrize("fit_intercept", [True, False])
241def test_sparse_input(sparse_container, solver, fit_intercept, global_random_seed):
242 """Test that sparse and dense X give same results."""
243 n_informative = 10
244 quantile_level = 0.6
245 X, y = make_regression(
246 n_samples=300,
247 n_features=20,
248 n_informative=10,
249 random_state=global_random_seed,
250 noise=1.0,
251 )
252 X_sparse = sparse_container(X)
253 alpha = 0.1
254 quant_dense = QuantileRegressor(
255 quantile=quantile_level, alpha=alpha, fit_intercept=fit_intercept
256 ).fit(X, y)
257 quant_sparse = QuantileRegressor(
258 quantile=quantile_level, alpha=alpha, fit_intercept=fit_intercept, solver=solver
259 ).fit(X_sparse, y)
260 assert_allclose(quant_sparse.coef_, quant_dense.coef_, rtol=1e-2)
261 sparse_support = quant_sparse.coef_ != 0
262 dense_support = quant_dense.coef_ != 0
263 assert dense_support.sum() == pytest.approx(n_informative, abs=1)
264 assert sparse_support.sum() == pytest.approx(n_informative, abs=1)
265 if fit_intercept:
266 assert quant_sparse.intercept_ == approx(quant_dense.intercept_)
267 # check that we still predict fraction
268 empirical_coverage = np.mean(y < quant_sparse.predict(X_sparse))
269 assert empirical_coverage == approx(quantile_level, abs=3e-2)
