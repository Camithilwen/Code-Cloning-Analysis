237 def test_more_step_and_impulse(self):
238 lambda1 = 0.5
239 lambda2 = 0.75
240 a = np.array([[lambda1, 0.0],
241 [0.0, lambda2]])
242 b = np.array([[1.0, 0.0],
243 [0.0, 1.0]])
244 c = np.array([[1.0, 1.0]])
245 d = np.array([[0.0, 0.0]])
246
247 n = 10
248
249 # Check a step response.
250 ts, ys = dstep((a, b, c, d, 1), n=n)
251
252 # Create the exact step response.
253 stp0 = (1.0 / (1 - lambda1)) * (1.0 - lambda1 ** np.arange(n))
254 stp1 = (1.0 / (1 - lambda2)) * (1.0 - lambda2 ** np.arange(n))
255
256 assert_allclose(ys[0][:, 0], stp0)
257 assert_allclose(ys[1][:, 0], stp1)
258
259 # Check an impulse response with an initial condition.
260 x0 = np.array([1.0, 1.0])
261 ti, yi = dimpulse((a, b, c, d, 1), n=n, x0=x0)
262
263 # Create the exact impulse response.
264 imp = (np.array([lambda1, lambda2]) **
265 np.arange(-1, n + 1).reshape(-1, 1))
266 imp[0, :] = 0.0
267 # Analytical solution to impulse response
268 y0 = imp[:n, 0] + np.dot(imp[1:n + 1, :], x0)
269 y1 = imp[:n, 1] + np.dot(imp[1:n + 1, :], x0)
270
271 assert_allclose(yi[0][:, 0], y0)
272 assert_allclose(yi[1][:, 0], y1)
273
274 # Check that dt=0.1, n=3 gives 3 time values.
275 system = ([1.0], [1.0, -0.5], 0.1)
276 t, (y,) = dstep(system, n=3)
277 assert_allclose(t, [0, 0.1, 0.2])
278 assert_array_equal(y.T, [[0, 1.0, 1.5]])
279 t, (y,) = dimpulse(system, n=3)
280 assert_allclose(t, [0, 0.1, 0.2])
281 assert_array_equal(y.T, [[0, 1, 0.5]])
