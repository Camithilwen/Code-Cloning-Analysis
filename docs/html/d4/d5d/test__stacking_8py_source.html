466 # convergence here
467 n_half_samples = len(y) // 2

469 [0.1] * n_half_samples + [0.9] * (len(y) - n_half_samples)

479 with ignore_warnings(category=ConvergenceWarning):
480 stacker.fit(X_train, y_train, sample_weight=np.ones(y_train.shape))
481 y_pred_unit_weight = stacker.predict(X_test)
482
483 assert_allclose(y_pred_no_weight, y_pred_unit_weight)
484
485 with ignore_warnings(category=ConvergenceWarning):
486 stacker.fit(X_train, y_train, sample_weight=sample_weight_train)
487 y_pred_biased = stacker.predict(X_test)
