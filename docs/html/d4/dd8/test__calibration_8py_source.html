63@pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
64@pytest.mark.parametrize("method", ["sigmoid", "isotonic"])
65@pytest.mark.parametrize("ensemble", [True, False])
66def test_calibration(data, method, csr_container, ensemble):
67 # Test calibration objects with isotonic and sigmoid
68 n_samples = N_SAMPLES // 2
69 X, y = data
70 sample_weight = np.random.RandomState(seed=42).uniform(size=y.size)
71
72 X -= X.min() # MultinomialNB only allows positive X
73
74 # split train and test
75 X_train, y_train, sw_train = X[:n_samples], y[:n_samples], sample_weight[:n_samples]
76 X_test, y_test = X[n_samples:], y[n_samples:]
77
78 # Naive-Bayes
79 clf = MultinomialNB().fit(X_train, y_train, sample_weight=sw_train)
80 prob_pos_clf = clf.predict_proba(X_test)[:, 1]
81
82 cal_clf = CalibratedClassifierCV(clf, cv=y.size + 1, ensemble=ensemble)
83 with pytest.raises(ValueError):
84 cal_clf.fit(X, y)
85
86 # Naive Bayes with calibration
87 for this_X_train, this_X_test in [
88 (X_train, X_test),
89 (csr_container(X_train), csr_container(X_test)),
90 ]:
91 cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
92 # Note that this fit overwrites the fit on the entire training
93 # set
94 cal_clf.fit(this_X_train, y_train, sample_weight=sw_train)
95 prob_pos_cal_clf = cal_clf.predict_proba(this_X_test)[:, 1]
96
97 # Check that brier score has improved after calibration
98 assert brier_score_loss(y_test, prob_pos_clf) > brier_score_loss(
99 y_test, prob_pos_cal_clf
100 )
101
102 # Check invariance against relabeling [0, 1] -> [1, 2]
103 cal_clf.fit(this_X_train, y_train + 1, sample_weight=sw_train)
104 prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
105 assert_array_almost_equal(prob_pos_cal_clf, prob_pos_cal_clf_relabeled)
106
107 # Check invariance against relabeling [0, 1] -> [-1, 1]
108 cal_clf.fit(this_X_train, 2 * y_train - 1, sample_weight=sw_train)
109 prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
110 assert_array_almost_equal(prob_pos_cal_clf, prob_pos_cal_clf_relabeled)
111
112 # Check invariance against relabeling [0, 1] -> [1, 0]
113 cal_clf.fit(this_X_train, (y_train + 1) % 2, sample_weight=sw_train)
114 prob_pos_cal_clf_relabeled = cal_clf.predict_proba(this_X_test)[:, 1]
115 if method == "sigmoid":
116 assert_array_almost_equal(prob_pos_cal_clf, 1 - prob_pos_cal_clf_relabeled)
117 else:
118 # Isotonic calibration is not invariant against relabeling
119 # but should improve in both cases
120 assert brier_score_loss(y_test, prob_pos_clf) > brier_score_loss(
121 (y_test + 1) % 2, prob_pos_cal_clf_relabeled
122 )

213@pytest.mark.parametrize("method", ["sigmoid", "isotonic"])
214@pytest.mark.parametrize("ensemble", [True, False])
215# increase the number of RNG seeds to assess the statistical stability of this
216# test:
217@pytest.mark.parametrize("seed", range(2))
218def test_calibration_multiclass(method, ensemble, seed):
219 def multiclass_brier(y_true, proba_pred, n_classes):
220 Y_onehot = np.eye(n_classes)[y_true]
221 return np.sum((Y_onehot - proba_pred) ** 2) / Y_onehot.shape[0]
222
223 # Test calibration for multiclass with classifier that implements
224 # only decision function.
225 clf = LinearSVC(random_state=7)
226 X, y = make_blobs(
227 n_samples=500, n_features=100, random_state=seed, centers=10, cluster_std=15.0
228 )
229
230 # Use an unbalanced dataset by collapsing 8 clusters into one class
231 # to make the naive calibration based on a softmax more unlikely
232 # to work.
233 y[y > 2] = 2
234 n_classes = np.unique(y).shape[0]
235 X_train, y_train = X[::2], y[::2]
236 X_test, y_test = X[1::2], y[1::2]
237
238 clf.fit(X_train, y_train)
239
240 cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
241 cal_clf.fit(X_train, y_train)
242 probas = cal_clf.predict_proba(X_test)
243 # Check probabilities sum to 1
244 assert_allclose(np.sum(probas, axis=1), np.ones(len(X_test)))
245
246 # Check that the dataset is not too trivial, otherwise it's hard
247 # to get interesting calibration data during the internal
248 # cross-validation loop.
249 assert 0.65 < clf.score(X_test, y_test) < 0.95
250
251 # Check that the accuracy of the calibrated model is never degraded
252 # too much compared to the original classifier.
253 assert cal_clf.score(X_test, y_test) > 0.95 * clf.score(X_test, y_test)
254
255 # Check that Brier loss of calibrated classifier is smaller than
256 # loss obtained by naively turning OvR decision function to
257 # probabilities via a softmax
258 uncalibrated_brier = multiclass_brier(
259 y_test, softmax(clf.decision_function(X_test)), n_classes=n_classes
260 )
261 calibrated_brier = multiclass_brier(y_test, probas, n_classes=n_classes)
262
263 assert calibrated_brier < 1.1 * uncalibrated_brier
264
265 # Test that calibration of a multiclass classifier decreases log-loss
266 # for RandomForestClassifier
267 clf = RandomForestClassifier(n_estimators=30, random_state=42)
268 clf.fit(X_train, y_train)
269 clf_probs = clf.predict_proba(X_test)
270 uncalibrated_brier = multiclass_brier(y_test, clf_probs, n_classes=n_classes)
271
272 cal_clf = CalibratedClassifierCV(clf, method=method, cv=5, ensemble=ensemble)
273 cal_clf.fit(X_train, y_train)
274 cal_clf_probs = cal_clf.predict_proba(X_test)
275 calibrated_brier = multiclass_brier(y_test, cal_clf_probs, n_classes=n_classes)
276 assert calibrated_brier < 1.1 * uncalibrated_brier

279def test_calibration_zero_probability():
280 # Test an edge case where _CalibratedClassifier avoids numerical errors
281 # in the multiclass normalization step if all the calibrators output
282 # are zero all at once for a given sample and instead fallback to uniform
283 # probabilities.
284 class ZeroCalibrator:
285 # This function is called from _CalibratedClassifier.predict_proba.
286 def predict(self, X):
287 return np.zeros(X.shape[0])
288
289 X, y = make_blobs(
290 n_samples=50, n_features=10, random_state=7, centers=10, cluster_std=15.0
291 )
292 clf = DummyClassifier().fit(X, y)
293 calibrator = ZeroCalibrator()
294 cal_clf = _CalibratedClassifier(
295 estimator=clf, calibrators=[calibrator], classes=clf.classes_
296 )
297
298 probas = cal_clf.predict_proba(X)
299
300 # Check that all probabilities are uniformly 1. / clf.n_classes_
301 assert_allclose(probas, 1.0 / clf.n_classes_)
