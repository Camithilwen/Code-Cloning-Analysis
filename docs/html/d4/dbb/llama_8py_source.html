78def truncate_code(code, max_lines=50):
79 '''! Truncates code to a limited number of lines for prompt fitting.
80
81 @param code Full code string.
82 @param max_lines Maximum number of lines to retain.
83 @return Truncated code string.'''
84 lines = code.splitlines()
85 return '\n'.join(lines[:max_lines]) if len(lines) > max_lines else code

87# --- ENSEMBLE ASSESSMENT ---
88def ensemble_assessment(code1, code2, model_name, n=3):
89 '''! Performs multiple assessments and aggregates predictions using voting.
90
91 @param code1 First code string.
92 @param code2 Second code string.
93 @param model_name The name of the LLM model to use.
94 @param thr Threshold for similarity.
95 @param n Number of repeated assessments for ensemble.
96 @return Tuple of (average type scores, majority predicted type, majority binary similarity).'''
97 predictions = []
98 for _ in range(n):
99 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name)
100 predictions.append((results, predicted_type, predicted_sim))
101 # Majority voting on predicted_sim
102 sim_votes = [pred[2] for pred in predictions]
103 final_sim = max(set(sim_votes), key=sim_votes.count)
104 # Most frequent predicted_type
105 type_votes = [pred[1] for pred in predictions]
106 final_type = max(set(type_votes), key=type_votes.count)
107 # Average results for reporting
108 avg_results = {t: sum(pred[0][t] for pred in predictions) / n for t in ["Type-1", "Type-2", "Type-3", "Type-4"]}
109 return avg_results, final_type, final_sim

124 '''! Sends a prompt to the LLM to classify clone type and score similarities.

128 @param model_name Name of the language model to query.
129 @param thr Threshold for determining clone similarity.

131 prompt = f"""You are a code similarity expert. Analyze the following code pair step by step for clone detection.
132
133Definitions:
134- Type-1: Identical except for whitespace, comments, layout.
135- Type-2: Identical except for variable/function names (plus Type-1 differences).
136- Type-3: Similar, but with some statements added/removed/modified (plus Type-2 differences).
137- Type-4: Syntactically different but functionally identical (same outputs for same inputs).
138
139Examples:
140Type-1: 'int a=5;' vs 'int a = 5; // set a'
141Type-2: 'int a=5;' vs 'int b=5;'
142Type-3: 'int a=5; print(a);' vs 'int a=5;'
143Type-4: 'for(int i=0;i<5;i++)sum+=i;' vs 'sum = sum_of_first_n(5);'
144
145Step-by-step:
1461. Are the outputs always identical for all inputs? (Type-4)
1472. Are they identical except for whitespace/comments? (Type-1)
1483. Are names changed but structure identical? (Type-2)
1494. Are there statement-level edits? (Type-3)
150Explain your reasoning for each type and provide a confidence score (0-1).
151
152Target Code:
153{code1}
154
155Similar Code:
156{code2}
157
158Respond ONLY with a JSON object with the following keys: Type-1, Type-2, Type-3, Type-4, each with a confidence score between 0 and 1.
159"""
