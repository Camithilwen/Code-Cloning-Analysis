148 def test_correctness(self):
149 """This test ensures the correctness of the implemented Taylor
150 Windowing function. A Taylor Window of 1024 points is created, its FFT
151 is taken, and the Peak Sidelobe Level (PSLL) and 3dB and 18dB bandwidth
152 are found and checked.
153
154 A publication from Sandia National Laboratories was used as reference
155 for the correctness values [1]_.
156
157 References
158 -----
159 .. [1] Armin Doerry, "Catalog of Window Taper Functions for
160 Sidelobe Control", 2017.
161 https://www.researchgate.net/profile/Armin_Doerry/publication/316281181_Catalog_of_Window_Taper_Functions_for_Sidelobe_Control/links/58f92cb2a6fdccb121c9d54d/Catalog-of-Window-Taper-Functions-for-Sidelobe-Control.pdf
162 """
163 M_win = 1024
164 N_fft = 131072
165 # Set norm=False for correctness as the values obtained from the
166 # scientific publication do not normalize the values. Normalizing
167 # changes the sidelobe level from the desired value.
168 w = windows.taylor(M_win, nbar=4, sll=35, norm=False, sym=False)
169 f = fft(w, N_fft)
170 spec = 20 * np.log10(np.abs(f / np.amax(f)))
171
172 first_zero = np.argmax(np.diff(spec) > 0)
173
174 PSLL = np.amax(spec[first_zero:-first_zero])
175
176 BW_3dB = 2*np.argmax(spec <= -3.0102999566398121) / N_fft * M_win
177 BW_18dB = 2*np.argmax(spec <= -18.061799739838872) / N_fft * M_win
178
179 assert_allclose(PSLL, -35.1672, atol=1)
180 assert_allclose(BW_3dB, 1.1822, atol=0.1)
181 assert_allclose(BW_18dB, 2.6112, atol=0.1)
