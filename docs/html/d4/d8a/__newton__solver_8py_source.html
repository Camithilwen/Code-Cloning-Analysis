22class NewtonSolver(ABC):
23 """Newton solver for GLMs.
24
25 This class implements Newton/2nd-order optimization routines for GLMs. Each Newton
26 iteration aims at finding the Newton step which is done by the inner solver. With
27 Hessian H, gradient g and coefficients coef, one step solves:
28
29 H @ coef_newton = -g
30
31 For our GLM / LinearModelLoss, we have gradient g and Hessian H:
32
33 g = X.T @ loss.gradient + l2_reg_strength * coef
34 H = X.T @ diag(loss.hessian) @ X + l2_reg_strength * identity
35
36 Backtracking line search updates coef = coef_old + t * coef_newton for some t in
37 (0, 1].
38
39 This is a base class, actual implementations (child classes) may deviate from the
40 above pattern and use structure specific tricks.
41
42 Usage pattern:
43 - initialize solver: sol = NewtonSolver(...)
44 - solve the problem: sol.solve(X, y, sample_weight)
45
46 References
47 ----------
48 - Jorge Nocedal, Stephen J. Wright. (2006) "Numerical Optimization"
49 2nd edition
50 https://doi.org/10.1007/978-0-387-40065-5
51
52 - Stephen P. Boyd, Lieven Vandenberghe. (2004) "Convex Optimization."
53 Cambridge University Press, 2004.
54 https://web.stanford.edu/~boyd/cvxbook/bv_cvxbook.pdf
55
56 Parameters
57 ----------
58 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
59 Initial coefficients of a linear model.
60 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
61 i.e. one reconstructs the 2d-array via
62 coef.reshape((n_classes, -1), order="F").
63
64 linear_loss : LinearModelLoss
65 The loss to be minimized.
66
67 l2_reg_strength : float, default=0.0
68 L2 regularization strength.
69
70 tol : float, default=1e-4
71 The optimization problem is solved when each of the following condition is
72 fulfilled:
73 1. maximum |gradient| <= tol
74 2. Newton decrement d: 1/2 * d^2 <= tol
75
76 max_iter : int, default=100
77 Maximum number of Newton steps allowed.
78
79 n_threads : int, default=1
80 Number of OpenMP threads to use for the computation of the Hessian and gradient
81 of the loss function.
82
83 Attributes
84 ----------
85 coef_old : ndarray of shape coef.shape
86 Coefficient of previous iteration.
87
88 coef_newton : ndarray of shape coef.shape
89 Newton step.
90
91 gradient : ndarray of shape coef.shape
92 Gradient of the loss w.r.t. the coefficients.
93
94 gradient_old : ndarray of shape coef.shape
95 Gradient of previous iteration.
96
97 loss_value : float
98 Value of objective function = loss + penalty.
99
100 loss_value_old : float
101 Value of objective function of previous itertion.
102
103 raw_prediction : ndarray of shape (n_samples,) or (n_samples, n_classes)
104
105 converged : bool
106 Indicator for convergence of the solver.
107
108 iteration : int
109 Number of Newton steps, i.e. calls to inner_solve
110
111 use_fallback_lbfgs_solve : bool
112 If set to True, the solver will resort to call LBFGS to finish the optimisation
113 procedure in case of convergence issues.
114
115 gradient_times_newton : float
116 gradient @ coef_newton, set in inner_solve and used by line_search. If the
117 Newton step is a descent direction, this is negative.
118 """

172 def fallback_lbfgs_solve(self, X, y, sample_weight):
173 """Fallback solver in case of emergency.
174
175 If a solver detects convergence problems, it may fall back to this methods in
176 the hope to exit with success instead of raising an error.
177
178 Sets:
179 - self.coef
180 - self.converged
181 """
182 max_iter = self.max_iter - self.iteration
183 opt_res = scipy.optimize.minimize(
184 self.linear_loss.loss_gradient,
185 self.coef,
186 method="L-BFGS-B",
187 jac=True,
188 options={
189 "maxiter": max_iter,
190 "maxls": 50, # default is 20
191 "gtol": self.tol,
192 "ftol": 64 * np.finfo(np.float64).eps,
193 **_get_additional_lbfgs_options_dict("iprint", self.verbose - 1),
194 },
195 args=(X, y, sample_weight, self.l2_reg_strength, self.n_threads),
196 )
197 self.iteration += _check_optimize_result("lbfgs", opt_res, max_iter=max_iter)
198 self.coef = opt_res.x
199 self.converged = opt_res.status == 0

201 def line_search(self, X, y, sample_weight):
202 """Backtracking line search.
203
204 Sets:
205 - self.coef_old
206 - self.coef
207 - self.loss_value_old
208 - self.loss_value
209 - self.gradient_old
210 - self.gradient
211 - self.raw_prediction
212 """
213 # line search parameters
214 beta, sigma = 0.5, 0.00048828125 # 1/2, 1/2**11
215 eps = 16 * np.finfo(self.loss_value.dtype).eps
216 t = 1 # step size
217
218 # gradient_times_newton = self.gradient @ self.coef_newton
219 # was computed in inner_solve.
220 armijo_term = sigma * self.gradient_times_newton
221 _, _, raw_prediction_newton = self.linear_loss.weight_intercept_raw(
222 self.coef_newton, X
223 )
224
225 self.coef_old = self.coef
226 self.loss_value_old = self.loss_value
227 self.gradient_old = self.gradient
228
229 # np.sum(np.abs(self.gradient_old))
230 sum_abs_grad_old = -1
231
232 is_verbose = self.verbose >= 2
233 if is_verbose:
234 print(" Backtracking Line Search")
235 print(f" eps=16 * finfo.eps={eps}")
236
237 for i in range(21): # until and including t = beta**20 ~ 1e-6
238 self.coef = self.coef_old + t * self.coef_newton
239 raw = self.raw_prediction + t * raw_prediction_newton
240 self.loss_value, self.gradient = self.linear_loss.loss_gradient(
241 coef=self.coef,
242 X=X,
243 y=y,
244 sample_weight=sample_weight,
245 l2_reg_strength=self.l2_reg_strength,
246 n_threads=self.n_threads,
247 raw_prediction=raw,
248 )
249 # Note: If coef_newton is too large, loss_gradient may produce inf values,
250 # potentially accompanied by a RuntimeWarning.
251 # This case will be captured by the Armijo condition.
252
253 # 1. Check Armijo / sufficient decrease condition.
254 # The smaller (more negative) the better.
255 loss_improvement = self.loss_value - self.loss_value_old
256 check = loss_improvement <= t * armijo_term
257 if is_verbose:
258 print(
259 f" line search iteration={i + 1}, step size={t}\n"
260 f" check loss improvement <= armijo term: {loss_improvement} "
261 f"<= {t * armijo_term} {check}"
262 )
263 if check:
264 break
265 # 2. Deal with relative loss differences around machine precision.
266 tiny_loss = np.abs(self.loss_value_old * eps)
267 check = np.abs(loss_improvement) <= tiny_loss
268 if is_verbose:
269 print(
270 " check loss |improvement| <= eps * |loss_old|:"
271 f" {np.abs(loss_improvement)} <= {tiny_loss} {check}"
272 )
273 if check:
274 if sum_abs_grad_old < 0:
275 sum_abs_grad_old = scipy.linalg.norm(self.gradient_old, ord=1)
276 # 2.1 Check sum of absolute gradients as alternative condition.
277 sum_abs_grad = scipy.linalg.norm(self.gradient, ord=1)
278 check = sum_abs_grad < sum_abs_grad_old
279 if is_verbose:
280 print(
281 " check sum(|gradient|) < sum(|gradient_old|): "
282 f"{sum_abs_grad} < {sum_abs_grad_old} {check}"
283 )
284 if check:
285 break
286
287 t *= beta
288 else:
289 warnings.warn(
290 (
291 f"Line search of Newton solver {self.__class__.__name__} at"
292 f" iteration #{self.iteration} did no converge after 21 line search"
293 " refinement iterations. It will now resort to lbfgs instead."
294 ),
295 ConvergenceWarning,
296 )
297 if self.verbose:
298 print(" Line search did not converge and resorts to lbfgs instead.")
299 self.use_fallback_lbfgs_solve = True
300 return
301
302 self.raw_prediction = raw
303 if is_verbose:
304 print(
305 f" line search successful after {i + 1} iterations with "
306 f"loss={self.loss_value}."
307 )

309 def check_convergence(self, X, y, sample_weight):
310 """Check for convergence.
311
312 Sets self.converged.
313 """
314 if self.verbose:
315 print(" Check Convergence")
316 # Note: Checking maximum relative change of coefficient <= tol is a bad
317 # convergence criterion because even a large step could have brought us close
318 # to the true minimum.
319 # coef_step = self.coef - self.coef_old
320 # change = np.max(np.abs(coef_step) / np.maximum(1, np.abs(self.coef_old)))
321 # check = change <= tol
322
323 # 1. Criterion: maximum |gradient| <= tol
324 # The gradient was already updated in line_search()
325 g_max_abs = np.max(np.abs(self.gradient))
326 check = g_max_abs <= self.tol
327 if self.verbose:
328 print(f" 1. max |gradient| {g_max_abs} <= {self.tol} {check}")
329 if not check:
330 return
331
332 # 2. Criterion: For Newton decrement d, check 1/2 * d^2 <= tol
333 # d = sqrt(grad @ hessian^-1 @ grad)
334 # = sqrt(coef_newton @ hessian @ coef_newton)
335 # See Boyd, Vanderberghe (2009) "Convex Optimization" Chapter 9.5.1.
336 d2 = self.coef_newton @ self.hessian @ self.coef_newton
337 check = 0.5 * d2 <= self.tol
338 if self.verbose:
339 print(f" 2. Newton decrement {0.5 * d2} <= {self.tol} {check}")
340 if not check:
341 return
342
343 if self.verbose:
344 loss_value = self.linear_loss.loss(
345 coef=self.coef,
346 X=X,
347 y=y,
348 sample_weight=sample_weight,
349 l2_reg_strength=self.l2_reg_strength,
350 n_threads=self.n_threads,
351 )
352 print(f" Solver did converge at loss = {loss_value}.")
353 self.converged = True

362 def solve(self, X, y, sample_weight):
363 """Solve the optimization problem.
364
365 This is the main routine.
366
367 Order of calls:
368 self.setup()
369 while iteration:
370 self.update_gradient_hessian()
371 self.inner_solve()
372 self.line_search()
373 self.check_convergence()
374 self.finalize()
375
376 Returns
377 -------
378 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
379 Solution of the optimization problem.
380 """
381 # setup usually:
382 # - initializes self.coef if needed
383 # - initializes and calculates self.raw_predictions, self.loss_value
384 self.setup(X=X, y=y, sample_weight=sample_weight)
385
386 self.iteration = 1
387 self.converged = False
388 self.use_fallback_lbfgs_solve = False
389
390 while self.iteration <= self.max_iter and not self.converged:
391 if self.verbose:
392 print(f"Newton iter={self.iteration}")
393
394 self.use_fallback_lbfgs_solve = False # Fallback solver.
395
396 # 1. Update Hessian and gradient
397 self.update_gradient_hessian(X=X, y=y, sample_weight=sample_weight)
398
399 # TODO:
400 # if iteration == 1:
401 # We might stop early, e.g. we already are close to the optimum,
402 # usually detected by zero gradients at this stage.
403
404 # 2. Inner solver
405 # Calculate Newton step/direction
406 # This usually sets self.coef_newton and self.gradient_times_newton.
407 self.inner_solve(X=X, y=y, sample_weight=sample_weight)
408 if self.use_fallback_lbfgs_solve:
409 break
410
411 # 3. Backtracking line search
412 # This usually sets self.coef_old, self.coef, self.loss_value_old
413 # self.loss_value, self.gradient_old, self.gradient,
414 # self.raw_prediction.
415 self.line_search(X=X, y=y, sample_weight=sample_weight)
416 if self.use_fallback_lbfgs_solve:
417 break
418
419 # 4. Check convergence
420 # Sets self.converged.
421 self.check_convergence(X=X, y=y, sample_weight=sample_weight)
422
423 # 5. Next iteration
424 self.iteration += 1
425
426 if not self.converged:
427 if self.use_fallback_lbfgs_solve:
428 # Note: The fallback solver circumvents check_convergence and relies on
429 # the convergence checks of lbfgs instead. Enough warnings have been
430 # raised on the way.
431 self.fallback_lbfgs_solve(X=X, y=y, sample_weight=sample_weight)
432 else:
433 warnings.warn(
434 (
435 f"Newton solver did not converge after {self.iteration - 1} "
436 "iterations."
437 ),
438 ConvergenceWarning,
439 )
440
441 self.iteration -= 1
442 self.finalize(X=X, y=y, sample_weight=sample_weight)
443 return self.coef

446class NewtonCholeskySolver(NewtonSolver):
447 """Cholesky based Newton solver.
448
449 Inner solver for finding the Newton step H w_newton = -g uses Cholesky based linear
450 solver.
451 """
452
453 def setup(self, X, y, sample_weight):
454 super().setup(X=X, y=y, sample_weight=sample_weight)
455 if self.linear_loss.base_loss.is_multiclass:
456 # Easier with ravelled arrays, e.g., for scipy.linalg.solve.
457 # As with LinearModelLoss, we always are contiguous in n_classes.
458 self.coef = self.coef.ravel(order="F")
459 # Note that the computation of gradient in LinearModelLoss follows the shape of
460 # coef.
461 self.gradient = np.empty_like(self.coef)
462 # But the hessian is always 2d.
463 n = self.coef.size
464 self.hessian = np.empty_like(self.coef, shape=(n, n))
465 # To help case distinctions.
466 self.is_multinomial_with_intercept = (
467 self.linear_loss.base_loss.is_multiclass and self.linear_loss.fit_intercept
468 )
469 self.is_multinomial_no_penalty = (
470 self.linear_loss.base_loss.is_multiclass and self.l2_reg_strength == 0
471 )
472
473 def update_gradient_hessian(self, X, y, sample_weight):
474 _, _, self.hessian_warning = self.linear_loss.gradient_hessian(
475 coef=self.coef,
476 X=X,
477 y=y,
478 sample_weight=sample_weight,
479 l2_reg_strength=self.l2_reg_strength,
480 n_threads=self.n_threads,
481 gradient_out=self.gradient,
482 hessian_out=self.hessian,
483 raw_prediction=self.raw_prediction, # this was updated in line_search
484 )

486 def inner_solve(self, X, y, sample_weight):
487 if self.hessian_warning:
488 warnings.warn(
489 (
490 f"The inner solver of {self.__class__.__name__} detected a "
491 "pointwise hessian with many negative values at iteration "
492 f"#{self.iteration}. It will now resort to lbfgs instead."
493 ),
494 ConvergenceWarning,
495 )
496 if self.verbose:
497 print(
498 " The inner solver detected a pointwise Hessian with many "
499 "negative values and resorts to lbfgs instead."
500 )
501 self.use_fallback_lbfgs_solve = True
502 return
503
504 # Note: The following case distinction could also be shifted to the
505 # implementation of HalfMultinomialLoss instead of here within the solver.
506 if self.is_multinomial_no_penalty:
507 # The multinomial loss is overparametrized for each unpenalized feature, so
508 # at least the intercepts. This can be seen by noting that predicted
509 # probabilities are invariant under shifting all coefficients of a single
510 # feature j for all classes by the same amount c:
511 # coef[k, :] -> coef[k, :] + c => proba stays the same
512 # where we have assumed coef.shape = (n_classes, n_features).
513 # Therefore, also the loss (-log-likelihood), gradient and hessian stay the
514 # same, see
515 # Noah Simon and Jerome Friedman and Trevor Hastie. (2013) "A Blockwise
516 # Descent Algorithm for Group-penalized Multiresponse and Multinomial
517 # Regression". https://doi.org/10.48550/arXiv.1311.6529
518 #
519 # We choose the standard approach and set all the coefficients of the last
520 # class to zero, for all features including the intercept.
521 n_classes = self.linear_loss.base_loss.n_classes
522 n_dof = self.coef.size // n_classes # degree of freedom per class
523 n = self.coef.size - n_dof # effective size
524 self.coef[n_classes - 1 :: n_classes] = 0
525 self.gradient[n_classes - 1 :: n_classes] = 0
526 self.hessian[n_classes - 1 :: n_classes, :] = 0
527 self.hessian[:, n_classes - 1 :: n_classes] = 0
528 # We also need the reduced variants of gradient and hessian where the
529 # entries set to zero are removed. For 2 features and 3 classes with
530 # arbitrary values, "x" means removed:
531 # gradient = [0, 1, x, 3, 4, x]
532 #
533 # hessian = [0, 1, x, 3, 4, x]
534 # [1, 7, x, 9, 10, x]
535 # [x, x, x, x, x, x]
536 # [3, 9, x, 21, 22, x]
537 # [4, 10, x, 22, 28, x]
538 # [x, x, x, x, x, x]
539 # The following slicing triggers copies of gradient and hessian.
540 gradient = self.gradient.reshape(-1, n_classes)[:, :-1].flatten()
541 hessian = self.hessian.reshape(n_dof, n_classes, n_dof, n_classes)[
542 :, :-1, :, :-1
543 ].reshape(n, n)
544 elif self.is_multinomial_with_intercept:
545 # Here, only intercepts are unpenalized. We again choose the last class and
546 # set its intercept to zero.
547 self.coef[-1] = 0
548 self.gradient[-1] = 0
549 self.hessian[-1, :] = 0
550 self.hessian[:, -1] = 0
551 gradient, hessian = self.gradient[:-1], self.hessian[:-1, :-1]
552 else:
553 gradient, hessian = self.gradient, self.hessian
554
555 try:
556 with warnings.catch_warnings():
557 warnings.simplefilter("error", scipy.linalg.LinAlgWarning)
558 self.coef_newton = scipy.linalg.solve(
559 hessian, -gradient, check_finite=False, assume_a="sym"
560 )
561 if self.is_multinomial_no_penalty:
562 self.coef_newton = np.c_[
563 self.coef_newton.reshape(n_dof, n_classes - 1), np.zeros(n_dof)
564 ].reshape(-1)
565 assert self.coef_newton.flags.f_contiguous
566 elif self.is_multinomial_with_intercept:
567 self.coef_newton = np.r_[self.coef_newton, 0]
568 self.gradient_times_newton = self.gradient @ self.coef_newton
569 if self.gradient_times_newton > 0:
570 if self.verbose:
571 print(
572 " The inner solver found a Newton step that is not a "
573 "descent direction and resorts to LBFGS steps instead."
574 )
575 self.use_fallback_lbfgs_solve = True
576 return
577 except (np.linalg.LinAlgError, scipy.linalg.LinAlgWarning) as e:
578 warnings.warn(
579 f"The inner solver of {self.__class__.__name__} stumbled upon a "
580 "singular or very ill-conditioned Hessian matrix at iteration "
581 f"{self.iteration}. It will now resort to lbfgs instead.\n"
582 "Further options are to use another solver or to avoid such situation "
583 "in the first place. Possible remedies are removing collinear features"
584 " of X or increasing the penalization strengths.\n"
585 "The original Linear Algebra message was:\n" + str(e),
586 scipy.linalg.LinAlgWarning,
587 )
588 # Possible causes:
589 # 1. hess_pointwise is negative. But this is already taken care in
590 # LinearModelLoss.gradient_hessian.
591 # 2. X is singular or ill-conditioned
592 # This might be the most probable cause.
593 #
594 # There are many possible ways to deal with this situation. Most of them
595 # add, explicitly or implicitly, a matrix to the hessian to make it
596 # positive definite, confer to Chapter 3.4 of Nocedal & Wright 2nd ed.
597 # Instead, we resort to lbfgs.
598 if self.verbose:
599 print(
600 " The inner solver stumbled upon an singular or ill-conditioned "
601 "Hessian matrix and resorts to LBFGS instead."
602 )
603 self.use_fallback_lbfgs_solve = True
604 return

606 def finalize(self, X, y, sample_weight):
607 if self.is_multinomial_no_penalty:
608 # Our convention is usually the symmetric parametrization where
609 # sum(coef[classes, features], axis=0) = 0.
610 # We convert now to this convention. Note that it does not change
611 # the predicted probabilities.
612 n_classes = self.linear_loss.base_loss.n_classes
613 self.coef = self.coef.reshape(n_classes, -1, order="F")
614 self.coef -= np.mean(self.coef, axis=0)
615 elif self.is_multinomial_with_intercept:
616 # Only the intercept needs an update to the symmetric parametrization.
617 n_classes = self.linear_loss.base_loss.n_classes
618 self.coef[-n_classes:] -= np.mean(self.coef[-n_classes:])
