30class _BisectingTree:
31 """Tree structure representing the hierarchical clusters of BisectingKMeans."""
32
33 def __init__(self, center, indices, score):
34 """Create a new cluster node in the tree.
35
36 The node holds the center of this cluster and the indices of the data points
37 that belong to it.
38 """
39 self.center = center
40 self.indices = indices
41 self.score = score
42
43 self.left = None
44 self.right = None
45
46 def split(self, labels, centers, scores):
47 """Split the cluster node into two subclusters."""
48 self.left = _BisectingTree(
49 indices=self.indices[labels == 0], center=centers[0], score=scores[0]
50 )
51 self.right = _BisectingTree(
52 indices=self.indices[labels == 1], center=centers[1], score=scores[1]
53 )
54
55 # reset the indices attribute to save memory
56 self.indices = None
57
58 def get_cluster_to_bisect(self):
59 """Return the cluster node to bisect next.
60
61 It's based on the score of the cluster, which can be either the number of
62 data points assigned to that cluster or the inertia of that cluster
63 (see `bisecting_strategy` for details).
64 """
65 max_score = None
66
67 for cluster_leaf in self.iter_leaves():
68 if max_score is None or cluster_leaf.score > max_score:
69 max_score = cluster_leaf.score
70 best_cluster_leaf = cluster_leaf
71
72 return best_cluster_leaf
73
74 def iter_leaves(self):
75 """Iterate over all the cluster leaves in the tree."""
76 if self.left is None:
77 yield self
78 else:
79 yield from self.left.iter_leaves()
80 yield from self.right.iter_leaves()

83class BisectingKMeans(_BaseKMeans):
84 """Bisecting K-Means clustering.
85
86 Read more in the :ref:`User Guide <bisect_k_means>`.
87
88 .. versionadded:: 1.1
89
90 Parameters
91 ----------
92 n_clusters : int, default=8
93 The number of clusters to form as well as the number of
94 centroids to generate.
95
96 init : {'k-means++', 'random'} or callable, default='random'
97 Method for initialization:
98
99 'k-means++' : selects initial cluster centers for k-mean
100 clustering in a smart way to speed up convergence. See section
101 Notes in k_init for more details.
102
103 'random': choose `n_clusters` observations (rows) at random from data
104 for the initial centroids.
105
106 If a callable is passed, it should take arguments X, n_clusters and a
107 random state and return an initialization.
108
109 n_init : int, default=1
110 Number of time the inner k-means algorithm will be run with different
111 centroid seeds in each bisection.
112 That will result producing for each bisection best output of n_init
113 consecutive runs in terms of inertia.
114
115 random_state : int, RandomState instance or None, default=None
116 Determines random number generation for centroid initialization
117 in inner K-Means. Use an int to make the randomness deterministic.
118 See :term:`Glossary <random_state>`.
119
120 max_iter : int, default=300
121 Maximum number of iterations of the inner k-means algorithm at each
122 bisection.
123
124 verbose : int, default=0
125 Verbosity mode.
126
127 tol : float, default=1e-4
128 Relative tolerance with regards to Frobenius norm of the difference
129 in the cluster centers of two consecutive iterations to declare
130 convergence. Used in inner k-means algorithm at each bisection to pick
131 best possible clusters.
132
133 copy_x : bool, default=True
134 When pre-computing distances it is more numerically accurate to center
135 the data first. If copy_x is True (default), then the original data is
136 not modified. If False, the original data is modified, and put back
137 before the function returns, but small numerical differences may be
138 introduced by subtracting and then adding the data mean. Note that if
139 the original data is not C-contiguous, a copy will be made even if
140 copy_x is False. If the original data is sparse, but not in CSR format,
141 a copy will be made even if copy_x is False.
142
143 algorithm : {"lloyd", "elkan"}, default="lloyd"
144 Inner K-means algorithm used in bisection.
145 The classical EM-style algorithm is `"lloyd"`.
146 The `"elkan"` variation can be more efficient on some datasets with
147 well-defined clusters, by using the triangle inequality. However it's
148 more memory intensive due to the allocation of an extra array of shape
149 `(n_samples, n_clusters)`.
150
151 bisecting_strategy : {"biggest_inertia", "largest_cluster"},\
152 default="biggest_inertia"
153 Defines how bisection should be performed:
154
155 - "biggest_inertia" means that BisectingKMeans will always check
156 all calculated cluster for cluster with biggest SSE
157 (Sum of squared errors) and bisect it. This approach concentrates on
158 precision, but may be costly in terms of execution time (especially for
159 larger amount of data points).
160
161 - "largest_cluster" - BisectingKMeans will always split cluster with
162 largest amount of points assigned to it from all clusters
163 previously calculated. That should work faster than picking by SSE
164 ('biggest_inertia') and may produce similar results in most cases.
165
166 Attributes
167 ----------
168 cluster_centers_ : ndarray of shape (n_clusters, n_features)
169 Coordinates of cluster centers. If the algorithm stops before fully
170 converging (see ``tol`` and ``max_iter``), these will not be
171 consistent with ``labels_``.
172
173 labels_ : ndarray of shape (n_samples,)
174 Labels of each point.
175
176 inertia_ : float
177 Sum of squared distances of samples to their closest cluster center,
178 weighted by the sample weights if provided.
179
180 n_features_in_ : int
181 Number of features seen during :term:`fit`.
182
183 feature_names_in_ : ndarray of shape (`n_features_in_`,)
184 Names of features seen during :term:`fit`. Defined only when `X`
185 has feature names that are all strings.
186
187 See Also
188 --------
189 KMeans : Original implementation of K-Means algorithm.
190
191 Notes
192 -----
193 It might be inefficient when n_cluster is less than 3, due to unnecessary
194 calculations for that case.
195
196 Examples
197 --------
198 >>> from sklearn.cluster import BisectingKMeans
199 >>> import numpy as np
200 >>> X = np.array([[1, 1], [10, 1], [3, 1],
201 ... [10, 0], [2, 1], [10, 2],
202 ... [10, 8], [10, 9], [10, 10]])
203 >>> bisect_means = BisectingKMeans(n_clusters=3, random_state=0).fit(X)
204 >>> bisect_means.labels_
205 array([0, 2, 0, 2, 0, 2, 1, 1, 1], dtype=int32)
206 >>> bisect_means.predict([[0, 0], [12, 3]])
207 array([0, 2], dtype=int32)
208 >>> bisect_means.cluster_centers_
209 array([[ 2., 1.],
210 [10., 9.],
211 [10., 1.]])
212
213 For a comparison between BisectingKMeans and K-Means refer to example
214 :ref:`sphx_glr_auto_examples_cluster_plot_bisect_kmeans.py`.
215 """

263 def _inertia_per_cluster(self, X, centers, labels, sample_weight):
264 """Calculate the sum of squared errors (inertia) per cluster.
265
266 Parameters
267 ----------
268 X : {ndarray, csr_matrix} of shape (n_samples, n_features)
269 The input samples.
270
271 centers : ndarray of shape (n_clusters=2, n_features)
272 The cluster centers.
273
274 labels : ndarray of shape (n_samples,)
275 Index of the cluster each sample belongs to.
276
277 sample_weight : ndarray of shape (n_samples,)
278 The weights for each observation in X.
279
280 Returns
281 -------
282 inertia_per_cluster : ndarray of shape (n_clusters=2,)
283 Sum of squared errors (inertia) for each cluster.
284 """
285 n_clusters = centers.shape[0] # = 2 since centers comes from a bisection
286 _inertia = _inertia_sparse if sp.issparse(X) else _inertia_dense
287
288 inertia_per_cluster = np.empty(n_clusters)
289 for label in range(n_clusters):
290 inertia_per_cluster[label] = _inertia(
291 X, sample_weight, centers, labels, self._n_threads, single_label=label
292 )
293
294 return inertia_per_cluster

296 def _bisect(self, X, x_squared_norms, sample_weight, cluster_to_bisect):
297 """Split a cluster into 2 subsclusters.
298
299 Parameters
300 ----------
301 X : {ndarray, csr_matrix} of shape (n_samples, n_features)
302 Training instances to cluster.
303
304 x_squared_norms : ndarray of shape (n_samples,)
305 Squared euclidean norm of each data point.
306
307 sample_weight : ndarray of shape (n_samples,)
308 The weights for each observation in X.
309
310 cluster_to_bisect : _BisectingTree node object
311 The cluster node to split.
312 """
313 X = X[cluster_to_bisect.indices]
314 x_squared_norms = x_squared_norms[cluster_to_bisect.indices]
315 sample_weight = sample_weight[cluster_to_bisect.indices]
316
317 best_inertia = None
318
319 # Split samples in X into 2 clusters.
320 # Repeating `n_init` times to obtain best clusters
321 for _ in range(self.n_init):
322 centers_init = self._init_centroids(
323 X,
324 x_squared_norms=x_squared_norms,
325 init=self.init,
326 random_state=self._random_state,
327 n_centroids=2,
328 sample_weight=sample_weight,
329 )
330
331 labels, inertia, centers, _ = self._kmeans_single(
332 X,
333 sample_weight,
334 centers_init,
335 max_iter=self.max_iter,
336 verbose=self.verbose,
337 tol=self.tol,
338 n_threads=self._n_threads,
339 )
340
341 # allow small tolerance on the inertia to accommodate for
342 # non-deterministic rounding errors due to parallel computation
343 if best_inertia is None or inertia < best_inertia * (1 - 1e-6):
344 best_labels = labels
345 best_centers = centers
346 best_inertia = inertia
347
348 if self.verbose:
349 print(f"New centroids from bisection: {best_centers}")
350
351 if self.bisecting_strategy == "biggest_inertia":
352 scores = self._inertia_per_cluster(
353 X, best_centers, best_labels, sample_weight
354 )
355 else: # bisecting_strategy == "largest_cluster"
356 # Using minlength to make sure that we have the counts for both labels even
357 # if all samples are labelled 0.
358 scores = np.bincount(best_labels, minlength=2)
359
360 cluster_to_bisect.split(best_labels, best_centers, scores)

362 @_fit_context(prefer_skip_nested_validation=True)
363 def fit(self, X, y=None, sample_weight=None):
364 """Compute bisecting k-means clustering.
365
366 Parameters
367 ----------
368 X : {array-like, sparse matrix} of shape (n_samples, n_features)
369
370 Training instances to cluster.
371
372 .. note:: The data will be converted to C ordering,
373 which will cause a memory copy
374 if the given data is not C-contiguous.
375
376 y : Ignored
377 Not used, present here for API consistency by convention.
378
379 sample_weight : array-like of shape (n_samples,), default=None
380 The weights for each observation in X. If None, all observations
381 are assigned equal weight. `sample_weight` is not used during
382 initialization if `init` is a callable.
383
384 Returns
385 -------
386 self
387 Fitted estimator.
388 """
389 X = validate_data(
390 self,
391 X,
392 accept_sparse="csr",
393 dtype=[np.float64, np.float32],
394 order="C",
395 copy=self.copy_x,
396 accept_large_sparse=False,
397 )
398
399 self._check_params_vs_input(X)
400
401 self._random_state = check_random_state(self.random_state)
402 sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
403 self._n_threads = _openmp_effective_n_threads()
404
405 if self.algorithm == "lloyd" or self.n_clusters == 1:
406 self._kmeans_single = _kmeans_single_lloyd
407 self._check_mkl_vcomp(X, X.shape[0])
408 else:
409 self._kmeans_single = _kmeans_single_elkan
410
411 # Subtract of mean of X for more accurate distance computations
412 if not sp.issparse(X):
413 self._X_mean = X.mean(axis=0)
414 X -= self._X_mean
415
416 # Initialize the hierarchical clusters tree
417 self._bisecting_tree = _BisectingTree(
418 indices=np.arange(X.shape[0]),
419 center=X.mean(axis=0),
420 score=0,
421 )
422
423 x_squared_norms = row_norms(X, squared=True)
424
425 for _ in range(self.n_clusters - 1):
426 # Chose cluster to bisect
427 cluster_to_bisect = self._bisecting_tree.get_cluster_to_bisect()
428
429 # Split this cluster into 2 subclusters
430 self._bisect(X, x_squared_norms, sample_weight, cluster_to_bisect)
431
432 # Aggregate final labels and centers from the bisecting tree
433 self.labels_ = np.full(X.shape[0], -1, dtype=np.int32)
434 self.cluster_centers_ = np.empty((self.n_clusters, X.shape[1]), dtype=X.dtype)
435
436 for i, cluster_node in enumerate(self._bisecting_tree.iter_leaves()):
437 self.labels_[cluster_node.indices] = i
438 self.cluster_centers_[i] = cluster_node.center
439 cluster_node.label = i # label final clusters for future prediction
440 cluster_node.indices = None # release memory
441
442 # Restore original data
443 if not sp.issparse(X):
444 X += self._X_mean
445 self.cluster_centers_ += self._X_mean
446
447 _inertia = _inertia_sparse if sp.issparse(X) else _inertia_dense
448 self.inertia_ = _inertia(
449 X, sample_weight, self.cluster_centers_, self.labels_, self._n_threads
450 )
451
452 self._n_features_out = self.cluster_centers_.shape[0]
453
454 return self

456 def predict(self, X):
457 """Predict which cluster each sample in X belongs to.
458
459 Prediction is made by going down the hierarchical tree
460 in searching of closest leaf cluster.
461
462 In the vector quantization literature, `cluster_centers_` is called
463 the code book and each value returned by `predict` is the index of
464 the closest code in the code book.
465
466 Parameters
467 ----------
468 X : {array-like, sparse matrix} of shape (n_samples, n_features)
469 New data to predict.
470
471 Returns
472 -------
473 labels : ndarray of shape (n_samples,)
474 Index of the cluster each sample belongs to.
475 """
476 check_is_fitted(self)
477
478 X = self._check_test_data(X)
479 x_squared_norms = row_norms(X, squared=True)
480
481 # sample weights are unused but necessary in cython helpers
482 sample_weight = np.ones_like(x_squared_norms)
483
484 labels = self._predict_recursive(X, sample_weight, self._bisecting_tree)
485
486 return labels

488 def _predict_recursive(self, X, sample_weight, cluster_node):
489 """Predict recursively by going down the hierarchical tree.
490
491 Parameters
492 ----------
493 X : {ndarray, csr_matrix} of shape (n_samples, n_features)
494 The data points, currently assigned to `cluster_node`, to predict between
495 the subclusters of this node.
496
497 sample_weight : ndarray of shape (n_samples,)
498 The weights for each observation in X.
499
500 cluster_node : _BisectingTree node object
501 The cluster node of the hierarchical tree.
502
503 Returns
504 -------
505 labels : ndarray of shape (n_samples,)
506 Index of the cluster each sample belongs to.
507 """
508 if cluster_node.left is None:
509 # This cluster has no subcluster. Labels are just the label of the cluster.
510 return np.full(X.shape[0], cluster_node.label, dtype=np.int32)
511
512 # Determine if data points belong to the left or right subcluster
513 centers = np.vstack((cluster_node.left.center, cluster_node.right.center))
514 if hasattr(self, "_X_mean"):
515 centers += self._X_mean
516
517 cluster_labels = _labels_inertia_threadpool_limit(
518 X,
519 sample_weight,
520 centers,
521 self._n_threads,
522 return_inertia=False,
523 )
524 mask = cluster_labels == 0
525
526 # Compute the labels for each subset of the data points.
527 labels = np.full(X.shape[0], -1, dtype=np.int32)
528
529 labels[mask] = self._predict_recursive(
530 X[mask], sample_weight[mask], cluster_node.left
531 )
532
533 labels[~mask] = self._predict_recursive(
534 X[~mask], sample_weight[~mask], cluster_node.right
535 )
536
537 return labels
