9class Rosenbrock:
10 """Rosenbrock function.
11
12 The following optimization problem:
13 minimize sum(100.0*(x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0)
14 """
15
16 def __init__(self, n=2, random_state=0):
17 rng = np.random.RandomState(random_state)
18 self.x0 = rng.uniform(-1, 1, n)
19 self.x_opt = np.ones(n)
20
21 def fun(self, x):
22 x = np.asarray(x)
23 r = np.sum(100.0 * (x[1:] - x[:-1]**2.0)**2.0 + (1 - x[:-1])**2.0,
24 axis=0)
25 return r
26
27 def grad(self, x):
28 x = np.asarray(x)
29 xm = x[1:-1]
30 xm_m1 = x[:-2]
31 xm_p1 = x[2:]
32 der = np.zeros_like(x)
33 der[1:-1] = (200 * (xm - xm_m1**2) -
34 400 * (xm_p1 - xm**2) * xm - 2 * (1 - xm))
35 der[0] = -400 * x[0] * (x[1] - x[0]**2) - 2 * (1 - x[0])
36 der[-1] = 200 * (x[-1] - x[-2]**2)
37 return der
38
39 def hess(self, x):
40 x = np.atleast_1d(x)
41 H = np.diag(-400 * x[:-1], 1) - np.diag(400 * x[:-1], -1)
42 diagonal = np.zeros(len(x), dtype=x.dtype)
43 diagonal[0] = 1200 * x[0]**2 - 400 * x[1] + 2
44 diagonal[-1] = 200
45 diagonal[1:-1] = 202 + 1200 * x[1:-1]**2 - 400 * x[2:]
46 H = H + np.diag(diagonal)
47 return H
