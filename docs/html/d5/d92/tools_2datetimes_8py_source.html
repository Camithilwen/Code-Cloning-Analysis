151def should_cache(
152 arg: ArrayConvertible, unique_share: float = 0.7, check_count: int | None = None
153) -> bool:
154 """
155 Decides whether to do caching.
156
157 If the percent of unique elements among `check_count` elements less
158 than `unique_share * 100` then we can do caching.
159
160 Parameters
161 ----------
162 arg: listlike, tuple, 1-d array, Series
163 unique_share: float, default=0.7, optional
164 0 < unique_share < 1
165 check_count: int, optional
166 0 <= check_count <= len(arg)
167
168 Returns
169 -------
170 do_caching: bool
171
172 Notes
173 -----
174 By default for a sequence of less than 50 items in size, we don't do
175 caching; for the number of elements less than 5000, we take ten percent of
176 all elements to check for a uniqueness share; if the sequence size is more
177 than 5000, then we check only the first 500 elements.
178 All constants were chosen empirically by.
179 """
180 do_caching = True
181
182 # default realization
183 if check_count is None:
184 # in this case, the gain from caching is negligible
185 if len(arg) <= start_caching_at:
186 return False
187
188 if len(arg) <= 5000:
189 check_count = len(arg) // 10
190 else:
191 check_count = 500
192 else:
193 assert (
194 0 <= check_count <= len(arg)
195 ), "check_count must be in next bounds: [0; len(arg)]"
196 if check_count == 0:
197 return False
198
199 assert 0 < unique_share < 1, "unique_share must be in next bounds: (0; 1)"
200
201 try:
202 # We can't cache if the items are not hashable.
203 unique_elements = set(islice(arg, check_count))
204 except TypeError:
205 return False
206 if len(unique_elements) > check_count * unique_share:
207 do_caching = False
208 return do_caching

555def _adjust_to_origin(arg, origin, unit):
556 """
557 Helper function for to_datetime.
558 Adjust input argument to the specified origin
559
560 Parameters
561 ----------
562 arg : list, tuple, ndarray, Series, Index
563 date to be adjusted
564 origin : 'julian' or Timestamp
565 origin offset for the arg
566 unit : str
567 passed unit from to_datetime, must be 'D'
568
569 Returns
570 -------
571 ndarray or scalar of adjusted date(s)
572 """
573 if origin == "julian":
574 original = arg
575 j0 = Timestamp(0).to_julian_date()
576 if unit != "D":
577 raise ValueError("unit must be 'D' for origin='julian'")
578 try:
579 arg = arg - j0
580 except TypeError as err:
581 raise ValueError(
582 "incompatible 'arg' type for given 'origin'='julian'"
583 ) from err
584
585 # preemptively check this for a nice range
586 j_max = Timestamp.max.to_julian_date() - j0
587 j_min = Timestamp.min.to_julian_date() - j0
588 if np.any(arg > j_max) or np.any(arg < j_min):
589 raise OutOfBoundsDatetime(
590 f"{original} is Out of Bounds for origin='julian'"
591 )
592 else:
593 # arg must be numeric
594 if not (
595 (is_integer(arg) or is_float(arg)) or is_numeric_dtype(np.asarray(arg))
596 ):
597 raise ValueError(
598 f"'{arg}' is not compatible with origin='{origin}'; "
599 "it must be numeric with a unit specified"
600 )
601
602 # we are going to offset back to unix / epoch time
603 try:
604 offset = Timestamp(origin, unit=unit)
605 except OutOfBoundsDatetime as err:
606 raise OutOfBoundsDatetime(f"origin {origin} is Out of Bounds") from err
607 except ValueError as err:
608 raise ValueError(
609 f"origin {origin} cannot be converted to a Timestamp"
610 ) from err
611
612 if offset.tz is not None:
613 raise ValueError(f"origin offset {offset} must be tz-naive")
614 td_offset = offset - Timestamp(0)
615
616 # convert the offset to the unit of the arg
617 # this should be lossless in terms of precision
618 ioffset = td_offset // Timedelta(1, unit=unit)
619
620 # scalars & ndarray-like can handle the addition
621 if is_list_like(arg) and not isinstance(arg, (ABCSeries, Index, np.ndarray)):
622 arg = np.asarray(arg)
623 arg = arg + ioffset
624 return arg

703 must be specified in 4-digit format.
704 errors : {'ignore', 'raise', 'coerce'}, default 'raise'
705 - If :const:`'raise'`, then invalid parsing will raise an exception.
706 - If :const:`'coerce'`, then invalid parsing will be set as :const:`NaT`.
707 - If :const:`'ignore'`, then invalid parsing will return the input.
708 dayfirst : bool, default False
709 Specify a date parse order if `arg` is str or is list-like.
710 If :const:`True`, parses dates with the day first, e.g. :const:`"10/11/12"`

712
713 .. warning::
714
715 ``dayfirst=True`` is not strict, but will prefer to parse
716 with day first.
717
718 yearfirst : bool, default False
719 Specify a date parse order if `arg` is str or is list-like.
720

722 :const:`"10/11/12"` is parsed as :const:`2010-11-12`.
723 - If both `dayfirst` and `yearfirst` are :const:`True`, `yearfirst` is
724 preceded (same as :mod:`dateutil`).
725
726 .. warning::
727
728 ``yearfirst=True`` is not strict, but will prefer to parse
729 with year first.
730
731 utc : bool, default False
732 Control timezone-related parsing, localization and conversion.
733
734 - If :const:`True`, the function *always* returns a timezone-aware
735 UTC-localized :class:`Timestamp`, :class:`Series` or
736 :class:`DatetimeIndex`. To do this, timezone-naive inputs are
737 *localized* as UTC, while timezone-aware inputs are *converted* to UTC.
738
739 - If :const:`False` (default), inputs will not be coerced to UTC.
740 Timezone-naive inputs will remain naive, while timezone-aware ones
741 will keep their time offsets. Limitations exist for mixed
742 offsets (typically, daylight savings), see :ref:`Examples
743 <to_datetime_tz_examples>` section for details.

746

756 #time-zone-handling>`_.
757
758 format : str, default None
759 The strftime to parse time, e.g. :const:`"%d/%m/%Y"`. See
760 `strftime documentation

779 - If :const:`False`, allow the `format` to match anywhere in the target
780 string.
781
782 Cannot be used alongside ``format='ISO8601'`` or ``format='mixed'``.
783 unit : str, default 'ns'

785 integer or float number. This will be based off the origin.
786 Example, with ``unit='ms'`` and ``origin='unix'``, this would calculate
787 the number of milliseconds to the unix epoch start.
788 infer_datetime_format : bool, default False
789 If :const:`True` and no `format` is given, attempt to infer the format
790 of the datetime strings based on the first non-NaN element,
791 and if it can be inferred, switch to a faster method of parsing them.
792 In some cases this can increase the parsing speed by ~5-10x.
793
794 .. deprecated:: 2.0.0
795 A strict version of this argument is now the default, passing it has
796 no effect.

798 origin : scalar, default 'unix'

800 of units (defined by `unit`) since this reference date.
801
802 - If :const:`'unix'` (or POSIX) time; origin is set to 1970-01-01.
803 - If :const:`'julian'`, unit must be :const:`'D'`, and origin is set to
804 beginning of Julian Calendar. Julian day number :const:`0` is assigned
805 to the day starting at noon on January 1, 4713 BC.
806 - If Timestamp convertible (Timestamp, dt.datetime, np.datetimt64 or date
807 string), origin is set to Timestamp identified by origin.
808 - If a float or integer, origin is the difference
809 (in units determined by the ``unit`` argument) relative to 1970-01-01.
810 cache : bool, default True
811 If :const:`True`, use a cache of unique, converted dates to apply the
812 datetime conversion. May produce significant speed-up when parsing
813 duplicate date strings, especially ones with timezone offsets. The cache
814 is only used when there are at least 50 values. The presence of
815 out-of-bounds values will render the cache unusable and may slow down
816 parsing.

820 datetime
821 If parsing succeeded.
822 Return type depends on input (types in parenthesis correspond to

824 parsing):

827 - array-like: :class:`DatetimeIndex` (or :class:`Series` with
828 :class:`object` dtype containing :class:`datetime.datetime`)
829 - Series: :class:`Series` of :class:`datetime64` dtype (or
830 :class:`Series` of :class:`object` dtype containing
831 :class:`datetime.datetime`)
832 - DataFrame: :class:`Series` of :class:`datetime64` dtype (or

852 Notes
853 -----
854
855 Many input types are supported, and lead to different output types:
856
857 - **scalars** can be int, float, str, datetime object (from stdlib :mod:`datetime`
858 module or :mod:`numpy`). They are converted to :class:`Timestamp` when
859 possible, otherwise they are converted to :class:`datetime.datetime`.
860 None/NaN/null scalars are converted to :const:`NaT`.
861
862 - **array-like** can contain int, float, str, datetime objects. They are
863 converted to :class:`DatetimeIndex` when possible, otherwise they are
864 converted to :class:`Index` with :class:`object` dtype, containing
865 :class:`datetime.datetime`. None/NaN/null entries are converted to
866 :const:`NaT` in both cases.
867
868 - **Series** are converted to :class:`Series` with :class:`datetime64`
869 dtype when possible, otherwise they are converted to :class:`Series` with
870 :class:`object` dtype, containing :class:`datetime.datetime`. None/NaN/null
871 entries are converted to :const:`NaT` in both cases.
872
873 - **DataFrame/dict-like** are converted to :class:`Series` with
874 :class:`datetime64` dtype. For each row a datetime is created from assembling
875 the various dataframe columns. Column keys can be common abbreviations
876 like ['year', 'month', 'day', 'minute', 'second', 'ms', 'us', 'ns']) or
877 plurals of the same.
878
879 The following causes are responsible for :class:`datetime.datetime` objects
880 being returned (possibly inside an :class:`Index` or a :class:`Series` with
881 :class:`object` dtype) instead of a proper pandas designated type
882 (:class:`Timestamp`, :class:`DatetimeIndex` or :class:`Series`
883 with :class:`datetime64` dtype):
884
885 - when any input element is before :const:`Timestamp.min` or after
886 :const:`Timestamp.max`, see `timestamp limitations
887 <https://pandas.pydata.org/pandas-docs/stable/user_guide/timeseries.html
888 #timeseries-timestamp-limits>`_.
889
890 - when ``utc=False`` (default) and the input is an array-like or
891 :class:`Series` containing mixed naive/aware datetime, or aware with mixed
892 time offsets. Note that this happens in the (quite frequent) situation when
893 the timezone has a daylight savings policy. In that case you may wish to
894 use ``utc=True``.
895
896 Examples
897 --------
898
899 **Handling various input formats**
900
901 Assembling a datetime from multiple columns of a :class:`DataFrame`. The keys
902 can be common abbreviations like ['year', 'month', 'day', 'minute', 'second',
903 'ms', 'us', 'ns']) or plurals of the same
904
905 >>> df = pd.DataFrame({'year': [2015, 2016],
906 ... 'month': [2, 3],
907 ... 'day': [4, 5]})
908 >>> pd.to_datetime(df)
909 0 2015-02-04
910 1 2016-03-05
911 dtype: datetime64[ns]

914
915 >>> pd.to_datetime(1490195805, unit='s')
916 Timestamp('2017-03-22 15:16:45')
917 >>> pd.to_datetime(1490195805433502912, unit='ns')

921 unexpected behavior use a fixed-width exact type.
922

924

926 ... origin=pd.Timestamp('1960-01-01'))
927 DatetimeIndex(['1960-01-02', '1960-01-03', '1960-01-04'],

931
932 :const:`"%f"` will parse all the way up to nanoseconds.

941 in addition to forcing non-dates (or non-parseable dates) to :const:`NaT`.
942

947
948 **Timezones and time offsets**
949
950 The default behaviour (``utc=False``) is as follows:
951
952 - Timezone-naive inputs are converted to timezone-naive :class:`DatetimeIndex`:
953
954 >>> pd.to_datetime(['2018-10-26 12:00:00', '2018-10-26 13:00:15'])
955 DatetimeIndex(['2018-10-26 12:00:00', '2018-10-26 13:00:15'],
956 dtype='datetime64[ns]', freq=None)
957
958 - Timezone-aware inputs *with constant time offset* are converted to
959 timezone-aware :class:`DatetimeIndex`:
960
961 >>> pd.to_datetime(['2018-10-26 12:00 -0500', '2018-10-26 13:00 -0500'])
962 DatetimeIndex(['2018-10-26 12:00:00-05:00', '2018-10-26 13:00:00-05:00'],
963 dtype='datetime64[ns, UTC-05:00]', freq=None)
964
965 - However, timezone-aware inputs *with mixed time offsets* (for example
966 issued from a timezone with daylight savings, such as Europe/Paris)
967 are **not successfully converted** to a :class:`DatetimeIndex`.
968 Parsing datetimes with mixed time zones will show a warning unless
969 `utc=True`. If you specify `utc=False` the warning below will be shown
970 and a simple :class:`Index` containing :class:`datetime.datetime`
971 objects will be returned:
972
973 >>> pd.to_datetime(['2020-10-25 02:00 +0200',
974 ... '2020-10-25 04:00 +0100']) # doctest: +SKIP
975 FutureWarning: In a future version of pandas, parsing datetimes with mixed
976 time zones will raise an error unless `utc=True`. Please specify `utc=True`
977 to opt in to the new behaviour and silence this warning. To create a `Series`
978 with mixed offsets and `object` dtype, please use `apply` and
979 `datetime.datetime.strptime`.
980 Index([2020-10-25 02:00:00+02:00, 2020-10-25 04:00:00+01:00],
981 dtype='object')
982
983 - A mix of timezone-aware and timezone-naive inputs is also converted to
984 a simple :class:`Index` containing :class:`datetime.datetime` objects:
985
986 >>> from datetime import datetime
987 >>> pd.to_datetime(["2020-01-01 01:00:00-01:00",
988 ... datetime(2020, 1, 1, 3, 0)]) # doctest: +SKIP
989 FutureWarning: In a future version of pandas, parsing datetimes with mixed
990 time zones will raise an error unless `utc=True`. Please specify `utc=True`
991 to opt in to the new behaviour and silence this warning. To create a `Series`
992 with mixed offsets and `object` dtype, please use `apply` and
993 `datetime.datetime.strptime`.
994 Index([2020-01-01 01:00:00-01:00, 2020-01-01 03:00:00], dtype='object')
995
996 |
997
998 Setting ``utc=True`` solves most of the above issues:
999
1000 - Timezone-naive inputs are *localized* as UTC
1001
1002 >>> pd.to_datetime(['2018-10-26 12:00', '2018-10-26 13:00'], utc=True)
1003 DatetimeIndex(['2018-10-26 12:00:00+00:00', '2018-10-26 13:00:00+00:00'],
1004 dtype='datetime64[ns, UTC]', freq=None)
1005
1006 - Timezone-aware inputs are *converted* to UTC (the output represents the
1007 exact same datetime, but viewed from the UTC time offset `+00:00`).
1008
1009 >>> pd.to_datetime(['2018-10-26 12:00 -0530', '2018-10-26 12:00 -0500'],
1010 ... utc=True)
1011 DatetimeIndex(['2018-10-26 17:30:00+00:00', '2018-10-26 17:00:00+00:00'],
1012 dtype='datetime64[ns, UTC]', freq=None)
1013
1014 - Inputs can contain both string or datetime, the above
1015 rules still apply
1016
1017 >>> pd.to_datetime(['2018-10-26 12:00', datetime(2020, 1, 1, 18)], utc=True)
1018 DatetimeIndex(['2018-10-26 12:00:00+00:00', '2020-01-01 18:00:00+00:00'],
1019 dtype='datetime64[ns, UTC]', freq=None)

1142def _assemble_from_unit_mappings(arg, errors: DateTimeErrorChoices, utc: bool):
1143 """
1144 assemble the unit specified fields from the arg (DataFrame)
1145 Return a Series for actual parsing
1146
1147 Parameters
1148 ----------
1149 arg : DataFrame
1150 errors : {'ignore', 'raise', 'coerce'}, default 'raise'
1151
1152 - If :const:`'raise'`, then invalid parsing will raise an exception
1153 - If :const:`'coerce'`, then invalid parsing will be set as :const:`NaT`
1154 - If :const:`'ignore'`, then invalid parsing will return the input
1155 utc : bool
1156 Whether to convert/localize timestamps to UTC.
1157
1158 Returns
1159 -------
1160 Series
1161 """
1162 from pandas import (
1163 DataFrame,
1164 to_numeric,
1165 to_timedelta,
1166 )
1167
1168 arg = DataFrame(arg)
1169 if not arg.columns.is_unique:
1170 raise ValueError("cannot assemble with duplicate keys")
1171
1172 # replace passed unit with _unit_map
1173 def f(value):
1174 if value in _unit_map:
1175 return _unit_map[value]
1176
1177 # m is case significant
1178 if value.lower() in _unit_map:
1179 return _unit_map[value.lower()]
1180
1181 return value
1182
1183 unit = {k: f(k) for k in arg.keys()}
1184 unit_rev = {v: k for k, v in unit.items()}
1185
1186 # we require at least Ymd
1187 required = ["year", "month", "day"]
1188 req = sorted(set(required) - set(unit_rev.keys()))
1189 if len(req):
1190 _required = ",".join(req)
1191 raise ValueError(
1192 "to assemble mappings requires at least that "
1193 f"[year, month, day] be specified: [{_required}] is missing"
1194 )
1195
1196 # keys we don't recognize
1197 excess = sorted(set(unit_rev.keys()) - set(_unit_map.values()))
1198 if len(excess):
1199 _excess = ",".join(excess)
1200 raise ValueError(
1201 f"extra keys have been passed to the datetime assemblage: [{_excess}]"
1202 )
1203
1204 def coerce(values):
1205 # we allow coercion to if errors allows
1206 values = to_numeric(values, errors=errors)
1207
1208 # prevent overflow in case of int8 or int16
1209 if is_integer_dtype(values.dtype):
1210 values = values.astype("int64", copy=False)
1211 return values
1212
1213 values = (
1214 coerce(arg[unit_rev["year"]]) * 10000
1215 + coerce(arg[unit_rev["month"]]) * 100
1216 + coerce(arg[unit_rev["day"]])
1217 )
1218 try:
1219 values = to_datetime(values, format="%Y%m%d", errors=errors, utc=utc)
1220 except (TypeError, ValueError) as err:
1221 raise ValueError(f"cannot assemble the datetimes: {err}") from err
1222
1223 units: list[UnitChoices] = ["h", "m", "s", "ms", "us", "ns"]
1224 for u in units:
1225 value = unit_rev.get(u)
1226 if value is not None and value in arg:
1227 try:
1228 values += to_timedelta(coerce(arg[value]), unit=u, errors=errors)
1229 except (TypeError, ValueError) as err:
1230 raise ValueError(
1231 f"cannot assemble the datetimes [{value}]: {err}"
1232 ) from err
1233 return values
