31def test_symlog_mask_nan():
32 # Use a transform round-trip to verify that the forward and inverse
33 # transforms work, and that they respect nans and/or masking.
34 slt = SymmetricalLogTransform(10, 2, 1)
35 slti = slt.inverted()
36
37 x = np.arange(-1.5, 5, 0.5)
38 out = slti.transform_non_affine(slt.transform_non_affine(x))
39 assert_allclose(out, x)
40 assert type(out) is type(x)
41
42 x[4] = np.nan
43 out = slti.transform_non_affine(slt.transform_non_affine(x))
44 assert_allclose(out, x)
45 assert type(out) is type(x)
46
47 x = np.ma.array(x)
48 out = slti.transform_non_affine(slt.transform_non_affine(x))
