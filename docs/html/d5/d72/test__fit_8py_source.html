96@pytest.mark.slow

98@pytest.mark.parametrize('method', ["MLE", "MM"])
99def test_cont_fit(distname, arg, method):
100 if distname in failing_fits[method]:
101 # Skip failing fits unless overridden
102 try:
103 xfail = not int(os.environ['SCIPY_XFAIL'])
104 except Exception:
105 xfail = True
106 if xfail:
107 msg = "Fitting %s doesn't work reliably yet" % distname
108 msg += (" [Set environment variable SCIPY_XFAIL=1 to run this"
109 " test nevertheless.]")
110 pytest.xfail(msg)
111
112 distfn = getattr(stats, distname)
113
114 truearg = np.hstack([arg, [0.0, 1.0]])
115 diffthreshold = np.max(np.vstack([truearg*thresh_percent,
116 np.full(distfn.numargs+2, thresh_min)]),
117 0)
118
119 for fit_size in fit_sizes:
120 # Note that if a fit succeeds, the other fit_sizes are skipped
121 np.random.seed(1234)
122
123 with np.errstate(all='ignore'):
124 rvs = distfn.rvs(size=fit_size, *arg)
125 if method == 'MLE' and distfn.name in mle_use_floc0:
126 kwds = {'floc': 0}
127 else:
128 kwds = {}
129 # start with default values
130 est = distfn.fit(rvs, method=method, **kwds)
131 if method == 'MLE':
132 # Trivial test of the use of CensoredData. The fit() method
133 # will check that data contains no actual censored data, and
134 # do a regular uncensored fit.
135 data1 = stats.CensoredData(rvs)
136 est1 = distfn.fit(data1, **kwds)
137 msg = ('Different results fitting uncensored data wrapped as'
138 f' CensoredData: {distfn.name}: est={est} est1={est1}')
139 assert_allclose(est1, est, rtol=1e-10, err_msg=msg)
140 if method == 'MLE' and distname not in fail_interval_censored:
141 # Convert the first `nic` values in rvs to interval-censored
142 # values. The interval is small, so est2 should be close to
143 # est.
144 nic = 15
145 interval = np.column_stack((rvs, rvs))
146 interval[:nic, 0] *= 0.99
147 interval[:nic, 1] *= 1.01
148 interval.sort(axis=1)
149 data2 = stats.CensoredData(interval=interval)
150 est2 = distfn.fit(data2, **kwds)
151 msg = ('Different results fitting interval-censored'
152 f' data: {distfn.name}: est={est} est2={est2}')
153 assert_allclose(est2, est, rtol=0.05, err_msg=msg)
154
155 diff = est - truearg
156
157 # threshold for location
158 diffthreshold[-2] = np.max([np.abs(rvs.mean())*thresh_percent,
159 thresh_min])
160
161 if np.any(np.isnan(est)):
162 raise AssertionError('nan returned in fit')
163 else:
164 if np.all(np.abs(diff) <= diffthreshold):
165 break
166 else:
167 txt = 'parameter: %s\n' % str(truearg)
168 txt += 'estimated: %s\n' % str(est)
169 txt += 'diff : %s\n' % str(diff)
170 raise AssertionError('fit not very good in %s\n' % distfn.name + txt)

483 def basic_fit_test(self, dist_name, method):
484
485 N = 5000
486 dist_data = dict(distcont + distdiscrete)
487 rng = np.random.default_rng(self.seed)
488 dist = getattr(stats, dist_name)
489 shapes = np.array(dist_data[dist_name])
490 bounds = np.empty((len(shapes) + 2, 2), dtype=np.float64)
491 bounds[:-2, 0] = shapes/10.**np.sign(shapes)
492 bounds[:-2, 1] = shapes*10.**np.sign(shapes)
493 bounds[-2] = (0, 10)
494 bounds[-1] = (1e-16, 10)
495 loc = rng.uniform(*bounds[-2])
496 scale = rng.uniform(*bounds[-1])
497 ref = list(dist_data[dist_name]) + [loc, scale]
498
499 if getattr(dist, 'pmf', False):
500 ref = ref[:-1]
501 ref[-1] = np.floor(loc)
502 data = dist.rvs(*ref, size=N, random_state=rng)
503 bounds = bounds[:-1]
504 if getattr(dist, 'pdf', False):
505 data = dist.rvs(*ref, size=N, random_state=rng)
506
507 with npt.suppress_warnings() as sup:
508 sup.filter(RuntimeWarning, "overflow encountered")
509 res = stats.fit(dist, data, bounds, method=method,
510 optimizer=self.opt)
511
512 nlff_names = {'mle': 'nnlf', 'mse': '_penalized_nlpsf'}
513 nlff_name = nlff_names[method]
514 assert_nlff_less_or_close(dist, data, res.params, ref, **self.tols,
515 nlff_name=nlff_name)
