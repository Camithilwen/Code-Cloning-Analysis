13def check_czt(x):
14 # Check that czt is the equivalent of normal fft
15 y = fft(x)
16 y1 = czt(x)

25def check_zoom_fft(x):
26 # Check that zoom_fft is the equivalent of normal fft
27 y = fft(x)
28 y1 = zoom_fft(x, [0, 2-2./len(y)], endpoint=True)
29 assert_allclose(y1, y, rtol=1e-11, atol=1e-14)
30 y1 = zoom_fft(x, [0, 2])
31 assert_allclose(y1, y, rtol=1e-11, atol=1e-14)
32
33 # Test fn scalar
34 y1 = zoom_fft(x, 2-2./len(y), endpoint=True)
35 assert_allclose(y1, y, rtol=1e-11, atol=1e-14)
36 y1 = zoom_fft(x, 2)
37 assert_allclose(y1, y, rtol=1e-11, atol=1e-14)
38
39 # Check that zoom_fft with oversampling is equivalent to zero padding
40 over = 10
41 yover = fft(x, over*len(x))
42 y2 = zoom_fft(x, [0, 2-2./len(yover)], m=len(yover), endpoint=True)
43 assert_allclose(y2, yover, rtol=1e-12, atol=1e-10)
44 y2 = zoom_fft(x, [0, 2], m=len(yover))
45 assert_allclose(y2, yover, rtol=1e-12, atol=1e-10)
46
47 # Check that zoom_fft works on a subrange
48 w = np.linspace(0, 2-2./len(x), len(x))
49 f1, f2 = w[3], w[6]
50 y3 = zoom_fft(x, [f1, f2], m=3*over+1, endpoint=True)
51 idx3 = slice(3*over, 6*over+1)
52 assert_allclose(y3, yover[idx3], rtol=1e-13)

152@pytest.mark.parametrize('w', (None, 0.98534 + 0.17055j))
153def test_czt_math(impulse, m, w, a):
154 # z-transform of an impulse is 1 everywhere
155 assert_allclose(czt(impulse[2:], m=m, w=w, a=a),
156 np.ones(m), rtol=1e-10)
157
158 # z-transform of a delayed impulse is z**-1
159 assert_allclose(czt(impulse[1:], m=m, w=w, a=a),
160 czt_points(m=m, w=w, a=a)**-1, rtol=1e-10)
161
162 # z-transform of a 2-delayed impulse is z**-2
163 assert_allclose(czt(impulse, m=m, w=w, a=a),
