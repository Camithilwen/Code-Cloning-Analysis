5def _check_feature_names(X, feature_names=None):
6 """Check feature names.
7
8 Parameters
9 ----------
10 X : array-like of shape (n_samples, n_features)
11 Input data.
12
13 feature_names : None or array-like of shape (n_names,), dtype=str
14 Feature names to check or `None`.
15
16 Returns
17 -------
18 feature_names : list of str
19 Feature names validated. If `feature_names` is `None`, then a list of
20 feature names is provided, i.e. the column names of a pandas dataframe
21 or a generic list of feature names (e.g. `["x0", "x1", ...]`) for a
22 NumPy array.
23 """
24 if feature_names is None:
25 if hasattr(X, "columns") and hasattr(X.columns, "tolist"):
26 # get the column names for a pandas dataframe
27 feature_names = X.columns.tolist()
28 else:
29 # define a list of numbered indices for a numpy array
30 feature_names = [f"x{i}" for i in range(X.shape[1])]
31 elif hasattr(feature_names, "tolist"):
32 # convert numpy array or pandas index to a list
33 feature_names = feature_names.tolist()
34 if len(set(feature_names)) != len(feature_names):
35 raise ValueError("feature_names should not contain duplicates.")
36
37 return feature_names
