51class Kernel(BuiltinFilter):
52 """
53 Create a convolution kernel. This only supports 3x3 and 5x5 integer and floating
54 point kernels.
55
56 Kernels can only be applied to "L" and "RGB" images.
57
58 :param size: Kernel size, given as (width, height). This must be (3,3) or (5,5).
59 :param kernel: A sequence containing kernel weights. The kernel will be flipped
60 vertically before being applied to the image.
61 :param scale: Scale factor. If given, the result for each pixel is divided by this
62 value. The default is the sum of the kernel weights.
63 :param offset: Offset. If given, this value is added to the result, after it has
64 been divided by the scale factor.
65 """
66
67 name = "Kernel"
68
69 def __init__(
70 self,
71 size: tuple[int, int],
72 kernel: Sequence[float],
73 scale: float | None = None,
74 offset: float = 0,
75 ) -> None:
76 if scale is None:
77 # default scale is sum of kernel
78 scale = functools.reduce(lambda a, b: a + b, kernel)
79 if size[0] * size[1] != len(kernel):
80 msg = "not enough coefficients in kernel"
81 raise ValueError(msg)
82 self.filterargs = size, scale, offset, kernel

86 """
87 Create a rank filter. The rank filter sorts all pixels in

89
90 :param size: The kernel size, in pixels.

173class GaussianBlur(MultibandFilter):
174 """Blurs the image with a sequence of extended box filters, which
175 approximates a Gaussian kernel. For details on accuracy see
176 <https://www.mia.uni-saarland.de/Publications/gwosdek-ssvm11.pdf>
177
178 :param radius: Standard deviation of the Gaussian kernel. Either a sequence of two
179 numbers for x and y, or a single number for both.
180 """
181
182 name = "GaussianBlur"
183
184 def __init__(self, radius: float | Sequence[float] = 2) -> None:
185 self.radius = radius
186
187 def filter(self, image: _imaging.ImagingCore) -> _imaging.ImagingCore:
188 xy = self.radius
189 if isinstance(xy, (int, float)):
190 xy = (xy, xy)
191 if xy == (0, 0):
192 return image.copy()
193 return image.gaussian_blur(xy)

196class BoxBlur(MultibandFilter):
197 """Blurs the image by setting each pixel to the average value of the pixels
198 in a square box extending radius pixels in each direction.
199 Supports float radius of arbitrary size. Uses an optimized implementation
200 which runs in linear time relative to the size of the image
201 for any radius value.
202
203 :param radius: Size of the box in a direction. Either a sequence of two numbers for
204 x and y, or a single number for both.
205
206 Radius 0 does not blur, returns an identical image.
207 Radius 1 takes 1 pixel in each direction, i.e. 9 pixels in total.
208 """
209
210 name = "BoxBlur"
211
212 def __init__(self, radius: float | Sequence[float]) -> None:
213 xy = radius if isinstance(radius, (tuple, list)) else (radius, radius)
214 if xy[0] < 0 or xy[1] < 0:
215 msg = "radius must be >= 0"
216 raise ValueError(msg)
217 self.radius = radius
218
219 def filter(self, image: _imaging.ImagingCore) -> _imaging.ImagingCore:
220 xy = self.radius
221 if isinstance(xy, (int, float)):
222 xy = (xy, xy)
223 if xy == (0, 0):
224 return image.copy()
225 return image.box_blur(xy)

370class Color3DLUT(MultibandFilter):
371 """Three-dimensional color lookup table.
372
373 Transforms 3-channel pixels using the values of the channels as coordinates
374 in the 3D lookup table and interpolating the nearest elements.
375
376 This method allows you to apply almost any color transformation
377 in constant time by using pre-calculated decimated tables.
378
379 .. versionadded:: 5.2.0
380
381 :param size: Size of the table. One int or tuple of (int, int, int).
382 Minimal size in any dimension is 2, maximum is 65.
383 :param table: Flat lookup table. A list of ``channels * size**3``
384 float elements or a list of ``size**3`` channels-sized
385 tuples with floats. Channels are changed first,
386 then first dimension, then second, then third.
387 Value 0.0 corresponds lowest value of output, 1.0 highest.
388 :param channels: Number of channels in the table. Could be 3 or 4.
389 Default is 3.
390 :param target_mode: A mode for the result image. Should have not less
391 than ``channels`` channels. Default is ``None``,
392 which means that mode wouldn't be changed.
393 """
394
395 name = "Color 3D LUT"

407 raise ValueError(msg)

409 self.channels = channels
410 self.mode = target_mode
411

413 # of the table if the table is specially made for the constructor.
414 copy_table = kwargs.get("_copy_table", True)

416 wrong_size = False

467 @staticmethod
468 def _check_size(size: Any) -> tuple[int, int, int]:
469 try:
470 _, _, _ = size
471 except ValueError as e:
472 msg = "Size should be either an integer or a tuple of three integers."
473 raise ValueError(msg) from e
474 except TypeError:

483 @classmethod
484 def generate(
485 cls,
486 size: int | tuple[int, int, int],
487 callback: Callable[[float, float, float], tuple[float, ...]],
488 channels: int = 3,
489 target_mode: str | None = None,
490 ) -> Color3DLUT:
491 """Generates new LUT using provided callback.
492
493 :param size: Size of the table. Passed to the constructor.
494 :param callback: Function with three parameters which correspond
495 three color channels. Will be called ``size**3``
496 times with values from 0.0 to 1.0 and should return
497 a tuple with ``channels`` elements.
498 :param channels: The number of channels which should return callback.
499 :param target_mode: Passed to the constructor of the resulting
500 lookup table.
501 """
502 size_1d, size_2d, size_3d = cls._check_size(size)
503 if channels not in (3, 4):
504 msg = "Only 3 or 4 output channels are supported"
505 raise ValueError(msg)
506
507 table: list[float] = [0] * (size_1d * size_2d * size_3d * channels)
508 idx_out = 0
509 for b in range(size_3d):
510 for g in range(size_2d):
511 for r in range(size_1d):
512 table[idx_out : idx_out + channels] = callback(
513 r / (size_1d - 1), g / (size_2d - 1), b / (size_3d - 1)
514 )
515 idx_out += channels
516
517 return cls(
518 (size_1d, size_2d, size_3d),
519 table,
520 channels=channels,
521 target_mode=target_mode,
522 _copy_table=False,
523 )
