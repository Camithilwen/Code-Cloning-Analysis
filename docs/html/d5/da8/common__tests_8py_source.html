153def check_named_args(distfn, x, shape_args, defaults, meths):
154
155
156 # check consistency of shapes, numargs and _parse signature
157 signature = _getfullargspec(distfn._parse_args)
158 npt.assert_(signature.varargs is None)
159 npt.assert_(signature.varkw is None)
160 npt.assert_(not signature.kwonlyargs)
161 npt.assert_(list(signature.defaults) == list(defaults))
162
163 shape_argnames = signature.args[:-len(defaults)] # a, b, loc=0, scale=1
164 if distfn.shapes:
165 shapes_ = distfn.shapes.replace(',', ' ').split()
166 else:
167 shapes_ = ''
168 npt.assert_(len(shapes_) == distfn.numargs)
169 npt.assert_(len(shapes_) == len(shape_argnames))
170
171 # check calling w/ named arguments
172 shape_args = list(shape_args)
173
174 vals = [meth(x, *shape_args) for meth in meths]
175 npt.assert_(np.all(np.isfinite(vals)))
176
177 names, a, k = shape_argnames[:], shape_args[:], {}
178 while names:
179 k.update({names.pop(): a.pop()})
180 v = [meth(x, *a, **k) for meth in meths]
181 npt.assert_array_equal(vals, v)
182 if 'n' not in k.keys():
183 # `n` is first parameter of moment(), so can't be used as named arg
184 npt.assert_equal(distfn.moment(1, *a, **k),
185 distfn.moment(1, *shape_args))

192def check_random_state_property(distfn, args):
193 # check the random_state attribute of a distribution *instance*
194
195 # This test fiddles with distfn.random_state. This breaks other tests,
196 # hence need to save it and then restore.
197 rndm = distfn.random_state
198
199 # baseline: this relies on the global state
200 np.random.seed(1234)
201 distfn.random_state = None
202 r0 = distfn.rvs(*args, size=8)
203
204 # use an explicit instance-level random_state
205 distfn.random_state = 1234
206 r1 = distfn.rvs(*args, size=8)
207 npt.assert_equal(r0, r1)
208
209 distfn.random_state = np.random.RandomState(1234)
210 r2 = distfn.rvs(*args, size=8)
211 npt.assert_equal(r0, r2)
212
213 # check that np.random.Generator can be used (numpy >= 1.17)
214 if hasattr(np.random, 'default_rng'):
215 # obtain a np.random.Generator object
216 rng = np.random.default_rng(1234)
217 distfn.rvs(*args, size=1, random_state=rng)
218
219 # can override the instance-level random_state for an individual .rvs call
220 distfn.random_state = 2
221 orig_state = distfn.random_state.get_state()
222
223 r3 = distfn.rvs(*args, size=8, random_state=np.random.RandomState(1234))
224 npt.assert_equal(r0, r3)
225
226 # ... and that does not alter the instance-level random_state!
227 npt.assert_equal(distfn.random_state.get_state(), orig_state)

257def check_cmplx_deriv(distfn, arg):
258 # Distributions allow complex arguments.
259 def deriv(f, x, *arg):
260 x = np.asarray(x)
261 h = 1e-10
262 return (f(x + h*1j, *arg)/h).imag
263
264 x0 = distfn.ppf([0.25, 0.51, 0.75], *arg)
265 x_cast = [x0.astype(tp) for tp in (np.int_, np.float16, np.float32,
266 np.float64)]
267
268 for x in x_cast:
269 # casting may have clipped the values, exclude those
270 distfn._argcheck(*arg)
271 x = x[(distfn.a < x) & (x < distfn.b)]
272
273 pdf, cdf, sf = distfn.pdf(x, *arg), distfn.cdf(x, *arg), distfn.sf(x, *arg)
274 assert_allclose(deriv(distfn.cdf, x, *arg), pdf, rtol=1e-5)
275 assert_allclose(deriv(distfn.logcdf, x, *arg), pdf/cdf, rtol=1e-5)
276
277 assert_allclose(deriv(distfn.sf, x, *arg), -pdf, rtol=1e-5)
278 assert_allclose(deriv(distfn.logsf, x, *arg), -pdf/sf, rtol=1e-5)
279
280 assert_allclose(deriv(distfn.logpdf, x, *arg),
281 deriv(distfn.pdf, x, *arg) / distfn.pdf(x, *arg),
282 rtol=1e-5)
