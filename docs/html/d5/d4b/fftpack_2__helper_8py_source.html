8def rfftfreq(n, d=1.0):
9 """DFT sample frequencies (for usage with rfft, irfft).
10
11 The returned float array contains the frequency bins in
12 cycles/unit (with zero at the start) given a window length `n` and a
13 sample spacing `d`::
14
15 f = [0,1,1,2,2,...,n/2-1,n/2-1,n/2]/(d*n) if n is even
16 f = [0,1,1,2,2,...,n/2-1,n/2-1,n/2,n/2]/(d*n) if n is odd
17
18 Parameters
19 ----------
20 n : int
21 Window length.
22 d : scalar, optional
23 Sample spacing. Default is 1.
24
25 Returns
26 -------
27 out : ndarray
28 The array of length `n`, containing the sample frequencies.
29
30 Examples
31 --------
32 >>> import numpy as np
33 >>> from scipy import fftpack
34 >>> sig = np.array([-2, 8, 6, 4, 1, 0, 3, 5], dtype=float)
35 >>> sig_fft = fftpack.rfft(sig)
36 >>> n = sig_fft.size
37 >>> timestep = 0.1
38 >>> freq = fftpack.rfftfreq(n, d=timestep)
39 >>> freq
40 array([ 0. , 1.25, 1.25, 2.5 , 2.5 , 3.75, 3.75, 5. ])
41
42 """
43 n = operator.index(n)
44 if n < 0:
45 raise ValueError("n = %s is not valid. "
46 "n must be a nonnegative integer." % n)
47
48 return (np.arange(1, n + 1, dtype=int) // 2) / float(n * d)

51def next_fast_len(target):
52 """
53 Find the next fast size of input data to `fft`, for zero-padding, etc.
54
55 SciPy's FFTPACK has efficient functions for radix {2, 3, 4, 5}, so this
56 returns the next composite of the prime factors 2, 3, and 5 which is
57 greater than or equal to `target`. (These are also known as 5-smooth
58 numbers, regular numbers, or Hamming numbers.)
59
60 Parameters
61 ----------
62 target : int
63 Length to start searching from. Must be a positive integer.
64
65 Returns
66 -------
67 out : int
68 The first 5-smooth number greater than or equal to `target`.
69
70 Notes
71 -----
72 .. versionadded:: 0.18.0
73
74 Examples
75 --------
76 On a particular machine, an FFT of prime length takes 133 ms:
77
78 >>> from scipy import fftpack
79 >>> import numpy as np
80 >>> rng = np.random.default_rng()
81 >>> min_len = 10007 # prime length is worst case for speed
82 >>> a = rng.standard_normal(min_len)
83 >>> b = fftpack.fft(a)
84
85 Zero-padding to the next 5-smooth length reduces computation time to
86 211 us, a speedup of 630 times:
87
88 >>> fftpack.next_fast_len(min_len)
89 10125
90 >>> b = fftpack.fft(a, 10125)
91
92 Rounding up to the next power of 2 is not optimal, taking 367 us to
93 compute, 1.7 times as long as the 5-smooth size:
94
95 >>> b = fftpack.fft(a, 16384)
96
97 """
98 # Real transforms use regular sizes so this is backwards compatible
99 return _helper.good_size(target, True)
