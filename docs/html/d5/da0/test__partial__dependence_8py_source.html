92def test_output_shape(
93 Estimator, method, data, grid_resolution, features, kind, use_custom_values
94):
95 # Check that partial_dependence has consistent output shape for different
96 # kinds of estimators:
97 # - classifiers with binary and multiclass settings
98 # - regressors
99 # - multi-task regressors
100
101 est = Estimator()
102 if hasattr(est, "n_estimators"):
103 est.set_params(n_estimators=2) # speed-up computations
104
105 # n_target corresponds to the number of classes (1 for binary classif) or
106 # the number of tasks / outputs in multi task settings. It's equal to 1 for
107 # classical regression_data.
108 (X, y), n_targets = data
109 n_instances = X.shape[0]
110
111 custom_values = None
112 if use_custom_values:
113 grid_resolution = 5
114 custom_values = {f: X[:grid_resolution, f] for f in features}
115
116 est.fit(X, y)
117 result = partial_dependence(
118 est,
119 X=X,
120 features=features,
121 method=method,
122 kind=kind,
123 grid_resolution=grid_resolution,
124 custom_values=custom_values,
125 )
126 pdp, axes = result, result["grid_values"]
127
128 expected_pdp_shape = (n_targets, *[grid_resolution for _ in range(len(features))])
129 expected_ice_shape = (
130 n_targets,
131 n_instances,
132 *[grid_resolution for _ in range(len(features))],
133 )
134 if kind == "average":
135 assert pdp.average.shape == expected_pdp_shape
136 elif kind == "individual":
137 assert pdp.individual.shape == expected_ice_shape
138 else: # 'both'
139 assert pdp.average.shape == expected_pdp_shape
140 assert pdp.individual.shape == expected_ice_shape
141
142 expected_axes_shape = (len(features), grid_resolution)

147def test_grid_from_X():
148 # tests for _grid_from_X: sanity check for output, and for shapes.
149
150 # Make sure that the grid is a cartesian product of the input (it will use
151 # the unique values instead of the percentiles)
152 percentiles = (0.05, 0.95)
153 grid_resolution = 100
154 is_categorical = [False, False]
155 X = np.asarray([[1, 2], [3, 4]])
156 grid, axes = _grid_from_X(X, percentiles, is_categorical, grid_resolution, {})
157 assert_array_equal(grid, [[1, 2], [1, 4], [3, 2], [3, 4]])
158 assert_array_equal(axes, X.T)
159
160 # test shapes of returned objects depending on the number of unique values
161 # for a feature.
162 rng = np.random.RandomState(0)
163 grid_resolution = 15
164
165 # n_unique_values > grid_resolution
166 X = rng.normal(size=(20, 2))
167 grid, axes = _grid_from_X(
168 X,
169 percentiles,
170 is_categorical,
171 grid_resolution=grid_resolution,
172 custom_values={},
173 )
174 assert grid.shape == (grid_resolution * grid_resolution, X.shape[1])
175 assert np.asarray(axes).shape == (2, grid_resolution)
176 assert grid.dtype == X.dtype
177
178 # n_unique_values < grid_resolution, will use actual values
179 n_unique_values = 12
180 X[n_unique_values - 1 :, 0] = 12345
181 rng.shuffle(X) # just to make sure the order is irrelevant
182 grid, axes = _grid_from_X(
183 X,
184 percentiles,
185 is_categorical,
186 grid_resolution=grid_resolution,
187 custom_values={},
188 )
189 assert grid.shape == (n_unique_values * grid_resolution, X.shape[1])
190 # axes is a list of arrays of different shapes
191 assert axes[0].shape == (n_unique_values,)
192 assert axes[1].shape == (grid_resolution,)
193 assert grid.dtype == X.dtype
194
195 # Check that uses custom_range
196 X = rng.normal(size=(20, 2))
197 X[n_unique_values - 1 :, 0] = 12345
198 col_1_range = [0, 2, 3]
199 grid, axes = _grid_from_X(
200 X,
201 percentiles,
202 is_categorical=is_categorical,
203 grid_resolution=grid_resolution,
204 custom_values={1: col_1_range},
205 )
206 assert grid.shape == (n_unique_values * len(col_1_range), X.shape[1])
207 # axes is a list of arrays of different shapes
208 assert axes[0].shape == (n_unique_values,)
209 assert axes[1].shape == (len(col_1_range),)
210 assert grid.dtype == X.dtype
211
212 # Check that grid_resolution does not impact custom_range
213 X = rng.normal(size=(20, 2))
214 col_0_range = [0, 2, 3, 4, 5, 6]
215 grid_resolution = 5
216 grid, axes = _grid_from_X(
217 X,
218 percentiles,
219 is_categorical=is_categorical,
220 grid_resolution=grid_resolution,
221 custom_values={0: col_0_range},
222 )
223 assert grid.shape == (grid_resolution * len(col_0_range), X.shape[1])
224 # axes is a list of arrays of different shapes
225 assert axes[0].shape == (len(col_0_range),)
226 assert axes[1].shape == (grid_resolution,)
227 assert grid.dtype == np.result_type(X, np.asarray(col_0_range).dtype)
228
229 X = np.array([[0, "a"], [1, "b"], [2, "c"]])
230
231 grid, axes = _grid_from_X(
232 X,
233 percentiles,
234 is_categorical=is_categorical,
235 grid_resolution=grid_resolution,
236 custom_values={1: ["a", "b", "c"]},

319@pytest.mark.parametrize("target_feature", range(5))
320@pytest.mark.parametrize(
321 "est, method",
322 [
323 (LinearRegression(), "brute"),
324 (GradientBoostingRegressor(random_state=0), "brute"),
325 (GradientBoostingRegressor(random_state=0), "recursion"),

327 (HistGradientBoostingRegressor(random_state=0), "recursion"),

330def test_partial_dependence_helpers(est, method, target_feature):
331 # Check that what is returned by _partial_dependence_brute or
332 # _partial_dependence_recursion is equivalent to manually setting a target
333 # feature to a given value, and computing the average prediction over all
334 # samples.
335 # This also checks that the brute and recursion methods give the same
336 # output.
337 # Note that even on the trainset, the brute and the recursion methods
338 # aren't always strictly equivalent, in particular when the slow method
339 # generates unrealistic samples that have low mass in the joint
340 # distribution of the input features, and when some of the features are
341 # dependent. Hence the high tolerance on the checks.
342
343 X, y = make_regression(random_state=0, n_features=5, n_informative=5)
344 # The 'init' estimator for GBDT (here the average prediction) isn't taken
345 # into account with the recursion method, for technical reasons. We set
346 # the mean to 0 to that this 'bug' doesn't have any effect.
347 y = y - y.mean()
348
349 # Clone is necessary to make the test thread-safe.
350 est = clone(est).fit(X, y)
351
352 # target feature will be set to .5 and then to 123
353 features = np.array([target_feature], dtype=np.intp)
354 grid = np.array([[0.5], [123]])
355
356 if method == "brute":
357 pdp, predictions = _partial_dependence_brute(
358 est, grid, features, X, response_method="auto"
359 )
360 else:
361 pdp = _partial_dependence_recursion(est, grid, features)
362
363 mean_predictions = []
364 for val in (0.5, 123):
365 X_ = X.copy()
366 X_[:, target_feature] = val
367 mean_predictions.append(est.predict(X_).mean())
368
369 pdp = pdp[0] # (shape is (1, 2) so make it (2,))
370
371 # allow for greater margin for error with recursion method
372 rtol = 1e-1 if method == "recursion" else 1e-3
373 assert np.allclose(pdp, mean_predictions, rtol=rtol)

376@pytest.mark.parametrize("seed", range(1))
377def test_recursion_decision_tree_vs_forest_and_gbdt(seed):
378 # Make sure that the recursion method gives the same results on a
379 # DecisionTreeRegressor and a GradientBoostingRegressor or a
380 # RandomForestRegressor with 1 tree and equivalent parameters.
381
382 rng = np.random.RandomState(seed)
383
384 # Purely random dataset to avoid correlated features
385 n_samples = 1000
386 n_features = 5
387 X = rng.randn(n_samples, n_features)
388 y = rng.randn(n_samples) * 10
389
390 # The 'init' estimator for GBDT (here the average prediction) isn't taken
391 # into account with the recursion method, for technical reasons. We set
392 # the mean to 0 to that this 'bug' doesn't have any effect.
393 y = y - y.mean()
394
395 # set max_depth not too high to avoid splits with same gain but different
396 # features
397 max_depth = 5
398
399 tree_seed = 0
400 forest = RandomForestRegressor(
401 n_estimators=1,
402 max_features=None,
403 bootstrap=False,
404 max_depth=max_depth,
405 random_state=tree_seed,
406 )
407 # The forest will use ensemble.base._set_random_states to set the
408 # random_state of the tree sub-estimator. We simulate this here to have
409 # equivalent estimators.
410 equiv_random_state = check_random_state(tree_seed).randint(np.iinfo(np.int32).max)
411 gbdt = GradientBoostingRegressor(
412 n_estimators=1,
413 learning_rate=1,
414 criterion="squared_error",
415 max_depth=max_depth,
416 random_state=equiv_random_state,
417 )
418 tree = DecisionTreeRegressor(max_depth=max_depth, random_state=equiv_random_state)
419
420 forest.fit(X, y)
421 gbdt.fit(X, y)
422 tree.fit(X, y)
423
424 # sanity check: if the trees aren't the same, the PD values won't be equal
425 try:
426 assert_is_subtree(tree.tree_, gbdt[0, 0].tree_)
427 assert_is_subtree(tree.tree_, forest[0].tree_)
428 except AssertionError:
429 # For some reason the trees aren't exactly equal on 32bits, so the PDs
430 # cannot be equal either. See
431 # https://github.com/scikit-learn/scikit-learn/issues/8853
432 assert _IS_32BIT, "this should only fail on 32 bit platforms"
433 return
434
435 grid = rng.randn(50).reshape(-1, 1)
436 for f in range(n_features):
437 features = np.array([f], dtype=np.intp)
438
439 pdp_forest = _partial_dependence_recursion(forest, grid, features)
440 pdp_gbdt = _partial_dependence_recursion(gbdt, grid, features)
441 pdp_tree = _partial_dependence_recursion(tree, grid, features)
442
443 np.testing.assert_allclose(pdp_gbdt, pdp_tree)
444 np.testing.assert_allclose(pdp_forest, pdp_tree)
