81def _convert_datetimes(sas_datetimes: pd.Series, unit: str) -> pd.Series:
82 """
83 Convert to Timestamp if possible, otherwise to datetime.datetime.
84 SAS float64 lacks precision for more than ms resolution so the fit
85 to datetime.datetime is ok.
86
87 Parameters
88 ----------
89 sas_datetimes : {Series, Sequence[float]}
90 Dates or datetimes in SAS
91 unit : {'d', 's'}
92 "d" if the floats represent dates, "s" for datetimes
93
94 Returns
95 -------
96 Series
97 Series of datetime64 dtype or datetime.datetime.
98 """
99 td = (_sas_origin - _unix_origin).as_unit("s")
100 if unit == "s":
101 millis = cast_from_unit_vectorized(
102 sas_datetimes._values, unit="s", out_unit="ms"
103 )
104 dt64ms = millis.view("M8[ms]") + td
105 return pd.Series(dt64ms, index=sas_datetimes.index, copy=False)
106 else:
107 vals = np.array(sas_datetimes, dtype="M8[D]") + td
108 return pd.Series(vals, dtype="M8[s]", index=sas_datetimes.index, copy=False)

141
142 Parameters

144 path_or_buf : path name or buffer

148 Column to use as index.

151 some rarely used SAS date formats may be unsupported.
152 blank_missing : bool, defaults to True

253 def _get_properties(self) -> None:
254 # Check magic number
255 self._path_or_buf.seek(0)
256 self._cached_page = self._path_or_buf.read(288)
257 if self._cached_page[0 : len(const.magic)] != const.magic:
258 raise ValueError("magic number mismatch (not a SAS file?)")
259
260 # Get alignment information
261 buf = self._read_bytes(const.align_1_offset, const.align_1_length)
262 if buf == const.u64_byte_checker_value:
263 self.U64 = True
264 self._int_length = 8
265 self._page_bit_offset = const.page_bit_offset_x64
266 self._subheader_pointer_length = const.subheader_pointer_length_x64
267 else:
268 self.U64 = False
269 self._page_bit_offset = const.page_bit_offset_x86
270 self._subheader_pointer_length = const.subheader_pointer_length_x86
271 self._int_length = 4
272 buf = self._read_bytes(const.align_2_offset, const.align_2_length)
273 if buf == const.align_1_checker_value:
274 align1 = const.align_2_value
275 else:
276 align1 = 0
277
278 # Get endianness information
279 buf = self._read_bytes(const.endianness_offset, const.endianness_length)
280 if buf == b"\x01":
281 self.byte_order = "<"
282 self.need_byteswap = sys.byteorder == "big"
283 else:
284 self.byte_order = ">"
285 self.need_byteswap = sys.byteorder == "little"
286
287 # Get encoding information
288 buf = self._read_bytes(const.encoding_offset, const.encoding_length)[0]
289 if buf in const.encoding_names:
290 self.inferred_encoding = const.encoding_names[buf]
291 if self.encoding == "infer":
292 self.encoding = self.inferred_encoding
293 else:
294 self.inferred_encoding = f"unknown (code={buf})"
295
296 # Timestamp is epoch 01/01/1960
297 epoch = datetime(1960, 1, 1)
298 x = self._read_float(
299 const.date_created_offset + align1, const.date_created_length
300 )
301 self.date_created = epoch + pd.to_timedelta(x, unit="s")
302 x = self._read_float(
303 const.date_modified_offset + align1, const.date_modified_length
304 )
305 self.date_modified = epoch + pd.to_timedelta(x, unit="s")
306
307 self.header_length = self._read_uint(
308 const.header_size_offset + align1, const.header_size_length
309 )
310
311 # Read the rest of the header into cached_page.
312 buf = self._path_or_buf.read(self.header_length - 288)
313 self._cached_page += buf
314 # error: Argument 1 to "len" has incompatible type "Optional[bytes]";
315 # expected "Sized"
316 if len(self._cached_page) != self.header_length: # type: ignore[arg-type]
317 raise ValueError("The SAS7BDAT file appears to be truncated.")
318
319 self._page_length = self._read_uint(
320 const.page_size_offset + align1, const.page_size_length
321 )

330 # Read a single float of the given width (4 or 8).
331 def _read_float(self, offset: int, width: int):
332 assert self._cached_page is not None
333 if width == 4:
334 return read_float_with_byteswap(
335 self._cached_page, offset, self.need_byteswap
336 )
337 elif width == 8:
338 return read_double_with_byteswap(
339 self._cached_page, offset, self.need_byteswap
340 )
341 else:
342 self.close()
343 raise ValueError("invalid float width")

345 # Read a single unsigned integer of the given width (1, 2, 4 or 8).
346 def _read_uint(self, offset: int, width: int) -> int:
347 assert self._cached_page is not None
348 if width == 1:
349 return self._read_bytes(offset, 1)[0]
350 elif width == 2:
351 return read_uint16_with_byteswap(
352 self._cached_page, offset, self.need_byteswap
353 )
354 elif width == 4:
355 return read_uint32_with_byteswap(
356 self._cached_page, offset, self.need_byteswap
357 )
358 elif width == 8:
359 return read_uint64_with_byteswap(
360 self._cached_page, offset, self.need_byteswap
361 )
362 else:
363 self.close()
364 raise ValueError("invalid int width")

367 assert self._cached_page is not None

378 def _parse_metadata(self) -> None:
379 done = False
380 while not done:
381 self._cached_page = self._path_or_buf.read(self._page_length)
382 if len(self._cached_page) <= 0:
383 break
384 if len(self._cached_page) != self._page_length:
385 raise ValueError("Failed to read a meta data page from the SAS file.")
386 done = self._process_page_meta()

388 def _process_page_meta(self) -> bool:
389 self._read_page_header()
390 pt = const.page_meta_types + [const.page_amd_type, const.page_mix_type]
391 if self._current_page_type in pt:
392 self._process_page_metadata()
393 is_data_page = self._current_page_type == const.page_data_type
394 is_mix_page = self._current_page_type == const.page_mix_type
395 return bool(
396 is_data_page
397 or is_mix_page
398 or self._current_page_data_subheader_pointers != []
399 )

401 def _read_page_header(self) -> None:
402 bit_offset = self._page_bit_offset
403 tx = const.page_type_offset + bit_offset
404 self._current_page_type = (
405 self._read_uint(tx, const.page_type_length) & const.page_type_mask2
406 )
407 tx = const.block_count_offset + bit_offset
408 self._current_page_block_count = self._read_uint(tx, const.block_count_length)
409 tx = const.subheader_count_offset + bit_offset
410 self._current_page_subheaders_count = self._read_uint(
411 tx, const.subheader_count_length
412 )

414 def _process_page_metadata(self) -> None:
415 bit_offset = self._page_bit_offset
416
417 for i in range(self._current_page_subheaders_count):
418 offset = const.subheader_pointers_offset + bit_offset
419 total_offset = offset + self._subheader_pointer_length * i
420
421 subheader_offset = self._read_uint(total_offset, self._int_length)
422 total_offset += self._int_length
423
424 subheader_length = self._read_uint(total_offset, self._int_length)
425 total_offset += self._int_length
426
427 subheader_compression = self._read_uint(total_offset, 1)
428 total_offset += 1
429
430 subheader_type = self._read_uint(total_offset, 1)
431
432 if (
433 subheader_length == 0
434 or subheader_compression == const.truncated_subheader_id
435 ):
436 continue
437
438 subheader_signature = self._read_bytes(subheader_offset, self._int_length)
439 subheader_index = get_subheader_index(subheader_signature)
440 subheader_processor = self._subheader_processors[subheader_index]
441
442 if subheader_processor is None:
443 f1 = subheader_compression in (const.compressed_subheader_id, 0)
444 f2 = subheader_type == const.compressed_subheader_type
445 if self.compression and f1 and f2:
446 self._current_page_data_subheader_pointers.append(
447 (subheader_offset, subheader_length)
448 )
449 else:
450 self.close()
451 raise ValueError(
452 f"Unknown subheader signature {subheader_signature}"
453 )
454 else:
455 subheader_processor(subheader_offset, subheader_length)

664 def read(self, nrows: int | None = None) -> DataFrame:
665 if (nrows is None) and (self.chunksize is not None):
666 nrows = self.chunksize
667 elif nrows is None:
668 nrows = self.row_count
669
670 if len(self._column_types) == 0:
671 self.close()
672 raise EmptyDataError("No columns to parse from file")
673
674 if nrows > 0 and self._current_row_in_file_index >= self.row_count:
675 return DataFrame()
676
677 nrows = min(nrows, self.row_count - self._current_row_in_file_index)
678
679 nd = self._column_types.count(b"d")
680 ns = self._column_types.count(b"s")
