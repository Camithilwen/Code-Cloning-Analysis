141def test_bin_seeds(global_dtype):
142 # Test the bin seeding technique which can be used in the mean shift
143 # algorithm
144 # Data is just 6 points in the plane
145 X = np.array(
146 [[1.0, 1.0], [1.4, 1.4], [1.8, 1.2], [2.0, 1.0], [2.1, 1.1], [0.0, 0.0]],
147 dtype=global_dtype,
148 )
149
150 # With a bin coarseness of 1.0 and min_bin_freq of 1, 3 bins should be
151 # found
152 ground_truth = {(1.0, 1.0), (2.0, 1.0), (0.0, 0.0)}
153 test_bins = get_bin_seeds(X, 1, 1)
154 test_result = set(tuple(p) for p in test_bins)
155 assert len(ground_truth.symmetric_difference(test_result)) == 0
156
157 # With a bin coarseness of 1.0 and min_bin_freq of 2, 2 bins should be
158 # found
159 ground_truth = {(1.0, 1.0), (2.0, 1.0)}
160 test_bins = get_bin_seeds(X, 1, 2)
161 test_result = set(tuple(p) for p in test_bins)
162 assert len(ground_truth.symmetric_difference(test_result)) == 0
163
164 # With a bin size of 0.01 and min_bin_freq of 1, 6 bins should be found
165 # we bail and use the whole data here.
166 with warnings.catch_warnings(record=True):
167 test_bins = get_bin_seeds(X, 0.01, 1)
168 assert_allclose(test_bins, X)
169
170 # tight clusters around [0, 0] and [1, 1], only get two bins
171 X, _ = make_blobs(

174 centers=[[0, 0], [1, 1]],

176 random_state=0,

178 X = X.astype(global_dtype, copy=False)

196def test_mean_shift_zero_bandwidth(global_dtype):
197 # Check that mean shift works when the estimated bandwidth is 0.
198 X = np.array([1, 1, 1, 2, 2, 2, 3, 3], dtype=global_dtype).reshape(-1, 1)
199
200 # estimate_bandwidth with default args returns 0 on this dataset
201 bandwidth = estimate_bandwidth(X)
202 assert bandwidth == 0
203
204 # get_bin_seeds with a 0 bin_size should return the dataset itself
205 assert get_bin_seeds(X, bin_size=bandwidth) is X
206
207 # MeanShift with binning and a 0 estimated bandwidth should be equivalent
208 # to no binning.
209 ms_binning = MeanShift(bin_seeding=True, bandwidth=None).fit(X)
210 ms_nobinning = MeanShift(bin_seeding=False).fit(X)
211 expected_labels = np.array([0, 0, 0, 1, 1, 1, 2, 2])
212
213 assert v_measure_score(ms_binning.labels_, expected_labels) == pytest.approx(1)
214 assert v_measure_score(ms_nobinning.labels_, expected_labels) == pytest.approx(1)
