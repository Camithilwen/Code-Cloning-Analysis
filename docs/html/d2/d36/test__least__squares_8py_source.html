68def fun_bvp(x):
69 n = int(np.sqrt(x.shape[0]))
70 u = np.zeros((n + 2, n + 2))
71 x = x.reshape((n, n))
72 u[1:-1, 1:-1] = x
73 y = u[:-2, 1:-1] + u[2:, 1:-1] + u[1:-1, :-2] + u[1:-1, 2:] - 4 * x + x**3
74 return y.ravel()

77class BroydenTridiagonal:
78 def __init__(self, n=100, mode='sparse'):
79 np.random.seed(0)
80
81 self.n = n
82
83 self.x0 = -np.ones(n)
84 self.lb = np.linspace(-2, -1.5, n)
85 self.ub = np.linspace(-0.8, 0.0, n)
86
87 self.lb += 0.1 * np.random.randn(n)
88 self.ub += 0.1 * np.random.randn(n)
89
90 self.x0 += 0.1 * np.random.randn(n)
91 self.x0 = make_strictly_feasible(self.x0, self.lb, self.ub)
92
93 if mode == 'sparse':
94 self.sparsity = lil_matrix((n, n), dtype=int)
95 i = np.arange(n)
96 self.sparsity[i, i] = 1
97 i = np.arange(1, n)
98 self.sparsity[i, i - 1] = 1
99 i = np.arange(n - 1)
100 self.sparsity[i, i + 1] = 1
101
102 self.jac = self._jac
103 elif mode == 'operator':
104 self.jac = lambda x: aslinearoperator(self._jac(x))
105 elif mode == 'dense':
106 self.sparsity = None
107 self.jac = lambda x: self._jac(x).toarray()
108 else:
109 assert_(False)
110
111 def fun(self, x):
112 f = (3 - x) * x + 1
113 f[1:] -= x[:-1]
114 f[:-1] -= 2 * x[1:]
115 return f
116
117 def _jac(self, x):
118 J = lil_matrix((self.n, self.n))
119 i = np.arange(self.n)
120 J[i, i] = 3 - 2 * x
121 i = np.arange(1, self.n)
122 J[i, i - 1] = -1
123 i = np.arange(self.n - 1)
124 J[i, i + 1] = -2
125 return J

128class ExponentialFittingProblem:
129 """Provide data and function for exponential fitting in the form
130 y = a + exp(b * x) + noise."""
131
132 def __init__(self, a, b, noise, n_outliers=1, x_range=(-1, 1),
133 n_points=11, random_seed=None):
134 np.random.seed(random_seed)
135 self.m = n_points
136 self.n = 2
137
138 self.p0 = np.zeros(2)
139 self.x = np.linspace(x_range[0], x_range[1], n_points)
140
141 self.y = a + np.exp(b * self.x)
142 self.y += noise * np.random.randn(self.m)
143
144 outliers = np.random.randint(0, self.m, n_outliers)
145 self.y[outliers] += 50 * noise * np.random.rand(n_outliers)
146
147 self.p_opt = np.array([a, b])
148
149 def fun(self, p):
150 return p[0] + np.exp(p[1] * self.x) - self.y
151
152 def jac(self, p):
153 J = np.empty((self.m, self.n))
154 J[:, 0] = 1
155 J[:, 1] = self.x * np.exp(p[1] * self.x)
156 return J

159def cubic_soft_l1(z):
160 rho = np.empty((3, z.size))
161
162 t = 1 + z
163 rho[0] = 3 * (t**(1/3) - 1)
164 rho[1] = t ** (-2/3)
165 rho[2] = -2/3 * t**(-5/3)
166
167 return rho

639 def test_jac(self):
640 # Test that res.jac.T.dot(res.jac) gives Gauss-Newton approximation
641 # of Hessian. This approximation is computed by doubly differentiating
642 # the cost function and dropping the part containing second derivative
643 # of f. For a scalar function it is computed as
644 # H = (rho' + 2 * rho'' * f**2) * f'**2, if the expression inside the
645 # brackets is less than EPS it is replaced by EPS. Here, we check
646 # against the root of H.
647
648 x = 2.0 # res.x will be this.
649 f = x**2 + 5 # res.fun will be this.
650
651 res = least_squares(fun_trivial, x, jac_trivial, loss='linear',
652 max_nfev=1, method=self.method)
653 assert_equal(res.jac, 2 * x)
654
655 # For `huber` loss the Jacobian correction is identically zero
656 # in outlier region, in such cases it is modified to be equal EPS**0.5.
657 res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
658 max_nfev=1, method=self.method)
659 assert_equal(res.jac, 2 * x * EPS**0.5)
660
661 # Now, let's apply `loss_scale` to turn the residual into an inlier.
662 # The loss function becomes linear.
663 res = least_squares(fun_trivial, x, jac_trivial, loss='huber',
664 f_scale=10, max_nfev=1)
665 assert_equal(res.jac, 2 * x)
666
667 # 'soft_l1' always gives a positive scaling.
668 res = least_squares(fun_trivial, x, jac_trivial, loss='soft_l1',
669 max_nfev=1, method=self.method)
670 assert_allclose(res.jac, 2 * x * (1 + f**2)**-0.75)
671
672 # For 'cauchy' the correction term turns out to be negative, and it
673 # replaced by EPS**0.5.
674 res = least_squares(fun_trivial, x, jac_trivial, loss='cauchy',
675 max_nfev=1, method=self.method)
676 assert_allclose(res.jac, 2 * x * EPS**0.5)
677
678 # Now use scaling to turn the residual to inlier.
679 res = least_squares(fun_trivial, x, jac_trivial, loss='cauchy',
680 f_scale=10, max_nfev=1, method=self.method)
681 fs = f / 10
682 assert_allclose(res.jac, 2 * x * (1 - fs**2)**0.5 / (1 + fs**2))
683
684 # 'arctan' gives an outlier.
685 res = least_squares(fun_trivial, x, jac_trivial, loss='arctan',
686 max_nfev=1, method=self.method)
687 assert_allclose(res.jac, 2 * x * EPS**0.5)
688
689 # Turn to inlier.
690 res = least_squares(fun_trivial, x, jac_trivial, loss='arctan',
691 f_scale=20.0, max_nfev=1, method=self.method)
692 fs = f / 20
693 assert_allclose(res.jac, 2 * x * (1 - 3 * fs**4)**0.5 / (1 + fs**4))
694
695 # cubic_soft_l1 will give an outlier.
696 res = least_squares(fun_trivial, x, jac_trivial, loss=cubic_soft_l1,
697 max_nfev=1)
698 assert_allclose(res.jac, 2 * x * EPS**0.5)
699
700 # Turn to inlier.
701 res = least_squares(fun_trivial, x, jac_trivial,
702 loss=cubic_soft_l1, f_scale=6, max_nfev=1)
703 fs = f / 6
704 assert_allclose(res.jac,
705 2 * x * (1 - fs**2 / 3)**0.5 * (1 + fs**2)**(-5/6))
