28def random_X_y_coef(
29 linear_model_loss, n_samples, n_features, coef_bound=(-2, 2), seed=42
30):
31 """Random generate y, X and coef in valid range."""
32 rng = np.random.RandomState(seed)
33 n_dof = n_features + linear_model_loss.fit_intercept
34 X = make_low_rank_matrix(
35 n_samples=n_samples,
36 n_features=n_features,
37 random_state=rng,
38 )
39 coef = linear_model_loss.init_zero_coef(X)
40
41 if linear_model_loss.base_loss.is_multiclass:
42 n_classes = linear_model_loss.base_loss.n_classes
43 coef.flat[:] = rng.uniform(
44 low=coef_bound[0],
45 high=coef_bound[1],
46 size=n_classes * n_dof,
47 )
48 if linear_model_loss.fit_intercept:
49 raw_prediction = X @ coef[:, :-1].T + coef[:, -1]
50 else:
51 raw_prediction = X @ coef.T
52 proba = linear_model_loss.base_loss.link.inverse(raw_prediction)
53
54 # y = rng.choice(np.arange(n_classes), p=proba) does not work.
55 # See https://stackoverflow.com/a/34190035/16761084
56 def choice_vectorized(items, p):
57 s = p.cumsum(axis=1)
58 r = rng.rand(p.shape[0])[:, None]
59 k = (s < r).sum(axis=1)
60 return items[k]
61
62 y = choice_vectorized(np.arange(n_classes), p=proba).astype(np.float64)
63 else:
64 coef.flat[:] = rng.uniform(
65 low=coef_bound[0],
66 high=coef_bound[1],
67 size=n_dof,
68 )
69 if linear_model_loss.fit_intercept:
70 raw_prediction = X @ coef[:-1] + coef[-1]
71 else:
72 raw_prediction = X @ coef
73 y = linear_model_loss.base_loss.link.inverse(
74 raw_prediction + rng.uniform(low=-1, high=1, size=n_samples)
75 )
76
77 return X, y, coef

207def test_loss_gradients_hessp_intercept(
208 base_loss, sample_weight, l2_reg_strength, X_container, global_random_seed
209):
210 """Test that loss and gradient handle intercept correctly."""
211 loss = LinearModelLoss(base_loss=base_loss(), fit_intercept=False)
212 loss_inter = LinearModelLoss(base_loss=base_loss(), fit_intercept=True)
213 n_samples, n_features = 10, 5
214 X, y, coef = random_X_y_coef(
215 linear_model_loss=loss,
216 n_samples=n_samples,
217 n_features=n_features,
218 seed=global_random_seed,
219 )
220
221 X[:, -1] = 1 # make last column of 1 to mimic intercept term
222 X_inter = X[
223 :, :-1
224 ] # exclude intercept column as it is added automatically by loss_inter
225
226 if X_container is not None:
227 X = X_container(X)
228
229 if sample_weight == "range":
230 sample_weight = np.linspace(1, y.shape[0], num=y.shape[0])
231
232 l, g = loss.loss_gradient(
233 coef, X, y, sample_weight=sample_weight, l2_reg_strength=l2_reg_strength
234 )
235 _, hessp = loss.gradient_hessian_product(
236 coef, X, y, sample_weight=sample_weight, l2_reg_strength=l2_reg_strength
237 )
238 l_inter, g_inter = loss_inter.loss_gradient(
239 coef, X_inter, y, sample_weight=sample_weight, l2_reg_strength=l2_reg_strength
240 )
241 _, hessp_inter = loss_inter.gradient_hessian_product(
242 coef, X_inter, y, sample_weight=sample_weight, l2_reg_strength=l2_reg_strength
243 )
244

248 )
249
250 g_inter_corrected = g_inter
251 g_inter_corrected.T[-1] += l2_reg_strength * coef.T[-1]
252 assert_allclose(g, g_inter_corrected)
253
254 s = np.random.RandomState(global_random_seed).randn(*coef.shape)
255 h = hessp(s)
256 h_inter = hessp_inter(s)
257 h_inter_corrected = h_inter
258 h_inter_corrected.T[-1] += l2_reg_strength * s.T[-1]
259 assert_allclose(h, h_inter_corrected)

262@pytest.mark.parametrize("base_loss", LOSSES)
263@pytest.mark.parametrize("fit_intercept", [False, True])
264@pytest.mark.parametrize("sample_weight", [None, "range"])
265@pytest.mark.parametrize("l2_reg_strength", [0, 1])
266def test_gradients_hessians_numerically(
267 base_loss, fit_intercept, sample_weight, l2_reg_strength, global_random_seed
268):
269 """Test gradients and hessians with numerical derivatives.
270
271 Gradient should equal the numerical derivatives of the loss function.
272 Hessians should equal the numerical derivatives of gradients.
273 """
274 loss = LinearModelLoss(base_loss=base_loss(), fit_intercept=fit_intercept)
275 n_samples, n_features = 10, 5
276 X, y, coef = random_X_y_coef(
277 linear_model_loss=loss,
278 n_samples=n_samples,
279 n_features=n_features,
280 seed=global_random_seed,
281 )
282 coef = coef.ravel(order="F") # this is important only for multinomial loss
283
284 if sample_weight == "range":
285 sample_weight = np.linspace(1, y.shape[0], num=y.shape[0])
286
287 # 1. Check gradients numerically
288 eps = 1e-6
289 g, hessp = loss.gradient_hessian_product(
290 coef, X, y, sample_weight=sample_weight, l2_reg_strength=l2_reg_strength
291 )
292 # Use a trick to get central finite difference of accuracy 4 (five-point stencil)
293 # https://en.wikipedia.org/wiki/Numerical_differentiation
294 # https://en.wikipedia.org/wiki/Finite_difference_coefficient
295 # approx_g1 = (f(x + eps) - f(x - eps)) / (2*eps)
296 approx_g1 = optimize.approx_fprime(
297 coef,
298 lambda coef: loss.loss(
299 coef - eps,
300 X,
301 y,
302 sample_weight=sample_weight,
303 l2_reg_strength=l2_reg_strength,
304 ),
305 2 * eps,
306 )
307 # approx_g2 = (f(x + 2*eps) - f(x - 2*eps)) / (4*eps)
308 approx_g2 = optimize.approx_fprime(
309 coef,
310 lambda coef: loss.loss(
311 coef - 2 * eps,
312 X,
313 y,
314 sample_weight=sample_weight,
315 l2_reg_strength=l2_reg_strength,
316 ),
317 4 * eps,
318 )
319 # Five-point stencil approximation
320 # See: https://en.wikipedia.org/wiki/Five-point_stencil#1D_first_derivative
321 approx_g = (4 * approx_g1 - approx_g2) / 3
322 assert_allclose(g, approx_g, rtol=1e-2, atol=1e-8)
323
324 # 2. Check hessp numerically along the second direction of the gradient
325 vector = np.zeros_like(g)
326 vector[1] = 1
327 hess_col = hessp(vector)
328 # Computation of the Hessian is particularly fragile to numerical errors when doing
329 # simple finite differences. Here we compute the grad along a path in the direction
330 # of the vector and then use a least-square regression to estimate the slope
331 eps = 1e-3
332 d_x = np.linspace(-eps, eps, 30)
333 d_grad = np.array(
334 [
335 loss.gradient(
336 coef + t * vector,
337 X,
338 y,
339 sample_weight=sample_weight,
340 l2_reg_strength=l2_reg_strength,
341 )
342 for t in d_x
343 ]
344 )
345 d_grad -= d_grad.mean(axis=0)
346 approx_hess_col = linalg.lstsq(d_x[:, np.newaxis], d_grad)[0].ravel()
347 assert_allclose(approx_hess_col, hess_col, rtol=1e-3)

391@pytest.mark.parametrize("sample_weight", [None, "range"])
392def test_multinomial_hessian_3_classes(sample_weight, global_random_seed):
393 """Test multinomial hessian for 3 classes and 2 points.
394
395 For n_classes = 3 and n_samples = 2, we have
396 p0 = [p0_0, p0_1]
397 p1 = [p1_0, p1_1]
398 p2 = [p2_0, p2_1]
399 and with 2 x 2 diagonal subblocks
400 H = [p0 * (1-p0), -p0 * p1, -p0 * p2]
401 [ -p0 * p1, p1 * (1-p1), -p1 * p2]
402 [ -p0 * p2, -p1 * p2, p2 * (1-p2)]
403 hess = X' H X
404 """
405 n_samples, n_features, n_classes = 2, 5, 3
406 loss = LinearModelLoss(
407 base_loss=HalfMultinomialLoss(n_classes=n_classes), fit_intercept=False
408 )
409 X, y, coef = random_X_y_coef(
410 linear_model_loss=loss,
411 n_samples=n_samples,
412 n_features=n_features,
413 seed=global_random_seed,
414 )
415 coef = coef.ravel(order="F") # this is important only for multinomial loss
416
417 if sample_weight == "range":
418 sample_weight = np.linspace(1, y.shape[0], num=y.shape[0])
419
420 grad, hess, _ = loss.gradient_hessian(
421 coef,
422 X,
423 y,
424 sample_weight=sample_weight,
425 l2_reg_strength=0,
426 )
427 # Hessian must be a symmetrix matrix.
428 assert_allclose(hess, hess.T)
429
430 weights, intercept, raw_prediction = loss.weight_intercept_raw(coef, X)
431 grad_pointwise, proba = loss.base_loss.gradient_proba(
432 y_true=y,
433 raw_prediction=raw_prediction,
434 sample_weight=sample_weight,
435 )
436 p0d, p1d, p2d, oned = (
437 np.diag(proba[:, 0]),
438 np.diag(proba[:, 1]),
439 np.diag(proba[:, 2]),
440 np.diag(np.ones(2)),
441 )
442 h = np.block(
443 [
444 [p0d * (oned - p0d), -p0d * p1d, -p0d * p2d],
445 [-p0d * p1d, p1d * (oned - p1d), -p1d * p2d],
446 [-p0d * p2d, -p1d * p2d, p2d * (oned - p2d)],
447 ]
448 )
449 h = h.reshape((n_classes, n_samples, n_classes, n_samples))
450 if sample_weight is None:
451 h /= n_samples
452 else:
453 h *= sample_weight / np.sum(sample_weight)
454 # hess_expected.shape = (n_features, n_classes, n_classes, n_features)
455 hess_expected = np.einsum("ij, mini, ik->jmnk", X, h, X)
456 hess_expected = np.moveaxis(hess_expected, 2, 3)
457 hess_expected = hess_expected.reshape(
458 n_classes * n_features, n_classes * n_features, order="C"
459 )
460 assert_allclose(hess_expected, hess_expected.T)
461 assert_allclose(hess, hess_expected)
