157def test_determinism_check(objects, fmt, usetex):
158 """
159 Output three times the same graphs and checks that the outputs are exactly
160 the same.
161
162 Parameters
163 ----------
164 objects : str
165 Objects to be included in the test document: 'm' for markers, 'h' for
166 hatch patterns, 'i' for images, and 'p' for paths.
167 fmt : {"pdf", "ps", "svg"}
168 Output format.
169 """
170 plots = [
171 subprocess_run_for_testing(
172 [sys.executable, "-R", "-c",
173 f"from matplotlib.tests.test_determinism import _save_figure;"
174 f"_save_figure({objects!r}, {fmt!r}, {usetex})"],
175 env={**os.environ, "SOURCE_DATE_EPOCH": "946684800",
176 "MPLBACKEND": "Agg"},
177 text=False, capture_output=True, check=True).stdout
178 for _ in range(3)
179 ]
180 for p in plots[1:]:
181 if fmt == "ps" and usetex:
182 if p != plots[0]:
183 pytest.skip("failed, maybe due to ghostscript timestamps")
184 else:
185 assert p == plots[0]
