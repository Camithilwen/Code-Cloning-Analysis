18def spline_filter(Iin, lmbda=5.0):
19 """Smoothing spline (cubic) filtering of a rank-2 array.
20
21 Filter an input data set, `Iin`, using a (cubic) smoothing spline of
22 fall-off `lmbda`.
23
24 Parameters
25 ----------
26 Iin : array_like
27 input data set
28 lmbda : float, optional
29 spline smooghing fall-off value, default is `5.0`.
30
31 Returns
32 -------
33 res : ndarray
34 filterd input data
35
36 Examples
37 --------
38 We can filter an multi dimentional signal (ex: 2D image) using cubic
39 B-spline filter:
40
41 >>> import numpy as np
42 >>> from scipy.signal import spline_filter
43 >>> import matplotlib.pyplot as plt
44 >>> orig_img = np.eye(20) # create an image
45 >>> orig_img[10, :] = 1.0
46 >>> sp_filter = spline_filter(orig_img, lmbda=0.1)
47 >>> f, ax = plt.subplots(1, 2, sharex=True)
48 >>> for ind, data in enumerate([[orig_img, "original image"],
49 ... [sp_filter, "spline filter"]]):
50 ... ax[ind].imshow(data[0], cmap='gray_r')
51 ... ax[ind].set_title(data[1])
52 >>> plt.tight_layout()
53 >>> plt.show()
54
55 """
56 intype = Iin.dtype.char
57 hcol = array([1.0, 4.0, 1.0], 'f') / 6.0
58 if intype in ['F', 'D']:
59 Iin = Iin.astype('F')
60 ckr = cspline2d(Iin.real, lmbda)
61 cki = cspline2d(Iin.imag, lmbda)
62 outr = sepfir2d(ckr, hcol, hcol)
63 outi = sepfir2d(cki, hcol, hcol)
64 out = (outr + 1j * outi).astype(intype)
65 elif intype in ['f', 'd']:
66 ckr = cspline2d(Iin, lmbda)
67 out = sepfir2d(ckr, hcol, hcol)
68 out = out.astype(intype)
69 else:
70 raise TypeError("Invalid data type for Iin")
71 return out

77def _bspline_piecefunctions(order):
78 """Returns the function defined over the left-side pieces for a bspline of
79 a given order.
80
81 The 0th piece is the first one less than 0. The last piece is a function
82 identical to 0 (returned as the constant 0). (There are order//2 + 2 total
83 pieces).
84
85 Also returns the condition functions that when evaluated return boolean
86 arrays for use with `numpy.piecewise`.
87 """
88 try:
89 return _splinefunc_cache[order]
90 except KeyError:
91 pass
92
93 def condfuncgen(num, val1, val2):
94 if num == 0:
95 return lambda x: logical_and(less_equal(x, val1),
96 greater_equal(x, val2))
97 elif num == 2:
98 return lambda x: less_equal(x, val2)
99 else:
100 return lambda x: logical_and(less(x, val1),
101 greater_equal(x, val2))
102
103 last = order // 2 + 2
104 if order % 2:
105 startbound = -1.0
106 else:
107 startbound = -0.5
108 condfuncs = [condfuncgen(0, 0, startbound)]
109 bound = startbound
110 for num in range(1, last - 1):
111 condfuncs.append(condfuncgen(1, bound, bound - 1))
112 bound = bound - 1
113 condfuncs.append(condfuncgen(2, 0, -(order + 1) / 2.0))
114
115 # final value of bound is used in piecefuncgen below
116
117 # the functions to evaluate are taken from the left-hand side
118 # in the general expression derived from the central difference
119 # operator (because they involve fewer terms).
120
121 fval = float_factorial(order)
122
123 def piecefuncgen(num):
124 Mk = order // 2 - num
125 if (Mk < 0):
126 return 0 # final function is 0
127 coeffs = [(1 - 2 * (k % 2)) * float(comb(order + 1, k, exact=1)) / fval
128 for k in range(Mk + 1)]
129 shifts = [-bound - k for k in range(Mk + 1)]
130
131 def thefunc(x):
132 res = 0.0
133 for k in range(Mk + 1):
134 res += coeffs[k] * (x + shifts[k]) ** order
135 return res
136 return thefunc
137
138 funclist = [piecefuncgen(k) for k in range(last)]
139
140 _splinefunc_cache[order] = (funclist, condfuncs)
141
142 return funclist, condfuncs

157@np.deprecate(message=msg_bspline)
158def bspline(x, n):
159 """B-spline basis function of order n.
160
161 Parameters
162 ----------
163 x : array_like
164 a knot vector
165 n : int
166 The order of the spline. Must be non-negative, i.e., n >= 0
167
168 Returns
169 -------
170 res : ndarray
171 B-spline basis function values
172
173 See Also
174 --------
175 cubic : A cubic B-spline.
176 quadratic : A quadratic B-spline.
177
178 Notes
179 -----
180 Uses numpy.piecewise and automatic function-generator.
181
182 Examples
183 --------
184 We can calculate B-Spline basis function of several orders:
185
186 >>> import numpy as np
187 >>> from scipy.signal import bspline, cubic, quadratic
188 >>> bspline(0.0, 1)
189 1
190
191 >>> knots = [-1.0, 0.0, -1.0]
192 >>> bspline(knots, 2)
193 array([0.125, 0.75, 0.125])
194
195 >>> np.array_equal(bspline(knots, 2), quadratic(knots))
196 True
197
198 >>> np.array_equal(bspline(knots, 3), cubic(knots))
199 True
200
201 """
202 ax = -abs(asarray(x, dtype=float))
203 # number of pieces on the left-side is (n+1)/2
204 funclist, condfuncs = _bspline_piecefunctions(n)
205 condlist = [func(ax) for func in condfuncs]
206 return piecewise(ax, condlist, funclist)

208def gauss_spline(x, n):
209 r"""Gaussian approximation to B-spline basis function of order n.
210
211 Parameters
212 ----------
213 x : array_like
214 a knot vector
215 n : int
216 The order of the spline. Must be non-negative, i.e., n >= 0
217
218 Returns
219 -------
220 res : ndarray
221 B-spline basis function values approximated by a zero-mean Gaussian
222 function.
223
224 Notes
225 -----
226 The B-spline basis function can be approximated well by a zero-mean
227 Gaussian function with standard-deviation equal to :math:`\sigma=(n+1)/12`
228 for large `n` :
229
230 .. math:: \frac{1}{\sqrt {2\pi\sigma^2}}exp(-\frac{x^2}{2\sigma})
231
232 References
233 ----------
234 .. [1] Bouma H., Vilanova A., Bescos J.O., ter Haar Romeny B.M., Gerritsen
235 F.A. (2007) Fast and Accurate Gaussian Derivatives Based on B-Splines. In:
236 Sgallari F., Murli A., Paragios N. (eds) Scale Space and Variational
237 Methods in Computer Vision. SSVM 2007. Lecture Notes in Computer
238 Science, vol 4485. Springer, Berlin, Heidelberg
239 .. [2] http://folk.uio.no/inf3330/scripting/doc/python/SciPy/tutorial/old/node24.html
240
241 Examples
242 --------
243 We can calculate B-Spline basis functions approximated by a gaussian
244 distribution:
245
246 >>> import numpy as np
247 >>> from scipy.signal import gauss_spline, bspline
248 >>> knots = np.array([-1.0, 0.0, -1.0])
249 >>> gauss_spline(knots, 3)
250 array([0.15418033, 0.6909883, 0.15418033]) # may vary
251
252 >>> bspline(knots, 3)
253 array([0.16666667, 0.66666667, 0.16666667]) # may vary
254
255 """
256 x = asarray(x)
257 signsq = (n + 1) / 12.0
258 return 1 / sqrt(2 * pi * signsq) * exp(-x ** 2 / 2 / signsq)

407def _coeff_smooth(lam):
408 xi = 1 - 96 * lam + 24 * lam * sqrt(3 + 144 * lam)
409 omeg = arctan2(sqrt(144 * lam - 1), sqrt(xi))
410 rho = (24 * lam - 1 - sqrt(xi)) / (24 * lam)
411 rho = rho * sqrt((48 * lam + 24 * lam * sqrt(3 + 144 * lam)) / xi)
412 return rho, omeg

415def _hc(k, cs, rho, omega):
416 return (cs / sin(omega) * (rho ** k) * sin(omega * (k + 1)) *
417 greater(k, -1))
418
419
420def _hs(k, cs, rho, omega):
421 c0 = (cs * cs * (1 + rho * rho) / (1 - rho * rho) /
422 (1 - 2 * rho * rho * cos(2 * omega) + rho ** 4))
423 gamma = (1 - rho * rho) / (1 + rho * rho) / tan(omega)
424 ak = abs(k)
425 return c0 * rho ** ak * (cos(omega * ak) + gamma * sin(omega * ak))

428def _cubic_smooth_coeff(signal, lamb):
429 rho, omega = _coeff_smooth(lamb)
430 cs = 1 - 2 * rho * cos(omega) + rho * rho
431 K = len(signal)
432 yp = zeros((K,), signal.dtype.char)
433 k = arange(K)
434 yp[0] = (_hc(0, cs, rho, omega) * signal[0] +
435 add.reduce(_hc(k + 1, cs, rho, omega) * signal))
436
437 yp[1] = (_hc(0, cs, rho, omega) * signal[0] +
438 _hc(1, cs, rho, omega) * signal[1] +
439 add.reduce(_hc(k + 2, cs, rho, omega) * signal))
440
441 for n in range(2, K):
442 yp[n] = (cs * signal[n] + 2 * rho * cos(omega) * yp[n - 1] -
443 rho * rho * yp[n - 2])
444
445 y = zeros((K,), signal.dtype.char)
446
447 y[K - 1] = add.reduce((_hs(k, cs, rho, omega) +
448 _hs(k + 1, cs, rho, omega)) * signal[::-1])
449 y[K - 2] = add.reduce((_hs(k - 1, cs, rho, omega) +
450 _hs(k + 2, cs, rho, omega)) * signal[::-1])
451
452 for n in range(K - 3, -1, -1):
453 y[n] = (cs * yp[n] + 2 * rho * cos(omega) * y[n + 1] -
454 rho * rho * y[n + 2])
455
456 return y

459def _cubic_coeff(signal):
460 zi = -2 + sqrt(3)
461 K = len(signal)
462 yplus = zeros((K,), signal.dtype.char)
463 powers = zi ** arange(K)
464 yplus[0] = signal[0] + zi * add.reduce(powers * signal)
465 for k in range(1, K):
466 yplus[k] = signal[k] + zi * yplus[k - 1]
467 output = zeros((K,), signal.dtype)
468 output[K - 1] = zi / (zi - 1) * yplus[K - 1]
469 for k in range(K - 2, -1, -1):
470 output[k] = zi * (output[k + 1] - yplus[k])
471 return output * 6.0

590def cspline1d_eval(cj, newx, dx=1.0, x0=0):
591 """Evaluate a cubic spline at the new set of points.
592
593 `dx` is the old sample-spacing while `x0` was the old origin. In
594 other-words the old-sample points (knot-points) for which the `cj`
595 represent spline coefficients were at equally-spaced points of:
596
597 oldx = x0 + j*dx j=0...N-1, with N=len(cj)
598
599 Edges are handled using mirror-symmetric boundary conditions.
600
601 Parameters
602 ----------
603 cj : ndarray
604 cublic spline coefficients
605 newx : ndarray
606 New set of points.
607 dx : float, optional
608 Old sample-spacing, the default value is 1.0.
609 x0 : int, optional
610 Old origin, the default value is 0.
611
612 Returns
613 -------
614 res : ndarray
615 Evaluated a cubic spline points.
616
617 See Also
618 --------
619 cspline1d : Compute cubic spline coefficients for rank-1 array.
620
621 Examples
622 --------
623 We can filter a signal to reduce and smooth out high-frequency noise with
624 a cubic spline:
625
626 >>> import numpy as np
627 >>> import matplotlib.pyplot as plt
628 >>> from scipy.signal import cspline1d, cspline1d_eval
629 >>> rng = np.random.default_rng()
630 >>> sig = np.repeat([0., 1., 0.], 100)
631 >>> sig += rng.standard_normal(len(sig))*0.05 # add noise
632 >>> time = np.linspace(0, len(sig))
633 >>> filtered = cspline1d_eval(cspline1d(sig), time)
634 >>> plt.plot(sig, label="signal")
635 >>> plt.plot(time, filtered, label="filtered")
636 >>> plt.legend()
637 >>> plt.show()
638
639 """
640 newx = (asarray(newx) - x0) / float(dx)
641 res = zeros_like(newx, dtype=cj.dtype)
642 if res.size == 0:
643 return res
644 N = len(cj)
645 cond1 = newx < 0
646 cond2 = newx > (N - 1)
647 cond3 = ~(cond1 | cond2)
648 # handle general mirror-symmetry
649 res[cond1] = cspline1d_eval(cj, -newx[cond1])
650 res[cond2] = cspline1d_eval(cj, 2 * (N - 1) - newx[cond2])
651 newx = newx[cond3]
652 if newx.size == 0:
653 return res
654 result = zeros_like(newx, dtype=cj.dtype)
655 jlower = floor(newx - 2).astype(int) + 1
656 for i in range(4):
657 thisj = jlower + i
658 indj = thisj.clip(0, N - 1) # handle edge cases
659 result += cj[indj] * _cubic(newx - thisj)
660 res[cond3] = result
661 return res
