140def test_pickle_load_from_subprocess(fig_test, fig_ref, tmp_path):
141 _generate_complete_test_figure(fig_ref)
142
143 fp = tmp_path / 'sinus.pickle'
144 assert not fp.exists()
145
146 with fp.open('wb') as file:
147 pickle.dump(fig_ref, file, pickle.HIGHEST_PROTOCOL)
148 assert fp.exists()
149
150 proc = subprocess_run_helper(
151 _pickle_load_subprocess,
152 timeout=60,
153 extra_env={
154 "PICKLE_FILE_PATH": str(fp),
155 "MPLBACKEND": "Agg",
156 # subprocess_run_helper will set SOURCE_DATE_EPOCH=0, so for a dirty tree,

158 # version compatibility warning, force setuptools-scm to use the same
159 # version as us.
160 "SETUPTOOLS_SCM_PRETEND_VERSION_FOR_MATPLOTLIB": mpl.__version__,
161 },
162 )
163
164 loaded_fig = pickle.loads(ast.literal_eval(proc.stdout))
165

221class TransformBlob:
222 def __init__(self):
223 self.identity = mtransforms.IdentityTransform()
224 self.identity2 = mtransforms.IdentityTransform()
225 # Force use of the more complex composition.
226 self.composite = mtransforms.CompositeGenericTransform(

228 self.identity2)
229 # Check parent -> child links of TransformWrapper.
230 self.wrapper = mtransforms.TransformWrapper(self.composite)

232 self.composite2 = mtransforms.CompositeGenericTransform(

234 self.identity)
