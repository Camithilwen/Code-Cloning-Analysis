14class Spine(mpatches.Patch):
15 """
16 An axis spine -- the line noting the data area boundaries.
17
18 Spines are the lines connecting the axis tick marks and noting the
19 boundaries of the data area. They can be placed at arbitrary
20 positions. See `~.Spine.set_position` for more information.
21
22 The default position is ``('outward', 0)``.
23
24 Spines are subclasses of `.Patch`, and inherit much of their behavior.
25
26 Spines draw a line, a circle, or an arc depending on if
27 `~.Spine.set_patch_line`, `~.Spine.set_patch_circle`, or
28 `~.Spine.set_patch_arc` has been called. Line-like is the default.
29
30 For examples see :ref:`spines_examples`.

300 def set_position(self, position):
301 """
302 Set the position of the spine.
303
304 Spine position is specified by a 2 tuple of (position type,
305 amount). The position types are:
306
307 * 'outward': place the spine out from the data area by the specified
308 number of points. (Negative values place the spine inwards.)
309 * 'axes': place the spine at the specified Axes coordinate (0 to 1).
310 * 'data': place the spine at the specified data coordinate.
311
312 Additionally, shorthand notations define a special positions:
313
314 * 'center' -> ``('axes', 0.5)``
315 * 'zero' -> ``('data', 0.0)``
316
317 Examples
318 --------
319 :doc:`/gallery/spines/spine_placement_demo`
320 """
321 if position in ('center', 'zero'): # special positions
322 pass
323 else:
324 if len(position) != 2:
325 raise ValueError("position should be 'center' or 2-tuple")
326 if position[0] not in ['outward', 'axes', 'data']:
327 raise ValueError("position[0] should be one of 'outward', "
328 "'axes', or 'data' ")
329 self._position = position
330 self.set_transform(self.get_spine_transform())
331 if self.axis is not None:
332 self.axis.reset_ticks()
333 self.stale = True

340 def get_spine_transform(self):
341 """Return the spine transform."""
342 self._ensure_position_is_set()
343
344 position = self._position
345 if isinstance(position, str):
346 if position == 'center':
347 position = ('axes', 0.5)
348 elif position == 'zero':
349 position = ('data', 0)
350 assert len(position) == 2, 'position should be 2-tuple'
351 position_type, amount = position
352 _api.check_in_list(['axes', 'outward', 'data'],
353 position_type=position_type)
354 if self.spine_type in ['left', 'right']:
355 base_transform = self.axes.get_yaxis_transform(which='grid')
356 elif self.spine_type in ['top', 'bottom']:
357 base_transform = self.axes.get_xaxis_transform(which='grid')
358 else:
359 raise ValueError(f'unknown spine spine_type: {self.spine_type!r}')
360
361 if position_type == 'outward':
362 if amount == 0: # short circuit commonest case
363 return base_transform
364 else:
365 offset_vec = {'left': (-1, 0), 'right': (1, 0),
366 'bottom': (0, -1), 'top': (0, 1),
367 }[self.spine_type]
368 # calculate x and y offset in dots
369 offset_dots = amount * np.array(offset_vec) / 72
370 return (base_transform
371 + mtransforms.ScaledTranslation(
372 *offset_dots, self.get_figure(root=False).dpi_scale_trans))
373 elif position_type == 'axes':
374 if self.spine_type in ['left', 'right']:
375 # keep y unchanged, fix x at amount
376 return (mtransforms.Affine2D.from_values(0, 0, 0, 1, amount, 0)
377 + base_transform)
378 elif self.spine_type in ['bottom', 'top']:
379 # keep x unchanged, fix y at amount
380 return (mtransforms.Affine2D.from_values(1, 0, 0, 0, 0, amount)
381 + base_transform)
382 elif position_type == 'data':
383 if self.spine_type in ('right', 'top'):
384 # The right and top spines have a default position of 1 in
385 # axes coordinates. When specifying the position in data
386 # coordinates, we need to calculate the position relative to 0.
387 amount -= 1
388 if self.spine_type in ('left', 'right'):
389 return mtransforms.blended_transform_factory(
390 mtransforms.Affine2D().translate(amount, 0)
391 + self.axes.transData,
392 self.axes.transData)
393 elif self.spine_type in ('bottom', 'top'):
394 return mtransforms.blended_transform_factory(
395 self.axes.transData,
396 mtransforms.Affine2D().translate(0, amount)
397 + self.axes.transData)

488class SpinesProxy:
489 """
490 A proxy to broadcast ``set_*()`` and ``set()`` method calls to contained `.Spines`.
491
492 The proxy cannot be used for any other operations on its members.
493
494 The supported methods are determined dynamically based on the contained
495 spines. If not all spines support a given method, it's executed only on
496 the subset of spines that support it.
497 """
498 def __init__(self, spine_dict):
499 self._spine_dict = spine_dict
500
501 def __getattr__(self, name):
502 broadcast_targets = [spine for spine in self._spine_dict.values()
503 if hasattr(spine, name)]
504 if (name != 'set' and not name.startswith('set_')) or not broadcast_targets:
505 raise AttributeError(
506 f"'SpinesProxy' object has no attribute '{name}'")
507
508 def x(_targets, _funcname, *args, **kwargs):
509 for spine in _targets:
510 getattr(spine, _funcname)(*args, **kwargs)
511 x = functools.partial(x, broadcast_targets, name)
512 x.__doc__ = broadcast_targets[0].__doc__
513 return x
514
515 def __dir__(self):
516 names = []
517 for spine in self._spine_dict.values():
518 names.extend(name
519 for name in dir(spine) if name.startswith('set_'))
520 return list(sorted(set(names)))
