331 def help_is_isomorphic_randperm(self, nobs, nclusters, noniso=False, nerrors=0):
332 for k in range(3):
333 a = np.int_(np.random.rand(nobs) * nclusters)
334 b = np.zeros(a.size, dtype=np.int_)
335 P = np.random.permutation(nclusters)
336 for i in range(0, a.shape[0]):
337 b[i] = P[a[i]]
338 if noniso:
339 Q = np.random.permutation(nobs)
340 b[Q[0:nerrors]] += 1

984def calculate_maximum_distances(Z):
985 # Used for testing correctness of maxdists.
986 n = Z.shape[0] + 1
987 B = np.zeros((n-1,))
988 q = np.zeros((3,))
989 for i in range(0, n - 1):
990 q[:] = 0.0
991 left = Z[i, 0]
992 right = Z[i, 1]
993 if left >= n:
994 q[0] = B[int(left) - n]
995 if right >= n:
996 q[1] = B[int(right) - n]
997 q[2] = Z[i, 2]
998 B[i] = q.max()
999 return B

1002def calculate_maximum_inconsistencies(Z, R, k=3):
1003 # Used for testing correctness of maxinconsts.
1004 n = Z.shape[0] + 1
1005 B = np.zeros((n-1,))
1006 q = np.zeros((3,))
1007 for i in range(0, n - 1):
1008 q[:] = 0.0
1009 left = Z[i, 0]
1010 right = Z[i, 1]
1011 if left >= n:
1012 q[0] = B[int(left) - n]
1013 if right >= n:
1014 q[1] = B[int(right) - n]
1015 q[2] = R[i, k]
1016 B[i] = q.max()
1017 return B
