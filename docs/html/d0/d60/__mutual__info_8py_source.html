20def _compute_mi_cc(x, y, n_neighbors):
21 """Compute mutual information between two continuous variables.
22
23 Parameters
24 ----------
25 x, y : ndarray, shape (n_samples,)
26 Samples of two continuous random variables, must have an identical
27 shape.
28
29 n_neighbors : int
30 Number of nearest neighbors to search for each point, see [1]_.
31
32 Returns
33 -------
34 mi : float
35 Estimated mutual information in nat units. If it turned out to be
36 negative it is replaced by 0.
37
38 Notes
39 -----
40 True mutual information can't be negative. If its estimate by a numerical
41 method is negative, it means (providing the method is adequate) that the
42 mutual information is close to 0 and replacing it by 0 is a reasonable
43 strategy.
44
45 References
46 ----------
47 .. [1] A. Kraskov, H. Stogbauer and P. Grassberger, "Estimating mutual
48 information". Phys. Rev. E 69, 2004.
49 """
50 n_samples = x.size
51
52 x = x.reshape((-1, 1))
53 y = y.reshape((-1, 1))
54 xy = np.hstack((x, y))
55
56 # Here we rely on NearestNeighbors to select the fastest algorithm.
57 nn = NearestNeighbors(metric="chebyshev", n_neighbors=n_neighbors)
58
59 nn.fit(xy)
60 radius = nn.kneighbors()[0]
61 radius = np.nextafter(radius[:, -1], 0)
62
63 # KDTree is explicitly fit to allow for the querying of number of
64 # neighbors within a specified radius
65 kd = KDTree(x, metric="chebyshev")
66 nx = kd.query_radius(x, radius, count_only=True, return_distance=False)
67 nx = np.array(nx) - 1.0
68
69 kd = KDTree(y, metric="chebyshev")
70 ny = kd.query_radius(y, radius, count_only=True, return_distance=False)
71 ny = np.array(ny) - 1.0
72
73 mi = (
74 digamma(n_samples)
75 + digamma(n_neighbors)
76 - np.mean(digamma(nx + 1))
77 - np.mean(digamma(ny + 1))
78 )
79
80 return max(0, mi)

83def _compute_mi_cd(c, d, n_neighbors):
84 """Compute mutual information between continuous and discrete variables.
85
86 Parameters
87 ----------
88 c : ndarray, shape (n_samples,)
89 Samples of a continuous random variable.
90
91 d : ndarray, shape (n_samples,)
92 Samples of a discrete random variable.
93
94 n_neighbors : int
95 Number of nearest neighbors to search for each point, see [1]_.
96
97 Returns
98 -------
99 mi : float
100 Estimated mutual information in nat units. If it turned out to be
101 negative it is replaced by 0.
102
103 Notes
104 -----
105 True mutual information can't be negative. If its estimate by a numerical
106 method is negative, it means (providing the method is adequate) that the
107 mutual information is close to 0 and replacing it by 0 is a reasonable
108 strategy.
109
110 References
111 ----------
112 .. [1] B. C. Ross "Mutual Information between Discrete and Continuous
113 Data Sets". PLoS ONE 9(2), 2014.
114 """
115 n_samples = c.shape[0]
116 c = c.reshape((-1, 1))
117
118 radius = np.empty(n_samples)
119 label_counts = np.empty(n_samples)
120 k_all = np.empty(n_samples)
121 nn = NearestNeighbors()
122 for label in np.unique(d):
123 mask = d == label
124 count = np.sum(mask)
125 if count > 1:
126 k = min(n_neighbors, count - 1)
127 nn.set_params(n_neighbors=k)
128 nn.fit(c[mask])
129 r = nn.kneighbors()[0]
130 radius[mask] = np.nextafter(r[:, -1], 0)
131 k_all[mask] = k
132 label_counts[mask] = count
133
134 # Ignore points with unique labels.
135 mask = label_counts > 1
136 n_samples = np.sum(mask)
137 label_counts = label_counts[mask]
138 k_all = k_all[mask]
139 c = c[mask]
140 radius = radius[mask]
141
142 kd = KDTree(c)
143 m_all = kd.query_radius(c, radius, count_only=True, return_distance=False)
144 m_all = np.array(m_all)
145
146 mi = (
147 digamma(n_samples)
148 + np.mean(digamma(k_all))
149 - np.mean(digamma(label_counts))
150 - np.mean(digamma(m_all))
151 )
152
153 return max(0, mi)

156def _compute_mi(x, y, x_discrete, y_discrete, n_neighbors=3):
157 """Compute mutual information between two variables.
158
159 This is a simple wrapper which selects a proper function to call based on
160 whether `x` and `y` are discrete or not.
161 """
162 if x_discrete and y_discrete:
163 return mutual_info_score(x, y)
164 elif x_discrete and not y_discrete:
165 return _compute_mi_cd(y, x, n_neighbors)
166 elif not x_discrete and y_discrete:
167 return _compute_mi_cd(x, y, n_neighbors)
168 else:
169 return _compute_mi_cc(x, y, n_neighbors)

172def _iterate_columns(X, columns=None):
173 """Iterate over columns of a matrix.
174
175 Parameters
176 ----------
177 X : ndarray or csc_matrix, shape (n_samples, n_features)
178 Matrix over which to iterate.
179
180 columns : iterable or None, default=None
181 Indices of columns to iterate over. If None, iterate over all columns.
182
183 Yields
184 ------
185 x : ndarray, shape (n_samples,)
186 Columns of `X` in dense format.
187 """
188 if columns is None:
189 columns = range(X.shape[1])
190
191 if issparse(X):
192 for i in columns:
193 x = np.zeros(X.shape[0])
194 start_ptr, end_ptr = X.indptr[i], X.indptr[i + 1]
195 x[X.indices[start_ptr:end_ptr]] = X.data[start_ptr:end_ptr]
196 yield x
197 else:
198 for i in columns:
199 yield X[:, i]

202def _estimate_mi(
203 X,
204 y,
205 *,
206 discrete_features="auto",
207 discrete_target=False,
208 n_neighbors=3,
209 copy=True,
210 random_state=None,
211 n_jobs=None,
212):
213 """Estimate mutual information between the features and the target.
214
215 Parameters
216 ----------
217 X : array-like or sparse matrix, shape (n_samples, n_features)
218 Feature matrix.
219
220 y : array-like of shape (n_samples,)
221 Target vector.
222
223 discrete_features : {'auto', bool, array-like}, default='auto'
224 If bool, then determines whether to consider all features discrete
225 or continuous. If array, then it should be either a boolean mask
226 with shape (n_features,) or array with indices of discrete features.
227 If 'auto', it is assigned to False for dense `X` and to True for
228 sparse `X`.
229
230 discrete_target : bool, default=False
231 Whether to consider `y` as a discrete variable.
232
233 n_neighbors : int, default=3
234 Number of neighbors to use for MI estimation for continuous variables,
235 see [1]_ and [2]_. Higher values reduce variance of the estimation, but
236 could introduce a bias.
237
238 copy : bool, default=True
239 Whether to make a copy of the given data. If set to False, the initial
240 data will be overwritten.
241
242 random_state : int, RandomState instance or None, default=None
243 Determines random number generation for adding small noise to
244 continuous variables in order to remove repeated values.
245 Pass an int for reproducible results across multiple function calls.
246 See :term:`Glossary <random_state>`.
247
248 n_jobs : int, default=None
249 The number of jobs to use for computing the mutual information.
250 The parallelization is done on the columns of `X`.
251 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
252 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
253 for more details.
254
255 .. versionadded:: 1.5
256
257
258 Returns
259 -------
260 mi : ndarray, shape (n_features,)
261 Estimated mutual information between each feature and the target in
262 nat units. A negative value will be replaced by 0.
263
264 References
265 ----------
266 .. [1] A. Kraskov, H. Stogbauer and P. Grassberger, "Estimating mutual
267 information". Phys. Rev. E 69, 2004.
268 .. [2] B. C. Ross "Mutual Information between Discrete and Continuous
269 Data Sets". PLoS ONE 9(2), 2014.
270 """
271 X, y = check_X_y(X, y, accept_sparse="csc", y_numeric=not discrete_target)
272 n_samples, n_features = X.shape
273
274 if isinstance(discrete_features, (str, bool)):
275 if isinstance(discrete_features, str):
276 if discrete_features == "auto":
277 discrete_features = issparse(X)
278 else:
279 raise ValueError("Invalid string value for discrete_features.")
280 discrete_mask = np.empty(n_features, dtype=bool)
281 discrete_mask.fill(discrete_features)
282 else:
283 discrete_features = check_array(discrete_features, ensure_2d=False)
284 if discrete_features.dtype != "bool":
285 discrete_mask = np.zeros(n_features, dtype=bool)
286 discrete_mask[discrete_features] = True
287 else:
288 discrete_mask = discrete_features
289
290 continuous_mask = ~discrete_mask
291 if np.any(continuous_mask) and issparse(X):
292 raise ValueError("Sparse matrix `X` can't have continuous features.")
293
294 rng = check_random_state(random_state)
295 if np.any(continuous_mask):
296 X = X.astype(np.float64, copy=copy)
297 X[:, continuous_mask] = scale(
298 X[:, continuous_mask], with_mean=False, copy=False
299 )
300
301 # Add small noise to continuous features as advised in Kraskov et. al.
302 means = np.maximum(1, np.mean(np.abs(X[:, continuous_mask]), axis=0))
303 X[:, continuous_mask] += (
304 1e-10
305 * means
306 * rng.standard_normal(size=(n_samples, np.sum(continuous_mask)))
307 )
308
309 if not discrete_target:
310 y = scale(y, with_mean=False)
311 y += (
312 1e-10
313 * np.maximum(1, np.mean(np.abs(y)))
314 * rng.standard_normal(size=n_samples)
315 )
316
317 mi = Parallel(n_jobs=n_jobs)(
318 delayed(_compute_mi)(x, y, discrete_feature, discrete_target, n_neighbors)
319 for x, discrete_feature in zip(_iterate_columns(X), discrete_mask)
320 )
321
322 return np.array(mi)

352 values mean higher dependency.
353

355 from k-nearest neighbors distances as described in [2]_ and [3]_. Both
356 methods are based on the idea originally proposed in [4]_.
357

370 If bool, then determines whether to consider all features discrete

372 with shape (n_features,) or array with indices of discrete features.
373 If 'auto', it is assigned to False for dense `X` and to True for
374 sparse `X`.
375
376 n_neighbors : int, default=3
377 Number of neighbors to use for MI estimation for continuous variables,
378 see [2]_ and [3]_. Higher values reduce variance of the estimation, but
379 could introduce a bias.
380

383 data will be overwritten.

387 continuous variables in order to remove repeated values.
388 Pass an int for reproducible results across multiple function calls.
389 See :term:`Glossary <random_state>`.
390
391 n_jobs : int, default=None

393 The parallelization is done on the columns of `X`.
394
395 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
396 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
397 for more details.
398
399 .. versionadded:: 1.5
400
401 Returns
402 -------
403 mi : ndarray, shape (n_features,)
404 Estimated mutual information between each feature and the target in
405 nat units.
406
407 Notes
