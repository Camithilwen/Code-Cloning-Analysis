19def _huber_loss_and_gradient(w, X, y, epsilon, alpha, sample_weight=None):
20 """Returns the Huber loss and the gradient.
21
22 Parameters
23 ----------
24 w : ndarray, shape (n_features + 1,) or (n_features + 2,)
25 Feature vector.
26 w[:n_features] gives the coefficients
27 w[-1] gives the scale factor and if the intercept is fit w[-2]
28 gives the intercept factor.
29
30 X : ndarray of shape (n_samples, n_features)
31 Input data.
32
33 y : ndarray of shape (n_samples,)
34 Target vector.
35
36 epsilon : float
37 Robustness of the Huber estimator.
38
39 alpha : float
40 Regularization parameter.
41
42 sample_weight : ndarray of shape (n_samples,), default=None
43 Weight assigned to each sample.
44
45 Returns
46 -------
47 loss : float
48 Huber loss.
49
50 gradient : ndarray, shape (len(w))
51 Returns the derivative of the Huber loss with respect to each
52 coefficient, intercept and the scale as a vector.
53 """
54 _, n_features = X.shape
55 fit_intercept = n_features + 2 == w.shape[0]
56 if fit_intercept:
57 intercept = w[-2]
58 sigma = w[-1]
59 w = w[:n_features]
60 n_samples = np.sum(sample_weight)
61
62 # Calculate the values where |y - X'w -c / sigma| > epsilon
63 # The values above this threshold are outliers.
64 linear_loss = y - safe_sparse_dot(X, w)
65 if fit_intercept:
66 linear_loss -= intercept
67 abs_linear_loss = np.abs(linear_loss)
68 outliers_mask = abs_linear_loss > epsilon * sigma
69
70 # Calculate the linear loss due to the outliers.
71 # This is equal to (2 * M * |y - X'w -c / sigma| - M**2) * sigma
72 outliers = abs_linear_loss[outliers_mask]
73 num_outliers = np.count_nonzero(outliers_mask)
74 n_non_outliers = X.shape[0] - num_outliers
75
76 # n_sq_outliers includes the weight give to the outliers while
77 # num_outliers is just the number of outliers.
78 outliers_sw = sample_weight[outliers_mask]
79 n_sw_outliers = np.sum(outliers_sw)
80 outlier_loss = (
81 2.0 * epsilon * np.sum(outliers_sw * outliers)
82 - sigma * n_sw_outliers * epsilon**2
83 )
84
85 # Calculate the quadratic loss due to the non-outliers.-
86 # This is equal to |(y - X'w - c)**2 / sigma**2| * sigma
87 non_outliers = linear_loss[~outliers_mask]
88 weighted_non_outliers = sample_weight[~outliers_mask] * non_outliers
89 weighted_loss = np.dot(weighted_non_outliers.T, non_outliers)
90 squared_loss = weighted_loss / sigma
91
92 if fit_intercept:
93 grad = np.zeros(n_features + 2)
94 else:
95 grad = np.zeros(n_features + 1)
96
97 # Gradient due to the squared loss.
98 X_non_outliers = -axis0_safe_slice(X, ~outliers_mask, n_non_outliers)
99 grad[:n_features] = (
100 2.0 / sigma * safe_sparse_dot(weighted_non_outliers, X_non_outliers)
101 )
102
103 # Gradient due to the linear loss.
104 signed_outliers = np.ones_like(outliers)
105 signed_outliers_mask = linear_loss[outliers_mask] < 0
106 signed_outliers[signed_outliers_mask] = -1.0
107 X_outliers = axis0_safe_slice(X, outliers_mask, num_outliers)
108 sw_outliers = sample_weight[outliers_mask] * signed_outliers
109 grad[:n_features] -= 2.0 * epsilon * (safe_sparse_dot(sw_outliers, X_outliers))
110
111 # Gradient due to the penalty.
112 grad[:n_features] += alpha * 2.0 * w
113
114 # Gradient due to sigma.
115 grad[-1] = n_samples
116 grad[-1] -= n_sw_outliers * epsilon**2
117 grad[-1] -= squared_loss / sigma
118
119 # Gradient due to the intercept.
120 if fit_intercept:
121 grad[-2] = -2.0 * np.sum(weighted_non_outliers) / sigma
122 grad[-2] -= 2.0 * epsilon * np.sum(sw_outliers)
123
124 loss = n_samples * sigma + squared_loss + outlier_loss
125 loss += alpha * np.dot(w, w)
126 return loss, grad

129class HuberRegressor(LinearModel, RegressorMixin, BaseEstimator):
130 """L2-regularized linear regression model that is robust to outliers.
131
132 The Huber Regressor optimizes the squared loss for the samples where
133 ``|(y - Xw - c) / sigma| < epsilon`` and the absolute loss for the samples
134 where ``|(y - Xw - c) / sigma| > epsilon``, where the model coefficients
135 ``w``, the intercept ``c`` and the scale ``sigma`` are parameters
136 to be optimized. The parameter `sigma` makes sure that if `y` is scaled up
137 or down by a certain factor, one does not need to rescale `epsilon` to
138 achieve the same robustness. Note that this does not take into account
139 the fact that the different features of `X` may be of different scales.
140
141 The Huber loss function has the advantage of not being heavily influenced
142 by the outliers while not completely ignoring their effect.
143
144 Read more in the :ref:`User Guide <huber_regression>`
145
146 .. versionadded:: 0.18
147
148 Parameters
149 ----------
150 epsilon : float, default=1.35
151 The parameter epsilon controls the number of samples that should be
152 classified as outliers. The smaller the epsilon, the more robust it is
153 to outliers. Epsilon must be in the range `[1, inf)`.
154
155 max_iter : int, default=100
156 Maximum number of iterations that
157 ``scipy.optimize.minimize(method="L-BFGS-B")`` should run for.
158
159 alpha : float, default=0.0001
160 Strength of the squared L2 regularization. Note that the penalty is
161 equal to ``alpha * ||w||^2``.
162 Must be in the range `[0, inf)`.
163
164 warm_start : bool, default=False
165 This is useful if the stored attributes of a previously used model
166 has to be reused. If set to False, then the coefficients will
167 be rewritten for every call to fit.
168 See :term:`the Glossary <warm_start>`.
169
170 fit_intercept : bool, default=True
171 Whether or not to fit the intercept. This can be set to False
172 if the data is already centered around the origin.
173
174 tol : float, default=1e-05
175 The iteration will stop when
176 ``max{|proj g_i | i = 1, ..., n}`` <= ``tol``
177 where pg_i is the i-th component of the projected gradient.
178
179 Attributes
180 ----------
181 coef_ : array, shape (n_features,)
182 Features got by optimizing the L2-regularized Huber loss.
183
184 intercept_ : float
185 Bias.
186
187 scale_ : float
188 The value by which ``|y - Xw - c|`` is scaled down.
189
190 n_features_in_ : int
191 Number of features seen during :term:`fit`.
192
193 .. versionadded:: 0.24
194
195 feature_names_in_ : ndarray of shape (`n_features_in_`,)
196 Names of features seen during :term:`fit`. Defined only when `X`
197 has feature names that are all strings.
198
199 .. versionadded:: 1.0
200
201 n_iter_ : int
202 Number of iterations that
203 ``scipy.optimize.minimize(method="L-BFGS-B")`` has run for.
204
205 .. versionchanged:: 0.20
206
207 In SciPy <= 1.0.0 the number of lbfgs iterations may exceed
208 ``max_iter``. ``n_iter_`` will now report at most ``max_iter``.
209
210 outliers_ : array, shape (n_samples,)
211 A boolean mask which is set to True where the samples are identified
212 as outliers.
213
214 See Also
215 --------
216 RANSACRegressor : RANSAC (RANdom SAmple Consensus) algorithm.
217 TheilSenRegressor : Theil-Sen Estimator robust multivariate regression model.
218 SGDRegressor : Fitted by minimizing a regularized empirical loss with SGD.
219
220 References
221 ----------
222 .. [1] Peter J. Huber, Elvezio M. Ronchetti, Robust Statistics
223 Concomitant scale estimates, p. 172
224 .. [2] Art B. Owen (2006), `A robust hybrid of lasso and ridge regression.
225 <https://artowen.su.domains/reports/hhu.pdf>`_
226
227 Examples
228 --------
229 >>> import numpy as np
230 >>> from sklearn.linear_model import HuberRegressor, LinearRegression
231 >>> from sklearn.datasets import make_regression
232 >>> rng = np.random.RandomState(0)
233 >>> X, y, coef = make_regression(
234 ... n_samples=200, n_features=2, noise=4.0, coef=True, random_state=0)
235 >>> X[:4] = rng.uniform(10, 20, (4, 2))
236 >>> y[:4] = rng.uniform(10, 20, 4)
237 >>> huber = HuberRegressor().fit(X, y)
238 >>> huber.score(X, y)
239 -7.284
240 >>> huber.predict(X[:1,])
241 array([806.7200])
242 >>> linear = LinearRegression().fit(X, y)
243 >>> print("True coefficients:", coef)
244 True coefficients: [20.4923... 34.1698...]
245 >>> print("Huber coefficients:", huber.coef_)
246 Huber coefficients: [17.7906... 31.0106...]
247 >>> print("Linear Regression coefficients:", linear.coef_)
248 Linear Regression coefficients: [-1.9221... 7.0226...]
249 """

277 @_fit_context(prefer_skip_nested_validation=True)
278 def fit(self, X, y, sample_weight=None):
279 """Fit the model according to the given training data.
280
281 Parameters
282 ----------
283 X : array-like, shape (n_samples, n_features)
284 Training vector, where `n_samples` is the number of samples and
285 `n_features` is the number of features.
286
287 y : array-like, shape (n_samples,)
288 Target vector relative to X.
289
290 sample_weight : array-like, shape (n_samples,)
291 Weight given to each sample.
292
293 Returns
294 -------
295 self : object
296 Fitted `HuberRegressor` estimator.
297 """
298 X, y = validate_data(
299 self,
300 X,
301 y,
302 copy=False,
303 accept_sparse=["csr"],
304 y_numeric=True,
305 dtype=[np.float64, np.float32],
306 )
307
308 sample_weight = _check_sample_weight(sample_weight, X)
309
310 if self.warm_start and hasattr(self, "coef_"):
311 parameters = np.concatenate((self.coef_, [self.intercept_, self.scale_]))
312 else:
313 if self.fit_intercept:
314 parameters = np.zeros(X.shape[1] + 2)
315 else:
316 parameters = np.zeros(X.shape[1] + 1)
317 # Make sure to initialize the scale parameter to a strictly
318 # positive value:
319 parameters[-1] = 1
320
321 # Sigma or the scale factor should be non-negative.
322 # Setting it to be zero might cause undefined bounds hence we set it
323 # to a value close to zero.
324 bounds = np.tile([-np.inf, np.inf], (parameters.shape[0], 1))
325 bounds[-1][0] = np.finfo(np.float64).eps * 10
326
327 opt_res = optimize.minimize(
328 _huber_loss_and_gradient,
329 parameters,
330 method="L-BFGS-B",
331 jac=True,
332 args=(X, y, self.epsilon, self.alpha, sample_weight),
333 options={
334 "maxiter": self.max_iter,
335 "gtol": self.tol,
336 **_get_additional_lbfgs_options_dict("iprint", -1),
337 },
338 bounds=bounds,
339 )
340
341 parameters = opt_res.x
342
343 if opt_res.status == 2:
344 raise ValueError(
345 "HuberRegressor convergence failed: l-BFGS-b solver terminated with %s"
346 % opt_res.message
347 )
348 self.n_iter_ = _check_optimize_result("lbfgs", opt_res, self.max_iter)
349 self.scale_ = parameters[-1]
350 if self.fit_intercept:
351 self.intercept_ = parameters[-2]
352 else:
353 self.intercept_ = 0.0
354 self.coef_ = parameters[: X.shape[1]]
355
356 residual = np.abs(y - safe_sparse_dot(X, self.coef_) - self.intercept_)
357 self.outliers_ = residual > self.scale_ * self.epsilon
358 return self
