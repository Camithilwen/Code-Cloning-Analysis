12def test_map_coordinates_dts():
13 # check that ndimage accepts different data types for interpolation
14 data = np.array([[4, 1, 3, 2],
15 [7, 6, 8, 5],
16 [3, 5, 3, 6]])
17 shifted_data = np.array([[0, 0, 0, 0],
18 [0, 4, 1, 3],
19 [0, 7, 6, 8]])
20 idx = np.indices(data.shape)
21 dts = (np.uint8, np.uint16, np.uint32, np.uint64,
22 np.int8, np.int16, np.int32, np.int64,
23 np.intp, np.uintp, np.float32, np.float64)
24 for order in range(0, 6):
25 for data_dt in dts:
26 these_data = data.astype(data_dt)
27 for coord_dt in dts:
28 # affine mapping
29 mat = np.eye(2, dtype=coord_dt)
30 off = np.zeros((2,), dtype=coord_dt)
31 out = ndimage.affine_transform(these_data, mat, off)
32 assert_array_almost_equal(these_data, out)
33 # map coordinates
34 coords_m1 = idx.astype(coord_dt) - 1
35 coords_p10 = idx.astype(coord_dt) + 10
36 out = ndimage.map_coordinates(these_data, coords_m1, order=order)
37 assert_array_almost_equal(out, shifted_data)
38 # check constant fill works
39 out = ndimage.map_coordinates(these_data, coords_p10, order=order)
40 assert_array_almost_equal(out, np.zeros((3,4)))
41 # check shift and zoom
42 out = ndimage.shift(these_data, 1)
43 assert_array_almost_equal(out, shifted_data)
44 out = ndimage.zoom(these_data, 1)
45 assert_array_almost_equal(these_data, out)

49def test_uint64_max():
50 # Test interpolation respects uint64 max. Reported to fail at least on
51 # win32 (due to the 32 bit visual C compiler using signed int64 when
52 # converting between uint64 to double) and Debian on s390x.
53 # Interpolation is always done in double precision floating point, so
54 # we use the largest uint64 value for which int(float(big)) still fits
55 # in a uint64.
56 big = 2**64 - 1025
57 arr = np.array([big, big, big], dtype=np.uint64)
58 # Tests geometric transform (map_coordinates, affine_transform)
59 inds = np.indices(arr.shape) - 0.1
60 x = ndimage.map_coordinates(arr, inds)
61 assert_(x[1] == int(float(big)))
62 assert_(x[2] == int(float(big)))
63 # Tests zoom / shift
64 x = ndimage.shift(arr, 0.1)
65 assert_(x[1] == int(float(big)))
66 assert_(x[2] == int(float(big)))
