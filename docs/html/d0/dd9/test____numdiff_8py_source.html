459 # check upper bound of error on the derivative for 2-point
460 def f(x):
461 return np.sin(x)
462 def g(x):
463 return np.cos(x)
464 def hess(x):
465 return -np.sin(x)
466
467 def calc_atol(h, x0, f, hess, EPS):
468 # truncation error
469 t0 = h / 2 * max(np.abs(hess(x0)), np.abs(hess(x0 + h)))
470 # roundoff error. There may be a divisor (>1) missing from
471 # the following line, so this contribution is possibly
472 # overestimated
473 t1 = EPS / h * max(np.abs(f(x0)), np.abs(f(x0 + h)))
474 return t0 + t1
475
476 for dtype in [np.float16, np.float32, np.float64]:
477 EPS = np.finfo(dtype).eps
478 x0 = np.array(1.0).astype(dtype)
479 h = _compute_absolute_step(None, x0, f(x0), '2-point')
480 atol = calc_atol(h, x0, f, hess, EPS)
481 err = approx_derivative(f, x0, method='2-point',
482 abs_step=h) - g(x0)
483 assert abs(err) < atol

520 def jac(self, x):
521 n = x.size
522 J = np.zeros((n, n))
523 J[0, 0] = -4 * x[0]
524 J[0, 1] = 6 * x[1]**2
525 for i in range(1, n - 1):

721def test_absolute_step_sign():
722 # test for gh12487
723 # if an absolute step is specified for 2-point differences make sure that
724 # the side corresponds to the step. i.e. if step is positive then forward
725 # differences should be used, if step is negative then backwards
726 # differences should be used.
727
728 # function has double discontinuity at x = [-1, -1]
729 # first component is \/, second component is /\
730 def f(x):
731 return -np.abs(x[0] + 1) + np.abs(x[1] + 1)
732
733 # check that the forward difference is used
734 grad = approx_derivative(f, [-1, -1], method='2-point', abs_step=1e-8)
735 assert_allclose(grad, [-1.0, 1.0])
736
737 # check that the backwards difference is used
738 grad = approx_derivative(f, [-1, -1], method='2-point', abs_step=-1e-8)
739 assert_allclose(grad, [1.0, -1.0])
740
741 # check that the forwards difference is used with a step for both
742 # parameters
743 grad = approx_derivative(
744 f, [-1, -1], method='2-point', abs_step=[1e-8, 1e-8]
745 )
746 assert_allclose(grad, [-1.0, 1.0])
747
748 # check that we can mix forward/backwards steps.
749 grad = approx_derivative(
750 f, [-1, -1], method='2-point', abs_step=[1e-8, -1e-8]
751 )
752 assert_allclose(grad, [-1.0, -1.0])
753 grad = approx_derivative(
754 f, [-1, -1], method='2-point', abs_step=[-1e-8, 1e-8]
755 )
756 assert_allclose(grad, [1.0, 1.0])
757
758 # the forward step should reverse to a backwards step if it runs into a
759 # bound
760 # This is kind of tested in TestAdjustSchemeToBounds, but only for a lower level
761 # function.
762 grad = approx_derivative(
763 f, [-1, -1], method='2-point', abs_step=1e-8,
764 bounds=(-np.inf, -1)
765 )
766 assert_allclose(grad, [1.0, -1.0])
767
768 grad = approx_derivative(
769 f, [-1, -1], method='2-point', abs_step=-1e-8, bounds=(-1, np.inf)
770 )
771 assert_allclose(grad, [-1.0, 1.0])
