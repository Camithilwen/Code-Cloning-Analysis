36class _BinaryGaussianProcessClassifierLaplace(BaseEstimator):
37 """Binary Gaussian process classification based on Laplace approximation.
38
39 The implementation is based on Algorithm 3.1, 3.2, and 5.1 from [RW2006]_.
40
41 Internally, the Laplace approximation is used for approximating the
42 non-Gaussian posterior by a Gaussian.
43
44 Currently, the implementation is restricted to using the logistic link
45 function.
46
47 .. versionadded:: 0.18
48
49 Parameters
50 ----------
51 kernel : kernel instance, default=None
52 The kernel specifying the covariance function of the GP. If None is
53 passed, the kernel "1.0 * RBF(1.0)" is used as default. Note that
54 the kernel's hyperparameters are optimized during fitting.
55
56 optimizer : 'fmin_l_bfgs_b' or callable, default='fmin_l_bfgs_b'
57 Can either be one of the internally supported optimizers for optimizing
58 the kernel's parameters, specified by a string, or an externally
59 defined optimizer passed as a callable. If a callable is passed, it
60 must have the signature::
61
62 def optimizer(obj_func, initial_theta, bounds):
63 # * 'obj_func' is the objective function to be maximized, which
64 # takes the hyperparameters theta as parameter and an
65 # optional flag eval_gradient, which determines if the
66 # gradient is returned additionally to the function value
67 # * 'initial_theta': the initial value for theta, which can be
68 # used by local optimizers
69 # * 'bounds': the bounds on the values of theta
70 ....
71 # Returned are the best found hyperparameters theta and
72 # the corresponding value of the target function.
73 return theta_opt, func_min
74
75 Per default, the 'L-BFGS-B' algorithm from scipy.optimize.minimize
76 is used. If None is passed, the kernel's parameters are kept fixed.
77 Available internal optimizers are::
78
79 'fmin_l_bfgs_b'
80
81 n_restarts_optimizer : int, default=0
82 The number of restarts of the optimizer for finding the kernel's
83 parameters which maximize the log-marginal likelihood. The first run
84 of the optimizer is performed from the kernel's initial parameters,
85 the remaining ones (if any) from thetas sampled log-uniform randomly
86 from the space of allowed theta-values. If greater than 0, all bounds
87 must be finite. Note that n_restarts_optimizer=0 implies that one
88 run is performed.
89
90 max_iter_predict : int, default=100
91 The maximum number of iterations in Newton's method for approximating
92 the posterior during predict. Smaller values will reduce computation
93 time at the cost of worse results.
94
95 warm_start : bool, default=False
96 If warm-starts are enabled, the solution of the last Newton iteration
97 on the Laplace approximation of the posterior mode is used as
98 initialization for the next call of _posterior_mode(). This can speed
99 up convergence when _posterior_mode is called several times on similar
100 problems as in hyperparameter optimization. See :term:`the Glossary
101 <warm_start>`.
102
103 copy_X_train : bool, default=True
104 If True, a persistent copy of the training data is stored in the
105 object. Otherwise, just a reference to the training data is stored,
106 which might cause predictions to change if the data is modified
107 externally.
108
109 random_state : int, RandomState instance or None, default=None
110 Determines random number generation used to initialize the centers.
111 Pass an int for reproducible results across multiple function calls.
112 See :term:`Glossary <random_state>`.
113
114 Attributes
115 ----------
116 X_train_ : array-like of shape (n_samples, n_features) or list of object
117 Feature vectors or other representations of training data (also
118 required for prediction).
119
120 y_train_ : array-like of shape (n_samples,)
121 Target values in training data (also required for prediction)
122
123 classes_ : array-like of shape (n_classes,)
124 Unique class labels.
125
126 kernel_ : kernl instance
127 The kernel used for prediction. The structure of the kernel is the
128 same as the one passed as parameter but with optimized hyperparameters
129
130 L_ : array-like of shape (n_samples, n_samples)
131 Lower-triangular Cholesky decomposition of the kernel in X_train_
132
133 pi_ : array-like of shape (n_samples,)
134 The probabilities of the positive class for the training points
135 X_train_
136
137 W_sr_ : array-like of shape (n_samples,)
138 Square root of W, the Hessian of log-likelihood of the latent function
139 values for the observed labels. Since W is diagonal, only the diagonal
140 of sqrt(W) is stored.
141
142 log_marginal_likelihood_value_ : float
143 The log-marginal-likelihood of ``self.kernel_.theta``
144
145 References
146 ----------
147 .. [RW2006] `Carl E. Rasmussen and Christopher K.I. Williams,
148 "Gaussian Processes for Machine Learning",
149 MIT Press 2006 <https://www.gaussianprocess.org/gpml/chapters/RW.pdf>`_
150 """

171 def fit(self, X, y):
172 """Fit Gaussian process classification model.
173
174 Parameters
175 ----------
176 X : array-like of shape (n_samples, n_features) or list of object
177 Feature vectors or other representations of training data.
178
179 y : array-like of shape (n_samples,)
180 Target values, must be binary.
181
182 Returns
183 -------
184 self : returns an instance of self.
185 """
186 if self.kernel is None: # Use an RBF kernel as default
187 self.kernel_ = C(1.0, constant_value_bounds="fixed") * RBF(
188 1.0, length_scale_bounds="fixed"
189 )
190 else:
191 self.kernel_ = clone(self.kernel)
192
193 self.rng = check_random_state(self.random_state)
194
195 self.X_train_ = np.copy(X) if self.copy_X_train else X
196
197 # Encode class labels and check that it is a binary classification
198 # problem
199 label_encoder = LabelEncoder()
200 self.y_train_ = label_encoder.fit_transform(y)
201 self.classes_ = label_encoder.classes_
202 if self.classes_.size > 2:
203 raise ValueError(
204 "%s supports only binary classification. y contains classes %s"
205 % (self.__class__.__name__, self.classes_)
206 )
207 elif self.classes_.size == 1:
208 raise ValueError(
209 "{0:s} requires 2 classes; got {1:d} class".format(
210 self.__class__.__name__, self.classes_.size
211 )
212 )
213
214 if self.optimizer is not None and self.kernel_.n_dims > 0:
215 # Choose hyperparameters based on maximizing the log-marginal
216 # likelihood (potentially starting from several initial values)
217 def obj_func(theta, eval_gradient=True):
218 if eval_gradient:
219 lml, grad = self.log_marginal_likelihood(
220 theta, eval_gradient=True, clone_kernel=False
221 )
222 return -lml, -grad
223 else:
224 return -self.log_marginal_likelihood(theta, clone_kernel=False)
225
226 # First optimize starting from theta specified in kernel
227 optima = [
228 self._constrained_optimization(
229 obj_func, self.kernel_.theta, self.kernel_.bounds
230 )
231 ]
232
233 # Additional runs are performed from log-uniform chosen initial
234 # theta
235 if self.n_restarts_optimizer > 0:
236 if not np.isfinite(self.kernel_.bounds).all():
237 raise ValueError(
238 "Multiple optimizer restarts (n_restarts_optimizer>0) "
239 "requires that all bounds are finite."
240 )
241 bounds = self.kernel_.bounds
242 for iteration in range(self.n_restarts_optimizer):
243 theta_initial = np.exp(self.rng.uniform(bounds[:, 0], bounds[:, 1]))
244 optima.append(
245 self._constrained_optimization(obj_func, theta_initial, bounds)
246 )
247 # Select result from run with minimal (negative) log-marginal
248 # likelihood
249 lml_values = list(map(itemgetter(1), optima))
250 self.kernel_.theta = optima[np.argmin(lml_values)][0]
251 self.kernel_._check_bounds_params()
252
253 self.log_marginal_likelihood_value_ = -np.min(lml_values)
254 else:
255 self.log_marginal_likelihood_value_ = self.log_marginal_likelihood(
256 self.kernel_.theta
257 )
258
259 # Precompute quantities required for predictions which are independent
260 # of actual query points
261 K = self.kernel_(self.X_train_)
262
263 _, (self.pi_, self.W_sr_, self.L_, _, _) = self._posterior_mode(
264 K, return_temporaries=True
265 )
266
267 return self

269 def predict(self, X):
270 """Perform classification on an array of test vectors X.
271
272 Parameters
273 ----------
274 X : array-like of shape (n_samples, n_features) or list of object
275 Query points where the GP is evaluated for classification.
276
277 Returns
278 -------
279 C : ndarray of shape (n_samples,)
280 Predicted target values for X, values are from ``classes_``
281 """
282 check_is_fitted(self)
283
284 # As discussed on Section 3.4.2 of GPML, for making hard binary
285 # decisions, it is enough to compute the MAP of the posterior and
286 # pass it through the link function
287 K_star = self.kernel_(self.X_train_, X) # K_star =k(x_star)
288 f_star = K_star.T.dot(self.y_train_ - self.pi_) # Algorithm 3.2,Line 4
289
290 return np.where(f_star > 0, self.classes_[1], self.classes_[0])

292 def predict_proba(self, X):
293 """Return probability estimates for the test vector X.
294
295 Parameters
296 ----------
297 X : array-like of shape (n_samples, n_features) or list of object
298 Query points where the GP is evaluated for classification.
299
300 Returns
301 -------
302 C : array-like of shape (n_samples, n_classes)
303 Returns the probability of the samples for each class in
304 the model. The columns correspond to the classes in sorted
305 order, as they appear in the attribute ``classes_``.
306 """
307 check_is_fitted(self)
308
309 # Compute the mean and variance of the latent function
310 # (Lines 4-6 of Algorithm 3.2 of GPML)
311 latent_mean, latent_var = self.latent_mean_and_variance(X)
312
313 # Line 7:
314 # Approximate \int log(z) * N(z | f_star, var_f_star)
315 # Approximation is due to Williams & Barber, "Bayesian Classification
316 # with Gaussian Processes", Appendix A: Approximate the logistic
317 # sigmoid by a linear combination of 5 error functions.
318 # For information on how this integral can be computed see
319 # blitiri.blogspot.de/2012/11/gaussian-integral-of-error-function.html
320 alpha = 1 / (2 * latent_var)
321 gamma = LAMBDAS * latent_mean
322 integrals = (
323 np.sqrt(np.pi / alpha)
324 * erf(gamma * np.sqrt(alpha / (alpha + LAMBDAS**2)))
325 / (2 * np.sqrt(latent_var * 2 * np.pi))
326 )
327 pi_star = (COEFS * integrals).sum(axis=0) + 0.5 * COEFS.sum()
328
329 return np.vstack((1 - pi_star, pi_star)).T

331 def log_marginal_likelihood(
332 self, theta=None, eval_gradient=False, clone_kernel=True
333 ):
334 """Returns log-marginal likelihood of theta for training data.
335
336 Parameters
337 ----------
338 theta : array-like of shape (n_kernel_params,), default=None
339 Kernel hyperparameters for which the log-marginal likelihood is
340 evaluated. If None, the precomputed log_marginal_likelihood
341 of ``self.kernel_.theta`` is returned.
342
343 eval_gradient : bool, default=False
344 If True, the gradient of the log-marginal likelihood with respect
345 to the kernel hyperparameters at position theta is returned
346 additionally. If True, theta must not be None.
347
348 clone_kernel : bool, default=True
349 If True, the kernel attribute is copied. If False, the kernel
350 attribute is modified, but may result in a performance improvement.
351
352 Returns
353 -------
354 log_likelihood : float
355 Log-marginal likelihood of theta for training data.
356
357 log_likelihood_gradient : ndarray of shape (n_kernel_params,), \
358 optional
359 Gradient of the log-marginal likelihood with respect to the kernel
360 hyperparameters at position theta.
361 Only returned when `eval_gradient` is True.
362 """
363 if theta is None:
364 if eval_gradient:
365 raise ValueError("Gradient can only be evaluated for theta!=None")
366 return self.log_marginal_likelihood_value_
367
368 if clone_kernel:
369 kernel = self.kernel_.clone_with_theta(theta)
370 else:
371 kernel = self.kernel_
372 kernel.theta = theta
373
374 if eval_gradient:
375 K, K_gradient = kernel(self.X_train_, eval_gradient=True)
376 else:
377 K = kernel(self.X_train_)
378
379 # Compute log-marginal-likelihood Z and also store some temporaries
380 # which can be reused for computing Z's gradient
381 Z, (pi, W_sr, L, b, a) = self._posterior_mode(K, return_temporaries=True)
382
383 if not eval_gradient:
384 return Z
385
386 # Compute gradient based on Algorithm 5.1 of GPML
387 d_Z = np.empty(theta.shape[0])
388 # XXX: Get rid of the np.diag() in the next line
389 R = W_sr[:, np.newaxis] * cho_solve((L, True), np.diag(W_sr)) # Line 7
390 C = solve(L, W_sr[:, np.newaxis] * K) # Line 8
391 # Line 9: (use einsum to compute np.diag(C.T.dot(C))))
392 s_2 = (
393 -0.5
394 * (np.diag(K) - np.einsum("ij, ij -> j", C, C))
395 * (pi * (1 - pi) * (1 - 2 * pi))
396 ) # third derivative
397
398 for j in range(d_Z.shape[0]):
399 C = K_gradient[:, :, j] # Line 11
400 # Line 12: (R.T.ravel().dot(C.ravel()) = np.trace(R.dot(C)))
401 s_1 = 0.5 * a.T.dot(C).dot(a) - 0.5 * R.T.ravel().dot(C.ravel())
402
403 b = C.dot(self.y_train_ - pi) # Line 13
404 s_3 = b - K.dot(R.dot(b)) # Line 14
405
406 d_Z[j] = s_1 + s_2.T.dot(s_3) # Line 15
407
408 return Z, d_Z

410 def latent_mean_and_variance(self, X):
411 """Compute the mean and variance of the latent function values.
412
413 Based on algorithm 3.2 of [RW2006]_, this function returns the latent
414 mean (Line 4) and variance (Line 6) of the Gaussian process
415 classification model.
416
417 Note that this function is only supported for binary classification.
418
419 Parameters
420 ----------
421 X : array-like of shape (n_samples, n_features) or list of object
422 Query points where the GP is evaluated for classification.
423
424 Returns
425 -------
426 latent_mean : array-like of shape (n_samples,)
427 Mean of the latent function values at the query points.
428
429 latent_var : array-like of shape (n_samples,)
430 Variance of the latent function values at the query points.
431 """
432 check_is_fitted(self)
433
434 # Based on Algorithm 3.2 of GPML
435 K_star = self.kernel_(self.X_train_, X) # K_star =k(x_star)
436 latent_mean = K_star.T.dot(self.y_train_ - self.pi_) # Line 4
437 v = solve(self.L_, self.W_sr_[:, np.newaxis] * K_star) # Line 5
438 # Line 6 (compute np.diag(v.T.dot(v)) via einsum)
439 latent_var = self.kernel_.diag(X) - np.einsum("ij,ij->j", v, v)
440
441 return latent_mean, latent_var

443 def _posterior_mode(self, K, return_temporaries=False):
444 """Mode-finding for binary Laplace GPC and fixed kernel.
445
446 This approximates the posterior of the latent function values for given
447 inputs and target observations with a Gaussian approximation and uses
448 Newton's iteration to find the mode of this approximation.
449 """
450 # Based on Algorithm 3.1 of GPML
451
452 # If warm_start are enabled, we reuse the last solution for the
453 # posterior mode as initialization; otherwise, we initialize with 0
454 if (
455 self.warm_start
456 and hasattr(self, "f_cached")
457 and self.f_cached.shape == self.y_train_.shape
458 ):
459 f = self.f_cached
460 else:
461 f = np.zeros_like(self.y_train_, dtype=np.float64)
462
463 # Use Newton's iteration method to find mode of Laplace approximation
464 log_marginal_likelihood = -np.inf
465 for _ in range(self.max_iter_predict):
466 # Line 4
467 pi = expit(f)
468 W = pi * (1 - pi)
469 # Line 5
470 W_sr = np.sqrt(W)
471 W_sr_K = W_sr[:, np.newaxis] * K
472 B = np.eye(W.shape[0]) + W_sr_K * W_sr
473 L = cholesky(B, lower=True)
474 # Line 6
475 b = W * f + (self.y_train_ - pi)
476 # Line 7
477 a = b - W_sr * cho_solve((L, True), W_sr_K.dot(b))
478 # Line 8
479 f = K.dot(a)
480
481 # Line 10: Compute log marginal likelihood in loop and use as
482 # convergence criterion
483 lml = (
484 -0.5 * a.T.dot(f)
485 - np.log1p(np.exp(-(self.y_train_ * 2 - 1) * f)).sum()
486 - np.log(np.diag(L)).sum()
487 )
488 # Check if we have converged (log marginal likelihood does
489 # not decrease)
490 # XXX: more complex convergence criterion
491 if lml - log_marginal_likelihood < 1e-10:
492 break
493 log_marginal_likelihood = lml
494
495 self.f_cached = f # Remember solution for later warm-starts
496 if return_temporaries:
497 return log_marginal_likelihood, (pi, W_sr, L, b, a)
498 else:
499 return log_marginal_likelihood

501 def _constrained_optimization(self, obj_func, initial_theta, bounds):
502 if self.optimizer == "fmin_l_bfgs_b":
503 opt_res = scipy.optimize.minimize(
504 obj_func, initial_theta, method="L-BFGS-B", jac=True, bounds=bounds
505 )
506 _check_optimize_result("lbfgs", opt_res)
507 theta_opt, func_min = opt_res.x, opt_res.fun
508 elif callable(self.optimizer):
509 theta_opt, func_min = self.optimizer(obj_func, initial_theta, bounds=bounds)
510 else:
511 raise ValueError("Unknown optimizer %s." % self.optimizer)
512
513 return theta_opt, func_min

516class GaussianProcessClassifier(ClassifierMixin, BaseEstimator):
517 """Gaussian process classification (GPC) based on Laplace approximation.
518
519 The implementation is based on Algorithm 3.1, 3.2, and 5.1 from [RW2006]_.
520
521 Internally, the Laplace approximation is used for approximating the
522 non-Gaussian posterior by a Gaussian.
523
524 Currently, the implementation is restricted to using the logistic link
525 function. For multi-class classification, several binary one-versus rest
526 classifiers are fitted. Note that this class thus does not implement
527 a true multi-class Laplace approximation.
528
529 Read more in the :ref:`User Guide <gaussian_process>`.
530
531 .. versionadded:: 0.18
532
533 Parameters
534 ----------
535 kernel : kernel instance, default=None
536 The kernel specifying the covariance function of the GP. If None is
537 passed, the kernel "1.0 * RBF(1.0)" is used as default. Note that
538 the kernel's hyperparameters are optimized during fitting. Also kernel
539 cannot be a `CompoundKernel`.
540
541 optimizer : 'fmin_l_bfgs_b', callable or None, default='fmin_l_bfgs_b'
542 Can either be one of the internally supported optimizers for optimizing
543 the kernel's parameters, specified by a string, or an externally
544 defined optimizer passed as a callable. If a callable is passed, it
545 must have the signature::
546
547 def optimizer(obj_func, initial_theta, bounds):
548 # * 'obj_func' is the objective function to be maximized, which
549 # takes the hyperparameters theta as parameter and an
550 # optional flag eval_gradient, which determines if the
551 # gradient is returned additionally to the function value
552 # * 'initial_theta': the initial value for theta, which can be
553 # used by local optimizers
554 # * 'bounds': the bounds on the values of theta
555 ....
556 # Returned are the best found hyperparameters theta and
557 # the corresponding value of the target function.
558 return theta_opt, func_min
559
560 Per default, the 'L-BFGS-B' algorithm from scipy.optimize.minimize
561 is used. If None is passed, the kernel's parameters are kept fixed.
562 Available internal optimizers are::
563
564 'fmin_l_bfgs_b'
565
566 n_restarts_optimizer : int, default=0
567 The number of restarts of the optimizer for finding the kernel's
568 parameters which maximize the log-marginal likelihood. The first run
569 of the optimizer is performed from the kernel's initial parameters,
570 the remaining ones (if any) from thetas sampled log-uniform randomly
571 from the space of allowed theta-values. If greater than 0, all bounds
572 must be finite. Note that n_restarts_optimizer=0 implies that one
573 run is performed.
574
575 max_iter_predict : int, default=100
576 The maximum number of iterations in Newton's method for approximating
577 the posterior during predict. Smaller values will reduce computation
578 time at the cost of worse results.
579
580 warm_start : bool, default=False
581 If warm-starts are enabled, the solution of the last Newton iteration
582 on the Laplace approximation of the posterior mode is used as
583 initialization for the next call of _posterior_mode(). This can speed
584 up convergence when _posterior_mode is called several times on similar
585 problems as in hyperparameter optimization. See :term:`the Glossary
586 <warm_start>`.
587
588 copy_X_train : bool, default=True
589 If True, a persistent copy of the training data is stored in the
590 object. Otherwise, just a reference to the training data is stored,
591 which might cause predictions to change if the data is modified
592 externally.
593
594 random_state : int, RandomState instance or None, default=None
595 Determines random number generation used to initialize the centers.
596 Pass an int for reproducible results across multiple function calls.
597 See :term:`Glossary <random_state>`.
598
599 multi_class : {'one_vs_rest', 'one_vs_one'}, default='one_vs_rest'
600 Specifies how multi-class classification problems are handled.
601 Supported are 'one_vs_rest' and 'one_vs_one'. In 'one_vs_rest',
602 one binary Gaussian process classifier is fitted for each class, which
603 is trained to separate this class from the rest. In 'one_vs_one', one
604 binary Gaussian process classifier is fitted for each pair of classes,
605 which is trained to separate these two classes. The predictions of
606 these binary predictors are combined into multi-class predictions.
607 Note that 'one_vs_one' does not support predicting probability
608 estimates.
609
610 n_jobs : int, default=None
611 The number of jobs to use for the computation: the specified
612 multiclass problems are computed in parallel.
613 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
614 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
615 for more details.
616
617 Attributes
618 ----------
619 base_estimator_ : ``Estimator`` instance
620 The estimator instance that defines the likelihood function
621 using the observed data.
622
623 kernel_ : kernel instance
624 The kernel used for prediction. In case of binary classification,
625 the structure of the kernel is the same as the one passed as parameter
626 but with optimized hyperparameters. In case of multi-class
627 classification, a CompoundKernel is returned which consists of the
628 different kernels used in the one-versus-rest classifiers.
629
630 log_marginal_likelihood_value_ : float
631 The log-marginal-likelihood of ``self.kernel_.theta``
632
633 classes_ : array-like of shape (n_classes,)
634 Unique class labels.
635
636 n_classes_ : int
637 The number of classes in the training data
638
639 n_features_in_ : int
640 Number of features seen during :term:`fit`.
641
642 .. versionadded:: 0.24
643
644 feature_names_in_ : ndarray of shape (`n_features_in_`,)
645 Names of features seen during :term:`fit`. Defined only when `X`
646 has feature names that are all strings.

851 def log_marginal_likelihood(
852 self, theta=None, eval_gradient=False, clone_kernel=True
853 ):
854 """Return log-marginal likelihood of theta for training data.
855
856 In the case of multi-class classification, the mean log-marginal
857 likelihood of the one-versus-rest classifiers are returned.
858
859 Parameters
860 ----------
861 theta : array-like of shape (n_kernel_params,), default=None
862 Kernel hyperparameters for which the log-marginal likelihood is
863 evaluated. In the case of multi-class classification, theta may
864 be the hyperparameters of the compound kernel or of an individual
865 kernel. In the latter case, all individual kernel get assigned the
866 same theta values. If None, the precomputed log_marginal_likelihood
867 of ``self.kernel_.theta`` is returned.
868
869 eval_gradient : bool, default=False
870 If True, the gradient of the log-marginal likelihood with respect
871 to the kernel hyperparameters at position theta is returned
872 additionally. Note that gradient computation is not supported
873 for non-binary classification. If True, theta must not be None.
874
875 clone_kernel : bool, default=True
876 If True, the kernel attribute is copied. If False, the kernel
877 attribute is modified, but may result in a performance improvement.
878
879 Returns
880 -------
881 log_likelihood : float
882 Log-marginal likelihood of theta for training data.
883
884 log_likelihood_gradient : ndarray of shape (n_kernel_params,), optional
885 Gradient of the log-marginal likelihood with respect to the kernel
886 hyperparameters at position theta.
887 Only returned when `eval_gradient` is True.
888 """
889 check_is_fitted(self)
890
891 if theta is None:
892 if eval_gradient:
893 raise ValueError("Gradient can only be evaluated for theta!=None")
894 return self.log_marginal_likelihood_value_
895
896 theta = np.asarray(theta)
897 if self.n_classes_ == 2:
898 return self.base_estimator_.log_marginal_likelihood(
899 theta, eval_gradient, clone_kernel=clone_kernel
900 )
901 else:
902 if eval_gradient:
903 raise NotImplementedError(
904 "Gradient of log-marginal-likelihood not implemented for "
905 "multi-class GPC."
906 )
907 estimators = self.base_estimator_.estimators_
908 n_dims = estimators[0].kernel_.n_dims
909 if theta.shape[0] == n_dims: # use same theta for all sub-kernels

913 theta, clone_kernel=clone_kernel
914 )
915 for i, estimator in enumerate(estimators)
916 ]
917 )
918 elif theta.shape[0] == n_dims * self.classes_.shape[0]:
919 # theta for compound kernel
