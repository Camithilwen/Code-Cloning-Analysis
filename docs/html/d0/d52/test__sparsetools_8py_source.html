28def test_threads():
29 # Smoke test for parallel threaded execution; doesn't actually
30 # check that code runs in parallel, but just that it produces
31 # expected results.
32 nthreads = 10
33 niter = 100
34
35 n = 20
36 a = csr_matrix(np.ones([n, n]))
37 bres = []
38
39 class Worker(threading.Thread):
40 def run(self):
41 b = a.copy()
42 for j in range(niter):
43 _sparsetools.csr_plus_csr(n, n,
44 a.indptr, a.indices, a.data,
45 a.indptr, a.indices, a.data,
46 b.indptr, b.indices, b.data)
47 bres.append(b)

49 threads = [Worker() for _ in range(nthreads)]
50 for thread in threads:

289def test_upcast():
290 a0 = csr_matrix([[np.pi, np.pi*1j], [3, 4]], dtype=complex)
291 b0 = np.array([256+1j, 2**32], dtype=complex)
292
293 for a_dtype in supported_dtypes:

297 if np.issubdtype(a_dtype, np.complexfloating):
298 a = a0.copy().astype(a_dtype)

300 a = a0.real.copy().astype(a_dtype)

307 # numpy >1.23
