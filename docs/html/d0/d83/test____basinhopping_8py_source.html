18def func1d(x):
19 f = cos(14.5 * x - 0.3) + (x + 0.2) * x
20 df = np.array(-14.5 * sin(14.5 * x - 0.3) + 2. * x + 0.2)
21 return f, df

29def func2d(x):
30 f = cos(14.5 * x[0] - 0.3) + (x[1] + 0.2) * x[1] + (x[0] + 0.2) * x[0]
31 df = np.zeros(2)
32 df[0] = -14.5 * sin(14.5 * x[0] - 0.3) + 2. * x[0] + 0.2
33 df[1] = 2. * x[1] + 0.2
34 return f, df

37def func2d_easyderiv(x):
38 f = 2.0*x[0]**2 + 2.0*x[0]*x[1] + 2.0*x[1]**2 - 6.0*x[0]
39 df = np.zeros(2)
40 df[0] = 4.0*x[0] + 2.0*x[1] - 6.0
41 df[1] = 2.0*x[0] + 4.0*x[1]
42
43 return f, df
