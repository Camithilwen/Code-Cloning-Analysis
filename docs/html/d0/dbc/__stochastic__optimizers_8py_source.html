72class SGDOptimizer(BaseOptimizer):
73 """Stochastic gradient descent optimizer with momentum
74
75 Parameters
76 ----------
77 params : list, length = len(coefs_) + len(intercepts_)
78 The concatenated list containing coefs_ and intercepts_ in MLP model.
79 Used for initializing velocities and updating params
80
81 learning_rate_init : float, default=0.1
82 The initial learning rate used. It controls the step-size in updating
83 the weights
84
85 lr_schedule : {'constant', 'adaptive', 'invscaling'}, default='constant'
86 Learning rate schedule for weight updates.
87
88 -'constant', is a constant learning rate given by
89 'learning_rate_init'.
90
91 -'invscaling' gradually decreases the learning rate 'learning_rate_' at
92 each time step 't' using an inverse scaling exponent of 'power_t'.
93 learning_rate_ = learning_rate_init / pow(t, power_t)
94
95 -'adaptive', keeps the learning rate constant to
96 'learning_rate_init' as long as the training keeps decreasing.
97 Each time 2 consecutive epochs fail to decrease the training loss by
98 tol, or fail to increase validation score by tol if 'early_stopping'
99 is on, the current learning rate is divided by 5.
100
101 momentum : float, default=0.9
102 Value of momentum used, must be larger than or equal to 0
103
104 nesterov : bool, default=True
105 Whether to use nesterov's momentum or not. Use nesterov's if True
106
107 power_t : float, default=0.5
108 Power of time step 't' in inverse scaling. See `lr_schedule` for
109 more details.
110
111 Attributes
112 ----------
113 learning_rate : float
114 the current learning rate
115
116 velocities : list, length = len(params)
117 velocities that are used to update params
118 """
119
120 def __init__(
121 self,
122 params,
123 learning_rate_init=0.1,
124 lr_schedule="constant",
125 momentum=0.9,
126 nesterov=True,
127 power_t=0.5,
128 ):
129 super().__init__(learning_rate_init)
130
131 self.lr_schedule = lr_schedule
132 self.momentum = momentum
133 self.nesterov = nesterov
134 self.power_t = power_t
135 self.velocities = [np.zeros_like(param) for param in params]
136
137 def iteration_ends(self, time_step):
138 """Perform updates to learning rate and potential other states at the
139 end of an iteration
140
141 Parameters
142 ----------
143 time_step : int
144 number of training samples trained on so far, used to update
145 learning rate for 'invscaling'
146 """
147 if self.lr_schedule == "invscaling":
148 self.learning_rate = (
149 float(self.learning_rate_init) / (time_step + 1) ** self.power_t
150 )
151
152 def trigger_stopping(self, msg, verbose):
153 if self.lr_schedule != "adaptive":
154 if verbose:
155 print(msg + " Stopping.")
156 return True
157
158 if self.learning_rate <= 1e-6:
159 if verbose:
160 print(msg + " Learning rate too small. Stopping.")
161 return True
162
163 self.learning_rate /= 5.0
164 if verbose:
165 print(msg + " Setting learning rate to %f" % self.learning_rate)
166 return False
167
168 def _get_updates(self, grads):
169 """Get the values used to update params with given gradients
170
171 Parameters
172 ----------
173 grads : list, length = len(coefs_) + len(intercepts_)
174 Containing gradients with respect to coefs_ and intercepts_ in MLP
175 model. So length should be aligned with params
176
177 Returns
178 -------
179 updates : list, length = len(grads)
180 The values to add to params
181 """
182 updates = [
183 self.momentum * velocity - self.learning_rate * grad
184 for velocity, grad in zip(self.velocities, grads)
185 ]
186 self.velocities = updates
187
188 if self.nesterov:
189 updates = [
190 self.momentum * velocity - self.learning_rate * grad
191 for velocity, grad in zip(self.velocities, grads)
192 ]
193
194 return updates

197class AdamOptimizer(BaseOptimizer):
198 """Stochastic gradient descent optimizer with Adam
199
200 Note: All default values are from the original Adam paper
201
202 Parameters
203 ----------
204 params : list, length = len(coefs_) + len(intercepts_)
205 The concatenated list containing coefs_ and intercepts_ in MLP model.
206 Used for initializing velocities and updating params
207
208 learning_rate_init : float, default=0.001
209 The initial learning rate used. It controls the step-size in updating
210 the weights
211
212 beta_1 : float, default=0.9
213 Exponential decay rate for estimates of first moment vector, should be
214 in [0, 1)
215
216 beta_2 : float, default=0.999
217 Exponential decay rate for estimates of second moment vector, should be
218 in [0, 1)
219
220 epsilon : float, default=1e-8
221 Value for numerical stability
222
223 Attributes
224 ----------
225 learning_rate : float
226 The current learning rate
227
228 t : int
229 Timestep
230
231 ms : list, length = len(params)
232 First moment vectors
233
234 vs : list, length = len(params)
235 Second moment vectors
236
237 References
238 ----------
239 :arxiv:`Kingma, Diederik, and Jimmy Ba (2014) "Adam: A method for
240 stochastic optimization." <1412.6980>
241 """
242
243 def __init__(
244 self, params, learning_rate_init=0.001, beta_1=0.9, beta_2=0.999, epsilon=1e-8
245 ):
246 super().__init__(learning_rate_init)
247
248 self.beta_1 = beta_1
249 self.beta_2 = beta_2
250 self.epsilon = epsilon
251 self.t = 0
252 self.ms = [np.zeros_like(param) for param in params]
253 self.vs = [np.zeros_like(param) for param in params]
254
255 def _get_updates(self, grads):
256 """Get the values used to update params with given gradients
257
258 Parameters
259 ----------
260 grads : list, length = len(coefs_) + len(intercepts_)
261 Containing gradients with respect to coefs_ and intercepts_ in MLP
262 model. So length should be aligned with params
263
264 Returns
265 -------
266 updates : list, length = len(grads)
267 The values to add to params
268 """
269 self.t += 1
270 self.ms = [
271 self.beta_1 * m + (1 - self.beta_1) * grad
272 for m, grad in zip(self.ms, grads)
273 ]
274 self.vs = [
275 self.beta_2 * v + (1 - self.beta_2) * (grad**2)
276 for v, grad in zip(self.vs, grads)
277 ]
278 self.learning_rate = (
279 self.learning_rate_init
280 * np.sqrt(1 - self.beta_2**self.t)
281 / (1 - self.beta_1**self.t)
282 )
283 updates = [
284 -self.learning_rate * m / (np.sqrt(v) + self.epsilon)
285 for m, v in zip(self.ms, self.vs)
286 ]
287 return updates
