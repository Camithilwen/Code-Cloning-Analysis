148def test_count_bitmaps():
149 def count_tag(fig, tag):
150 with BytesIO() as fd:
151 fig.savefig(fd, format='svg')
152 buf = fd.getvalue().decode()
153 return buf.count(f"<{tag}")
154
155 # No rasterized elements
156 fig1 = plt.figure()
157 ax1 = fig1.add_subplot(1, 1, 1)
158 ax1.set_axis_off()
159 for n in range(5):
160 ax1.plot([0, 20], [0, n], "b-", rasterized=False)
161 assert count_tag(fig1, "image") == 0
162 assert count_tag(fig1, "path") == 6 # axis patch plus lines
163
164 # rasterized can be merged
165 fig2 = plt.figure()
166 ax2 = fig2.add_subplot(1, 1, 1)
167 ax2.set_axis_off()
168 for n in range(5):
169 ax2.plot([0, 20], [0, n], "b-", rasterized=True)
170 assert count_tag(fig2, "image") == 1
171 assert count_tag(fig2, "path") == 1 # axis patch
172
173 # rasterized can't be merged without affecting draw order
174 fig3 = plt.figure()
175 ax3 = fig3.add_subplot(1, 1, 1)
176 ax3.set_axis_off()
177 for n in range(5):
178 ax3.plot([0, 20], [n, 0], "b-", rasterized=False)
179 ax3.plot([0, 20], [0, n], "b-", rasterized=True)
180 assert count_tag(fig3, "image") == 5
181 assert count_tag(fig3, "path") == 6
182
183 # rasterized whole axes
184 fig4 = plt.figure()
185 ax4 = fig4.add_subplot(1, 1, 1)
186 ax4.set_axis_off()
187 ax4.set_rasterized(True)
188 for n in range(5):
189 ax4.plot([0, 20], [n, 0], "b-", rasterized=False)
190 ax4.plot([0, 20], [0, n], "b-", rasterized=True)
191 assert count_tag(fig4, "image") == 1
192 assert count_tag(fig4, "path") == 1
193
194 # rasterized can be merged, but inhibited by suppressComposite
195 fig5 = plt.figure()
196 fig5.suppressComposite = True
197 ax5 = fig5.add_subplot(1, 1, 1)
198 ax5.set_axis_off()
199 for n in range(5):
200 ax5.plot([0, 20], [0, n], "b-", rasterized=True)
201 assert count_tag(fig5, "image") == 5

240def test_gid():
241 """Test that object gid appears in output svg."""
242 from matplotlib.offsetbox import OffsetBox
243 from matplotlib.axis import Tick
244
245 fig = plt.figure()
246
247 ax1 = fig.add_subplot(131)
248 ax1.imshow([[1., 2.], [2., 3.]], aspect="auto")
249 ax1.scatter([1, 2, 3], [1, 2, 3], label="myscatter")
250 ax1.plot([2, 3, 1], label="myplot")
251 ax1.legend()
252 ax1a = ax1.twinx()
253 ax1a.bar([1, 2, 3], [1, 2, 3])
254
255 ax2 = fig.add_subplot(132, projection="polar")
256 ax2.plot([0, 1.5, 3], [1, 2, 3])
257
258 ax3 = fig.add_subplot(133, projection="3d")
259 ax3.plot([1, 2], [1, 2], [1, 2])
260
261 fig.canvas.draw()
262
263 gdic = {}
264 for idx, obj in enumerate(fig.findobj(include_self=True)):
265 if obj.get_visible():
266 gid = f"test123{obj.__class__.__name__}_{idx}"
267 gdic[gid] = obj
268 obj.set_gid(gid)
269
270 with BytesIO() as fd:
271 fig.savefig(fd, format='svg')
272 buf = fd.getvalue().decode()
273
274 def include(gid, obj):
275 # we need to exclude certain objects which will not appear in the svg
276 if isinstance(obj, OffsetBox):
277 return False
278 if isinstance(obj, Text):
279 if obj.get_text() == "":
280 return False
281 elif obj.axes is None:
282 return False
283 if isinstance(obj, plt.Line2D):
284 xdata, ydata = obj.get_data()
285 if len(xdata) == len(ydata) == 1:
286 return False
287 elif not hasattr(obj, "axes") or obj.axes is None:
288 return False
289 if isinstance(obj, Tick):
290 loc = obj.get_loc()
291 if loc == 0:
292 return False
293 vi = obj.get_view_interval()
294 if loc < min(vi) or loc > max(vi):
295 return False
296 return True
297
298 for gid, obj in gdic.items():
299 if include(gid, obj):
300 assert gid in buf
