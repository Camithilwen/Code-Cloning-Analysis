18def get_joblib_version(joblib_version=joblib.__version__):
19 """Normalize joblib version by removing suffix.
20
21 >>> get_joblib_version('0.8.4')
22 '0.8.4'
23 >>> get_joblib_version('0.8.4b1')
24 '0.8.4'
25 >>> get_joblib_version('0.9.dev0')
26 '0.9'
27 """
28 matches = [re.match(r"(\d+).*", each) for each in joblib_version.split(".")]
29 return ".".join([m.group(1) for m in matches if m is not None])

32def write_test_pickle(to_pickle, args):
33 kwargs = {}
34 compress = args.compress
35 method = args.method
36 joblib_version = get_joblib_version()
37 py_version = "{0[0]}{0[1]}".format(sys.version_info)
38 numpy_version = "".join(np.__version__.split(".")[:2])

40 # The game here is to generate the right filename according to the options.
41 body = "_compressed" if (compress and method == "zlib") else ""
42 if compress:
43 if method == "zlib":
44 kwargs["compress"] = True
45 extension = ".gz"
46 else:
47 kwargs["compress"] = (method, 3)
48 extension = ".pkl.{}".format(method)
49 if args.cache_size:
50 kwargs["cache_size"] = 0
51 body += "_cache_size"
52 else:
53 extension = ".pkl"
54
55 pickle_filename = "joblib_{}{}_pickle_py{}_np{}{}".format(
56 joblib_version, body, py_version, numpy_version, extension
57 )
58
59 try:
60 joblib.dump(to_pickle, pickle_filename, **kwargs)
61 except Exception as e:
62 # With old python version (=< 3.3.), we can arrive there when
63 # dumping compressed pickle with LzmaFile.
64 print(
65 "Error: cannot generate file '{}' with arguments '{}'. "
66 "Error was: {}".format(pickle_filename, kwargs, e)
67 )
68 else:
