31def _make_edges_3d(n_x, n_y, n_z=1):
32 """Returns a list of edges for a 3D image.
33
34 Parameters
35 ----------
36 n_x : int
37 The size of the grid in the x direction.
38 n_y : int
39 The size of the grid in the y direction.
40 n_z : integer, default=1
41 The size of the grid in the z direction, defaults to 1
42 """
43 vertices = np.arange(n_x * n_y * n_z).reshape((n_x, n_y, n_z))
44 edges_deep = np.vstack((vertices[:, :, :-1].ravel(), vertices[:, :, 1:].ravel()))
45 edges_right = np.vstack((vertices[:, :-1].ravel(), vertices[:, 1:].ravel()))
46 edges_down = np.vstack((vertices[:-1].ravel(), vertices[1:].ravel()))
47 edges = np.hstack((edges_deep, edges_right, edges_down))
48 return edges

51def _compute_gradient_3d(edges, img):
52 _, n_y, n_z = img.shape
53 gradient = np.abs(
54 img[
55 edges[0] // (n_y * n_z),
56 (edges[0] % (n_y * n_z)) // n_z,
57 (edges[0] % (n_y * n_z)) % n_z,
58 ]
59 - img[
60 edges[1] // (n_y * n_z),
61 (edges[1] % (n_y * n_z)) // n_z,
62 (edges[1] % (n_y * n_z)) % n_z,
63 ]
64 )
65 return gradient

71def _mask_edges_weights(mask, edges, weights=None):
72 """Apply a mask to edges (weighted or not)"""
73 inds = np.arange(mask.size)
74 inds = inds[mask.ravel()]
75 ind_mask = np.logical_and(np.isin(edges[0], inds), np.isin(edges[1], inds))
76 edges = edges[:, ind_mask]
77 if weights is not None:
78 weights = weights[ind_mask]
79 if len(edges.ravel()):
80 maxval = edges.max()
81 else:
82 maxval = 0
83 order = np.searchsorted(np.flatnonzero(mask), np.arange(maxval + 1))
84 edges = order[edges]
85 if weights is None:
86 return edges
87 else:
88 return edges, weights

91def _to_graph(
92 n_x, n_y, n_z, mask=None, img=None, return_as=sparse.coo_matrix, dtype=None
93):
94 """Auxiliary function for img_to_graph and grid_to_graph"""
95 edges = _make_edges_3d(n_x, n_y, n_z)
96
97 if dtype is None: # To not overwrite input dtype
98 if img is None:
99 dtype = int
100 else:
101 dtype = img.dtype
102
103 if img is not None:
104 img = np.atleast_3d(img)
105 weights = _compute_gradient_3d(edges, img)
106 if mask is not None:
107 edges, weights = _mask_edges_weights(mask, edges, weights)
108 diag = img.squeeze()[mask]
109 else:
110 diag = img.ravel()
111 n_voxels = diag.size
112 else:
113 if mask is not None:
114 mask = mask.astype(dtype=bool, copy=False)
115 edges = _mask_edges_weights(mask, edges)
116 n_voxels = np.sum(mask)
117 else:
118 n_voxels = n_x * n_y * n_z
119 weights = np.ones(edges.shape[1], dtype=dtype)
120 diag = np.ones(n_voxels, dtype=dtype)
121
122 diag_idx = np.arange(n_voxels)
123 i_idx = np.hstack((edges[0], edges[1]))
124 j_idx = np.hstack((edges[1], edges[0]))
125 graph = sparse.coo_matrix(
126 (
127 np.hstack((weights, weights, diag)),
128 (np.hstack((i_idx, diag_idx)), np.hstack((j_idx, diag_idx))),
129 ),
130 (n_voxels, n_voxels),
131 dtype=dtype,
132 )
133 if return_as is np.ndarray:
134 return graph.toarray()
135 return return_as(graph)

254def _compute_n_patches(i_h, i_w, p_h, p_w, max_patches=None):
255 """Compute the number of patches that will be extracted in an image.
256
257 Read more in the :ref:`User Guide <image_feature_extraction>`.
258
259 Parameters
260 ----------
261 i_h : int
262 The image height
263 i_w : int
264 The image with
265 p_h : int
266 The height of a patch
267 p_w : int
268 The width of a patch
269 max_patches : int or float, default=None
270 The maximum number of patches to extract. If `max_patches` is a float
271 between 0 and 1, it is taken to be a proportion of the total number
272 of patches. If `max_patches` is None, all possible patches are extracted.
273 """
274 n_h = i_h - p_h + 1
275 n_w = i_w - p_w + 1
276 all_patches = n_h * n_w
277
278 if max_patches:
279 if isinstance(max_patches, (Integral)) and max_patches < all_patches:
280 return max_patches
281 elif isinstance(max_patches, (Integral)) and max_patches >= all_patches:
282 return all_patches
283 elif isinstance(max_patches, (Real)) and 0 < max_patches < 1:
284 return int(max_patches * all_patches)
285 else:
286 raise ValueError("Invalid value for max_patches: %r" % max_patches)
287 else:
288 return all_patches

291def _extract_patches(arr, patch_shape=8, extraction_step=1):
292 """Extracts patches of any n-dimensional array in place using strides.
293
294 Given an n-dimensional array it will return a 2n-dimensional array with
295 the first n dimensions indexing patch position and the last n indexing
296 the patch content. This operation is immediate (O(1)). A reshape
297 performed on the first n dimensions will cause numpy to copy data, leading
298 to a list of extracted patches.
299
300 Read more in the :ref:`User Guide <image_feature_extraction>`.
301
302 Parameters
303 ----------
304 arr : ndarray
305 n-dimensional array of which patches are to be extracted
306
307 patch_shape : int or tuple of length arr.ndim.default=8
308 Indicates the shape of the patches to be extracted. If an
309 integer is given, the shape will be a hypercube of
310 sidelength given by its value.
311
312 extraction_step : int or tuple of length arr.ndim, default=1
313 Indicates step size at which extraction shall be performed.
314 If integer is given, then the step is uniform in all dimensions.
315
316
317 Returns
318 -------
319 patches : strided ndarray
320 2n-dimensional array indexing patches on first n dimensions and
321 containing patches on the last n dimensions. These dimensions
322 are fake, but this way no data is copied. A simple reshape invokes
323 a copying operation to obtain a list of patches:
324 result.reshape([-1] + list(patch_shape))
325 """
326
327 arr_ndim = arr.ndim
328
329 if isinstance(patch_shape, Number):
330 patch_shape = tuple([patch_shape] * arr_ndim)
331 if isinstance(extraction_step, Number):
332 extraction_step = tuple([extraction_step] * arr_ndim)
333
334 patch_strides = arr.strides
335
336 slices = tuple(slice(None, None, st) for st in extraction_step)
337 indexing_strides = arr[slices].strides
338
339 patch_indices_shape = (
340 (np.array(arr.shape) - np.array(patch_shape)) // np.array(extraction_step)
341 ) + 1
342
343 shape = tuple(list(patch_indices_shape) + list(patch_shape))
344 strides = tuple(list(indexing_strides) + list(patch_strides))
345
346 patches = as_strided(arr, shape=shape, strides=strides)
347 return patches

363def extract_patches_2d(image, patch_size, *, max_patches=None, random_state=None):
364 """Reshape a 2D image into a collection of patches.
365
366 The resulting patches are allocated in a dedicated array.
367
368 Read more in the :ref:`User Guide <image_feature_extraction>`.
369
370 Parameters
371 ----------
372 image : ndarray of shape (image_height, image_width) or \
373 (image_height, image_width, n_channels)
374 The original image data. For color images, the last dimension specifies
375 the channel: a RGB image would have `n_channels=3`.
376
377 patch_size : tuple of int (patch_height, patch_width)
378 The dimensions of one patch.
379
380 max_patches : int or float, default=None
381 The maximum number of patches to extract. If `max_patches` is a float
382 between 0 and 1, it is taken to be a proportion of the total number
383 of patches. If `max_patches` is None it corresponds to the total number
384 of patches that can be extracted.
385
386 random_state : int, RandomState instance, default=None
387 Determines the random number generator used for random sampling when
388 `max_patches` is not None. Use an int to make the randomness
389 deterministic.
390 See :term:`Glossary <random_state>`.
391
392 Returns
393 -------
394 patches : array of shape (n_patches, patch_height, patch_width) or \
395 (n_patches, patch_height, patch_width, n_channels)
396 The collection of patches extracted from the image, where `n_patches`
397 is either `max_patches` or the total number of patches that can be
398 extracted.
399
400 Examples
401 --------
402 >>> from sklearn.datasets import load_sample_image
403 >>> from sklearn.feature_extraction import image
404 >>> # Use the array data from the first image in this dataset:
405 >>> one_image = load_sample_image("china.jpg")
406 >>> print('Image shape: {}'.format(one_image.shape))
407 Image shape: (427, 640, 3)
408 >>> patches = image.extract_patches_2d(one_image, (2, 2))
409 >>> print('Patches shape: {}'.format(patches.shape))
410 Patches shape: (272214, 2, 2, 3)
411 >>> # Here are just two of these patches:
412 >>> print(patches[1])
413 [[[174 201 231]
414 [174 201 231]]
415 [[173 200 230]
416 [173 200 230]]]
417 >>> print(patches[800])
418 [[[187 214 243]
419 [188 215 244]]
420 [[187 214 243]
421 [188 215 244]]]
422 """
423 i_h, i_w = image.shape[:2]
424 p_h, p_w = patch_size
425
426 if p_h > i_h:
427 raise ValueError(
428 "Height of the patch should be less than the height of the image."
429 )
430
431 if p_w > i_w:
432 raise ValueError(
433 "Width of the patch should be less than the width of the image."
434 )
435
436 image = check_array(image, allow_nd=True)
437 image = image.reshape((i_h, i_w, -1))
438 n_colors = image.shape[-1]
439
440 extracted_patches = _extract_patches(
441 image, patch_shape=(p_h, p_w, n_colors), extraction_step=1
442 )
443
444 n_patches = _compute_n_patches(i_h, i_w, p_h, p_w, max_patches)
445 if max_patches:
446 rng = check_random_state(random_state)
447 i_s = rng.randint(i_h - p_h + 1, size=n_patches)
448 j_s = rng.randint(i_w - p_w + 1, size=n_patches)
449 patches = extracted_patches[i_s, j_s, 0]
450 else:
451 patches = extracted_patches
452
453 patches = patches.reshape(-1, p_h, p_w, n_colors)
454 # remove the color dimension if useless
455 if patches.shape[-1] == 1:
456 return patches.reshape((n_patches, p_h, p_w))
457 else:
458 return patches

465def reconstruct_from_patches_2d(patches, image_size):
466 """Reconstruct the image from all of its patches.
467
468 Patches are assumed to overlap and the image is constructed by filling in
469 the patches from left to right, top to bottom, averaging the overlapping
470 regions.
471
472 Read more in the :ref:`User Guide <image_feature_extraction>`.
473
474 Parameters
475 ----------
476 patches : ndarray of shape (n_patches, patch_height, patch_width) or \
477 (n_patches, patch_height, patch_width, n_channels)
478 The complete set of patches. If the patches contain colour information,
479 channels are indexed along the last dimension: RGB patches would
480 have `n_channels=3`.
481
482 image_size : tuple of int (image_height, image_width) or \
483 (image_height, image_width, n_channels)
484 The size of the image that will be reconstructed.
485
486 Returns
487 -------
488 image : ndarray of shape image_size
489 The reconstructed image.
490
491 Examples
492 --------
493 >>> from sklearn.datasets import load_sample_image
494 >>> from sklearn.feature_extraction import image
495 >>> one_image = load_sample_image("china.jpg")
496 >>> print('Image shape: {}'.format(one_image.shape))
497 Image shape: (427, 640, 3)
498 >>> image_patches = image.extract_patches_2d(image=one_image, patch_size=(10, 10))
499 >>> print('Patches shape: {}'.format(image_patches.shape))
500 Patches shape: (263758, 10, 10, 3)
501 >>> image_reconstructed = image.reconstruct_from_patches_2d(
502 ... patches=image_patches,
503 ... image_size=one_image.shape
504 ... )
505 >>> print(f"Reconstructed shape: {image_reconstructed.shape}")
506 Reconstructed shape: (427, 640, 3)
507 """
508 i_h, i_w = image_size[:2]
509 p_h, p_w = patches.shape[1:3]
510 img = np.zeros(image_size)
511 # compute the dimensions of the patches array
512 n_h = i_h - p_h + 1
513 n_w = i_w - p_w + 1
514 for p, (i, j) in zip(patches, product(range(n_h), range(n_w))):
515 img[i : i + p_h, j : j + p_w] += p
516
517 for i in range(i_h):
518 for j in range(i_w):
519 # divide by the amount of overlap
520 # XXX: is this the most efficient way? memory-wise yes, cpu wise?
521 img[i, j] /= float(min(i + 1, p_h, i_h - i) * min(j + 1, p_w, i_w - j))
522 return img

525class PatchExtractor(TransformerMixin, BaseEstimator):
526 """Extracts patches from a collection of images.
527
528 Read more in the :ref:`User Guide <image_feature_extraction>`.
529
530 .. versionadded:: 0.9
531
532 Parameters
533 ----------
534 patch_size : tuple of int (patch_height, patch_width), default=None
535 The dimensions of one patch. If set to None, the patch size will be
536 automatically set to `(img_height // 10, img_width // 10)`, where
537 `img_height` and `img_width` are the dimensions of the input images.
538
539 max_patches : int or float, default=None
540 The maximum number of patches per image to extract. If `max_patches` is
541 a float in (0, 1), it is taken to mean a proportion of the total number
542 of patches. If set to None, extract all possible patches.
543
544 random_state : int, RandomState instance, default=None
545 Determines the random number generator used for random sampling when
546 `max_patches is not None`. Use an int to make the randomness
547 deterministic.
548 See :term:`Glossary <random_state>`.
549
550 See Also
551 --------
552 reconstruct_from_patches_2d : Reconstruct image from all of its patches.
553
554 Notes
555 -----
556 This estimator is stateless and does not need to be fitted. However, we
557 recommend to call :meth:`fit_transform` instead of :meth:`transform`, as
558 parameter validation is only performed in :meth:`fit`.
559
560 Examples
561 --------
562 >>> from sklearn.datasets import load_sample_images
563 >>> from sklearn.feature_extraction import image
564 >>> # Use the array data from the second image in this dataset:
565 >>> X = load_sample_images().images[1]
566 >>> X = X[None, ...]
567 >>> print(f"Image shape: {X.shape}")
568 Image shape: (1, 427, 640, 3)
569 >>> pe = image.PatchExtractor(patch_size=(10, 10))
570 >>> pe_trans = pe.transform(X)
571 >>> print(f"Patches shape: {pe_trans.shape}")
572 Patches shape: (263758, 10, 10, 3)
573 >>> X_reconstructed = image.reconstruct_from_patches_2d(pe_trans, X.shape[1:])
574 >>> print(f"Reconstructed shape: {X_reconstructed.shape}")
575 Reconstructed shape: (427, 640, 3)
576 """

618 def transform(self, X):
619 """Transform the image samples in `X` into a matrix of patch data.
620
621 Parameters
622 ----------
623 X : ndarray of shape (n_samples, image_height, image_width) or \
624 (n_samples, image_height, image_width, n_channels)
625 Array of images from which to extract patches. For color images,
626 the last dimension specifies the channel: a RGB image would have
627 `n_channels=3`.
628
629 Returns
630 -------
631 patches : array of shape (n_patches, patch_height, patch_width) or \
632 (n_patches, patch_height, patch_width, n_channels)
633 The collection of patches extracted from the images, where
634 `n_patches` is either `n_samples * max_patches` or the total
635 number of patches that can be extracted.
636 """
637 X = validate_data(
638 self,
639 X=X,
640 ensure_2d=False,
641 allow_nd=True,
642 ensure_min_samples=1,
643 ensure_min_features=1,
644 reset=False,
645 )
646 random_state = check_random_state(self.random_state)
647 n_imgs, img_height, img_width = X.shape[:3]
648 if self.patch_size is None:
649 patch_size = img_height // 10, img_width // 10
650 else:
651 if len(self.patch_size) != 2:
652 raise ValueError(
653 "patch_size must be a tuple of two integers. Got"
654 f" {self.patch_size} instead."
655 )
656 patch_size = self.patch_size
657
658 n_imgs, img_height, img_width = X.shape[:3]
659 X = np.reshape(X, (n_imgs, img_height, img_width, -1))
660 n_channels = X.shape[-1]
661
662 # compute the dimensions of the patches array
663 patch_height, patch_width = patch_size
664 n_patches = _compute_n_patches(
665 img_height, img_width, patch_height, patch_width, self.max_patches
666 )
667 patches_shape = (n_imgs * n_patches,) + patch_size
668 if n_channels > 1:
669 patches_shape += (n_channels,)
670
671 # extract the patches
672 patches = np.empty(patches_shape)
673 for ii, image in enumerate(X):
674 patches[ii * n_patches : (ii + 1) * n_patches] = extract_patches_2d(
675 image,
676 patch_size,
677 max_patches=self.max_patches,
678 random_state=random_state,
679 )
680 return patches
