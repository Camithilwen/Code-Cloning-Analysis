34def _deprecate_positional_args(func=None, *, version="1.3"):
35 """Decorator for methods that issues warnings for positional arguments.
36
37 Using the keyword-only argument syntax in pep 3102, arguments after the
38 * will issue a warning when passed as a positional argument.
39
40 Parameters
41 ----------
42 func : callable, default=None
43 Function to check arguments on.
44 version : callable, default="1.3"
45 The version when positional arguments will result in error.
46 """
47
48 def _inner_deprecate_positional_args(f):
49 sig = signature(f)
50 kwonly_args = []
51 all_args = []
52
53 for name, param in sig.parameters.items():
54 if param.kind == Parameter.POSITIONAL_OR_KEYWORD:
55 all_args.append(name)
56 elif param.kind == Parameter.KEYWORD_ONLY:
57 kwonly_args.append(name)
58
59 @wraps(f)
60 def inner_f(*args, **kwargs):
61 extra_args = len(args) - len(all_args)
62 if extra_args <= 0:
63 return f(*args, **kwargs)
64
65 # extra_args > 0
66 args_msg = [
67 "{}={}".format(name, arg)
68 for name, arg in zip(kwonly_args[:extra_args], args[-extra_args:])
69 ]
70 args_msg = ", ".join(args_msg)
71 warnings.warn(
72 (
73 f"Pass {args_msg} as keyword args. From version "
74 f"{version} passing these as positional arguments "
75 "will result in an error"
76 ),
77 FutureWarning,
78 )
79 kwargs.update(zip(sig.parameters, args))
80 return f(**kwargs)
81
82 return inner_f
83
84 if func is not None:
85 return _inner_deprecate_positional_args(func)
86
87 return _inner_deprecate_positional_args

328def _num_features(X):
329 """Return the number of features in an array-like X.
330
331 This helper function tries hard to avoid to materialize an array version
332 of X unless necessary. For instance, if X is a list of lists,
333 this function will return the length of the first element, assuming
334 that subsequent elements are all lists of the same length without
335 checking.
336 Parameters
337 ----------
338 X : array-like
339 array-like to get the number of features.
340
341 Returns
342 -------
343 features : int
344 Number of features
345 """
346 type_ = type(X)
347 if type_.__module__ == "builtins":
348 type_name = type_.__qualname__
349 else:
350 type_name = f"{type_.__module__}.{type_.__qualname__}"
351 message = f"Unable to find the number of features from X of type {type_name}"
352 if not hasattr(X, "__len__") and not hasattr(X, "shape"):
353 if not hasattr(X, "__array__"):
354 raise TypeError(message)
355 # Only convert X to a numpy array if there is no cheaper, heuristic
356 # option.
357 X = np.asarray(X)
358
359 if hasattr(X, "shape"):
360 if not hasattr(X.shape, "__len__") or len(X.shape) <= 1:
361 message += f" with shape {X.shape}"
362 raise TypeError(message)
363 return X.shape[1]
364
365 first_sample = X[0]
366
367 # Do not consider an array-like of strings or dicts to be a 2D array
368 if isinstance(first_sample, (str, bytes, dict)):
369 message += f" where the samples are of type {type(first_sample).__qualname__}"
370 raise TypeError(message)
371
372 try:
373 # If X is a list of lists, for instance, we assume that all nested
374 # lists have the same length without checking or converting to
375 # a numpy array to keep this function call as cheap as possible.
376 return len(first_sample)
377 except Exception as err:
378 raise TypeError(message) from err

1554def check_symmetric(array, *, tol=1e-10, raise_warning=True, raise_exception=False):
1555 """Make sure that array is 2D, square and symmetric.
1556
1557 If the array is not symmetric, then a symmetrized version is returned.
1558 Optionally, a warning or exception is raised if the matrix is not
1559 symmetric.
1560
1561 Parameters
1562 ----------
1563 array : {ndarray, sparse matrix}
1564 Input object to check / convert. Must be two-dimensional and square,
1565 otherwise a ValueError will be raised.
1566
1567 tol : float, default=1e-10
1568 Absolute tolerance for equivalence of arrays. Default = 1E-10.
1569
1570 raise_warning : bool, default=True
1571 If True then raise a warning if conversion is required.
1572
1573 raise_exception : bool, default=False
1574 If True then raise an exception if array is not symmetric.
1575
1576 Returns
1577 -------
1578 array_sym : {ndarray, sparse matrix}
1579 Symmetrized version of the input array, i.e. the average of array
1580 and array.transpose(). If sparse, then duplicate entries are first
1581 summed and zeros are eliminated.
1582
1583 Examples
1584 --------
1585 >>> import numpy as np
1586 >>> from sklearn.utils.validation import check_symmetric
1587 >>> symmetric_array = np.array([[0, 1, 2], [1, 0, 1], [2, 1, 0]])
1588 >>> check_symmetric(symmetric_array)
1589 array([[0, 1, 2],
1590 [1, 0, 1],
1591 [2, 1, 0]])
1592 >>> from scipy.sparse import csr_matrix
1593 >>> sparse_symmetric_array = csr_matrix(symmetric_array)
1594 >>> check_symmetric(sparse_symmetric_array)
1595 <Compressed Sparse Row sparse matrix of dtype 'int64'
1596 with 6 stored elements and shape (3, 3)>
1597 """
1598 if (array.ndim != 2) or (array.shape[0] != array.shape[1]):
1599 raise ValueError(
1600 "array must be 2-dimensional and square. shape = {0}".format(array.shape)
1601 )
1602
1603 if sp.issparse(array):
1604 diff = array - array.T
1605 # only csr, csc, and coo have `data` attribute
1606 if diff.format not in ["csr", "csc", "coo"]:
1607 diff = diff.tocsr()
1608 symmetric = np.all(abs(diff.data) < tol)
1609 else:
1610 symmetric = np.allclose(array, array.T, atol=tol)
1611
1612 if not symmetric:
1613 if raise_exception:
1614 raise ValueError("Array must be symmetric")
1615 if raise_warning:
1616 warnings.warn(
1617 (
1618 "Array is not symmetric, and will be converted "
1619 "to symmetric by average with its transpose."
1620 ),
1621 stacklevel=2,
1622 )
1623 if sp.issparse(array):
1624 conversion = "to" + array.format
1625 array = getattr(0.5 * (array + array.T), conversion)()
1626 else:
1627 array = 0.5 * (array + array.T)
1628
1629 return array

1758 """Check if we can delegate a method to the underlying estimator.
1759
1760 We check the `delegates` in the order they are passed. By default, we first check
1761 the fitted estimator if available, otherwise we check the unfitted estimator.
1762

1955def _check_psd_eigenvalues(lambdas, enable_warnings=False):
1956 """Check the eigenvalues of a positive semidefinite (PSD) matrix.
1957
1958 Checks the provided array of PSD matrix eigenvalues for numerical or
1959 conditioning issues and returns a fixed validated version. This method
1960 should typically be used if the PSD matrix is user-provided (e.g. a
1961 Gram matrix) or computed using a user-provided dissimilarity metric
1962 (e.g. kernel function), or if the decomposition process uses approximation
1963 methods (randomized SVD, etc.).
1964
1965 It checks for three things:
1966
1967 - that there are no significant imaginary parts in eigenvalues (more than
1968 1e-5 times the maximum real part). If this check fails, it raises a
1969 ``ValueError``. Otherwise all non-significant imaginary parts that may
1970 remain are set to zero. This operation is traced with a
1971 ``PositiveSpectrumWarning`` when ``enable_warnings=True``.
1972
1973 - that eigenvalues are not all negative. If this check fails, it raises a
1974 ``ValueError``
1975
1976 - that there are no significant negative eigenvalues with absolute value
1977 more than 1e-10 (1e-6) and more than 1e-5 (5e-3) times the largest
1978 positive eigenvalue in double (simple) precision. If this check fails,
1979 it raises a ``ValueError``. Otherwise all negative eigenvalues that may
1980 remain are set to zero. This operation is traced with a
1981 ``PositiveSpectrumWarning`` when ``enable_warnings=True``.
1982
1983 Finally, all the positive eigenvalues that are too small (with a value
1984 smaller than the maximum eigenvalue multiplied by 1e-12 (2e-7)) are set to
1985 zero. This operation is traced with a ``PositiveSpectrumWarning`` when
1986 ``enable_warnings=True``.
1987
1988 Parameters
1989 ----------
1990 lambdas : array-like of shape (n_eigenvalues,)
1991 Array of eigenvalues to check / fix.
1992
1993 enable_warnings : bool, default=False
1994 When this is set to ``True``, a ``PositiveSpectrumWarning`` will be
1995 raised when there are imaginary parts, negative eigenvalues, or
1996 extremely small non-zero eigenvalues. Otherwise no warning will be
1997 raised. In both cases, imaginary parts, negative eigenvalues, and
1998 extremely small non-zero eigenvalues will be set to zero.
1999
2000 Returns
2001 -------
2002 lambdas_fixed : ndarray of shape (n_eigenvalues,)
2003 A fixed validated copy of the array of eigenvalues.
2004
2005 Examples
2006 --------
2007 >>> from sklearn.utils.validation import _check_psd_eigenvalues
2008 >>> _check_psd_eigenvalues([1, 2]) # nominal case
2009 array([1, 2])
2010 >>> _check_psd_eigenvalues([5, 5j]) # significant imag part
2011 Traceback (most recent call last):
2012 ...
2013 ValueError: There are significant imaginary parts in eigenvalues (1
2014 of the maximum real part). Either the matrix is not PSD, or there was
2015 an issue while computing the eigendecomposition of the matrix.
2016 >>> _check_psd_eigenvalues([5, 5e-5j]) # insignificant imag part
2017 array([5., 0.])
2018 >>> _check_psd_eigenvalues([-5, -1]) # all negative
2019 Traceback (most recent call last):
2020 ...
2021 ValueError: All eigenvalues are negative (maximum is -1). Either the
2022 matrix is not PSD, or there was an issue while computing the
2023 eigendecomposition of the matrix.
2024 >>> _check_psd_eigenvalues([5, -1]) # significant negative
2025 Traceback (most recent call last):
2026 ...
2027 ValueError: There are significant negative eigenvalues (0.2 of the
2028 maximum positive). Either the matrix is not PSD, or there was an issue
2029 while computing the eigendecomposition of the matrix.
2030 >>> _check_psd_eigenvalues([5, -5e-5]) # insignificant negative
2031 array([5., 0.])
2032 >>> _check_psd_eigenvalues([5, 4e-12]) # bad conditioning (too small)
2033 array([5., 0.])
2034
2035 """
2036
2037 lambdas = np.array(lambdas)
2038 is_double_precision = lambdas.dtype == np.float64
2039
2040 # note: the minimum value available is
2041 # - single-precision: np.finfo('float32').eps = 1.2e-07
2042 # - double-precision: np.finfo('float64').eps = 2.2e-16
2043
2044 # the various thresholds used for validation
2045 # we may wish to change the value according to precision.
2046 significant_imag_ratio = 1e-5
2047 significant_neg_ratio = 1e-5 if is_double_precision else 5e-3
2048 significant_neg_value = 1e-10 if is_double_precision else 1e-6
2049 small_pos_ratio = 1e-12 if is_double_precision else 2e-7
2050
2051 # Check that there are no significant imaginary parts
2052 if not np.isreal(lambdas).all():
2053 max_imag_abs = np.abs(np.imag(lambdas)).max()
2054 max_real_abs = np.abs(np.real(lambdas)).max()
2055 if max_imag_abs > significant_imag_ratio * max_real_abs:
2056 raise ValueError(
2057 "There are significant imaginary parts in eigenvalues (%g "
2058 "of the maximum real part). Either the matrix is not PSD, or "
2059 "there was an issue while computing the eigendecomposition "
2060 "of the matrix." % (max_imag_abs / max_real_abs)
2061 )
2062
2063 # warn about imaginary parts being removed
2064 if enable_warnings:
2065 warnings.warn(
2066 "There are imaginary parts in eigenvalues (%g "
2067 "of the maximum real part). Either the matrix is not"
2068 " PSD, or there was an issue while computing the "
2069 "eigendecomposition of the matrix. Only the real "
