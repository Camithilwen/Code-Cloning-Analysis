11class DetCurveDisplay(_BinaryClassifierCurveDisplayMixin):
12 """Detection Error Tradeoff (DET) curve visualization.
13
14 It is recommended to use :func:`~sklearn.metrics.DetCurveDisplay.from_estimator`
15 or :func:`~sklearn.metrics.DetCurveDisplay.from_predictions` to create a
16 visualizer. All parameters are stored as attributes.
17
18 For general information regarding `scikit-learn` visualization tools, see
19 the :ref:`Visualization Guide <visualizations>`.
20 For guidance on interpreting these plots, refer to the
21 :ref:`Model Evaluation Guide <det_curve>`.
22
23 .. versionadded:: 0.24
24
25 Parameters
26 ----------
27 fpr : ndarray
28 False positive rate.
29
30 fnr : ndarray
31 False negative rate.
32
33 estimator_name : str, default=None
34 Name of estimator. If None, the estimator name is not shown.
35
36 pos_label : int, float, bool or str, default=None
37 The label of the positive class.
38
39 Attributes
40 ----------
41 line_ : matplotlib Artist
42 DET Curve.
43
44 ax_ : matplotlib Axes
45 Axes with DET Curve.
46
47 figure_ : matplotlib Figure
48 Figure containing the curve.
49
50 See Also
51 --------
52 det_curve : Compute error rates for different probability thresholds.
53 DetCurveDisplay.from_estimator : Plot DET curve given an estimator and
54 some data.
55 DetCurveDisplay.from_predictions : Plot DET curve given the true and
56 predicted labels.
57
58 Examples
59 --------
60 >>> import matplotlib.pyplot as plt
61 >>> from sklearn.datasets import make_classification
62 >>> from sklearn.metrics import det_curve, DetCurveDisplay
63 >>> from sklearn.model_selection import train_test_split
64 >>> from sklearn.svm import SVC
65 >>> X, y = make_classification(n_samples=1000, random_state=0)
66 >>> X_train, X_test, y_train, y_test = train_test_split(
67 ... X, y, test_size=0.4, random_state=0)
68 >>> clf = SVC(random_state=0).fit(X_train, y_train)
69 >>> y_pred = clf.decision_function(X_test)
70 >>> fpr, fnr, _ = det_curve(y_test, y_pred)
71 >>> display = DetCurveDisplay(
72 ... fpr=fpr, fnr=fnr, estimator_name="SVC"
73 ... )
74 >>> display.plot()
75 <...>
76 >>> plt.show()

308 def plot(self, ax=None, *, name=None, **kwargs):
309 """Plot visualization.
310
311 Parameters
312 ----------
313 ax : matplotlib axes, default=None
314 Axes object to plot on. If `None`, a new figure and axes is
315 created.
316
317 name : str, default=None
318 Name of DET curve for labeling. If `None`, use `estimator_name` if
319 it is not `None`, otherwise no labeling is shown.
320
321 **kwargs : dict
322 Additional keywords arguments passed to matplotlib `plot` function.
323
324 Returns
325 -------
326 display : :class:`~sklearn.metrics.DetCurveDisplay`
327 Object that stores computed values.
328 """
329 self.ax_, self.figure_, name = self._validate_plot_params(ax=ax, name=name)
330
331 line_kwargs = {} if name is None else {"label": name}
332 line_kwargs.update(**kwargs)
333
334 # We have the following bounds:
335 # sp.stats.norm.ppf(0.0) = -np.inf
336 # sp.stats.norm.ppf(1.0) = np.inf
337 # We therefore clip to eps and 1 - eps to not provide infinity to matplotlib.
338 eps = np.finfo(self.fpr.dtype).eps
339 self.fpr = self.fpr.clip(eps, 1 - eps)
340 self.fnr = self.fnr.clip(eps, 1 - eps)
341
342 (self.line_,) = self.ax_.plot(
343 sp.stats.norm.ppf(self.fpr),
344 sp.stats.norm.ppf(self.fnr),
345 **line_kwargs,
346 )
347 info_pos_label = (
348 f" (Positive label: {self.pos_label})" if self.pos_label is not None else ""
349 )
350
351 xlabel = "False Positive Rate" + info_pos_label
352 ylabel = "False Negative Rate" + info_pos_label
353 self.ax_.set(xlabel=xlabel, ylabel=ylabel)
354
355 if "label" in line_kwargs:
356 self.ax_.legend(loc="lower right")
357
358 ticks = [0.001, 0.01, 0.05, 0.20, 0.5, 0.80, 0.95, 0.99, 0.999]
359 tick_locations = sp.stats.norm.ppf(ticks)
360 tick_labels = [
361 "{:.0%}".format(s) if (100 * s).is_integer() else "{:.1%}".format(s)
362 for s in ticks
363 ]
364 self.ax_.set_xticks(tick_locations)
365 self.ax_.set_xticklabels(tick_labels)
366 self.ax_.set_xlim(-3, 3)
367 self.ax_.set_yticks(tick_locations)
368 self.ax_.set_yticklabels(tick_labels)
369 self.ax_.set_ylim(-3, 3)
370
371 return self
