44def _append_path(ctx, path, transform, clip=None):
45 for points, code in path.iter_segments(
46 transform, remove_nans=True, clip=clip):
47 if code == Path.MOVETO:
48 ctx.move_to(*points)
49 elif code == Path.CLOSEPOLY:
50 ctx.close_path()
51 elif code == Path.LINETO:
52 ctx.line_to(*points)
53 elif code == Path.CURVE3:
54 cur = np.asarray(ctx.get_current_point())
55 a = points[:2]
56 b = points[-2:]
57 ctx.curve_to(*(cur / 3 + a * 2 / 3), *(a * 2 / 3 + b / 3), *b)
58 elif code == Path.CURVE4:
59 ctx.curve_to(*points)

62def _cairo_font_args_from_font_prop(prop):
63 """
64 Convert a `.FontProperties` or a `.FontEntry` to arguments that can be
65 passed to `.Context.select_font_face`.
66 """
67 def attr(field):
68 try:
69 return getattr(prop, f"get_{field}")()
70 except AttributeError:
71 return getattr(prop, field)
72
73 name = attr("name")
74 slant = getattr(cairo, f"FONT_SLANT_{attr('style').upper()}")
75 weight = attr("weight")
76 weight = (cairo.FONT_WEIGHT_NORMAL
77 if font_manager.weight_dict.get(weight, weight) < 550
78 else cairo.FONT_WEIGHT_BOLD)
79 return name, slant, weight

156 def draw_markers(self, gc, marker_path, marker_trans, path, transform,
157 rgbFace=None):
158 # docstring inherited
159
160 ctx = gc.ctx
161 ctx.new_path()
162 # Create the path for the marker; it needs to be flipped here already!
163 _append_path(ctx, marker_path, marker_trans + Affine2D().scale(1, -1))
164 marker_path = ctx.copy_path_flat()
165
166 # Figure out whether the path has a fill
167 x1, y1, x2, y2 = ctx.fill_extents()
168 if x1 == 0 and y1 == 0 and x2 == 0 and y2 == 0:
169 filled = False
170 # No fill, just unset this (so we don't try to fill it later on)
171 rgbFace = None
172 else:
173 filled = True
174
175 transform = (transform
176 + Affine2D().scale(1, -1).translate(0, self.height))
177
178 ctx.new_path()
179 for i, (vertices, codes) in enumerate(
180 path.iter_segments(transform, simplify=False)):
181 if len(vertices):
182 x, y = vertices[-2:]
183 ctx.save()
184
185 # Translate and apply path
186 ctx.translate(x, y)
187 ctx.append_path(marker_path)
188
189 ctx.restore()
190
191 # Slower code path if there is a fill; we need to draw
192 # the fill and stroke for each marker at the same time.
193 # Also flush out the drawing every once in a while to
194 # prevent the paths from getting way too long.
195 if filled or i % 1000 == 0:
196 self._fill_and_stroke(
197 ctx, rgbFace, gc.get_alpha(), gc.get_forced_alpha())
198
199 # Fast path, if there is no fill, draw everything in one step
200 if not filled:
201 self._fill_and_stroke(
202 ctx, rgbFace, gc.get_alpha(), gc.get_forced_alpha())

416 def copy_from_bbox(self, bbox):
417 surface = self._renderer.gc.ctx.get_target()
418 if not isinstance(surface, cairo.ImageSurface):
419 raise RuntimeError(
420 "copy_from_bbox only works when rendering to an ImageSurface")
421 sw = surface.get_width()
422 sh = surface.get_height()
423 x0 = math.ceil(bbox.x0)
424 x1 = math.floor(bbox.x1)
425 y0 = math.ceil(sh - bbox.y1)
426 y1 = math.floor(sh - bbox.y0)
427 if not (0 <= x0 and x1 <= sw and bbox.x0 <= bbox.x1
428 and 0 <= y0 and y1 <= sh and bbox.y0 <= bbox.y1):
429 raise ValueError("Invalid bbox")
430 sls = slice(y0, y0 + max(y1 - y0, 0)), slice(x0, x0 + max(x1 - x0, 0))
431 data = (np.frombuffer(surface.get_data(), np.uint32)
432 .reshape((sh, sw))[sls].copy())
433 return _CairoRegion(sls, data)

468 def _save(self, fmt, fobj, *, orientation='portrait'):
469 # save PDF/PS/SVG
470
471 dpi = 72
472 self.figure.dpi = dpi
473 w_in, h_in = self.figure.get_size_inches()
474 width_in_points, height_in_points = w_in * dpi, h_in * dpi
475
476 if orientation == 'landscape':
477 width_in_points, height_in_points = (
478 height_in_points, width_in_points)
479
480 if fmt == 'ps':
481 if not hasattr(cairo, 'PSSurface'):
482 raise RuntimeError('cairo has not been compiled with PS '
483 'support enabled')
484 surface = cairo.PSSurface(fobj, width_in_points, height_in_points)
485 elif fmt == 'pdf':
486 if not hasattr(cairo, 'PDFSurface'):
487 raise RuntimeError('cairo has not been compiled with PDF '
488 'support enabled')
489 surface = cairo.PDFSurface(fobj, width_in_points, height_in_points)
490 elif fmt in ('svg', 'svgz'):
491 if not hasattr(cairo, 'SVGSurface'):
492 raise RuntimeError('cairo has not been compiled with SVG '
493 'support enabled')
494 if fmt == 'svgz':
495 if isinstance(fobj, str):
496 fobj = gzip.GzipFile(fobj, 'wb')
497 else:
498 fobj = gzip.GzipFile(None, 'wb', fileobj=fobj)
499 surface = cairo.SVGSurface(fobj, width_in_points, height_in_points)
500 else:
501 raise ValueError(f"Unknown format: {fmt!r}")
502
503 self._renderer.dpi = self.figure.dpi
504 self._renderer.set_context(cairo.Context(surface))
505 ctx = self._renderer.gc.ctx
506
507 if orientation == 'landscape':
508 ctx.rotate(np.pi / 2)
509 ctx.translate(0, -height_in_points)
510 # Perhaps add an '%%Orientation: Landscape' comment?
511
512 self.figure.draw(self._renderer)
513
514 ctx.show_page()
515 surface.finish()
516 if fmt == 'svgz':
517 fobj.close()
518
519 print_pdf = functools.partialmethod(_save, "pdf")
520 print_ps = functools.partialmethod(_save, "ps")
521 print_svg = functools.partialmethod(_save, "svg")
522 print_svgz = functools.partialmethod(_save, "svgz")
