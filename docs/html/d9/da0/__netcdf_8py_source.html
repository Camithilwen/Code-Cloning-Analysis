98class netcdf_file:
99 """
100 A file object for NetCDF data.
101
102 A `netcdf_file` object has two standard attributes: `dimensions` and
103 `variables`. The values of both are dictionaries, mapping dimension
104 names to their associated lengths and variable names to variables,
105 respectively. Application programs should never modify these
106 dictionaries.
107
108 All other attributes correspond to global attributes defined in the
109 NetCDF file. Global file attributes are created by assigning to an
110 attribute of the `netcdf_file` object.
111
112 Parameters
113 ----------
114 filename : string or file-like
115 string -> filename
116 mode : {'r', 'w', 'a'}, optional
117 read-write-append mode, default is 'r'
118 mmap : None or bool, optional
119 Whether to mmap `filename` when reading. Default is True
120 when `filename` is a file name, False when `filename` is a
121 file-like object. Note that when mmap is in use, data arrays
122 returned refer directly to the mmapped data on disk, and the
123 file cannot be closed as long as references to it exist.
124 version : {1, 2}, optional
125 version of netcdf to read / write, where 1 means *Classic
126 format* and 2 means *64-bit offset format*. Default is 1. See
127 `here <https://www.unidata.ucar.edu/software/netcdf/docs/netcdf_introduction.html#select_format>`__
128 for more info.
129 maskandscale : bool, optional
130 Whether to automatically scale and/or mask data based on attributes.
131 Default is False.
132
133 Notes
134 -----
135 The major advantage of this module over other modules is that it doesn't
136 require the code to be linked to the NetCDF libraries. This module is
137 derived from `pupynere <https://bitbucket.org/robertodealmeida/pupynere/>`_.
138
139 NetCDF files are a self-describing binary data format. The file contains
140 metadata that describes the dimensions and variables in the file. More
141 details about NetCDF files can be found `here
142 <https://www.unidata.ucar.edu/software/netcdf/guide_toc.html>`__. There
143 are three main sections to a NetCDF data structure:
144
145 1. Dimensions
146 2. Variables
147 3. Attributes
148
149 The dimensions section records the name and length of each dimension used
150 by the variables. The variables would then indicate which dimensions it
151 uses and any attributes such as data units, along with containing the data
152 values for the variable. It is good practice to include a
153 variable that is the same name as a dimension to provide the values for
154 that axes. Lastly, the attributes section would contain additional
155 information such as the name of the file creator or the instrument used to
156 collect the data.
157
158 When writing data to a NetCDF file, there is often the need to indicate the
159 'record dimension'. A record dimension is the unbounded dimension for a
160 variable. For example, a temperature variable may have dimensions of
161 latitude, longitude and time. If one wants to add more temperature data to
162 the NetCDF file as time progresses, then the temperature variable should
163 have the time dimension flagged as the record dimension.
164
165 In addition, the NetCDF file header contains the position of the data in
166 the file, so access can be done in an efficient manner without loading
167 unnecessary data into memory. It uses the ``mmap`` module to create
168 Numpy arrays mapped to the data on disk, for the same purpose.
169
170 Note that when `netcdf_file` is used to open a file with mmap=True
171 (default for read-only), arrays returned by it refer to data
172 directly on the disk. The file should not be closed, and cannot be cleanly
173 closed when asked, if such arrays are alive. You may want to copy data arrays
174 obtained from mmapped Netcdf file if they are to be processed after the file
175 is closed, see the example below.
176
177 Examples
178 --------
179 To create a NetCDF file:
180
181 >>> from scipy.io import netcdf_file
182 >>> import numpy as np
183 >>> f = netcdf_file('simple.nc', 'w')
184 >>> f.history = 'Created for a test'
185 >>> f.createDimension('time', 10)
186 >>> time = f.createVariable('time', 'i', ('time',))
187 >>> time[:] = np.arange(10)
188 >>> time.units = 'days since 2008-01-01'
189 >>> f.close()
190
191 Note the assignment of ``arange(10)`` to ``time[:]``. Exposing the slice
192 of the time variable allows for the data to be set in the object, rather
193 than letting ``arange(10)`` overwrite the ``time`` variable.
194
195 To read the NetCDF file we just created:
196
197 >>> from scipy.io import netcdf_file
198 >>> f = netcdf_file('simple.nc', 'r')
199 >>> print(f.history)
200 b'Created for a test'
201 >>> time = f.variables['time']
202 >>> print(time.units)
203 b'days since 2008-01-01'
204 >>> print(time.shape)
205 (10,)
206 >>> print(time[-1])
207 9
208
209 NetCDF files, when opened read-only, return arrays that refer
210 directly to memory-mapped data on disk:
211
212 >>> data = time[:]
213
214 If the data is to be processed after the file is closed, it needs
215 to be copied to main memory:
216
217 >>> data = time[:].copy()
218 >>> f.close()
219 >>> data.mean()
220 4.5
221
222 A NetCDF file can also be used as context manager:
223
224 >>> from scipy.io import netcdf_file
225 >>> with netcdf_file('simple.nc', 'r') as f:
226 ... print(f.history)
227 b'Created for a test'
228

289 def close(self):
290 """Closes the NetCDF file."""
291 if hasattr(self, 'fp') and not self.fp.closed:
292 try:
293 self.flush()
294 finally:
295 self.variables = {}
296 if self._mm_buf is not None:
297 ref = weakref.ref(self._mm_buf)
298 self._mm_buf = None
299 if ref() is None:
300 # self._mm_buf is gc'd, and we can close the mmap
301 self._mm.close()
302 else:
303 # we cannot close self._mm, since self._mm_buf is
304 # alive and there may still be arrays referring to it
305 warnings.warn((
306 "Cannot close a netcdf_file opened with mmap=True, when "
307 "netcdf_variables or arrays referring to its data still exist. "
308 "All data arrays obtained from such files refer directly to "
309 "data on disk, and must be copied before the file can be cleanly "
310 "closed. (See netcdf_file docstring for more information on mmap.)"
311 ), category=RuntimeWarning)
312 self._mm = None
313 self.fp.close()
314 __del__ = close

348 def createVariable(self, name, type, dimensions):
349 """
350 Create an empty variable for the `netcdf_file` object, specifying its data
351 type and the dimensions it uses.
352
353 Parameters
354 ----------
355 name : str
356 Name of the new variable.
357 type : dtype or str
358 Data type of the variable.
359 dimensions : sequence of str
360 List of the dimension names used by the variable, in the desired order.
361
362 Returns
363 -------
364 variable : netcdf_variable
365 The newly created ``netcdf_variable`` object.
366 This object has also been added to the `netcdf_file` object as well.
367
368 See Also
369 --------
370 createDimension
371

406 def _write(self):
407 self.fp.seek(0)
408 self.fp.write(b'CDF')
409 self.fp.write(array(self.version_byte, '>b').tobytes())

417 def _write_numrecs(self):
418 # Get highest record count from all record variables.
419 for var in self.variables.values():
420 if var.isrec and len(var.data) > self._recs:
421 self.__dict__['_recs'] = len(var.data)
422 self._pack_int(self._recs)

424 def _write_dim_array(self):
425 if self.dimensions:
426 self.fp.write(NC_DIMENSION)
427 self._pack_int(len(self.dimensions))
428 for name in self._dims:
429 self._pack_string(name)
430 length = self.dimensions[name]
431 self._pack_int(length or 0) # replace None with 0 for record dimension
432 else:
433 self.fp.write(ABSENT)

435 def _write_gatt_array(self):
436 self._write_att_array(self._attributes)

438 def _write_att_array(self, attributes):
439 if attributes:
440 self.fp.write(NC_ATTRIBUTE)
441 self._pack_int(len(attributes))
442 for name, values in attributes.items():
443 self._pack_string(name)
444 self._write_att_values(values)
445 else:

448 def _write_var_array(self):
449 if self.variables:
450 self.fp.write(NC_VARIABLE)
451 self._pack_int(len(self.variables))
452
453 # Sort variable names non-recs first, then recs.
454 def sortkey(n):
455 v = self.variables[n]
456 if v.isrec:
457 return (-1,)
458 return v._shape
459 variables = sorted(self.variables, key=sortkey, reverse=True)
460
461 # Set the metadata for all variables.
462 for name in variables:
463 self._write_var_metadata(name)
464 # Now that we have the metadata, we know the vsize of
465 # each record variable, so we can calculate recsize.
466 self.__dict__['_recsize'] = sum([
467 var._vsize for var in self.variables.values()
468 if var.isrec])
469 # Set the data for all variables.
470 for name in variables:
471 self._write_var_data(name)
472 else:
473 self.fp.write(ABSENT)

475 def _write_var_metadata(self, name):
476 var = self.variables[name]
477
478 self._pack_string(name)
479 self._pack_int(len(var.dimensions))
480 for dimname in var.dimensions:
481 dimid = self._dims.index(dimname)
482 self._pack_int(dimid)
483
484 self._write_att_array(var._attributes)
485
486 nc_type = REVERSE[var.typecode(), var.itemsize()]
487 self.fp.write(nc_type)
488
489 if not var.isrec:
490 vsize = var.data.size * var.data.itemsize
491 vsize += -vsize % 4
492 else: # record variable
493 try:
494 vsize = var.data[0].size * var.data.itemsize
495 except IndexError:
496 vsize = 0
497 rec_vars = len([v for v in self.variables.values()
498 if v.isrec])
499 if rec_vars > 1:
500 vsize += -vsize % 4
501 self.variables[name].__dict__['_vsize'] = vsize
502 self._pack_int(vsize)
503
504 # Pack a bogus begin, and set the real value later.
505 self.variables[name].__dict__['_begin'] = self.fp.tell()
506 self._pack_begin(0)

508 def _write_var_data(self, name):
509 var = self.variables[name]
510
511 # Set begin in file header.
512 the_beguine = self.fp.tell()
513 self.fp.seek(var._begin)
514 self._pack_begin(the_beguine)
515 self.fp.seek(the_beguine)
516
517 # Write data.
518 if not var.isrec:
519 self.fp.write(var.data.tobytes())
520 count = var.data.size * var.data.itemsize
521 self._write_var_padding(var, var._vsize - count)
522 else: # record variable
523 # Handle rec vars with shape[0] < nrecs.
524 if self._recs > len(var.data):
525 shape = (self._recs,) + var.data.shape[1:]
526 # Resize in-place does not always work since
527 # the array might not be single-segment
528 try:
529 var.data.resize(shape)
530 except ValueError:
531 var.__dict__['data'] = np.resize(var.data, shape).astype(var.data.dtype)
532
533 pos0 = pos = self.fp.tell()
534 for rec in var.data:
535 # Apparently scalars cannot be converted to big endian. If we
536 # try to convert a ``=i4`` scalar to, say, '>i4' the dtype
537 # will remain as ``=i4``.
538 if not rec.shape and (rec.dtype.byteorder == '<' or
539 (rec.dtype.byteorder == '=' and LITTLE_ENDIAN)):
540 rec = rec.byteswap()
541 self.fp.write(rec.tobytes())
542 # Padding
543 count = rec.size * rec.itemsize
544 self._write_var_padding(var, var._vsize - count)
545 pos += self._recsize
546 self.fp.seek(pos)
547 self.fp.seek(pos0 + var._vsize)

549 def _write_var_padding(self, var, size):
550 encoded_fill_value = var._get_encoded_fill_value()
551 num_fills = size // len(encoded_fill_value)
552 self.fp.write(encoded_fill_value * num_fills)

554 def _write_att_values(self, values):
555 if hasattr(values, 'dtype'):
556 nc_type = REVERSE[values.dtype.char, values.dtype.itemsize]
557 else:
558 types = [(int, NC_INT), (float, NC_FLOAT), (str, NC_CHAR)]
559
560 # bytes index into scalars in py3k. Check for "string" types
561 if isinstance(values, (str, bytes)):
562 sample = values
563 else:
564 try:
565 sample = values[0] # subscriptable?
566 except TypeError:
567 sample = values # scalar
568
569 for class_, nc_type in types:
570 if isinstance(sample, class_):
571 break
572
573 typecode, size = TYPEMAP[nc_type]
574 dtype_ = '>%s' % typecode
575 # asarray() dies with bytes and '>c' in py3k. Change to 'S'
576 dtype_ = 'S' if dtype_ == '>c' else dtype_
577
578 values = asarray(values, dtype=dtype_)
579
580 self.fp.write(nc_type)
581
582 if values.dtype.char == 'S':
583 nelems = values.itemsize
584 else:
585 nelems = values.size
586 self._pack_int(nelems)
587
588 if not values.shape and (values.dtype.byteorder == '<' or
589 (values.dtype.byteorder == '=' and LITTLE_ENDIAN)):
590 values = values.byteswap()
591 self.fp.write(values.tobytes())
592 count = values.size * values.itemsize
593 self.fp.write(b'\x00' * (-count % 4)) # pad

595 def _read(self):
596 # Check magic bytes and version
597 magic = self.fp.read(3)
598 if not magic == b'CDF':
599 raise TypeError("Error: %s is not a valid NetCDF 3 file" %
600 self.filename)
601 self.__dict__['version_byte'] = frombuffer(self.fp.read(1), '>b')[0]
602
603 # Read file headers and set data.
604 self._read_numrecs()
605 self._read_dim_array()
606 self._read_gatt_array()
607 self._read_var_array()
608
609 def _read_numrecs(self):
610 self.__dict__['_recs'] = self._unpack_int()
611
612 def _read_dim_array(self):
613 header = self.fp.read(4)
614 if header not in [ZERO, NC_DIMENSION]:
615 raise ValueError("Unexpected header.")
616 count = self._unpack_int()
617
618 for dim in range(count):
619 name = self._unpack_string().decode('latin1')
620 length = self._unpack_int() or None # None for record dimension
621 self.dimensions[name] = length
622 self._dims.append(name) # preserve order
623
624 def _read_gatt_array(self):
625 for k, v in self._read_att_array().items():
626 self.__setattr__(k, v)
627
628 def _read_att_array(self):
629 header = self.fp.read(4)
630 if header not in [ZERO, NC_ATTRIBUTE]:
631 raise ValueError("Unexpected header.")
632 count = self._unpack_int()
633
634 attributes = {}
635 for attr in range(count):
636 name = self._unpack_string().decode('latin1')
637 attributes[name] = self._read_att_values()
638 return attributes
639
640 def _read_var_array(self):
641 header = self.fp.read(4)
642 if header not in [ZERO, NC_VARIABLE]:
643 raise ValueError("Unexpected header.")
644
645 begin = 0
646 dtypes = {'names': [], 'formats': []}
647 rec_vars = []
648 count = self._unpack_int()
649 for var in range(count):
650 (name, dimensions, shape, attributes,
651 typecode, size, dtype_, begin_, vsize) = self._read_var()
652 # https://www.unidata.ucar.edu/software/netcdf/guide_toc.html
653 # Note that vsize is the product of the dimension lengths
654 # (omitting the record dimension) and the number of bytes
655 # per value (determined from the type), increased to the
656 # next multiple of 4, for each variable. If a record
657 # variable, this is the amount of space per record. The
658 # netCDF "record size" is calculated as the sum of the
659 # vsize's of all the record variables.
660 #
661 # The vsize field is actually redundant, because its value
662 # may be computed from other information in the header. The
663 # 32-bit vsize field is not large enough to contain the size
664 # of variables that require more than 2^32 - 4 bytes, so
665 # 2^32 - 1 is used in the vsize field for such variables.
666 if shape and shape[0] is None: # record variable
667 rec_vars.append(name)
668 # The netCDF "record size" is calculated as the sum of
669 # the vsize's of all the record variables.
670 self.__dict__['_recsize'] += vsize
671 if begin == 0:
672 begin = begin_
673 dtypes['names'].append(name)
674 dtypes['formats'].append(str(shape[1:]) + dtype_)
675
676 # Handle padding with a virtual variable.
677 if typecode in 'bch':
678 actual_size = reduce(mul, (1,) + shape[1:]) * size
679 padding = -actual_size % 4
680 if padding:
681 dtypes['names'].append('_padding_%d' % var)
682 dtypes['formats'].append('(%d,)>b' % padding)
683
684 # Data will be set later.
685 data = None
686 else: # not a record variable
687 # Calculate size to avoid problems with vsize (above)
688 a_size = reduce(mul, shape, 1) * size
689 if self.use_mmap:
690 data = self._mm_buf[begin_:begin_+a_size].view(dtype=dtype_)
691 data.shape = shape
692 else:
693 pos = self.fp.tell()
694 self.fp.seek(begin_)

703 maskandscale=self.maskandscale)

709 dtypes['formats'] = dtypes['formats'][:1]

712 if self.use_mmap:
713 rec_array = self._mm_buf[begin:begin+self._recs*self._recsize].view(dtype=dtypes)

718 rec_array = frombuffer(self.fp.read(self._recs*self._recsize),
719 dtype=dtypes).copy()

721 self.fp.seek(pos)

724 self.variables[var].__dict__['data'] = rec_array[var]

726 def _read_var(self):
727 name = self._unpack_string().decode('latin1')
728 dimensions = []
729 shape = []
730 dims = self._unpack_int()
731
732 for i in range(dims):
733 dimid = self._unpack_int()
734 dimname = self._dims[dimid]
735 dimensions.append(dimname)
736 dim = self.dimensions[dimname]
737 shape.append(dim)
738 dimensions = tuple(dimensions)
739 shape = tuple(shape)

741 attributes = self._read_att_array()
742 nc_type = self.fp.read(4)

744 begin = [self._unpack_int, self._unpack_int64][self.version_byte-1]()
745
746 typecode, size = TYPEMAP[nc_type]
747 dtype_ = '>%s' % typecode

751 def _read_att_values(self):

802class netcdf_variable:
803 """
804 A data object for netcdf files.
805
806 `netcdf_variable` objects are constructed by calling the method
807 `netcdf_file.createVariable` on the `netcdf_file` object. `netcdf_variable`
808 objects behave much like array objects defined in numpy, except that their
809 data resides in a file. Data is read by indexing and written by assigning
810 to an indexed subset; the entire array can be accessed by the index ``[:]``
811 or (for scalars) by using the methods `getValue` and `assignValue`.
812 `netcdf_variable` objects also have attribute `shape` with the same meaning
813 as for arrays, but the shape cannot be modified. There is another read-only
814 attribute `dimensions`, whose value is the tuple of dimension names.
815
816 All other attributes correspond to variable attributes defined in
817 the NetCDF file. Variable attributes are created by assigning to an
818 attribute of the `netcdf_variable` object.
819
820 Parameters
821 ----------
822 data : array_like
823 The data array that holds the values for the variable.
824 Typically, this is initialized as empty, but with the proper shape.
825 typecode : dtype character code
826 Desired data-type for the data array.
827 size : int
828 Desired element size for the data array.
829 shape : sequence of ints
830 The shape of the array. This should match the lengths of the
831 variable's dimensions.
832 dimensions : sequence of strings
833 The names of the dimensions used by the variable. Must be in the
834 same order of the dimension lengths given by `shape`.
835 attributes : dict, optional
836 Attribute values (any type) keyed by string names. These attributes
837 become attributes for the netcdf_variable object.
838 maskandscale : bool, optional
839 Whether to automatically scale and/or mask data based on attributes.
840 Default is False.
841
842
843 Attributes
844 ----------
845 dimensions : list of str
846 List of names of dimensions used by the variable object.
847 isrec, shape
848 Properties
849
850 See also
851 --------
852 isrec, shape
853
854 """
855 def __init__(self, data, typecode, size, shape, dimensions,
856 attributes=None,
857 maskandscale=False):
858 self.data = data
859 self._typecode = typecode
860 self._size = size
861 self._shape = shape
862 self.dimensions = dimensions
863 self.maskandscale = maskandscale
864
865 self._attributes = attributes or {}
866 for k, v in self._attributes.items():
867 self.__dict__[k] = v

963 def __getitem__(self, index):
964 if not self.maskandscale:
965 return self.data[index]
966
967 data = self.data[index].copy()
968 missing_value = self._get_missing_value()
969 data = self._apply_missing_value(data, missing_value)
970 scale_factor = self._attributes.get('scale_factor')
971 add_offset = self._attributes.get('add_offset')
972 if add_offset is not None or scale_factor is not None:
973 data = data.astype(np.float64)
974 if scale_factor is not None:
975 data = data * scale_factor
976 if add_offset is not None:
977 data += add_offset
978
979 return data

981 def __setitem__(self, index, data):
982 if self.maskandscale:
983 missing_value = (
984 self._get_missing_value() or
985 getattr(data, 'fill_value', 999999))
986 self._attributes.setdefault('missing_value', missing_value)
987 self._attributes.setdefault('_FillValue', missing_value)
988 data = ((data - self._attributes.get('add_offset', 0.0)) /
989 self._attributes.get('scale_factor', 1.0))
990 data = np.ma.asarray(data).filled(missing_value)
991 if self._typecode not in 'fd' and data.dtype.kind == 'f':
992 data = np.round(data)
993
994 # Expand data for record vars?
995 if self.isrec:
996 if isinstance(index, tuple):
997 rec_index = index[0]
998 else:
999 rec_index = index
1000 if isinstance(rec_index, slice):
1001 recs = (rec_index.start or 0) + len(data)
1002 else:
1003 recs = rec_index + 1
1004 if recs > len(self.data):
1005 shape = (recs,) + self._shape[1:]
1006 # Resize in-place does not always work since
1007 # the array might not be single-segment
1008 try:
1009 self.data.resize(shape)
1010 except ValueError:
1011 self.__dict__['data'] = np.resize(self.data, shape).astype(self.data.dtype)
1012 self.data[index] = data

1063 Returns a numpy.ma array, with any value equal to missing_value masked
1064 out (unless missing_value is None, in which case the original array is

1069 newdata = data
1070 else:
1071 try:
1072 missing_value_isnan = np.isnan(missing_value)

1079 else:
1080 mymask = (data == missing_value)
1081
1082 newdata = np.ma.masked_where(mymask, data)
1083
