10def get_equivalent_estimator(estimator, lib="lightgbm", n_classes=None):
11 """Return an unfitted estimator from another lib with matching hyperparams.
12
13 This utility function takes care of renaming the sklearn parameters into
14 their LightGBM, XGBoost or CatBoost equivalent parameters.
15
16 # unmapped XGB parameters:
17 # - min_samples_leaf
18 # - min_data_in_bin
19 # - min_split_gain (there is min_split_loss though?)
20
21 # unmapped Catboost parameters:
22 # max_leaves
23 # min_*
24 """
25
26 if lib not in ("lightgbm", "xgboost", "catboost"):
27 raise ValueError(
28 "accepted libs are lightgbm, xgboost, and catboost. got {}".format(lib)
29 )
30
31 sklearn_params = estimator.get_params()
32
33 if sklearn_params["loss"] == "auto":
34 raise ValueError(
35 "auto loss is not accepted. We need to know if "
36 "the problem is binary or multiclass classification."
37 )
38 if sklearn_params["early_stopping"]:
39 raise NotImplementedError("Early stopping should be deactivated.")
40
41 lightgbm_loss_mapping = {
42 "squared_error": "regression_l2",
43 "absolute_error": "regression_l1",
44 "log_loss": "binary" if n_classes == 2 else "multiclass",
45 "gamma": "gamma",
46 "poisson": "poisson",
47 }
48
49 lightgbm_params = {
50 "objective": lightgbm_loss_mapping[sklearn_params["loss"]],
51 "learning_rate": sklearn_params["learning_rate"],
52 "n_estimators": sklearn_params["max_iter"],
53 "num_leaves": sklearn_params["max_leaf_nodes"],
54 "max_depth": sklearn_params["max_depth"],
55 "min_data_in_leaf": sklearn_params["min_samples_leaf"],
56 "reg_lambda": sklearn_params["l2_regularization"],
57 "max_bin": sklearn_params["max_bins"],
58 "min_data_in_bin": 1,
59 "min_sum_hessian_in_leaf": 1e-3,
60 "min_split_gain": 0,
61 "verbosity": 10 if sklearn_params["verbose"] else -10,
62 "boost_from_average": True,
63 "enable_bundle": False, # also makes feature order consistent
64 "subsample_for_bin": _BinMapper().subsample,
65 "poisson_max_delta_step": 1e-12,
66 "feature_fraction_bynode": sklearn_params["max_features"],
67 }
68
69 if sklearn_params["loss"] == "log_loss" and n_classes > 2:
70 # LightGBM multiplies hessians by 2 in multiclass loss.
71 lightgbm_params["min_sum_hessian_in_leaf"] *= 2
72 # LightGBM 3.0 introduced a different scaling of the hessian for the multiclass
73 # case.
74 # It is equivalent of scaling the learning rate.
75 # See https://github.com/microsoft/LightGBM/pull/3256.
76 if n_classes is not None:
77 lightgbm_params["learning_rate"] *= n_classes / (n_classes - 1)
78
79 # XGB
80 xgboost_loss_mapping = {
81 "squared_error": "reg:linear",
82 "absolute_error": "LEAST_ABSOLUTE_DEV_NOT_SUPPORTED",
83 "log_loss": "reg:logistic" if n_classes == 2 else "multi:softmax",
84 "gamma": "reg:gamma",
85 "poisson": "count:poisson",
86 }
87
88 xgboost_params = {
89 "tree_method": "hist",
90 "grow_policy": "lossguide", # so that we can set max_leaves
91 "objective": xgboost_loss_mapping[sklearn_params["loss"]],
92 "learning_rate": sklearn_params["learning_rate"],
93 "n_estimators": sklearn_params["max_iter"],
94 "max_leaves": sklearn_params["max_leaf_nodes"],
95 "max_depth": sklearn_params["max_depth"] or 0,
96 "lambda": sklearn_params["l2_regularization"],
97 "max_bin": sklearn_params["max_bins"],
98 "min_child_weight": 1e-3,
99 "verbosity": 2 if sklearn_params["verbose"] else 0,
100 "silent": sklearn_params["verbose"] == 0,
101 "n_jobs": -1,
102 "colsample_bynode": sklearn_params["max_features"],
103 }
104
105 # Catboost
106 catboost_loss_mapping = {
107 "squared_error": "RMSE",
108 # catboost does not support MAE when leaf_estimation_method is Newton
109 "absolute_error": "LEAST_ASBOLUTE_DEV_NOT_SUPPORTED",
110 "log_loss": "Logloss" if n_classes == 2 else "MultiClass",
111 "gamma": None,
112 "poisson": "Poisson",
113 }
114
115 catboost_params = {
116 "loss_function": catboost_loss_mapping[sklearn_params["loss"]],
117 "learning_rate": sklearn_params["learning_rate"],
118 "iterations": sklearn_params["max_iter"],
119 "depth": sklearn_params["max_depth"],
120 "reg_lambda": sklearn_params["l2_regularization"],
121 "max_bin": sklearn_params["max_bins"],
122 "feature_border_type": "Median",
123 "leaf_estimation_method": "Newton",
124 "verbose": bool(sklearn_params["verbose"]),
125 }
126
127 if lib == "lightgbm":
128 from lightgbm import LGBMClassifier, LGBMRegressor
129
130 if is_classifier(estimator):
131 return LGBMClassifier(**lightgbm_params)
132 else:
133 return LGBMRegressor(**lightgbm_params)
134
135 elif lib == "xgboost":
136 from xgboost import XGBClassifier, XGBRegressor
137
138 if is_classifier(estimator):
139 return XGBClassifier(**xgboost_params)
140 else:
141 return XGBRegressor(**xgboost_params)
142
143 else:
144 from catboost import CatBoostClassifier, CatBoostRegressor
145
146 if is_classifier(estimator):
147 return CatBoostClassifier(**catboost_params)
148 else:
149 return CatBoostRegressor(**catboost_params)
