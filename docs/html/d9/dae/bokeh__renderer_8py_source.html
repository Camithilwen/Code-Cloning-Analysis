29class BokehRenderer(Renderer):
30 """Utility renderer using Bokeh to render a grid of plots over the same (x, y) range.
31
32 Args:
33 nrows (int, optional): Number of rows of plots, default ``1``.
34 ncols (int, optional): Number of columns of plots, default ``1``.
35 figsize (tuple(float, float), optional): Figure size in inches (assuming 100 dpi), default
36 ``(9, 9)``.
37 show_frame (bool, optional): Whether to show frame and axes ticks, default ``True``.
38 want_svg (bool, optional): Whether output is required in SVG format or not, default
39 ``False``.
40
41 Warning:
42 :class:`~.BokehRenderer`, unlike :class:`~.MplRenderer`, needs to be told in advance if
43 output to SVG format will be required later, otherwise it will assume PNG output.
44 """
45 _figures: list[figure]
46 _layout: GridPlot
47 _palette: Palette
48 _want_svg: bool

50 def __init__(
51 self,
52 nrows: int = 1,
53 ncols: int = 1,
54 figsize: tuple[float, float] = (9, 9),
55 show_frame: bool = True,
56 want_svg: bool = False,
57 ) -> None:
58 self._want_svg = want_svg
59 self._palette = Category10[10]
60
61 total_size = 100*np.asarray(figsize, dtype=int) # Assuming 100 dpi.
62
63 nfigures = nrows*ncols
64 self._figures = []
65 backend: OutputBackendType = "svg" if self._want_svg else "canvas"
66 for _ in range(nfigures):
67 fig = figure(output_backend=backend)
68 fig.xgrid.visible = False # type: ignore[attr-defined]
69 fig.ygrid.visible = False # type: ignore[attr-defined]
70 self._figures.append(fig)
71 if not show_frame:
72 fig.outline_line_color = None
73 fig.axis.visible = False # type: ignore[attr-defined]
74
75 self._layout = gridplot(
76 self._figures, ncols=ncols, toolbar_location=None, # type: ignore[arg-type]
77 width=total_size[0] // ncols, height=total_size[1] // nrows)

90 def filled(
91 self,
92 filled: FillReturn,
93 fill_type: FillType | str,
94 ax: figure | int = 0,
95 color: str = "C0",
96 alpha: float = 0.7,
97 ) -> None:
98 """Plot filled contours on a single plot.
99
100 Args:
101 filled (sequence of arrays): Filled contour data as returned by
102 :meth:`~.ContourGenerator.filled`.
103 fill_type (FillType or str): Type of :meth:`~.ContourGenerator.filled` data as returned
104 by :attr:`~.ContourGenerator.fill_type`, or a string equivalent.
105 ax (int or Bokeh Figure, optional): Which plot to use, default ``0``.
106 color (str, optional): Color to plot with. May be a string color or the letter ``"C"``
107 followed by an integer in the range ``"C0"`` to ``"C9"`` to use a color from the
108 ``Category10`` palette. Default ``"C0"``.
109 alpha (float, optional): Opacity to plot with, default ``0.7``.
110 """
111 fill_type = as_fill_type(fill_type)
112 fig = self._get_figure(ax)
113 color = self._convert_color(color)
114 xs, ys = filled_to_bokeh(filled, fill_type)
115 if len(xs) > 0:
116 fig.multi_polygons(xs=[xs], ys=[ys], color=color, fill_alpha=alpha, line_width=0) # type: ignore[arg-type]

118 def grid(
119 self,
120 x: ArrayLike,
121 y: ArrayLike,
122 ax: figure | int = 0,
123 color: str = "black",
124 alpha: float = 0.1,
125 point_color: str | None = None,
126 quad_as_tri_alpha: float = 0,
127 ) -> None:
128 """Plot quad grid lines on a single plot.
129
130 Args:
131 x (array-like of shape (ny, nx) or (nx,)): The x-coordinates of the grid points.
132 y (array-like of shape (ny, nx) or (ny,)): The y-coordinates of the grid points.
133 ax (int or Bokeh Figure, optional): Which plot to use, default ``0``.
134 color (str, optional): Color to plot grid lines, default ``"black"``.
135 alpha (float, optional): Opacity to plot lines with, default ``0.1``.
136 point_color (str, optional): Color to plot grid points or ``None`` if grid points
137 should not be plotted, default ``None``.
138 quad_as_tri_alpha (float, optional): Opacity to plot ``quad_as_tri`` grid, default
139 ``0``.
140
141 Colors may be a string color or the letter ``"C"`` followed by an integer in the range
142 ``"C0"`` to ``"C9"`` to use a color from the ``Category10`` palette.
143
144 Warning:
145 ``quad_as_tri_alpha > 0`` plots all quads as though they are unmasked.
146 """
147 fig = self._get_figure(ax)
148 x, y = self._grid_as_2d(x, y)
149 xs = list(x) + list(x.T)
150 ys = list(y) + list(y.T)
151 kwargs = {"line_color": color, "alpha": alpha}
152 fig.multi_line(xs, ys, **kwargs)
153 if quad_as_tri_alpha > 0:
154 # Assumes no quad mask.
155 xmid = (0.25*(x[:-1, :-1] + x[1:, :-1] + x[:-1, 1:] + x[1:, 1:])).ravel()
156 ymid = (0.25*(y[:-1, :-1] + y[1:, :-1] + y[:-1, 1:] + y[1:, 1:])).ravel()
157 fig.multi_line(
158 list(np.stack((x[:-1, :-1].ravel(), xmid, x[1:, 1:].ravel()), axis=1)),
159 list(np.stack((y[:-1, :-1].ravel(), ymid, y[1:, 1:].ravel()), axis=1)),
160 **kwargs)
161 fig.multi_line(
162 list(np.stack((x[:-1, 1:].ravel(), xmid, x[1:, :-1].ravel()), axis=1)),
163 list(np.stack((y[:-1, 1:].ravel(), ymid, y[1:, :-1].ravel()), axis=1)),
164 **kwargs)
165 if point_color is not None:
166 fig.scatter(
167 x=x.ravel(), y=y.ravel(), fill_color=color, line_color=None, alpha=alpha,
168 marker="circle", size=8)

170 def lines(
171 self,
172 lines: LineReturn,
173 line_type: LineType | str,
174 ax: figure | int = 0,
175 color: str = "C0",
176 alpha: float = 1.0,
177 linewidth: float = 1,

179 """Plot contour lines on a single plot.
180
181 Args:
182 lines (sequence of arrays): Contour line data as returned by
183 :meth:`~.ContourGenerator.lines`.
184 line_type (LineType or str): Type of :meth:`~.ContourGenerator.lines` data as returned
185 by :attr:`~.ContourGenerator.line_type`, or a string equivalent.
186 ax (int or Bokeh Figure, optional): Which plot to use, default ``0``.
187 color (str, optional): Color to plot lines. May be a string color or the letter ``"C"``
188 followed by an integer in the range ``"C0"`` to ``"C9"`` to use a color from the

190 alpha (float, optional): Opacity to plot lines with, default ``1.0``.
191 linewidth (float, optional): Width of lines, default ``1``.
192
193 Note:
194 Assumes all lines are open line strips not closed line loops.

196 line_type = as_line_type(line_type)
197 fig = self._get_figure(ax)
198 color = self._convert_color(color)
199 xs, ys = lines_to_bokeh(lines, line_type)
200 if xs is not None:
201 assert ys is not None
202 fig.line(xs, ys, line_color=color, line_alpha=alpha, line_width=linewidth)

204 def mask(
205 self,
206 x: ArrayLike,
207 y: ArrayLike,
208 z: ArrayLike | np.ma.MaskedArray[Any, Any],
209 ax: figure | int = 0,
210 color: str = "black",

212 """Plot masked out grid points as circles on a single plot.
213
214 Args:
215 x (array-like of shape (ny, nx) or (nx,)): The x-coordinates of the grid points.

230 self,
231 filename: str,

234 webdriver: WebDriver | None = None,

236 """Save plots to SVG or PNG file.
237

296 def z_values(
297 self,
298 x: ArrayLike,
299 y: ArrayLike,
300 z: ArrayLike,
301 ax: figure | int = 0,
302 color: str = "green",
303 fmt: str = ".1f",
304 quad_as_tri: bool = False,
305 ) -> None:
306 """Show ``z`` values on a single plot.
307
308 Args:
309 x (array-like of shape (ny, nx) or (nx,)): The x-coordinates of the grid points.
310 y (array-like of shape (ny, nx) or (ny,)): The y-coordinates of the grid points.
311 z (array-like of shape (ny, nx): z-values.
312 ax (int or Bokeh Figure, optional): Which plot to use, default ``0``.
313 color (str, optional): Color of added text. May be a string color or the letter ``"C"``
314 followed by an integer in the range ``"C0"`` to ``"C9"`` to use a color from the
315 ``Category10`` palette. Default ``"green"``.
316 fmt (str, optional): Format to display z-values, default ``".1f"``.
317 quad_as_tri (bool, optional): Whether to show z-values at the ``quad_as_tri`` centres
318 of quads.
319
320 Warning:
321 ``quad_as_tri=True`` shows z-values for all quads, even if masked.
322 """
323 fig = self._get_figure(ax)
324 color = self._convert_color(color)
325 x, y = self._grid_as_2d(x, y)
326 z = np.asarray(z)
327 ny, nx = z.shape
328 kwargs = {"text_color": color, "text_align": "center", "text_baseline": "middle"}
329 for j in range(ny):
330 for i in range(nx):
331 label = Label(x=x[j, i], y=y[j, i], text=f"{z[j, i]:{fmt}}", **kwargs) # type: ignore[arg-type]
332 fig.add_layout(label)
333 if quad_as_tri:
334 for j in range(ny-1):
335 for i in range(nx-1):
336 xx = np.mean(x[j:j+2, i:i+2])
337 yy = np.mean(y[j:j+2, i:i+2])
338 zz = np.mean(z[j:j+2, i:i+2])
339 fig.add_layout(Label(x=xx, y=yy, text=f"{zz:{fmt}}", **kwargs)) # type: ignore[arg-type]
