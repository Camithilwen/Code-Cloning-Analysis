19def _relative_degree(z, p):
20 """
21 Return relative degree of transfer function from zeros and poles.
22
23 This is simply len(p) - len(z), which must be nonnegative.
24 A ValueError is raised if len(p) < len(z).
25 """
26 degree = len(p) - len(z)
27 if degree < 0:
28 raise ValueError("Improper transfer function. "
29 "Must have at least as many poles as zeros.")
30 return degree

33def _zpkbilinear(z, p, k, fs):
34 """Bilinear transformation to convert a filter from analog to digital."""
35
36 degree = _relative_degree(z, p)
37
38 fs2 = 2*fs
39
40 # Bilinear transform the poles and zeros
41 z_z = [(fs2 + z1) / (fs2 - z1) for z1 in z]
42 p_z = [(fs2 + p1) / (fs2 - p1) for p1 in p]
43
44 # Any zeros that were at infinity get moved to the Nyquist frequency
45 z_z.extend([-1] * degree)
46
47 # Compensate for gain change
48 numer = _prod(fs2 - z1 for z1 in z)
49 denom = _prod(fs2 - p1 for p1 in p)
50 k_z = k * numer / denom
51
52 return z_z, p_z, k_z.real

55def _zpklp2lp(z, p, k, wo=1):
56 """Transform a lowpass filter to a different cutoff frequency."""
57
58 degree = _relative_degree(z, p)
59
60 # Scale all points radially from origin to shift cutoff frequency
61 z_lp = [wo * z1 for z1 in z]
62 p_lp = [wo * p1 for p1 in p]
63
64 # Each shifted pole decreases gain by wo, each shifted zero increases it.
65 # Cancel out the net change to keep overall gain the same
66 k_lp = k * wo**degree
67
68 return z_lp, p_lp, k_lp

71def _butter_analog_poles(n):
72 """
73 Poles of an analog Butterworth lowpass filter.
74
75 This is the same calculation as scipy.signal.buttap(n) or
76 scipy.signal.butter(n, 1, analog=True, output='zpk'), but mpmath is used,
77 and only the poles are returned.
78 """
79 poles = [-mpmath.exp(1j*mpmath.pi*k/(2*n)) for k in range(-n+1, n, 2)]
