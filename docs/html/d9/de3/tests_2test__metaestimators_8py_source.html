87def test_metaestimator_delegation():
88 # Ensures specified metaestimators have methods iff subestimator does
89 def hides(method):
90 @property
91 def wrapper(obj):
92 if obj.hidden_method == method.__name__:
93 raise AttributeError("%r is hidden" % obj.hidden_method)
94 return functools.partial(method, obj)
95
96 return wrapper

98 class SubEstimator(BaseEstimator):
99 def __init__(self, param=1, hidden_method=None):
100 self.param = param
101 self.hidden_method = hidden_method
102
103 def fit(self, X, y=None, *args, **kwargs):
104 self.coef_ = np.arange(X.shape[1])
105 self.classes_ = []
106 return True

108 def _check_fit(self):
109 check_is_fitted(self)
110
111 @hides
112 def inverse_transform(self, X, *args, **kwargs):
113 self._check_fit()
114 return X
115
116 @hides
117 def transform(self, X, *args, **kwargs):
118 self._check_fit()
119 return X
120
121 @hides
122 def predict(self, X, *args, **kwargs):
123 self._check_fit()
124 return np.ones(X.shape[0])
125
126 @hides
127 def predict_proba(self, X, *args, **kwargs):
128 self._check_fit()
129 return np.ones(X.shape[0])
130
131 @hides
132 def predict_log_proba(self, X, *args, **kwargs):
133 self._check_fit()
134 return np.ones(X.shape[0])
135
136 @hides
137 def decision_function(self, X, *args, **kwargs):
138 self._check_fit()
139 return np.ones(X.shape[0])
140
141 @hides
142 def score(self, X, y, *args, **kwargs):
143 self._check_fit()
144 return 1.0

146 methods = [
147 k
148 for k in SubEstimator.__dict__.keys()
149 if not k.startswith("_") and not k.startswith("fit")
150 ]
151 methods.sort()
152
153 for delegator_data in DELEGATING_METAESTIMATORS:
154 delegate = SubEstimator()
155 delegator = delegator_data.construct(delegate)
156 for method in methods:
157 if method in delegator_data.skip_methods:
158 continue
159 assert hasattr(delegate, method)
160 assert hasattr(delegator, method), (
161 "%s does not have method %r when its delegate does"
162 % (
163 delegator_data.name,
164 method,
165 )
166 )
167 # delegation before fit raises a NotFittedError
168 if method == "score":
169 with pytest.raises(NotFittedError):
170 getattr(delegator, method)(
171 delegator_data.fit_args[0], delegator_data.fit_args[1]
172 )
173 else:
174 with pytest.raises(NotFittedError):
175 getattr(delegator, method)(delegator_data.fit_args[0])
176
177 delegator.fit(*delegator_data.fit_args)
178 for method in methods:
179 if method in delegator_data.skip_methods:
180 continue
181 # smoke test delegation
182 if method == "score":
183 getattr(delegator, method)(
184 delegator_data.fit_args[0], delegator_data.fit_args[1]
185 )
186 else:
187 getattr(delegator, method)(delegator_data.fit_args[0])
188

191 continue
192 delegate = SubEstimator(hidden_method=method)
193 delegator = delegator_data.construct(delegate)

204def _get_instance_with_pipeline(meta_estimator, init_params):
205 """Given a single meta-estimator instance, generate an instance with a pipeline"""
206 if {"estimator", "base_estimator", "regressor"} & init_params:
207 if is_regressor(meta_estimator):
208 estimator = make_pipeline(TfidfVectorizer(), Ridge())
209 param_grid = {"ridge__alpha": [0.1, 1.0]}
210 else:
211 estimator = make_pipeline(TfidfVectorizer(), LogisticRegression())
212 param_grid = {"logisticregression__C": [0.1, 1.0]}
213
214 if init_params.intersection(
215 {"param_grid", "param_distributions"}
216 ): # SearchCV estimators
217 extra_params = {"n_iter": 2} if "n_iter" in init_params else {}
218 return type(meta_estimator)(estimator, param_grid, **extra_params)
219 else:
220 return type(meta_estimator)(estimator)
221
222 if "transformer_list" in init_params:
223 # FeatureUnion
224 transformer_list = [
225 ("trans1", make_pipeline(TfidfVectorizer(), MaxAbsScaler())),
226 (
227 "trans2",
228 make_pipeline(TfidfVectorizer(), StandardScaler(with_mean=False)),
229 ),
230 ]
231 return type(meta_estimator)(transformer_list)
232
233 if "estimators" in init_params:
234 # stacking, voting
235 if is_regressor(meta_estimator):
236 estimator = [
237 ("est1", make_pipeline(TfidfVectorizer(), Ridge(alpha=0.1))),
238 ("est2", make_pipeline(TfidfVectorizer(), Ridge(alpha=1))),
239 ]
240 else:
241 estimator = [
242 (
243 "est1",
244 make_pipeline(TfidfVectorizer(), LogisticRegression(C=0.1)),
245 ),
246 ("est2", make_pipeline(TfidfVectorizer(), LogisticRegression(C=1))),
247 ]
248 return type(meta_estimator)(estimator)

251def _generate_meta_estimator_instances_with_pipeline():
252 """Generate instances of meta-estimators fed with a pipeline
253
254 Are considered meta-estimators all estimators accepting one of "estimator",
255 "base_estimator" or "estimators".
256 """
257 print("estimators: ", len(all_estimators()))
258 for _, Estimator in sorted(all_estimators()):
259 sig = set(signature(Estimator).parameters)
260
261 print("\n", Estimator.__name__, sig)
262 if not sig.intersection(
263 {
264 "estimator",
265 "base_estimator",
266 "regressor",
267 "transformer_list",
268 "estimators",
269 }
270 ):
271 continue
272
273 with suppress(SkipTest):
274 for meta_estimator in _construct_instances(Estimator):
275 print(meta_estimator)
276 yield _get_instance_with_pipeline(meta_estimator, sig)
