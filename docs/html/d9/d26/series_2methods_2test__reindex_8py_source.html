25def test_reindex(datetime_series, string_series):
26 identity = string_series.reindex(string_series.index)
27
28 assert tm.shares_memory(string_series.index, identity.index)
29
30 assert identity.index.is_(string_series.index)
31 assert identity.index.identical(string_series.index)

33 subIndex = string_series.index[10:20]
34 subSeries = string_series.reindex(subIndex)
35
36 for idx, val in subSeries.items():
37 assert val == string_series[idx]
