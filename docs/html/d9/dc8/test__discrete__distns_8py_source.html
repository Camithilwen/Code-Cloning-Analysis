320 @np.vectorize
321 def Hns(n, s):
322 """Naive implementation of harmonic sum"""
323 return (1/np.arange(1, n+1)**s).sum()

325 @np.vectorize
326 def pzip(k, a, n):
327 """Naive implementation of zipfian pmf"""
328 if k < 1 or k > n:
329 return 0.
330 else:
331 return 1 / k**a / Hns(n, a)

333 k = np.arange(n+1)

336 mean = np.average(k, weights=pmf)

338 std = var**0.5

340 kurtosis = np.average(((k-mean)/std)**4, weights=pmf) - 3

342 assert_allclose(zipfian.cdf(k, a, n), cdf)
343 assert_allclose(zipfian.stats(a, n, moments="mvsk"),

408 # test against a very simple implementation
409
410 np.random.seed(2)
411 shape = (2, 4, 3)
412 max_m = 100
413 m1 = np.random.randint(1, max_m, size=shape)
414 m2 = np.random.randint(1, max_m, size=shape)
415 N = m1 + m2

417 xl = np.maximum(0, n-m2)

422 def support(N, m1, n, w):
423 m2 = N - m1
424 xl = np.maximum(0, n-m2)
425 xu = np.minimum(n, m1)
426 return xl, xu

428 @np.vectorize
429 def mean(N, m1, n, w):
430 m2 = N - m1
431 xl, xu = support(N, m1, n, w)
432
433 def fun(u):
434 return u/m1 + (1 - (n-u)/m2)**w - 1
435
436 return root_scalar(fun, bracket=(xl, xu)).root
437
438 with suppress_warnings() as sup:
439 sup.filter(RuntimeWarning,
440 message="invalid value encountered in mean")
441 assert_allclose(nchypergeom_wallenius.mean(N, m1, n, w),
442 mean(N, m1, n, w), rtol=2e-2)
443
444 @np.vectorize
445 def variance(N, m1, n, w):
446 m2 = N - m1
447 u = mean(N, m1, n, w)
448 a = u * (m1 - u)
449 b = (n-u)*(u + m2 - n)
450 return N*a*b / ((N-1) * (m1*b + m2*a))
451
452 with suppress_warnings() as sup:
453 sup.filter(RuntimeWarning,
454 message="invalid value encountered in mean")
455 assert_allclose(
456 nchypergeom_wallenius.stats(N, m1, n, w, moments='v'),
457 variance(N, m1, n, w),
458 rtol=5e-2
459 )

461 @np.vectorize
462 def pmf(x, N, m1, n, w):
463 m2 = N - m1
464 xl, xu = support(N, m1, n, w)
465
466 def integrand(t):
467 D = w*(m1 - x) + (m2 - (n-x))
468 res = (1-t**(w/D))**x * (1-t**(1/D))**(n-x)
469 return res
470
471 def f(x):
472 t1 = special_binom(m1, x)
473 t2 = special_binom(m2, n - x)
474 the_integral = quad(integrand, 0, 1,
475 epsrel=1e-16, epsabs=1e-16)
476 return t1 * t2 * the_integral[0]
477
478 return f(x)
479
480 pmf0 = pmf(x, N, m1, n, w)
481 pmf1 = nchypergeom_wallenius.pmf(x, N, m1, n, w)
482
483 atol, rtol = 1e-6, 1e-6
484 i = np.abs(pmf1 - pmf0) < atol + rtol*np.abs(pmf0)
485 assert i.sum() > np.prod(shape) / 2 # works at least half the time
486
487 # for those that fail, discredit the naive implementation
488 for N, m1, n, w in zip(N[~i], m1[~i], n[~i], w[~i]):
489 # get the support
490 m2 = N - m1

496 assert pmf(x, N, m1, n, w).sum() < .5
497 assert_allclose(nchypergeom_wallenius.pmf(x, N, m1, n, w).sum(), 1)
