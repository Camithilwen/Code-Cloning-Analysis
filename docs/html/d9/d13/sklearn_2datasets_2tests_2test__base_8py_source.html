339def test_bunch_pickle_generated_with_0_16_and_read_with_0_17():
340 bunch = Bunch(key="original")
341 # This reproduces a problem when Bunch pickles have been created
342 # with scikit-learn 0.16 and are read with 0.17. Basically there
343 # is a surprising behaviour because reading bunch.key uses
344 # bunch.__dict__ (which is non empty for 0.16 Bunch objects)
345 # whereas assigning into bunch.key uses bunch.__setattr__. See
346 # https://github.com/scikit-learn/scikit-learn/issues/6196 for
347 # more details
348 bunch.__dict__["key"] = "set from __dict__"
349 bunch_from_pkl = loads(dumps(bunch))
350 # After loading from pickle the __dict__ should have been ignored
351 assert bunch_from_pkl.key == "original"
352 assert bunch_from_pkl["key"] == "original"
353 # Making sure that changing the attr does change the value
354 # associated with __getitem__ as well
355 bunch_from_pkl.key = "changed"
356 assert bunch_from_pkl.key == "changed"
357 assert bunch_from_pkl["key"] == "changed"
