23def test_pandas_adapter():
24 """Check pandas adapter has expected behavior."""
25 pd = pytest.importorskip("pandas")

28 index = np.asarray([1, 2])

30 X_ser_orig = pd.Series([2, 3], index=index)
31
32 adapter = ADAPTERS_MANAGER.adapters["pandas"]

34 assert isinstance(X_container, pd.DataFrame)

36 assert_array_equal(X_container.index, index)
37
38 # use original index when the original is a series
39 X_container = adapter.create_container(X_np, X_ser_orig, columns=lambda: columns)
40 assert isinstance(X_container, pd.DataFrame)
41 assert_array_equal(X_container.columns, columns)
42 assert_array_equal(X_container.index, index)
43
44 # Input dataframe's index does not change
45 new_columns = np.asarray(["f0", "f1"], dtype=object)
46 X_df = pd.DataFrame([[1, 2], [1, 3]], index=[10, 12])
47 new_df = adapter.create_container(X_df, X_df_orig, columns=new_columns)
48 assert_array_equal(new_df.columns, new_columns)
49 assert_array_equal(new_df.index, X_df.index)
50
51 assert adapter.is_supported_container(X_df)
52 assert not adapter.is_supported_container(X_np)
53
54 # adapter.update_columns updates the columns
55 new_columns = np.array(["a", "c"], dtype=object)
56 new_df = adapter.rename_columns(X_df, new_columns)
57 assert_array_equal(new_df.columns, new_columns)
