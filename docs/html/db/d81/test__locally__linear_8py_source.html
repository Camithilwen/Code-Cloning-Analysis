44def test_lle_simple_grid(global_dtype):
45 # note: ARPACK is numerically unstable, so this test will fail for
46 # some random seeds. We choose 42 because the tests pass.
47 # for arm64 platforms 2 makes the test fail.
48 # TODO: rewrite this test to make less sensitive to the random seed,
49 # irrespective of the platform.
50 rng = np.random.RandomState(42)
51
52 # grid of equidistant points in 2D, n_components = n_dim
53 X = np.array(list(product(range(5), repeat=2)))
54 X = X + 1e-10 * rng.uniform(size=X.shape)
55 X = X.astype(global_dtype, copy=False)
56
57 n_components = 2
58 clf = manifold.LocallyLinearEmbedding(
59 n_neighbors=5, n_components=n_components, random_state=rng
60 )
61 tol = 0.1
62
63 N = barycenter_kneighbors_graph(X, clf.n_neighbors).toarray()
64 reconstruction_error = linalg.norm(np.dot(N, X) - X, "fro")
65 assert reconstruction_error < tol
66
67 for solver in eigen_solvers:
68 clf.set_params(eigen_solver=solver)
69 clf.fit(X)
70 assert clf.embedding_.shape[1] == n_components
71 reconstruction_error = (
72 linalg.norm(np.dot(N, clf.embedding_) - clf.embedding_, "fro") ** 2
73 )
74
75 assert reconstruction_error < tol
76 assert_allclose(clf.reconstruction_error_, reconstruction_error, atol=1e-1)
77
78 # re-embed a noisy version of X using the transform method
79 noise = rng.randn(*X.shape).astype(global_dtype, copy=False) / 100
80 X_reembedded = clf.transform(X + noise)
81 assert linalg.norm(X_reembedded - clf.embedding_) < tol
