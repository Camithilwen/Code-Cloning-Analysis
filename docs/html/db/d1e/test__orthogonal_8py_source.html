221 G4 = orth.sh_jacobi(4,p,q)

478def test_roots_hermite_asy():
479 # Recursion for Hermite functions
480 def hermite_recursion(n, nodes):
481 H = np.zeros((n, nodes.size))
482 H[0,:] = np.pi**(-0.25) * np.exp(-0.5*nodes**2)
483 if n > 1:
484 H[1,:] = sqrt(2.0) * nodes * H[0,:]
485 for k in range(2, n):
486 H[k,:] = sqrt(2.0/k) * nodes * H[k-1,:] - sqrt((k-1.0)/k) * H[k-2,:]
487 return H
488
489 # This tests only the nodes
490 def test(N, rtol=1e-15, atol=1e-14):
491 x, w = orth._roots_hermite_asy(N)

493 assert_allclose(H[-1,:], np.zeros(N), rtol, atol)
494 assert_allclose(sum(w), sqrt(np.pi), rtol, atol)
