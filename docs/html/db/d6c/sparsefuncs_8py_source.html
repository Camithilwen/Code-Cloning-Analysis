39def inplace_csr_column_scale(X, scale):
40 """Inplace column scaling of a CSR matrix.
41
42 Scale each feature of the data matrix by multiplying with specific scale
43 provided by the caller assuming a (n_samples, n_features) shape.
44
45 Parameters
46 ----------
47 X : sparse matrix of shape (n_samples, n_features)
48 Matrix to normalize using the variance of the features.
49 It should be of CSR format.
50
51 scale : ndarray of shape (n_features,), dtype={np.float32, np.float64}
52 Array of precomputed feature-wise values to use for scaling.
53
54 Examples
55 --------
56 >>> from sklearn.utils import sparsefuncs
57 >>> from scipy import sparse
58 >>> import numpy as np
59 >>> indptr = np.array([0, 3, 4, 4, 4])
60 >>> indices = np.array([0, 1, 2, 2])
61 >>> data = np.array([8, 1, 2, 5])
62 >>> scale = np.array([2, 3, 2])
63 >>> csr = sparse.csr_matrix((data, indices, indptr))
64 >>> csr.todense()
65 matrix([[8, 1, 2],
66 [0, 0, 5],
67 [0, 0, 0],
68 [0, 0, 0]])
69 >>> sparsefuncs.inplace_csr_column_scale(csr, scale)
70 >>> csr.todense()
71 matrix([[16, 3, 4],
72 [ 0, 0, 10],
73 [ 0, 0, 0],
74 [ 0, 0, 0]])
75 """
76 assert scale.shape[0] == X.shape[1]
77 X.data *= scale.take(X.indices, mode="clip")

98def mean_variance_axis(X, axis, weights=None, return_sum_weights=False):
99 """Compute mean and variance along an axis on a CSR or CSC matrix.
100
101 Parameters
102 ----------
103 X : sparse matrix of shape (n_samples, n_features)
104 Input data. It can be of CSR or CSC format.
105
106 axis : {0, 1}
107 Axis along which the axis should be computed.
108
109 weights : ndarray of shape (n_samples,) or (n_features,), default=None
110 If axis is set to 0 shape is (n_samples,) or
111 if axis is set to 1 shape is (n_features,).
112 If it is set to None, then samples are equally weighted.
113
114 .. versionadded:: 0.24
115
116 return_sum_weights : bool, default=False
117 If True, returns the sum of weights seen for each feature
118 if `axis=0` or each sample if `axis=1`.
119
120 .. versionadded:: 0.24
121
122 Returns
123 -------
124
125 means : ndarray of shape (n_features,), dtype=floating
126 Feature-wise means.
127
128 variances : ndarray of shape (n_features,), dtype=floating
129 Feature-wise variances.
130
131 sum_weights : ndarray of shape (n_features,), dtype=floating
132 Returned if `return_sum_weights` is `True`.
133
134 Examples
135 --------
136 >>> from sklearn.utils import sparsefuncs
137 >>> from scipy import sparse
138 >>> import numpy as np
139 >>> indptr = np.array([0, 3, 4, 4, 4])
140 >>> indices = np.array([0, 1, 2, 2])
141 >>> data = np.array([8, 1, 2, 5])
142 >>> scale = np.array([2, 3, 2])
143 >>> csr = sparse.csr_matrix((data, indices, indptr))
144 >>> csr.todense()
145 matrix([[8, 1, 2],
146 [0, 0, 5],
147 [0, 0, 0],
148 [0, 0, 0]])
149 >>> sparsefuncs.mean_variance_axis(csr, axis=0)
150 (array([2. , 0.25, 1.75]), array([12. , 0.1875, 4.1875]))
151 """
152 _raise_error_wrong_axis(axis)
153
154 if sp.issparse(X) and X.format == "csr":
155 if axis == 0:
156 return _csr_mean_var_axis0(
157 X, weights=weights, return_sum_weights=return_sum_weights
158 )
159 else:
160 return _csc_mean_var_axis0(
161 X.T, weights=weights, return_sum_weights=return_sum_weights
162 )
163 elif sp.issparse(X) and X.format == "csc":
164 if axis == 0:
165 return _csc_mean_var_axis0(
166 X, weights=weights, return_sum_weights=return_sum_weights
167 )
168 else:
169 return _csr_mean_var_axis0(
170 X.T, weights=weights, return_sum_weights=return_sum_weights
171 )
172 else:
173 _raise_typeerror(X)

176def incr_mean_variance_axis(X, *, axis, last_mean, last_var, last_n, weights=None):
177 """Compute incremental mean and variance along an axis on a CSR or CSC matrix.
178
179 last_mean, last_var are the statistics computed at the last step by this
180 function. Both must be initialized to 0-arrays of the proper size, i.e.
181 the number of features in X. last_n is the number of samples encountered
182 until now.
183
184 Parameters
185 ----------
186 X : CSR or CSC sparse matrix of shape (n_samples, n_features)
187 Input data.
188
189 axis : {0, 1}
190 Axis along which the axis should be computed.
191
192 last_mean : ndarray of shape (n_features,) or (n_samples,), dtype=floating
193 Array of means to update with the new data X.
194 Should be of shape (n_features,) if axis=0 or (n_samples,) if axis=1.
195
196 last_var : ndarray of shape (n_features,) or (n_samples,), dtype=floating
197 Array of variances to update with the new data X.
198 Should be of shape (n_features,) if axis=0 or (n_samples,) if axis=1.
199
200 last_n : float or ndarray of shape (n_features,) or (n_samples,), \
201 dtype=floating
202 Sum of the weights seen so far, excluding the current weights
203 If not float, it should be of shape (n_features,) if
204 axis=0 or (n_samples,) if axis=1. If float it corresponds to
205 having same weights for all samples (or features).
206
207 weights : ndarray of shape (n_samples,) or (n_features,), default=None
208 If axis is set to 0 shape is (n_samples,) or
209 if axis is set to 1 shape is (n_features,).
210 If it is set to None, then samples are equally weighted.
211
212 .. versionadded:: 0.24
213
214 Returns
215 -------
216 means : ndarray of shape (n_features,) or (n_samples,), dtype=floating
217 Updated feature-wise means if axis = 0 or
218 sample-wise means if axis = 1.
219
220 variances : ndarray of shape (n_features,) or (n_samples,), dtype=floating
221 Updated feature-wise variances if axis = 0 or
222 sample-wise variances if axis = 1.
223
224 n : ndarray of shape (n_features,) or (n_samples,), dtype=integral
225 Updated number of seen samples per feature if axis=0
226 or number of seen features per sample if axis=1.
227
228 If weights is not None, n is a sum of the weights of the seen
229 samples or features instead of the actual number of seen
230 samples or features.
231
232 Notes
233 -----
234 NaNs are ignored in the algorithm.
235
236 Examples
237 --------
238 >>> from sklearn.utils import sparsefuncs
239 >>> from scipy import sparse
240 >>> import numpy as np
241 >>> indptr = np.array([0, 3, 4, 4, 4])
242 >>> indices = np.array([0, 1, 2, 2])
243 >>> data = np.array([8, 1, 2, 5])
244 >>> scale = np.array([2, 3, 2])
245 >>> csr = sparse.csr_matrix((data, indices, indptr))
246 >>> csr.todense()
247 matrix([[8, 1, 2],
248 [0, 0, 5],
249 [0, 0, 0],
250 [0, 0, 0]])
251 >>> sparsefuncs.incr_mean_variance_axis(
252 ... csr, axis=0, last_mean=np.zeros(3), last_var=np.zeros(3), last_n=2
253 ... )
254 (array([1.33, 0.167, 1.17]), array([8.88, 0.139, 3.47]),
255 array([6., 6., 6.]))
256 """
257 _raise_error_wrong_axis(axis)
258
259 if not (sp.issparse(X) and X.format in ("csc", "csr")):
260 _raise_typeerror(X)
261
262 if np.size(last_n) == 1:
263 last_n = np.full(last_mean.shape, last_n, dtype=last_mean.dtype)
264
265 if not (np.size(last_mean) == np.size(last_var) == np.size(last_n)):
266 raise ValueError("last_mean, last_var, last_n do not have the same shapes.")
267
268 if axis == 1:
269 if np.size(last_mean) != X.shape[0]:
270 raise ValueError(
271 "If axis=1, then last_mean, last_n, last_var should be of "
272 f"size n_samples {X.shape[0]} (Got {np.size(last_mean)})."
273 )
274 else: # axis == 0
275 if np.size(last_mean) != X.shape[1]:
276 raise ValueError(
277 "If axis=0, then last_mean, last_n, last_var should be of "
278 f"size n_features {X.shape[1]} (Got {np.size(last_mean)})."
279 )
280
281 X = X.T if axis == 1 else X
282
283 if weights is not None:
284 weights = _check_sample_weight(weights, X, dtype=X.dtype)
285
286 return _incr_mean_var_axis0(
287 X, last_mean=last_mean, last_var=last_var, last_n=last_n, weights=weights
288 )

380def inplace_swap_row_csc(X, m, n):
381 """Swap two rows of a CSC matrix in-place.
382
383 Parameters
384 ----------
385 X : sparse matrix of shape (n_samples, n_features)
386 Matrix whose two rows are to be swapped. It should be of
387 CSC format.
388
389 m : int
390 Index of the row of X to be swapped.
391
392 n : int
393 Index of the row of X to be swapped.
394 """
395 for t in [m, n]:
396 if isinstance(t, np.ndarray):
397 raise TypeError("m and n should be valid integers")
398
399 if m < 0:
400 m += X.shape[0]
401 if n < 0:
402 n += X.shape[0]
403
404 m_mask = X.indices == m
405 X.indices[X.indices == n] = m
406 X.indices[m_mask] = n

409def inplace_swap_row_csr(X, m, n):
410 """Swap two rows of a CSR matrix in-place.
411
412 Parameters
413 ----------
414 X : sparse matrix of shape (n_samples, n_features)
415 Matrix whose two rows are to be swapped. It should be of
416 CSR format.
417
418 m : int
419 Index of the row of X to be swapped.
420
421 n : int
422 Index of the row of X to be swapped.
423 """
424 for t in [m, n]:
425 if isinstance(t, np.ndarray):
426 raise TypeError("m and n should be valid integers")
427
428 if m < 0:
429 m += X.shape[0]
430 if n < 0:
431 n += X.shape[0]
432
433 # The following swapping makes life easier since m is assumed to be the
434 # smaller integer below.
435 if m > n:
436 m, n = n, m
437
438 indptr = X.indptr
439 m_start = indptr[m]
440 m_stop = indptr[m + 1]
441 n_start = indptr[n]
442 n_stop = indptr[n + 1]
443 nz_m = m_stop - m_start
444 nz_n = n_stop - n_start
445
446 if nz_m != nz_n:
447 # Modify indptr first
448 X.indptr[m + 2 : n] += nz_n - nz_m
449 X.indptr[m + 1] = m_start + nz_n
450 X.indptr[n] = n_stop - nz_m
451
452 X.indices = np.concatenate(
453 [
454 X.indices[:m_start],
455 X.indices[n_start:n_stop],
456 X.indices[m_stop:n_start],
457 X.indices[m_start:m_stop],
458 X.indices[n_stop:],
459 ]
460 )
461 X.data = np.concatenate(
462 [
463 X.data[:m_start],
464 X.data[n_start:n_stop],
465 X.data[m_stop:n_start],
466 X.data[m_start:m_stop],
467 X.data[n_stop:],
468 ]
469 )

602def count_nonzero(X, axis=None, sample_weight=None):
603 """A variant of X.getnnz() with extension to weighting on axis 0.
604
605 Useful in efficiently calculating multilabel metrics.
606
607 Parameters
608 ----------
609 X : sparse matrix of shape (n_samples, n_labels)
610 Input data. It should be of CSR format.
611
612 axis : {0, 1}, default=None
613 The axis on which the data is aggregated.
614
615 sample_weight : array-like of shape (n_samples,), default=None
616 Weight for each row of X.
617
618 Returns
619 -------
620 nnz : int, float, ndarray of shape (n_samples,) or ndarray of shape (n_features,)
621 Number of non-zero values in the array along a given axis. Otherwise,
622 the total number of non-zero values in the array is returned.
623 """
624 if axis == -1:
625 axis = 1
626 elif axis == -2:
627 axis = 0
628 elif X.format != "csr":
629 raise TypeError("Expected CSR sparse format, got {0}".format(X.format))
630
631 # We rely here on the fact that np.diff(Y.indptr) for a CSR
632 # will return the number of nonzero entries in each row.
633 # A bincount over Y.indices will return the number of nonzeros
634 # in each column. See ``csr_matrix.getnnz`` in scipy >= 0.14.
635 if axis is None:
636 if sample_weight is None:
637 return X.nnz
638 else:
639 return np.dot(np.diff(X.indptr), sample_weight)
640 elif axis == 1:
641 out = np.diff(X.indptr)
642 if sample_weight is None:
643 # astype here is for consistency with axis=0 dtype
644 return out.astype("intp")
645 return out * sample_weight
646 elif axis == 0:
647 if sample_weight is None:
648 return np.bincount(X.indices, minlength=X.shape[1])
649 else:
650 weights = np.repeat(sample_weight, np.diff(X.indptr))
651 return np.bincount(X.indices, minlength=X.shape[1], weights=weights)
652 else:
653 raise ValueError("Unsupported axis: {0}".format(axis))

656def _get_median(data, n_zeros):
657 """Compute the median of data with n_zeros additional zeros.
658
659 This function is used to support sparse matrices; it modifies data
660 in-place.
661 """
662 n_elems = len(data) + n_zeros
663 if not n_elems:
664 return np.nan
665 n_negative = np.count_nonzero(data < 0)
666 middle, is_odd = divmod(n_elems, 2)
667 data.sort()
668
669 if is_odd:
670 return _get_elem_at_rank(middle, data, n_negative, n_zeros)
671
672 return (
673 _get_elem_at_rank(middle - 1, data, n_negative, n_zeros)
674 + _get_elem_at_rank(middle, data, n_negative, n_zeros)
675 ) / 2.0
676
677
678def _get_elem_at_rank(rank, data, n_negative, n_zeros):
679 """Find the value in data augmented with n_zeros for the given rank"""
680 if rank < n_negative:
681 return data[rank]
682 if rank - n_negative < n_zeros:
683 return 0
684 return data[rank - n_zeros]

687def csc_median_axis_0(X):
688 """Find the median across axis 0 of a CSC matrix.
689
690 It is equivalent to doing np.median(X, axis=0).
691
692 Parameters
693 ----------
694 X : sparse matrix of shape (n_samples, n_features)
695 Input data. It should be of CSC format.
696
697 Returns
698 -------
699 median : ndarray of shape (n_features,)
700 Median.
701 """
702 if not (sp.issparse(X) and X.format == "csc"):
703 raise TypeError("Expected matrix of CSC format, got %s" % X.format)
704
705 indptr = X.indptr
706 n_samples, n_features = X.shape
