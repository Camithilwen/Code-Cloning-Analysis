17def _average_binary_score(binary_metric, y_true, y_score, average, sample_weight=None):
18 """Average a binary metric for multilabel classification.
19
20 Parameters
21 ----------
22 y_true : array, shape = [n_samples] or [n_samples, n_classes]
23 True binary labels in binary label indicators.
24
25 y_score : array, shape = [n_samples] or [n_samples, n_classes]
26 Target scores, can either be probability estimates of the positive
27 class, confidence values, or binary decisions.
28
29 average : {None, 'micro', 'macro', 'samples', 'weighted'}, default='macro'
30 If ``None``, the scores for each class are returned. Otherwise,
31 this determines the type of averaging performed on the data:
32
33 ``'micro'``:
34 Calculate metrics globally by considering each element of the label
35 indicator matrix as a label.
36 ``'macro'``:
37 Calculate metrics for each label, and find their unweighted
38 mean. This does not take label imbalance into account.
39 ``'weighted'``:
40 Calculate metrics for each label, and find their average, weighted
41 by support (the number of true instances for each label).
42 ``'samples'``:
43 Calculate metrics for each instance, and find their average.
44
45 Will be ignored when ``y_true`` is binary.
46
47 sample_weight : array-like of shape (n_samples,), default=None
48 Sample weights.
49
50 binary_metric : callable, returns shape [n_classes]
51 The binary metric function to use.
52
53 Returns
54 -------
55 score : float or array of shape [n_classes]
56 If not ``None``, average the score, else return the score for each
57 classes.
58
59 """
60 average_options = (None, "micro", "macro", "weighted", "samples")
61 if average not in average_options:
62 raise ValueError("average has to be one of {0}".format(average_options))
63
64 y_type = type_of_target(y_true)
65 if y_type not in ("binary", "multilabel-indicator"):
66 raise ValueError("{0} format is not supported".format(y_type))
67
68 if y_type == "binary":
69 return binary_metric(y_true, y_score, sample_weight=sample_weight)
70
71 check_consistent_length(y_true, y_score, sample_weight)
72 y_true = check_array(y_true)
73 y_score = check_array(y_score)
74
75 not_average_axis = 1
76 score_weight = sample_weight
77 average_weight = None
78
79 if average == "micro":
80 if score_weight is not None:
81 score_weight = np.repeat(score_weight, y_true.shape[1])
82 y_true = y_true.ravel()
83 y_score = y_score.ravel()
84
85 elif average == "weighted":
86 if score_weight is not None:
87 average_weight = np.sum(
88 np.multiply(y_true, np.reshape(score_weight, (-1, 1))), axis=0
89 )
90 else:
91 average_weight = np.sum(y_true, axis=0)
92 if np.isclose(average_weight.sum(), 0.0):
93 return 0
94
95 elif average == "samples":
96 # swap average_weight <-> score_weight
97 average_weight = score_weight
98 score_weight = None
99 not_average_axis = 0
100
101 if y_true.ndim == 1:
102 y_true = y_true.reshape((-1, 1))
103
104 if y_score.ndim == 1:
105 y_score = y_score.reshape((-1, 1))
106
107 n_classes = y_score.shape[not_average_axis]
108 score = np.zeros((n_classes,))
109 for c in range(n_classes):
110 y_true_c = y_true.take([c], axis=not_average_axis).ravel()
111 y_score_c = y_score.take([c], axis=not_average_axis).ravel()
112 score[c] = binary_metric(y_true_c, y_score_c, sample_weight=score_weight)
113
114 # Average the results
115 if average is not None:
116 if average_weight is not None:
117 # Scores with 0 weights are forced to be 0, preventing the average
118 # score from being affected by 0-weighted NaN elements.
119 average_weight = np.asarray(average_weight)
120 score[average_weight == 0] = 0
121 return float(np.average(score, weights=average_weight))
122 else:
123 return score

126def _average_multiclass_ovo_score(binary_metric, y_true, y_score, average="macro"):
127 """Average one-versus-one scores for multiclass classification.
128
129 Uses the binary metric for one-vs-one multiclass classification,
130 where the score is computed according to the Hand & Till (2001) algorithm.
131
132 Parameters
133 ----------
134 binary_metric : callable
135 The binary metric function to use that accepts the following as input:
136 y_true_target : array, shape = [n_samples_target]
137 Some sub-array of y_true for a pair of classes designated
138 positive and negative in the one-vs-one scheme.
139 y_score_target : array, shape = [n_samples_target]
140 Scores corresponding to the probability estimates
141 of a sample belonging to the designated positive class label
142
143 y_true : array-like of shape (n_samples,)
144 True multiclass labels.
145
146 y_score : array-like of shape (n_samples, n_classes)
147 Target scores corresponding to probability estimates of a sample
148 belonging to a particular class.
149
150 average : {'macro', 'weighted'}, default='macro'
151 Determines the type of averaging performed on the pairwise binary
152 metric scores:
153 ``'macro'``:
154 Calculate metrics for each label, and find their unweighted
155 mean. This does not take label imbalance into account. Classes
156 are assumed to be uniformly distributed.
157 ``'weighted'``:
158 Calculate metrics for each label, taking into account the
159 prevalence of the classes.
160
161 Returns
162 -------
163 score : float
164 Average of the pairwise binary metric scores.
165 """
166 check_consistent_length(y_true, y_score)
167
168 y_true_unique = np.unique(y_true)
169 n_classes = y_true_unique.shape[0]
170 n_pairs = n_classes * (n_classes - 1) // 2
171 pair_scores = np.empty(n_pairs)
172
173 is_weighted = average == "weighted"
174 prevalence = np.empty(n_pairs) if is_weighted else None
175
176 # Compute scores treating a as positive class and b as negative class,
177 # then b as positive class and a as negative class
178 for ix, (a, b) in enumerate(combinations(y_true_unique, 2)):
179 a_mask = y_true == a
180 b_mask = y_true == b
181 ab_mask = np.logical_or(a_mask, b_mask)
182
183 if is_weighted:
184 prevalence[ix] = np.average(ab_mask)
185
186 a_true = a_mask[ab_mask]
187 b_true = b_mask[ab_mask]
188
189 a_true_score = binary_metric(a_true, y_score[ab_mask, a])
190 b_true_score = binary_metric(b_true, y_score[ab_mask, b])
191 pair_scores[ix] = (a_true_score + b_true_score) / 2
192
193 return np.average(pair_scores, weights=prevalence)
