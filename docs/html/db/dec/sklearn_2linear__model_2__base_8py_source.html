55def make_dataset(X, y, sample_weight, random_state=None):
56 """Create ``Dataset`` abstraction for sparse and dense inputs.
57
58 This also returns the ``intercept_decay`` which is different
59 for sparse datasets.
60
61 Parameters
62 ----------
63 X : array-like, shape (n_samples, n_features)
64 Training data
65
66 y : array-like, shape (n_samples, )
67 Target values.
68
69 sample_weight : numpy array of shape (n_samples,)
70 The weight of each sample
71
72 random_state : int, RandomState instance or None (default)
73 Determines random number generation for dataset random sampling. It is not
74 used for dataset shuffling.
75 Pass an int for reproducible output across multiple function calls.
76 See :term:`Glossary <random_state>`.
77
78 Returns
79 -------
80 dataset
81 The ``Dataset`` abstraction
82 intercept_decay
83 The intercept decay
84 """
85
86 rng = check_random_state(random_state)
87 # seed should never be 0 in SequentialDataset64
88 seed = rng.randint(1, np.iinfo(np.int32).max)
89
90 if X.dtype == np.float32:
91 CSRData = CSRDataset32
92 ArrayData = ArrayDataset32
93 else:
94 CSRData = CSRDataset64
95 ArrayData = ArrayDataset64
96
97 if sp.issparse(X):
98 dataset = CSRData(X.data, X.indptr, X.indices, y, sample_weight, seed=seed)
99 intercept_decay = SPARSE_INTERCEPT_DECAY
100 else:
101 X = np.ascontiguousarray(X)
102 dataset = ArrayData(X, y, sample_weight, seed=seed)
103 intercept_decay = 1.0
104
105 return dataset, intercept_decay

108def _preprocess_data(
109 X,
110 y,
111 *,
112 fit_intercept,
113 copy=True,
114 copy_y=True,
115 sample_weight=None,
116 check_input=True,
117):
118 """Common data preprocessing for fitting linear models.
119
120 This helper is in charge of the following steps:
121
122 - Ensure that `sample_weight` is an array or `None`.
123 - If `check_input=True`, perform standard input validation of `X`, `y`.
124 - Perform copies if requested to avoid side-effects in case of inplace
125 modifications of the input.
126
127 Then, if `fit_intercept=True` this preprocessing centers both `X` and `y` as
128 follows:
129 - if `X` is dense, center the data and
130 store the mean vector in `X_offset`.
131 - if `X` is sparse, store the mean in `X_offset`
132 without centering `X`. The centering is expected to be handled by the
133 linear solver where appropriate.
134 - in either case, always center `y` and store the mean in `y_offset`.
135 - both `X_offset` and `y_offset` are always weighted by `sample_weight`
136 if not set to `None`.
137
138 If `fit_intercept=False`, no centering is performed and `X_offset`, `y_offset`
139 are set to zero.
140
141 Returns
142 -------
143 X_out : {ndarray, sparse matrix} of shape (n_samples, n_features)
144 If copy=True a copy of the input X is triggered, otherwise operations are
145 inplace.
146 If input X is dense, then X_out is centered.
147 y_out : {ndarray, sparse matrix} of shape (n_samples,) or (n_samples, n_targets)
148 Centered version of y. Possibly performed inplace on input y depending
149 on the copy_y parameter.
150 X_offset : ndarray of shape (n_features,)
151 The mean per column of input X.
152 y_offset : float or ndarray of shape (n_features,)
153 X_scale : ndarray of shape (n_features,)
154 Always an array of ones. TODO: refactor the code base to make it
155 possible to remove this unused variable.
156 """
157 xp, _, device_ = get_namespace_and_device(X, y, sample_weight)
158 n_samples, n_features = X.shape
159 X_is_sparse = sp.issparse(X)
160
161 if isinstance(sample_weight, numbers.Number):
162 sample_weight = None
163 if sample_weight is not None:
164 sample_weight = xp.asarray(sample_weight)
165
166 if check_input:
167 X = check_array(
168 X, copy=copy, accept_sparse=["csr", "csc"], dtype=supported_float_dtypes(xp)
169 )
170 y = check_array(y, dtype=X.dtype, copy=copy_y, ensure_2d=False)
171 else:
172 y = xp.astype(y, X.dtype, copy=copy_y)
173 if copy:
174 if X_is_sparse:
175 X = X.copy()
176 else:
177 X = _asarray_with_order(X, order="K", copy=True, xp=xp)
178
179 dtype_ = X.dtype
180
181 if fit_intercept:
182 if X_is_sparse:
183 X_offset, X_var = mean_variance_axis(X, axis=0, weights=sample_weight)
184 else:
185 X_offset = _average(X, axis=0, weights=sample_weight, xp=xp)
186
187 X_offset = xp.astype(X_offset, X.dtype, copy=False)
188 X -= X_offset
189
190 y_offset = _average(y, axis=0, weights=sample_weight, xp=xp)
191 y -= y_offset
192 else:
193 X_offset = xp.zeros(n_features, dtype=X.dtype, device=device_)
194 if y.ndim == 1:
195 y_offset = xp.asarray(0.0, dtype=dtype_, device=device_)
196 else:
197 y_offset = xp.zeros(y.shape[1], dtype=dtype_, device=device_)
198
199 # XXX: X_scale is no longer needed. It is an historic artifact from the
200 # time where linear model exposed the normalize parameter.
201 X_scale = xp.ones(n_features, dtype=X.dtype, device=device_)
202 return X, y, X_offset, y_offset, X_scale

210def _rescale_data(X, y, sample_weight, inplace=False):
211 """Rescale data sample-wise by square root of sample_weight.
212
213 For many linear models, this enables easy support for sample_weight because
214
215 (y - X w)' S (y - X w)
216
217 with S = diag(sample_weight) becomes
218
219 ||y_rescaled - X_rescaled w||_2^2
220
221 when setting
222
223 y_rescaled = sqrt(S) y
224 X_rescaled = sqrt(S) X
225
226 Returns
227 -------
228 X_rescaled : {array-like, sparse matrix}
229
230 y_rescaled : {array-like, sparse matrix}
231 """
232 # Assume that _validate_data and _check_sample_weight have been called by
233 # the caller.
234 xp, _ = get_namespace(X, y, sample_weight)
235 n_samples = X.shape[0]
236 sample_weight_sqrt = xp.sqrt(sample_weight)
237
238 if sp.issparse(X) or sp.issparse(y):
239 sw_matrix = sparse.dia_matrix(
240 (sample_weight_sqrt, 0), shape=(n_samples, n_samples)
241 )
242
243 if sp.issparse(X):
244 X = safe_sparse_dot(sw_matrix, X)
245 else:
246 if inplace:
247 X *= sample_weight_sqrt[:, None]
248 else:
249 X = X * sample_weight_sqrt[:, None]
250
251 if sp.issparse(y):
252 y = safe_sparse_dot(sw_matrix, y)
253 else:
254 if inplace:
255 if y.ndim == 1:
256 y *= sample_weight_sqrt
257 else:
258 y *= sample_weight_sqrt[:, None]
259 else:
260 if y.ndim == 1:
261 y = y * sample_weight_sqrt
262 else:
263 y = y * sample_weight_sqrt[:, None]
264 return X, y, sample_weight_sqrt

267class LinearModel(BaseEstimator, metaclass=ABCMeta):
268 """Base class for Linear Models"""
269
270 @abstractmethod
271 def fit(self, X, y):
272 """Fit model."""
273
274 def _decision_function(self, X):
275 check_is_fitted(self)
276
277 X = validate_data(self, X, accept_sparse=["csr", "csc", "coo"], reset=False)
278 coef_ = self.coef_
279 if coef_.ndim == 1:
280 return X @ coef_ + self.intercept_
281 else:
282 return X @ coef_.T + self.intercept_
283
284 def predict(self, X):
285 """
286 Predict using the linear model.
287
288 Parameters
289 ----------
290 X : array-like or sparse matrix, shape (n_samples, n_features)
291 Samples.
292
293 Returns
294 -------
295 C : array, shape (n_samples,)
296 Returns predicted values.
297 """
298 return self._decision_function(X)
299
300 def _set_intercept(self, X_offset, y_offset, X_scale):
301 """Set the intercept_"""
302
303 xp, _ = get_namespace(X_offset, y_offset, X_scale)
304
305 if self.fit_intercept:
306 # We always want coef_.dtype=X.dtype. For instance, X.dtype can differ from
307 # coef_.dtype if warm_start=True.
308 coef_ = xp.astype(self.coef_, X_scale.dtype, copy=False)
309 coef_ = self.coef_ = xp.divide(coef_, X_scale)
310
311 if coef_.ndim == 1:
312 intercept_ = y_offset - X_offset @ coef_
313 else:
314 intercept_ = y_offset - X_offset @ coef_.T
315
316 self.intercept_ = intercept_
317
318 else:
319 self.intercept_ = 0.0

330 def decision_function(self, X):
331 """
332 Predict confidence scores for samples.
333
334 The confidence score for a sample is proportional to the signed
335 distance of that sample to the hyperplane.
336
337 Parameters
338 ----------
339 X : {array-like, sparse matrix} of shape (n_samples, n_features)
340 The data matrix for which we want to get the confidence scores.
341
342 Returns
343 -------
344 scores : ndarray of shape (n_samples,) or (n_samples, n_classes)
345 Confidence scores per `(n_samples, n_classes)` combination. In the
346 binary case, confidence score for `self.classes_[1]` where >0 means
347 this class would be predicted.
348 """
349 check_is_fitted(self)
350 xp, _ = get_namespace(X)
351
352 X = validate_data(self, X, accept_sparse="csr", reset=False)
353 scores = safe_sparse_dot(X, self.coef_.T, dense_output=True) + self.intercept_
354 return (
355 xp.reshape(scores, (-1,))
356 if (scores.ndim > 1 and scores.shape[1] == 1)
357 else scores
358 )
359
360 def predict(self, X):
361 """
362 Predict class labels for samples in X.
363
364 Parameters
365 ----------
366 X : {array-like, sparse matrix} of shape (n_samples, n_features)
367 The data matrix for which we want to get the predictions.
368
369 Returns
370 -------
371 y_pred : ndarray of shape (n_samples,)
372 Vector containing the class labels for each sample.
373 """
374 xp, _ = get_namespace(X)
375 scores = self.decision_function(X)
376 if len(scores.shape) == 1:
377 indices = xp.astype(scores > 0, indexing_dtype(xp))
378 else:
379 indices = xp.argmax(scores, axis=1)
380
381 return xp.take(self.classes_, indices, axis=0)

383 def _predict_proba_lr(self, X):
384 """Probability estimation for OvR logistic regression.
385
386 Positive class probabilities are computed as
387 1. / (1. + np.exp(-self.decision_function(X)));
388 multiclass is handled by normalizing that over all classes.
389 """
390 prob = self.decision_function(X)
391 expit(prob, out=prob)
392 if prob.ndim == 1:
393 return np.vstack([1 - prob, prob]).T
394 else:
395 # OvR normalization, like LibLinear's predict_probability
396 prob /= prob.sum(axis=1).reshape((prob.shape[0], -1))
397 return prob

406 def densify(self):
407 """
408 Convert coefficient matrix to dense array format.
409
410 Converts the ``coef_`` member (back) to a numpy.ndarray. This is the
411 default format of ``coef_`` and is required for fitting, so calling
412 this method is only required on models that have previously been
413 sparsified; otherwise, it is a no-op.
414
415 Returns
416 -------
417 self
418 Fitted estimator.
419 """
420 msg = "Estimator, %(name)s, must be fitted before densifying."
421 check_is_fitted(self, msg=msg)
422 if sp.issparse(self.coef_):
423 self.coef_ = self.coef_.toarray()
424 return self
425
426 def sparsify(self):
427 """
428 Convert coefficient matrix to sparse format.
429
430 Converts the ``coef_`` member to a scipy.sparse matrix, which for
431 L1-regularized models can be much more memory- and storage-efficient
432 than the usual numpy.ndarray representation.
433
434 The ``intercept_`` member is not converted.
435
436 Returns
437 -------
438 self
439 Fitted estimator.
440
441 Notes
442 -----
443 For non-sparse models, i.e. when there are not many zeros in ``coef_``,
444 this may actually *increase* memory usage, so use this method with
445 care. A rule of thumb is that the number of zero elements, which can
446 be computed with ``(coef_ == 0).sum()``, must be more than 50% for this
447 to provide significant benefits.
448
449 After calling this method, further fitting with the partial_fit
450 method (if any) will not work until you call densify.
451 """
452 msg = "Estimator, %(name)s, must be fitted before sparsifying."
453 check_is_fitted(self, msg=msg)
454 self.coef_ = sp.csr_matrix(self.coef_)
455 return self

458class LinearRegression(MultiOutputMixin, RegressorMixin, LinearModel):
459 """
460 Ordinary least squares Linear Regression.
461
462 LinearRegression fits a linear model with coefficients w = (w1, ..., wp)
463 to minimize the residual sum of squares between the observed targets in
464 the dataset, and the targets predicted by the linear approximation.
465
466 Parameters
467 ----------
468 fit_intercept : bool, default=True
469 Whether to calculate the intercept for this model. If set
470 to False, no intercept will be used in calculations
471 (i.e. data is expected to be centered).
472
473 copy_X : bool, default=True
474 If True, X will be copied; else, it may be overwritten.
475
476 tol : float, default=1e-6
477 The precision of the solution (`coef_`) is determined by `tol` which
478 specifies a different convergence criterion for the `lsqr` solver.
479 `tol` is set as `atol` and `btol` of `scipy.sparse.linalg.lsqr` when
480 fitting on sparse training data. This parameter has no effect when fitting
481 on dense data.
482
483 .. versionadded:: 1.7
484
485 n_jobs : int, default=None
486 The number of jobs to use for the computation. This will only provide
487 speedup in case of sufficiently large problems, that is if firstly
488 `n_targets > 1` and secondly `X` is sparse or if `positive` is set
489 to `True`. ``None`` means 1 unless in a
490 :obj:`joblib.parallel_backend` context. ``-1`` means using all
491 processors. See :term:`Glossary <n_jobs>` for more details.
492
493 positive : bool, default=False
494 When set to ``True``, forces the coefficients to be positive. This
495 option is only supported for dense arrays.
496
497 For a comparison between a linear regression model with positive constraints
498 on the regression coefficients and a linear regression without such constraints,
499 see :ref:`sphx_glr_auto_examples_linear_model_plot_nnls.py`.
500
501 .. versionadded:: 0.24
502
503 Attributes
504 ----------
505 coef_ : array of shape (n_features, ) or (n_targets, n_features)
506 Estimated coefficients for the linear regression problem.
507 If multiple targets are passed during the fit (y 2D), this
508 is a 2D array of shape (n_targets, n_features), while if only
509 one target is passed, this is a 1D array of length n_features.
510
511 rank_ : int
512 Rank of matrix `X`. Only available when `X` is dense.
513
514 singular_ : array of shape (min(X, y),)
515 Singular values of `X`. Only available when `X` is dense.
516
517 intercept_ : float or array of shape (n_targets,)
518 Independent term in the linear model. Set to 0.0 if
519 `fit_intercept = False`.
520
521 n_features_in_ : int
522 Number of features seen during :term:`fit`.
523
524 .. versionadded:: 0.24
525
526 feature_names_in_ : ndarray of shape (`n_features_in_`,)
527 Names of features seen during :term:`fit`. Defined only when `X`
528 has feature names that are all strings.
529
530 .. versionadded:: 1.0
531
532 See Also
533 --------
534 Ridge : Ridge regression addresses some of the
535 problems of Ordinary Least Squares by imposing a penalty on the
536 size of the coefficients with l2 regularization.
537 Lasso : The Lasso is a linear model that estimates
538 sparse coefficients with l1 regularization.
539 ElasticNet : Elastic-Net is a linear regression
540 model trained with both l1 and l2 -norm regularization of the
541 coefficients.
542
543 Notes
544 -----
545 From the implementation point of view, this is just plain Ordinary
546 Least Squares (scipy.linalg.lstsq) or Non Negative Least Squares
547 (scipy.optimize.nnls) wrapped as a predictor object.
548
549 Examples
550 --------
551 >>> import numpy as np
552 >>> from sklearn.linear_model import LinearRegression
553 >>> X = np.array([[1, 1], [1, 2], [2, 2], [2, 3]])
554 >>> # y = 1 * x_0 + 2 * x_1 + 3
555 >>> y = np.dot(X, np.array([1, 2])) + 3
556 >>> reg = LinearRegression().fit(X, y)
557 >>> reg.score(X, y)
558 1.0
559 >>> reg.coef_
560 array([1., 2.])
561 >>> reg.intercept_
562 np.float64(3.0)
563 >>> reg.predict(np.array([[3, 5]]))
564 array([16.])
565 """
566
567 _parameter_constraints: dict = {
568 "fit_intercept": ["boolean"],
569 "copy_X": ["boolean"],
570 "n_jobs": [None, Integral],
571 "positive": ["boolean"],
572 "tol": [Interval(Real, 0, None, closed="left")],
573 }
574
575 def __init__(
576 self,
577 *,
578 fit_intercept=True,
579 copy_X=True,
580 tol=1e-6,
581 n_jobs=None,
582 positive=False,
583 ):
584 self.fit_intercept = fit_intercept
585 self.copy_X = copy_X
586 self.tol = tol
587 self.n_jobs = n_jobs
588 self.positive = positive
589
590 @_fit_context(prefer_skip_nested_validation=True)
591 def fit(self, X, y, sample_weight=None):
592 """
593 Fit linear model.
594
595 Parameters
596 ----------
597 X : {array-like, sparse matrix} of shape (n_samples, n_features)
598 Training data.
599
600 y : array-like of shape (n_samples,) or (n_samples, n_targets)
601 Target values. Will be cast to X's dtype if necessary.
602
603 sample_weight : array-like of shape (n_samples,), default=None
604 Individual weights for each sample.
605
606 .. versionadded:: 0.17
607 parameter *sample_weight* support to LinearRegression.
608
609 Returns
610 -------
611 self : object
612 Fitted Estimator.
613 """
614 n_jobs_ = self.n_jobs
615
616 accept_sparse = False if self.positive else ["csr", "csc", "coo"]
617
618 X, y = validate_data(
619 self,
620 X,
621 y,
622 accept_sparse=accept_sparse,
623 y_numeric=True,
624 multi_output=True,
625 force_writeable=True,
626 )
627
628 has_sw = sample_weight is not None
629 if has_sw:
630 sample_weight = _check_sample_weight(
631 sample_weight, X, dtype=X.dtype, ensure_non_negative=True
632 )
633
634 # Note that neither _rescale_data nor the rest of the fit method of
635 # LinearRegression can benefit from in-place operations when X is a
636 # sparse matrix. Therefore, let's not copy X when it is sparse.
637 copy_X_in_preprocess_data = self.copy_X and not sp.issparse(X)
638
639 X, y, X_offset, y_offset, X_scale = _preprocess_data(
640 X,
641 y,
642 fit_intercept=self.fit_intercept,
643 copy=copy_X_in_preprocess_data,
644 sample_weight=sample_weight,
645 )
646
647 if has_sw:
648 # Sample weight can be implemented via a simple rescaling. Note
649 # that we safely do inplace rescaling when _preprocess_data has
650 # already made a copy if requested.
651 X, y, sample_weight_sqrt = _rescale_data(
652 X, y, sample_weight, inplace=copy_X_in_preprocess_data
653 )
654
655 if self.positive:
656 if y.ndim < 2:
657 self.coef_ = optimize.nnls(X, y)[0]
658 else:
659 # scipy.optimize.nnls cannot handle y with shape (M, K)
660 outs = Parallel(n_jobs=n_jobs_)(
661 delayed(optimize.nnls)(X, y[:, j]) for j in range(y.shape[1])
662 )
663 self.coef_ = np.vstack([out[0] for out in outs])
664 elif sp.issparse(X):
665 X_offset_scale = X_offset / X_scale
666
667 if has_sw:
668
669 def matvec(b):
670 return X.dot(b) - sample_weight_sqrt * b.dot(X_offset_scale)
671
672 def rmatvec(b):
673 return X.T.dot(b) - X_offset_scale * b.dot(sample_weight_sqrt)
674
675 else:
676
677 def matvec(b):
678 return X.dot(b) - b.dot(X_offset_scale)
679
680 def rmatvec(b):
681 return X.T.dot(b) - X_offset_scale * b.sum()
682
683 X_centered = sparse.linalg.LinearOperator(
684 shape=X.shape, matvec=matvec, rmatvec=rmatvec
685 )
686
687 if y.ndim < 2:
688 self.coef_ = lsqr(X_centered, y, atol=self.tol, btol=self.tol)[0]
689 else:
690 # sparse_lstsq cannot handle y with shape (M, K)
691 outs = Parallel(n_jobs=n_jobs_)(
692 delayed(lsqr)(
693 X_centered, y[:, j].ravel(), atol=self.tol, btol=self.tol
694 )
695 for j in range(y.shape[1])
696 )
697 self.coef_ = np.vstack([out[0] for out in outs])
698 else:
699 # cut-off ratio for small singular values
700 cond = max(X.shape) * np.finfo(X.dtype).eps
701 self.coef_, _, self.rank_, self.singular_ = linalg.lstsq(X, y, cond=cond)
702 self.coef_ = self.coef_.T
703
704 if y.ndim == 1:
705 self.coef_ = np.ravel(self.coef_)
706 self._set_intercept(X_offset, y_offset, X_scale)
707 return self

715def _check_precomputed_gram_matrix(
716 X, precompute, X_offset, X_scale, rtol=None, atol=1e-5
717):
718 """Computes a single element of the gram matrix and compares it to
719 the corresponding element of the user supplied gram matrix.
720
721 If the values do not match a ValueError will be thrown.
722
723 Parameters
724 ----------
725 X : ndarray of shape (n_samples, n_features)
726 Data array.
727
728 precompute : array-like of shape (n_features, n_features)
729 User-supplied gram matrix.
730
731 X_offset : ndarray of shape (n_features,)
732 Array of feature means used to center design matrix.
733
734 X_scale : ndarray of shape (n_features,)
735 Array of feature scale factors used to normalize design matrix.
736
737 rtol : float, default=None
738 Relative tolerance; see numpy.allclose
739 If None, it is set to 1e-4 for arrays of dtype numpy.float32 and 1e-7
740 otherwise.
741
742 atol : float, default=1e-5
743 absolute tolerance; see :func`numpy.allclose`. Note that the default
744 here is more tolerant than the default for
745 :func:`numpy.testing.assert_allclose`, where `atol=0`.
746
747 Raises
748 ------
749 ValueError
750 Raised when the provided Gram matrix is not consistent.
751 """
752
753 n_features = X.shape[1]
754 f1 = n_features // 2
755 f2 = min(f1 + 1, n_features - 1)
756
757 v1 = (X[:, f1] - X_offset[f1]) * X_scale[f1]
758 v2 = (X[:, f2] - X_offset[f2]) * X_scale[f2]
759
760 expected = np.dot(v1, v2)
761 actual = precompute[f1, f2]
762
763 dtypes = [precompute.dtype, expected.dtype]
764 if rtol is None:
765 rtols = [1e-4 if dtype == np.float32 else 1e-7 for dtype in dtypes]
766 rtol = max(rtols)
767
768 if not np.isclose(expected, actual, rtol=rtol, atol=atol):
769 raise ValueError(
770 "Gram matrix passed in via 'precompute' parameter "
771 "did not pass validation when a single element was "
772 "checked - please check that it was computed "
773 f"properly. For element ({f1},{f2}) we computed "
774 f"{expected} but the user-supplied value was "
775 f"{actual}."
776 )
