60def composite_images(images, renderer, magnification=1.0):
61 """
62 Composite a number of RGBA images into one. The images are
63 composited in the order in which they appear in the *images* list.
64
65 Parameters
66 ----------
67 images : list of Images
68 Each must have a `make_image` method. For each image,
69 `can_composite` should return `True`, though this is not
70 enforced by this function. Each image must have a purely
71 affine transformation with no shear.
72
73 renderer : `.RendererBase`
74
75 magnification : float, default: 1
76 The additional magnification to apply for the renderer in use.
77
78 Returns
79 -------
80 image : (M, N, 4) `numpy.uint8` array
81 The composited RGBA image.
82 offset_x, offset_y : float
83 The (left, bottom) offset where the composited image should be placed
84 in the output figure.
85 """
86 if len(images) == 0:
87 return np.empty((0, 0, 4), dtype=np.uint8), 0, 0
88
89 parts = []
90 bboxes = []
91 for image in images:
92 data, x, y, trans = image.make_image(renderer, magnification)
93 if data is not None:
94 x *= magnification
95 y *= magnification
96 parts.append((data, x, y, image._get_scalar_alpha()))
97 bboxes.append(
98 Bbox([[x, y], [x + data.shape[1], y + data.shape[0]]]))
99
100 if len(parts) == 0:
101 return np.empty((0, 0, 4), dtype=np.uint8), 0, 0
102
103 bbox = Bbox.union(bboxes)
104
105 output = np.zeros(
106 (int(bbox.height), int(bbox.width), 4), dtype=np.uint8)
107
108 for data, x, y, alpha in parts:
109 trans = Affine2D().translate(x - bbox.x0, y - bbox.y0)
110 _image.resample(data, output, trans, _image.NEAREST,
111 resample=False, alpha=alpha)
112
113 return output, bbox.x0 / magnification, bbox.y0 / magnification

163def _resample(
164 image_obj, data, out_shape, transform, *, resample=None, alpha=1):
165 """
166 Convenience wrapper around `._image.resample` to resample *data* to
167 *out_shape* (with a third dimension if *data* is RGBA) that takes care of
168 allocating the output array and fetching the relevant properties from the
169 Image object *image_obj*.
170 """
171 # AGG can only handle coordinates smaller than 24-bit signed integers,
172 # so raise errors if the input data is larger than _image.resample can
173 # handle.
174 msg = ('Data with more than {n} cannot be accurately displayed. '
175 'Downsampling to less than {n} before displaying. '
176 'To remove this warning, manually downsample your data.')
177 if data.shape[1] > 2**23:
178 warnings.warn(msg.format(n='2**23 columns'))
179 step = int(np.ceil(data.shape[1] / 2**23))
180 data = data[:, ::step]
181 transform = Affine2D().scale(step, 1) + transform
182 if data.shape[0] > 2**24:
183 warnings.warn(msg.format(n='2**24 rows'))
184 step = int(np.ceil(data.shape[0] / 2**24))
185 data = data[::step, :]
186 transform = Affine2D().scale(1, step) + transform
187 # decide if we need to apply anti-aliasing if the data is upsampled:
188 # compare the number of displayed pixels to the number of
189 # the data pixels.
190 interpolation = image_obj.get_interpolation()
191 if interpolation in ['antialiased', 'auto']:
192 # don't antialias if upsampling by an integer number or
193 # if zooming in more than a factor of 3
194 pos = np.array([[0, 0], [data.shape[1], data.shape[0]]])
195 disp = transform.transform(pos)
196 dispx = np.abs(np.diff(disp[:, 0]))
197 dispy = np.abs(np.diff(disp[:, 1]))
198 if ((dispx > 3 * data.shape[1] or
199 dispx == data.shape[1] or
200 dispx == 2 * data.shape[1]) and
201 (dispy > 3 * data.shape[0] or
202 dispy == data.shape[0] or
203 dispy == 2 * data.shape[0])):
204 interpolation = 'nearest'
205 else:
206 interpolation = 'hanning'
207 out = np.zeros(out_shape + data.shape[2:], data.dtype) # 2D->2D, 3D->3D.
208 if resample is None:
209 resample = image_obj.get_resample()
210 _image.resample(data, out, transform,
211 _interpd_[interpolation],
212 resample,
213 alpha,
214 image_obj.get_filternorm(),
215 image_obj.get_filterrad())
216 return out

337 def _make_image(self, A, in_bbox, out_bbox, clip_bbox, magnification=1.0,
338 unsampled=False, round_to_pixel_border=True):
339 """
340 Normalize, rescale, and colormap the image *A* from the given *in_bbox*
341 (in data space), to the given *out_bbox* (in pixel space) clipped to
342 the given *clip_bbox* (also in pixel space), and magnified by the
343 *magnification* factor.
344
345 Parameters
346 ----------
347 A : ndarray
348
349 - a (M, N) array interpreted as scalar (greyscale) image,
350 with one of the dtypes `~numpy.float32`, `~numpy.float64`,
351 `~numpy.float128`, `~numpy.uint16` or `~numpy.uint8`.
352 - (M, N, 4) RGBA image with a dtype of `~numpy.float32`,
353 `~numpy.float64`, `~numpy.float128`, or `~numpy.uint8`.
354
355 in_bbox : `~matplotlib.transforms.Bbox`
356
357 out_bbox : `~matplotlib.transforms.Bbox`
358
359 clip_bbox : `~matplotlib.transforms.Bbox`
360
361 magnification : float, default: 1
362
363 unsampled : bool, default: False
364 If True, the image will not be scaled, but an appropriate
365 affine transformation will be returned instead.
366
367 round_to_pixel_border : bool, default: True
368 If True, the output image size will be rounded to the nearest pixel
369 boundary. This makes the images align correctly with the Axes.
370 It should not be used if exact scaling is needed, such as for
371 `.FigureImage`.
372
373 Returns
374 -------
375 image : (M, N, 4) `numpy.uint8` array
376 The RGBA image, resampled unless *unsampled* is True.
377 x, y : float
378 The upper left corner where the image should be drawn, in pixel
379 space.
380 trans : `~matplotlib.transforms.Affine2D`
381 The affine transformation from image to pixel space.
382 """
383 if A is None:
384 raise RuntimeError('You must first set the image '
385 'array or the image attribute')
386 if A.size == 0:
387 raise RuntimeError("_make_image must get a non-empty image. "
388 "Your Artist's draw method must filter before "
389 "this method is called.")
390
391 clipped_bbox = Bbox.intersection(out_bbox, clip_bbox)
392
393 if clipped_bbox is None:
394 return None, 0, 0, None
395
396 out_width_base = clipped_bbox.width * magnification
397 out_height_base = clipped_bbox.height * magnification
398
399 if out_width_base == 0 or out_height_base == 0:
400 return None, 0, 0, None
401
402 if self.origin == 'upper':
403 # Flip the input image using a transform. This avoids the
404 # problem with flipping the array, which results in a copy
405 # when it is converted to contiguous in the C wrapper
406 t0 = Affine2D().translate(0, -A.shape[0]).scale(1, -1)
407 else:
408 t0 = IdentityTransform()
409
410 t0 += (
411 Affine2D()
412 .scale(
413 in_bbox.width / A.shape[1],
414 in_bbox.height / A.shape[0])
415 .translate(in_bbox.x0, in_bbox.y0)
416 + self.get_transform())
417
418 t = (t0
419 + (Affine2D()
420 .translate(-clipped_bbox.x0, -clipped_bbox.y0)
421 .scale(magnification)))
422
423 # So that the image is aligned with the edge of the Axes, we want to
424 # round up the output width to the next integer. This also means
425 # scaling the transform slightly to account for the extra subpixel.
426 if ((not unsampled) and t.is_affine and round_to_pixel_border and
427 (out_width_base % 1.0 != 0.0 or out_height_base % 1.0 != 0.0)):
428 out_width = math.ceil(out_width_base)
429 out_height = math.ceil(out_height_base)
430 extra_width = (out_width - out_width_base) / out_width_base
431 extra_height = (out_height - out_height_base) / out_height_base
432 t += Affine2D().scale(1.0 + extra_width, 1.0 + extra_height)
433 else:
434 out_width = int(out_width_base)
435 out_height = int(out_height_base)
436 out_shape = (out_height, out_width)
437
438 if not unsampled:
439 if not (A.ndim == 2 or A.ndim == 3 and A.shape[-1] in (3, 4)):
440 raise ValueError(f"Invalid shape {A.shape} for image data")
441
442 # if antialiased, this needs to change as window sizes
443 # change:
444 interpolation_stage = self._interpolation_stage
445 if interpolation_stage in ['antialiased', 'auto']:
446 pos = np.array([[0, 0], [A.shape[1], A.shape[0]]])
447 disp = t.transform(pos)
448 dispx = np.abs(np.diff(disp[:, 0])) / A.shape[1]
449 dispy = np.abs(np.diff(disp[:, 1])) / A.shape[0]
450 if (dispx < 3) or (dispy < 3):
451 interpolation_stage = 'rgba'
452 else:
453 interpolation_stage = 'data'
454
455 if A.ndim == 2 and interpolation_stage == 'data':
456 # if we are a 2D array, then we are running through the
457 # norm + colormap transformation. However, in general the
458 # input data is not going to match the size on the screen so we
459 # have to resample to the correct number of pixels
460
461 if A.dtype.kind == 'f': # Float dtype: scale to same dtype.
462 scaled_dtype = np.dtype("f8" if A.dtype.itemsize > 4 else "f4")
463 if scaled_dtype.itemsize < A.dtype.itemsize:
464 _api.warn_external(f"Casting input data from {A.dtype}"
465 f" to {scaled_dtype} for imshow.")
466 else: # Int dtype, likely.
467 # TODO slice input array first
468 # Scale to appropriately sized float: use float32 if the
469 # dynamic range is small, to limit the memory footprint.
470 da = A.max().astype("f8") - A.min().astype("f8")
471 scaled_dtype = "f8" if da > 1e8 else "f4"
472
473 # resample the input data to the correct resolution and shape
474 A_resampled = _resample(self, A.astype(scaled_dtype), out_shape, t)
475
476 # if using NoNorm, cast back to the original datatype
477 if isinstance(self.norm, mcolors.NoNorm):
478 A_resampled = A_resampled.astype(A.dtype)

481 # pixels) and out_alpha (to what extent screen pixels are
482 # covered by data pixels: 0 outside the data extent, 1 inside
483 # (even for bad data), and intermediate values at the edges).
484 mask = (np.where(A.mask, np.float32(np.nan), np.float32(1))

486 else np.ones_like(A, np.float32))

488 # non-affine transformations

499 output = self.norm(resampled_masked)

502 self.norm.autoscale_None(A)
503 A = self.to_rgba(A)
504 alpha = self.get_alpha()
505 if alpha is None: # alpha parameter not specified
506 if A.shape[2] == 3: # image has no alpha channel
507 output_alpha = 255 if A.dtype == np.uint8 else 1.0
508 else:
509 output_alpha = _resample( # resample alpha channel
510 self, A[..., 3], out_shape, t)
511 output = _resample( # resample rgb channels
512 self, _rgb_to_rgba(A[..., :3]), out_shape, t)
513 elif np.ndim(alpha) > 0: # Array alpha
514 # user-specified array alpha overrides the existing alpha channel
515 output_alpha = _resample(self, alpha, out_shape, t)
516 output = _resample(
517 self, _rgb_to_rgba(A[..., :3]), out_shape, t)
518 else: # Scalar alpha
519 if A.shape[2] == 3: # broadcast scalar alpha
520 output_alpha = (255 * alpha) if A.dtype == np.uint8 else alpha
521 else: # or apply scalar alpha to existing alpha channel
522 output_alpha = _resample(self, A[..., 3], out_shape, t) * alpha
523 output = _resample(
524 self, _rgb_to_rgba(A[..., :3]), out_shape, t)
525 output[..., 3] = output_alpha # recombine rgb and alpha
526
527 # output is now either a 2D array of normed (int or float) data
528 # or an RGBA array of re-sampled input
529 output = self.to_rgba(output, bytes=True, norm=False)
530 # output is now a correctly sized RGBA array of uint8
531
532 # Apply alpha *after* if the input was greyscale without a mask
533 if A.ndim == 2:
534 alpha = self._get_scalar_alpha()
535 alpha_channel = output[:, :, 3]
536 alpha_channel[:] = ( # Assignment will cast to uint8.
537 alpha_channel.astype(np.float32) * out_alpha * alpha)
538
539 else:
540 if self._imcache is None:
541 self._imcache = self.to_rgba(A, bytes=True, norm=(A.ndim == 2))
542 output = self._imcache
543
544 # Subset the input image to only the part that will be displayed.
545 subset = TransformedBbox(clip_bbox, t0.inverted()).frozen()
546 output = output[
547 int(max(subset.ymin, 0)):
548 int(min(subset.ymax + 1, output.shape[0])),
549 int(max(subset.xmin, 0)):
550 int(min(subset.xmax + 1, output.shape[1]))]
551
552 t = Affine2D().translate(
553 int(max(subset.xmin, 0)), int(max(subset.ymin, 0))) + t
554
555 return output, clipped_bbox.x0, clipped_bbox.y0, t
