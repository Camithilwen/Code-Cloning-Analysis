12def test_solve_equivalence():
13 # For toeplitz matrices, solve_toeplitz() should be equivalent to solve().
14 random = np.random.RandomState(1234)
15 for n in (1, 2, 3, 10):
16 c = random.randn(n)
17 if random.rand() < 0.5:
18 c = c + 1j * random.randn(n)
19 r = random.randn(n)
20 if random.rand() < 0.5:
21 r = r + 1j * random.randn(n)
22 y = random.randn(n)
23 if random.rand() < 0.5:
24 y = y + 1j * random.randn(n)
25
26 # Check equivalence when both the column and row are provided.
27 actual = solve_toeplitz((c,r), y)
28 desired = solve(toeplitz(c, r=r), y)
29 assert_allclose(actual, desired)
30
31 # Check equivalence when the column is provided but not the row.
32 actual = solve_toeplitz(c, b=y)
33 desired = solve(toeplitz(c), y)
34 assert_allclose(actual, desired)

81def test_reflection_coeffs():
82 # check that the partial solutions are given by the reflection
83 # coefficients
84
85 random = np.random.RandomState(1234)
86 y_d = random.randn(10)
87 y_z = random.randn(10) + 1j
88 reflection_coeffs_d = [1]
89 reflection_coeffs_z = [1]
90 for i in range(2, 10):
91 reflection_coeffs_d.append(solve_toeplitz(y_d[:(i-1)], b=y_d[1:i])[-1])
92 reflection_coeffs_z.append(solve_toeplitz(y_z[:(i-1)], b=y_z[1:i])[-1])
93
94 y_d_concat = np.concatenate((y_d[-2:0:-1], y_d[:-1]))
95 y_z_concat = np.concatenate((y_z[-2:0:-1].conj(), y_z[:-1]))
96 _, ref_d = levinson(y_d_concat, b=y_d[1:])
97 _, ref_z = levinson(y_z_concat, b=y_z[1:])
98
99 assert_allclose(reflection_coeffs_d, ref_d[:-1])
100 assert_allclose(reflection_coeffs_z, ref_z[:-1])
