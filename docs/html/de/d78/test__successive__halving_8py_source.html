33class FastClassifier(DummyClassifier):
34 """Dummy classifier that accepts parameters a, b, ... z.
35
36 These parameter don't affect the predictions and are useful for fast
37 grid searching."""
38
39 # update the constraints such that we accept all parameters from a to z
40 _parameter_constraints: dict = {
41 **DummyClassifier._parameter_constraints,
42 **{chr(key): "no_validation" for key in range(ord("a"), ord("z") + 1)},
43 }
44
45 def __init__(
46 self, strategy="stratified", random_state=None, constant=None, **kwargs

51

53 params = super().get_params(deep=deep)

59class SometimesFailClassifier(DummyClassifier):
60 def __init__(
61 self,
62 strategy="stratified",
63 random_state=None,
64 constant=None,
65 n_estimators=10,
66 fail_fit=False,
67 fail_predict=False,
68 a=0,
69 ):
70 self.fail_fit = fail_fit
71 self.fail_predict = fail_predict
72 self.n_estimators = n_estimators

554@pytest.mark.parametrize("Est", (HalvingRandomSearchCV, HalvingGridSearchCV))
555def test_cv_results(Est):
556 # test that the cv_results_ matches correctly the logic of the
557 # tournament: in particular that the candidates continued in each
558 # successive iteration are those that were best in the previous iteration
559 pd = pytest.importorskip("pandas")
560
561 rng = np.random.RandomState(0)
562
563 n_samples = 1000
564 X, y = make_classification(n_samples=n_samples, random_state=0)
565 param_grid = {"a": ("l1", "l2"), "b": list(range(30))}
566 base_estimator = FastClassifier()
567
568 # generate random scores: we want to avoid ties, which would otherwise
569 # mess with the ordering and make testing harder
570 def scorer(est, X, y):
571 return rng.rand()
572
573 sh = Est(base_estimator, param_grid, factor=2, scoring=scorer)
574 if Est is HalvingRandomSearchCV:
575 # same number of candidates as with the grid
576 sh.set_params(n_candidates=2 * 30, min_resources="exhaust")
577
578 sh.fit(X, y)
579
580 # non-regression check for
581 # https://github.com/scikit-learn/scikit-learn/issues/19203
582 assert isinstance(sh.cv_results_["iter"], np.ndarray)
583 assert isinstance(sh.cv_results_["n_resources"], np.ndarray)
584
585 cv_results_df = pd.DataFrame(sh.cv_results_)
586
587 # just make sure we don't have ties
588 assert len(cv_results_df["mean_test_score"].unique()) == len(cv_results_df)
589
590 cv_results_df["params_str"] = cv_results_df["params"].apply(str)
591 table = cv_results_df.pivot(
592 index="params_str", columns="iter", values="mean_test_score"
593 )
594
595 # table looks like something like this:
596 # iter 0 1 2 3 4 5
597 # params_str
598 # {'a': 'l2', 'b': 23} 0.75 NaN NaN NaN NaN NaN
599 # {'a': 'l1', 'b': 30} 0.90 0.875 NaN NaN NaN NaN
600 # {'a': 'l1', 'b': 0} 0.75 NaN NaN NaN NaN NaN
601 # {'a': 'l2', 'b': 3} 0.85 0.925 0.9125 0.90625 NaN NaN
602 # {'a': 'l1', 'b': 5} 0.80 NaN NaN NaN NaN NaN
603 # ...
604
605 # where a NaN indicates that the candidate wasn't evaluated at a given
606 # iteration, because it wasn't part of the top-K at some previous
607 # iteration. We here make sure that candidates that aren't in the top-k at
608 # any given iteration are indeed not evaluated at the subsequent
609 # iterations.
610 nan_mask = pd.isna(table)
611 n_iter = sh.n_iterations_
612 for it in range(n_iter - 1):
613 already_discarded_mask = nan_mask[it]
614
615 # make sure that if a candidate is already discarded, we don't evaluate
616 # it later
617 assert (
618 already_discarded_mask & nan_mask[it + 1] == already_discarded_mask
619 ).all()
620
621 # make sure that the number of discarded candidate is correct
622 discarded_now_mask = ~already_discarded_mask & nan_mask[it + 1]
623 kept_mask = ~already_discarded_mask & ~discarded_now_mask
624 assert kept_mask.sum() == sh.n_candidates_[it + 1]
625
626 # make sure that all discarded candidates have a lower score than the
627 # kept candidates
628 discarded_max_score = table[it].where(discarded_now_mask).max()
629 kept_min_score = table[it].where(kept_mask).min()
630 assert discarded_max_score < kept_min_score
631
632 # We now make sure that the best candidate is chosen only from the last
633 # iteration.
634 # We also make sure this is true even if there were higher scores in
635 # earlier rounds (this isn't generally the case, but worth ensuring it's
636 # possible).
637
638 last_iter = cv_results_df["iter"].max()
639 idx_best_last_iter = cv_results_df[cv_results_df["iter"] == last_iter][
640 "mean_test_score"
641 ].idxmax()
642 idx_best_all_iters = cv_results_df["mean_test_score"].idxmax()
643
644 assert sh.best_params_ == cv_results_df.iloc[idx_best_last_iter]["params"]
645 assert (
646 cv_results_df.iloc[idx_best_last_iter]["mean_test_score"]
647 < cv_results_df.iloc[idx_best_all_iters]["mean_test_score"]
648 )
649 assert (
650 cv_results_df.iloc[idx_best_last_iter]["params"]
651 != cv_results_df.iloc[idx_best_all_iters]["params"]
652 )
