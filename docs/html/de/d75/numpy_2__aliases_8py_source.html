85# asarray also adds the copy keyword, which is not present in numpy 1.0.
86# asarray() is different enough between numpy, cupy, and dask, the logic
87# complicated enough that it's easier to define it separately for each module
88# rather than trying to combine everything into one function in common/
89def asarray(
90 obj: Array | complex | NestedSequence[complex] | SupportsBufferProtocol,
91 /,
92 *,
93 dtype: DType | None = None,
94 device: Device | None = None,
95 copy: _Copy | None = None,
96 **kwargs: Any,
97) -> Array:
98 """
99 Array API compatibility wrapper for asarray().
100
101 See the corresponding documentation in the array library and/or the array API
102 specification for more details.
103 """
104 _helpers._check_device(np, device)
105
106 if copy is None:
107 copy = np._CopyMode.IF_NEEDED
108 elif copy is False:
109 copy = np._CopyMode.NEVER
110 elif copy is True:
111 copy = np._CopyMode.ALWAYS
112
113 return np.array(obj, copy=copy, dtype=dtype, **kwargs) # pyright: ignore

128# count_nonzero returns a python int for axis=None and keepdims=False
129# https://github.com/numpy/numpy/issues/17562
130def count_nonzero(
131 x: Array,
132 axis: int | tuple[int, ...] | None = None,
133 keepdims: py_bool = False,
134) -> Array:
135 # NOTE: this is currently incorrectly typed in numpy, but will be fixed in
136 # numpy 2.2.5 and 2.3.0: https://github.com/numpy/numpy/pull/28750
137 result = cast("Any", np.count_nonzero(x, axis=axis, keepdims=keepdims)) # pyright: ignore[reportArgumentType, reportCallIssue]
138 if axis is None and not keepdims:
139 return np.asarray(result)
140 return result
