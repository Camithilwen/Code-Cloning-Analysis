40 def check_response(self, h, expected_response, tol=.05):
41 N = len(h)
42 alpha = 0.5 * (N-1)
43 m = np.arange(0,N) - alpha # time indices of taps
44 for freq, expected in expected_response:
45 actual = abs(np.sum(h*np.exp(-1.j*np.pi*m*freq)))
46 mse = abs(actual-expected)**2
47 assert_(mse < tol, 'response not as expected, mse=%g > %g'
48 % (mse, tol))

82 def mse(self, h, bands):
83 """Compute mean squared error versus ideal response across frequency
84 band.
85 h -- coefficients
86 bands -- list of (left, right) tuples relative to 1==Nyquist of
87 passbands
88 """
89 w, H = freqz(h, worN=1024)
90 f = w/np.pi
91 passIndicator = np.zeros(len(w), bool)
92 for left, right in bands:
93 passIndicator |= (f >= left) & (f < right)
94 Hideal = np.where(passIndicator, 1, 0)
95 mse = np.mean(abs(abs(H)-Hideal)**2)
96 return mse

98 def test_scaling(self):
99 """
100 For one lowpass, bandpass, and highpass example filter, this test
101 checks two things:
102 - the mean squared error over the frequency domain of the unscaled
103 filter is smaller than the scaled filter (true for rectangular
104 window)
105 - the response of the scaled filter is exactly unity at the center
106 of the first passband
107 """
108 N = 11
109 cases = [
110 ([.5], True, (0, 1)),
111 ([0.2, .6], False, (.4, 1)),
112 ([.5], False, (1, 1)),
113 ]
114 for cutoff, pass_zero, expected_response in cases:
115 h = firwin(N, cutoff, scale=False, pass_zero=pass_zero, window='ones')
116 hs = firwin(N, cutoff, scale=True, pass_zero=pass_zero, window='ones')
117 if len(cutoff) == 1:
118 if pass_zero:
119 cutoff = [0] + cutoff
120 else:
121 cutoff = cutoff + [1]
122 assert_(self.mse(h, [cutoff]) < self.mse(hs, [cutoff]),
123 'least squares violation')
124 self.check_response(hs, [expected_response], 1e-12)

526 def test_firls(self):
527 N = 11 # number of taps in the filter
528 a = 0.1 # width of the transition band
529
530 # design a halfband symmetric low-pass filter
531 h = firls(11, [0, a, 0.5-a, 0.5], [1, 1, 0, 0], fs=1.0)
532
533 # make sure the filter has correct # of taps
534 assert_equal(len(h), N)
535
536 # make sure it is symmetric
537 midx = (N-1) // 2
538 assert_array_almost_equal(h[:midx], h[:-midx-1:-1])
539
540 # make sure the center tap is 0.5
541 assert_almost_equal(h[midx], 0.5)
542
543 # For halfband symmetric, odd coefficients (except the center)
544 # should be zero (really small)
545 hodd = np.hstack((h[1:midx:2], h[-midx+1::2]))
546 assert_array_almost_equal(hodd, 0)
547
548 # now check the frequency response
549 w, H = freqz(h, 1)
550 f = w/2/np.pi
551 Hmag = np.abs(H)
552
553 # check that the pass band is close to unity
554 idx = np.logical_and(f > 0, f < a)
555 assert_array_almost_equal(Hmag[idx], 1, decimal=3)
556
557 # check that the stop band is close to zero
558 idx = np.logical_and(f > 0.5-a, f < 0.5)
559 assert_array_almost_equal(Hmag[idx], 0, decimal=3)
