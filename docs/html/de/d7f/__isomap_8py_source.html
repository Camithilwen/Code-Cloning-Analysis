28class Isomap(ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator):
29 """Isomap Embedding.
30
31 Non-linear dimensionality reduction through Isometric Mapping
32
33 Read more in the :ref:`User Guide <isomap>`.
34
35 Parameters
36 ----------
37 n_neighbors : int or None, default=5
38 Number of neighbors to consider for each point. If `n_neighbors` is an int,
39 then `radius` must be `None`.
40
41 radius : float or None, default=None
42 Limiting distance of neighbors to return. If `radius` is a float,
43 then `n_neighbors` must be set to `None`.
44
45 .. versionadded:: 1.1
46
47 n_components : int, default=2
48 Number of coordinates for the manifold.
49
50 eigen_solver : {'auto', 'arpack', 'dense'}, default='auto'
51 'auto' : Attempt to choose the most efficient solver
52 for the given problem.
53
54 'arpack' : Use Arnoldi decomposition to find the eigenvalues
55 and eigenvectors.
56
57 'dense' : Use a direct solver (i.e. LAPACK)
58 for the eigenvalue decomposition.
59
60 tol : float, default=0
61 Convergence tolerance passed to arpack or lobpcg.
62 not used if eigen_solver == 'dense'.
63
64 max_iter : int, default=None
65 Maximum number of iterations for the arpack solver.
66 not used if eigen_solver == 'dense'.
67
68 path_method : {'auto', 'FW', 'D'}, default='auto'
69 Method to use in finding shortest path.
70
71 'auto' : attempt to choose the best algorithm automatically.
72
73 'FW' : Floyd-Warshall algorithm.
74
75 'D' : Dijkstra's algorithm.
76
77 neighbors_algorithm : {'auto', 'brute', 'kd_tree', 'ball_tree'}, \
78 default='auto'
79 Algorithm to use for nearest neighbors search,
80 passed to neighbors.NearestNeighbors instance.
81
82 n_jobs : int or None, default=None
83 The number of parallel jobs to run.
84 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
85 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
86 for more details.
87
88 metric : str, or callable, default="minkowski"
89 The metric to use when calculating distance between instances in a
90 feature array. If metric is a string or callable, it must be one of
91 the options allowed by :func:`sklearn.metrics.pairwise_distances` for
92 its metric parameter.
93 If metric is "precomputed", X is assumed to be a distance matrix and
94 must be square. X may be a :term:`Glossary <sparse graph>`.
95
96 .. versionadded:: 0.22
97
98 p : float, default=2
99 Parameter for the Minkowski metric from
100 sklearn.metrics.pairwise.pairwise_distances. When p = 1, this is
101 equivalent to using manhattan_distance (l1), and euclidean_distance
102 (l2) for p = 2. For arbitrary p, minkowski_distance (l_p) is used.
103
104 .. versionadded:: 0.22
105
106 metric_params : dict, default=None
107 Additional keyword arguments for the metric function.
108
109 .. versionadded:: 0.22
110
111 Attributes
112 ----------
113 embedding_ : array-like, shape (n_samples, n_components)
114 Stores the embedding vectors.
115
116 kernel_pca_ : object
117 :class:`~sklearn.decomposition.KernelPCA` object used to implement the
118 embedding.
119
120 nbrs_ : sklearn.neighbors.NearestNeighbors instance
121 Stores nearest neighbors instance, including BallTree or KDtree
122 if applicable.
123
124 dist_matrix_ : array-like, shape (n_samples, n_samples)
125 Stores the geodesic distance matrix of training data.
126
127 n_features_in_ : int
128 Number of features seen during :term:`fit`.
129
130 .. versionadded:: 0.24
131
132 feature_names_in_ : ndarray of shape (`n_features_in_`,)
133 Names of features seen during :term:`fit`. Defined only when `X`
134 has feature names that are all strings.
135
136 .. versionadded:: 1.0
137
138 See Also
139 --------
140 sklearn.decomposition.PCA : Principal component analysis that is a linear
141 dimensionality reduction method.
142 sklearn.decomposition.KernelPCA : Non-linear dimensionality reduction using
143 kernels and PCA.
144 MDS : Manifold learning using multidimensional scaling.
145 TSNE : T-distributed Stochastic Neighbor Embedding.
146 LocallyLinearEmbedding : Manifold learning using Locally Linear Embedding.
147 SpectralEmbedding : Spectral embedding for non-linear dimensionality.
148
149 References
150 ----------
151
152 .. [1] Tenenbaum, J.B.; De Silva, V.; & Langford, J.C. A global geometric
153 framework for nonlinear dimensionality reduction. Science 290 (5500)

155 Examples
156 --------
157 >>> from sklearn.datasets import load_digits
158 >>> from sklearn.manifold import Isomap
159 >>> X, _ = load_digits(return_X_y=True)
160 >>> X.shape
161 (1797, 64)
162 >>> embedding = Isomap(n_components=2)
163 >>> X_transformed = embedding.fit_transform(X[:100])
164 >>> X_transformed.shape
165 (100, 2)

212 def _fit_transform(self, X):
213 if self.n_neighbors is not None and self.radius is not None:
214 raise ValueError(
215 "Both n_neighbors and radius are provided. Use"
216 f" Isomap(radius={self.radius}, n_neighbors=None) if intended to use"
217 " radius-based neighbors"
218 )
219
220 self.nbrs_ = NearestNeighbors(
221 n_neighbors=self.n_neighbors,
222 radius=self.radius,
223 algorithm=self.neighbors_algorithm,
224 metric=self.metric,
225 p=self.p,
226 metric_params=self.metric_params,
227 n_jobs=self.n_jobs,
228 )
229 self.nbrs_.fit(X)
230 self.n_features_in_ = self.nbrs_.n_features_in_
231 if hasattr(self.nbrs_, "feature_names_in_"):
232 self.feature_names_in_ = self.nbrs_.feature_names_in_
233
234 self.kernel_pca_ = KernelPCA(
235 n_components=self.n_components,
236 kernel="precomputed",
237 eigen_solver=self.eigen_solver,
238 tol=self.tol,
239 max_iter=self.max_iter,
240 n_jobs=self.n_jobs,
241 ).set_output(transform="default")
242
243 if self.n_neighbors is not None:
244 nbg = kneighbors_graph(
245 self.nbrs_,
246 self.n_neighbors,
247 metric=self.metric,
248 p=self.p,
249 metric_params=self.metric_params,
250 mode="distance",
251 n_jobs=self.n_jobs,
252 )
253 else:
254 nbg = radius_neighbors_graph(
255 self.nbrs_,
256 radius=self.radius,
257 metric=self.metric,
258 p=self.p,
259 metric_params=self.metric_params,
260 mode="distance",
261 n_jobs=self.n_jobs,
262 )
263
264 # Compute the number of connected components, and connect the different
265 # components to be able to compute a shortest path between all pairs
266 # of samples in the graph.
267 # Similar fix to cluster._agglomerative._fix_connectivity.
268 n_connected_components, labels = connected_components(nbg)
269 if n_connected_components > 1:
270 if self.metric == "precomputed" and issparse(X):
271 raise RuntimeError(
272 "The number of connected components of the neighbors graph"
273 f" is {n_connected_components} > 1. The graph cannot be "
274 "completed with metric='precomputed', and Isomap cannot be"
275 "fitted. Increase the number of neighbors to avoid this "
276 "issue, or precompute the full distance matrix instead "
277 "of passing a sparse neighbors graph."
278 )
279 warnings.warn(
280 (
281 "The number of connected components of the neighbors graph "
282 f"is {n_connected_components} > 1. Completing the graph to fit"
283 " Isomap might be slow. Increase the number of neighbors to "
284 "avoid this issue."
285 ),
286 stacklevel=2,
287 )
288
289 # use array validated by NearestNeighbors
290 nbg = _fix_connected_components(
291 X=self.nbrs_._fit_X,
292 graph=nbg,
293 n_connected_components=n_connected_components,
294 component_labels=labels,
295 mode="distance",
296 metric=self.nbrs_.effective_metric_,
297 **self.nbrs_.effective_metric_params_,
298 )
299
300 self.dist_matrix_ = shortest_path(nbg, method=self.path_method, directed=False)
301
302 if self.nbrs_._fit_X.dtype == np.float32:
303 self.dist_matrix_ = self.dist_matrix_.astype(
304 self.nbrs_._fit_X.dtype, copy=False
305 )
306
307 G = self.dist_matrix_**2
308 G *= -0.5
309
310 self.embedding_ = self.kernel_pca_.fit_transform(G)
311 self._n_features_out = self.embedding_.shape[1]

313 def reconstruction_error(self):
314 """Compute the reconstruction error for the embedding.
315
316 Returns
317 -------
318 reconstruction_error : float
319 Reconstruction error.
320
321 Notes
322 -----
323 The cost function of an isomap embedding is
324
325 ``E = frobenius_norm[K(D) - K(D_fit)] / n_samples``
326
327 Where D is the matrix of distances for the input data X,
328 D_fit is the matrix of distances for the output embedding X_fit,
329 and K is the isomap kernel:
330
331 ``K(D) = -0.5 * (I - 1/n_samples) * D^2 * (I - 1/n_samples)``
332 """
333 G = -0.5 * self.dist_matrix_**2
334 G_center = KernelCenterer().fit_transform(G)
335 evals = self.kernel_pca_.eigenvalues_
336 return np.sqrt(np.sum(G_center**2) - np.sum(evals**2)) / G.shape[0]

387 def transform(self, X):
388 """Transform X.
389
390 This is implemented by linking the points X into the graph of geodesic
391 distances of the training data. First the `n_neighbors` nearest
392 neighbors of X are found in the training data, and from these the
393 shortest geodesic distances from each point in X to each point in
394 the training data are computed in order to construct the kernel.
395 The embedding of X is the projection of this kernel onto the
396 embedding vectors of the training set.
397
398 Parameters
399 ----------
400 X : {array-like, sparse matrix}, shape (n_queries, n_features)
401 If neighbors_algorithm='precomputed', X is assumed to be a
402 distance matrix or a sparse graph of shape
403 (n_queries, n_samples_fit).
404
405 Returns
406 -------
407 X_new : array-like, shape (n_queries, n_components)
408 X transformed in the new space.
409 """
410 check_is_fitted(self)
411 if self.n_neighbors is not None:
412 distances, indices = self.nbrs_.kneighbors(X, return_distance=True)
413 else:
414 distances, indices = self.nbrs_.radius_neighbors(X, return_distance=True)
415
416 # Create the graph of shortest distances from X to
417 # training data via the nearest neighbors of X.
418 # This can be done as a single array operation, but it potentially
419 # takes a lot of memory. To avoid that, use a loop:
420
421 n_samples_fit = self.nbrs_.n_samples_fit_
422 n_queries = distances.shape[0]
423
424 if hasattr(X, "dtype") and X.dtype == np.float32:
425 dtype = np.float32
426 else:
427 dtype = np.float64
428
429 G_X = np.zeros((n_queries, n_samples_fit), dtype)
430 for i in range(n_queries):
431 G_X[i] = np.min(self.dist_matrix_[indices[i]] + distances[i][:, None], 0)
432
433 G_X **= 2
434 G_X *= -0.5
435
436 return self.kernel_pca_.transform(G_X)
