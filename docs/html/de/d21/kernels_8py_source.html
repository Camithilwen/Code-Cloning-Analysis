1"""A set of kernels that can be combined by operators and used in Gaussian processes."""

3# Kernels for Gaussian process regression and classification.
4#
5# The kernels in this module allow kernel-engineering, i.e., they can be
6# combined via the "+" and "*" operators or be exponentiated with a scalar
7# via "**". These sum and product expressions can also contain scalar values,
8# which are automatically converted to a constant kernel.
9#
10# All kernels allow (analytic) gradient-based hyperparameter optimization.
11# The space of hyperparameters can be specified by giving lower und upper
12# boundaries for the value of each hyperparameter (the search space is thus
13# rectangular). Instead of specifying bounds, hyperparameters can also be
14# declared to be "fixed", which causes these hyperparameters to be excluded from
15# optimization.
16

22# package.

52class Hyperparameter(
53 namedtuple(
54 "Hyperparameter", ("name", "value_type", "bounds", "n_elements", "fixed")
55 )
56):
57 """A kernel hyperparameter's specification in form of a namedtuple.
58
59 .. versionadded:: 0.18
60
61 Attributes
62 ----------
63 name : str
64 The name of the hyperparameter. Note that a kernel using a
65 hyperparameter with name "x" must have the attributes self.x and
66 self.x_bounds
67
68 value_type : str
69 The type of the hyperparameter. Currently, only "numeric"
70 hyperparameters are supported.
71
72 bounds : pair of floats >= 0 or "fixed"
73 The lower and upper bound on the parameter. If n_elements>1, a pair
74 of 1d array with n_elements each may be given alternatively. If
75 the string "fixed" is passed as bounds, the hyperparameter's value
76 cannot be changed.
77
78 n_elements : int, default=1
79 The number of elements of the hyperparameter value. Defaults to 1,
80 which corresponds to a scalar hyperparameter. n_elements > 1
81 corresponds to a hyperparameter which is vector-valued,
82 such as, e.g., anisotropic length-scales.
83
84 fixed : bool, default=None
85 Whether the value of this hyperparameter is fixed, i.e., cannot be
86 changed during hyperparameter tuning. If None is passed, the "fixed" is
87 derived based on the given bounds.
88
89 Examples
90 --------
91 >>> from sklearn.gaussian_process.kernels import ConstantKernel
92 >>> from sklearn.datasets import make_friedman2
93 >>> from sklearn.gaussian_process import GaussianProcessRegressor
94 >>> from sklearn.gaussian_process.kernels import Hyperparameter
95 >>> X, y = make_friedman2(n_samples=50, noise=0, random_state=0)
96 >>> kernel = ConstantKernel(constant_value=1.0,
97 ... constant_value_bounds=(0.0, 10.0))
98
99 We can access each hyperparameter:
100
101 >>> for hyperparameter in kernel.hyperparameters:
102 ... print(hyperparameter)
103 Hyperparameter(name='constant_value', value_type='numeric',
104 bounds=array([[ 0., 10.]]), n_elements=1, fixed=False)
105
106 >>> params = kernel.get_params()
107 >>> for key in sorted(params): print(f"{key} : {params[key]}")
108 constant_value : 1.0
109 constant_value_bounds : (0.0, 10.0)
110 """
111
112 # A raw namedtuple is very memory efficient as it packs the attributes
113 # in a struct to get rid of the __dict__ of attributes in particular it
114 # does not copy the string for the keys on each instance.
115 # By deriving a namedtuple class just to introduce the __init__ method we
116 # would also reintroduce the __dict__ on the instance. By telling the
117 # Python interpreter that this subclass uses static __slots__ instead of
118 # dynamic attributes. Furthermore we don't need any additional slot in the
119 # subclass so we set __slots__ to the empty tuple.
120 __slots__ = ()
121
122 def __new__(cls, name, value_type, bounds, n_elements=1, fixed=None):
123 if not isinstance(bounds, str) or bounds != "fixed":
124 bounds = np.atleast_2d(bounds)
125 if n_elements > 1: # vector-valued parameter
126 if bounds.shape[0] == 1:
127 bounds = np.repeat(bounds, n_elements, 0)
128 elif bounds.shape[0] != n_elements:
129 raise ValueError(
130 "Bounds on %s should have either 1 or "
131 "%d dimensions. Given are %d"
132 % (name, n_elements, bounds.shape[0])
133 )
134
135 if fixed is None:
136 fixed = isinstance(bounds, str) and bounds == "fixed"
137 return super().__new__(cls, name, value_type, bounds, n_elements, fixed)

142 return (

144 and self.value_type == other.value_type
145 and np.all(self.bounds == other.bounds)

178 def get_params(self, deep=True):
179 """Get parameters of this kernel.
180
181 Parameters
182 ----------
183 deep : bool, default=True
184 If True, will return the parameters for this estimator and
185 contained subobjects that are estimators.
186
187 Returns
188 -------
189 params : dict
190 Parameter names mapped to their values.
191 """
192 params = dict()
193
194 # introspect the constructor arguments to find the model parameters
195 # to represent
196 cls = self.__class__
197 init = getattr(cls.__init__, "deprecated_original", cls.__init__)
198 init_sign = signature(init)
199 args, varargs = [], []
200 for parameter in init_sign.parameters.values():
201 if parameter.kind != parameter.VAR_KEYWORD and parameter.name != "self":
202 args.append(parameter.name)
203 if parameter.kind == parameter.VAR_POSITIONAL:
204 varargs.append(parameter.name)
205
206 if len(varargs) != 0:
207 raise RuntimeError(
208 "scikit-learn kernels should always "
209 "specify their parameters in the signature"
210 " of their __init__ (no varargs)."
211 " %s doesn't follow this convention." % (cls,)
212 )
213 for arg in args:
214 params[arg] = getattr(self, arg)
215
216 return params

218 def set_params(self, **params):
219 """Set the parameters of this kernel.
220
221 The method works on simple kernels as well as on nested kernels.
222 The latter have parameters of the form ``<component>__<parameter>``
223 so that it's possible to update each component of a nested object.
224
225 Returns
226 -------
227 self
228 """
229 if not params:
230 # Simple optimisation to gain speed (inspect is slow)
231 return self
232 valid_params = self.get_params(deep=True)
233 for key, value in params.items():
234 split = key.split("__", 1)
235 if len(split) > 1:
236 # nested objects case
237 name, sub_name = split
238 if name not in valid_params:
239 raise ValueError(
240 "Invalid parameter %s for kernel %s. "
241 "Check the list of available parameters "
242 "with `kernel.get_params().keys()`." % (name, self)
243 )
244 sub_object = valid_params[name]
245 sub_object.set_params(**{sub_name: value})
246 else:
247 # simple objects case
248 if key not in valid_params:
249 raise ValueError(
250 "Invalid parameter %s for kernel %s. "
251 "Check the list of available parameters "
252 "with `kernel.get_params().keys()`."
253 % (key, self.__class__.__name__)
254 )
255 setattr(self, key, value)
256 return self

285 @property
286 def theta(self):
287 """Returns the (flattened, log-transformed) non-fixed hyperparameters.
288
289 Note that theta are typically the log-transformed values of the
290 kernel's hyperparameters as this representation of the search space
291 is more amenable for hyperparameter search, as hyperparameters like
292 length-scales naturally live on a log-scale.
293
294 Returns
295 -------
296 theta : ndarray of shape (n_dims,)
297 The non-fixed, log-transformed hyperparameters of the kernel
298 """
299 theta = []
300 params = self.get_params()
301 for hyperparameter in self.hyperparameters:
302 if not hyperparameter.fixed:
303 theta.append(params[hyperparameter.name])
304 if len(theta) > 0:
305 return np.log(np.hstack(theta))
306 else:
307 return np.array([])

597 def __call__(self, X, Y=None, eval_gradient=False):
598 """Return the kernel k(X, Y) and optionally its gradient.
599
600 Note that this compound kernel returns the results of all simple kernel
601 stacked along an additional axis.
602
603 Parameters
604 ----------
605 X : array-like of shape (n_samples_X, n_features) or list of object, \
606 default=None
607 Left argument of the returned kernel k(X, Y)
608
609 Y : array-like of shape (n_samples_X, n_features) or list of object, \
610 default=None
611 Right argument of the returned kernel k(X, Y). If None, k(X, X)
612 is evaluated instead.
613

615 Determines whether the gradient with respect to the log of the
616 kernel hyperparameter is computed.

618 Returns
619 -------
620 K : ndarray of shape (n_samples_X, n_samples_Y, n_kernels)
621 Kernel k(X, Y)
622
623 K_gradient : ndarray of shape \
624 (n_samples_X, n_samples_X, n_dims, n_kernels), optional
625 The gradient of the kernel k(X, X) with respect to the log of the
626 hyperparameter of the kernel. Only returned when `eval_gradient`
627 is True.

629 if eval_gradient:
630 K = []

632 for kernel in self.kernels:
633 K_single, K_grad_single = kernel(X, Y, eval_gradient)
634 K.append(K_single)

637 else:
638 return np.dstack([kernel(X, Y, eval_gradient) for kernel in self.kernels])

1443class RBF(StationaryKernelMixin, NormalizedKernelMixin, Kernel):
1444 """Radial basis function kernel (aka squared-exponential kernel).
1445
1446 The RBF kernel is a stationary kernel. It is also known as the
1447 "squared exponential" kernel. It is parameterized by a length scale
1448 parameter :math:`l>0`, which can either be a scalar (isotropic variant
1449 of the kernel) or a vector with the same number of dimensions as the inputs
1450 X (anisotropic variant of the kernel). The kernel is given by:
1451
1452 .. math::
1453 k(x_i, x_j) = \\exp\\left(- \\frac{d(x_i, x_j)^2}{2l^2} \\right)
1454
1455 where :math:`l` is the length scale of the kernel and
1456 :math:`d(\\cdot,\\cdot)` is the Euclidean distance.
1457 For advice on how to set the length scale parameter, see e.g. [1]_.
1458
1459 This kernel is infinitely differentiable, which implies that GPs with this
1460 kernel as covariance function have mean square derivatives of all orders,
1461 and are thus very smooth.
1462 See [2]_, Chapter 4, Section 4.2, for further details of the RBF kernel.
1463
1464 Read more in the :ref:`User Guide <gp_kernels>`.
1465
1466 .. versionadded:: 0.18
1467
1468 Parameters
1469 ----------
1470 length_scale : float or ndarray of shape (n_features,), default=1.0
1471 The length scale of the kernel. If a float, an isotropic kernel is
1472 used. If an array, an anisotropic kernel is used where each dimension
1473 of l defines the length-scale of the respective feature dimension.
1474
1475 length_scale_bounds : pair of floats >= 0 or "fixed", default=(1e-5, 1e5)
1476 The lower and upper bound on 'length_scale'.
1477 If set to "fixed", 'length_scale' cannot be changed during
1478 hyperparameter tuning.
1479
1480 References
1481 ----------
1482 .. [1] `David Duvenaud (2014). "The Kernel Cookbook:
1483 Advice on Covariance functions".
1484 <https://www.cs.toronto.edu/~duvenaud/cookbook/>`_
1485
1486 .. [2] `Carl Edward Rasmussen, Christopher K. I. Williams (2006).
1487 "Gaussian Processes for Machine Learning". The MIT Press.
1488 <http://www.gaussianprocess.org/gpml/>`_
1489
1490 Examples
1491 --------
1492 >>> from sklearn.datasets import load_iris

1596class Matern(RBF):
1597 """Matern kernel.
1598
1599 The class of Matern kernels is a generalization of the :class:`RBF`.
1600 It has an additional parameter :math:`\\nu` which controls the
1601 smoothness of the resulting function. The smaller :math:`\\nu`,
1602 the less smooth the approximated function is.
1603 As :math:`\\nu\\rightarrow\\infty`, the kernel becomes equivalent to
1604 the :class:`RBF` kernel. When :math:`\\nu = 1/2`, the Matérn kernel
1605 becomes identical to the absolute exponential kernel.
1606 Important intermediate values are
1607 :math:`\\nu=1.5` (once differentiable functions)
1608 and :math:`\\nu=2.5` (twice differentiable functions).
1609
1610 The kernel is given by:
1611
1612 .. math::
1613 k(x_i, x_j) = \\frac{1}{\\Gamma(\\nu)2^{\\nu-1}}\\Bigg(
1614 \\frac{\\sqrt{2\\nu}}{l} d(x_i , x_j )
1615 \\Bigg)^\\nu K_\\nu\\Bigg(
1616 \\frac{\\sqrt{2\\nu}}{l} d(x_i , x_j )\\Bigg)
1617
1618
1619
1620 where :math:`d(\\cdot,\\cdot)` is the Euclidean distance,
1621 :math:`K_{\\nu}(\\cdot)` is a modified Bessel function and
1622 :math:`\\Gamma(\\cdot)` is the gamma function.
1623 See [1]_, Chapter 4, Section 4.2, for details regarding the different
1624 variants of the Matern kernel.
1625
1626 Read more in the :ref:`User Guide <gp_kernels>`.
1627
1628 .. versionadded:: 0.18
1629
1630 Parameters
1631 ----------
1632 length_scale : float or ndarray of shape (n_features,), default=1.0
1633 The length scale of the kernel. If a float, an isotropic kernel is
1634 used. If an array, an anisotropic kernel is used where each dimension
1635 of l defines the length-scale of the respective feature dimension.
1636
1637 length_scale_bounds : pair of floats >= 0 or "fixed", default=(1e-5, 1e5)
1638 The lower and upper bound on 'length_scale'.
1639 If set to "fixed", 'length_scale' cannot be changed during
1640 hyperparameter tuning.
1641
1642 nu : float, default=1.5
1643 The parameter nu controlling the smoothness of the learned function.
1644 The smaller nu, the less smooth the approximated function is.
1645 For nu=inf, the kernel becomes equivalent to the RBF kernel and for
1646 nu=0.5 to the absolute exponential kernel. Important intermediate
1647 values are nu=1.5 (once differentiable functions) and nu=2.5
1648 (twice differentiable functions). Note that values of nu not in
1649 [0.5, 1.5, 2.5, inf] incur a considerably higher computational cost
1650 (appr. 10 times higher) since they require to evaluate the modified
1651 Bessel function. Furthermore, in contrast to l, nu is kept fixed to
1652 its initial value and not optimized.
1653
1654 References
1655 ----------
1656 .. [1] `Carl Edward Rasmussen, Christopher K. I. Williams (2006).
1657 "Gaussian Processes for Machine Learning". The MIT Press.
1658 <http://www.gaussianprocess.org/gpml/>`_
1659
1660 Examples
1661 --------
1662 >>> from sklearn.datasets import load_iris
1663 >>> from sklearn.gaussian_process import GaussianProcessClassifier
1664 >>> from sklearn.gaussian_process.kernels import Matern
1665 >>> X, y = load_iris(return_X_y=True)
1666 >>> kernel = 1.0 * Matern(length_scale=1.0, nu=1.5)
1667 >>> gpc = GaussianProcessClassifier(kernel=kernel,
1668 ... random_state=0).fit(X, y)
1669 >>> gpc.score(X, y)
1670 0.9866
1671 >>> gpc.predict_proba(X[:2,:])
1672 array([[0.8513, 0.0368, 0.1117],
1673 [0.8086, 0.0693, 0.1220]])
1674 """
1675
1676 def __init__(self, length_scale=1.0, length_scale_bounds=(1e-5, 1e5), nu=1.5):
1677 super().__init__(length_scale, length_scale_bounds)
1678 self.nu = nu
1679
1680 def __call__(self, X, Y=None, eval_gradient=False):
1681 """Return the kernel k(X, Y) and optionally its gradient.
1682
1683 Parameters
1684 ----------
1685 X : ndarray of shape (n_samples_X, n_features)
1686 Left argument of the returned kernel k(X, Y)
1687
1688 Y : ndarray of shape (n_samples_Y, n_features), default=None
1689 Right argument of the returned kernel k(X, Y). If None, k(X, X)
1690 if evaluated instead.
1691
1692 eval_gradient : bool, default=False
1693 Determines whether the gradient with respect to the log of
1694 the kernel hyperparameter is computed.
1695 Only supported when Y is None.
1696
1697 Returns
1698 -------
1699 K : ndarray of shape (n_samples_X, n_samples_Y)
1700 Kernel k(X, Y)
1701
1702 K_gradient : ndarray of shape (n_samples_X, n_samples_X, n_dims), \
1703 optional
1704 The gradient of the kernel k(X, X) with respect to the log of the
1705 hyperparameter of the kernel. Only returned when `eval_gradient`
1706 is True.
1707 """
1708 X = np.atleast_2d(X)
1709 length_scale = _check_length_scale(X, self.length_scale)
1710 if Y is None:
1711 dists = pdist(X / length_scale, metric="euclidean")
1712 else:
1713 if eval_gradient:
1714 raise ValueError("Gradient can only be evaluated when Y is None.")
1715 dists = cdist(X / length_scale, Y / length_scale, metric="euclidean")
1716
1717 if self.nu == 0.5:
1718 K = np.exp(-dists)
1719 elif self.nu == 1.5:
1720 K = dists * math.sqrt(3)
1721 K = (1.0 + K) * np.exp(-K)
1722 elif self.nu == 2.5:
1723 K = dists * math.sqrt(5)
1724 K = (1.0 + K + K**2 / 3.0) * np.exp(-K)
1725 elif self.nu == np.inf:
1726 K = np.exp(-(dists**2) / 2.0)
1727 else: # general case; expensive to evaluate
1728 K = dists
1729 K[K == 0.0] += np.finfo(float).eps # strict zeros result in nan
1730 tmp = math.sqrt(2 * self.nu) * K
1731 K.fill((2 ** (1.0 - self.nu)) / gamma(self.nu))
1732 K *= tmp**self.nu
1733 K *= kv(self.nu, tmp)
1734
1735 if Y is None:
1736 # convert from upper-triangular matrix to square matrix
1737 K = squareform(K)
1738 np.fill_diagonal(K, 1)
1739
1740 if eval_gradient:
1741 if self.hyperparameter_length_scale.fixed:
1742 # Hyperparameter l kept fixed
1743 K_gradient = np.empty((X.shape[0], X.shape[0], 0))
1744 return K, K_gradient
1745
1746 # We need to recompute the pairwise dimension-wise distances
1747 if self.anisotropic:
1748 D = (X[:, np.newaxis, :] - X[np.newaxis, :, :]) ** 2 / (length_scale**2)
1749 else:
1750 D = squareform(dists**2)[:, :, np.newaxis]
1751
1752 if self.nu == 0.5:
1753 denominator = np.sqrt(D.sum(axis=2))[:, :, np.newaxis]
1754 divide_result = np.zeros_like(D)
1755 np.divide(
1756 D,
1757 denominator,
1758 out=divide_result,
1759 where=denominator != 0,
1760 )
1761 K_gradient = K[..., np.newaxis] * divide_result
1762 elif self.nu == 1.5:
1763 K_gradient = 3 * D * np.exp(-np.sqrt(3 * D.sum(-1)))[..., np.newaxis]
1764 elif self.nu == 2.5:
1765 tmp = np.sqrt(5 * D.sum(-1))[..., np.newaxis]
1766 K_gradient = 5.0 / 3.0 * D * (tmp + 1) * np.exp(-tmp)
1767 elif self.nu == np.inf:
1768 K_gradient = D * K[..., np.newaxis]
1769 else:
1770 # approximate gradient numerically
1771 def f(theta): # helper function
1772 return self.clone_with_theta(theta)(X, Y)
1773
1774 return K, _approx_fprime(self.theta, f, 1e-10)
1775
1776 if not self.anisotropic:
1777 return K, K_gradient[:, :].sum(-1)[:, :, np.newaxis]
1778 else:
1779 return K, K_gradient
1780 else:
1781 return K
1782
1783 def __repr__(self):
1784 if self.anisotropic:
1785 return "{0}(length_scale=[{1}], nu={2:.3g})".format(
1786 self.__class__.__name__,
1787 ", ".join(map("{0:.3g}".format, self.length_scale)),
1788 self.nu,
1789 )
1790 else:
1791 return "{0}(length_scale={1:.3g}, nu={2:.3g})".format(
1792 self.__class__.__name__, np.ravel(self.length_scale)[0], self.nu

2035

2037 Right argument of the returned kernel k(X, Y). If None, k(X, X)
2038 if evaluated instead.
2039
2040 eval_gradient : bool, default=False
2041 Determines whether the gradient with respect to the log of
2042 the kernel hyperparameter is computed.

2102 of :math:`x_d (d = 1, . . . , D)` and a prior of :math:`N(0, \sigma_0^2)`
2103 on the bias. The DotProduct kernel is invariant to a rotation of
2104 the coordinates about the origin, but not translations.
2105 It is parameterized by a parameter sigma_0 :math:`\sigma`
2106 which controls the inhomogenity of the kernel. For :math:`\sigma_0^2 =0`,
2107 the kernel is called the homogeneous linear kernel, otherwise
2108 it is inhomogeneous. The kernel is given by
2109
2110 .. math::
2111 k(x_i, x_j) = \sigma_0 ^ 2 + x_i \cdot x_j
2112
2113 The DotProduct kernel is commonly combined with exponentiation.
2114
2115 See [1]_, Chapter 4, Section 4.2, for further details regarding the
2116 DotProduct kernel.

2118 Read more in the :ref:`User Guide <gp_kernels>`.
