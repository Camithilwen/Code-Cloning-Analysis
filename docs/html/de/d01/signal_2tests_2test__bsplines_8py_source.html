69 # Verify with theoretical results at integer points up to order 5
70 with suppress_warnings() as sup:
71 sup.filter(DeprecationWarning)

184def test_sepfir2d_invalid_filter():
185 filt = np.array([1.0, 2.0, 4.0, 2.0, 1.0])
186 image = np.random.rand(7, 9)
187 # No error for odd lengths
188 signal.sepfir2d(image, filt, filt[2:])
189
190 # Row or column filter must be odd
191 with pytest.raises(ValueError, match="odd length"):
192 signal.sepfir2d(image, filt, filt[1:])
193 with pytest.raises(ValueError, match="odd length"):
194 signal.sepfir2d(image, filt[1:], filt)
195
196 # Filters must be 1-dimensional
197 with pytest.raises(ValueError, match="object too deep"):
198 signal.sepfir2d(image, filt.reshape(1, -1), filt)
199 with pytest.raises(ValueError, match="object too deep"):
200 signal.sepfir2d(image, filt, filt.reshape(1, -1))
