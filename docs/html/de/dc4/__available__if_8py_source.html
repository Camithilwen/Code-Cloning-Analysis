8class _AvailableIfDescriptor:
9 """Implements a conditional property using the descriptor protocol.
10
11 Using this class to create a decorator will raise an ``AttributeError``
12 if check(self) returns a falsey value. Note that if check raises an error
13 this will also result in hasattr returning false.
14
15 See https://docs.python.org/3/howto/descriptor.html for an explanation of
16 descriptors.
17 """
18
19 def __init__(self, fn, check, attribute_name):
20 self.fn = fn
21 self.check = check
22 self.attribute_name = attribute_name
23
24 # update the docstring of the descriptor
25 update_wrapper(self, fn)
26
27 def _check(self, obj, owner):
28 attr_err_msg = (
29 f"This {owner.__name__!r} has no attribute {self.attribute_name!r}"
30 )
31 try:
32 check_result = self.check(obj)
33 except Exception as e:
34 raise AttributeError(attr_err_msg) from e
35
36 if not check_result:
37 raise AttributeError(attr_err_msg)
38
39 def __get__(self, obj, owner=None):
40 if obj is not None:
41 # delegate only on instances, not the classes.
42 # this is to allow access to the docstrings.
43 self._check(obj, owner=owner)
44 out = MethodType(self.fn, obj)
45
46 else:
47 # This makes it possible to use the decorated method as an unbound method,
48 # for instance when monkeypatching.
49 @wraps(self.fn)
50 def out(*args, **kwargs):
51 self._check(args[0], owner=owner)
52 return self.fn(*args, **kwargs)
53
54 return out

57def available_if(check):
58 """An attribute that is available only if check returns a truthy value.
59
60 Parameters
61 ----------
62 check : callable
63 When passed the object with the decorated method, this should return
64 a truthy value if the attribute is available, and either return False
65 or raise an AttributeError if not available.
66
67 Returns
68 -------
69 callable
70 Callable makes the decorated method available if `check` returns
71 a truthy value, otherwise the decorated method is unavailable.
72
73 Examples
74 --------
75 >>> from sklearn.utils.metaestimators import available_if
76 >>> class HelloIfEven:
77 ... def __init__(self, x):
78 ... self.x = x
79 ...
80 ... def _x_is_even(self):
81 ... return self.x % 2 == 0
82 ...
83 ... @available_if(_x_is_even)
84 ... def say_hello(self):
85 ... print("Hello")
86 ...
87 >>> obj = HelloIfEven(1)
88 >>> hasattr(obj, "say_hello")
89 False
90 >>> obj.x = 2
91 >>> hasattr(obj, "say_hello")
92 True
93 >>> obj.say_hello()
94 Hello
95 """
