25def _log_dirichlet_norm(dirichlet_concentration):
26 """Compute the log of the Dirichlet distribution normalization term.
27
28 Parameters
29 ----------
30 dirichlet_concentration : array-like of shape (n_samples,)
31 The parameters values of the Dirichlet distribution.
32
33 Returns
34 -------
35 log_dirichlet_norm : float
36 The log normalization of the Dirichlet distribution.
37 """
38 return gammaln(np.sum(dirichlet_concentration)) - np.sum(
39 gammaln(dirichlet_concentration)
40 )

43def _log_wishart_norm(degrees_of_freedom, log_det_precisions_chol, n_features):
44 """Compute the log of the Wishart distribution normalization term.
45
46 Parameters
47 ----------
48 degrees_of_freedom : array-like of shape (n_components,)
49 The number of degrees of freedom on the covariance Wishart
50 distributions.
51
52 log_det_precision_chol : array-like of shape (n_components,)
53 The determinant of the precision matrix for each component.
54
55 n_features : int
56 The number of features.
57
58 Return
59 ------
60 log_wishart_norm : array-like of shape (n_components,)
61 The log normalization of the Wishart distribution.
62 """
63 # To simplify the computation we have removed the np.log(np.pi) term
64 return -(
65 degrees_of_freedom * log_det_precisions_chol
66 + degrees_of_freedom * n_features * 0.5 * math.log(2.0)
67 + np.sum(
68 gammaln(0.5 * (degrees_of_freedom - np.arange(n_features)[:, np.newaxis])),
69 0,
70 )
71 )

74class BayesianGaussianMixture(BaseMixture):
75 """Variational Bayesian estimation of a Gaussian mixture.
76
77 This class allows to infer an approximate posterior distribution over the
78 parameters of a Gaussian mixture distribution. The effective number of
79 components can be inferred from the data.
80
81 This class implements two types of prior for the weights distribution: a
82 finite mixture model with Dirichlet distribution and an infinite mixture
83 model with the Dirichlet Process. In practice Dirichlet Process inference
84 algorithm is approximated and uses a truncated distribution with a fixed
85 maximum number of components (called the Stick-breaking representation).
86 The number of components actually used almost always depends on the data.
87
88 .. versionadded:: 0.18
89
90 Read more in the :ref:`User Guide <bgmm>`.
91
92 Parameters
93 ----------
94 n_components : int, default=1
95 The number of mixture components. Depending on the data and the value
96 of the `weight_concentration_prior` the model can decide to not use
97 all the components by setting some component `weights_` to values very
98 close to zero. The number of effective components is therefore smaller
99 than n_components.
100
101 covariance_type : {'full', 'tied', 'diag', 'spherical'}, default='full'
102 String describing the type of covariance parameters to use.
103 Must be one of:
104
105 - 'full' (each component has its own general covariance matrix),
106 - 'tied' (all components share the same general covariance matrix),
107 - 'diag' (each component has its own diagonal covariance matrix),
108 - 'spherical' (each component has its own single variance).
109
110 tol : float, default=1e-3
111 The convergence threshold. EM iterations will stop when the
112 lower bound average gain on the likelihood (of the training data with
113 respect to the model) is below this threshold.
114
115 reg_covar : float, default=1e-6
116 Non-negative regularization added to the diagonal of covariance.
117 Allows to assure that the covariance matrices are all positive.
118
119 max_iter : int, default=100
120 The number of EM iterations to perform.
121
122 n_init : int, default=1
123 The number of initializations to perform. The result with the highest
124 lower bound value on the likelihood is kept.
125
126 init_params : {'kmeans', 'k-means++', 'random', 'random_from_data'}, \
127 default='kmeans'
128 The method used to initialize the weights, the means and the
129 covariances. String must be one of:
130
131 - 'kmeans': responsibilities are initialized using kmeans.
132 - 'k-means++': use the k-means++ method to initialize.
133 - 'random': responsibilities are initialized randomly.
134 - 'random_from_data': initial means are randomly selected data points.
135
136 .. versionchanged:: v1.1
137 `init_params` now accepts 'random_from_data' and 'k-means++' as
138 initialization methods.
139
140 weight_concentration_prior_type : {'dirichlet_process', 'dirichlet_distribution'}, \
141 default='dirichlet_process'
142 String describing the type of the weight concentration prior.
143
144 weight_concentration_prior : float or None, default=None
145 The dirichlet concentration of each component on the weight
146 distribution (Dirichlet). This is commonly called gamma in the
147 literature. The higher concentration puts more mass in
148 the center and will lead to more components being active, while a lower
149 concentration parameter will lead to more mass at the edge of the
150 mixture weights simplex. The value of the parameter must be greater
151 than 0. If it is None, it's set to ``1. / n_components``.
152
153 mean_precision_prior : float or None, default=None
154 The precision prior on the mean distribution (Gaussian).
155 Controls the extent of where means can be placed. Larger
156 values concentrate the cluster means around `mean_prior`.
157 The value of the parameter must be greater than 0.
158 If it is None, it is set to 1.
159
160 mean_prior : array-like, shape (n_features,), default=None
161 The prior on the mean distribution (Gaussian).
162 If it is None, it is set to the mean of X.
163
164 degrees_of_freedom_prior : float or None, default=None
165 The prior of the number of degrees of freedom on the covariance
166 distributions (Wishart). If it is None, it's set to `n_features`.
167
168 covariance_prior : float or array-like, default=None
169 The prior on the covariance distribution (Wishart).
170 If it is None, the emiprical covariance prior is initialized using the
171 covariance of X. The shape depends on `covariance_type`::
172
173 (n_features, n_features) if 'full',
174 (n_features, n_features) if 'tied',
175 (n_features) if 'diag',
176 float if 'spherical'
177
178 random_state : int, RandomState instance or None, default=None
179 Controls the random seed given to the method chosen to initialize the
180 parameters (see `init_params`).
181 In addition, it controls the generation of random samples from the
182 fitted distribution (see the method `sample`).
183 Pass an int for reproducible output across multiple function calls.
184 See :term:`Glossary <random_state>`.
185
186 warm_start : bool, default=False
187 If 'warm_start' is True, the solution of the last fitting is used as
188 initialization for the next call of fit(). This can speed up
189 convergence when fit is called several times on similar problems.
190 See :term:`the Glossary <warm_start>`.
191
192 verbose : int, default=0
193 Enable verbose output. If 1 then it prints the current
194 initialization and each iteration step. If greater than 1 then
195 it prints also the log probability and the time needed
196 for each step.
197
198 verbose_interval : int, default=10
199 Number of iteration done before the next print.
200
201 Attributes
202 ----------
203 weights_ : array-like of shape (n_components,)
204 The weights of each mixture components.
205
206 means_ : array-like of shape (n_components, n_features)
207 The mean of each mixture component.
208
209 covariances_ : array-like
210 The covariance of each mixture component.
211 The shape depends on `covariance_type`::
212
213 (n_components,) if 'spherical',
214 (n_features, n_features) if 'tied',
215 (n_components, n_features) if 'diag',
216 (n_components, n_features, n_features) if 'full'
217
218 precisions_ : array-like
219 The precision matrices for each component in the mixture. A precision
220 matrix is the inverse of a covariance matrix. A covariance matrix is
221 symmetric positive definite so the mixture of Gaussian can be
222 equivalently parameterized by the precision matrices. Storing the
223 precision matrices instead of the covariance matrices makes it more
224 efficient to compute the log-likelihood of new samples at test time.
225 The shape depends on ``covariance_type``::
226
227 (n_components,) if 'spherical',
228 (n_features, n_features) if 'tied',
229 (n_components, n_features) if 'diag',
230 (n_components, n_features, n_features) if 'full'
231
232 precisions_cholesky_ : array-like
233 The cholesky decomposition of the precision matrices of each mixture
234 component. A precision matrix is the inverse of a covariance matrix.
235 A covariance matrix is symmetric positive definite so the mixture of
236 Gaussian can be equivalently parameterized by the precision matrices.
237 Storing the precision matrices instead of the covariance matrices makes
238 it more efficient to compute the log-likelihood of new samples at test
239 time. The shape depends on ``covariance_type``::
240
241 (n_components,) if 'spherical',
242 (n_features, n_features) if 'tied',
243 (n_components, n_features) if 'diag',
244 (n_components, n_features, n_features) if 'full'
245
246 converged_ : bool
247 True when convergence of the best fit of inference was reached, False otherwise.
248
249 n_iter_ : int
250 Number of step used by the best fit of inference to reach the
251 convergence.
252
253 lower_bound_ : float
254 Lower bound value on the model evidence (of the training data) of the
255 best fit of inference.
256
257 lower_bounds_ : array-like of shape (`n_iter_`,)
258 The list of lower bound values on the model evidence from each iteration
259 of the best fit of inference.
260
261 weight_concentration_prior_ : tuple or float
262 The dirichlet concentration of each component on the weight
263 distribution (Dirichlet). The type depends on
264 ``weight_concentration_prior_type``::
265
266 (float, float) if 'dirichlet_process' (Beta parameters),
267 float if 'dirichlet_distribution' (Dirichlet parameters).
268
269 The higher concentration puts more mass in
270 the center and will lead to more components being active, while a lower
271 concentration parameter will lead to more mass at the edge of the
272 simplex.
273
274 weight_concentration_ : array-like of shape (n_components,)
275 The dirichlet concentration of each component on the weight
276 distribution (Dirichlet).
277
278 mean_precision_prior_ : float
279 The precision prior on the mean distribution (Gaussian).
280 Controls the extent of where means can be placed.
281 Larger values concentrate the cluster means around `mean_prior`.
282 If mean_precision_prior is set to None, `mean_precision_prior_` is set
283 to 1.
284
285 mean_precision_ : array-like of shape (n_components,)
286 The precision of each components on the mean distribution (Gaussian).
287
288 mean_prior_ : array-like of shape (n_features,)
289 The prior on the mean distribution (Gaussian).
290
291 degrees_of_freedom_prior_ : float
292 The prior of the number of degrees of freedom on the covariance
293 distributions (Wishart).
294
295 degrees_of_freedom_ : array-like of shape (n_components,)
296 The number of degrees of freedom of each components in the model.
297
298 covariance_prior_ : float or array-like
299 The prior on the covariance distribution (Wishart).
300 The shape depends on `covariance_type`::
301
302 (n_features, n_features) if 'full',
303 (n_features, n_features) if 'tied',
304 (n_features) if 'diag',
305 float if 'spherical'
306
307 n_features_in_ : int
308 Number of features seen during :term:`fit`.
309
310 .. versionadded:: 0.24
311
312 feature_names_in_ : ndarray of shape (`n_features_in_`,)
313 Names of features seen during :term:`fit`. Defined only when `X`
314 has feature names that are all strings.
315
316 .. versionadded:: 1.0
317

474 def _checkcovariance_prior_parameter(self, X):
475 """Check the `covariance_prior_`.
476
477 Parameters
478 ----------
479 X : array-like of shape (n_samples, n_features)
480 """
481 _, n_features = X.shape
482
483 if self.covariance_prior is None:
484 self.covariance_prior_ = {
485 "full": np.atleast_2d(np.cov(X.T)),
486 "tied": np.atleast_2d(np.cov(X.T)),
487 "diag": np.var(X, axis=0, ddof=1),
488 "spherical": np.var(X, axis=0, ddof=1).mean(),
489 }[self.covariance_type]
490
491 elif self.covariance_type in ["full", "tied"]:
492 self.covariance_prior_ = check_array(
493 self.covariance_prior, dtype=[np.float64, np.float32], ensure_2d=False

496 self.covariance_prior_,
497 (n_features, n_features),
498 "%s covariance_prior" % self.covariance_type,
499 )
500 _check_precision_matrix(self.covariance_prior_, self.covariance_type)
501 elif self.covariance_type == "diag":
502 self.covariance_prior_ = check_array(
503 self.covariance_prior, dtype=[np.float64, np.float32], ensure_2d=False

505 _check_shape(

508 "%s covariance_prior" % self.covariance_type,

510 _check_precision_positivity(self.covariance_prior_, self.covariance_type)

513 self.covariance_prior_ = self.covariance_prior

532 def _estimate_weights(self, nk):
533 """Estimate the parameters of the Dirichlet distribution.
534
535 Parameters
536 ----------
537 nk : array-like of shape (n_components,)
538 """
539 if self.weight_concentration_prior_type == "dirichlet_process":
540 # For dirichlet process weight_concentration will be a tuple
541 # containing the two parameters of the beta distribution
542 self.weight_concentration_ = (
543 1.0 + nk,
544 (
545 self.weight_concentration_prior_
546 + np.hstack((np.cumsum(nk[::-1])[-2::-1], 0))
547 ),
548 )
549 else:
550 # case Variational Gaussian mixture with dirichlet distribution
551 self.weight_concentration_ = self.weight_concentration_prior_ + nk
552
553 def _estimate_means(self, nk, xk):
554 """Estimate the parameters of the Gaussian distribution.
555
556 Parameters
557 ----------
558 nk : array-like of shape (n_components,)
559
560 xk : array-like of shape (n_components, n_features)
561 """
562 self.mean_precision_ = self.mean_precision_prior_ + nk
563 self.means_ = (
564 self.mean_precision_prior_ * self.mean_prior_ + nk[:, np.newaxis] * xk
565 ) / self.mean_precision_[:, np.newaxis]

567 def _estimate_precisions(self, nk, xk, sk):
568 """Estimate the precisions parameters of the precision distribution.
569
570 Parameters
571 ----------
572 nk : array-like of shape (n_components,)
573
574 xk : array-like of shape (n_components, n_features)
575
576 sk : array-like
577 The shape depends of `covariance_type`:
578 'full' : (n_components, n_features, n_features)
579 'tied' : (n_features, n_features)
580 'diag' : (n_components, n_features)
581 'spherical' : (n_components,)
582 """
583 {
584 "full": self._estimate_wishart_full,
585 "tied": self._estimate_wishart_tied,
586 "diag": self._estimate_wishart_diag,
587 "spherical": self._estimate_wishart_spherical,
588 }[self.covariance_type](nk, xk, sk)
589
590 self.precisions_cholesky_ = _compute_precision_cholesky(
591 self.covariances_, self.covariance_type
592 )

594 def _estimate_wishart_full(self, nk, xk, sk):
595 """Estimate the full Wishart distribution parameters.
596
597 Parameters
598 ----------
599 X : array-like of shape (n_samples, n_features)
600
601 nk : array-like of shape (n_components,)
602
603 xk : array-like of shape (n_components, n_features)
604
605 sk : array-like of shape (n_components, n_features, n_features)
606 """
607 _, n_features = xk.shape
608
609 # Warning : in some Bishop book, there is a typo on the formula 10.63
610 # `degrees_of_freedom_k = degrees_of_freedom_0 + Nk` is
611 # the correct formula
612 self.degrees_of_freedom_ = self.degrees_of_freedom_prior_ + nk
613
614 self.covariances_ = np.empty((self.n_components, n_features, n_features))
615
616 for k in range(self.n_components):
617 diff = xk[k] - self.mean_prior_
618 self.covariances_[k] = (
619 self.covariance_prior_
620 + nk[k] * sk[k]
621 + nk[k]
622 * self.mean_precision_prior_
623 / self.mean_precision_[k]
624 * np.outer(diff, diff)
625 )
626
627 # Contrary to the original bishop book, we normalize the covariances
628 self.covariances_ /= self.degrees_of_freedom_[:, np.newaxis, np.newaxis]
629
630 def _estimate_wishart_tied(self, nk, xk, sk):
631 """Estimate the tied Wishart distribution parameters.
632
633 Parameters
634 ----------
635 X : array-like of shape (n_samples, n_features)
636
637 nk : array-like of shape (n_components,)
638
639 xk : array-like of shape (n_components, n_features)
640
641 sk : array-like of shape (n_features, n_features)
642 """
643 _, n_features = xk.shape
644
645 # Warning : in some Bishop book, there is a typo on the formula 10.63
646 # `degrees_of_freedom_k = degrees_of_freedom_0 + Nk`
647 # is the correct formula
648 self.degrees_of_freedom_ = (
649 self.degrees_of_freedom_prior_ + nk.sum() / self.n_components
650 )
651
652 diff = xk - self.mean_prior_
653 self.covariances_ = (
654 self.covariance_prior_
655 + sk * nk.sum() / self.n_components
656 + self.mean_precision_prior_
657 / self.n_components
658 * np.dot((nk / self.mean_precision_) * diff.T, diff)
659 )
660
661 # Contrary to the original bishop book, we normalize the covariances
662 self.covariances_ /= self.degrees_of_freedom_

664 def _estimate_wishart_diag(self, nk, xk, sk):
665 """Estimate the diag Wishart distribution parameters.
666
667 Parameters
668 ----------
669 X : array-like of shape (n_samples, n_features)
670
671 nk : array-like of shape (n_components,)
672
673 xk : array-like of shape (n_components, n_features)
674
675 sk : array-like of shape (n_components, n_features)
676 """
677 _, n_features = xk.shape
678
679 # Warning : in some Bishop book, there is a typo on the formula 10.63
680 # `degrees_of_freedom_k = degrees_of_freedom_0 + Nk`
681 # is the correct formula
682 self.degrees_of_freedom_ = self.degrees_of_freedom_prior_ + nk
683
684 diff = xk - self.mean_prior_
685 self.covariances_ = self.covariance_prior_ + nk[:, np.newaxis] * (
686 sk
687 + (self.mean_precision_prior_ / self.mean_precision_)[:, np.newaxis]
688 * np.square(diff)
689 )
690
691 # Contrary to the original bishop book, we normalize the covariances
692 self.covariances_ /= self.degrees_of_freedom_[:, np.newaxis]

694 def _estimate_wishart_spherical(self, nk, xk, sk):
695 """Estimate the spherical Wishart distribution parameters.
696
697 Parameters
698 ----------
699 X : array-like of shape (n_samples, n_features)
700
701 nk : array-like of shape (n_components,)
702
703 xk : array-like of shape (n_components, n_features)
704
705 sk : array-like of shape (n_components,)
706 """
707 _, n_features = xk.shape
708
709 # Warning : in some Bishop book, there is a typo on the formula 10.63
710 # `degrees_of_freedom_k = degrees_of_freedom_0 + Nk`
711 # is the correct formula
712 self.degrees_of_freedom_ = self.degrees_of_freedom_prior_ + nk
713

715 self.covariances_ = self.covariance_prior_ + nk * (

717 + self.mean_precision_prior_
718 / self.mean_precision_
719 * np.mean(np.square(diff), 1)

745 def _estimate_log_weights(self):
746 if self.weight_concentration_prior_type == "dirichlet_process":
747 digamma_sum = digamma(
748 self.weight_concentration_[0] + self.weight_concentration_[1]
749 )
750 digamma_a = digamma(self.weight_concentration_[0])
751 digamma_b = digamma(self.weight_concentration_[1])

753 digamma_a
754 - digamma_sum
755 + np.hstack((0, np.cumsum(digamma_b - digamma_sum)[:-1]))

763 def _estimate_log_prob(self, X):
764 _, n_features = X.shape
765 # We remove `n_features * np.log(self.degrees_of_freedom_)` because
766 # the precision matrix is normalized
767 log_gauss = _estimate_log_gaussian_prob(
768 X, self.means_, self.precisions_cholesky_, self.covariance_type
769 ) - 0.5 * n_features * np.log(self.degrees_of_freedom_)

771 log_lambda = n_features * np.log(2.0) + np.sum(
772 digamma(
773 0.5
774 * (self.degrees_of_freedom_ - np.arange(0, n_features)[:, np.newaxis])
775 ),
776 0,

781 def _compute_lower_bound(self, log_resp, log_prob_norm):
782 """Estimate the lower bound of the model.
783

785 the model) is used to detect the convergence and has to increase at
786 each iteration.
787
788 Parameters
789 ----------

798
799 Returns
800 -------
801 lower_bound : float

804 # and removed all the constant terms.
805 (n_features,) = self.mean_prior_.shape

807 # We removed `.5 * n_features * np.log(self.degrees_of_freedom_)`
808 # because the precision matrix is normalized.
809 log_det_precisions_chol = _compute_log_det_cholesky(
810 self.precisions_cholesky_, self.covariance_type, n_features
811 ) - 0.5 * n_features * np.log(self.degrees_of_freedom_)

814 log_wishart = self.n_components * np.float64(
815 _log_wishart_norm(
816 self.degrees_of_freedom_, log_det_precisions_chol, n_features
817 )
818 )
819 else:

822 self.degrees_of_freedom_, log_det_precisions_chol, n_features

828 betaln(self.weight_concentration_[0], self.weight_concentration_[1])
829 )

833 return (

835 - log_wishart
836 - log_norm_weight
837 - 0.5 * n_features * np.sum(np.log(self.mean_precision_))
838 )
