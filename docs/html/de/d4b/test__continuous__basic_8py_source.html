389# mpmath with mpmath.mp.dps = 50 and output at 20:
390#
391# def ks(x, n):
392# x = mpmath.mpf(x)

401 [(2.0e-5, 1000000000,

411 # Regression test for gh-17775. In scipy 1.9.3 and earlier,
412 # these test would fail.
413 #
414 # KS one asymptotic sf ~ e^(-(6nx+1)^2 / 18n)
415 # Given a large 32-bit integer n, 6n will overflow in the c implementation.
416 # Example of broken behaviour:
417 # ksone.sf(2.0e-5, 1000000000) == 0.9374359693473666
