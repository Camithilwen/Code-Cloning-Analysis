52 def _get_pyarrow_options(self) -> None:
53 """
54 Rename some arguments to pass to pyarrow
55 """
56 mapping = {
57 "usecols": "include_columns",
58 "na_values": "null_values",
59 "escapechar": "escape_char",
60 "skip_blank_lines": "ignore_empty_lines",
61 "decimal": "decimal_point",
62 "quotechar": "quote_char",
63 }
64 for pandas_name, pyarrow_name in mapping.items():
65 if pandas_name in self.kwds and self.kwds.get(pandas_name) is not None:
66 self.kwds[pyarrow_name] = self.kwds.pop(pandas_name)
67
68 # Date format handling
69 # If we get a string, we need to convert it into a list for pyarrow
70 # If we get a dict, we want to parse those separately
71 date_format = self.date_format
72 if isinstance(date_format, str):
73 date_format = [date_format]
74 else:
75 # In case of dict, we don't want to propagate through, so
76 # just set to pyarrow default of None
77
78 # Ideally, in future we disable pyarrow dtype inference (read in as string)
79 # to prevent misreads.
80 date_format = None
81 self.kwds["timestamp_parsers"] = date_format
82
83 self.parse_options = {
84 option_name: option_value
85 for option_name, option_value in self.kwds.items()
86 if option_value is not None
87 and option_name
88 in ("delimiter", "quote_char", "escape_char", "ignore_empty_lines")
89 }
90
91 on_bad_lines = self.kwds.get("on_bad_lines")
92 if on_bad_lines is not None:
93 if callable(on_bad_lines):
94 self.parse_options["invalid_row_handler"] = on_bad_lines
95 elif on_bad_lines == ParserBase.BadLineHandleMethod.ERROR:
96 self.parse_options[
97 "invalid_row_handler"
98 ] = None # PyArrow raises an exception by default
99 elif on_bad_lines == ParserBase.BadLineHandleMethod.WARN:
100
101 def handle_warning(invalid_row) -> str:
102 warnings.warn(
103 f"Expected {invalid_row.expected_columns} columns, but found "
104 f"{invalid_row.actual_columns}: {invalid_row.text}",
105 ParserWarning,
106 stacklevel=find_stack_level(),
107 )
108 return "skip"
109
110 self.parse_options["invalid_row_handler"] = handle_warning
111 elif on_bad_lines == ParserBase.BadLineHandleMethod.SKIP:
112 self.parse_options["invalid_row_handler"] = lambda _: "skip"
113
114 self.convert_options = {
115 option_name: option_value
116 for option_name, option_value in self.kwds.items()
117 if option_value is not None
118 and option_name
119 in (
120 "include_columns",
121 "null_values",
122 "true_values",
123 "false_values",
124 "decimal_point",
125 "timestamp_parsers",
126 )
127 }
128 self.convert_options["strings_can_be_null"] = "" in self.kwds["null_values"]
129 # autogenerated column names are prefixed with 'f' in pyarrow.csv
130 if self.header is None and "include_columns" in self.convert_options:

132 f"f{n}" for n in self.convert_options["include_columns"]
133 ]
134
135 self.read_options = {
136 "autogenerate_column_names": self.header is None,
137 "skip_rows": self.header
138 if self.header is not None
139 else self.kwds["skiprows"],

143 def _finalize_pandas_output(self, frame: DataFrame) -> DataFrame:
144 """
145 Processes data read in based on kwargs.
146
147 Parameters
148 ----------
149 frame: DataFrame
150 The DataFrame to process.
151
152 Returns
153 -------
154 DataFrame
155 The processed DataFrame.
156 """
157 num_cols = len(frame.columns)
158 multi_index_named = True
159 if self.header is None:
160 if self.names is None:
161 if self.header is None:
162 self.names = range(num_cols)
163 if len(self.names) != num_cols:
164 # usecols is passed through to pyarrow, we only handle index col here
165 # The only way self.names is not the same length as number of cols is
166 # if we have int index_col. We should just pad the names(they will get
167 # removed anyways) to expected length then.
168 columns_prefix = [str(x) for x in range(num_cols - len(self.names))]
169 self.names = columns_prefix + self.names
170 multi_index_named = False
171 frame.columns = self.names
172 # we only need the frame not the names
173 _, frame = self._do_date_conversions(frame.columns, frame)
174 if self.index_col is not None:
175 index_to_set = self.index_col.copy()
176 for i, item in enumerate(self.index_col):
177 if is_integer(item):
178 index_to_set[i] = frame.columns[item]
179 # String case
180 elif item not in frame.columns:
181 raise ValueError(f"Index {item} invalid")
182
183 # Process dtype for index_col and drop from dtypes
184 if self.dtype is not None:
185 key, new_dtype = (
186 (item, self.dtype.get(item))
187 if self.dtype.get(item) is not None
188 else (frame.columns[item], self.dtype.get(frame.columns[item]))
189 )
190 if new_dtype is not None:
191 frame[key] = frame[key].astype(new_dtype)
192 del self.dtype[key]
193
194 frame.set_index(index_to_set, drop=True, inplace=True)
195 # Clear names if headerless and no name given
196 if self.header is None and not multi_index_named:
197 frame.index.names = [None] * len(frame.index.names)
198
199 if self.dtype is not None:
200 # Ignore non-existent columns from dtype mapping
201 # like other parsers do
202 if isinstance(self.dtype, dict):
203 self.dtype = {
204 k: pandas_dtype(v)
205 for k, v in self.dtype.items()
206 if k in frame.columns
207 }
208 else:
209 self.dtype = pandas_dtype(self.dtype)
210 try:
211 frame = frame.astype(self.dtype)
212 except TypeError as e:
213 # GH#44901 reraise to keep api consistent
214 raise ValueError(e)
215 return frame

228 def read(self) -> DataFrame:
229 """
230 Reads the contents of a CSV file into a DataFrame and
231 processes it according to the kwargs passed in the
232 constructor.
233
234 Returns
235 -------
236 DataFrame
237 The DataFrame created from the CSV file.
238 """
239 pa = import_optional_dependency("pyarrow")
240 pyarrow_csv = import_optional_dependency("pyarrow.csv")
241 self._get_pyarrow_options()
242
243 try:
244 convert_options = pyarrow_csv.ConvertOptions(**self.convert_options)
245 except TypeError:
246 include = self.convert_options.get("include_columns", None)
247 if include is not None:
248 self._validate_usecols(include)
249
250 nulls = self.convert_options.get("null_values", set())
251 if not lib.is_list_like(nulls) or not all(
252 isinstance(x, str) for x in nulls
253 ):
254 raise TypeError(
255 "The 'pyarrow' engine requires all na_values to be strings"
256 )
257
258 raise
259
260 try:
261 table = pyarrow_csv.read_csv(
262 self.src,
263 read_options=pyarrow_csv.ReadOptions(**self.read_options),
264 parse_options=pyarrow_csv.ParseOptions(**self.parse_options),
265 convert_options=convert_options,
266 )
267 except pa.ArrowInvalid as e:
268 raise ParserError(e) from e
269
270 dtype_backend = self.kwds["dtype_backend"]
271
272 # Convert all pa.null() cols -> float64 (non nullable)
273 # else Int64 (nullable case, see below)
274 if dtype_backend is lib.no_default:
275 new_schema = table.schema
276 new_type = pa.float64()
277 for i, arrow_type in enumerate(table.schema.types):
278 if pa.types.is_null(arrow_type):
279 new_schema = new_schema.set(
280 i, new_schema.field(i).with_type(new_type)
281 )
282
283 table = table.cast(new_schema)
284
285 with warnings.catch_warnings():
286 warnings.filterwarnings(
287 "ignore",
288 "make_block is deprecated",
289 DeprecationWarning,
290 )
291 frame = arrow_table_to_pandas(
292 table, dtype_backend=dtype_backend, null_to_int64=True
293 )
294
295 return self._finalize_pandas_output(frame)
