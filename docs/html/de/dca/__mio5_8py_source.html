9=================================
10 Note on functions and mat files
11=================================
12
13The document above does not give any hints as to the storage of matlab
14function handles, or anonymous function handles. I had, therefore, to
15guess the format of matlab arrays of ``mxFUNCTION_CLASS`` and
16``mxOPAQUE_CLASS`` by looking at example mat files.
17
18``mxFUNCTION_CLASS`` stores all types of matlab functions. It seems to
19contain a struct matrix with a set pattern of fields. For anonymous
20functions, a sub-fields of one of these fields seems to contain the
21well-named ``mxOPAQUE_CLASS``. This seems to contain:
22
23* array flags as for any matlab matrix
24* 3 int8 strings
25* a matrix
26
27It seems that whenever the mat file contains a ``mxOPAQUE_CLASS``
28instance, there is also an un-named matrix (name == '') at the end of
29the mat file. I'll call this the ``__function_workspace__`` matrix.
30
31When I saved two anonymous functions in a mat file, or appended another
32anonymous function to the mat file, there was still only one
33``__function_workspace__`` un-named matrix at the end, but larger than
34that for a mat file with a single anonymous function, suggesting that
35the workspaces for the two functions had been merged.
36
37The ``__function_workspace__`` matrix appears to be of double class
38(``mxCLASS_DOUBLE``), but stored as uint8, the memory for which is in
39the format of a mini .mat file, without the first 124 bytes of the file
40header (the description and the subsystem_offset), but with the version
41U2 bytes, and the S2 endian test bytes. There follow 4 zero bytes,
42presumably for 8 byte padding, and then a series of ``miMATRIX``
43entries, as in a standard mat file. The ``miMATRIX`` entries appear to
44be series of un-named (name == '') matrices, and may also contain arrays
45of this same mini-mat format.
46
47I guess that:
48
49* saving an anonymous function back to a mat file will need the
50 associated ``__function_workspace__`` matrix saved as well for the
51 anonymous function to work correctly.
52* appending to a mat file that has a ``__function_workspace__`` would
53 involve first pulling off this workspace, appending, checking whether
54 there were any more anonymous functions appended, and then somehow
55 merging the relevant workspaces, and saving at the end of the mat
56 file.
57
58The mat files I was playing with are in ``tests/data``:
59
60* sqr.mat
61* parabola.mat
62* some_functions.mat
63
64See ``tests/test_mio.py:test_mio_funcs.py`` for the debugging
65script I was working with.

104def _has_struct(elem):
105 """Determine if elem is an array and if first array item is a struct."""
106 return (isinstance(elem, np.ndarray) and (elem.size > 0) and (elem.ndim > 0) and
107 isinstance(elem[0], mat_struct))

110def _inspect_cell_array(ndarray):
111 """Construct lists from cell arrays (loaded as numpy ndarrays), recursing
112 into items if they contain mat_struct objects."""
113 elem_list = []
114 for sub_elem in ndarray:
115 if isinstance(sub_elem, mat_struct):
116 elem_list.append(_matstruct_to_dict(sub_elem))
117 elif _has_struct(sub_elem):
118 elem_list.append(_inspect_cell_array(sub_elem))
119 else:
120 elem_list.append(sub_elem)
121 return elem_list
122
123
124def _matstruct_to_dict(matobj):
125 """Construct nested dicts from mat_struct objects."""
126 d = {}
127 for f in matobj._fieldnames:
128 elem = matobj.__dict__[f]
129 if isinstance(elem, mat_struct):
130 d[f] = _matstruct_to_dict(elem)
131 elif _has_struct(elem):
132 d[f] = _inspect_cell_array(elem)
133 else:
134 d[f] = elem
135 return d

138def _simplify_cells(d):
139 """Convert mat objects in dict to nested dicts."""
140 for key in d:
141 if isinstance(d[key], mat_struct):
142 d[key] = _matstruct_to_dict(d[key])
143 elif _has_struct(d[key]):
144 d[key] = _inspect_cell_array(d[key])
145 return d

207 def guess_byte_order(self):
208 ''' Guess byte order.
209 Sets stream pointer to 0'''
210 self.mat_stream.seek(126)

215 def read_file_header(self):
216 ''' Read in mat 5 file header '''
217 hdict = {}
218 hdr_dtype = MDTYPES[self.byte_order]['dtypes']['file_header']
219 hdr = read_dtype(self.mat_stream, hdr_dtype)
220 hdict['__header__'] = hdr['description'].item().strip(b' \t\n\000')

238 def read_var_header(self):
239 ''' Read header, return header, next position
240
241 Header has to define at least .name and .is_global
242
243 Parameters
244 ----------
245 None
246
247 Returns
248 -------
249 header : object
250 object that can be passed to self.read_var_array, and that
251 has attributes .name and .is_global
252 next_position : int
253 position in stream of next variable
254 '''
255 mdtype, byte_count = self._file_reader.read_full_tag()
256 if not byte_count > 0:
257 raise ValueError("Did not read any bytes")
258 next_pos = self.mat_stream.tell() + byte_count
259 if mdtype == miCOMPRESSED:
260 # Make new stream from compressed data
261 stream = ZlibInputStream(self.mat_stream, byte_count)
262 self._matrix_reader.set_stream(stream)
263 check_stream_limit = self.verify_compressed_data_integrity
264 mdtype, byte_count = self._matrix_reader.read_full_tag()
265 else:
266 check_stream_limit = False
267 self._matrix_reader.set_stream(self.mat_stream)
268 if not mdtype == miMATRIX:
269 raise TypeError('Expecting miMATRIX type here, got %d' % mdtype)
270 header = self._matrix_reader.read_header(check_stream_limit)
271 return header, next_pos

292 def get_variables(self, variable_names=None):
293 ''' get variables from stream as dictionary
294
295 variable_names - optional list of variable names to get
296
297 If variable_names is None, then get all variables in file
298 '''
299 if isinstance(variable_names, str):
300 variable_names = [variable_names]
301 elif variable_names is not None:
302 variable_names = list(variable_names)
303
304 self.mat_stream.seek(0)
305 # Here we pass all the parameters in self to the reading objects
306 self.initialize_read()
307 mdict = self.read_file_header()
308 mdict['__globals__'] = []
309 while not self.end_of_stream():
310 hdr, next_position = self.read_var_header()
311 name = 'None' if hdr.name is None else hdr.name.decode('latin1')
312 if name in mdict:
313 warnings.warn('Duplicate variable name "%s" in stream'
314 ' - replacing previous with new\n'
315 'Consider mio5.varmats_from_mat to split '
316 'file into single variable files' % name,
317 MatReadWarning, stacklevel=2)
318 if name == '':
319 # can only be a matlab 7 function workspace
320 name = '__function_workspace__'
321 # We want to keep this raw because mat_dtype processing
322 # will break the format (uint8 as mxDOUBLE_CLASS)
323 process = False
324 else:
325 process = True
326 if variable_names is not None and name not in variable_names:
327 self.mat_stream.seek(next_position)
328 continue
329 try:
330 res = self.read_var_array(hdr, process)
331 except MatReadError as err:
332 warnings.warn(
333 'Unreadable variable "%s", because "%s"' %
334 (name, err),
335 Warning, stacklevel=2)
336 res = "Read error: %s" % err
337 self.mat_stream.seek(next_position)
338 mdict[name] = res
339 if hdr.is_global:
340 mdict['__globals__'].append(name)
341 if variable_names is not None:
342 variable_names.remove(name)
343 if len(variable_names) == 0:
344 break
345 if self.simplify_cells:
346 return _simplify_cells(mdict)
347 else:
348 return mdict

350 def list_variables(self):
351 ''' list variables from stream '''
352 self.mat_stream.seek(0)
353 # Here we pass all the parameters in self to the reading objects
354 self.initialize_read()
355 self.read_file_header()
356 vars = []
357 while not self.end_of_stream():
358 hdr, next_position = self.read_var_header()
359 name = 'None' if hdr.name is None else hdr.name.decode('latin1')
360 if name == '':
361 # can only be a matlab 7 function workspace
362 name = '__function_workspace__'
363
364 shape = self._matrix_reader.shape_from_header(hdr)
365 if hdr.is_logical:
366 info = 'logical'
367 else:
368 info = mclass_info.get(hdr.mclass, 'unknown')
369 vars.append((name, shape, info))
370
371 self.mat_stream.seek(next_position)
372 return vars

375def varmats_from_mat(file_obj):
376 """ Pull variables out of mat 5 file as a sequence of mat file objects
377
378 This can be useful with a difficult mat file, containing unreadable
379 variables. This routine pulls the variables out in raw form and puts them,
380 unread, back into a file stream for saving or reading. Another use is the
381 pathological case where there is more than one variable of the same name in
382 the file; this routine returns the duplicates, whereas the standard reader
383 will overwrite duplicates in the returned dictionary.
384
385 The file pointer in `file_obj` will be undefined. File pointers for the
386 returned file-like objects are set at 0.
387
388 Parameters
389 ----------
390 file_obj : file-like
391 file object containing mat file
392
393 Returns
394 -------
395 named_mats : list
396 list contains tuples of (name, BytesIO) where BytesIO is a file-like
397 object containing mat file contents as for a single variable. The
398 BytesIO contains a string with the original header and a single var. If
399 ``var_file_obj`` is an individual BytesIO instance, then save as a mat
400 file with something like ``open('test.mat',
401 'wb').write(var_file_obj.read())``
402
403 Examples
404 --------
405 >>> import scipy.io
406
407 BytesIO is from the ``io`` module in Python 3, and is ``cStringIO`` for
408 Python < 3.
409
410 >>> mat_fileobj = BytesIO()
411 >>> scipy.io.savemat(mat_fileobj, {'b': np.arange(10), 'a': 'a string'})
412 >>> varmats = varmats_from_mat(mat_fileobj)
413 >>> sorted([name for name, str_obj in varmats])
414 ['a', 'b']
415 """
416 rdr = MatFile5Reader(file_obj)
417 file_obj.seek(0)
418 # Raw read of top-level file header
419 hdr_len = MDTYPES[native_code]['dtypes']['file_header'].itemsize
420 raw_hdr = file_obj.read(hdr_len)
421 # Initialize variable reading
422 file_obj.seek(0)
423 rdr.initialize_read()
424 rdr.read_file_header()
425 next_position = file_obj.tell()
426 named_mats = []
427 while not rdr.end_of_stream():
428 start_position = next_position
429 hdr, next_position = rdr.read_var_header()
430 name = 'None' if hdr.name is None else hdr.name.decode('latin1')
431 # Read raw variable string
432 file_obj.seek(start_position)
433 byte_count = next_position - start_position
434 var_str = file_obj.read(byte_count)
435 # write to stringio object
436 out_obj = BytesIO()
437 out_obj.write(raw_hdr)
438 out_obj.write(var_str)
439 out_obj.seek(0)
440 named_mats.append((name, out_obj))
441 return named_mats

448def to_writeable(source):
449 ''' Convert input object ``source`` to something we can write
450
451 Parameters
452 ----------
453 source : object
454
455 Returns
456 -------
457 arr : None or ndarray or EmptyStructMarker
458 If `source` cannot be converted to something we can write to a matfile,
459 return None. If `source` is equivalent to an empty dictionary, return
460 ``EmptyStructMarker``. Otherwise return `source` converted to an
461 ndarray with contents for writing to matfile.
462 '''
463 if isinstance(source, np.ndarray):
464 return source
465 if source is None:
466 return None
467 if hasattr(source, "__array__"):
468 return np.asarray(source)
469 # Objects that implement mappings
470 is_mapping = (hasattr(source, 'keys') and hasattr(source, 'values') and
471 hasattr(source, 'items'))
472 # Objects that don't implement mappings, but do have dicts
473 if isinstance(source, np.generic):
474 # NumPy scalars are never mappings (PyPy issue workaround)
475 pass
476 elif not is_mapping and hasattr(source, '__dict__'):
477 source = {key: value for key, value in source.__dict__.items()
478 if not key.startswith('_')}
479 is_mapping = True
480 if is_mapping:
481 dtype = []
482 values = []
483 for field, value in source.items():
484 if (isinstance(field, str) and
485 field[0] not in '_0123456789'):
486 dtype.append((str(field), object))
487 values.append(value)
488 if dtype:
489 return np.array([tuple(values)], dtype)
490 else:
491 return EmptyStructMarker
492 # Next try and convert to an array
493 try:
494 narr = np.asanyarray(source)
495 except ValueError:
496 narr = np.asanyarray(source, dtype=object)
497 if narr.dtype.type in (object, np.object_) and \
498 narr.shape == () and narr == source:
499 # No interesting conversion possible
500 return None
501 return narr

531 def write_element(self, arr, mdtype=None):
532 ''' write tag and data '''
533 if mdtype is None:
534 mdtype = NP_TO_MTYPES[arr.dtype.str[1:]]
535 # Array needs to be in native byte order
536 if arr.dtype.byteorder == swapped_code:
537 arr = arr.byteswap().newbyteorder()
538 byte_count = arr.size*arr.itemsize
539 if byte_count <= 4:
540 self.write_smalldata_element(arr, mdtype, byte_count)
541 else:
542 self.write_regular_element(arr, mdtype, byte_count)

544 def write_smalldata_element(self, arr, mdtype, byte_count):
545 # write tag with embedded data
546 tag = np.zeros((), NDT_TAG_SMALL)
547 tag['byte_count_mdtype'] = (byte_count << 16) + mdtype
548 # if arr.tobytes is < 4, the element will be zero-padded as needed.
549 tag['data'] = arr.tobytes(order='F')
550 self.write_bytes(tag)
551
552 def write_regular_element(self, arr, mdtype, byte_count):
553 # write tag, data
554 tag = np.zeros((), NDT_TAG_FULL)
555 tag['mdtype'] = mdtype
556 tag['byte_count'] = byte_count
557 self.write_bytes(tag)
558 self.write_bytes(arr)
559 # pad to next 64-bit boundary
560 bc_mod_8 = byte_count % 8
561 if bc_mod_8:
562 self.file_stream.write(b'\x00' * (8-bc_mod_8))

564 def write_header(self,
565 shape,
566 mclass,
567 is_complex=False,
568 is_logical=False,
569 nzmax=0):
570 ''' Write header for given data options
571 shape : sequence
572 array shape
573 mclass - mat5 matrix class
574 is_complex - True if matrix is complex
575 is_logical - True if matrix is logical
576 nzmax - max non zero elements for sparse arrays
577
578 We get the name and the global flag from the object, and reset
579 them to defaults after we've used them
580 '''
581 # get name and is_global from one-shot object store
582 name = self._var_name
583 is_global = self._var_is_global
584 # initialize the top-level matrix tag, store position
585 self._mat_tag_pos = self.file_stream.tell()
586 self.write_bytes(self.mat_tag)
587 # write array flags (complex, global, logical, class, nzmax)
588 af = np.zeros((), NDT_ARRAY_FLAGS)
589 af['data_type'] = miUINT32
590 af['byte_count'] = 8
591 flags = is_complex << 3 | is_global << 2 | is_logical << 1
592 af['flags_class'] = mclass | flags << 8
593 af['nzmax'] = nzmax
594 self.write_bytes(af)
595 # shape
596 self.write_element(np.array(shape, dtype='i4'))
597 # write name
598 name = np.asarray(name)
599 if name == '': # empty string zero-terminated
600 self.write_smalldata_element(name, miINT8, 0)
601 else:
602 self.write_element(name, miINT8)
603 # reset the one-shot store to defaults
604 self._var_name = ''
605 self._var_is_global = False

607 def update_matrix_tag(self, start_pos):
608 curr_pos = self.file_stream.tell()
609 self.file_stream.seek(start_pos)
610 byte_count = curr_pos - start_pos - 8

612 raise MatWriteError("Matrix too large to save with Matlab "
613 "5 format")
614 self.mat_tag['byte_count'] = byte_count
615 self.write_bytes(self.mat_tag)
616 self.file_stream.seek(curr_pos)

638 def write(self, arr):
639 ''' Write `arr` to stream at top and sub levels
640
641 Parameters
642 ----------
643 arr : array_like
644 array-like object to create writer for
645 '''
646 # store position, so we can update the matrix tag
647 mat_tag_pos = self.file_stream.tell()
648 # First check if these are sparse
649 if scipy.sparse.issparse(arr):
650 self.write_sparse(arr)
651 self.update_matrix_tag(mat_tag_pos)
652 return
653 # Try to convert things that aren't arrays
654 narr = to_writeable(arr)
655 if narr is None:
656 raise TypeError('Could not convert %s (type %s) to array'
657 % (arr, type(arr)))
658 if isinstance(narr, MatlabObject):
659 self.write_object(narr)
660 elif isinstance(narr, MatlabFunction):
661 raise MatWriteError('Cannot write matlab functions')
662 elif narr is EmptyStructMarker: # empty struct array
663 self.write_empty_struct()
664 elif narr.dtype.fields: # struct array
665 self.write_struct(narr)
666 elif narr.dtype.hasobject: # cell array
667 self.write_cells(narr)
668 elif narr.dtype.kind in ('U', 'S'):
669 if self.unicode_strings:
670 codec = 'UTF8'
671 else:
672 codec = 'ascii'
673 self.write_char(narr, codec)
674 else:
675 self.write_numeric(narr)
676 self.update_matrix_tag(mat_tag_pos)

678 def write_numeric(self, arr):
679 imagf = arr.dtype.kind == 'c'
680 logif = arr.dtype.kind == 'b'
681 try:
682 mclass = NP_TO_MXTYPES[arr.dtype.str[1:]]
683 except KeyError:
684 # No matching matlab type, probably complex256 / float128 / float96
685 # Cast data to complex128 / float64.
686 if imagf:
687 arr = arr.astype('c128')
688 elif logif:
689 arr = arr.astype('i1') # Should only contain 0/1
690 else:
691 arr = arr.astype('f8')
692 mclass = mxDOUBLE_CLASS
693 self.write_header(matdims(arr, self.oned_as),
694 mclass,
695 is_complex=imagf,
696 is_logical=logif)
697 if imagf:
698 self.write_element(arr.real)
699 self.write_element(arr.imag)
700 else:
701 self.write_element(arr)

703 def write_char(self, arr, codec='ascii'):
704 ''' Write string array `arr` with given `codec`
705 '''
706 if arr.size == 0 or np.all(arr == ''):
707 # This an empty string array or a string array containing
708 # only empty strings. Matlab cannot distinguish between a
709 # string array that is empty, and a string array containing
710 # only empty strings, because it stores strings as arrays of
711 # char. There is no way of having an array of char that is
712 # not empty, but contains an empty string. We have to
713 # special-case the array-with-empty-strings because even
714 # empty strings have zero padding, which would otherwise
715 # appear in matlab as a string with a space.
716 shape = (0,) * np.max([arr.ndim, 2])
717 self.write_header(shape, mxCHAR_CLASS)
718 self.write_smalldata_element(arr, miUTF8, 0)
719 return
720 # non-empty string.
721 #
722 # Convert to char array
723 arr = arr_to_chars(arr)
724 # We have to write the shape directly, because we are going
725 # recode the characters, and the resulting stream of chars
726 # may have a different length
727 shape = arr.shape
728 self.write_header(shape, mxCHAR_CLASS)
729 if arr.dtype.kind == 'U' and arr.size:
730 # Make one long string from all the characters. We need to
731 # transpose here, because we're flattening the array, before
732 # we write the bytes. The bytes have to be written in
733 # Fortran order.
734 n_chars = np.prod(shape)
735 st_arr = np.ndarray(shape=(),
736 dtype=arr_dtype_number(arr, n_chars),
737 buffer=arr.T.copy()) # Fortran order
738 # Recode with codec to give byte string
739 st = st_arr.item().encode(codec)
740 # Reconstruct as 1-D byte array
741 arr = np.ndarray(shape=(len(st),),
742 dtype='S1',
743 buffer=st)
744 self.write_element(arr, mdtype=miUTF8)
745
746 def write_sparse(self, arr):
747 ''' Sparse matrices are 2D
748 '''
749 A = arr.tocsc() # convert to sparse CSC format
750 A.sort_indices() # MATLAB expects sorted row indices
751 is_complex = (A.dtype.kind == 'c')
752 is_logical = (A.dtype.kind == 'b')
753 nz = A.nnz
754 self.write_header(matdims(arr, self.oned_as),
755 mxSPARSE_CLASS,
756 is_complex=is_complex,
757 is_logical=is_logical,
758 # matlab won't load file with 0 nzmax
759 nzmax=1 if nz == 0 else nz)
760 self.write_element(A.indices.astype('i4'))
761 self.write_element(A.indptr.astype('i4'))
762 self.write_element(A.data.real)
763 if is_complex:
764 self.write_element(A.data.imag)
765
766 def write_cells(self, arr):
767 self.write_header(matdims(arr, self.oned_as),
768 mxCELL_CLASS)
769 # loop over data, column major
770 A = np.atleast_2d(arr).flatten('F')
771 for el in A:
772 self.write(el)

774 def write_empty_struct(self):
775 self.write_header((1, 1), mxSTRUCT_CLASS)
776 # max field name length set to 1 in an example matlab struct
777 self.write_element(np.array(1, dtype=np.int32))
778 # Field names element is empty
779 self.write_element(np.array([], dtype=np.int8))

781 def write_struct(self, arr):
782 self.write_header(matdims(arr, self.oned_as),
783 mxSTRUCT_CLASS)
784 self._write_items(arr)
785
786 def _write_items(self, arr):
787 # write fieldnames
788 fieldnames = [f[0] for f in arr.dtype.descr]
789 length = max([len(fieldname) for fieldname in fieldnames])+1
790 max_length = (self.long_field_names and 64) or 32
791 if length > max_length:
792 raise ValueError("Field names are restricted to %d characters" %
793 (max_length-1))
794 self.write_element(np.array([length], dtype='i4'))
795 self.write_element(
796 np.array(fieldnames, dtype='S%d' % (length)),
797 mdtype=miINT8)
798 A = np.atleast_2d(arr).flatten('F')
799 for el in A:
800 for f in fieldnames:
801 self.write(el[f])
802
803 def write_object(self, arr):
804 '''Same as writing structs, except different mx class, and extra
805 classname element after header
806 '''
807 self.write_header(matdims(arr, self.oned_as),
808 mxOBJECT_CLASS)
809 self.write_element(np.array(arr.classname, dtype='S'),
810 mdtype=miINT8)
811 self._write_items(arr)
