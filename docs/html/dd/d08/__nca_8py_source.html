34class NeighborhoodComponentsAnalysis(
35 ClassNamePrefixFeaturesOutMixin, TransformerMixin, BaseEstimator
36):
37 """Neighborhood Components Analysis.
38
39 Neighborhood Component Analysis (NCA) is a machine learning algorithm for
40 metric learning. It learns a linear transformation in a supervised fashion
41 to improve the classification accuracy of a stochastic nearest neighbors
42 rule in the transformed space.
43
44 Read more in the :ref:`User Guide <nca>`.
45
46 Parameters
47 ----------
48 n_components : int, default=None
49 Preferred dimensionality of the projected space.
50 If None it will be set to `n_features`.
51
52 init : {'auto', 'pca', 'lda', 'identity', 'random'} or ndarray of shape \
53 (n_features_a, n_features_b), default='auto'
54 Initialization of the linear transformation. Possible options are
55 `'auto'`, `'pca'`, `'lda'`, `'identity'`, `'random'`, and a numpy
56 array of shape `(n_features_a, n_features_b)`.
57
58 - `'auto'`
59 Depending on `n_components`, the most reasonable initialization
60 is chosen. If `n_components <= min(n_features, n_classes - 1)`
61 we use `'lda'`, as it uses labels information. If not, but
62 `n_components < min(n_features, n_samples)`, we use `'pca'`, as
63 it projects data in meaningful directions (those of higher
64 variance). Otherwise, we just use `'identity'`.
65
66 - `'pca'`
67 `n_components` principal components of the inputs passed
68 to :meth:`fit` will be used to initialize the transformation.
69 (See :class:`~sklearn.decomposition.PCA`)
70
71 - `'lda'`
72 `min(n_components, n_classes)` most discriminative
73 components of the inputs passed to :meth:`fit` will be used to
74 initialize the transformation. (If `n_components > n_classes`,
75 the rest of the components will be zero.) (See
76 :class:`~sklearn.discriminant_analysis.LinearDiscriminantAnalysis`)
77
78 - `'identity'`
79 If `n_components` is strictly smaller than the
80 dimensionality of the inputs passed to :meth:`fit`, the identity
81 matrix will be truncated to the first `n_components` rows.
82
83 - `'random'`
84 The initial transformation will be a random array of shape
85 `(n_components, n_features)`. Each value is sampled from the
86 standard normal distribution.
87
88 - numpy array
89 `n_features_b` must match the dimensionality of the inputs passed
90 to :meth:`fit` and n_features_a must be less than or equal to that.
91 If `n_components` is not `None`, `n_features_a` must match it.
92
93 warm_start : bool, default=False
94 If `True` and :meth:`fit` has been called before, the solution of the
95 previous call to :meth:`fit` is used as the initial linear
96 transformation (`n_components` and `init` will be ignored).
97
98 max_iter : int, default=50
99 Maximum number of iterations in the optimization.
100
101 tol : float, default=1e-5
102 Convergence tolerance for the optimization.
103
104 callback : callable, default=None
105 If not `None`, this function is called after every iteration of the
106 optimizer, taking as arguments the current solution (flattened
107 transformation matrix) and the number of iterations. This might be
108 useful in case one wants to examine or store the transformation
109 found after each iteration.
110
111 verbose : int, default=0
112 If 0, no progress messages will be printed.
113 If 1, progress messages will be printed to stdout.
114 If > 1, progress messages will be printed and the `disp`
115 parameter of :func:`scipy.optimize.minimize` will be set to
116 `verbose - 2`.
117
118 random_state : int or numpy.RandomState, default=None
119 A pseudo random number generator object or a seed for it if int. If
120 `init='random'`, `random_state` is used to initialize the random
121 transformation. If `init='pca'`, `random_state` is passed as an
122 argument to PCA when initializing the transformation. Pass an int
123 for reproducible results across multiple function calls.
124 See :term:`Glossary <random_state>`.
125
126 Attributes
127 ----------
128 components_ : ndarray of shape (n_components, n_features)
129 The linear transformation learned during fitting.
130
131 n_features_in_ : int
132 Number of features seen during :term:`fit`.
133
134 .. versionadded:: 0.24
135
136 n_iter_ : int
137 Counts the number of iterations performed by the optimizer.
138
139 random_state_ : numpy.RandomState
140 Pseudo random number generator object used during initialization.
141
142 feature_names_in_ : ndarray of shape (`n_features_in_`,)
143 Names of features seen during :term:`fit`. Defined only when `X`
144 has feature names that are all strings.
145
146 .. versionadded:: 1.0
147
148 See Also
149 --------
150 sklearn.discriminant_analysis.LinearDiscriminantAnalysis : Linear
151 Discriminant Analysis.
152 sklearn.decomposition.PCA : Principal component analysis (PCA).
153
154 References
155 ----------
156 .. [1] J. Goldberger, G. Hinton, S. Roweis, R. Salakhutdinov.
157 "Neighbourhood Components Analysis". Advances in Neural Information
158 Processing Systems. 17, 513-520, 2005.
159 http://www.cs.nyu.edu/~roweis/papers/ncanips.pdf
160
161 .. [2] Wikipedia entry on Neighborhood Components Analysis
162 https://en.wikipedia.org/wiki/Neighbourhood_components_analysis

164 Examples
165 --------
166 >>> from sklearn.neighbors import NeighborhoodComponentsAnalysis
167 >>> from sklearn.neighbors import KNeighborsClassifier
168 >>> from sklearn.datasets import load_iris
169 >>> from sklearn.model_selection import train_test_split
170 >>> X, y = load_iris(return_X_y=True)
171 >>> X_train, X_test, y_train, y_test = train_test_split(X, y,
172 ... stratify=y, test_size=0.7, random_state=42)
173 >>> nca = NeighborhoodComponentsAnalysis(random_state=42)
174 >>> nca.fit(X_train, y_train)
175 NeighborhoodComponentsAnalysis(...)
176 >>> knn = KNeighborsClassifier(n_neighbors=3)
177 >>> knn.fit(X_train, y_train)
178 KNeighborsClassifier(...)
179 >>> print(knn.score(X_test, y_test))
180 0.933333...
181 >>> knn.fit(nca.transform(X_train), y_train)
182 KNeighborsClassifier(...)
183 >>> print(knn.score(nca.transform(X_test), y_test))
184 0.961904...
185 """

225 @_fit_context(prefer_skip_nested_validation=True)
226 def fit(self, X, y):
227 """Fit the model according to the given training data.
228
229 Parameters
230 ----------
231 X : array-like of shape (n_samples, n_features)
232 The training samples.
233
234 y : array-like of shape (n_samples,)
235 The corresponding training labels.
236
237 Returns
238 -------
239 self : object
240 Fitted estimator.
241 """
242 # Validate the inputs X and y, and converts y to numerical classes.
243 X, y = validate_data(self, X, y, ensure_min_samples=2)
244 check_classification_targets(y)
245 y = LabelEncoder().fit_transform(y)
246
247 # Check the preferred dimensionality of the projected space
248 if self.n_components is not None and self.n_components > X.shape[1]:
249 raise ValueError(
250 "The preferred dimensionality of the "
251 f"projected space `n_components` ({self.n_components}) cannot "
252 "be greater than the given data "
253 f"dimensionality ({X.shape[1]})!"
254 )
255 # If warm_start is enabled, check that the inputs are consistent
256 if (
257 self.warm_start
258 and hasattr(self, "components_")
259 and self.components_.shape[1] != X.shape[1]
260 ):
261 raise ValueError(
262 f"The new inputs dimensionality ({X.shape[1]}) does not "
263 "match the input dimensionality of the "
264 f"previously learned transformation ({self.components_.shape[1]})."
265 )
266 # Check how the linear transformation should be initialized
267 init = self.init
268 if isinstance(init, np.ndarray):
269 init = check_array(init)
270 # Assert that init.shape[1] = X.shape[1]
271 if init.shape[1] != X.shape[1]:
272 raise ValueError(
273 f"The input dimensionality ({init.shape[1]}) of the given "
274 "linear transformation `init` must match the "
275 f"dimensionality of the given inputs `X` ({X.shape[1]})."
276 )
277 # Assert that init.shape[0] <= init.shape[1]
278 if init.shape[0] > init.shape[1]:
279 raise ValueError(
280 f"The output dimensionality ({init.shape[0]}) of the given "
281 "linear transformation `init` cannot be "
282 f"greater than its input dimensionality ({init.shape[1]})."
283 )
284 # Assert that self.n_components = init.shape[0]
285 if self.n_components is not None and self.n_components != init.shape[0]:
286 raise ValueError(
287 "The preferred dimensionality of the "
288 f"projected space `n_components` ({self.n_components}) does"
289 " not match the output dimensionality of "
290 "the given linear transformation "
291 f"`init` ({init.shape[0]})!"
292 )
293
294 # Initialize the random generator
295 self.random_state_ = check_random_state(self.random_state)
296
297 # Measure the total training time
298 t_train = time.time()
299
300 # Compute a mask that stays fixed during optimization:
301 same_class_mask = y[:, np.newaxis] == y[np.newaxis, :]
302 # (n_samples, n_samples)
303
304 # Initialize the transformation
305 transformation = np.ravel(self._initialize(X, y, init))
306
307 # Create a dictionary of parameters to be passed to the optimizer
308 disp = self.verbose - 2 if self.verbose > 1 else -1
309 optimizer_params = {
310 "method": "L-BFGS-B",
311 "fun": self._loss_grad_lbfgs,
312 "args": (X, same_class_mask, -1.0),
313 "jac": True,
314 "x0": transformation,
315 "tol": self.tol,
316 "options": dict(
317 maxiter=self.max_iter,
318 **_get_additional_lbfgs_options_dict("disp", disp),
319 ),
320 "callback": self._callback,
321 }
322
323 # Call the optimizer
324 self.n_iter_ = 0
325 opt_result = minimize(**optimizer_params)
326
327 # Reshape the solution found by the optimizer
328 self.components_ = opt_result.x.reshape(-1, X.shape[1])
329
330 # Stop timer
331 t_train = time.time() - t_train
332 if self.verbose:
333 cls_name = self.__class__.__name__
334
335 # Warn the user if the algorithm did not converge
336 if not opt_result.success:
337 warn(
338 "[{}] NCA did not converge: {}".format(
339 cls_name, opt_result.message
340 ),
341 ConvergenceWarning,
342 )
343
344 print("[{}] Training took {:8.2f}s.".format(cls_name, t_train))
345
346 return self

348 def transform(self, X):
349 """Apply the learned transformation to the given data.
350
351 Parameters
352 ----------
353 X : array-like of shape (n_samples, n_features)
354 Data samples.
355
356 Returns
357 -------
358 X_embedded: ndarray of shape (n_samples, n_components)
359 The data samples transformed.
360
361 Raises
362 ------
363 NotFittedError
364 If :meth:`fit` has not been called before.
365 """
366
367 check_is_fitted(self)
368 X = validate_data(self, X, reset=False)
369
370 return np.dot(X, self.components_.T)

372 def _initialize(self, X, y, init):
373 """Initialize the transformation.
374
375 Parameters
376 ----------
377 X : array-like of shape (n_samples, n_features)
378 The training samples.
379
380 y : array-like of shape (n_samples,)
381 The training labels.
382
383 init : str or ndarray of shape (n_features_a, n_features_b)
384 The validated initialization of the linear transformation.
385
386 Returns
387 -------
388 transformation : ndarray of shape (n_components, n_features)
389 The initialized linear transformation.
390
391 """
392
393 transformation = init
394 if self.warm_start and hasattr(self, "components_"):
395 transformation = self.components_
396 elif isinstance(init, np.ndarray):
397 pass
398 else:
399 n_samples, n_features = X.shape
400 n_components = self.n_components or n_features
401 if init == "auto":
402 n_classes = len(np.unique(y))
403 if n_components <= min(n_features, n_classes - 1):
404 init = "lda"
405 elif n_components < min(n_features, n_samples):
406 init = "pca"
407 else:
408 init = "identity"
409 if init == "identity":
410 transformation = np.eye(n_components, X.shape[1])
411 elif init == "random":
412 transformation = self.random_state_.standard_normal(
413 size=(n_components, X.shape[1])
414 )
415 elif init in {"pca", "lda"}:
416 init_time = time.time()
417 if init == "pca":
418 pca = PCA(
419 n_components=n_components, random_state=self.random_state_
420 )
421 if self.verbose:
422 print("Finding principal components... ", end="")
423 sys.stdout.flush()
424 pca.fit(X)
425 transformation = pca.components_
426 elif init == "lda":
427 from ..discriminant_analysis import LinearDiscriminantAnalysis
428
429 lda = LinearDiscriminantAnalysis(n_components=n_components)
430 if self.verbose:
431 print("Finding most discriminative components... ", end="")
432 sys.stdout.flush()
433 lda.fit(X, y)
434 transformation = lda.scalings_.T[:n_components]
435 if self.verbose:
436 print("done in {:5.2f}s".format(time.time() - init_time))
437 return transformation

452 def _loss_grad_lbfgs(self, transformation, X, same_class_mask, sign=1.0):
453 """Compute the loss and the loss gradient w.r.t. `transformation`.
454
455 Parameters
456 ----------
457 transformation : ndarray of shape (n_components * n_features,)
458 The raveled linear transformation on which to compute loss and
459 evaluate gradient.
460
461 X : ndarray of shape (n_samples, n_features)
462 The training samples.
463
464 same_class_mask : ndarray of shape (n_samples, n_samples)
465 A mask where `mask[i, j] == 1` if `X[i]` and `X[j]` belong
466 to the same class, and `0` otherwise.
467
468 Returns
469 -------
470 loss : float
471 The loss computed for the given transformation.
472
473 gradient : ndarray of shape (n_components * n_features,)
474 The new (flattened) gradient of the loss.
475 """
476
477 if self.n_iter_ == 0:
478 self.n_iter_ += 1
479 if self.verbose:
480 header_fields = ["Iteration", "Objective Value", "Time(s)"]
481 header_fmt = "{:>10} {:>20} {:>10}"
482 header = header_fmt.format(*header_fields)
483 cls_name = self.__class__.__name__
484 print("[{}]".format(cls_name))
485 print(
486 "[{}] {}\n[{}] {}".format(
487 cls_name, header, cls_name, "-" * len(header)
488 )
489 )
490
491 t_funcall = time.time()
492
493 transformation = transformation.reshape(-1, X.shape[1])
494 X_embedded = np.dot(X, transformation.T) # (n_samples, n_components)
495
496 # Compute softmax distances
497 p_ij = pairwise_distances(X_embedded, squared=True)
498 np.fill_diagonal(p_ij, np.inf)
499 p_ij = softmax(-p_ij) # (n_samples, n_samples)
500
501 # Compute loss
502 masked_p_ij = p_ij * same_class_mask
503 p = np.sum(masked_p_ij, axis=1, keepdims=True) # (n_samples, 1)
504 loss = np.sum(p)
505
506 # Compute gradient of loss w.r.t. `transform`
507 weighted_p_ij = masked_p_ij - p_ij * p
508 weighted_p_ij_sym = weighted_p_ij + weighted_p_ij.T
509 np.fill_diagonal(weighted_p_ij_sym, -weighted_p_ij.sum(axis=0))
510 gradient = 2 * X_embedded.T.dot(weighted_p_ij_sym).dot(X)
511 # time complexity of the gradient: O(n_components x n_samples x (
512 # n_samples + n_features))
513
514 if self.verbose:
515 t_funcall = time.time() - t_funcall
516 values_fmt = "[{}] {:>10} {:>20.6e} {:>10.2f}"
517 print(
518 values_fmt.format(
519 self.__class__.__name__, self.n_iter_, loss, t_funcall
520 )
521 )
522 sys.stdout.flush()
523
524 return sign * loss, sign * gradient.ravel()
