167 def test_method(self, ishigami_ref_indices):
168 def jansen_sobol(f_A, f_B, f_AB):
169 """Jansen for S and Sobol' for St.
170
171 From Saltelli2010, table 2 formulations (c) and (e)."""
172 var = np.var([f_A, f_B], axis=(0, -1))
173
174 s = (var - 0.5*np.mean((f_B - f_AB)**2, axis=-1)) / var
175 st = np.mean(f_A*(f_A - f_AB), axis=-1) / var
176
177 return s.T, st.T
178
179 rng = np.random.default_rng(28631265345463262246170309650372465332)
180 res = sobol_indices(
181 func=f_ishigami, n=4096,
182 dists=self.dists,
183 method=jansen_sobol,
184 random_state=rng
185 )
186
187 assert_allclose(res.first_order, ishigami_ref_indices[0], atol=1e-2)
188 assert_allclose(res.total_order, ishigami_ref_indices[1], atol=1e-2)
