27def _check_ns_shape_dtype(
28 actual: Array, desired: Array
29) -> ModuleType: # numpydoc ignore=RT03
30 """
31 Assert that namespace, shape and dtype of the two arrays match.
32
33 Parameters
34 ----------
35 actual : Array
36 The array produced by the tested function.
37 desired : Array
38 The expected array (typically hardcoded).
39
40 Returns
41 -------
42 Arrays namespace.
43 """
44 actual_xp = array_namespace(actual) # Raises on scalars and lists
45 desired_xp = array_namespace(desired)
46
47 msg = f"namespaces do not match: {actual_xp} != f{desired_xp}"
48 assert actual_xp == desired_xp, msg
49
50 actual_shape = actual.shape
51 desired_shape = desired.shape
52 if is_dask_namespace(desired_xp):
53 # Dask uses nan instead of None for unknown shapes
54 if any(math.isnan(i) for i in cast(tuple[float, ...], actual_shape)):
55 actual_shape = actual.compute().shape # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
56 if any(math.isnan(i) for i in cast(tuple[float, ...], desired_shape)):
57 desired_shape = desired.compute().shape # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
58
59 msg = f"shapes do not match: {actual_shape} != f{desired_shape}"
60 assert actual_shape == desired_shape, msg
61
62 msg = f"dtypes do not match: {actual.dtype} != {desired.dtype}"
63 assert actual.dtype == desired.dtype, msg
64
65 return desired_xp

68def xp_assert_equal(actual: Array, desired: Array, err_msg: str = "") -> None:
69 """
70 Array-API compatible version of `np.testing.assert_array_equal`.
71
72 Parameters
73 ----------
74 actual : Array
75 The array produced by the tested function.
76 desired : Array
77 The expected array (typically hardcoded).
78 err_msg : str, optional
79 Error message to display on failure.
80
81 See Also
82 --------
83 xp_assert_close : Similar function for inexact equality checks.
84 numpy.testing.assert_array_equal : Similar function for NumPy arrays.
85 """
86 xp = _check_ns_shape_dtype(actual, desired)
87
88 if is_cupy_namespace(xp):
89 xp.testing.assert_array_equal(actual, desired, err_msg=err_msg)
90 elif is_torch_namespace(xp):
91 # PyTorch recommends using `rtol=0, atol=0` like this
92 # to test for exact equality
93 xp.testing.assert_close(
94 actual,
95 desired,
96 rtol=0,
97 atol=0,
98 equal_nan=True,
99 check_dtype=False,
100 msg=err_msg or None,
101 )
102 else:
103 import numpy as np # pylint: disable=import-outside-toplevel
104
105 if is_pydata_sparse_namespace(xp):
106 actual = actual.todense() # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
107 desired = desired.todense() # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
108
109 actual_np = None
110 desired_np = None
111 if is_array_api_strict_namespace(xp):
112 # __array__ doesn't work on array-api-strict device arrays
113 # We need to convert to the CPU device first
114 actual_np = np.asarray(xp.asarray(actual, device=xp.Device("CPU_DEVICE")))
115 desired_np = np.asarray(xp.asarray(desired, device=xp.Device("CPU_DEVICE")))
116
117 # JAX/Dask arrays work with `np.testing`
118 actual_np = actual if actual_np is None else actual_np
119 desired_np = desired if desired_np is None else desired_np

123def xp_assert_close(
124 actual: Array,
125 desired: Array,
126 *,
127 rtol: float | None = None,
128 atol: float = 0,
129 err_msg: str = "",
130) -> None:
131 """
132 Array-API compatible version of `np.testing.assert_allclose`.
133
134 Parameters
135 ----------
136 actual : Array
137 The array produced by the tested function.
138 desired : Array
139 The expected array (typically hardcoded).
140 rtol : float, optional
141 Relative tolerance. Default: dtype-dependent.
142 atol : float, optional
143 Absolute tolerance. Default: 0.
144 err_msg : str, optional
145 Error message to display on failure.
146
147 See Also
148 --------
149 xp_assert_equal : Similar function for exact equality checks.
150 isclose : Public function for checking closeness.
151 numpy.testing.assert_allclose : Similar function for NumPy arrays.
152
153 Notes
154 -----
155 The default `atol` and `rtol` differ from `xp.all(xpx.isclose(a, b))`.
156 """
157 xp = _check_ns_shape_dtype(actual, desired)
158
159 floating = xp.isdtype(actual.dtype, ("real floating", "complex floating"))
160 if rtol is None and floating:
161 # multiplier of 4 is used as for `np.float64` this puts the default `rtol`
162 # roughly half way between sqrt(eps) and the default for
163 # `numpy.testing.assert_allclose`, 1e-7
164 rtol = xp.finfo(actual.dtype).eps ** 0.5 * 4
165 elif rtol is None:
166 rtol = 1e-7
167
168 if is_cupy_namespace(xp):
169 xp.testing.assert_allclose(
170 actual, desired, rtol=rtol, atol=atol, err_msg=err_msg
171 )
172 elif is_torch_namespace(xp):
173 xp.testing.assert_close(
174 actual, desired, rtol=rtol, atol=atol, equal_nan=True, msg=err_msg or None
175 )
176 else:
177 import numpy as np # pylint: disable=import-outside-toplevel
178
179 if is_pydata_sparse_namespace(xp):
180 actual = actual.todense() # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
181 desired = desired.todense() # type: ignore[attr-defined] # pyright: ignore[reportAttributeAccessIssue]
182
183 actual_np = None
184 desired_np = None
185 if is_array_api_strict_namespace(xp):
186 # __array__ doesn't work on array-api-strict device arrays
187 # We need to convert to the CPU device first
188 actual_np = np.asarray(xp.asarray(actual, device=xp.Device("CPU_DEVICE")))
189 desired_np = np.asarray(xp.asarray(desired, device=xp.Device("CPU_DEVICE")))
190
191 # JAX/Dask arrays work with `np.testing`
192 actual_np = actual if actual_np is None else actual_np
193 desired_np = desired if desired_np is None else desired_np
194
195 assert isinstance(rtol, float)
196 np.testing.assert_allclose( # pyright: ignore[reportCallIssue]
197 actual_np, # type: ignore[arg-type] # pyright: ignore[reportArgumentType]
198 desired_np, # type: ignore[arg-type] # pyright: ignore[reportArgumentType]
199 rtol=rtol,
200 atol=atol,
201 err_msg=err_msg,
