200def assert_tree_equal(d, s, message):
201 assert s.node_count == d.node_count, (
202 "{0}: inequal number of node ({1} != {2})".format(
203 message, s.node_count, d.node_count
204 )
205 )
206
207 assert_array_equal(
208 d.children_right, s.children_right, message + ": inequal children_right"
209 )
210 assert_array_equal(
211 d.children_left, s.children_left, message + ": inequal children_left"
212 )
213
214 external = d.children_right == TREE_LEAF
215 internal = np.logical_not(external)
216
217 assert_array_equal(
218 d.feature[internal], s.feature[internal], message + ": inequal features"
219 )
220 assert_array_equal(
221 d.threshold[internal], s.threshold[internal], message + ": inequal threshold"
222 )
223 assert_array_equal(
224 d.n_node_samples.sum(),
225 s.n_node_samples.sum(),
226 message + ": inequal sum(n_node_samples)",
227 )
228 assert_array_equal(
229 d.n_node_samples, s.n_node_samples, message + ": inequal n_node_samples"
230 )
231
232 assert_almost_equal(d.impurity, s.impurity, err_msg=message + ": inequal impurity")
233
234 assert_array_almost_equal(
235 d.value[external], s.value[external], err_msg=message + ": inequal value"

817def test_min_impurity_decrease(global_random_seed):
818 # test if min_impurity_decrease ensure that a split is made only if
819 # if the impurity decrease is at least that value
820 X, y = datasets.make_classification(n_samples=100, random_state=global_random_seed)
821
822 # test both DepthFirstTreeBuilder and BestFirstTreeBuilder
823 # by setting max_leaf_nodes
824 for max_leaf_nodes, name in product((None, 1000), ALL_TREES.keys()):
825 TreeEstimator = ALL_TREES[name]
826
827 # Check default value of min_impurity_decrease, 1e-7
828 est1 = TreeEstimator(max_leaf_nodes=max_leaf_nodes, random_state=0)
829 # Check with explicit value of 0.05
830 est2 = TreeEstimator(
831 max_leaf_nodes=max_leaf_nodes, min_impurity_decrease=0.05, random_state=0
832 )
833 # Check with a much lower value of 0.0001
834 est3 = TreeEstimator(
835 max_leaf_nodes=max_leaf_nodes, min_impurity_decrease=0.0001, random_state=0
836 )
837 # Check with a much lower value of 0.1
838 est4 = TreeEstimator(
839 max_leaf_nodes=max_leaf_nodes, min_impurity_decrease=0.1, random_state=0
840 )
841
842 for est, expected_decrease in (
843 (est1, 1e-7),
844 (est2, 0.05),
845 (est3, 0.0001),
846 (est4, 0.1),
847 ):
848 assert est.min_impurity_decrease <= expected_decrease, (
849 "Failed, min_impurity_decrease = {0} > {1}".format(
850 est.min_impurity_decrease, expected_decrease
851 )
852 )
853 est.fit(X, y)
854 for node in range(est.tree_.node_count):
855 # If current node is a not leaf node, check if the split was
856 # justified w.r.t the min_impurity_decrease
857 if est.tree_.children_left[node] != TREE_LEAF:
858 imp_parent = est.tree_.impurity[node]
859 wtd_n_node = est.tree_.weighted_n_node_samples[node]
860
861 left = est.tree_.children_left[node]
862 wtd_n_left = est.tree_.weighted_n_node_samples[left]
863 imp_left = est.tree_.impurity[left]
864 wtd_imp_left = wtd_n_left * imp_left
865
866 right = est.tree_.children_right[node]
867 wtd_n_right = est.tree_.weighted_n_node_samples[right]
868 imp_right = est.tree_.impurity[right]
869 wtd_imp_right = wtd_n_right * imp_right
870
871 wtd_avg_left_right_imp = wtd_imp_right + wtd_imp_left
872 wtd_avg_left_right_imp /= wtd_n_node
873
874 fractional_node_weight = (
875 est.tree_.weighted_n_node_samples[node] / X.shape[0]
876 )
877
878 actual_decrease = fractional_node_weight * (
879 imp_parent - wtd_avg_left_right_imp
880 )
881
882 assert actual_decrease >= expected_decrease, (
883 "Failed with {0} expected min_impurity_decrease={1}".format(
884 actual_decrease, expected_decrease
885 )

1663def test_mae():
1664 """Check MAE criterion produces correct results on small toy dataset:
1665
1666 ------------------
1667 | X | y | weight |
1668 ------------------
1669 | 3 | 3 | 0.1 |
1670 | 5 | 3 | 0.3 |
1671 | 8 | 4 | 1.0 |
1672 | 3 | 6 | 0.6 |
1673 | 5 | 7 | 0.3 |
1674 ------------------
1675 |sum wt:| 2.3 |
1676 ------------------
1677
1678 Because we are dealing with sample weights, we cannot find the median by
1679 simply choosing/averaging the centre value(s), instead we consider the
1680 median where 50% of the cumulative weight is found (in a y sorted data set)
1681 . Therefore with regards to this test data, the cumulative weight is >= 50%
1682 when y = 4. Therefore:
1683 Median = 4
1684
1685 For all the samples, we can get the total error by summing:
1686 Absolute(Median - y) * weight
1687
1688 I.e., total error = (Absolute(4 - 3) * 0.1)
1689 + (Absolute(4 - 3) * 0.3)
1690 + (Absolute(4 - 4) * 1.0)
1691 + (Absolute(4 - 6) * 0.6)
1692 + (Absolute(4 - 7) * 0.3)
1693 = 2.5
1694
1695 Impurity = Total error / total weight
1696 = 2.5 / 2.3
1697 = 1.08695652173913
1698 ------------------
1699
1700 From this root node, the next best split is between X values of 3 and 5.
1701 Thus, we have left and right child nodes:
1702
1703 LEFT RIGHT
1704 ------------------ ------------------
1705 | X | y | weight | | X | y | weight |
1706 ------------------ ------------------
1707 | 3 | 3 | 0.1 | | 5 | 3 | 0.3 |
1708 | 3 | 6 | 0.6 | | 8 | 4 | 1.0 |
1709 ------------------ | 5 | 7 | 0.3 |
1710 |sum wt:| 0.7 | ------------------
1711 ------------------ |sum wt:| 1.6 |
1712 ------------------
1713
1714 Impurity is found in the same way:
1715 Left node Median = 6
1716 Total error = (Absolute(6 - 3) * 0.1)
1717 + (Absolute(6 - 6) * 0.6)
1718 = 0.3
1719
1720 Left Impurity = Total error / total weight
1721 = 0.3 / 0.7
1722 = 0.428571428571429
1723 -------------------
1724
1725 Likewise for Right node:
1726 Right node Median = 4
1727 Total error = (Absolute(4 - 3) * 0.3)
1728 + (Absolute(4 - 4) * 1.0)
1729 + (Absolute(4 - 7) * 0.3)
1730 = 1.2
1731
1732 Right Impurity = Total error / total weight
1733 = 1.2 / 1.6
1734 = 0.75
1735 ------
1736 """
