24class KNNImputer(_BaseImputer):
25 """Imputation for completing missing values using k-Nearest Neighbors.
26
27 Each sample's missing values are imputed using the mean value from
28 `n_neighbors` nearest neighbors found in the training set. Two samples are
29 close if the features that neither is missing are close.
30
31 Read more in the :ref:`User Guide <knnimpute>`.
32
33 .. versionadded:: 0.22
34
35 Parameters
36 ----------
37 missing_values : int, float, str, np.nan or None, default=np.nan
38 The placeholder for the missing values. All occurrences of
39 `missing_values` will be imputed. For pandas' dataframes with
40 nullable integer dtypes with missing values, `missing_values`
41 should be set to np.nan, since `pd.NA` will be converted to np.nan.
42
43 n_neighbors : int, default=5
44 Number of neighboring samples to use for imputation.
45
46 weights : {'uniform', 'distance'} or callable, default='uniform'
47 Weight function used in prediction. Possible values:
48
49 - 'uniform' : uniform weights. All points in each neighborhood are
50 weighted equally.
51 - 'distance' : weight points by the inverse of their distance.
52 in this case, closer neighbors of a query point will have a
53 greater influence than neighbors which are further away.
54 - callable : a user-defined function which accepts an
55 array of distances, and returns an array of the same shape
56 containing the weights.
57
58 metric : {'nan_euclidean'} or callable, default='nan_euclidean'
59 Distance metric for searching neighbors. Possible values:
60
61 - 'nan_euclidean'
62 - callable : a user-defined function which conforms to the definition
63 of ``func_metric(x, y, *, missing_values=np.nan)``. `x` and `y`
64 corresponds to a row (i.e. 1-D arrays) of `X` and `Y`, respectively.
65 The callable should returns a scalar distance value.
66
67 copy : bool, default=True
68 If True, a copy of X will be created. If False, imputation will
69 be done in-place whenever possible.
70
71 add_indicator : bool, default=False
72 If True, a :class:`MissingIndicator` transform will stack onto the
73 output of the imputer's transform. This allows a predictive estimator
74 to account for missingness despite imputation. If a feature has no
75 missing values at fit/train time, the feature won't appear on the
76 missing indicator even if there are missing values at transform/test
77 time.
78
79 keep_empty_features : bool, default=False
80 If True, features that consist exclusively of missing values when
81 `fit` is called are returned in results when `transform` is called.
82 The imputed value is always `0`.
83
84 .. versionadded:: 1.2
85
86 Attributes
87 ----------
88 indicator_ : :class:`~sklearn.impute.MissingIndicator`
89 Indicator used to add binary indicators for missing values.
90 ``None`` if add_indicator is False.
91
92 n_features_in_ : int
93 Number of features seen during :term:`fit`.
94
95 .. versionadded:: 0.24
96
97 feature_names_in_ : ndarray of shape (`n_features_in_`,)
98 Names of features seen during :term:`fit`. Defined only when `X`
99 has feature names that are all strings.
100
101 .. versionadded:: 1.0
102
103 See Also
104 --------
105 SimpleImputer : Univariate imputer for completing missing values
106 with simple strategies.
107 IterativeImputer : Multivariate imputer that estimates values to impute for
108 each feature with missing values from all the others.
109
110 References
111 ----------
112 * `Olga Troyanskaya, Michael Cantor, Gavin Sherlock, Pat Brown, Trevor
113 Hastie, Robert Tibshirani, David Botstein and Russ B. Altman, Missing
114 value estimation methods for DNA microarrays, BIOINFORMATICS Vol. 17
115 no. 6, 2001 Pages 520-525.
116 <https://academic.oup.com/bioinformatics/article/17/6/520/272365>`_
117
118 Examples
119 --------
120 >>> import numpy as np
121 >>> from sklearn.impute import KNNImputer
122 >>> X = [[1, 2, np.nan], [3, 4, 3], [np.nan, 6, 5], [8, 8, 7]]
123 >>> imputer = KNNImputer(n_neighbors=2)
124 >>> imputer.fit_transform(X)
125 array([[1. , 2. , 4. ],
126 [3. , 4. , 3. ],
127 [5.5, 6. , 5. ],
128 [8. , 8. , 7. ]])
129
130 For a more detailed example see
131 :ref:`sphx_glr_auto_examples_impute_plot_missing_values.py`.
132 """

163 def _calc_impute(self, dist_pot_donors, n_neighbors, fit_X_col, mask_fit_X_col):
164 """Helper function to impute a single column.
165
166 Parameters
167 ----------
168 dist_pot_donors : ndarray of shape (n_receivers, n_potential_donors)
169 Distance matrix between the receivers and potential donors from
170 training set. There must be at least one non-nan distance between
171 a receiver and a potential donor.
172
173 n_neighbors : int
174 Number of neighbors to consider.
175
176 fit_X_col : ndarray of shape (n_potential_donors,)
177 Column of potential donors from training set.
178
179 mask_fit_X_col : ndarray of shape (n_potential_donors,)
180 Missing mask for fit_X_col.
181
182 Returns
183 -------
184 imputed_values: ndarray of shape (n_receivers,)
185 Imputed values for receiver.
186 """
187 # Get donors
188 donors_idx = np.argpartition(dist_pot_donors, n_neighbors - 1, axis=1)[
189 :, :n_neighbors
190 ]
191
192 # Get weight matrix from distance matrix
193 donors_dist = dist_pot_donors[
194 np.arange(donors_idx.shape[0])[:, None], donors_idx
195 ]
196
197 weight_matrix = _get_weights(donors_dist, self.weights)
198
199 # fill nans with zeros
200 if weight_matrix is not None:
201 weight_matrix[np.isnan(weight_matrix)] = 0.0
202 else:
203 weight_matrix = np.ones_like(donors_dist)
204 weight_matrix[np.isnan(donors_dist)] = 0.0
205
206 # Retrieve donor values and calculate kNN average
207 donors = fit_X_col.take(donors_idx)
208 donors_mask = mask_fit_X_col.take(donors_idx)
209 donors = np.ma.array(donors, mask=donors_mask)
210
211 return np.ma.average(donors, axis=1, weights=weight_matrix).data

214 def fit(self, X, y=None):
215 """Fit the imputer on X.
216
217 Parameters
218 ----------
219 X : array-like shape of (n_samples, n_features)
220 Input data, where `n_samples` is the number of samples and
221 `n_features` is the number of features.
222
223 y : Ignored
224 Not used, present here for API consistency by convention.
225
226 Returns
227 -------
228 self : object
229 The fitted `KNNImputer` class instance.
230 """
231 # Check data integrity and calling arguments
232 if not is_scalar_nan(self.missing_values):
233 ensure_all_finite = True
234 else:
235 ensure_all_finite = "allow-nan"
236
237 X = validate_data(
238 self,
239 X,
240 accept_sparse=False,
241 dtype=FLOAT_DTYPES,
242 ensure_all_finite=ensure_all_finite,
243 copy=self.copy,
244 )
245
246 self._fit_X = X
247 self._mask_fit_X = _get_mask(self._fit_X, self.missing_values)
248 self._valid_mask = ~np.all(self._mask_fit_X, axis=0)
249
250 super()._fit_indicator(self._mask_fit_X)
251
252 return self

254 def transform(self, X):
255 """Impute all missing values in X.
256
257 Parameters
258 ----------
259 X : array-like of shape (n_samples, n_features)
260 The input data to complete.
261
262 Returns
263 -------
264 X : array-like of shape (n_samples, n_output_features)
265 The imputed dataset. `n_output_features` is the number of features
266 that is not always missing during `fit`.
267 """
268
269 check_is_fitted(self)
270 if not is_scalar_nan(self.missing_values):
271 ensure_all_finite = True
272 else:
273 ensure_all_finite = "allow-nan"
274 X = validate_data(
275 self,
276 X,
277 accept_sparse=False,
278 dtype=FLOAT_DTYPES,
279 force_writeable=True,
280 ensure_all_finite=ensure_all_finite,
281 copy=self.copy,
282 reset=False,
283 )
284
285 mask = _get_mask(X, self.missing_values)
286 mask_fit_X = self._mask_fit_X
287 valid_mask = self._valid_mask
288
289 X_indicator = super()._transform_indicator(mask)
290
291 # Removes columns where the training data is all nan
292 if not np.any(mask[:, valid_mask]):
293 # No missing values in X
294 if self.keep_empty_features:
295 Xc = X
296 Xc[:, ~valid_mask] = 0
297 else:
298 Xc = X[:, valid_mask]
299
300 # Even if there are no missing values in X, we still concatenate Xc
301 # with the missing value indicator matrix, X_indicator.
302 # This is to ensure that the output maintains consistency in terms
303 # of columns, regardless of whether missing values exist in X or not.
304 return super()._concatenate_indicator(Xc, X_indicator)
305
306 row_missing_idx = np.flatnonzero(mask[:, valid_mask].any(axis=1))
307
308 non_missing_fix_X = np.logical_not(mask_fit_X)
309
310 # Maps from indices from X to indices in dist matrix
311 dist_idx_map = np.zeros(X.shape[0], dtype=int)
312 dist_idx_map[row_missing_idx] = np.arange(row_missing_idx.shape[0])
313
314 def process_chunk(dist_chunk, start):
315 row_missing_chunk = row_missing_idx[start : start + len(dist_chunk)]
316
317 # Find and impute missing by column
318 for col in range(X.shape[1]):
319 if not valid_mask[col]:
320 # column was all missing during training
321 continue
322
323 col_mask = mask[row_missing_chunk, col]
324 if not np.any(col_mask):
325 # column has no missing values
326 continue
327
328 (potential_donors_idx,) = np.nonzero(non_missing_fix_X[:, col])
329
330 # receivers_idx are indices in X
331 receivers_idx = row_missing_chunk[np.flatnonzero(col_mask)]
332
333 # distances for samples that needed imputation for column
334 dist_subset = dist_chunk[dist_idx_map[receivers_idx] - start][
335 :, potential_donors_idx
336 ]
337
338 # receivers with all nan distances impute with mean
339 all_nan_dist_mask = np.isnan(dist_subset).all(axis=1)
340 all_nan_receivers_idx = receivers_idx[all_nan_dist_mask]
341
342 if all_nan_receivers_idx.size:
343 col_mean = np.ma.array(
344 self._fit_X[:, col], mask=mask_fit_X[:, col]
345 ).mean()
346 X[all_nan_receivers_idx, col] = col_mean
347
348 if len(all_nan_receivers_idx) == len(receivers_idx):
349 # all receivers imputed with mean
350 continue
351
352 # receivers with at least one defined distance
353 receivers_idx = receivers_idx[~all_nan_dist_mask]
354 dist_subset = dist_chunk[dist_idx_map[receivers_idx] - start][
355 :, potential_donors_idx
356 ]
357
358 n_neighbors = min(self.n_neighbors, len(potential_donors_idx))
359 value = self._calc_impute(
360 dist_subset,
361 n_neighbors,
362 self._fit_X[potential_donors_idx, col],
363 mask_fit_X[potential_donors_idx, col],
364 )
365 X[receivers_idx, col] = value
366
367 # process in fixed-memory chunks
368 gen = pairwise_distances_chunked(
369 X[row_missing_idx, :],
370 self._fit_X,
371 metric=self.metric,
372 missing_values=self.missing_values,
373 ensure_all_finite=ensure_all_finite,
374 reduce_func=process_chunk,
375 )
376 for chunk in gen:
377 # process_chunk modifies X in place. No return value.
378 pass
379
380 if self.keep_empty_features:
381 Xc = X
382 Xc[:, ~valid_mask] = 0
383 else:
384 Xc = X[:, valid_mask]
385
386 return super()._concatenate_indicator(Xc, X_indicator)
