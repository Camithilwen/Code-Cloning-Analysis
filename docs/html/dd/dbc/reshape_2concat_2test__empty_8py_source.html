146 def int_result_type(dtype, dtype2):
147 typs = {dtype.kind, dtype2.kind}
148 if not len(typs - {"i", "u", "b"}) and (
149 dtype.kind == "i" or dtype2.kind == "i"
150 ):
151 return "i"
152 elif not len(typs - {"u", "b"}) and (
153 dtype.kind == "u" or dtype2.kind == "u"
154 ):
155 return "u"
156 return None
157
158 def float_result_type(dtype, dtype2):
159 typs = {dtype.kind, dtype2.kind}
160 if not len(typs - {"f", "i", "u"}) and (
161 dtype.kind == "f" or dtype2.kind == "f"
162 ):
163 return "f"
164 return None
165
166 def get_result_type(dtype, dtype2):
167 result = float_result_type(dtype, dtype2)
168 if result is not None:
169 return result
170 result = int_result_type(dtype, dtype2)
171 if result is not None:
172 return result
173 return "O"
174
175 dtype = np.dtype(dtype)
176 dtype2 = np.dtype(dtype2)
177 expected = get_result_type(dtype, dtype2)
178 result = concat([Series(dtype=dtype), Series(dtype=dtype2)]).dtype
179 assert result.kind == expected
