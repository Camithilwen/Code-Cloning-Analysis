132 def _validate_curve_kwargs(
133 n_curves,
134 name,
135 legend_metric,
136 legend_metric_name,
137 curve_kwargs,
138 **kwargs,
139 ):
140 """Get validated line kwargs for each curve.
141
142 Parameters
143 ----------
144 n_curves : int
145 Number of curves.
146
147 name : list of str or None
148 Name for labeling legend entries.
149
150 legend_metric : dict
151 Dictionary with "mean" and "std" keys, or "metric" key of metric
152 values for each curve. If None, "label" will not contain metric values.
153
154 legend_metric_name : str
155 Name of the summary value provided in `legend_metrics`.
156
157 curve_kwargs : dict or list of dict or None
158 Dictionary with keywords passed to the matplotlib's `plot` function
159 to draw the individual curves. If a list is provided, the
160 parameters are applied to the curves sequentially. If a single
161 dictionary is provided, the same parameters are applied to all
162 curves.
163
164 **kwargs : dict
165 Deprecated. Keyword arguments to be passed to matplotlib's `plot`.
166 """
167 # TODO(1.9): Remove deprecated **kwargs
168 if curve_kwargs and kwargs:
169 raise ValueError(
170 "Cannot provide both `curve_kwargs` and `kwargs`. `**kwargs` is "
171 "deprecated in 1.7 and will be removed in 1.9. Pass all matplotlib "
172 "arguments to `curve_kwargs` as a dictionary."
173 )
174 if kwargs:
175 warnings.warn(
176 "`**kwargs` is deprecated and will be removed in 1.9. Pass all "
177 "matplotlib arguments to `curve_kwargs` as a dictionary instead.",
178 FutureWarning,
179 )
180 curve_kwargs = kwargs
181
182 if isinstance(curve_kwargs, list) and len(curve_kwargs) != n_curves:
183 raise ValueError(
184 f"`curve_kwargs` must be None, a dictionary or a list of length "
185 f"{n_curves}. Got: {curve_kwargs}."
186 )
187
188 # Ensure valid `name` and `curve_kwargs` combination.
189 if (
190 isinstance(name, list)
191 and len(name) != 1
192 and not isinstance(curve_kwargs, list)
193 ):
194 raise ValueError(
195 "To avoid labeling individual curves that have the same appearance, "
196 f"`curve_kwargs` should be a list of {n_curves} dictionaries. "
197 "Alternatively, set `name` to `None` or a single string to label "
198 "a single legend entry with mean ROC AUC score of all curves."
199 )
200
201 # Ensure `name` is of the correct length
202 if isinstance(name, str):
203 name = [name]
204 if isinstance(name, list) and len(name) == 1:
205 name = name * n_curves
206 name = [None] * n_curves if name is None else name
207
208 # Ensure `curve_kwargs` is of correct length
209 if isinstance(curve_kwargs, Mapping):
210 curve_kwargs = [curve_kwargs] * n_curves
211
212 default_multi_curve_kwargs = {"alpha": 0.5, "linestyle": "--", "color": "blue"}
213 if curve_kwargs is None:
214 if n_curves > 1:
215 curve_kwargs = [default_multi_curve_kwargs] * n_curves
216 else:
217 curve_kwargs = [{}]
218
219 labels = []
220 if "mean" in legend_metric:
221 label_aggregate = _BinaryClassifierCurveDisplayMixin._get_legend_label(
222 legend_metric["mean"], name[0], legend_metric_name
223 )
224 # Note: "std" always `None` when "mean" is `None` - no metric value added
225 # to label in this case
226 if legend_metric["std"] is not None:
227 # Add the "+/- std" to the end (in brackets if name provided)
228 if name[0] is not None:
229 label_aggregate = (
230 label_aggregate[:-1] + f" +/- {legend_metric['std']:0.2f})"
231 )
232 else:
233 label_aggregate = (
234 label_aggregate + f" +/- {legend_metric['std']:0.2f}"
235 )
236 # Add `label` for first curve only, set to `None` for remaining curves
237 labels.extend([label_aggregate] + [None] * (n_curves - 1))
238 else:
239 for curve_legend_metric, curve_name in zip(legend_metric["metric"], name):
240 labels.append(
241 _BinaryClassifierCurveDisplayMixin._get_legend_label(
242 curve_legend_metric, curve_name, legend_metric_name
243 )
244 )
245
246 curve_kwargs_ = [
247 _validate_style_kwargs({"label": label}, curve_kwargs[fold_idx])
248 for fold_idx, label in enumerate(labels)
249 ]

253def _validate_score_name(score_name, scoring, negate_score):
254 """Validate the `score_name` parameter.
255
256 If `score_name` is provided, we just return it as-is.
257 If `score_name` is `None`, we use `Score` if `negate_score` is `False` and
258 `Negative score` otherwise.
259 If `score_name` is a string or a callable, we infer the name. We replace `_` by
260 spaces and capitalize the first letter. We remove `neg_` and replace it by
261 `"Negative"` if `negate_score` is `False` or just remove it otherwise.
262 """
263 if score_name is not None:
264 return score_name
265 elif scoring is None:
266 return "Negative score" if negate_score else "Score"
267 else:
268 score_name = scoring.__name__ if callable(scoring) else scoring
269 if negate_score:
270 if score_name.startswith("neg_"):
271 score_name = score_name[4:]
272 else:
273 score_name = f"Negative {score_name}"
274 elif score_name.startswith("neg_"):
275 score_name = f"Negative {score_name[4:]}"
276 score_name = score_name.replace("_", " ")
277 return score_name.capitalize()

280def _interval_max_min_ratio(data):
281 """Compute the ratio between the largest and smallest inter-point distances.
282
283 A value larger than 5 typically indicates that the parameter range would
284 better be displayed with a log scale while a linear scale would be more
285 suitable otherwise.
286 """
287 diff = np.diff(np.sort(data))
288 return diff.max() / diff.min()

291def _validate_style_kwargs(default_style_kwargs, user_style_kwargs):
292 """Create valid style kwargs by avoiding Matplotlib alias errors.
293
294 Matplotlib raises an error when, for example, 'color' and 'c', or 'linestyle' and
295 'ls', are specified together. To avoid this, we automatically keep only the one
296 specified by the user and raise an error if the user specifies both.
297
298 Parameters
299 ----------
300 default_style_kwargs : dict
301 The Matplotlib style kwargs used by default in the scikit-learn display.
302 user_style_kwargs : dict
303 The user-defined Matplotlib style kwargs.
304
305 Returns
306 -------
307 valid_style_kwargs : dict
308 The validated style kwargs taking into account both default and user-defined
309 Matplotlib style kwargs.
310 """
311
312 invalid_to_valid_kw = {
313 "ls": "linestyle",
314 "c": "color",
315 "ec": "edgecolor",
316 "fc": "facecolor",
317 "lw": "linewidth",
318 "mec": "markeredgecolor",
319 "mfcalt": "markerfacecoloralt",
320 "ms": "markersize",
321 "mew": "markeredgewidth",
322 "mfc": "markerfacecolor",
323 "aa": "antialiased",
324 "ds": "drawstyle",
325 "font": "fontproperties",
326 "family": "fontfamily",
327 "name": "fontname",
328 "size": "fontsize",
329 "stretch": "fontstretch",
330 "style": "fontstyle",
331 "variant": "fontvariant",
332 "weight": "fontweight",
333 "ha": "horizontalalignment",
334 "va": "verticalalignment",
335 "ma": "multialignment",
336 }
337 for invalid_key, valid_key in invalid_to_valid_kw.items():
338 if invalid_key in user_style_kwargs and valid_key in user_style_kwargs:
339 raise TypeError(
340 f"Got both {invalid_key} and {valid_key}, which are aliases of one "
341 "another"
342 )
343 valid_style_kwargs = default_style_kwargs.copy()
344
345 for key in user_style_kwargs.keys():
346 if key in invalid_to_valid_kw:
347 valid_style_kwargs[invalid_to_valid_kw[key]] = user_style_kwargs[key]
348 else:
349 valid_style_kwargs[key] = user_style_kwargs[key]
350
351 return valid_style_kwargs
