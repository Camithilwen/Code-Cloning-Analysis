593def test_as_euler_asymmetric_axes():
594 # helper function for mean error tests
595 def test_stats(error, mean_max, rms_max):
596 mean = np.mean(error, axis=0)
597 std = np.std(error, axis=0)
598 rms = np.hypot(mean, std)
599 assert np.all(np.abs(mean) < mean_max)
600 assert np.all(rms < rms_max)
601
602 rnd = np.random.RandomState(0)
603 n = 1000
604 angles = np.empty((n, 3))
605 angles[:, 0] = rnd.uniform(low=-np.pi, high=np.pi, size=(n,))
606 angles[:, 1] = rnd.uniform(low=-np.pi / 2, high=np.pi / 2, size=(n,))
607 angles[:, 2] = rnd.uniform(low=-np.pi, high=np.pi, size=(n,))
608
609 for seq_tuple in permutations('xyz'):
610 # Extrinsic rotations
611 seq = ''.join(seq_tuple)
612 rotation = Rotation.from_euler(seq, angles)
613 angles_quat = rotation.as_euler(seq)
614 angles_mat = rotation._as_euler_from_matrix(seq)

1282 key_quats = key_rots.as_quat()

1285 interpolator = Slerp(key_times, key_rots)
1286

1289 interp_quats = interp_rots.as_quat()
1290
1291 # Dot products are affected by sign of quaternions
1292 interp_quats[interp_quats[:, -1] < 0] *= -1
1293 # Checking for quaternion equality, perform same operation
1294 key_quats[key_quats[:, -1] < 0] *= -1

1297 assert_allclose(interp_quats[0], key_quats[0])

1299 assert_allclose(interp_quats[5], key_quats[2])
1300 assert_allclose(interp_quats[7], key_quats[3])
1301 assert_allclose(interp_quats[10], key_quats[4])
1302
1303 # Constant angular velocity between keyframes. Check by equating
1304 # cos(theta) between quaternion pairs with equal time difference.
1305 cos_theta1 = np.sum(interp_quats[0] * interp_quats[2])
1306 cos_theta2 = np.sum(interp_quats[2] * interp_quats[1])
1307 assert_allclose(cos_theta1, cos_theta2)
1308

1310 cos_theta5 = np.sum(interp_quats[4] * interp_quats[5])
1311 assert_allclose(cos_theta4, cos_theta5)

1314 # Use double angle formula for double the time difference
