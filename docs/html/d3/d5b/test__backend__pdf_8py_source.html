234
235 test_url = 'https://test_text_urls.matplotlib.org/'
236
237 fig = plt.figure(figsize=(2, 1))

242 fig.savefig(fd, format='pdf')

244 with pikepdf.Pdf.open(fd) as pdf:

247 # Iteration over Annots must occur within the context manager,

371def test_glyphs_subset():
372 fpath = str(_get_data_path("fonts/ttf/DejaVuSerif.ttf"))
373 chars = "these should be subsetted! 1234567890"
374
375 # non-subsetted FT2Font
376 nosubfont = FT2Font(fpath)
377 nosubfont.set_text(chars)
378
379 # subsetted FT2Font
380 with get_glyphs_subset(fpath, chars) as subset:
381 subfont = FT2Font(font_as_file(subset))
382 subfont.set_text(chars)
383
384 nosubcmap = nosubfont.get_charmap()
385 subcmap = subfont.get_charmap()
386
387 # all unique chars must be available in subsetted font
388 assert {*chars} == {chr(key) for key in subcmap}
389
390 # subsetted font's charmap should have less entries
391 assert len(subcmap) < len(nosubcmap)
392
393 # since both objects are assigned same characters
394 assert subfont.get_num_glyphs() == nosubfont.get_num_glyphs()
