132def test_transform_target_regressor_1d_transformer(X, y):
133 # All transformer in scikit-learn expect 2D data. FunctionTransformer with
134 # validate=False lift this constraint without checking that the input is a
135 # 2D vector. We check the consistency of the data shape using a 1D and 2D y
136 # array.
137 transformer = FunctionTransformer(
138 func=lambda x: x + 1, inverse_func=lambda x: x - 1
139 )
140 regr = TransformedTargetRegressor(
141 regressor=LinearRegression(), transformer=transformer
142 )
143 y_pred = regr.fit(X, y).predict(X)
144 assert y.shape == y_pred.shape
145 # consistency forward transform
146 y_tran = regr.transformer_.transform(y)
147 _check_shifted_by_one(y, y_tran)
148 assert y.shape == y_pred.shape
149 # consistency inverse transform
150 assert_allclose(y, regr.transformer_.inverse_transform(y_tran).squeeze())
151 # consistency of the regressor
152 lr = LinearRegression()
153 transformer2 = clone(transformer)
154 lr.fit(X, transformer2.fit_transform(y))
155 y_lr_pred = lr.predict(X)
156 assert_allclose(y_pred, transformer2.inverse_transform(y_lr_pred))
157 assert_allclose(regr.regressor_.coef_, lr.coef_)
