283 def _check_barycentric_transforms(self, tri, err_msg="",
284 unit_cube=False,
285 unit_cube_tol=0):
286 """Check that a triangulation has reasonable barycentric transforms"""
287 vertices = tri.points[tri.simplices]
288 sc = 1/(tri.ndim + 1.0)
289 centroids = vertices.sum(axis=1) * sc
290
291 # Either: (i) the simplex has a `nan` barycentric transform,
292 # or, (ii) the centroid is in the simplex
293
294 def barycentric_transform(tr, x):
295 r = tr[:,-1,:]
296 Tinv = tr[:,:-1,:]
297 return np.einsum('ijk,ik->ij', Tinv, x - r)
298
299 eps = np.finfo(float).eps
300
301 c = barycentric_transform(tri.transform, centroids)
302 with np.errstate(invalid="ignore"):
303 ok = np.isnan(c).all(axis=1) | (abs(c - sc)/sc < 0.1).all(axis=1)
304
305 assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
306
307 # Invalid simplices must be (nearly) zero volume
308 q = vertices[:,:-1,:] - vertices[:,-1,None,:]
309 volume = np.array([np.linalg.det(q[k,:,:])
310 for k in range(tri.nsimplex)])
311 ok = np.isfinite(tri.transform[:,0,0]) | (volume < np.sqrt(eps))
312 assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
313
314 # Also, find_simplex for the centroid should end up in some
315 # simplex for the non-degenerate cases
316 j = tri.find_simplex(centroids)
317 ok = (j != -1) | np.isnan(tri.transform[:,0,0])
318 assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")
319
320 if unit_cube:
321 # If in unit cube, no interior point should be marked out of hull
322 at_boundary = (centroids <= unit_cube_tol).any(axis=1)
323 at_boundary |= (centroids >= 1 - unit_cube_tol).any(axis=1)
324
325 ok = (j != -1) | at_boundary
326 assert_(ok.all(), f"{err_msg} {np.nonzero(~ok)}")

550 if facets_1 != facets_2 and points.shape[1] == 2:

552 # --- then the convex hull from Delaunay differs (due
553 # to rounding error etc.) from the hull computed
554 # otherwise, by the question whether (tricoplanar)
555 # points that lie almost exactly on the hull are
556 # included as vertices of the hull or not.
557 #
558 # So we check the result, and accept it if the Delaunay
559 # hull line segments are a subset of the usual hull.
560
561 eps = 1000 * np.finfo(float).eps
562
563 for a, b in facets_1:
564 for ap, bp in facets_2:
565 t = points[bp] - points[ap]
566 t /= np.linalg.norm(t) # tangent
567 n = np.array([-t[1], t[0]]) # normal
568
569 # check that the two line segments are parallel
570 # to the same line
571 c1 = np.dot(n, points[b] - points[ap])
572 c2 = np.dot(n, points[a] - points[ap])
573 if not np.allclose(np.dot(c1, n), 0):
574 continue
575 if not np.allclose(np.dot(c2, n), 0):
576 continue
577
578 # Check that the segment (a, b) is contained in (ap, bp)
579 c1 = np.dot(t, points[a] - points[ap])
580 c2 = np.dot(t, points[b] - points[ap])
581 c3 = np.dot(t, points[bp] - points[ap])
582 if c1 < -eps or c1 > c3 + eps:
583 continue
584 if c2 < -eps or c2 > c3 + eps:
585 continue
586
587 # OK:
588 break
589 else:
590 raise AssertionError("comparison fails")
591
592 # it was OK
593 return
594
595 assert_equal(facets_1, facets_2)
