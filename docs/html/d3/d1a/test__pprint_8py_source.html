628def test_kwargs_in_init():
629 # Make sure the changed_only=True mode is OK when an argument is passed as
630 # kwargs.
631 # Non-regression test for
632 # https://github.com/scikit-learn/scikit-learn/issues/17206
633
634 class WithKWargs(BaseEstimator):
635 # Estimator with a kwargs argument. These need to hack around
636 # set_params and get_params. Here we mimic what LightGBM does.
637 def __init__(self, a="willchange", b="unchanged", **kwargs):
638 self.a = a
639 self.b = b
640 self._other_params = {}
641 self.set_params(**kwargs)
642
643 def get_params(self, deep=True):
644 params = super().get_params(deep=deep)
645 params.update(self._other_params)
646 return params
647
648 def set_params(self, **params):
649 for key, value in params.items():
650 setattr(self, key, value)
651 self._other_params[key] = value
652 return self
653
