49def _retry_with_clean_cache(
50 openml_path: str,
51 data_home: Optional[str],
52 no_retry_exception: Optional[Exception] = None,
53) -> Callable:
54 """If the first call to the decorated function fails, the local cached
55 file is removed, and the function is called again. If ``data_home`` is
56 ``None``, then the function is called once. We can provide a specific
57 exception to not retry on using `no_retry_exception` parameter.
58 """
59
60 def decorator(f):
61 @wraps(f)
62 def wrapper(*args, **kw):
63 if data_home is None:
64 return f(*args, **kw)
65 try:
66 return f(*args, **kw)
67 except URLError:
68 raise
69 except Exception as exc:
70 if no_retry_exception is not None and isinstance(
71 exc, no_retry_exception
72 ):
73 raise
74 warn("Invalid cache, redownloading file", RuntimeWarning)
75 local_path = _get_local_path(openml_path, data_home)
76 if os.path.exists(local_path):
77 os.unlink(local_path)
78 return f(*args, **kw)
79
80 return wrapper
81
82 return decorator

85def _retry_on_network_error(
86 n_retries: int = 3, delay: float = 1.0, url: str = ""
87) -> Callable:
88 """If the function call results in a network error, call the function again
89 up to ``n_retries`` times with a ``delay`` between each call. If the error
90 has a 412 status code, don't call the function again as this is a specific
91 OpenML error.
92 The url parameter is used to give more information to the user about the
93 error.
94 """
95
96 def decorator(f):
97 @wraps(f)
98 def wrapper(*args, **kwargs):
99 retry_counter = n_retries
100 while True:
101 try:
102 return f(*args, **kwargs)
103 except (URLError, TimeoutError) as e:
104 # 412 is a specific OpenML error code.
105 if isinstance(e, HTTPError) and e.code == 412:
106 raise
107 if retry_counter == 0:
108 raise
109 warn(
110 f"A network error occurred while downloading {url}. Retrying..."
111 )
112 retry_counter -= 1
113 time.sleep(delay)
114
115 return wrapper
116
117 return decorator

120def _open_openml_url(
121 url: str, data_home: Optional[str], n_retries: int = 3, delay: float = 1.0
122):
123 """
124 Returns a resource from OpenML.org. Caches it to data_home if required.
125
126 Parameters
127 ----------
128 url : str
129 OpenML URL that will be downloaded and cached locally. The path component
130 of the URL is used to replicate the tree structure as sub-folders of the local
131 cache folder.
132
133 data_home : str
134 Directory to which the files will be cached. If None, no caching will
135 be applied.
136
137 n_retries : int, default=3
138 Number of retries when HTTP errors are encountered. Error with status
139 code 412 won't be retried as they represent OpenML generic errors.
140
141 delay : float, default=1.0
142 Number of seconds between retries.
143
144 Returns
145 -------
146 result : stream
147 A stream to the OpenML resource.
148 """
149
150 def is_gzip_encoded(_fsrc):
151 return _fsrc.info().get("Content-Encoding", "") == "gzip"
152
153 req = Request(url)
154 req.add_header("Accept-encoding", "gzip")
155
156 if data_home is None:
157 fsrc = _retry_on_network_error(n_retries, delay, req.full_url)(urlopen)(req)
158 if is_gzip_encoded(fsrc):
159 return gzip.GzipFile(fileobj=fsrc, mode="rb")
160 return fsrc
161
162 openml_path = urlparse(url).path.lstrip("/")
163 local_path = _get_local_path(openml_path, data_home)
164 dir_name, file_name = os.path.split(local_path)
165 if not os.path.exists(local_path):
166 os.makedirs(dir_name, exist_ok=True)
167 try:
168 # Create a tmpdir as a subfolder of dir_name where the final file will
169 # be moved to if the download is successful. This guarantees that the
170 # renaming operation to the final location is atomic to ensure the
171 # concurrence safety of the dataset caching mechanism.
172 with TemporaryDirectory(dir=dir_name) as tmpdir:
173 with closing(
174 _retry_on_network_error(n_retries, delay, req.full_url)(urlopen)(
175 req
176 )
177 ) as fsrc:
178 opener: Callable
179 if is_gzip_encoded(fsrc):
180 opener = open
181 else:
182 opener = gzip.GzipFile
183 with opener(os.path.join(tmpdir, file_name), "wb") as fdst:
184 shutil.copyfileobj(fsrc, fdst)
185 shutil.move(fdst.name, local_path)
186 except Exception:
187 if os.path.exists(local_path):
188 os.unlink(local_path)
189 raise
190
191 # XXX: First time, decompression will not be necessary (by using fsrc), but
192 # it will happen nonetheless
193 return gzip.GzipFile(local_path, "rb")

202def _get_json_content_from_openml_api(
203 url: str,
204 error_message: Optional[str],
205 data_home: Optional[str],
206 n_retries: int = 3,
207 delay: float = 1.0,
208) -> Dict:
209 """
210 Loads json data from the openml api.
211
212 Parameters
213 ----------
214 url : str
215 The URL to load from. Should be an official OpenML endpoint.
216
217 error_message : str or None
218 The error message to raise if an acceptable OpenML error is thrown
219 (acceptable error is, e.g., data id not found. Other errors, like 404's
220 will throw the native error message).
221
222 data_home : str or None
223 Location to cache the response. None if no cache is required.
224
225 n_retries : int, default=3
226 Number of retries when HTTP errors are encountered. Error with status
227 code 412 won't be retried as they represent OpenML generic errors.
228
229 delay : float, default=1.0
230 Number of seconds between retries.
231
232 Returns
233 -------
234 json_data : json
235 the json result from the OpenML server if the call was successful.
236 An exception otherwise.
237 """
238
239 @_retry_with_clean_cache(url, data_home=data_home)
240 def _load_json():
241 with closing(
242 _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
243 ) as response:
244 return json.loads(response.read().decode("utf-8"))
245
246 try:
247 return _load_json()
248 except HTTPError as error:
249 # 412 is an OpenML specific error code, indicating a generic error
250 # (e.g., data not found)
251 if error.code != 412:
252 raise error
253
254 # 412 error, not in except for nicer traceback
255 raise OpenMLError(error_message)

258def _get_data_info_by_name(
259 name: str,
260 version: Union[int, str],
261 data_home: Optional[str],
262 n_retries: int = 3,
263 delay: float = 1.0,
264):
265 """
266 Utilizes the openml dataset listing api to find a dataset by
267 name/version
268 OpenML api function:
269 https://www.openml.org/api_docs#!/data/get_data_list_data_name_data_name
270
271 Parameters
272 ----------
273 name : str
274 name of the dataset
275
276 version : int or str
277 If version is an integer, the exact name/version will be obtained from
278 OpenML. If version is a string (value: "active") it will take the first
279 version from OpenML that is annotated as active. Any other string
280 values except "active" are treated as integer.
281
282 data_home : str or None
283 Location to cache the response. None if no cache is required.
284
285 n_retries : int, default=3
286 Number of retries when HTTP errors are encountered. Error with status
287 code 412 won't be retried as they represent OpenML generic errors.
288
289 delay : float, default=1.0
290 Number of seconds between retries.
291
292 Returns
293 -------
294 first_dataset : json
295 json representation of the first dataset object that adhired to the
296 search criteria
297
298 """
299 if version == "active":
300 # situation in which we return the oldest active version
301 url = _SEARCH_NAME.format(name) + "/status/active/"
302 error_msg = "No active dataset {} found.".format(name)
303 json_data = _get_json_content_from_openml_api(
304 url,
305 error_msg,
306 data_home=data_home,
307 n_retries=n_retries,
308 delay=delay,
309 )
310 res = json_data["data"]["dataset"]
311 if len(res) > 1:
312 first_version = version = res[0]["version"]
313 warning_msg = (
314 "Multiple active versions of the dataset matching the name"
315 f" {name} exist. Versions may be fundamentally different, "
316 f"returning version {first_version}. "
317 "Available versions:\n"
318 )
319 for r in res:
320 warning_msg += f"- version {r['version']}, status: {r['status']}\n"
321 warning_msg += (
322 f" url: https://www.openml.org/search?type=data&id={r['did']}\n"
323 )
324 warn(warning_msg)
325 return res[0]
326
327 # an integer version has been provided
328 url = (_SEARCH_NAME + "/data_version/{}").format(name, version)
329 try:
330 json_data = _get_json_content_from_openml_api(
331 url,
332 error_message=None,
333 data_home=data_home,
334 n_retries=n_retries,
335 delay=delay,
336 )
337 except OpenMLError:
338 # we can do this in 1 function call if OpenML does not require the
339 # specification of the dataset status (i.e., return datasets with a
340 # given name / version regardless of active, deactivated, etc. )
341 # TODO: feature request OpenML.
342 url += "/status/deactivated"
343 error_msg = "Dataset {} with version {} not found.".format(name, version)
344 json_data = _get_json_content_from_openml_api(
345 url,
346 error_msg,
347 data_home=data_home,
348 n_retries=n_retries,
349 delay=delay,
350 )
351
352 return json_data["data"]["dataset"][0]

437def _load_arff_response(
438 url: str,
439 data_home: Optional[str],
440 parser: str,
441 output_type: str,
442 openml_columns_info: dict,
443 feature_names_to_select: List[str],
444 target_names_to_select: List[str],
445 shape: Optional[Tuple[int, int]],
446 md5_checksum: str,
447 n_retries: int = 3,
448 delay: float = 1.0,
449 read_csv_kwargs: Optional[Dict] = None,
450):
451 """Load the ARFF data associated with the OpenML URL.
452
453 In addition of loading the data, this function will also check the
454 integrity of the downloaded file from OpenML using MD5 checksum.
455
456 Parameters
457 ----------
458 url : str
459 The URL of the ARFF file on OpenML.
460
461 data_home : str
462 The location where to cache the data.
463
464 parser : {"liac-arff", "pandas"}
465 The parser used to parse the ARFF file.
466
467 output_type : {"numpy", "pandas", "sparse"}
468 The type of the arrays that will be returned. The possibilities are:
469
470 - `"numpy"`: both `X` and `y` will be NumPy arrays;
471 - `"sparse"`: `X` will be sparse matrix and `y` will be a NumPy array;
472 - `"pandas"`: `X` will be a pandas DataFrame and `y` will be either a
473 pandas Series or DataFrame.
474
475 openml_columns_info : dict
476 The information provided by OpenML regarding the columns of the ARFF
477 file.
478
479 feature_names_to_select : list of str
480 The list of the features to be selected.
481
482 target_names_to_select : list of str
483 The list of the target variables to be selected.
484
485 shape : tuple or None
486 With `parser="liac-arff"`, when using a generator to load the data,
487 one needs to provide the shape of the data beforehand.
488
489 md5_checksum : str
490 The MD5 checksum provided by OpenML to check the data integrity.
491
492 n_retries : int, default=3
493 The number of times to retry downloading the data if it fails.
494
495 delay : float, default=1.0
496 The delay between two consecutive downloads in seconds.
497
498 read_csv_kwargs : dict, default=None
499 Keyword arguments to pass to `pandas.read_csv` when using the pandas parser.
500 It allows to overwrite the default options.
501
502 .. versionadded:: 1.3
503
504 Returns
505 -------
506 X : {ndarray, sparse matrix, dataframe}
507 The data matrix.
508
509 y : {ndarray, dataframe, series}
510 The target.
511
512 frame : dataframe or None
513 A dataframe containing both `X` and `y`. `None` if
514 `output_array_type != "pandas"`.
515
516 categories : list of str or None
517 The names of the features that are categorical. `None` if
518 `output_array_type == "pandas"`.
519 """
520 gzip_file = _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
521 with closing(gzip_file):
522 md5 = hashlib.md5()
523 for chunk in iter(lambda: gzip_file.read(4096), b""):
524 md5.update(chunk)
525 actual_md5_checksum = md5.hexdigest()
526
527 if actual_md5_checksum != md5_checksum:
528 raise ValueError(
529 f"md5 checksum of local file for {url} does not match description: "
530 f"expected: {md5_checksum} but got {actual_md5_checksum}. "
531 "Downloaded file could have been modified / corrupted, clean cache "
532 "and retry..."
533 )

535 def _open_url_and_load_gzip_file(url, data_home, n_retries, delay, arff_params):
536 gzip_file = _open_openml_url(url, data_home, n_retries=n_retries, delay=delay)
537 with closing(gzip_file):
538 return load_arff_from_gzip_file(gzip_file, **arff_params)
539
540 arff_params: Dict = dict(
541 parser=parser,
542 output_type=output_type,
543 openml_columns_info=openml_columns_info,
544 feature_names_to_select=feature_names_to_select,
545 target_names_to_select=target_names_to_select,
546 shape=shape,
547 read_csv_kwargs=read_csv_kwargs or {},
548 )
549 try:
550 X, y, frame, categories = _open_url_and_load_gzip_file(
551 url, data_home, n_retries, delay, arff_params
552 )
553 except Exception as exc:
554 if parser != "pandas":
555 raise
556
557 from pandas.errors import ParserError
558
559 if not isinstance(exc, ParserError):
560 raise
561
562 # A parsing error could come from providing the wrong quotechar
563 # to pandas. By default, we use a double quote. Thus, we retry
564 # with a single quote before to raise the error.
565 arff_params["read_csv_kwargs"].update(quotechar="'")
566 X, y, frame, categories = _open_url_and_load_gzip_file(
567 url, data_home, n_retries, delay, arff_params
568 )
569
570 return X, y, frame, categories

675 )
676

678 if parser == "pandas":
679 # If we get a ParserError with pandas, then we don't want to retry and we raise
680 # early.
681 from pandas.errors import ParserError
682
683 no_retry_exception = ParserError
684
685 X, y, frame, categories = _retry_with_clean_cache(
686 url, data_home, no_retry_exception
687 )(_load_arff_response)(
688 url,
689 data_home,
690 parser=parser,
691 output_type=output_type,
692 openml_columns_info=features_dict,
693 feature_names_to_select=data_columns,
694 target_names_to_select=target_columns,
695 shape=shape,
696 md5_checksum=md5_checksum,
697 n_retries=n_retries,
698 delay=delay,
699 read_csv_kwargs=read_csv_kwargs,
700 )
701
702 return Bunch(
703 data=X,
704 target=y,
705 frame=frame,
706 categories=categories,
707 feature_names=data_columns,
708 target_names=target_columns,

712def _verify_target_data_type(features_dict, target_columns):
713 # verifies the data type of the y array in case there are multiple targets
714 # (throws an error if these targets do not comply with sklearn support)
715 if not isinstance(target_columns, list):
716 raise ValueError("target_column should be list, got: %s" % type(target_columns))
717 found_types = set()
718 for target_column in target_columns:
719 if target_column not in features_dict:
720 raise KeyError(f"Could not find target_column='{target_column}'")
721 if features_dict[target_column]["data_type"] == "numeric":
722 found_types.add(np.float64)
723 else:
724 found_types.add(object)
725
726 # note: we compare to a string, not boolean
727 if features_dict[target_column]["is_ignore"] == "true":
728 warn(f"target_column='{target_column}' has flag is_ignore.")
729 if features_dict[target_column]["is_row_identifier"] == "true":
730 warn(f"target_column='{target_column}' has flag is_row_identifier.")
731 if len(found_types) > 1:
732 raise ValueError(
733 "Can only handle homogeneous multi-target datasets, "
734 "i.e., all targets are either numeric or "
735 "categorical."
736 )
