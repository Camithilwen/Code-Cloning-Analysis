35# Some notes on function parameters:
36#
37# `cutoff` and `width` are given as numbers between 0 and 1. These are
38# relative frequencies, expressed as a fraction of the Nyquist frequency.
39# For example, if the Nyquist frequency is 2 KHz, then width=0.15 is a width
40# of 300 Hz.
41#
42# The `order` of a FIR filter is one less than the number of taps.
43# This is a potential source of confusion, so in the following code,
44# we will always use the number of taps as the parameterization of
45# the 'size' of the filter. The "number of taps" means the number
46# of coefficients, which is the same as the length of the impulse
47# response of the filter.

50def kaiser_beta(a):
51 """Compute the Kaiser parameter `beta`, given the attenuation `a`.
52
53 Parameters
54 ----------
55 a : float
56 The desired attenuation in the stopband and maximum ripple in
57 the passband, in dB. This should be a *positive* number.
58
59 Returns
60 -------
61 beta : float
62 The `beta` parameter to be used in the formula for a Kaiser window.
63
64 References
65 ----------
66 Oppenheim, Schafer, "Discrete-Time Signal Processing", p.475-476.
67
68 Examples
69 --------
70 Suppose we want to design a lowpass filter, with 65 dB attenuation
71 in the stop band. The Kaiser window parameter to be used in the
72 window method is computed by ``kaiser_beta(65)``:
73
74 >>> from scipy.signal import kaiser_beta
75 >>> kaiser_beta(65)
76 6.20426
77
78 """
79 if a > 50:
80 beta = 0.1102 * (a - 8.7)
81 elif a > 21:
82 beta = 0.5842 * (a - 21) ** 0.4 + 0.07886 * (a - 21)
83 else:
84 beta = 0.0
85 return beta

88def kaiser_atten(numtaps, width):
89 """Compute the attenuation of a Kaiser FIR filter.
90
91 Given the number of taps `N` and the transition width `width`, compute the
92 attenuation `a` in dB, given by Kaiser's formula:
93
94 a = 2.285 * (N - 1) * pi * width + 7.95
95
96 Parameters
97 ----------
98 numtaps : int
99 The number of taps in the FIR filter.
100 width : float
101 The desired width of the transition region between passband and
102 stopband (or, in general, at any discontinuity) for the filter,
103 expressed as a fraction of the Nyquist frequency.
104
105 Returns
106 -------
107 a : float
108 The attenuation of the ripple, in dB.
109
110 See Also
111 --------
112 kaiserord, kaiser_beta
113
114 Examples
115 --------
116 Suppose we want to design a FIR filter using the Kaiser window method
117 that will have 211 taps and a transition width of 9 Hz for a signal that
118 is sampled at 480 Hz. Expressed as a fraction of the Nyquist frequency,
119 the width is 9/(0.5*480) = 0.0375. The approximate attenuation (in dB)
120 is computed as follows:
121
122 >>> from scipy.signal import kaiser_atten
123 >>> kaiser_atten(211, 0.0375)
124 64.48099630593983
125
126 """
127 a = 2.285 * (numtaps - 1) * np.pi * width + 7.95
128 return a

131def kaiserord(ripple, width):
132 """
133 Determine the filter window parameters for the Kaiser window method.
134
135 The parameters returned by this function are generally used to create
136 a finite impulse response filter using the window method, with either
137 `firwin` or `firwin2`.
138
139 Parameters
140 ----------
141 ripple : float
142 Upper bound for the deviation (in dB) of the magnitude of the
143 filter's frequency response from that of the desired filter (not
144 including frequencies in any transition intervals). That is, if w
145 is the frequency expressed as a fraction of the Nyquist frequency,
146 A(w) is the actual frequency response of the filter and D(w) is the
147 desired frequency response, the design requirement is that::
148
149 abs(A(w) - D(w))) < 10**(-ripple/20)
150
151 for 0 <= w <= 1 and w not in a transition interval.
152 width : float
153 Width of transition region, normalized so that 1 corresponds to pi
154 radians / sample. That is, the frequency is expressed as a fraction
155 of the Nyquist frequency.
156
157 Returns
158 -------
159 numtaps : int
160 The length of the Kaiser window.
161 beta : float
162 The beta parameter for the Kaiser window.
163
164 See Also
165 --------
166 kaiser_beta, kaiser_atten
167
168 Notes
169 -----
170 There are several ways to obtain the Kaiser window:
171
172 - ``signal.windows.kaiser(numtaps, beta, sym=True)``
173 - ``signal.get_window(beta, numtaps)``
174 - ``signal.get_window(('kaiser', beta), numtaps)``
175
176 The empirical equations discovered by Kaiser are used.
177
178 References
179 ----------
180 Oppenheim, Schafer, "Discrete-Time Signal Processing", pp.475-476.
181
182 Examples
183 --------
184 We will use the Kaiser window method to design a lowpass FIR filter
185 for a signal that is sampled at 1000 Hz.
186
187 We want at least 65 dB rejection in the stop band, and in the pass
188 band the gain should vary no more than 0.5%.
189
190 We want a cutoff frequency of 175 Hz, with a transition between the
191 pass band and the stop band of 24 Hz. That is, in the band [0, 163],
192 the gain varies no more than 0.5%, and in the band [187, 500], the
193 signal is attenuated by at least 65 dB.
194
195 >>> import numpy as np
196 >>> from scipy.signal import kaiserord, firwin, freqz
197 >>> import matplotlib.pyplot as plt
198 >>> fs = 1000.0
199 >>> cutoff = 175
200 >>> width = 24
201
202 The Kaiser method accepts just a single parameter to control the pass
203 band ripple and the stop band rejection, so we use the more restrictive
204 of the two. In this case, the pass band ripple is 0.005, or 46.02 dB,
205 so we will use 65 dB as the design parameter.
206
207 Use `kaiserord` to determine the length of the filter and the
208 parameter for the Kaiser window.
209
210 >>> numtaps, beta = kaiserord(65, width/(0.5*fs))
211 >>> numtaps
212 167
213 >>> beta
214 6.20426
215
216 Use `firwin` to create the FIR filter.
217
218 >>> taps = firwin(numtaps, cutoff, window=('kaiser', beta),
219 ... scale=False, fs=fs)
220
221 Compute the frequency response of the filter. ``w`` is the array of
222 frequencies, and ``h`` is the corresponding complex array of frequency
223 responses.
224
225 >>> w, h = freqz(taps, worN=8000)
226 >>> w *= 0.5*fs/np.pi # Convert w to Hz.
227
228 Compute the deviation of the magnitude of the filter's response from
229 that of the ideal lowpass filter. Values in the transition region are
230 set to ``nan``, so they won't appear in the plot.
231
232 >>> ideal = w < cutoff # The "ideal" frequency response.
233 >>> deviation = np.abs(np.abs(h) - ideal)
234 >>> deviation[(w > cutoff - 0.5*width) & (w < cutoff + 0.5*width)] = np.nan
235
236 Plot the deviation. A close look at the left end of the stop band shows
237 that the requirement for 65 dB attenuation is violated in the first lobe
238 by about 0.125 dB. This is not unusual for the Kaiser window method.
239
240 >>> plt.plot(w, 20*np.log10(np.abs(deviation)))
241 >>> plt.xlim(0, 0.5*fs)
242 >>> plt.ylim(-90, -60)
243 >>> plt.grid(alpha=0.25)
244 >>> plt.axhline(-65, color='r', ls='--', alpha=0.3)
245 >>> plt.xlabel('Frequency (Hz)')
246 >>> plt.ylabel('Deviation from ideal (dB)')
247 >>> plt.title('Lowpass Filter Frequency Response')
248 >>> plt.show()
249
250 """
251 A = abs(ripple) # in case somebody is confused as to what's meant
252 if A < 8:
253 # Formula for N is not valid in this range.
254 raise ValueError("Requested maximum ripple attentuation %f is too "
255 "small for the Kaiser formula." % A)
256 beta = kaiser_beta(A)
257
258 # Kaiser's formula (as given in Oppenheim and Schafer) is for the filter
259 # order, so we have to add 1 to get the number of taps.
260 numtaps = (A - 7.95) / 2.285 / (np.pi * width) + 1
261
262 return int(ceil(numtaps)), beta

265def firwin(numtaps, cutoff, width=None, window='hamming', pass_zero=True,
266 scale=True, nyq=None, fs=None):
267 """
268 FIR filter design using the window method.
269
270 This function computes the coefficients of a finite impulse response
271 filter. The filter will have linear phase; it will be Type I if
272 `numtaps` is odd and Type II if `numtaps` is even.
273
274 Type II filters always have zero response at the Nyquist frequency, so a
275 ValueError exception is raised if firwin is called with `numtaps` even and
276 having a passband whose right end is at the Nyquist frequency.
277
278 Parameters
279 ----------
280 numtaps : int
281 Length of the filter (number of coefficients, i.e. the filter
282 order + 1). `numtaps` must be odd if a passband includes the
283 Nyquist frequency.
284 cutoff : float or 1-D array_like
285 Cutoff frequency of filter (expressed in the same units as `fs`)
286 OR an array of cutoff frequencies (that is, band edges). In the
287 latter case, the frequencies in `cutoff` should be positive and
288 monotonically increasing between 0 and `fs/2`. The values 0 and
289 `fs/2` must not be included in `cutoff`.
290 width : float or None, optional
291 If `width` is not None, then assume it is the approximate width
292 of the transition region (expressed in the same units as `fs`)
293 for use in Kaiser FIR filter design. In this case, the `window`
294 argument is ignored.
295 window : string or tuple of string and parameter values, optional
296 Desired window to use. See `scipy.signal.get_window` for a list
297 of windows and required parameters.
298 pass_zero : {True, False, 'bandpass', 'lowpass', 'highpass', 'bandstop'}, optional
299 If True, the gain at the frequency 0 (i.e., the "DC gain") is 1.
300 If False, the DC gain is 0. Can also be a string argument for the
301 desired filter type (equivalent to ``btype`` in IIR design functions).
302
303 .. versionadded:: 1.3.0
304 Support for string arguments.
305 scale : bool, optional
306 Set to True to scale the coefficients so that the frequency
307 response is exactly unity at a certain frequency.
308 That frequency is either:
309
310 - 0 (DC) if the first passband starts at 0 (i.e. pass_zero
311 is True)
312 - `fs/2` (the Nyquist frequency) if the first passband ends at
313 `fs/2` (i.e the filter is a single band highpass filter);
314 center of first passband otherwise
315
316 nyq : float, optional, deprecated
317 This is the Nyquist frequency. Each frequency in `cutoff` must be
318 between 0 and `nyq`. Default is 1.
319
320 .. deprecated:: 1.0.0
321 `firwin` keyword argument `nyq` is deprecated in favour of `fs` and
322 will be removed in SciPy 1.12.0.
323 fs : float, optional
324 The sampling frequency of the signal. Each frequency in `cutoff`
325 must be between 0 and ``fs/2``. Default is 2.
326
327 Returns
328 -------
329 h : (numtaps,) ndarray
330 Coefficients of length `numtaps` FIR filter.
331
332 Raises
333 ------
334 ValueError
335 If any value in `cutoff` is less than or equal to 0 or greater
336 than or equal to ``fs/2``, if the values in `cutoff` are not strictly
337 monotonically increasing, or if `numtaps` is even but a passband
338 includes the Nyquist frequency.
339
340 See Also
341 --------
342 firwin2
343 firls
344 minimum_phase
345 remez
346
347 Examples
348 --------
349 Low-pass from 0 to f:
350
351 >>> from scipy import signal
352 >>> numtaps = 3
353 >>> f = 0.1
354 >>> signal.firwin(numtaps, f)
355 array([ 0.06799017, 0.86401967, 0.06799017])
356
357 Use a specific window function:
358
359 >>> signal.firwin(numtaps, f, window='nuttall')
360 array([ 3.56607041e-04, 9.99286786e-01, 3.56607041e-04])
361
362 High-pass ('stop' from 0 to f):
363
364 >>> signal.firwin(numtaps, f, pass_zero=False)
365 array([-0.00859313, 0.98281375, -0.00859313])
366
367 Band-pass:
368
369 >>> f1, f2 = 0.1, 0.2
370 >>> signal.firwin(numtaps, [f1, f2], pass_zero=False)
371 array([ 0.06301614, 0.88770441, 0.06301614])
372
373 Band-stop:
374
375 >>> signal.firwin(numtaps, [f1, f2])
376 array([-0.00801395, 1.0160279 , -0.00801395])
377
378 Multi-band (passbands are [0, f1], [f2, f3] and [f4, 1]):
379
380 >>> f3, f4 = 0.3, 0.4
381 >>> signal.firwin(numtaps, [f1, f2, f3, f4])
382 array([-0.01376344, 1.02752689, -0.01376344])
383
384 Multi-band (passbands are [f1, f2] and [f3,f4]):
385
386 >>> signal.firwin(numtaps, [f1, f2, f3, f4], pass_zero=False)
387 array([ 0.04890915, 0.91284326, 0.04890915])
388
389 """ # noqa: E501
390 # The major enhancements to this function added in November 2010 were
391 # developed by Tom Krauss (see ticket #902).
392
393 nyq = 0.5 * _get_fs(fs, nyq)
394
395 cutoff = np.atleast_1d(cutoff) / float(nyq)
396
397 # Check for invalid input.
398 if cutoff.ndim > 1:
399 raise ValueError("The cutoff argument must be at most "
400 "one-dimensional.")
401 if cutoff.size == 0:
402 raise ValueError("At least one cutoff frequency must be given.")
403 if cutoff.min() <= 0 or cutoff.max() >= 1:
404 raise ValueError("Invalid cutoff frequency: frequencies must be "
405 "greater than 0 and less than fs/2.")
406 if np.any(np.diff(cutoff) <= 0):
407 raise ValueError("Invalid cutoff frequencies: the frequencies "
408 "must be strictly increasing.")
409
410 if width is not None:
411 # A width was given. Find the beta parameter of the Kaiser window
412 # and set `window`. This overrides the value of `window` passed in.
413 atten = kaiser_atten(numtaps, float(width) / nyq)
414 beta = kaiser_beta(atten)
415 window = ('kaiser', beta)
416
417 if isinstance(pass_zero, str):
418 if pass_zero in ('bandstop', 'lowpass'):
419 if pass_zero == 'lowpass':
420 if cutoff.size != 1:
421 raise ValueError('cutoff must have one element if '
422 'pass_zero=="lowpass", got %s'
423 % (cutoff.shape,))
424 elif cutoff.size <= 1:
425 raise ValueError('cutoff must have at least two elements if '
426 'pass_zero=="bandstop", got %s'
427 % (cutoff.shape,))
428 pass_zero = True
429 elif pass_zero in ('bandpass', 'highpass'):
430 if pass_zero == 'highpass':
431 if cutoff.size != 1:
432 raise ValueError('cutoff must have one element if '
433 'pass_zero=="highpass", got %s'
434 % (cutoff.shape,))
435 elif cutoff.size <= 1:
436 raise ValueError('cutoff must have at least two elements if '
437 'pass_zero=="bandpass", got %s'
438 % (cutoff.shape,))
439 pass_zero = False
440 else:
441 raise ValueError('pass_zero must be True, False, "bandpass", '
442 '"lowpass", "highpass", or "bandstop", got '
443 '{}'.format(pass_zero))
444 pass_zero = bool(operator.index(pass_zero)) # ensure bool-like
445
446 pass_nyquist = bool(cutoff.size & 1) ^ pass_zero
447 if pass_nyquist and numtaps % 2 == 0:
448 raise ValueError("A filter with an even number of coefficients must "
449 "have zero response at the Nyquist frequency.")
450
451 # Insert 0 and/or 1 at the ends of cutoff so that the length of cutoff
452 # is even, and each pair in cutoff corresponds to passband.
453 cutoff = np.hstack(([0.0] * pass_zero, cutoff, [1.0] * pass_nyquist))
454
455 # `bands` is a 2-D array; each row gives the left and right edges of
456 # a passband.
457 bands = cutoff.reshape(-1, 2)
458
459 # Build up the coefficients.
460 alpha = 0.5 * (numtaps - 1)
461 m = np.arange(0, numtaps) - alpha
462 h = 0
463 for left, right in bands:
464 h += right * sinc(right * m)
465 h -= left * sinc(left * m)
466
467 # Get and apply the window function.
468 from .windows import get_window
469 win = get_window(window, numtaps, fftbins=False)
470 h *= win
471
472 # Now handle scaling if desired.
473 if scale:
474 # Get the first passband.
475 left, right = bands[0]
476 if left == 0:
477 scale_frequency = 0.0
478 elif right == 1:
479 scale_frequency = 1.0
480 else:
481 scale_frequency = 0.5 * (left + right)
482 c = np.cos(np.pi * m * scale_frequency)
483 s = np.sum(h * c)
484 h /= s
485
486 return h

493def firwin2(numtaps, freq, gain, nfreqs=None, window='hamming', nyq=None,
494 antisymmetric=False, fs=None):
495 """
496 FIR filter design using the window method.
497
498 From the given frequencies `freq` and corresponding gains `gain`,
499 this function constructs an FIR filter with linear phase and
500 (approximately) the given frequency response.
501
502 Parameters
503 ----------
504 numtaps : int
505 The number of taps in the FIR filter. `numtaps` must be less than
506 `nfreqs`.
507 freq : array_like, 1-D
508 The frequency sampling points. Typically 0.0 to 1.0 with 1.0 being
509 Nyquist. The Nyquist frequency is half `fs`.
510 The values in `freq` must be nondecreasing. A value can be repeated
511 once to implement a discontinuity. The first value in `freq` must
512 be 0, and the last value must be ``fs/2``. Values 0 and ``fs/2`` must
513 not be repeated.
514 gain : array_like
515 The filter gains at the frequency sampling points. Certain
516 constraints to gain values, depending on the filter type, are applied,
517 see Notes for details.
518 nfreqs : int, optional
519 The size of the interpolation mesh used to construct the filter.
520 For most efficient behavior, this should be a power of 2 plus 1
521 (e.g, 129, 257, etc). The default is one more than the smallest
522 power of 2 that is not less than `numtaps`. `nfreqs` must be greater
523 than `numtaps`.
524 window : string or (string, float) or float, or None, optional
525 Window function to use. Default is "hamming". See
526 `scipy.signal.get_window` for the complete list of possible values.
527 If None, no window function is applied.
528 nyq : float, optional, deprecated
529 This is the Nyquist frequency. Each frequency in `freq` must be
530 between 0 and `nyq`. Default is 1.
531
532 .. deprecated:: 1.0.0
533 `firwin2` keyword argument `nyq` is deprecated in favour of `fs` and
534 will be removed in SciPy 1.12.0.
535 antisymmetric : bool, optional
536 Whether resulting impulse response is symmetric/antisymmetric.
537 See Notes for more details.
538 fs : float, optional
539 The sampling frequency of the signal. Each frequency in `cutoff`
540 must be between 0 and ``fs/2``. Default is 2.
541
542 Returns
543 -------
544 taps : ndarray
545 The filter coefficients of the FIR filter, as a 1-D array of length
546 `numtaps`.
547
548 See Also
549 --------
550 firls
551 firwin
552 minimum_phase
553 remez
554
555 Notes
556 -----
557 From the given set of frequencies and gains, the desired response is
558 constructed in the frequency domain. The inverse FFT is applied to the
559 desired response to create the associated convolution kernel, and the
560 first `numtaps` coefficients of this kernel, scaled by `window`, are
561 returned.
562
563 The FIR filter will have linear phase. The type of filter is determined by
564 the value of 'numtaps` and `antisymmetric` flag.
565 There are four possible combinations:
566
567 - odd `numtaps`, `antisymmetric` is False, type I filter is produced
568 - even `numtaps`, `antisymmetric` is False, type II filter is produced
569 - odd `numtaps`, `antisymmetric` is True, type III filter is produced
570 - even `numtaps`, `antisymmetric` is True, type IV filter is produced
571
572 Magnitude response of all but type I filters are subjects to following
573 constraints:
574
575 - type II -- zero at the Nyquist frequency
576 - type III -- zero at zero and Nyquist frequencies
577 - type IV -- zero at zero frequency
578
579 .. versionadded:: 0.9.0
580
581 References
582 ----------
583 .. [1] Oppenheim, A. V. and Schafer, R. W., "Discrete-Time Signal
584 Processing", Prentice-Hall, Englewood Cliffs, New Jersey (1989).
585 (See, for example, Section 7.4.)
586
587 .. [2] Smith, Steven W., "The Scientist and Engineer's Guide to Digital
588 Signal Processing", Ch. 17. http://www.dspguide.com/ch17/1.htm
589
590 Examples
591 --------
592 A lowpass FIR filter with a response that is 1 on [0.0, 0.5], and
593 that decreases linearly on [0.5, 1.0] from 1 to 0:
594
595 >>> from scipy import signal
596 >>> taps = signal.firwin2(150, [0.0, 0.5, 1.0], [1.0, 1.0, 0.0])
597 >>> print(taps[72:78])
598 [-0.02286961 -0.06362756 0.57310236 0.57310236 -0.06362756 -0.02286961]
599
600 """
601 nyq = 0.5 * _get_fs(fs, nyq)
602
603 if len(freq) != len(gain):
604 raise ValueError('freq and gain must be of same length.')
605
606 if nfreqs is not None and numtaps >= nfreqs:
607 raise ValueError(('ntaps must be less than nfreqs, but firwin2 was '
608 'called with ntaps=%d and nfreqs=%s') %
609 (numtaps, nfreqs))
610
611 if freq[0] != 0 or freq[-1] != nyq:
612 raise ValueError('freq must start with 0 and end with fs/2.')
613 d = np.diff(freq)
614 if (d < 0).any():
615 raise ValueError('The values in freq must be nondecreasing.')
616 d2 = d[:-1] + d[1:]
617 if (d2 == 0).any():
618 raise ValueError('A value in freq must not occur more than twice.')
619 if freq[1] == 0:
620 raise ValueError('Value 0 must not be repeated in freq')
621 if freq[-2] == nyq:
622 raise ValueError('Value fs/2 must not be repeated in freq')
623
624 if antisymmetric:
625 if numtaps % 2 == 0:
626 ftype = 4
627 else:
628 ftype = 3
629 else:
630 if numtaps % 2 == 0:
631 ftype = 2
632 else:
633 ftype = 1
634
635 if ftype == 2 and gain[-1] != 0.0:
636 raise ValueError("A Type II filter must have zero gain at the "
637 "Nyquist frequency.")
638 elif ftype == 3 and (gain[0] != 0.0 or gain[-1] != 0.0):
639 raise ValueError("A Type III filter must have zero gain at zero "
640 "and Nyquist frequencies.")
641 elif ftype == 4 and gain[0] != 0.0:
642 raise ValueError("A Type IV filter must have zero gain at zero "
643 "frequency.")
644
645 if nfreqs is None:
646 nfreqs = 1 + 2 ** int(ceil(log(numtaps, 2)))
647
648 if (d == 0).any():
649 # Tweak any repeated values in freq so that interp works.
650 freq = np.array(freq, copy=True)
651 eps = np.finfo(float).eps * nyq
652 for k in range(len(freq) - 1):
653 if freq[k] == freq[k + 1]:
654 freq[k] = freq[k] - eps
655 freq[k + 1] = freq[k + 1] + eps
656 # Check if freq is strictly increasing after tweak
657 d = np.diff(freq)
658 if (d <= 0).any():
659 raise ValueError("freq cannot contain numbers that are too close "
660 "(within eps * (fs/2): "
661 "{}) to a repeated value".format(eps))
662
663 # Linearly interpolate the desired response on a uniform mesh `x`.
664 x = np.linspace(0.0, nyq, nfreqs)
665 fx = np.interp(x, freq, gain)
666
667 # Adjust the phases of the coefficients so that the first `ntaps` of the
668 # inverse FFT are the desired filter coefficients.
669 shift = np.exp(-(numtaps - 1) / 2. * 1.j * np.pi * x / nyq)
670 if ftype > 2:
671 shift *= 1j
672
673 fx2 = fx * shift
674
675 # Use irfft to compute the inverse FFT.
676 out_full = irfft(fx2)
677
678 if window is not None:
679 # Create the window to apply to the filter coefficients.
680 from .windows import get_window
681 wind = get_window(window, numtaps, fftbins=False)
682 else:
683 wind = 1
684
685 # Keep only the first `numtaps` coefficients in `out`, and multiply by
686 # the window.
687 out = out_full[:numtaps] * wind
688
689 if ftype == 3:
690 out[out.size // 2] = 0.0
691
692 return out

695def remez(numtaps, bands, desired, weight=None, Hz=None, type='bandpass',
696 maxiter=25, grid_density=16, fs=None):
697 """
698 Calculate the minimax optimal filter using the Remez exchange algorithm.
699
700 Calculate the filter-coefficients for the finite impulse response
701 (FIR) filter whose transfer function minimizes the maximum error
702 between the desired gain and the realized gain in the specified
703 frequency bands using the Remez exchange algorithm.
704
705 Parameters
706 ----------
707 numtaps : int
708 The desired number of taps in the filter. The number of taps is
709 the number of terms in the filter, or the filter order plus one.
710 bands : array_like
711 A monotonic sequence containing the band edges.
712 All elements must be non-negative and less than half the sampling
713 frequency as given by `fs`.
714 desired : array_like
715 A sequence half the size of bands containing the desired gain
716 in each of the specified bands.
717 weight : array_like, optional
718 A relative weighting to give to each band region. The length of
719 `weight` has to be half the length of `bands`.
720 Hz : scalar, optional, deprecated
721 The sampling frequency in Hz. Default is 1.
722
723 .. deprecated:: 1.0.0
724 `remez` keyword argument `Hz` is deprecated in favour of `fs` and
725 will be removed in SciPy 1.12.0.

727 The type of filter:
728

730
731 * 'differentiator' : frequency proportional response in bands.
732
733 * 'hilbert' : filter with odd symmetry, that is, type III
734 (for even order) or type IV (for odd order)
735 linear phase filters.
736
737 maxiter : int, optional
738 Maximum number of iterations of the algorithm. Default is 25.
739 grid_density : int, optional
740 Grid density. The dense grid used in `remez` is of size
741 ``(numtaps + 1) * grid_density``. Default is 16.
742 fs : float, optional
743 The sampling frequency of the signal. Default is 1.
744
745 Returns
746 -------
747 out : ndarray
748 A rank-1 array containing the coefficients of the optimal
749 (in a minimax sense) filter.
750
751 See Also
752 --------
753 firls
754 firwin
755 firwin2
756 minimum_phase
757
758 References
759 ----------
760 .. [1] J. H. McClellan and T. W. Parks, "A unified approach to the
761 design of optimum FIR linear phase digital filters",
762 IEEE Trans. Circuit Theory, vol. CT-20, pp. 697-701, 1973.
763 .. [2] J. H. McClellan, T. W. Parks and L. R. Rabiner, "A Computer
764 Program for Designing Optimum FIR Linear Phase Digital
765 Filters", IEEE Trans. Audio Electroacoust., vol. AU-21,
766 pp. 506-525, 1973.
767
768 Examples
769 --------
770 In these examples, `remez` is used to design low-pass, high-pass,
771 band-pass and band-stop filters. The parameters that define each filter

774

776 example, the desired gain in each band is either 0 (for a stop band)
777 or 1 (for a pass band).
778

781 the response.

791 ... fig = plt.figure()

793 ... ax.plot(w, 20*np.log10(np.abs(h)))
794 ... ax.set_ylim(-40, 5)
795 ... ax.grid(True)
796 ... ax.set_xlabel('Frequency (Hz)')
797 ... ax.set_ylabel('Gain (dB)')
798 ... ax.set_title(title)

805 >>> trans_width = 100 # Width of transition from pass to stop, Hz
806 >>> numtaps = 325 # Size of the FIR filter.
807 >>> taps = signal.remez(numtaps, [0, cutoff, cutoff + trans_width, 0.5*fs],
808 ... [1, 0], fs=fs)
809 >>> w, h = signal.freqz(taps, [1], worN=2000, fs=fs)

819 ... [0, 1], fs=fs)
820 >>> w, h = signal.freqz(taps, [1], worN=2000, fs=fs)
821 >>> plot_response(w, h, "High-pass Filter")

830 >>> numtaps = 63 # Size of the FIR filter.
831 >>> edges = [0, band[0] - trans_width, band[0], band[1],

840 The next example shows a band-stop filter.
841

843 >>> trans_width = 200 # Width of transition from pass to stop, Hz
844 >>> numtaps = 175 # Size of the FIR filter.
845 >>> edges = [0, band[0] - trans_width, band[0], band[1],

850 >>> plt.show()

865 tnum = {'bandpass': 1, 'differentiator': 2, 'hilbert': 3}[type]

879def firls(numtaps, bands, desired, weight=None, nyq=None, fs=None):
880 """
881 FIR filter design using least-squares error minimization.
882
883 Calculate the filter coefficients for the linear-phase finite
884 impulse response (FIR) filter which has the best approximation
885 to the desired frequency response described by `bands` and
886 `desired` in the least squares sense (i.e., the integral of the
887 weighted mean-squared error within the specified bands is
888 minimized).
889
890 Parameters
891 ----------
892 numtaps : int
893 The number of taps in the FIR filter. `numtaps` must be odd.
894 bands : array_like
895 A monotonic nondecreasing sequence containing the band edges in
896 Hz. All elements must be non-negative and less than or equal to
897 the Nyquist frequency given by `nyq`. The bands are specified as
898 frequency pairs, thus, if using a 1D array, its length must be
899 even, e.g., `np.array([0, 1, 2, 3, 4, 5])`. Alternatively, the
900 bands can be specified as an nx2 sized 2D array, where n is the
901 number of bands, e.g, `np.array([[0, 1], [2, 3], [4, 5]])`.
902 desired : array_like
903 A sequence the same size as `bands` containing the desired gain
904 at the start and end point of each band.
905 weight : array_like, optional
906 A relative weighting to give to each band region when solving
907 the least squares problem. `weight` has to be half the size of
908 `bands`.
909 nyq : float, optional, deprecated
910 This is the Nyquist frequency. Each frequency in `bands` must be
911 between 0 and `nyq` (inclusive). Default is 1.
912
913 .. deprecated:: 1.0.0
914 `firls` keyword argument `nyq` is deprecated in favour of `fs` and
915 will be removed in SciPy 1.12.0.
916 fs : float, optional
917 The sampling frequency of the signal. Each frequency in `bands`
918 must be between 0 and ``fs/2`` (inclusive). Default is 2.
919
920 Returns
921 -------
922 coeffs : ndarray
923 Coefficients of the optimal (in a least squares sense) FIR filter.
924
925 See Also
926 --------
927 firwin
928 firwin2
929 minimum_phase
930 remez
931
932 Notes
933 -----
934 This implementation follows the algorithm given in [1]_.
935 As noted there, least squares design has multiple advantages:
936
937 1. Optimal in a least-squares sense.
938 2. Simple, non-iterative method.
939 3. The general solution can obtained by solving a linear
940 system of equations.
941 4. Allows the use of a frequency dependent weighting function.
942
943 This function constructs a Type I linear phase FIR filter, which
944 contains an odd number of `coeffs` satisfying for :math:`n < numtaps`:
945
946 .. math:: coeffs(n) = coeffs(numtaps - 1 - n)
947
948 The odd number of coefficients and filter symmetry avoid boundary
949 conditions that could otherwise occur at the Nyquist and 0 frequencies
950 (e.g., for Type II, III, or IV variants).
951
952 .. versionadded:: 0.18
953
954 References
955 ----------
956 .. [1] Ivan Selesnick, Linear-Phase Fir Filter Design By Least Squares.
957 OpenStax CNX. Aug 9, 2005.
958 http://cnx.org/contents/eb1ecb35-03a9-4610-ba87-41cd771c95f2@7
959
960 Examples
961 --------
962 We want to construct a band-pass filter. Note that the behavior in the
963 frequency ranges between our stop bands and pass bands is unspecified,
964 and thus may overshoot depending on the parameters of our filter:
965
966 >>> import numpy as np
967 >>> from scipy import signal
968 >>> import matplotlib.pyplot as plt
969 >>> fig, axs = plt.subplots(2)
970 >>> fs = 10.0 # Hz
971 >>> desired = (0, 0, 1, 1, 0, 0)
972 >>> for bi, bands in enumerate(((0, 1, 2, 3, 4, 5), (0, 1, 2, 4, 4.5, 5))):
973 ... fir_firls = signal.firls(73, bands, desired, fs=fs)
974 ... fir_remez = signal.remez(73, bands, desired[::2], fs=fs)
975 ... fir_firwin2 = signal.firwin2(73, bands, desired, fs=fs)
976 ... hs = list()
977 ... ax = axs[bi]
978 ... for fir in (fir_firls, fir_remez, fir_firwin2):
979 ... freq, response = signal.freqz(fir)
980 ... hs.append(ax.semilogy(0.5*fs*freq/np.pi, np.abs(response))[0])
981 ... for band, gains in zip(zip(bands[::2], bands[1::2]),
982 ... zip(desired[::2], desired[1::2])):
983 ... ax.semilogy(band, np.maximum(gains, 1e-7), 'k--', linewidth=2)
984 ... if bi == 0:
985 ... ax.legend(hs, ('firls', 'remez', 'firwin2'),
986 ... loc='lower center', frameon=False)
987 ... else:
988 ... ax.set_xlabel('Frequency (Hz)')
989 ... ax.grid(True)
990 ... ax.set(title='Band-pass %d-%d Hz' % bands[2:4], ylabel='Magnitude')
991 ...
992 >>> fig.tight_layout()
993 >>> plt.show()
994
995 """ # noqa
996 nyq = 0.5 * _get_fs(fs, nyq)
997
998 numtaps = int(numtaps)
999 if numtaps % 2 == 0 or numtaps < 1:
1000 raise ValueError("numtaps must be odd and >= 1")
1001 M = (numtaps-1) // 2
1002
1003 # normalize bands 0->1 and make it 2 columns
1004 nyq = float(nyq)
1005 if nyq <= 0:
1006 raise ValueError('nyq must be positive, got %s <= 0.' % nyq)
1007 bands = np.asarray(bands).flatten() / nyq
1008 if len(bands) % 2 != 0:
1009 raise ValueError("bands must contain frequency pairs.")
1010 if (bands < 0).any() or (bands > 1).any():
1011 raise ValueError("bands must be between 0 and 1 relative to Nyquist")
1012 bands.shape = (-1, 2)
1013
1014 # check remaining params
1015 desired = np.asarray(desired).flatten()
1016 if bands.size != desired.size:
1017 raise ValueError("desired must have one entry per frequency, got %s "
1018 "gains for %s frequencies."
1019 % (desired.size, bands.size))
1020 desired.shape = (-1, 2)
1021 if (np.diff(bands) <= 0).any() or (np.diff(bands[:, 0]) < 0).any():
1022 raise ValueError("bands must be monotonically nondecreasing and have "
1023 "width > 0.")
1024 if (bands[:-1, 1] > bands[1:, 0]).any():
1025 raise ValueError("bands must not overlap.")
1026 if (desired < 0).any():
1027 raise ValueError("desired must be non-negative.")
1028 if weight is None:
1029 weight = np.ones(len(desired))
1030 weight = np.asarray(weight).flatten()
1031 if len(weight) != len(desired):
1032 raise ValueError("weight must be the same size as the number of "
1033 "band pairs ({}).".format(len(bands)))
1034 if (weight < 0).any():
1035 raise ValueError("weight must be non-negative.")
1036
1037 # Set up the linear matrix equation to be solved, Qa = b
1038
1039 # We can express Q(k,n) = 0.5 Q1(k,n) + 0.5 Q2(k,n)
1040 # where Q1(k,n)=q(k-n) and Q2(k,n)=q(k+n), i.e. a Toeplitz plus Hankel.
1041
1042 # We omit the factor of 0.5 above, instead adding it during coefficient
1043 # calculation.
1044
1045 # We also omit the 1/ from both Q and b equations, as they cancel
1046 # during solving.
1047
1048 # We have that:
1049 # q(n) = 1/ W()cos(n)d (over 0->)
1050 # Using our nomalization =f and with a constant weight W over each
1051 # interval f1->f2 we get:
1052 # q(n) = Wcos(nf)df (0->1) = Wf sin(nf)/nf
1053 # integrated over each f1->f2 pair (i.e., value at f2 - value at f1).
1054 n = np.arange(numtaps)[:, np.newaxis, np.newaxis]
1055 q = np.dot(np.diff(np.sinc(bands * n) * bands, axis=2)[:, :, 0], weight)
1056
1057 # Now we assemble our sum of Toeplitz and Hankel
1058 Q1 = toeplitz(q[:M+1])
1059 Q2 = hankel(q[:M+1], q[M:])
1060 Q = Q1 + Q2
1061
1062 # Now for b(n) we have that:
1063 # b(n) = 1/ W()D()cos(n)d (over 0->)
1064 # Using our normalization =f and with a constant weight W over each
1065 # interval and a linear term for D() we get (over each f1->f2 interval):
1066 # b(n) = W (mf+c)cos(nf)df
1067 # = f(mf+c)sin(nf)/nf + mf**2 cos(nf)/(nf)**2
1068 # integrated over each f1->f2 pair (i.e., value at f2 - value at f1).
1069 n = n[:M + 1] # only need this many coefficients here
1070 # Choose m and c such that we are at the start and end weights
1071 m = (np.diff(desired, axis=1) / np.diff(bands, axis=1))
1072 c = desired[:, [0]] - bands[:, [0]] * m
1073 b = bands * (m*bands + c) * np.sinc(bands * n)
1074 # Use L'Hospital's rule here for cos(nf)/(nf)**2 @ n=0
1075 b[0] -= m * bands * bands / 2.
1076 b[1:] += m * np.cos(n[1:] * np.pi * bands) / (np.pi * n[1:]) ** 2
1077 b = np.dot(np.diff(b, axis=2)[:, :, 0], weight)
1078
1079 # Now we can solve the equation
1080 try: # try the fast way
1081 with warnings.catch_warnings(record=True) as w:
1082 warnings.simplefilter('always')
1083 a = solve(Q, b, assume_a="pos", check_finite=False)
1084 for ww in w:
1085 if (ww.category == LinAlgWarning and
1086 str(ww.message).startswith('Ill-conditioned matrix')):
1087 raise LinAlgError(str(ww.message))
1088 except LinAlgError: # in case Q is rank deficient
1089 # This is faster than pinvh, even though we don't explicitly use
1090 # the symmetry here. gelsy was faster than gelsd and gelss in
1091 # some non-exhaustive tests.
1092 a = lstsq(Q, b, lapack_driver='gelsy')[0]
1093
1094 # make coefficients symmetric (linear phase)
1095 coeffs = np.hstack((a[:0:-1], 2 * a[0], a[1:]))
1096 return coeffs

1121def minimum_phase(h, method='homomorphic', n_fft=None):
1122 """Convert a linear-phase FIR filter to minimum phase
1123
1124 Parameters
1125 ----------
1126 h : array
1127 Linear-phase FIR filter coefficients.
1128 method : {'hilbert', 'homomorphic'}
1129 The method to use:
1130
1131 'homomorphic' (default)
1132 This method [4]_ [5]_ works best with filters with an
1133 odd number of taps, and the resulting minimum phase filter
1134 will have a magnitude response that approximates the square
1135 root of the original filter's magnitude response.
1136
1137 'hilbert'
1138 This method [1]_ is designed to be used with equiripple
1139 filters (e.g., from `remez`) with unity or zero gain
1140 regions.
1141
1142 n_fft : int
1143 The number of points to use for the FFT. Should be at least a
1144 few times larger than the signal length (see Notes).
1145
1146 Returns
1147 -------
1148 h_minimum : array
1149 The minimum-phase version of the filter, with length
1150 ``(length(h) + 1) // 2``.
1151
1152 See Also
1153 --------
1154 firwin
1155 firwin2
1156 remez
1157
1158 Notes
1159 -----
1160 Both the Hilbert [1]_ or homomorphic [4]_ [5]_ methods require selection
1161 of an FFT length to estimate the complex cepstrum of the filter.
1162
1163 In the case of the Hilbert method, the deviation from the ideal
1164 spectrum ``epsilon`` is related to the number of stopband zeros
1165 ``n_stop`` and FFT length ``n_fft`` as::
1166
1167 epsilon = 2. * n_stop / n_fft
1168
1169 For example, with 100 stopband zeros and a FFT length of 2048,
1170 ``epsilon = 0.0976``. If we conservatively assume that the number of
1171 stopband zeros is one less than the filter length, we can take the FFT
1172 length to be the next power of 2 that satisfies ``epsilon=0.01`` as::
1173
1174 n_fft = 2 ** int(np.ceil(np.log2(2 * (len(h) - 1) / 0.01)))
1175
1176 This gives reasonable results for both the Hilbert and homomorphic
1177 methods, and gives the value used when ``n_fft=None``.
1178
1179 Alternative implementations exist for creating minimum-phase filters,
1180 including zero inversion [2]_ and spectral factorization [3]_ [4]_.
1181 For more information, see:
1182
1183 http://dspguru.com/dsp/howtos/how-to-design-minimum-phase-fir-filters
1184
1185 References
1186 ----------
1187 .. [1] N. Damera-Venkata and B. L. Evans, "Optimal design of real and
1188 complex minimum phase digital FIR filters," Acoustics, Speech,
1189 and Signal Processing, 1999. Proceedings., 1999 IEEE International
1190 Conference on, Phoenix, AZ, 1999, pp. 1145-1148 vol.3.
1191 :doi:`10.1109/ICASSP.1999.756179`
1192 .. [2] X. Chen and T. W. Parks, "Design of optimal minimum phase FIR
1193 filters by direct factorization," Signal Processing,
1194 vol. 10, no. 4, pp. 369-383, Jun. 1986.
1195 .. [3] T. Saramaki, "Finite Impulse Response Filter Design," in
1196 Handbook for Digital Signal Processing, chapter 4,
1197 New York: Wiley-Interscience, 1993.
1198 .. [4] J. S. Lim, Advanced Topics in Signal Processing.
1199 Englewood Cliffs, N.J.: Prentice Hall, 1988.
1200 .. [5] A. V. Oppenheim, R. W. Schafer, and J. R. Buck,
1201 "Discrete-Time Signal Processing," 2nd edition.
1202 Upper Saddle River, N.J.: Prentice Hall, 1999.
1203
1204 Examples
1205 --------
1206 Create an optimal linear-phase filter, then convert it to minimum phase:
1207
1208 >>> import numpy as np
1209 >>> from scipy.signal import remez, minimum_phase, freqz, group_delay
1210 >>> import matplotlib.pyplot as plt
1211 >>> freq = [0, 0.2, 0.3, 1.0]
1212 >>> desired = [1, 0]
1213 >>> h_linear = remez(151, freq, desired, fs=2.)
1214
1215 Convert it to minimum phase:
1216
1217 >>> h_min_hom = minimum_phase(h_linear, method='homomorphic')
1218 >>> h_min_hil = minimum_phase(h_linear, method='hilbert')
1219
1220 Compare the three filters:
1221
1222 >>> fig, axs = plt.subplots(4, figsize=(4, 8))
1223 >>> for h, style, color in zip((h_linear, h_min_hom, h_min_hil),
1224 ... ('-', '-', '--'), ('k', 'r', 'c')):
1225 ... w, H = freqz(h)
1226 ... w, gd = group_delay((h, 1))
1227 ... w /= np.pi
1228 ... axs[0].plot(h, color=color, linestyle=style)
1229 ... axs[1].plot(w, np.abs(H), color=color, linestyle=style)
1230 ... axs[2].plot(w, 20 * np.log10(np.abs(H)), color=color, linestyle=style)
1231 ... axs[3].plot(w, gd, color=color, linestyle=style)
1232 >>> for ax in axs:
1233 ... ax.grid(True, color='0.5')
1234 ... ax.fill_between(freq[1:3], *ax.get_ylim(), color='#ffeeaa', zorder=1)
1235 >>> axs[0].set(xlim=[0, len(h_linear) - 1], ylabel='Amplitude', xlabel='Samples')
1236 >>> axs[1].legend(['Linear', 'Min-Hom', 'Min-Hil'], title='Phase')
1237 >>> for ax, ylim in zip(axs[1:], ([0, 1.1], [-150, 10], [-60, 60])):
1238 ... ax.set(xlim=[0, 1], ylim=ylim, xlabel='Frequency')
1239 >>> axs[1].set(ylabel='Magnitude')
1240 >>> axs[2].set(ylabel='Magnitude (dB)')
1241 >>> axs[3].set(ylabel='Group delay')
1242 >>> plt.tight_layout()
1243
1244 """ # noqa
1245 h = np.asarray(h)
1246 if np.iscomplexobj(h):
1247 raise ValueError('Complex filters not supported')
1248 if h.ndim != 1 or h.size <= 2:
1249 raise ValueError('h must be 1-D and at least 2 samples long')
1250 n_half = len(h) // 2
1251 if not np.allclose(h[-n_half:][::-1], h[:n_half]):
1252 warnings.warn('h does not appear to by symmetric, conversion may '
1253 'fail', RuntimeWarning)
1254 if not isinstance(method, str) or method not in \
1255 ('homomorphic', 'hilbert',):
1256 raise ValueError('method must be "homomorphic" or "hilbert", got %r'
1257 % (method,))
