119def make_sparse_data(
120 sparse_container,
121 n_samples=100,
122 n_features=100,
123 n_informative=10,
124 seed=42,
125 positive=False,
126 n_targets=1,
127):
128 random_state = np.random.RandomState(seed)
129
130 # build an ill-posed linear regression problem with many noisy features and
131 # comparatively few samples
132
133 # generate a ground truth model
134 w = random_state.randn(n_features, n_targets)
135 w[n_informative:] = 0.0 # only the top features are impacting the model
136 if positive:
137 w = np.abs(w)
138
139 X = random_state.randn(n_samples, n_features)
140 rnd = random_state.uniform(size=(n_samples, n_features))
141 X[rnd > 0.5] = 0.0 # 50% of zeros in input signal
142
143 # generate training ground truth labels
144 y = np.dot(X, w)
145 X = sparse_container(X)
146 if n_targets == 1:
147 y = np.ravel(y)
148 return X, y

151@pytest.mark.parametrize("csc_container", CSC_CONTAINERS)
152@pytest.mark.parametrize(
153 "alpha, fit_intercept, positive",
154 [(0.1, False, False), (0.1, True, False), (1e-3, False, True), (1e-3, True, True)],

156def test_sparse_enet_not_as_toy_dataset(csc_container, alpha, fit_intercept, positive):
157 n_samples, n_features, max_iter = 100, 100, 1000
158 n_informative = 10
159
160 X, y = make_sparse_data(
161 csc_container, n_samples, n_features, n_informative, positive=positive
162 )
163
164 X_train, X_test = X[n_samples // 2 :], X[: n_samples // 2]
165 y_train, y_test = y[n_samples // 2 :], y[: n_samples // 2]
166
167 s_clf = ElasticNet(
168 alpha=alpha,
169 l1_ratio=0.8,
170 fit_intercept=fit_intercept,
171 max_iter=max_iter,
172 tol=1e-7,
173 positive=positive,
174 warm_start=True,
175 )
176 s_clf.fit(X_train, y_train)
177
178 assert_almost_equal(s_clf.dual_gap_, 0, 4)
179 assert s_clf.score(X_test, y_test) > 0.85
180
181 # check the convergence is the same as the dense version
182 d_clf = ElasticNet(
183 alpha=alpha,
184 l1_ratio=0.8,
185 fit_intercept=fit_intercept,
186 max_iter=max_iter,
187 tol=1e-7,
188 positive=positive,
189 warm_start=True,
190 )
191 d_clf.fit(X_train.toarray(), y_train)
192
193 assert_almost_equal(d_clf.dual_gap_, 0, 4)
194 assert d_clf.score(X_test, y_test) > 0.85
195
196 assert_almost_equal(s_clf.coef_, d_clf.coef_, 5)
197 assert_almost_equal(s_clf.intercept_, d_clf.intercept_, 5)
198
199 # check that the coefs are sparse
200 assert np.sum(s_clf.coef_ != 0.0) < 2 * n_informative
