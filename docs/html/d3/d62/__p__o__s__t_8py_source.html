284def unpackPStrings(data, n):
285 # extract n Pascal strings from data.
286 # if there is not enough data, use ""
287
288 strings = []
289 index = 0
290 dataLen = len(data)
291
292 for _ in range(n):
293 if dataLen <= index:
294 length = 0
295 else:
296 length = byteord(data[index])
297 index += 1
298
299 if dataLen <= index + length - 1:
300 name = ""
301 else:
302 name = tostr(data[index : index + length], encoding="latin1")
303 strings.append(name)
304 index += length
305
306 if index < dataLen:
307 log.warning("%d extra bytes in post.stringData array", dataLen - index)
308
309 elif dataLen < index:
310 log.warning("not enough data in post.stringData array")
311
312 return strings
