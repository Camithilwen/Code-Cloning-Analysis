41class _VisualBlock:
42 """HTML Representation of Estimator
43
44 Parameters
45 ----------
46 kind : {'serial', 'parallel', 'single'}
47 kind of HTML block
48
49 estimators : list of estimators or `_VisualBlock`s or a single estimator
50 If kind != 'single', then `estimators` is a list of
51 estimators.
52 If kind == 'single', then `estimators` is a single estimator.
53
54 names : list of str, default=None
55 If kind != 'single', then `names` corresponds to estimators.
56 If kind == 'single', then `names` is a single string corresponding to
57 the single estimator.
58
59 name_details : list of str, str, or None, default=None
60 If kind != 'single', then `name_details` corresponds to `names`.
61 If kind == 'single', then `name_details` is a single string
62 corresponding to the single estimator.
63
64 name_caption : str, default=None
65 The caption below the name. `None` stands for no caption.
66 Only active when kind == 'single'.
67
68 doc_link_label : str, default=None
69 The label for the documentation link. If provided, the label would be
70 "Documentation for {doc_link_label}". Otherwise it will look for `names`.
71 Only active when kind == 'single'.
72
73 dash_wrapped : bool, default=True
74 If true, wrapped HTML element will be wrapped with a dashed border.
75 Only active when kind != 'single'.
76 """

84 name_details=None,
85 name_caption=None,
86 doc_link_label=None,

102 self.name_details = name_details

108def _write_label_html(
109 out,
110 params,
111 name,
112 name_details,
113 name_caption=None,
114 doc_link_label=None,
115 outer_class="sk-label-container",
116 inner_class="sk-label",
117 checked=False,
118 doc_link="",
119 is_fitted_css_class="",
120 is_fitted_icon="",
121 param_prefix="",
122):
123 """Write labeled html with or without a dropdown with named details.
124
125 Parameters
126 ----------
127 out : file-like object
128 The file to write the HTML representation to.
129 params: str
130 If estimator has `get_params` method, this is the HTML representation
131 of the estimator's parameters and their values. When the estimator
132 does not have `get_params`, it is an empty string.
133 name : str
134 The label for the estimator. It corresponds either to the estimator class name
135 for a simple estimator or in the case of a `Pipeline` and `ColumnTransformer`,
136 it corresponds to the name of the step.
137 name_details : str
138 The details to show as content in the dropdown part of the toggleable label. It
139 can contain information such as non-default parameters or column information for
140 `ColumnTransformer`.
141 name_caption : str, default=None
142 The caption below the name. If `None`, no caption will be created.
143 doc_link_label : str, default=None
144 The label for the documentation link. If provided, the label would be
145 "Documentation for {doc_link_label}". Otherwise it will look for `name`.
146 outer_class : {"sk-label-container", "sk-item"}, default="sk-label-container"
147 The CSS class for the outer container.
148 inner_class : {"sk-label", "sk-estimator"}, default="sk-label"
149 The CSS class for the inner container.
150 checked : bool, default=False
151 Whether the dropdown is folded or not. With a single estimator, we intend to
152 unfold the content.
153 doc_link : str, default=""
154 The link to the documentation for the estimator. If an empty string, no link is
155 added to the diagram. This can be generated for an estimator if it uses the
156 `_HTMLDocumentationLinkMixin`.
157 is_fitted_css_class : {"", "fitted"}
158 The CSS class to indicate whether or not the estimator is fitted. The
159 empty string means that the estimator is not fitted and "fitted" means that the
160 estimator is fitted.
161 is_fitted_icon : str, default=""
162 The HTML representation to show the fitted information in the diagram. An empty
163 string means that no information is shown.
164 param_prefix : str, default=""
165 The prefix to prepend to parameter names for nested estimators.
166 """
167 out.write(
168 f'<div class="{outer_class}"><div'
169 f' class="{inner_class} {is_fitted_css_class} sk-toggleable">'
170 )
171 name = html.escape(name)
172 if name_details is not None:
173 name_details = html.escape(str(name_details))
174 checked_str = "checked" if checked else ""
175 est_id = _ESTIMATOR_ID_COUNTER.get_id()
176
177 if doc_link:
178 doc_label = "<span>Online documentation</span>"
179 if doc_link_label is not None:
180 doc_label = f"<span>Documentation for {doc_link_label}</span>"
181 elif name is not None:
182 doc_label = f"<span>Documentation for {name}</span>"
183 doc_link = (
184 f'<a class="sk-estimator-doc-link {is_fitted_css_class}"'
185 f' rel="noreferrer" target="_blank" href="{doc_link}">?{doc_label}</a>'
186 )
187
188 name_caption_div = (
189 ""
190 if name_caption is None
191 else f'<div class="caption">{html.escape(name_caption)}</div>'
192 )
193 name_caption_div = f"<div><div>{name}</div>{name_caption_div}</div>"
194 links_div = (
195 f"<div>{doc_link}{is_fitted_icon}</div>"
196 if doc_link or is_fitted_icon
197 else ""
198 )
199
200 label_html = (
201 f'<label for="{est_id}" class="sk-toggleable__label {is_fitted_css_class} '
202 f'sk-toggleable__label-arrow">{name_caption_div}{links_div}</label>'
203 )
204
205 fmt_str = (
206 f'<input class="sk-toggleable__control sk-hidden--visually" id="{est_id}" '
207 f'type="checkbox" {checked_str}>{label_html}<div '
208 f'class="sk-toggleable__content {is_fitted_css_class}" '
209 f'data-param-prefix="{html.escape(param_prefix)}">'
210 )
211
212 if params:
213 fmt_str = "".join([fmt_str, f"{params}</div>"])
214 elif name_details and ("Pipeline" not in name):
215 fmt_str = "".join([fmt_str, f"<pre>{name_details}</pre></div>"])
216
217 out.write(fmt_str)
218 else:
219 out.write(f"<label>{name}</label>")
220 out.write("</div></div>") # outer_class inner_class

223def _get_visual_block(estimator):
224 """Generate information about how to display an estimator."""
225 if hasattr(estimator, "_sk_visual_block_"):
226 try:
227 return estimator._sk_visual_block_()
228 except Exception:
229 return _VisualBlock(
230 "single",
231 estimator,
232 names=estimator.__class__.__name__,
233 name_details=str(estimator),
234 )
235
236 if isinstance(estimator, str):
237 return _VisualBlock(
238 "single", estimator, names=estimator, name_details=estimator
239 )
240 elif estimator is None:
241 return _VisualBlock("single", estimator, names="None", name_details="None")
242
243 # check if estimator looks like a meta estimator (wraps estimators)
244 if hasattr(estimator, "get_params") and not isclass(estimator):
245 estimators = [
246 (key, est)
247 for key, est in estimator.get_params(deep=False).items()
248 if hasattr(est, "get_params") and hasattr(est, "fit") and not isclass(est)
249 ]
250 if estimators:
251 return _VisualBlock(
252 "parallel",
253 [est for _, est in estimators],
254 names=[f"{key}: {est.__class__.__name__}" for key, est in estimators],
255 name_details=[str(est) for _, est in estimators],
256 )
257
258 return _VisualBlock(
259 "single",
260 estimator,
261 names=estimator.__class__.__name__,
262 name_details=str(estimator),
263 )

266def _write_estimator_html(
267 out,
268 estimator,
269 estimator_label,
270 estimator_label_details,
271 is_fitted_css_class,
272 is_fitted_icon="",
273 first_call=False,
274 param_prefix="",
275):
276 """Write estimator to html in serial, parallel, or by itself (single).
277
278 For multiple estimators, this function is called recursively.
279
280 Parameters
281 ----------
282 out : file-like object
283 The file to write the HTML representation to.
284 estimator : estimator object
285 The estimator to visualize.
286 estimator_label : str
287 The label for the estimator. It corresponds either to the estimator class name
288 for simple estimator or in the case of `Pipeline` and `ColumnTransformer`, it
289 corresponds to the name of the step.
290 estimator_label_details : str
291 The details to show as content in the dropdown part of the toggleable label.
292 It can contain information as non-default parameters or column information for
293 `ColumnTransformer`.
294 is_fitted_css_class : {"", "fitted"}
295 The CSS class to indicate whether or not the estimator is fitted or not. The
296 empty string means that the estimator is not fitted and "fitted" means that the
297 estimator is fitted.
298 is_fitted_icon : str, default=""
299 The HTML representation to show the fitted information in the diagram. An empty
300 string means that no information is shown. If the estimator to be shown is not
301 the first estimator (i.e. `first_call=False`), `is_fitted_icon` is always an
302 empty string.
303 first_call : bool, default=False
304 Whether this is the first time this function is called.
305 param_prefix : str, default=""
306 The prefix to prepend to parameter names for nested estimators.
307 For example, in a pipeline this might be "pipeline__stepname__".
308 """
309 if first_call:
310 est_block = _get_visual_block(estimator)
311 else:
312 is_fitted_icon = ""
313 with config_context(print_changed_only=True):
314 est_block = _get_visual_block(estimator)
315 # `estimator` can also be an instance of `_VisualBlock`
316 if hasattr(estimator, "_get_doc_link"):
317 doc_link = estimator._get_doc_link()
318 else:
319 doc_link = ""
320 if est_block.kind in ("serial", "parallel"):
321 dashed_wrapped = first_call or est_block.dash_wrapped
322 dash_cls = " sk-dashed-wrapped" if dashed_wrapped else ""
323 out.write(f'<div class="sk-item{dash_cls}">')
324
325 if estimator_label:
326 if hasattr(estimator, "get_params") and hasattr(
327 estimator, "_get_params_html"
328 ):
329 params = estimator._get_params_html(deep=False)._repr_html_inner()
330 else:
331 params = ""
332
333 _write_label_html(
334 out,
335 params,
336 estimator_label,
337 estimator_label_details,
338 doc_link=doc_link,
339 is_fitted_css_class=is_fitted_css_class,
340 is_fitted_icon=is_fitted_icon,
341 param_prefix=param_prefix,
342 )
343
344 kind = est_block.kind
345 out.write(f'<div class="sk-{kind}">')
346 est_infos = zip(est_block.estimators, est_block.names, est_block.name_details)
347
348 for est, name, name_details in est_infos:
349 # Build the parameter prefix for nested estimators
350
351 if param_prefix and hasattr(name, "split"):
352 # If we already have a prefix, append the new component
353 new_prefix = f"{param_prefix}{name.split(':')[0]}__"
354 elif hasattr(name, "split"):
355 # If this is the first level, start the prefix
356 new_prefix = f"{name.split(':')[0]}__" if name else ""
357 else:
358 new_prefix = param_prefix
359
360 if kind == "serial":
361 _write_estimator_html(
362 out,
363 est,
364 name,
365 name_details,
366 is_fitted_css_class=is_fitted_css_class,
367 param_prefix=new_prefix,
368 )
369 else: # parallel
370 out.write('<div class="sk-parallel-item">')
371 # wrap element in a serial visualblock
372 serial_block = _VisualBlock("serial", [est], dash_wrapped=False)
373 _write_estimator_html(
374 out,
375 serial_block,
376 name,
377 name_details,
378 is_fitted_css_class=is_fitted_css_class,
379 param_prefix=new_prefix,
380 )
381 out.write("</div>") # sk-parallel-item
382
383 out.write("</div></div>")
384 elif est_block.kind == "single":
385 if hasattr(estimator, "_get_params_html"):
386 params = estimator._get_params_html()._repr_html_inner()
387 else:
388 params = ""
389
390 _write_label_html(
391 out,
392 params,
393 est_block.names,
394 est_block.name_details,
395 est_block.name_caption,
396 est_block.doc_link_label,
397 outer_class="sk-item",
398 inner_class="sk-estimator",
399 checked=first_call,
400 doc_link=doc_link,
401 is_fitted_css_class=is_fitted_css_class,
402 is_fitted_icon=is_fitted_icon,
403 param_prefix=param_prefix,
404 )

407def estimator_html_repr(estimator):
408 """Build a HTML representation of an estimator.
409
410 Read more in the :ref:`User Guide <visualizing_composite_estimators>`.
411
412 Parameters
413 ----------
414 estimator : estimator object
415 The estimator to visualize.
416
417 Returns
418 -------
419 html: str
420 HTML representation of estimator.
421
422 Examples
423 --------
424 >>> from sklearn.utils._repr_html.estimator import estimator_html_repr
425 >>> from sklearn.linear_model import LogisticRegression
426 >>> estimator_html_repr(LogisticRegression())
427 '<style>#sk-container-id...'
428 """
429 from sklearn.exceptions import NotFittedError
430 from sklearn.utils.validation import check_is_fitted
431
432 if not hasattr(estimator, "fit"):
433 status_label = "<span>Not fitted</span>"
434 is_fitted_css_class = ""
435 else:
436 try:
437 check_is_fitted(estimator)
438 status_label = "<span>Fitted</span>"
439 is_fitted_css_class = "fitted"
440 except NotFittedError:
441 status_label = "<span>Not fitted</span>"
442 is_fitted_css_class = ""
443
444 is_fitted_icon = (
445 f'<span class="sk-estimator-doc-link {is_fitted_css_class}">'
446 f"i{status_label}</span>"
447 )
448 with closing(StringIO()) as out:
449 container_id = _CONTAINER_ID_COUNTER.get_id()
450 style_template = Template(_CSS_STYLE)
451 style_with_id = style_template.substitute(id=container_id)
452 estimator_str = str(estimator)
453
454 # The fallback message is shown by default and loading the CSS sets
455 # div.sk-text-repr-fallback to display: none to hide the fallback message.
456 #
457 # If the notebook is trusted, the CSS is loaded which hides the fallback
458 # message. If the notebook is not trusted, then the CSS is not loaded and the
459 # fallback message is shown by default.
460 #
461 # The reverse logic applies to HTML repr div.sk-container.
462 # div.sk-container is hidden by default and the loading the CSS displays it.
463 fallback_msg = (
464 "In a Jupyter environment, please rerun this cell to show the HTML"
465 " representation or trust the notebook. <br />On GitHub, the"
466 " HTML representation is unable to render, please try loading this page"
467 " with nbviewer.org."
468 )
469 html_template = (
470 f"<style>{style_with_id}</style>"
471 f"<body>"
472 f'<div id="{container_id}" class="sk-top-container">'
473 '<div class="sk-text-repr-fallback">'
474 f"<pre>{html.escape(estimator_str)}</pre><b>{fallback_msg}</b>"
475 "</div>"
476 '<div class="sk-container" hidden>'
477 )
478
479 out.write(html_template)
480 _write_estimator_html(
481 out,
482 estimator,
483 estimator.__class__.__name__,
484 estimator_str,
485 first_call=True,
486 is_fitted_css_class=is_fitted_css_class,
487 is_fitted_icon=is_fitted_icon,
488 )
489 with open(str(Path(__file__).parent / "estimator.js"), "r") as f:
490 script = f.read()
491
492 html_end = f"</div></div><script>{script}</script></body>"
493
494 out.write(html_end)
495
496 html_output = out.getvalue()
497 return html_output
