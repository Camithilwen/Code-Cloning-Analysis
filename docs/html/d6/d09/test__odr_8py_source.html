44 def explicit_fcn(self, B, x):
45 ret = B[0] + B[1] * np.power(np.exp(B[2]*x) - 1.0, 2)
46 return ret

48 def explicit_fjd(self, B, x):
49 eBx = np.exp(B[2]*x)

54 eBx = np.exp(B[2]*x)
55 res = np.vstack([np.ones(x.shape[-1]),
56 np.power(eBx-1.0, 2),
57 B[1]*2.0*(eBx-1.0)*eBx*x])
58 return res

151 # Multi-variable Example
152
153 def multi_fcn(self, B, x):
154 if (x < 0.0).any():
155 raise OdrStop
156 theta = pi*B[3]/2.
157 ctheta = np.cos(theta)
158 stheta = np.sin(theta)
159 omega = np.power(2.*pi*x*np.exp(-B[2]), B[3])
160 phi = np.arctan2((omega*stheta), (1.0 + omega*ctheta))
161 r = (B[0] - B[1]) * np.power(np.sqrt(np.power(1.0 + omega*ctheta, 2) +
162 np.power(omega*stheta, 2)), -B[4])
163 ret = np.vstack([B[1] + r*np.cos(B[4]*phi),
164 r*np.sin(B[4]*phi)])
165 return ret
