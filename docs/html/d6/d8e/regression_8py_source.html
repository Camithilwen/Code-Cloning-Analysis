13class PredictionErrorDisplay:
14 """Visualization of the prediction error of a regression model.
15
16 This tool can display "residuals vs predicted" or "actual vs predicted"
17 using scatter plots to qualitatively assess the behavior of a regressor,
18 preferably on held-out data points.
19
20 See the details in the docstrings of
21 :func:`~sklearn.metrics.PredictionErrorDisplay.from_estimator` or
22 :func:`~sklearn.metrics.PredictionErrorDisplay.from_predictions` to
23 create a visualizer. All parameters are stored as attributes.
24
25 For general information regarding `scikit-learn` visualization tools, read
26 more in the :ref:`Visualization Guide <visualizations>`.
27 For details regarding interpreting these plots, refer to the
28 :ref:`Model Evaluation Guide <visualization_regression_evaluation>`.
29
30 .. versionadded:: 1.2
31
32 Parameters
33 ----------
34 y_true : ndarray of shape (n_samples,)
35 True values.
36
37 y_pred : ndarray of shape (n_samples,)
38 Prediction values.
39
40 Attributes
41 ----------
42 line_ : matplotlib Artist
43 Optimal line representing `y_true == y_pred`. Therefore, it is a
44 diagonal line for `kind="predictions"` and a horizontal line for
45 `kind="residuals"`.
46
47 errors_lines_ : matplotlib Artist or None
48 Residual lines. If `with_errors=False`, then it is set to `None`.
49
50 scatter_ : matplotlib Artist
51 Scatter data points.
52
53 ax_ : matplotlib Axes
54 Axes with the different matplotlib axis.
55
56 figure_ : matplotlib Figure
57 Figure containing the scatter and lines.
58
59 See Also
60 --------
61 PredictionErrorDisplay.from_estimator : Prediction error visualization
62 given an estimator and some data.
63 PredictionErrorDisplay.from_predictions : Prediction error visualization
64 given the true and predicted targets.
65
66 Examples
67 --------
68 >>> import matplotlib.pyplot as plt
69 >>> from sklearn.datasets import load_diabetes
70 >>> from sklearn.linear_model import Ridge
71 >>> from sklearn.metrics import PredictionErrorDisplay
72 >>> X, y = load_diabetes(return_X_y=True)
73 >>> ridge = Ridge().fit(X, y)
74 >>> y_pred = ridge.predict(X)
75 >>> display = PredictionErrorDisplay(y_true=y, y_pred=y_pred)
76 >>> display.plot()
77 <...>
78 >>> plt.show()
79 """
80

85 def plot(
86 self,
87 ax=None,
88 *,
89 kind="residual_vs_predicted",
90 scatter_kwargs=None,
91 line_kwargs=None,
92 ):
93 """Plot visualization.
94
95 Extra keyword arguments will be passed to matplotlib's ``plot``.
96
97 Parameters
98 ----------
99 ax : matplotlib axes, default=None
100 Axes object to plot on. If `None`, a new figure and axes is
101 created.
102
103 kind : {"actual_vs_predicted", "residual_vs_predicted"}, \
104 default="residual_vs_predicted"
105 The type of plot to draw:
106
107 - "actual_vs_predicted" draws the observed values (y-axis) vs.
108 the predicted values (x-axis).
109 - "residual_vs_predicted" draws the residuals, i.e. difference
110 between observed and predicted values, (y-axis) vs. the predicted
111 values (x-axis).
112
113 scatter_kwargs : dict, default=None
114 Dictionary with keywords passed to the `matplotlib.pyplot.scatter`
115 call.
116
117 line_kwargs : dict, default=None
118 Dictionary with keyword passed to the `matplotlib.pyplot.plot`
119 call to draw the optimal line.
120
121 Returns
122 -------
123 display : :class:`~sklearn.metrics.PredictionErrorDisplay`
124
125 Object that stores computed values.
126 """
127 check_matplotlib_support(f"{self.__class__.__name__}.plot")
128
129 expected_kind = ("actual_vs_predicted", "residual_vs_predicted")
130 if kind not in expected_kind:
131 raise ValueError(
132 f"`kind` must be one of {', '.join(expected_kind)}. "
133 f"Got {kind!r} instead."
134 )
135
136 import matplotlib.pyplot as plt
137
138 if scatter_kwargs is None:
139 scatter_kwargs = {}
140 if line_kwargs is None:
141 line_kwargs = {}
142
143 default_scatter_kwargs = {"color": "tab:blue", "alpha": 0.8}
144 default_line_kwargs = {"color": "black", "alpha": 0.7, "linestyle": "--"}
145
146 scatter_kwargs = _validate_style_kwargs(default_scatter_kwargs, scatter_kwargs)
147 line_kwargs = _validate_style_kwargs(default_line_kwargs, line_kwargs)
148
149 scatter_kwargs = {**default_scatter_kwargs, **scatter_kwargs}
150 line_kwargs = {**default_line_kwargs, **line_kwargs}
151
152 if ax is None:
153 _, ax = plt.subplots()
154
155 if kind == "actual_vs_predicted":
156 max_value = max(np.max(self.y_true), np.max(self.y_pred))
157 min_value = min(np.min(self.y_true), np.min(self.y_pred))
158 self.line_ = ax.plot(
159 [min_value, max_value], [min_value, max_value], **line_kwargs
160 )[0]
161
162 x_data, y_data = self.y_pred, self.y_true
163 xlabel, ylabel = "Predicted values", "Actual values"
164
165 self.scatter_ = ax.scatter(x_data, y_data, **scatter_kwargs)
166
167 # force to have a squared axis
168 ax.set_aspect("equal", adjustable="datalim")
169 ax.set_xticks(np.linspace(min_value, max_value, num=5))
170 ax.set_yticks(np.linspace(min_value, max_value, num=5))
171 else: # kind == "residual_vs_predicted"
172 self.line_ = ax.plot(
173 [np.min(self.y_pred), np.max(self.y_pred)],
174 [0, 0],
175 **line_kwargs,
176 )[0]
177 self.scatter_ = ax.scatter(
178 self.y_pred, self.y_true - self.y_pred, **scatter_kwargs
179 )
180 xlabel, ylabel = "Predicted values", "Residuals (actual - predicted)"
181
182 ax.set(xlabel=xlabel, ylabel=ylabel)
183
184 self.ax_ = ax
185 self.figure_ = ax.figure
186
187 return self
