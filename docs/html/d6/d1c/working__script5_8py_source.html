32# --- ENSEMBLE ASSESSMENT ---
33def ensemble_assessment(code1, code2, model_name, n=3):
34 predictions = []
35 for _ in range(n):
36 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name)
37 predictions.append((results, predicted_type, predicted_sim))
38 # Majority voting on predicted_sim
39 sim_votes = [pred[2] for pred in predictions]
40 final_sim = max(set(sim_votes), key=sim_votes.count)
41 # Most frequent predicted_type
42 type_votes = [pred[1] for pred in predictions]
43 final_type = max(set(type_votes), key=type_votes.count)
44 # Average results for reporting
45 avg_results = {t: sum(pred[0][t] for pred in predictions) / n for t in ["Type-1", "Type-2", "Type-3", "Type-4"]}
46 return avg_results, final_type, final_sim

55# --- PROMPT V2 WITH STEP-BY-STEP REASONING AND EXAMPLES ---
56def rag_similarity_assessment(code1, code2, model_name):
57 prompt = f"""You are a code similarity expert. Analyze the following code pair step by step for clone detection.
58
59Definitions:
60- Type-1: Identical except for whitespace, comments, layout.
61- Type-2: Identical except for variable/function names (plus Type-1 differences).
62- Type-3: Similar, but with some statements added/removed/modified (plus Type-2 differences).
63- Type-4: Syntactically different but functionally identical (same outputs for same inputs).
64
65Examples:
66Type-1: 'int a=5;' vs 'int a = 5; // set a'
67Type-2: 'int a=5;' vs 'int b=5;'
68Type-3: 'int a=5; print(a);' vs 'int a=5;'
69Type-4: 'for(int i=0;i<5;i++)sum+=i;' vs 'sum = sum_of_first_n(5);'
70
71Step-by-step:
721. Are the outputs always identical for all inputs? (Type-4)
732. Are they identical except for whitespace/comments? (Type-1)
743. Are names changed but structure identical? (Type-2)
754. Are there statement-level edits? (Type-3)
76Explain your reasoning for each type and provide a confidence score (0-1).
77
78Target Code:
79{code1}
80
81Similar Code:
82{code2}
83
84Respond ONLY with a JSON object with the following keys: Type-1, Type-2, Type-3, Type-4, each with a confidence score between 0 and 1.
85"""
