384@given(DATETIME_NO_TZ)
385@pytest.mark.parametrize("delimiter", list(" -./"))
386@pytest.mark.parametrize("dayfirst", [True, False])
387@pytest.mark.parametrize(
388 "date_format",
389 ["%d %m %Y", "%m %d %Y", "%m %Y", "%Y %m %d", "%y %m %d", "%Y%m%d", "%y%m%d"],
390)
391def test_hypothesis_delimited_date(
392 request, date_format, dayfirst, delimiter, test_datetime
393):
394 if date_format == "%m %Y" and delimiter == ".":
395 request.applymarker(
396 pytest.mark.xfail(
397 reason="parse_datetime_string cannot reliably tell whether "
398 "e.g. %m.%Y is a float or a date"
399 )
400 )
401 date_string = test_datetime.strftime(date_format.replace(" ", delimiter))
402
403 except_out_dateutil, result = _helper_hypothesis_delimited_date(
404 parsing.py_parse_datetime_string, date_string, dayfirst=dayfirst
405 )
406 except_in_dateutil, expected = _helper_hypothesis_delimited_date(
407 du_parse,
408 date_string,
409 default=datetime(1, 1, 1),
410 dayfirst=dayfirst,
411 yearfirst=False,
412 )
413
414 assert except_out_dateutil == except_in_dateutil
415 assert result == expected
