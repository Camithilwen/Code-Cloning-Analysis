152@pytest.mark.parametrize("fmt", ["png", "jpg", "jpeg", "tiff"])
153def test_imsave(fmt):
154 has_alpha = fmt not in ["jpg", "jpeg"]
155
156 # The goal here is that the user can specify an output logical DPI
157 # for the image, but this will not actually add any extra pixels
158 # to the image, it will merely be used for metadata purposes.
159
160 # So we do the traditional case (dpi == 1), and the new case (dpi
161 # == 100) and read the resulting PNG files back in and make sure
162 # the data is 100% identical.
163 np.random.seed(1)
164 # The height of 1856 pixels was selected because going through creating an
165 # actual dpi=100 figure to save the image to a Pillow-provided format would
166 # cause a rounding error resulting in a final image of shape 1855.
167 data = np.random.rand(1856, 2)
168
169 buff_dpi1 = io.BytesIO()
170 plt.imsave(buff_dpi1, data, format=fmt, dpi=1)
171
172 buff_dpi100 = io.BytesIO()
173 plt.imsave(buff_dpi100, data, format=fmt, dpi=100)
174
175 buff_dpi1.seek(0)
176 arr_dpi1 = plt.imread(buff_dpi1, format=fmt)
177
178 buff_dpi100.seek(0)
179 arr_dpi100 = plt.imread(buff_dpi100, format=fmt)
180
181 assert arr_dpi1.shape == (1856, 2, 3 + has_alpha)
182 assert arr_dpi100.shape == (1856, 2, 3 + has_alpha)
183
184 assert_array_equal(arr_dpi1, arr_dpi100)

1227# A basic ndarray subclass that implements a quantity
1228# It does not implement an entire unit system or all quantity math.
1229# There is just enough implemented to test handling of ndarray
1230# subclasses.
1231class QuantityND(np.ndarray):
1232 def __new__(cls, input_array, units):
1233 obj = np.asarray(input_array).view(cls)
1234 obj.units = units
1235 return obj
1236
1237 def __array_finalize__(self, obj):
1238 self.units = getattr(obj, "units", None)
1239
1240 def __getitem__(self, item):
1241 units = getattr(self, "units", None)
1242 ret = super().__getitem__(item)
1243 if isinstance(ret, QuantityND) or units is not None:
1244 ret = QuantityND(ret, units)
1245 return ret
1246
1247 def __array_ufunc__(self, ufunc, method, *inputs, **kwargs):
1248 func = getattr(ufunc, method)
1249 if "out" in kwargs:
1250 return NotImplemented
1251 if len(inputs) == 1:
1252 i0 = inputs[0]
1253 unit = getattr(i0, "units", "dimensionless")
1254 out_arr = func(np.asarray(i0), **kwargs)
1255 elif len(inputs) == 2:
1256 i0 = inputs[0]
1257 i1 = inputs[1]
1258 u0 = getattr(i0, "units", "dimensionless")
1259 u1 = getattr(i1, "units", "dimensionless")
1260 u0 = u1 if u0 is None else u0
1261 u1 = u0 if u1 is None else u1
1262 if ufunc in [np.add, np.subtract]:
1263 if u0 != u1:
1264 raise ValueError
1265 unit = u0
1266 elif ufunc == np.multiply:
1267 unit = f"{u0}*{u1}"
1268 elif ufunc == np.divide:
1269 unit = f"{u0}/({u1})"
1270 elif ufunc in (np.greater, np.greater_equal,
1271 np.equal, np.not_equal,
1272 np.less, np.less_equal):
1273 # Comparisons produce unitless booleans for output
1274 unit = None
1275 else:
1276 return NotImplemented
1277 out_arr = func(i0.view(np.ndarray), i1.view(np.ndarray), **kwargs)
1278 else:
1279 return NotImplemented
1280 if unit is None:
1281 out_arr = np.array(out_arr)
1282 else:
1283 out_arr = QuantityND(out_arr, unit)
1284 return out_arr

1286 @property
1287 def v(self):
1288 return self.view(np.ndarray)
