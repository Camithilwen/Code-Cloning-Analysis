44):
45 S = np.array(
46 [
47 [1.0, 1.0, 1.0, 0.2, 0.0, 0.0, 0.0],

53 [0.0, 0.0, 0.0, 1.0, 1.0, 1.0, 1.0],
54 ]
55 )
56
57 for mat in (S, csr_container(S)):
58 model = SpectralClustering(

152 kernels_available = kernel_metrics()
153 for kern in kernels_available:
154 # Additive chi^2 gives a negative similarity matrix which

157 sp = SpectralClustering(n_clusters=2, affinity=kern, random_state=0)
158 labels = sp.fit(X).labels_

162 labels = sp.fit(X).labels_
163 assert (X.shape[0],) == labels.shape

165 def histogram(x, y, **kwargs):
166 # Histogram kernel implemented as a callable.

168 return np.minimum(x, y).sum()

171 labels = sp.fit(X).labels_

175def test_cluster_qr(global_random_seed):
176 # cluster_qr by itself should not be used for clustering generic data
177 # other than the rows of the eigenvectors within spectral clustering,
178 # but cluster_qr must still preserve the labels for different dtypes
179 # of the generic fixed input even if the labels may be meaningless.
180 random_state = np.random.RandomState(seed=global_random_seed)
181 n_samples, n_components = 10, 5
182 data = random_state.randn(n_samples, n_components)
183 labels_float64 = cluster_qr(data.astype(np.float64))
184 # Each sample is assigned a cluster identifier
185 assert labels_float64.shape == (n_samples,)
186 # All components should be covered by the assignment
187 assert np.array_equal(np.unique(labels_float64), np.arange(n_components))
188 # Single precision data should yield the same cluster assignments
189 labels_float32 = cluster_qr(data.astype(np.float32))
190 assert np.array_equal(labels_float64, labels_float32)

205@pytest.mark.parametrize("coo_container", COO_CONTAINERS)
206@pytest.mark.parametrize("n_samples", [50, 100, 150, 500])
207def test_discretize(n_samples, coo_container, global_random_seed):
208 # Test the discretize using a noise assignment matrix
209 random_state = np.random.RandomState(seed=global_random_seed)
210 for n_class in range(2, 10):
211 # random class labels
212 y_true = random_state.randint(0, n_class + 1, n_samples)
213 y_true = np.array(y_true, float)
214 # noise class assignment matrix
215 y_indicator = coo_container(
216 (np.ones(n_samples), (np.arange(n_samples), y_true)),
217 shape=(n_samples, n_class + 1),
218 )
219 y_true_noisy = y_indicator.toarray() + 0.1 * random_state.randn(
220 n_samples, n_class + 1
221 )
222 y_pred = discretize(y_true_noisy, random_state=random_state)
223 assert adjusted_rand_score(y_true, y_pred) > 0.8

226def test_spectral_clustering_with_arpack_amg_solvers(global_random_seed):
227 # Test that spectral_clustering is the same for arpack and amg solver
228 # Based on toy example from plot_segmentation_toy.py
229
230 # a small two coin image
231 x, y = np.indices((40, 40))
232
233 center1, center2 = (14, 12), (20, 25)
234 radius1, radius2 = 8, 7
235
236 circle1 = (x - center1[0]) ** 2 + (y - center1[1]) ** 2 < radius1**2
237 circle2 = (x - center2[0]) ** 2 + (y - center2[1]) ** 2 < radius2**2
238
239 circles = circle1 | circle2
240 mask = circles.copy()
241 img = circles.astype(float)
242
243 graph = img_to_graph(img, mask=mask)
244 graph.data = np.exp(-graph.data / graph.data.std())
245
246 labels_arpack = spectral_clustering(
247 graph, n_clusters=2, eigen_solver="arpack", random_state=global_random_seed
248 )
249
250 assert len(np.unique(labels_arpack)) == 2
251
252 if amg_loaded:
253 labels_amg = spectral_clustering(
254 graph, n_clusters=2, eigen_solver="amg", random_state=global_random_seed
255 )
256 assert adjusted_rand_score(labels_arpack, labels_amg) == 1
257 else:
258 with pytest.raises(ValueError):
259 spectral_clustering(graph, n_clusters=2, eigen_solver="amg", random_state=0)
