46def _fix_connectivity(X, connectivity, affinity):
47 """
48 Fixes the connectivity matrix.
49
50 The different steps are:
51
52 - copies it
53 - makes it symmetric
54 - converts it to LIL if necessary
55 - completes it if necessary.
56
57 Parameters
58 ----------
59 X : array-like of shape (n_samples, n_features)
60 Feature matrix representing `n_samples` samples to be clustered.
61
62 connectivity : sparse matrix, default=None
63 Connectivity matrix. Defines for each sample the neighboring samples
64 following a given structure of the data. The matrix is assumed to
65 be symmetric and only the upper triangular half is used.
66 Default is `None`, i.e, the Ward algorithm is unstructured.
67
68 affinity : {"euclidean", "precomputed"}, default="euclidean"
69 Which affinity to use. At the moment `precomputed` and
70 ``euclidean`` are supported. `euclidean` uses the
71 negative squared Euclidean distance between points.
72
73 Returns
74 -------
75 connectivity : sparse matrix
76 The fixed connectivity matrix.
77
78 n_connected_components : int
79 The number of connected components in the graph.
80 """
81 n_samples = X.shape[0]
82 if connectivity.shape[0] != n_samples or connectivity.shape[1] != n_samples:
83 raise ValueError(
84 "Wrong shape for connectivity matrix: %s when X is %s"
85 % (connectivity.shape, X.shape)
86 )
87
88 # Make the connectivity matrix symmetric:
89 connectivity = connectivity + connectivity.T
90
91 # Convert connectivity matrix to LIL
92 if not sparse.issparse(connectivity):
93 connectivity = sparse.lil_matrix(connectivity)
94
95 # `connectivity` is a sparse matrix at this point
96 if connectivity.format != "lil":
97 connectivity = connectivity.tolil()
98
99 # Compute the number of nodes
100 n_connected_components, labels = connected_components(connectivity)
101
102 if n_connected_components > 1:
103 warnings.warn(
104 "the number of connected components of the "
105 "connectivity matrix is %d > 1. Completing it to avoid "
106 "stopping the tree early." % n_connected_components,
107 stacklevel=2,
108 )
109 # XXX: Can we do without completing the matrix?
110 connectivity = _fix_connected_components(
111 X=X,
112 graph=connectivity,
113 n_connected_components=n_connected_components,
114 component_labels=labels,
115 metric=affinity,
116 mode="connectivity",
117 )
118
119 return connectivity, n_connected_components

122def _single_linkage_tree(
123 connectivity,
124 n_samples,
125 n_nodes,
126 n_clusters,
127 n_connected_components,
128 return_distance,
129):
130 """
131 Perform single linkage clustering on sparse data via the minimum
132 spanning tree from scipy.sparse.csgraph, then using union-find to label.
133 The parent array is then generated by walking through the tree.
134 """
135 from scipy.sparse.csgraph import minimum_spanning_tree
136
137 # explicitly cast connectivity to ensure safety
138 connectivity = connectivity.astype(np.float64, copy=False)
139
140 # Ensure zero distances aren't ignored by setting them to "epsilon"
141 epsilon_value = np.finfo(dtype=connectivity.data.dtype).eps
142 connectivity.data[connectivity.data == 0] = epsilon_value
143
144 # Use scipy.sparse.csgraph to generate a minimum spanning tree
145 mst = minimum_spanning_tree(connectivity.tocsr())
146
147 # Convert the graph to scipy.cluster.hierarchy array format
148 mst = mst.tocoo()
149
150 # Undo the epsilon values
151 mst.data[mst.data == epsilon_value] = 0
152
153 mst_array = np.vstack([mst.row, mst.col, mst.data]).T
154
155 # Sort edges of the min_spanning_tree by weight
156 mst_array = mst_array[np.argsort(mst_array.T[2], kind="mergesort"), :]
157
158 # Convert edge list into standard hierarchical clustering format
159 single_linkage_tree = _hierarchical._single_linkage_label(mst_array)
160 children_ = single_linkage_tree[:, :2].astype(int)
161
162 # Compute parents
163 parent = np.arange(n_nodes, dtype=np.intp)
164 for i, (left, right) in enumerate(children_, n_samples):
165 if n_clusters is not None and i >= n_nodes:
166 break
167 if left < n_nodes:
168 parent[left] = i
169 if right < n_nodes:
170 parent[right] = i
171
172 if return_distance:
173 distances = single_linkage_tree[:, 2]
174 return children_, n_connected_components, n_samples, parent, distances
175 return children_, n_connected_components, n_samples, parent

191def ward_tree(X, *, connectivity=None, n_clusters=None, return_distance=False):
192 """Ward clustering based on a Feature matrix.
193
194 Recursively merges the pair of clusters that minimally increases
195 within-cluster variance.
196
197 The inertia matrix uses a Heapq-based representation.
198
199 This is the structured version, that takes into account some topological
200 structure between samples.
201
202 Read more in the :ref:`User Guide <hierarchical_clustering>`.
203
204 Parameters
205 ----------
206 X : array-like of shape (n_samples, n_features)
207 Feature matrix representing `n_samples` samples to be clustered.
208
209 connectivity : {array-like, sparse matrix}, default=None
210 Connectivity matrix. Defines for each sample the neighboring samples
211 following a given structure of the data. The matrix is assumed to
212 be symmetric and only the upper triangular half is used.
213 Default is None, i.e, the Ward algorithm is unstructured.
214
215 n_clusters : int, default=None
216 `n_clusters` should be less than `n_samples`. Stop early the
217 construction of the tree at `n_clusters.` This is useful to decrease
218 computation time if the number of clusters is not small compared to the
219 number of samples. In this case, the complete tree is not computed, thus
220 the 'children' output is of limited use, and the 'parents' output should
221 rather be used. This option is valid only when specifying a connectivity
222 matrix.
223
224 return_distance : bool, default=False
225 If `True`, return the distance between the clusters.
226
227 Returns
228 -------
229 children : ndarray of shape (n_nodes-1, 2)
230 The children of each non-leaf node. Values less than `n_samples`
231 correspond to leaves of the tree which are the original samples.
232 A node `i` greater than or equal to `n_samples` is a non-leaf
233 node and has children `children_[i - n_samples]`. Alternatively
234 at the i-th iteration, children[i][0] and children[i][1]
235 are merged to form node `n_samples + i`.
236
237 n_connected_components : int
238 The number of connected components in the graph.
239
240 n_leaves : int
241 The number of leaves in the tree.
242
243 parents : ndarray of shape (n_nodes,) or None
244 The parent of each node. Only returned when a connectivity matrix
245 is specified, elsewhere 'None' is returned.
246
247 distances : ndarray of shape (n_nodes-1,)
248 Only returned if `return_distance` is set to `True` (for compatibility).
249 The distances between the centers of the nodes. `distances[i]`
250 corresponds to a weighted Euclidean distance between
251 the nodes `children[i, 1]` and `children[i, 2]`. If the nodes refer to
252 leaves of the tree, then `distances[i]` is their unweighted Euclidean
253 distance. Distances are updated in the following way
254 (from scipy.hierarchy.linkage):
255
256 The new entry :math:`d(u,v)` is computed as follows,
257
258 .. math::
259
260 d(u,v) = \\sqrt{\\frac{|v|+|s|}
261 {T}d(v,s)^2
262 + \\frac{|v|+|t|}
263 {T}d(v,t)^2
264 - \\frac{|v|}
265 {T}d(s,t)^2}
266
267 where :math:`u` is the newly joined cluster consisting of
268 clusters :math:`s` and :math:`t`, :math:`v` is an unused
269 cluster in the forest, :math:`T=|v|+|s|+|t|`, and
270 :math:`|*|` is the cardinality of its argument. This is also
271 known as the incremental algorithm.
272
273 Examples
274 --------
275 >>> import numpy as np
276 >>> from sklearn.cluster import ward_tree
277 >>> X = np.array([[1, 2], [1, 4], [1, 0],
278 ... [4, 2], [4, 4], [4, 0]])
279 >>> children, n_connected_components, n_leaves, parents = ward_tree(X)
280 >>> children
281 array([[0, 1],
282 [3, 5],
283 [2, 6],
284 [4, 7],
285 [8, 9]])
286 >>> n_connected_components
287 1
288 >>> n_leaves
289 6
290 """
291 X = np.asarray(X)
292 if X.ndim == 1:
293 X = np.reshape(X, (-1, 1))
294 n_samples, n_features = X.shape
295
296 if connectivity is None:
297 from scipy.cluster import hierarchy # imports PIL
298
299 if n_clusters is not None:
300 warnings.warn(
301 (
302 "Partial build of the tree is implemented "
303 "only for structured clustering (i.e. with "
304 "explicit connectivity). The algorithm "
305 "will build the full tree and only "
306 "retain the lower branches required "
307 "for the specified number of clusters"
308 ),
309 stacklevel=2,
310 )
311 X = np.require(X, requirements="W")
312 out = hierarchy.ward(X)
313 children_ = out[:, :2].astype(np.intp)
314
315 if return_distance:
316 distances = out[:, 2]
317 return children_, 1, n_samples, None, distances
318 else:
319 return children_, 1, n_samples, None
320
321 connectivity, n_connected_components = _fix_connectivity(
322 X, connectivity, affinity="euclidean"
323 )
324 if n_clusters is None:
325 n_nodes = 2 * n_samples - 1
326 else:
327 if n_clusters > n_samples:
328 raise ValueError(
329 "Cannot provide more clusters than samples. "
330 "%i n_clusters was asked, and there are %i "
331 "samples." % (n_clusters, n_samples)
332 )
333 n_nodes = 2 * n_samples - n_clusters
334
335 # create inertia matrix
336 coord_row = []
337 coord_col = []
338 A = []
339 for ind, row in enumerate(connectivity.rows):
340 A.append(row)
341 # We keep only the upper triangular for the moments
342 # Generator expressions are faster than arrays on the following
343 row = [i for i in row if i < ind]
344 coord_row.extend(
345 len(row)
346 * [
347 ind,
348 ]
349 )
350 coord_col.extend(row)
351
352 coord_row = np.array(coord_row, dtype=np.intp, order="C")
353 coord_col = np.array(coord_col, dtype=np.intp, order="C")
354
355 # build moments as a list
356 moments_1 = np.zeros(n_nodes, order="C")
357 moments_1[:n_samples] = 1
358 moments_2 = np.zeros((n_nodes, n_features), order="C")
359 moments_2[:n_samples] = X
360 inertia = np.empty(len(coord_row), dtype=np.float64, order="C")
361 _hierarchical.compute_ward_dist(moments_1, moments_2, coord_row, coord_col, inertia)
362 inertia = list(zip(inertia, coord_row, coord_col))
363 heapify(inertia)
364
365 # prepare the main fields
366 parent = np.arange(n_nodes, dtype=np.intp)
367 used_node = np.ones(n_nodes, dtype=bool)
368 children = []
369 if return_distance:
370 distances = np.empty(n_nodes - n_samples)
371
372 not_visited = np.empty(n_nodes, dtype=bool, order="C")
373
374 # recursive merge loop
375 for k in range(n_samples, n_nodes):
376 # identify the merge
377 while True:
378 inert, i, j = heappop(inertia)
379 if used_node[i] and used_node[j]:
380 break
381 parent[i], parent[j] = k, k
382 children.append((i, j))
383 used_node[i] = used_node[j] = False
384 if return_distance: # store inertia value
385 distances[k - n_samples] = inert
386
387 # update the moments
388 moments_1[k] = moments_1[i] + moments_1[j]
389 moments_2[k] = moments_2[i] + moments_2[j]
390
391 # update the structure matrix A and the inertia matrix
392 coord_col = []
393 not_visited.fill(1)
394 not_visited[k] = 0
395 _hierarchical._get_parents(A[i], coord_col, parent, not_visited)
396 _hierarchical._get_parents(A[j], coord_col, parent, not_visited)
397 # List comprehension is faster than a for loop
398 [A[col].append(k) for col in coord_col]
399 A.append(coord_col)
400 coord_col = np.array(coord_col, dtype=np.intp, order="C")
401 coord_row = np.empty(coord_col.shape, dtype=np.intp, order="C")
402 coord_row.fill(k)
403 n_additions = len(coord_row)
404 ini = np.empty(n_additions, dtype=np.float64, order="C")
405
406 _hierarchical.compute_ward_dist(moments_1, moments_2, coord_row, coord_col, ini)
407
408 # List comprehension is faster than a for loop
409 [heappush(inertia, (ini[idx], k, coord_col[idx])) for idx in range(n_additions)]
410
411 # Separate leaves in children (empty lists up to now)
412 n_leaves = n_samples
413 # sort children to get consistent output with unstructured version
414 children = [c[::-1] for c in children]
415 children = np.array(children) # return numpy array for efficient caching
416
417 if return_distance:
418 # 2 is scaling factor to compare w/ unstructured version
419 distances = np.sqrt(2.0 * distances)
420 return children, n_connected_components, n_leaves, parent, distances
421 else:
422 return children, n_connected_components, n_leaves, parent

425# single average and complete linkage
426def linkage_tree(
427 X,
428 connectivity=None,
429 n_clusters=None,
430 linkage="complete",
431 affinity="euclidean",
432 return_distance=False,
433):
434 """Linkage agglomerative clustering based on a Feature matrix.
435
436 The inertia matrix uses a Heapq-based representation.
437
438 This is the structured version, that takes into account some topological
439 structure between samples.
440
441 Read more in the :ref:`User Guide <hierarchical_clustering>`.
442
443 Parameters
444 ----------
445 X : array-like of shape (n_samples, n_features)
446 Feature matrix representing `n_samples` samples to be clustered.
447
448 connectivity : sparse matrix, default=None
449 Connectivity matrix. Defines for each sample the neighboring samples
450 following a given structure of the data. The matrix is assumed to
451 be symmetric and only the upper triangular half is used.
452 Default is `None`, i.e, the Ward algorithm is unstructured.
453
454 n_clusters : int, default=None
455 Stop early the construction of the tree at `n_clusters`. This is
456 useful to decrease computation time if the number of clusters is
457 not small compared to the number of samples. In this case, the
458 complete tree is not computed, thus the 'children' output is of
459 limited use, and the 'parents' output should rather be used.
460 This option is valid only when specifying a connectivity matrix.
461
462 linkage : {"average", "complete", "single"}, default="complete"
463 Which linkage criteria to use. The linkage criterion determines which
464 distance to use between sets of observation.
465 - "average" uses the average of the distances of each observation of
466 the two sets.
467 - "complete" or maximum linkage uses the maximum distances between
468 all observations of the two sets.
469 - "single" uses the minimum of the distances between all
470 observations of the two sets.
471
472 affinity : str or callable, default='euclidean'
473 Which metric to use. Can be 'euclidean', 'manhattan', or any
474 distance known to paired distance (see metric.pairwise).
475
476 return_distance : bool, default=False
477 Whether or not to return the distances between the clusters.
478
479 Returns
480 -------
481 children : ndarray of shape (n_nodes-1, 2)
482 The children of each non-leaf node. Values less than `n_samples`
483 correspond to leaves of the tree which are the original samples.
484 A node `i` greater than or equal to `n_samples` is a non-leaf
485 node and has children `children_[i - n_samples]`. Alternatively
486 at the i-th iteration, children[i][0] and children[i][1]
487 are merged to form node `n_samples + i`.
488
489 n_connected_components : int
490 The number of connected components in the graph.
491
492 n_leaves : int
493 The number of leaves in the tree.
494
495 parents : ndarray of shape (n_nodes, ) or None
496 The parent of each node. Only returned when a connectivity matrix
497 is specified, elsewhere 'None' is returned.
498
499 distances : ndarray of shape (n_nodes-1,)
500 Returned when `return_distance` is set to `True`.
501
502 distances[i] refers to the distance between children[i][0] and
503 children[i][1] when they are merged.
504
505 See Also
506 --------
507 ward_tree : Hierarchical clustering with ward linkage.
508 """
509 X = np.asarray(X)
510 if X.ndim == 1:
511 X = np.reshape(X, (-1, 1))
512 n_samples, n_features = X.shape
513
514 linkage_choices = {
515 "complete": _hierarchical.max_merge,
516 "average": _hierarchical.average_merge,
517 "single": None,
518 } # Single linkage is handled differently
519 try:
520 join_func = linkage_choices[linkage]
521 except KeyError as e:
522 raise ValueError(
523 "Unknown linkage option, linkage should be one of %s, but %s was given"
524 % (linkage_choices.keys(), linkage)
525 ) from e
526
527 if affinity == "cosine" and np.any(~np.any(X, axis=1)):
528 raise ValueError("Cosine affinity cannot be used when X contains zero vectors")
529
530 if connectivity is None:
531 from scipy.cluster import hierarchy # imports PIL
532
533 if n_clusters is not None:
534 warnings.warn(
535 (
536 "Partial build of the tree is implemented "
537 "only for structured clustering (i.e. with "
538 "explicit connectivity). The algorithm "
539 "will build the full tree and only "
540 "retain the lower branches required "
541 "for the specified number of clusters"
542 ),
543 stacklevel=2,
544 )
545
546 if affinity == "precomputed":
547 # for the linkage function of hierarchy to work on precomputed
548 # data, provide as first argument an ndarray of the shape returned
549 # by sklearn.metrics.pairwise_distances.
550 if X.shape[0] != X.shape[1]:
551 raise ValueError(
552 f"Distance matrix should be square, got matrix of shape {X.shape}"
553 )
554 i, j = np.triu_indices(X.shape[0], k=1)
555 X = X[i, j]
556 elif affinity == "l2":
557 # Translate to something understood by scipy
558 affinity = "euclidean"
559 elif affinity in ("l1", "manhattan"):
560 affinity = "cityblock"
561 elif callable(affinity):
562 X = affinity(X)
563 i, j = np.triu_indices(X.shape[0], k=1)
564 X = X[i, j]
565 if (
566 linkage == "single"
567 and affinity != "precomputed"
568 and not callable(affinity)
569 and affinity in METRIC_MAPPING64
570 ):
571 # We need the fast cythonized metric from neighbors
572 dist_metric = DistanceMetric.get_metric(affinity)
573
574 # The Cython routines used require contiguous arrays
575 X = np.ascontiguousarray(X, dtype=np.double)
576
577 mst = _hierarchical.mst_linkage_core(X, dist_metric)
578 # Sort edges of the min_spanning_tree by weight
579 mst = mst[np.argsort(mst.T[2], kind="mergesort"), :]
580
581 # Convert edge list into standard hierarchical clustering format
582 out = _hierarchical.single_linkage_label(mst)
583 else:
584 out = hierarchy.linkage(X, method=linkage, metric=affinity)
585 children_ = out[:, :2].astype(int, copy=False)
586
587 if return_distance:
588 distances = out[:, 2]
589 return children_, 1, n_samples, None, distances
590 return children_, 1, n_samples, None
591
592 connectivity, n_connected_components = _fix_connectivity(
593 X, connectivity, affinity=affinity
594 )
595 connectivity = connectivity.tocoo()
596 # Put the diagonal to zero
597 diag_mask = connectivity.row != connectivity.col
598 connectivity.row = connectivity.row[diag_mask]
599 connectivity.col = connectivity.col[diag_mask]
600 connectivity.data = connectivity.data[diag_mask]
601 del diag_mask
602
603 if affinity == "precomputed":
604 distances = X[connectivity.row, connectivity.col].astype(np.float64, copy=False)
605 else:
606 # FIXME We compute all the distances, while we could have only computed
607 # the "interesting" distances
608 distances = paired_distances(
609 X[connectivity.row], X[connectivity.col], metric=affinity
610 )
611 connectivity.data = distances
612
613 if n_clusters is None:
614 n_nodes = 2 * n_samples - 1
615 else:
616 assert n_clusters <= n_samples
617 n_nodes = 2 * n_samples - n_clusters
618
619 if linkage == "single":
620 return _single_linkage_tree(
621 connectivity,
622 n_samples,
623 n_nodes,
624 n_clusters,
625 n_connected_components,
626 return_distance,
627 )
628
629 if return_distance:
630 distances = np.empty(n_nodes - n_samples)
631 # create inertia heap and connection matrix
632 A = np.empty(n_nodes, dtype=object)
633 inertia = list()
634
635 # LIL seems to the best format to access the rows quickly,
636 # without the numpy overhead of slicing CSR indices and data.
637 connectivity = connectivity.tolil()
638 # We are storing the graph in a list of IntFloatDict
639 for ind, (data, row) in enumerate(zip(connectivity.data, connectivity.rows)):
640 A[ind] = IntFloatDict(
641 np.asarray(row, dtype=np.intp), np.asarray(data, dtype=np.float64)
642 )
643 # We keep only the upper triangular for the heap
644 # Generator expressions are faster than arrays on the following
645 inertia.extend(
646 _hierarchical.WeightedEdge(d, ind, r) for r, d in zip(row, data) if r < ind
647 )
648 del connectivity
649
650 heapify(inertia)
651
652 # prepare the main fields
653 parent = np.arange(n_nodes, dtype=np.intp)
654 used_node = np.ones(n_nodes, dtype=np.intp)
655 children = []
656
657 # recursive merge loop
658 for k in range(n_samples, n_nodes):
659 # identify the merge
660 while True:
661 edge = heappop(inertia)
662 if used_node[edge.a] and used_node[edge.b]:
663 break
664 i = edge.a
665 j = edge.b
666
667 if return_distance:
668 # store distances
669 distances[k - n_samples] = edge.weight
670
671 parent[i] = parent[j] = k
672 children.append((i, j))
673 # Keep track of the number of elements per cluster
674 n_i = used_node[i]
675 n_j = used_node[j]
676 used_node[k] = n_i + n_j
677 used_node[i] = used_node[j] = False
678
679 # update the structure matrix A and the inertia matrix
680 # a clever 'min', or 'max' operation between A[i] and A[j]
681 coord_col = join_func(A[i], A[j], used_node, n_i, n_j)
682 for col, d in coord_col:
683 A[col].append(k, d)
684 # Here we use the information from coord_col (containing the
685 # distances) to update the heap
686 heappush(inertia, _hierarchical.WeightedEdge(d, k, col))
687 A[k] = coord_col
688 # Clear A[i] and A[j] to save memory
689 A[i] = A[j] = 0
690
691 # Separate leaves in children (empty lists up to now)
692 n_leaves = n_samples
693
694 # # return numpy array for efficient caching
695 children = np.array(children)[:, ::-1]
696
697 if return_distance:
698 return children, n_connected_components, n_leaves, parent, distances
699 return children, n_connected_components, n_leaves, parent

729def _hc_cut(n_clusters, children, n_leaves):
730 """Function cutting the ward tree for a given number of clusters.
731
732 Parameters
733 ----------
734 n_clusters : int or ndarray
735 The number of clusters to form.
736
737 children : ndarray of shape (n_nodes-1, 2)
738 The children of each non-leaf node. Values less than `n_samples`
739 correspond to leaves of the tree which are the original samples.
740 A node `i` greater than or equal to `n_samples` is a non-leaf
741 node and has children `children_[i - n_samples]`. Alternatively
742 at the i-th iteration, children[i][0] and children[i][1]
743 are merged to form node `n_samples + i`.
744
745 n_leaves : int
746 Number of leaves of the tree.
747
748 Returns
749 -------
750 labels : array [n_samples]
751 Cluster labels for each point.
752 """
753 if n_clusters > n_leaves:
754 raise ValueError(
755 "Cannot extract more clusters than samples: "
756 f"{n_clusters} clusters were given for a tree with {n_leaves} leaves."
757 )
758 # In this function, we store nodes as a heap to avoid recomputing
759 # the max of the nodes: the first element is always the smallest
760 # We use negated indices as heaps work on smallest elements, and we
761 # are interested in largest elements
762 # children[-1] is the root of the tree
763 nodes = [-(max(children[-1]) + 1)]
764 for _ in range(n_clusters - 1):
765 # As we have a heap, nodes[0] is the smallest element
766 these_children = children[-nodes[0] - n_leaves]
767 # Insert the 2 children and remove the largest node
768 heappush(nodes, -these_children[0])
769 heappushpop(nodes, -these_children[1])
770 label = np.zeros(n_leaves, dtype=np.intp)
771 for i, node in enumerate(nodes):
772 label[_hierarchical._hc_get_descendent(-node, children, n_leaves)] = i
773 return label

993 def _fit(self, X):
994 """Fit without validation
995
996 Parameters
997 ----------
998 X : ndarray of shape (n_samples, n_features) or (n_samples, n_samples)
999 Training instances to cluster, or distances between instances if
1000 ``metric='precomputed'``.
1001
1002 Returns
1003 -------
1004 self : object
1005 Returns the fitted instance.
1006 """
1007 memory = check_memory(self.memory)
1008
1009 if not ((self.n_clusters is None) ^ (self.distance_threshold is None)):
1010 raise ValueError(
1011 "Exactly one of n_clusters and "
1012 "distance_threshold has to be set, and the other "
1013 "needs to be None."
1014 )
1015
1016 if self.distance_threshold is not None and not self.compute_full_tree:
1017 raise ValueError(
1018 "compute_full_tree must be True if distance_threshold is set."
1019 )
1020
1021 if self.linkage == "ward" and self.metric != "euclidean":
1022 raise ValueError(
1023 f"{self.metric} was provided as metric. Ward can only "
1024 "work with euclidean distances."
1025 )
1026
1027 tree_builder = _TREE_BUILDERS[self.linkage]
1028
1029 connectivity = self.connectivity
1030 if self.connectivity is not None:
1031 if callable(self.connectivity):
1032 connectivity = self.connectivity(X)
1033 connectivity = check_array(
1034 connectivity, accept_sparse=["csr", "coo", "lil"]
1035 )
1036
1037 n_samples = len(X)
1038 compute_full_tree = self.compute_full_tree
1039 if self.connectivity is None:
1040 compute_full_tree = True
1041 if compute_full_tree == "auto":
1042 if self.distance_threshold is not None:
1043 compute_full_tree = True
1044 else:
1045 # Early stopping is likely to give a speed up only for
1046 # a large number of clusters. The actual threshold
1047 # implemented here is heuristic
1048 compute_full_tree = self.n_clusters < max(100, 0.02 * n_samples)
1049 n_clusters = self.n_clusters
1050 if compute_full_tree:
1051 n_clusters = None
1052
1053 # Construct the tree
1054 kwargs = {}
1055 if self.linkage != "ward":
1056 kwargs["linkage"] = self.linkage
1057 kwargs["affinity"] = self.metric
1058
1059 distance_threshold = self.distance_threshold
1060
1061 return_distance = (distance_threshold is not None) or self.compute_distances
1062
1063 out = memory.cache(tree_builder)(
1064 X,
1065 connectivity=connectivity,
1066 n_clusters=n_clusters,
1067 return_distance=return_distance,
1068 **kwargs,
1069 )
1070 (self.children_, self.n_connected_components_, self.n_leaves_, parents) = out[
1071 :4
1072 ]
1073
1074 if return_distance:
1075 self.distances_ = out[-1]
1076
1077 if self.distance_threshold is not None: # distance_threshold is used
1078 self.n_clusters_ = (
1079 np.count_nonzero(self.distances_ >= distance_threshold) + 1
1080 )
1081 else: # n_clusters is used
1082 self.n_clusters_ = self.n_clusters
1083
1084 # Cut the tree
1085 if compute_full_tree:
1086 self.labels_ = _hc_cut(self.n_clusters_, self.children_, self.n_leaves_)
1087 else:
1088 labels = _hierarchical.hc_get_heads(parents, copy=False)
1089 # copy to avoid holding a reference on the original array
1090 labels = np.copy(labels[:n_samples])
1091 # Reassign cluster numbers
1092 self.labels_ = np.searchsorted(np.unique(labels), labels)
1093 return self
