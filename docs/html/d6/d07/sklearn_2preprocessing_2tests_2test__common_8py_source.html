56def test_missing_value_handling(
57 est, func, support_sparse, strictly_positive, omit_kwargs
58):
59 # check that the preprocessing method let pass nan
60 rng = np.random.RandomState(42)
61 X = iris.data.copy()
62 n_missing = 50
63 X[
64 rng.randint(X.shape[0], size=n_missing), rng.randint(X.shape[1], size=n_missing)
65 ] = np.nan
66 if strictly_positive:
67 X += np.nanmin(X) + 0.1
68 X_train, X_test = train_test_split(X, random_state=1)
69 # sanity check
70 assert not np.all(np.isnan(X_train), axis=0).any()
71 assert np.any(np.isnan(X_train), axis=0).all()
72 assert np.any(np.isnan(X_test), axis=0).all()
73 X_test[:, 0] = np.nan # make sure this boundary case is tested
74
75 with warnings.catch_warnings():
76 warnings.simplefilter("error", RuntimeWarning)
77 Xt = est.fit(X_train).transform(X_test)
78 # ensure no warnings are raised
79 # missing values should still be missing, and only them
80 assert_array_equal(np.isnan(Xt), np.isnan(X_test))
81
82 # check that the function leads to the same results as the class
83 with warnings.catch_warnings():
84 warnings.simplefilter("error", RuntimeWarning)
85 Xt_class = est.transform(X_train)
86 kwargs = est.get_params()
87 # remove the parameters which should be omitted because they
88 # are not defined in the counterpart function of the preprocessing class
89 for kwarg in omit_kwargs:
90 _ = kwargs.pop(kwarg)
91 Xt_func = func(X_train, **kwargs)
92 assert_array_equal(np.isnan(Xt_func), np.isnan(Xt_class))
93 assert_allclose(Xt_func[~np.isnan(Xt_func)], Xt_class[~np.isnan(Xt_class)])
94
95 # check that the inverse transform keep NaN
96 Xt_inv = est.inverse_transform(Xt)
97 assert_array_equal(np.isnan(Xt_inv), np.isnan(X_test))
98 # FIXME: we can introduce equal_nan=True in recent version of numpy.
99 # For the moment which just check that non-NaN values are almost equal.
100 assert_allclose(Xt_inv[~np.isnan(Xt_inv)], X_test[~np.isnan(X_test)])
101
102 for i in range(X.shape[1]):
103 # train only on non-NaN
104 est.fit(_get_valid_samples_by_column(X_train, i))
105 # check transforming with NaN works even when training without NaN
106 with warnings.catch_warnings():
107 warnings.simplefilter("error", RuntimeWarning)
108 Xt_col = est.transform(X_test[:, [i]])
109 assert_allclose(Xt_col, Xt[:, [i]])
110 # check non-NaN is handled as before - the 1st column is all nan
111 if not np.isnan(X_test[:, i]).all():
112 Xt_col_nonan = est.transform(_get_valid_samples_by_column(X_test, i))
113 assert_array_equal(Xt_col_nonan, Xt_col[~np.isnan(Xt_col.squeeze())])
