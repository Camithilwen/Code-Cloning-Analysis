43def generate_data(
44 n_samples, n_features, weights, means, precisions, covariance_type, dtype=np.float64
45):
46 rng = np.random.RandomState(0)
47
48 X = []
49 if covariance_type == "spherical":
50 for _, (w, m, c) in enumerate(zip(weights, means, precisions["spherical"])):
51 X.append(
52 rng.multivariate_normal(
53 m, c * np.eye(n_features), int(np.round(w * n_samples))
54 ).astype(dtype)
55 )
56 if covariance_type == "diag":
57 for _, (w, m, c) in enumerate(zip(weights, means, precisions["diag"])):
58 X.append(
59 rng.multivariate_normal(
60 m, np.diag(c), int(np.round(w * n_samples))
61 ).astype(dtype)
62 )
63 if covariance_type == "tied":
64 for _, (w, m) in enumerate(zip(weights, means)):
65 X.append(
66 rng.multivariate_normal(
67 m, precisions["tied"], int(np.round(w * n_samples))
68 ).astype(dtype)
69 )
70 if covariance_type == "full":
71 for _, (w, m, c) in enumerate(zip(weights, means, precisions["full"])):
72 X.append(
73 rng.multivariate_normal(m, c, int(np.round(w * n_samples))).astype(
74 dtype
75 )
76 )
77
78 X = np.vstack(X)
79 return X

82class RandomData:
83 def __init__(
84 self,
85 rng,
86 n_samples=200,
87 n_components=2,
88 n_features=2,
89 scale=50,
90 dtype=np.float64,
91 ):
92 self.n_samples = n_samples
93 self.n_components = n_components
94 self.n_features = n_features
95
96 self.weights = rng.rand(n_components).astype(dtype)
97 self.weights = self.weights.astype(dtype) / self.weights.sum()
98 self.means = rng.rand(n_components, n_features).astype(dtype) * scale
99 self.covariances = {
100 "spherical": 0.5 + rng.rand(n_components).astype(dtype),
101 "diag": (0.5 + rng.rand(n_components, n_features).astype(dtype)) ** 2,
102 "tied": make_spd_matrix(n_features, random_state=rng).astype(dtype),
103 "full": np.array(
104 [
105 make_spd_matrix(n_features, random_state=rng).astype(dtype) * 0.5
106 for _ in range(n_components)
107 ]
108 ),
109 }
110 self.precisions = {
111 "spherical": 1.0 / self.covariances["spherical"],
112 "diag": 1.0 / self.covariances["diag"],
113 "tied": linalg.inv(self.covariances["tied"]),
114 "full": np.array(
115 [linalg.inv(covariance) for covariance in self.covariances["full"]]
116 ),
117 }
118
119 self.X = dict(
120 zip(
121 COVARIANCE_TYPE,
122 [
123 generate_data(
124 n_samples,
125 n_features,
126 self.weights,
127 self.means,
128 self.covariances,
129 covar_type,
130 dtype=dtype,
131 )
132 for covar_type in COVARIANCE_TYPE
133 ],
134 )
135 )
136 self.Y = np.hstack(
137 [
138 np.full(int(np.round(w * n_samples)), k, dtype=int)
139 for k, w in enumerate(self.weights)
140 ]
141 )

343 rng = np.random.RandomState(0)
344 n_samples, n_features, n_components = 500, 2, 2

420def test_compute_log_det_cholesky(global_dtype):
421 n_features = 2
422 rand_data = RandomData(np.random.RandomState(0), dtype=global_dtype)
423
424 for covar_type in COVARIANCE_TYPE:
425 covariance = rand_data.covariances[covar_type]
426
427 if covar_type == "full":
428 predected_det = np.array([linalg.det(cov) for cov in covariance])
429 elif covar_type == "tied":
430 predected_det = linalg.det(covariance)
431 elif covar_type == "diag":
432 predected_det = np.array([np.prod(cov) for cov in covariance])
433 elif covar_type == "spherical":
434 predected_det = covariance**n_features
435
436 # We compute the cholesky decomposition of the covariance matrix
437 assert covariance.dtype == global_dtype

439 _compute_precision_cholesky(covariance, covar_type),
440 covar_type,
441 n_features=n_features,
442 )
443 assert_array_almost_equal(expected_det, -0.5 * np.log(predected_det))
444 assert expected_det.dtype == global_dtype

447def _naive_lmvnpdf_diag(X, means, covars):
448 resp = np.empty((len(X), len(means)))
449 stds = np.sqrt(covars)
450 for i, (mean, std) in enumerate(zip(means, stds)):
451 resp[:, i] = stats.norm.logpdf(X, mean, std).sum(axis=1)
452 return resp

1308def test_gaussian_mixture_precisions_init_diag(global_dtype):
1309 """Check that we properly initialize `precision_cholesky_` when we manually
1310 provide the precision matrix.
1311
1312 In this regard, we check the consistency between estimating the precision
1313 matrix and providing the same precision matrix as initialization. It should
1314 lead to the same results with the same number of iterations.
1315
1316 If the initialization is wrong then the number of iterations will increase.
1317
1318 Non-regression test for:
1319 https://github.com/scikit-learn/scikit-learn/issues/16944
1320 """
1321 # generate a toy dataset
1322 n_samples = 300
1323 rng = np.random.RandomState(0)
1324 shifted_gaussian = rng.randn(n_samples, 2) + np.array([20, 20])
1325 C = np.array([[0.0, -0.7], [3.5, 0.7]])
1326 stretched_gaussian = np.dot(rng.randn(n_samples, 2), C)
1327 X = np.vstack([shifted_gaussian, stretched_gaussian]).astype(global_dtype)
1328
1329 # common parameters to check the consistency of precision initialization
1330 n_components, covariance_type, reg_covar, random_state = 2, "diag", 1e-6, 0
1331
1332 # execute the manual initialization to compute the precision matrix:
1333 # - run KMeans to have an initial guess
1334 # - estimate the covariance
1335 # - compute the precision matrix from the estimated covariance
1336 resp = np.zeros((X.shape[0], n_components)).astype(global_dtype)
1337 label = (
1338 KMeans(n_clusters=n_components, n_init=1, random_state=random_state)
1339 .fit(X)
1340 .labels_
1341 )
1342 resp[np.arange(X.shape[0]), label] = 1
1343 _, _, covariance = _estimate_gaussian_parameters(
1344 X, resp, reg_covar=reg_covar, covariance_type=covariance_type
1345 )
1346 assert covariance.dtype == global_dtype
1347 precisions_init = 1 / covariance
1348
1349 gm_with_init = GaussianMixture(
1350 n_components=n_components,
1351 covariance_type=covariance_type,
1352 reg_covar=reg_covar,
1353 precisions_init=precisions_init,
1354 random_state=random_state,
1355 ).fit(X)
1356 assert gm_with_init.means_.dtype == global_dtype
1357 assert gm_with_init.covariances_.dtype == global_dtype
1358 assert gm_with_init.precisions_cholesky_.dtype == global_dtype
1359
1360 gm_without_init = GaussianMixture(
1361 n_components=n_components,
1362 covariance_type=covariance_type,
1363 reg_covar=reg_covar,
1364 random_state=random_state,
1365 ).fit(X)
1366 assert gm_without_init.means_.dtype == global_dtype
1367 assert gm_without_init.covariances_.dtype == global_dtype
1368 assert gm_without_init.precisions_cholesky_.dtype == global_dtype
1369
1370 assert gm_without_init.n_iter_ == gm_with_init.n_iter_
1371 assert_allclose(
1372 gm_with_init.precisions_cholesky_, gm_without_init.precisions_cholesky_
1373 )

1387 for the given covariance type.
