66 -MAXFLT, INF, -INF, qNAN, sNAN,

68
69 // [-1.0, 1.0]

72 }
73 // (0.0, INF]
74 else if ((func == "log2") || (func == "log10")) {
75 append_random_array<T>(input, 0.0, 1.0, 200);
