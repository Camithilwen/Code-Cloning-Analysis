29def _modified_weiszfeld_step(X, x_old):
30 """Modified Weiszfeld step.
31
32 This function defines one iteration step in order to approximate the
33 spatial median (L1 median). It is a form of an iteratively re-weighted
34 least squares method.
35
36 Parameters
37 ----------
38 X : array-like of shape (n_samples, n_features)
39 Training vector, where `n_samples` is the number of samples and
40 `n_features` is the number of features.
41
42 x_old : ndarray of shape = (n_features,)
43 Current start vector.
44
45 Returns
46 -------
47 x_new : ndarray of shape (n_features,)
48 New iteration step.
49
50 References
51 ----------
52 - On Computation of Spatial Median for Robust Data Mining, 2005
53 T. Kärkkäinen and S. Äyrämö
54 http://users.jyu.fi/~samiayr/pdf/ayramo_eurogen05.pdf
55 """
56 diff = X - x_old
57 diff_norm = np.sqrt(np.sum(diff**2, axis=1))
58 mask = diff_norm >= _EPSILON
59 # x_old equals one of our samples
60 is_x_old_in_X = int(mask.sum() < X.shape[0])
61
62 diff = diff[mask]
63 diff_norm = diff_norm[mask][:, np.newaxis]
64 quotient_norm = linalg.norm(np.sum(diff / diff_norm, axis=0))
65
66 if quotient_norm > _EPSILON: # to avoid division by zero
67 new_direction = np.sum(X[mask, :] / diff_norm, axis=0) / np.sum(
68 1 / diff_norm, axis=0
69 )
70 else:
71 new_direction = 1.0
72 quotient_norm = 1.0
73
74 return (
75 max(0.0, 1.0 - is_x_old_in_X / quotient_norm) * new_direction
76 + min(1.0, is_x_old_in_X / quotient_norm) * x_old
77 )

80def _spatial_median(X, max_iter=300, tol=1.0e-3):
81 """Spatial median (L1 median).
82
83 The spatial median is member of a class of so-called M-estimators which
84 are defined by an optimization problem. Given a number of p points in an
85 n-dimensional space, the point x minimizing the sum of all distances to the
86 p other points is called spatial median.
87
88 Parameters
89 ----------
90 X : array-like of shape (n_samples, n_features)
91 Training vector, where `n_samples` is the number of samples and
92 `n_features` is the number of features.
93
94 max_iter : int, default=300
95 Maximum number of iterations.
96
97 tol : float, default=1.e-3
98 Stop the algorithm if spatial_median has converged.
99
100 Returns
101 -------
102 spatial_median : ndarray of shape = (n_features,)
103 Spatial median.
104
105 n_iter : int
106 Number of iterations needed.
107
108 References
109 ----------
110 - On Computation of Spatial Median for Robust Data Mining, 2005
111 T. Kärkkäinen and S. Äyrämö
112 http://users.jyu.fi/~samiayr/pdf/ayramo_eurogen05.pdf
113 """
114 if X.shape[1] == 1:
115 return 1, np.median(X.ravel(), keepdims=True)
116
117 tol **= 2 # We are computing the tol on the squared norm
118 spatial_median_old = np.mean(X, axis=0)
119
120 for n_iter in range(max_iter):
121 spatial_median = _modified_weiszfeld_step(X, spatial_median_old)
122 if np.sum((spatial_median_old - spatial_median) ** 2) < tol:
123 break
124 else:
125 spatial_median_old = spatial_median
126 else:
127 warnings.warn(
128 "Maximum number of iterations {max_iter} reached in "
129 "spatial median for TheilSen regressor."
130 "".format(max_iter=max_iter),
131 ConvergenceWarning,
132 )
133 return n_iter, spatial_median

136def _breakdown_point(n_samples, n_subsamples):
137 """Approximation of the breakdown point.
138
139 Parameters
140 ----------
141 n_samples : int
142 Number of samples.
143
144 n_subsamples : int
145 Number of subsamples to consider.
146
147 Returns
148 -------
149 breakdown_point : float
150 Approximation of breakdown point.
151 """
152 return (
153 1
154 - (
155 0.5 ** (1 / n_subsamples) * (n_samples - n_subsamples + 1)
156 + n_subsamples
157 - 1
158 )
159 / n_samples
160 )

163def _lstsq(X, y, indices, fit_intercept):
164 """Least Squares Estimator for TheilSenRegressor class.
165
166 This function calculates the least squares method on a subset of rows of X
167 and y defined by the indices array. Optionally, an intercept column is
168 added if intercept is set to true.
169
170 Parameters
171 ----------
172 X : array-like of shape (n_samples, n_features)
173 Design matrix, where `n_samples` is the number of samples and
174 `n_features` is the number of features.
175
176 y : ndarray of shape (n_samples,)
177 Target vector, where `n_samples` is the number of samples.
178
179 indices : ndarray of shape (n_subpopulation, n_subsamples)
180 Indices of all subsamples with respect to the chosen subpopulation.
181
182 fit_intercept : bool
183 Fit intercept or not.
184
185 Returns
186 -------
187 weights : ndarray of shape (n_subpopulation, n_features + intercept)
188 Solution matrix of n_subpopulation solved least square problems.
189 """
190 fit_intercept = int(fit_intercept)
191 n_features = X.shape[1] + fit_intercept
192 n_subsamples = indices.shape[1]
193 weights = np.empty((indices.shape[0], n_features))
194 X_subpopulation = np.ones((n_subsamples, n_features))
195 # gelss need to pad y_subpopulation to be of the max dim of X_subpopulation
196 y_subpopulation = np.zeros((max(n_subsamples, n_features)))
197 (lstsq,) = get_lapack_funcs(("gelss",), (X_subpopulation, y_subpopulation))
198
199 for index, subset in enumerate(indices):
200 X_subpopulation[:, fit_intercept:] = X[subset, :]
201 y_subpopulation[:n_subsamples] = y[subset]
202 weights[index] = lstsq(X_subpopulation, y_subpopulation)[1][:n_features]
203
204 return weights

207class TheilSenRegressor(RegressorMixin, LinearModel):
208 """Theil-Sen Estimator: robust multivariate regression model.
209
210 The algorithm calculates least square solutions on subsets with size
211 n_subsamples of the samples in X. Any value of n_subsamples between the
212 number of features and samples leads to an estimator with a compromise
213 between robustness and efficiency. Since the number of least square
214 solutions is "n_samples choose n_subsamples", it can be extremely large
215 and can therefore be limited with max_subpopulation. If this limit is
216 reached, the subsets are chosen randomly. In a final step, the spatial
217 median (or L1 median) is calculated of all least square solutions.
218
219 Read more in the :ref:`User Guide <theil_sen_regression>`.
220
221 Parameters
222 ----------
223 fit_intercept : bool, default=True
224 Whether to calculate the intercept for this model. If set
225 to false, no intercept will be used in calculations.
226
227 copy_X : bool, default=True
228 If True, X will be copied; else, it may be overwritten.
229
230 .. deprecated:: 1.6
231 `copy_X` was deprecated in 1.6 and will be removed in 1.8.
232 It has no effect as a copy is always made.
233
234 max_subpopulation : int, default=1e4
235 Instead of computing with a set of cardinality 'n choose k', where n is
236 the number of samples and k is the number of subsamples (at least
237 number of features), consider only a stochastic subpopulation of a
238 given maximal size if 'n choose k' is larger than max_subpopulation.
239 For other than small problem sizes this parameter will determine
240 memory usage and runtime if n_subsamples is not changed. Note that the
241 data type should be int but floats such as 1e4 can be accepted too.
242
243 n_subsamples : int, default=None
244 Number of samples to calculate the parameters. This is at least the
245 number of features (plus 1 if fit_intercept=True) and the number of
246 samples as a maximum. A lower number leads to a higher breakdown
247 point and a low efficiency while a high number leads to a low
248 breakdown point and a high efficiency. If None, take the
249 minimum number of subsamples leading to maximal robustness.
250 If n_subsamples is set to n_samples, Theil-Sen is identical to least
251 squares.
252
253 max_iter : int, default=300
254 Maximum number of iterations for the calculation of spatial median.
255
256 tol : float, default=1e-3
257 Tolerance when calculating spatial median.
258
259 random_state : int, RandomState instance or None, default=None
260 A random number generator instance to define the state of the random
261 permutations generator. Pass an int for reproducible output across
262 multiple function calls.
263 See :term:`Glossary <random_state>`.
264
265 n_jobs : int, default=None
266 Number of CPUs to use during the cross validation.
267 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
268 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
269 for more details.
270
271 verbose : bool, default=False
272 Verbose mode when fitting the model.
273
274 Attributes
275 ----------
276 coef_ : ndarray of shape (n_features,)
277 Coefficients of the regression model (median of distribution).
278
279 intercept_ : float
280 Estimated intercept of regression model.
281
282 breakdown_ : float
283 Approximated breakdown point.
284
285 n_iter_ : int
286 Number of iterations needed for the spatial median.
287
288 n_subpopulation_ : int
289 Number of combinations taken into account from 'n choose k', where n is
290 the number of samples and k is the number of subsamples.
291
292 n_features_in_ : int
293 Number of features seen during :term:`fit`.
294
295 .. versionadded:: 0.24
296
297 feature_names_in_ : ndarray of shape (`n_features_in_`,)
298 Names of features seen during :term:`fit`. Defined only when `X`
299 has feature names that are all strings.
300
301 .. versionadded:: 1.0
302
303 See Also
304 --------
305 HuberRegressor : Linear regression model that is robust to outliers.
306 RANSACRegressor : RANSAC (RANdom SAmple Consensus) algorithm.
307 SGDRegressor : Fitted by minimizing a regularized empirical loss with SGD.
308
309 References
310 ----------
311 - Theil-Sen Estimators in a Multiple Linear Regression Model, 2009
312 Xin Dang, Hanxiang Peng, Xueqin Wang and Heping Zhang
313 http://home.olemiss.edu/~xdang/papers/MTSE.pdf
314
315 Examples
316 --------
317 >>> from sklearn.linear_model import TheilSenRegressor
318 >>> from sklearn.datasets import make_regression
319 >>> X, y = make_regression(
320 ... n_samples=200, n_features=2, noise=4.0, random_state=0)
321 >>> reg = TheilSenRegressor(random_state=0).fit(X, y)
322 >>> reg.score(X, y)
323 0.9884
324 >>> reg.predict(X[:1,])
325 array([-31.5871])
326 """

364 def _check_subparams(self, n_samples, n_features):
365 n_subsamples = self.n_subsamples
366
367 if self.fit_intercept:
368 n_dim = n_features + 1
369 else:
370 n_dim = n_features
371
372 if n_subsamples is not None:
373 if n_subsamples > n_samples:
374 raise ValueError(
375 "Invalid parameter since n_subsamples > "
376 "n_samples ({0} > {1}).".format(n_subsamples, n_samples)
377 )
378 if n_samples >= n_features:
379 if n_dim > n_subsamples:
380 plus_1 = "+1" if self.fit_intercept else ""
381 raise ValueError(
382 "Invalid parameter since n_features{0} "
383 "> n_subsamples ({1} > {2})."
384 "".format(plus_1, n_dim, n_subsamples)
385 )
386 else: # if n_samples < n_features
387 if n_subsamples != n_samples:
388 raise ValueError(
389 "Invalid parameter since n_subsamples != "
390 "n_samples ({0} != {1}) while n_samples "
391 "< n_features.".format(n_subsamples, n_samples)
392 )
393 else:
394 n_subsamples = min(n_dim, n_samples)
395
396 all_combinations = max(1, np.rint(binom(n_samples, n_subsamples)))
397 n_subpopulation = int(min(self.max_subpopulation, all_combinations))
398
399 return n_subsamples, n_subpopulation

401 @_fit_context(prefer_skip_nested_validation=True)
402 def fit(self, X, y):
403 """Fit linear model.
404
405 Parameters
406 ----------
407 X : ndarray of shape (n_samples, n_features)
408 Training data.
409 y : ndarray of shape (n_samples,)
410 Target values.
411
412 Returns
413 -------
414 self : returns an instance of self.
415 Fitted `TheilSenRegressor` estimator.
416 """
417 if self.copy_X != "deprecated":
418 warnings.warn(
419 "`copy_X` was deprecated in 1.6 and will be removed in 1.8 since it "
420 "has no effect internally. Simply leave this parameter to its default "
421 "value to avoid this warning.",
422 FutureWarning,
423 )
424
425 random_state = check_random_state(self.random_state)
426 X, y = validate_data(self, X, y, y_numeric=True)
427 n_samples, n_features = X.shape
428 n_subsamples, self.n_subpopulation_ = self._check_subparams(
429 n_samples, n_features
430 )
431 self.breakdown_ = _breakdown_point(n_samples, n_subsamples)
432
433 if self.verbose:
434 print("Breakdown point: {0}".format(self.breakdown_))
435 print("Number of samples: {0}".format(n_samples))
436 tol_outliers = int(self.breakdown_ * n_samples)
437 print("Tolerable outliers: {0}".format(tol_outliers))
438 print("Number of subpopulations: {0}".format(self.n_subpopulation_))
439
440 # Determine indices of subpopulation
441 if np.rint(binom(n_samples, n_subsamples)) <= self.max_subpopulation:
442 indices = list(combinations(range(n_samples), n_subsamples))
443 else:
444 indices = [
445 random_state.choice(n_samples, size=n_subsamples, replace=False)
446 for _ in range(self.n_subpopulation_)
447 ]
448
449 n_jobs = effective_n_jobs(self.n_jobs)
450 index_list = np.array_split(indices, n_jobs)
451 weights = Parallel(n_jobs=n_jobs, verbose=self.verbose)(
452 delayed(_lstsq)(X, y, index_list[job], self.fit_intercept)
453 for job in range(n_jobs)
454 )
455 weights = np.vstack(weights)
456 self.n_iter_, coefs = _spatial_median(
457 weights, max_iter=self.max_iter, tol=self.tol
458 )
459
460 if self.fit_intercept:
461 self.intercept_ = coefs[0]
462 self.coef_ = coefs[1:]
463 else:
464 self.intercept_ = 0.0
465 self.coef_ = coefs
466
467 return self
