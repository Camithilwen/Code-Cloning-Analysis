103@pytest.mark.parametrize("range_, n_pts", [(_MAX_UNSIGNED_INT, 10000), (100, 25)])
104def test_newrand_bounded_rand_int(range_, n_pts):
105 """Test that `bounded_rand_int` follows a uniform distribution"""
106 # XXX: this test is very seed sensitive: either it is wrong (too strict?)
107 # or the wrapped RNG is not uniform enough, at least on some platforms.
108 set_seed_wrap(42)
109 n_iter = 100
110 ks_pvals = []
111 uniform_dist = stats.uniform(loc=0, scale=range_)
112 # perform multiple samplings to make chance of outlier sampling negligible
113 for _ in range(n_iter):
114 # Deterministic random sampling
115 sample = [bounded_rand_int_wrap(range_) for _ in range(n_pts)]
116 res = stats.kstest(sample, uniform_dist.cdf)
117 ks_pvals.append(res.pvalue)
118 # Null hypothesis = samples come from an uniform distribution.
119 # Under the null hypothesis, p-values should be uniformly distributed
120 # and not concentrated on low values
121 # (this may seem counter-intuitive but is backed by multiple refs)
122 # So we can do two checks:
123
124 # (1) check uniformity of p-values
125 uniform_p_vals_dist = stats.uniform(loc=0, scale=1)
126 res_pvals = stats.kstest(ks_pvals, uniform_p_vals_dist.cdf)
127 assert res_pvals.pvalue > 0.05, (
128 "Null hypothesis rejected: generated random numbers are not uniform."
129 " Details: the (meta) p-value of the test of uniform distribution"
130 f" of p-values is {res_pvals.pvalue} which is not > 0.05"
131 )
132
133 # (2) (safety belt) check that 90% of p-values are above 0.05
134 min_10pct_pval = np.percentile(ks_pvals, q=10)
135 # lower 10th quantile pvalue <= 0.05 means that the test rejects the
136 # null hypothesis that the sample came from the uniform distribution
137 assert min_10pct_pval > 0.05, (
138 "Null hypothesis rejected: generated random numbers are not uniform. "
139 f"Details: lower 10th quantile p-value of {min_10pct_pval} not > 0.05."
140 )
