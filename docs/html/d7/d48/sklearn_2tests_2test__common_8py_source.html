143def test_import_all_consistency():
144 sklearn_path = [os.path.dirname(sklearn.__file__)]
145 # Smoke test to check that any name in a __all__ list is actually defined
146 # in the namespace of the module or package.
147 pkgs = pkgutil.walk_packages(
148 path=sklearn_path, prefix="sklearn.", onerror=lambda _: None
149 )
150 submods = [modname for _, modname, _ in pkgs]
151 for modname in submods + ["sklearn"]:
152 if ".tests." in modname or "sklearn.externals" in modname:
153 continue
154 # Avoid test suite depending on build dependencies, for example Cython
155 if "sklearn._build_utils" in modname:
156 continue
157 package = __import__(modname, fromlist="dummy")
158 for name in getattr(package, "__all__", ()):
159 assert hasattr(package, name), "Module '{0}' has no attribute '{1}'".format(
160 modname, name
161 )

175def test_all_tests_are_importable():
176 # Ensure that for each contentful subpackage, there is a test directory
177 # within it that is also a subpackage (i.e. a directory with __init__.py)
178
179 HAS_TESTS_EXCEPTIONS = re.compile(
180 r"""(?x)
181 \.externals(\.|$)|
182 \.tests(\.|$)|
183 \._
184 """
185 )
186 resource_modules = {
187 "sklearn.datasets.data",
188 "sklearn.datasets.descr",
189 "sklearn.datasets.images",
190 }
191 sklearn_path = [os.path.dirname(sklearn.__file__)]
192 lookup = {
193 name: ispkg
194 for _, name, ispkg in pkgutil.walk_packages(sklearn_path, prefix="sklearn.")
195 }
196 missing_tests = [
197 name
198 for name, ispkg in lookup.items()
199 if ispkg
200 and name not in resource_modules
201 and not HAS_TESTS_EXCEPTIONS.search(name)
202 and name + ".tests" not in lookup
203 ]
204 assert missing_tests == [], (
205 "{0} do not have `tests` subpackages. "
206 "Perhaps they require "
207 "__init__.py or a meson.build "
208 "in the parent "
209 "directory".format(missing_tests)
210 )

225def _estimators_that_predict_in_fit():
226 for estimator in _tested_estimators():
227 est_params = set(estimator.get_params())
228 if "oob_score" in est_params:
229 yield estimator.set_params(oob_score=True, bootstrap=True)
230 elif "early_stopping" in est_params:
231 est = estimator.set_params(early_stopping=True, n_iter_no_change=1)
232 if est.__class__.__name__ in {"MLPClassifier", "MLPRegressor"}:
233 # TODO: FIX MLP to not check validation set during MLP
234 yield pytest.param(
235 est, marks=pytest.mark.xfail(msg="MLP still validates in fit")
236 )
237 else:
238 yield est
239 elif "n_iter_no_change" in est_params:
240 yield estimator.set_params(n_iter_no_change=1)
