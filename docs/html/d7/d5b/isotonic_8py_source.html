33def check_increasing(x, y):
34 """Determine whether y is monotonically correlated with x.
35
36 y is found increasing or decreasing with respect to x based on a Spearman
37 correlation test.
38
39 Parameters
40 ----------
41 x : array-like of shape (n_samples,)
42 Training data.
43
44 y : array-like of shape (n_samples,)
45 Training target.
46
47 Returns
48 -------
49 increasing_bool : boolean
50 Whether the relationship is increasing or decreasing.
51
52 Notes
53 -----
54 The Spearman correlation coefficient is estimated from the data, and the
55 sign of the resulting estimate is used as the result.
56
57 In the event that the 95% confidence interval based on Fisher transform
58 spans zero, a warning is raised.
59
60 References
61 ----------
62 Fisher transformation. Wikipedia.
63 https://en.wikipedia.org/wiki/Fisher_transformation
64
65 Examples
66 --------
67 >>> from sklearn.isotonic import check_increasing
68 >>> x, y = [1, 2, 3, 4, 5], [2, 4, 6, 8, 10]
69 >>> check_increasing(x, y)
70 np.True_
71 >>> y = [10, 8, 6, 4, 2]
72 >>> check_increasing(x, y)
73 np.False_
74 """
75
76 # Calculate Spearman rho estimate and set return accordingly.
77 rho, _ = spearmanr(x, y)
78 increasing_bool = rho >= 0
79
80 # Run Fisher transform to get the rho CI, but handle rho=+/-1
81 if rho not in [-1.0, 1.0] and len(x) > 3:
82 F = 0.5 * math.log((1.0 + rho) / (1.0 - rho))
83 F_se = 1 / math.sqrt(len(x) - 3)
84
85 # Use a 95% CI, i.e., +/-1.96 S.E.
86 # https://en.wikipedia.org/wiki/Fisher_transformation
87 rho_0 = math.tanh(F - 1.96 * F_se)
88 rho_1 = math.tanh(F + 1.96 * F_se)
89
90 # Warn if the CI spans zero.
91 if np.sign(rho_0) != np.sign(rho_1):
92 warnings.warn(
93 "Confidence interval of the Spearman "
94 "correlation coefficient spans zero. "
95 "Determination of ``increasing`` may be "
96 "suspect."
97 )
98
99 return increasing_bool

112def isotonic_regression(
113 y, *, sample_weight=None, y_min=None, y_max=None, increasing=True
114):
115 """Solve the isotonic regression model.
116
117 Read more in the :ref:`User Guide <isotonic>`.
118
119 Parameters
120 ----------
121 y : array-like of shape (n_samples,)
122 The data.
123
124 sample_weight : array-like of shape (n_samples,), default=None
125 Weights on each point of the regression.
126 If None, weight is set to 1 (equal weights).
127
128 y_min : float, default=None
129 Lower bound on the lowest predicted value (the minimum value may
130 still be higher). If not set, defaults to -inf.
131
132 y_max : float, default=None
133 Upper bound on the highest predicted value (the maximum may still be
134 lower). If not set, defaults to +inf.
135
136 increasing : bool, default=True
137 Whether to compute ``y_`` is increasing (if set to True) or decreasing
138 (if set to False).
139
140 Returns
141 -------
142 y_ : ndarray of shape (n_samples,)
143 Isotonic fit of y.
144
145 References
146 ----------
147 "Active set algorithms for isotonic regression; A unifying framework"
148 by Michael J. Best and Nilotpal Chakravarti, section 3.
149
150 Examples
151 --------
152 >>> from sklearn.isotonic import isotonic_regression
153 >>> isotonic_regression([5, 3, 1, 2, 8, 10, 7, 9, 6, 4])
154 array([2.75 , 2.75 , 2.75 , 2.75 , 7.33,
155 7.33, 7.33, 7.33, 7.33, 7.33])
156 """
157 y = check_array(y, ensure_2d=False, input_name="y", dtype=[np.float64, np.float32])
158 if sp_base_version >= parse_version("1.12.0"):
159 res = optimize.isotonic_regression(
160 y=y, weights=sample_weight, increasing=increasing
161 )
162 y = np.asarray(res.x, dtype=y.dtype)
163 else:
164 # TODO: remove this branch when Scipy 1.12 is the minimum supported version
165 # Also remove _inplace_contiguous_isotonic_regression.
166 order = np.s_[:] if increasing else np.s_[::-1]
167 y = np.array(y[order], dtype=y.dtype)
168 sample_weight = _check_sample_weight(sample_weight, y, dtype=y.dtype, copy=True)
169 sample_weight = np.ascontiguousarray(sample_weight[order])
170 _inplace_contiguous_isotonic_regression(y, sample_weight)
171 y = y[order]
172
173 if y_min is not None or y_max is not None:
174 # Older versions of np.clip don't accept None as a bound, so use np.inf
175 if y_min is None:
176 y_min = -np.inf
177 if y_max is None:
178 y_max = np.inf
179 np.clip(y, y_min, y_max, y)
180 return y

183class IsotonicRegression(RegressorMixin, TransformerMixin, BaseEstimator):
184 """Isotonic regression model.
185
186 Read more in the :ref:`User Guide <isotonic>`.
187
188 .. versionadded:: 0.13
189
190 Parameters
191 ----------
192 y_min : float, default=None
193 Lower bound on the lowest predicted value (the minimum value may
194 still be higher). If not set, defaults to -inf.
195
196 y_max : float, default=None
197 Upper bound on the highest predicted value (the maximum may still be
198 lower). If not set, defaults to +inf.
199
200 increasing : bool or 'auto', default=True
201 Determines whether the predictions should be constrained to increase
202 or decrease with `X`. 'auto' will decide based on the Spearman
203 correlation estimate's sign.
204
205 out_of_bounds : {'nan', 'clip', 'raise'}, default='nan'
206 Handles how `X` values outside of the training domain are handled
207 during prediction.
208
209 - 'nan', predictions will be NaN.
210 - 'clip', predictions will be set to the value corresponding to
211 the nearest train interval endpoint.
212 - 'raise', a `ValueError` is raised.
213
214 Attributes
215 ----------
216 X_min_ : float
217 Minimum value of input array `X_` for left bound.
218
219 X_max_ : float
220 Maximum value of input array `X_` for right bound.
221
222 X_thresholds_ : ndarray of shape (n_thresholds,)
223 Unique ascending `X` values used to interpolate
224 the y = f(X) monotonic function.
225
226 .. versionadded:: 0.24
227
228 y_thresholds_ : ndarray of shape (n_thresholds,)
229 De-duplicated `y` values suitable to interpolate the y = f(X)
230 monotonic function.
231
232 .. versionadded:: 0.24
233
234 f_ : function
235 The stepwise interpolating function that covers the input domain ``X``.
236
237 increasing_ : bool
238 Inferred value for ``increasing``.
239
240 See Also
241 --------
242 sklearn.linear_model.LinearRegression : Ordinary least squares Linear
243 Regression.
244 sklearn.ensemble.HistGradientBoostingRegressor : Gradient boosting that
245 is a non-parametric model accepting monotonicity constraints.
246 isotonic_regression : Function to solve the isotonic regression model.
247
248 Notes
249 -----
250 Ties are broken using the secondary method from de Leeuw, 1977.
251
252 References
253 ----------
254 Isotonic Median Regression: A Linear Programming Approach
255 Nilotpal Chakravarti
256 Mathematics of Operations Research
257 Vol. 14, No. 2 (May, 1989), pp. 303-308
258
259 Isotone Optimization in R : Pool-Adjacent-Violators
260 Algorithm (PAVA) and Active Set Methods
261 de Leeuw, Hornik, Mair
262 Journal of Statistical Software 2009
263
264 Correctness of Kruskal's algorithms for monotone regression with ties
265 de Leeuw, Psychometrica, 1977
266
267 Examples
268 --------
269 >>> from sklearn.datasets import make_regression
270 >>> from sklearn.isotonic import IsotonicRegression
271 >>> X, y = make_regression(n_samples=10, n_features=1, random_state=41)
272 >>> iso_reg = IsotonicRegression().fit(X, y)
273 >>> iso_reg.predict([.1, .2])
274 array([1.8628, 3.7256])
275 """

302 def _build_f(self, X, y):
303 """Build the f_ interp1d function."""
304
305 bounds_error = self.out_of_bounds == "raise"
306 if len(y) == 1:
307 # single y, constant prediction
308 self.f_ = lambda x: y.repeat(x.shape)
309 else:
310 self.f_ = interpolate.interp1d(
311 X, y, kind="linear", bounds_error=bounds_error
312 )

314 def _build_y(self, X, y, sample_weight, trim_duplicates=True):
315 """Build the y_ IsotonicRegression."""
316 self._check_input_data_shape(X)
317 X = X.reshape(-1) # use 1d view
318
319 # Determine increasing if auto-determination requested
320 if self.increasing == "auto":
321 self.increasing_ = check_increasing(X, y)
322 else:
323 self.increasing_ = self.increasing
324
325 # If sample_weights is passed, removed zero-weight values and clean
326 # order
327 sample_weight = _check_sample_weight(sample_weight, X, dtype=X.dtype)
328 mask = sample_weight > 0
329 X, y, sample_weight = X[mask], y[mask], sample_weight[mask]
330
331 order = np.lexsort((y, X))
332 X, y, sample_weight = [array[order] for array in [X, y, sample_weight]]
333 unique_X, unique_y, unique_sample_weight = _make_unique(X, y, sample_weight)
334
335 X = unique_X
336 y = isotonic_regression(
337 unique_y,
338 sample_weight=unique_sample_weight,
339 y_min=self.y_min,
340 y_max=self.y_max,
341 increasing=self.increasing_,
342 )
343
344 # Handle the left and right bounds on X
345 self.X_min_, self.X_max_ = np.min(X), np.max(X)
346
347 if trim_duplicates:
348 # Remove unnecessary points for faster prediction
349 keep_data = np.ones((len(y),), dtype=bool)
350 # Aside from the 1st and last point, remove points whose y values
351 # are equal to both the point before and the point after it.
352 keep_data[1:-1] = np.logical_or(
353 np.not_equal(y[1:-1], y[:-2]), np.not_equal(y[1:-1], y[2:])
354 )
355 return X[keep_data], y[keep_data]
356 else:
357 # The ability to turn off trim_duplicates is only used to it make
358 # easier to unit test that removing duplicates in y does not have
359 # any impact the resulting interpolation function (besides
360 # prediction speed).
361 return X, y

363 @_fit_context(prefer_skip_nested_validation=True)
364 def fit(self, X, y, sample_weight=None):
365 """Fit the model using X, y as training data.
366
367 Parameters
368 ----------
369 X : array-like of shape (n_samples,) or (n_samples, 1)
370 Training data.
371
372 .. versionchanged:: 0.24
373 Also accepts 2d array with 1 feature.
374
375 y : array-like of shape (n_samples,)
376 Training target.
377
378 sample_weight : array-like of shape (n_samples,), default=None
379 Weights. If set to None, all weights will be set to 1 (equal
380 weights).
381
382 Returns
383 -------
384 self : object
385 Returns an instance of self.
386
387 Notes
388 -----
389 X is stored for future use, as :meth:`transform` needs X to interpolate
390 new input data.
391 """
392 check_params = dict(accept_sparse=False, ensure_2d=False)
393 X = check_array(
394 X, input_name="X", dtype=[np.float64, np.float32], **check_params
395 )
396 y = check_array(y, input_name="y", dtype=X.dtype, **check_params)
397 check_consistent_length(X, y, sample_weight)
398
399 # Transform y by running the isotonic regression algorithm and
400 # transform X accordingly.
401 X, y = self._build_y(X, y, sample_weight)
402
403 # It is necessary to store the non-redundant part of the training set
404 # on the model to make it possible to support model persistence via
405 # the pickle module as the object built by scipy.interp1d is not
406 # picklable directly.
407 self.X_thresholds_, self.y_thresholds_ = X, y
408
409 # Build the interpolation function
410 self._build_f(X, y)
411 return self

413 def _transform(self, T):
414 """`_transform` is called by both `transform` and `predict` methods.
415
416 Since `transform` is wrapped to output arrays of specific types (e.g.
417 NumPy arrays, pandas DataFrame), we cannot make `predict` call `transform`
418 directly.
419
420 The above behaviour could be changed in the future, if we decide to output
421 other type of arrays when calling `predict`.
422 """
423 if hasattr(self, "X_thresholds_"):
424 dtype = self.X_thresholds_.dtype
425 else:
426 dtype = np.float64
427
428 T = check_array(T, dtype=dtype, ensure_2d=False)
429
430 self._check_input_data_shape(T)
431 T = T.reshape(-1) # use 1d view
432
433 if self.out_of_bounds == "clip":
434 T = np.clip(T, self.X_min_, self.X_max_)
435
436 res = self.f_(T)
437
438 # on scipy 0.17, interp1d up-casts to float64, so we cast back
439 res = res.astype(T.dtype)
440
441 return res
