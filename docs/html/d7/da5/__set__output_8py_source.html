113class PandasAdapter:
114 container_lib = "pandas"
115
116 def create_container(self, X_output, X_original, columns, inplace=True):
117 pd = check_library_installed("pandas")
118 columns = get_columns(columns)
119
120 if not inplace or not isinstance(X_output, pd.DataFrame):
121 # In all these cases, we need to create a new DataFrame
122
123 # Unfortunately, we cannot use `getattr(container, "index")`
124 # because `list` exposes an `index` attribute.
125 if isinstance(X_output, pd.DataFrame):
126 index = X_output.index
127 elif isinstance(X_original, (pd.DataFrame, pd.Series)):
128 index = X_original.index
129 else:
130 index = None
131
132 # We don't pass columns here because it would intend columns selection
133 # instead of renaming.
134 X_output = pd.DataFrame(X_output, index=index, copy=not inplace)
135
136 if columns is not None:
137 return self.rename_columns(X_output, columns)
138 return X_output
139
140 def is_supported_container(self, X):
141 pd = check_library_installed("pandas")
142 return isinstance(X, pd.DataFrame)
143
144 def rename_columns(self, X, columns):
145 # we cannot use `rename` since it takes a dictionary and at this stage we have
146 # potentially duplicate column names in `X`
147 X.columns = columns
148 return X
149
150 def hstack(self, Xs):
151 pd = check_library_installed("pandas")
152 return pd.concat(Xs, axis=1)

158 def create_container(self, X_output, X_original, columns, inplace=True):
159 pl = check_library_installed("polars")
160 columns = get_columns(columns)
161 columns = columns.tolist() if isinstance(columns, np.ndarray) else columns
162
163 if not inplace or not isinstance(X_output, pl.DataFrame):
164 # In all these cases, we need to create a new DataFrame
165 return pl.DataFrame(X_output, schema=columns, orient="row")

168 return self.rename_columns(X_output, columns)

230def _get_output_config(method, estimator=None):
231 """Get output config based on estimator and global configuration.
232
233 Parameters
234 ----------
235 method : {"transform"}
236 Estimator's method for which the output container is looked up.
237
238 estimator : estimator instance or None
239 Estimator to get the output configuration from. If `None`, check global
240 configuration is used.
241
242 Returns
243 -------
244 config : dict
245 Dictionary with keys:
246
247 - "dense": specifies the dense container for `method`. This can be
248 `"default"` or `"pandas"`.
249 """
250 est_sklearn_output_config = getattr(estimator, "_sklearn_output_config", {})
251 if method in est_sklearn_output_config:
252 dense_config = est_sklearn_output_config[method]
253 else:
254 dense_config = get_config()[f"{method}_output"]
255
256 supported_outputs = ADAPTERS_MANAGER.supported_outputs
257 if dense_config not in supported_outputs:
258 raise ValueError(
259 f"output config must be in {sorted(supported_outputs)}, got {dense_config}"
260 )
261
262 return {"dense": dense_config}

265def _wrap_data_with_container(method, data_to_wrap, original_input, estimator):
266 """Wrap output with container based on an estimator's or global config.
267
268 Parameters
269 ----------
270 method : {"transform"}
271 Estimator's method to get container output for.
272
273 data_to_wrap : {ndarray, dataframe}
274 Data to wrap with container.
275
276 original_input : {ndarray, dataframe}
277 Original input of function.
278
279 estimator : estimator instance
280 Estimator with to get the output configuration from.
281
282 Returns
283 -------
284 output : {ndarray, dataframe}
285 If the output config is "default" or the estimator is not configured
286 for wrapping return `data_to_wrap` unchanged.
287 If the output config is "pandas", return `data_to_wrap` as a pandas
288 DataFrame.
289 """
290 output_config = _get_output_config(method, estimator)
291
292 if output_config["dense"] == "default" or not _auto_wrap_is_configured(estimator):
293 return data_to_wrap
294
295 dense_config = output_config["dense"]
296 if issparse(data_to_wrap):
297 raise ValueError(
298 "The transformer outputs a scipy sparse matrix. "
299 "Try to set the transformer output to a dense array or disable "
300 f"{dense_config.capitalize()} output with set_output(transform='default')."
301 )
302
303 adapter = ADAPTERS_MANAGER.adapters[dense_config]
304 return adapter.create_container(
305 data_to_wrap,
306 original_input,
307 columns=estimator.get_feature_names_out,
308 )

311def _wrap_method_output(f, method):
312 """Wrapper used by `_SetOutputMixin` to automatically wrap methods."""
313
314 @wraps(f)
315 def wrapped(self, X, *args, **kwargs):
316 data_to_wrap = f(self, X, *args, **kwargs)
317 if isinstance(data_to_wrap, tuple):
318 # only wrap the first output for cross decomposition
319 return_tuple = (
320 _wrap_data_with_container(method, data_to_wrap[0], X, self),
321 *data_to_wrap[1:],
322 )
323 # Support for namedtuples `_make` is a documented API for namedtuples:
324 # https://docs.python.org/3/library/collections.html#collections.somenamedtuple._make
325 if hasattr(type(data_to_wrap), "_make"):
326 return type(data_to_wrap)._make(return_tuple)
327 return return_tuple
328
329 return _wrap_data_with_container(method, data_to_wrap, X, self)
330
331 return wrapped

347class _SetOutputMixin:
348 """Mixin that dynamically wraps methods to return container based on config.
349
350 Currently `_SetOutputMixin` wraps `transform` and `fit_transform` and configures
351 it based on `set_output` of the global configuration.
352
353 `set_output` is only defined if `get_feature_names_out` is defined and
354 `auto_wrap_output_keys` is the default value.
355 """
356
357 def __init_subclass__(cls, auto_wrap_output_keys=("transform",), **kwargs):
358 super().__init_subclass__(**kwargs)
359
360 # Dynamically wraps `transform` and `fit_transform` and configure it's
361 # output based on `set_output`.
362 if not (
363 isinstance(auto_wrap_output_keys, tuple) or auto_wrap_output_keys is None
364 ):
365 raise ValueError("auto_wrap_output_keys must be None or a tuple of keys.")
366
367 if auto_wrap_output_keys is None:
368 cls._sklearn_auto_wrap_output_keys = set()
369 return
370
371 # Mapping from method to key in configurations
372 method_to_key = {
373 "transform": "transform",
374 "fit_transform": "transform",
375 }
376 cls._sklearn_auto_wrap_output_keys = set()
377
378 for method, key in method_to_key.items():
379 if not hasattr(cls, method) or key not in auto_wrap_output_keys:
380 continue
381 cls._sklearn_auto_wrap_output_keys.add(key)
382
383 # Only wrap methods defined by cls itself
384 if method not in cls.__dict__:
385 continue
386 wrapped_method = _wrap_method_output(getattr(cls, method), key)
387 setattr(cls, method, wrapped_method)

389 @available_if(_auto_wrap_is_configured)
390 def set_output(self, *, transform=None):
391 """Set output container.
392
393 See :ref:`sphx_glr_auto_examples_miscellaneous_plot_set_output.py`
394 for an example on how to use the API.
395
396 Parameters
397 ----------
398 transform : {"default", "pandas", "polars"}, default=None
399 Configure output of `transform` and `fit_transform`.
400
401 - `"default"`: Default output format of a transformer
402 - `"pandas"`: DataFrame output
403 - `"polars"`: Polars output
404 - `None`: Transform configuration is unchanged
405
406 .. versionadded:: 1.4
407 `"polars"` option was added.
408
409 Returns
410 -------
411 self : estimator instance
412 Estimator instance.
413 """
414 if transform is None:
415 return self
416
417 if not hasattr(self, "_sklearn_output_config"):
418 self._sklearn_output_config = {}
419
420 self._sklearn_output_config["transform"] = transform
421 return self

424def _safe_set_output(estimator, *, transform=None):
425 """Safely call estimator.set_output and error if it not available.
426
427 This is used by meta-estimators to set the output for child estimators.
428
429 Parameters
430 ----------
431 estimator : estimator instance
432 Estimator instance.
433
434 transform : {"default", "pandas", "polars"}, default=None
435 Configure output of the following estimator's methods:
436
437 - `"transform"`
438 - `"fit_transform"`
439
440 If `None`, this operation is a no-op.
441
442 Returns
443 -------
444 estimator : estimator instance
445 Estimator instance.
446 """
447 set_output_for_transform = hasattr(estimator, "transform") or (
448 hasattr(estimator, "fit_transform") and transform is not None
449 )
450 if not set_output_for_transform:
451 # If estimator can not transform, then `set_output` does not need to be
452 # called.
453 return
454
455 if not hasattr(estimator, "set_output"):
456 raise ValueError(
457 f"Unable to configure output for {estimator} because `set_output` "
458 "is not available."
459 )
460 return estimator.set_output(transform=transform)
