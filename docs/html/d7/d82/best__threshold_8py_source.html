33# --- ENSEMBLE ASSESSMENT ---
34def ensemble_assessment(code1, code2, model_name, thr, n=3):
35 predictions = []
36 for _ in range(n):
37 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name, thr)
38 predictions.append((results, predicted_type, predicted_sim))
39 # Majority voting on predicted_sim
40 sim_votes = [pred[2] for pred in predictions]
41 final_sim = max(set(sim_votes), key=sim_votes.count)
42 # Most frequent predicted_type
43 type_votes = [pred[1] for pred in predictions]
44 final_type = max(set(type_votes), key=type_votes.count)
45 # Average results for reporting
46 avg_results = {t: sum(pred[0][t] for pred in predictions) / n for t in ["Type-1", "Type-2", "Type-3", "Type-4"]}
47 return avg_results, final_type, final_sim

49# --- THRESHOLD-BASED SIMILARITY DECISION ---
50def determine_similarity(results, min_threshold=0.1):
51 # Sort by score, then by type severity
52 type_priority = {"Type-4": 4, "Type-3": 3, "Type-2": 2, "Type-1": 1}
53 sorted_types = sorted(results.items(), key=lambda x: (x[1], type_priority[x[0]]), reverse=True)
54 max_type, max_score = sorted_types[0]
55 if max_score >= min_threshold:
56 return max_type, 1
57 else:
58 return "Non-clone", 0
