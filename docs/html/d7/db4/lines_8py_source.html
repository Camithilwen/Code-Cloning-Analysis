33def _get_dash_pattern(style):
34 """Convert linestyle to dash pattern."""
35 # go from short hand -> full strings
36 if isinstance(style, str):
37 style = ls_mapper.get(style, style)
38 # un-dashed styles
39 if style in ['solid', 'None']:
40 offset = 0
41 dashes = None
42 # dashed styles
43 elif style in ['dashed', 'dashdot', 'dotted']:
44 offset = 0
45 dashes = tuple(mpl.rcParams[f'lines.{style}_pattern'])
46 #
47 elif isinstance(style, tuple):
48 offset, dashes = style
49 if offset is None:
50 raise ValueError(f'Unrecognized linestyle: {style!r}')
51 else:
52 raise ValueError(f'Unrecognized linestyle: {style!r}')
53
54 # normalize offset to be positive and shorter than the dash cycle
55 if dashes is not None:
56 dsum = sum(dashes)
57 if dsum:
58 offset %= dsum
59
60 return offset, dashes

98def segment_hits(cx, cy, x, y, radius):
99 """
100 Return the indices of the segments in the polyline with coordinates (*cx*,
101 *cy*) that are within a distance *radius* of the point (*x*, *y*).
102 """
103 # Process single points specially
104 if len(x) <= 1:
105 res, = np.nonzero((cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2)
106 return res
107
108 # We need to lop the last element off a lot.
109 xr, yr = x[:-1], y[:-1]
110
111 # Only look at line segments whose nearest point to C on the line
112 # lies within the segment.
113 dx, dy = x[1:] - xr, y[1:] - yr
114 Lnorm_sq = dx ** 2 + dy ** 2 # Possibly want to eliminate Lnorm==0
115 u = ((cx - xr) * dx + (cy - yr) * dy) / Lnorm_sq
116 candidates = (u >= 0) & (u <= 1)
117
118 # Note that there is a little area near one side of each point
119 # which will be near neither segment, and another which will
120 # be near both, depending on the angle of the lines. The
121 # following radius test eliminates these ambiguities.
122 point_hits = (cx - x) ** 2 + (cy - y) ** 2 <= radius ** 2
123 candidates = candidates & ~(point_hits[:-1] | point_hits[1:])
124
125 # For those candidates which remain, determine how far they lie away
126 # from the line.
127 px, py = xr + u * dx, yr + u * dy
128 line_hits = (cx - px) ** 2 + (cy - py) ** 2 <= radius ** 2
129 line_hits = line_hits & candidates
130 points, = point_hits.ravel().nonzero()
131 lines, = line_hits.ravel().nonzero()
132 return np.concatenate((points, lines))

135def _mark_every_path(markevery, tpath, affine, ax):
136 """
137 Helper function that sorts out how to deal the input
138 `markevery` and returns the points where markers should be drawn.
139
140 Takes in the `markevery` value and the line path and returns the
141 sub-sampled path.
142 """
143 # pull out the two bits of data we want from the path
144 codes, verts = tpath.codes, tpath.vertices
145
146 def _slice_or_none(in_v, slc):
147 """Helper function to cope with `codes` being an ndarray or `None`."""
148 if in_v is None:
149 return None
150 return in_v[slc]
151
152 # if just an int, assume starting at 0 and make a tuple
153 if isinstance(markevery, Integral):
154 markevery = (0, markevery)
155 # if just a float, assume starting at 0.0 and make a tuple
156 elif isinstance(markevery, Real):
157 markevery = (0.0, markevery)
158
159 if isinstance(markevery, tuple):
160 if len(markevery) != 2:
161 raise ValueError('`markevery` is a tuple but its len is not 2; '
162 f'markevery={markevery}')
163 start, step = markevery
164 # if step is an int, old behavior
165 if isinstance(step, Integral):
166 # tuple of 2 int is for backwards compatibility,
167 if not isinstance(start, Integral):
168 raise ValueError(
169 '`markevery` is a tuple with len 2 and second element is '
170 'an int, but the first element is not an int; '
171 f'markevery={markevery}')
172 # just return, we are done here
173
174 return Path(verts[slice(start, None, step)],
175 _slice_or_none(codes, slice(start, None, step)))
176
177 elif isinstance(step, Real):
178 if not isinstance(start, Real):
179 raise ValueError(
180 '`markevery` is a tuple with len 2 and second element is '
181 'a float, but the first element is not a float or an int; '
182 f'markevery={markevery}')
183 if ax is None:
184 raise ValueError(
185 "markevery is specified relative to the Axes size, but "
186 "the line does not have a Axes as parent")
187
188 # calc cumulative distance along path (in display coords):
189 fin = np.isfinite(verts).all(axis=1)
190 fverts = verts[fin]
191 disp_coords = affine.transform(fverts)
192
193 delta = np.empty((len(disp_coords), 2))
194 delta[0, :] = 0
195 delta[1:, :] = disp_coords[1:, :] - disp_coords[:-1, :]
196 delta = np.hypot(*delta.T).cumsum()
197 # calc distance between markers along path based on the Axes
198 # bounding box diagonal being a distance of unity:
199 (x0, y0), (x1, y1) = ax.transAxes.transform([[0, 0], [1, 1]])
200 scale = np.hypot(x1 - x0, y1 - y0)
201 marker_delta = np.arange(start * scale, delta[-1], step * scale)
202 # find closest actual data point that is closest to
203 # the theoretical distance along the path:
204 inds = np.abs(delta[np.newaxis, :] - marker_delta[:, np.newaxis])
205 inds = inds.argmin(axis=1)
206 inds = np.unique(inds)
207 # return, we are done here
208 return Path(fverts[inds], _slice_or_none(codes, inds))
209 else:
210 raise ValueError(
211 f"markevery={markevery!r} is a tuple with len 2, but its "
212 f"second element is not an int or a float")
213
214 elif isinstance(markevery, slice):
215 # mazol tov, it's already a slice, just return
216 return Path(verts[markevery], _slice_or_none(codes, markevery))
217
218 elif np.iterable(markevery):
219 # fancy indexing
220 try:
221 return Path(verts[markevery], _slice_or_none(codes, markevery))
222 except (ValueError, IndexError) as err:
223 raise ValueError(
224 f"markevery={markevery!r} is iterable but not a valid numpy "
225 f"fancy index") from err
226 else:
227 raise ValueError(f"markevery={markevery!r} is not a recognized value")

442 def contains(self, mouseevent):
443 """
444 Test whether *mouseevent* occurred on the line.
445
446 An event is deemed to have occurred "on" the line if it is less
447 than ``self.pickradius`` (default: 5 points) away from it. Use
448 `~.Line2D.get_pickradius` or `~.Line2D.set_pickradius` to get or set
449 the pick radius.
450
451 Parameters
452 ----------
453 mouseevent : `~matplotlib.backend_bases.MouseEvent`
454
455 Returns
456 -------
457 contains : bool
458 Whether any values are within the radius.
459 details : dict
460 A dictionary ``{'ind': pointlist}``, where *pointlist* is a
461 list of points of the line that are within the pickradius around
462 the event position.
463
464 TODO: sort returned indices by distance
465 """
466 if self._different_canvas(mouseevent):
467 return False, {}
468
469 # Make sure we have data to plot
470 if self._invalidy or self._invalidx:
471 self.recache()
472 if len(self._xy) == 0:
473 return False, {}
474
475 # Convert points to pixels
476 transformed_path = self._get_transformed_path()
477 path, affine = transformed_path.get_transformed_path_and_affine()
478 path = affine.transform_path(path)
479 xy = path.vertices
480 xt = xy[:, 0]
481 yt = xy[:, 1]
482
483 # Convert pick radius from points to pixels
484 fig = self.get_figure(root=True)
485 if fig is None:
486 _log.warning('no figure set when check if mouse is on line')
487 pixels = self._pickradius
488 else:
489 pixels = fig.dpi / 72. * self._pickradius
490
491 # The math involved in checking for containment (here and inside of
492 # segment_hits) assumes that it is OK to overflow, so temporarily set
493 # the error flags accordingly.
494 with np.errstate(all='ignore'):
495 # Check for collision
496 if self._linestyle in ['None', None]:
497 # If no line, return the nearby point(s)
498 ind, = np.nonzero(
499 (xt - mouseevent.x) ** 2 + (yt - mouseevent.y) ** 2
500 <= pixels ** 2)
501 else:
502 # If line, return the nearby segment(s)
503 ind = segment_hits(mouseevent.x, mouseevent.y, xt, yt, pixels)
504 if self._drawstyle.startswith("steps"):
505 ind //= 2
506
507 ind += self.ind_offset
508
509 # Return the point(s) within radius
510 return len(ind) > 0, dict(ind=ind)

568

570
571 Parameters
572 ----------
573 every : None or int or (int, int) or slice or list[int] or float or \
574(float, float) or list[bool]
575 Which markers to plot.
576
577 - ``every=None``: every point will be plotted.
578 - ``every=N``: every N-th marker will be plotted starting with
579 marker 0.
580 - ``every=(start, N)``: every N-th marker, starting at index
581 *start*, will be plotted.
582 - ``every=slice(start, end, N)``: every N-th marker, starting at

591 approximately equal visual distances along the line; the distance
592 along the line between markers is determined by multiplying the
593 display-coordinate distance of the Axes bounding-box diagonal

595 - ``every=(0.5, 0.1)`` (i.e. a length-2 tuple of float): similar

605 Setting *markevery* will still only draw markers at actual data points.

610

612 be from the first data point which may be different from the first
613 the visible data point if the plot is zoomed in.
614
615 If zooming in on a plot when using float arguments then the actual
616 data points that have markers will change because the distance between
617 markers is always determined from the display-coordinates
618 axes-bounding-box-diagonal regardless of the actual axes data limits.
619

622 self.stale = True

779 from matplotlib.patheffects import PathEffectRenderer
780 renderer = PathEffectRenderer(self.get_path_effects(), renderer)

783 if self._lineStyles[self._linestyle] != '_draw_nothing':
784 tpath, affine = (self._get_transformed_path()
785 .get_transformed_path_and_affine())
786 if len(tpath.vertices):
787 gc = renderer.new_gc()
788 self._set_gc_clip(gc)
789 gc.set_url(self.get_url())
790
791 gc.set_antialiased(self._antialiased)
792 gc.set_linewidth(self._linewidth)
793
794 if self.is_dashed():
795 cap = self._dashcapstyle
796 join = self._dashjoinstyle
797 else:
798 cap = self._solidcapstyle
799 join = self._solidjoinstyle
800 gc.set_joinstyle(join)
801 gc.set_capstyle(cap)
802 gc.set_snap(self.get_snap())
803 if self.get_sketch_params() is not None:
804 gc.set_sketch_params(*self.get_sketch_params())
805
806 # We first draw a path within the gaps if needed.
807 if self.is_dashed() and self._gapcolor is not None:
808 lc_rgba = mcolors.to_rgba(self._gapcolor, self._alpha)
809 gc.set_foreground(lc_rgba, isRGBA=True)
810
811 offset_gaps, gaps = _get_inverse_dash_pattern(
812 *self._dash_pattern)
813
814 gc.set_dashes(offset_gaps, gaps)
815 renderer.draw_path(gc, tpath, affine.frozen())
816
817 lc_rgba = mcolors.to_rgba(self._color, self._alpha)
818 gc.set_foreground(lc_rgba, isRGBA=True)
819
820 gc.set_dashes(*self._dash_pattern)
821 renderer.draw_path(gc, tpath, affine.frozen())
822 gc.restore()
823
824 if self._marker and self._markersize > 0:
825 gc = renderer.new_gc()
826 self._set_gc_clip(gc)
827 gc.set_url(self.get_url())
828 gc.set_linewidth(self._markeredgewidth)
829 gc.set_antialiased(self._antialiased)
830
831 ec_rgba = mcolors.to_rgba(
832 self.get_markeredgecolor(), self._alpha)
833 fc_rgba = mcolors.to_rgba(
834 self._get_markerfacecolor(), self._alpha)
835 fcalt_rgba = mcolors.to_rgba(
836 self._get_markerfacecolor(alt=True), self._alpha)
837 # If the edgecolor is "auto", it is set according to the *line*
838 # color but inherits the alpha value of the *face* color, if any.
839 if (cbook._str_equal(self._markeredgecolor, "auto")
840 and not cbook._str_lower_equal(
841 self.get_markerfacecolor(), "none")):
842 ec_rgba = ec_rgba[:3] + (fc_rgba[3],)
843 gc.set_foreground(ec_rgba, isRGBA=True)
844 if self.get_sketch_params() is not None:
845 scale, length, randomness = self.get_sketch_params()
846 gc.set_sketch_params(scale/2, length/2, 2*randomness)
847
848 marker = self._marker
849
850 # Markers *must* be drawn ignoring the drawstyle (but don't pay the
851 # recaching if drawstyle is already "default").
852 if self.get_drawstyle() != "default":
853 with cbook._setattr_cm(
854 self, _drawstyle="default", _transformed_path=None):
855 self.recache()
856 self._transform_path(subslice)
857 tpath, affine = (self._get_transformed_path()
858 .get_transformed_points_and_affine())
859 else:
860 tpath, affine = (self._get_transformed_path()
861 .get_transformed_points_and_affine())
862
863 if len(tpath.vertices):
864 # subsample the markers if markevery is not None
865 markevery = self.get_markevery()
866 if markevery is not None:
867 subsampled = _mark_every_path(
868 markevery, tpath, affine, self.axes)
869 else:
870 subsampled = tpath
871
872 snap = marker.get_snap_threshold()
873 if isinstance(snap, Real):
874 snap = renderer.points_to_pixels(self._markersize) >= snap
875 gc.set_snap(snap)
876 gc.set_joinstyle(marker.get_joinstyle())
877 gc.set_capstyle(marker.get_capstyle())
878 marker_path = marker.get_path()
879 marker_trans = marker.get_transform()
880 w = renderer.points_to_pixels(self._markersize)
881
882 if cbook._str_equal(marker.get_marker(), ","):
883 gc.set_linewidth(0)
884 else:
885 # Don't scale for pixels, and don't stroke them
886 marker_trans = marker_trans.scale(w)
887 renderer.draw_markers(gc, marker_path, marker_trans,
888 subsampled, affine.frozen(),
889 fc_rgba)
890
891 alt_marker_path = marker.get_alt_path()
892 if alt_marker_path:
893 alt_marker_trans = marker.get_alt_transform()
894 alt_marker_trans = alt_marker_trans.scale(w)
895 renderer.draw_markers(
896 gc, alt_marker_path, alt_marker_trans, subsampled,
897 affine.frozen(), fcalt_rgba)
898
899 gc.restore()
900
901 renderer.close_group('line2d')
902 self.stale = False
