208def test_lasso_dual_gap():
209 """
210 Check that Lasso.dual_gap_ matches its objective formulation, with the
211 datafit normalized by n_samples
212 """
213 X, y, _, _ = build_dataset(n_samples=10, n_features=30)
214 n_samples = len(y)
215 alpha = 0.01 * np.max(np.abs(X.T @ y)) / n_samples
216 clf = Lasso(alpha=alpha, fit_intercept=False).fit(X, y)
217 w = clf.coef_
218 R = y - X @ w
219 primal = 0.5 * np.mean(R**2) + clf.alpha * np.sum(np.abs(w))
220 # dual pt: R / n_samples, dual constraint: norm(X.T @ theta, inf) <= alpha
221 R /= np.max(np.abs(X.T @ R) / (n_samples * alpha))
222 dual = 0.5 * (np.mean(y**2) - np.mean((y - R) ** 2))
223 assert_allclose(clf.dual_gap_, primal - dual)

331def _scale_alpha_inplace(estimator, n_samples):
332 """Rescale the parameter alpha from when the estimator is evoked with
333 normalize set to True as if it were evoked in a Pipeline with normalize set
334 to False and with a StandardScaler.
335 """
336 if ("alpha" not in estimator.get_params()) and (
337 "alphas" not in estimator.get_params()
338 ):
339 return
340
341 if isinstance(estimator, (RidgeCV, RidgeClassifierCV)):
342 # alphas is not validated at this point and can be a list.
343 # We convert it to a np.ndarray to make sure broadcasting
344 # is used.
345 alphas = np.asarray(estimator.alphas) * n_samples
346 return estimator.set_params(alphas=alphas)
347 if isinstance(estimator, (Lasso, LassoLars, MultiTaskLasso)):
348 alpha = estimator.alpha * np.sqrt(n_samples)
349 if isinstance(estimator, (Ridge, RidgeClassifier)):
350 alpha = estimator.alpha * n_samples
351 if isinstance(estimator, (ElasticNet, MultiTaskElasticNet)):
352 if estimator.l1_ratio == 1:
353 alpha = estimator.alpha * np.sqrt(n_samples)
354 elif estimator.l1_ratio == 0:
355 alpha = estimator.alpha * n_samples
356 else:
357 # To avoid silent errors in case of refactoring
358 raise NotImplementedError
359
360 estimator.set_params(alpha=alpha)
