45 This boolean attribute indicates whether the data (`X`),

252def default_tags(estimator) -> Tags:
253 """Get the default tags for an estimator.
254
255 This ignores any ``__sklearn_tags__`` method that the estimator may have.
256
257 If the estimator is a classifier or a regressor, ``target_tags.required``
258 will be set to ``True``, otherwise it will be set to ``False``.
259
260 ``transformer_tags`` will be set to :class:`~.sklearn.utils. TransformerTags` if the
261 estimator has a ``transform`` or ``fit_transform`` method, otherwise it will be set
262 to ``None``.
263
264 ``classifier_tags`` will be set to :class:`~.sklearn.utils.ClassifierTags` if the
265 estimator is a classifier, otherwise it will be set to ``None``.
266 a classifier, otherwise it will be set to ``None``.
267
268 ``regressor_tags`` will be set to :class:`~.sklearn.utils.RegressorTags` if the
269 estimator is a regressor, otherwise it will be set to ``None``.
270
271 Parameters
272 ----------
273 estimator : estimator object
274 The estimator for which to get the default tags.
275
276 Returns
277 -------
278 tags : Tags
279 The default tags for the estimator.
280 """
281 est_is_classifier = getattr(estimator, "_estimator_type", None) == "classifier"
282 est_is_regressor = getattr(estimator, "_estimator_type", None) == "regressor"
283 target_required = est_is_classifier or est_is_regressor
284
285 return Tags(
286 estimator_type=getattr(estimator, "_estimator_type", None),
287 target_tags=TargetTags(required=target_required),
288 transformer_tags=(
289 TransformerTags()
290 if hasattr(estimator, "transform") or hasattr(estimator, "fit_transform")
291 else None
292 ),
293 classifier_tags=ClassifierTags() if est_is_classifier else None,
294 regressor_tags=RegressorTags() if est_is_regressor else None,
295 )

298def get_tags(estimator) -> Tags:
299 """Get estimator tags.
300
301 :class:`~sklearn.BaseEstimator` provides the estimator tags machinery.
302 However, if an estimator does not inherit from this base class, we should
303 fall-back to the default tags.
304
305 For scikit-learn built-in estimators, we should still rely on
306 `self.__sklearn_tags__()`. `get_tags(est)` should be used when we
307 are not sure where `est` comes from: typically
308 `get_tags(self.estimator)` where `self` is a meta-estimator, or in
309 the common checks.
310
311 .. versionadded:: 1.6
312
313 Parameters
314 ----------
315 estimator : estimator object
316 The estimator from which to get the tag.
317
318 Returns
319 -------
320 tags : :class:`~.sklearn.utils.Tags`
321 The estimator tags.
322 """
323
324 try:
325 tags = estimator.__sklearn_tags__()
326 except AttributeError as exc:
327 # TODO(1.8): turn the warning into an error
328 if "object has no attribute '__sklearn_tags__'" in str(exc):
329 # Fall back to the default tags if the estimator does not
330 # implement __sklearn_tags__.
331 # In particular, workaround the regression reported in
332 # https://github.com/scikit-learn/scikit-learn/issues/30479
333 # `__sklearn_tags__` is implemented by calling
334 # `super().__sklearn_tags__()` but there is no `__sklearn_tags__`
335 # method in the base class. Typically happens when only inheriting
336 # from Mixins.
337
338 warnings.warn(
339 f"The following error was raised: {exc}. It seems that "
340 "there are no classes that implement `__sklearn_tags__` "
341 "in the MRO and/or all classes in the MRO call "
342 "`super().__sklearn_tags__()`. Make sure to inherit from "
343 "`BaseEstimator` which implements `__sklearn_tags__` (or "
344 "alternatively define `__sklearn_tags__` but we don't recommend "
345 "this approach). Note that `BaseEstimator` needs to be on the "
346 "right side of other Mixins in the inheritance order. The "
347 "default are now used instead since retrieving tags failed. "
348 "This warning will be replaced by an error in 1.8.",
349 category=DeprecationWarning,
350 )
351 tags = default_tags(estimator)
352 else:
353 raise
354
355 return tags
