298def is_terminal() -> bool:
299 """
300 Detect if Python is running in a terminal.
301
302 Returns True if Python is running in a terminal or False if not.
303 """
304 try:
305 # error: Name 'get_ipython' is not defined
306 ip = get_ipython() # type: ignore[name-defined]
307 except NameError: # assume standard Python interpreter in a terminal
308 return True
309 else:
310 if hasattr(ip, "kernel"): # IPython as a Jupyter kernel
311 return False
312 else: # IPython in a terminal
313 return True
