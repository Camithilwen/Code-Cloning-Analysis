27def wrap_constraints(g):
28 cons = []
29 if g is not None:
30 if (type(g) is not tuple) and (type(g) is not list):
31 g = (g,)
32 else:
33 pass
34 for g in g:
35 cons.append({'type': 'ineq',
36 'fun': g})
37 cons = tuple(cons)
38 else:
39 cons = None
40 return cons
