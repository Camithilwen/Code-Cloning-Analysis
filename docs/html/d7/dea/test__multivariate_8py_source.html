275def _random_covariance(dim, evals, rng, singular=False):
276 # Generates random covariance matrix with dimensionality `dim` and
277 # eigenvalues `evals` using provided Generator `rng`. Randomly sets
278 # some evals to zero if `singular` is True.
279 A = rng.random((dim, dim))
280 A = A @ A.T
281 _, v = np.linalg.eigh(A)
282 if singular:
283 zero_eigs = rng.normal(size=dim) > 0
284 evals[zero_eigs] = 0
285 cov = v @ np.diag(evals) @ v.T
286 return cov
