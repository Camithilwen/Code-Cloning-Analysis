41def estimate_bandwidth(X, *, quantile=0.3, n_samples=None, random_state=0, n_jobs=None):
42 """Estimate the bandwidth to use with the mean-shift algorithm.
43
44 This function takes time at least quadratic in `n_samples`. For large
45 datasets, it is wise to subsample by setting `n_samples`. Alternatively,
46 the parameter `bandwidth` can be set to a small value without estimating
47 it.
48
49 Parameters
50 ----------
51 X : array-like of shape (n_samples, n_features)
52 Input points.
53
54 quantile : float, default=0.3
55 Should be between [0, 1]
56 0.5 means that the median of all pairwise distances is used.
57
58 n_samples : int, default=None
59 The number of samples to use. If not given, all samples are used.
60
61 random_state : int, RandomState instance, default=None
62 The generator used to randomly select the samples from input points
63 for bandwidth estimation. Use an int to make the randomness
64 deterministic.
65 See :term:`Glossary <random_state>`.
66
67 n_jobs : int, default=None
68 The number of parallel jobs to run for neighbors search.
69 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
70 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
71 for more details.
72
73 Returns
74 -------
75 bandwidth : float
76 The bandwidth parameter.
77
78 Examples
79 --------
80 >>> import numpy as np
81 >>> from sklearn.cluster import estimate_bandwidth
82 >>> X = np.array([[1, 1], [2, 1], [1, 0],
83 ... [4, 7], [3, 5], [3, 6]])
84 >>> estimate_bandwidth(X, quantile=0.5)
85 np.float64(1.61)
86 """
87 X = check_array(X)
88
89 random_state = check_random_state(random_state)
90 if n_samples is not None:
91 idx = random_state.permutation(X.shape[0])[:n_samples]
92 X = X[idx]
93 n_neighbors = int(X.shape[0] * quantile)
94 if n_neighbors < 1: # cannot fit NearestNeighbors with n_neighbors = 0
95 n_neighbors = 1
96 nbrs = NearestNeighbors(n_neighbors=n_neighbors, n_jobs=n_jobs)
97 nbrs.fit(X)
98
99 bandwidth = 0.0
100 for batch in gen_batches(len(X), 500):
101 d, _ = nbrs.kneighbors(X[batch, :], return_distance=True)
102 bandwidth += np.max(d, axis=1).sum()
103
104 return bandwidth / X.shape[0]

107# separate function for each seed's iterative loop
108def _mean_shift_single_seed(my_mean, X, nbrs, max_iter):
109 # For each seed, climb gradient until convergence or max_iter
110 bandwidth = nbrs.get_params()["radius"]
111 stop_thresh = 1e-3 * bandwidth # when mean has converged
112 completed_iterations = 0
113 while True:
114 # Find mean of points within bandwidth
115 i_nbrs = nbrs.radius_neighbors([my_mean], bandwidth, return_distance=False)[0]
116 points_within = X[i_nbrs]
117 if len(points_within) == 0:
118 break # Depending on seeding strategy this condition may occur
119 my_old_mean = my_mean # save the old mean
120 my_mean = np.mean(points_within, axis=0)
121 # If converged or at max_iter, adds the cluster
122 if (
123 np.linalg.norm(my_mean - my_old_mean) <= stop_thresh
124 or completed_iterations == max_iter
125 ):
126 break
127 completed_iterations += 1
128 return tuple(my_mean), len(points_within), completed_iterations

146 """Perform mean shift clustering of data using a flat kernel.
147
148 Read more in the :ref:`User Guide <mean_shift>`.
149
150 Parameters
151 ----------
152
153 X : array-like of shape (n_samples, n_features)
154 Input data.
155
156 bandwidth : float, default=None
157 Kernel bandwidth. If not None, must be in the range [0, +inf).
158
159 If None, the bandwidth is determined using a heuristic based on
160 the median of all pairwise distances. This will take quadratic time in
161 the number of samples. The sklearn.cluster.estimate_bandwidth function
162 can be used to do this more efficiently.
163
164 seeds : array-like of shape (n_seeds, n_features) or None
165 Point used as initial kernel locations. If None and bin_seeding=False,
166 each data point is used as a seed. If None and bin_seeding=True,
167 see bin_seeding.
168
169 bin_seeding : bool, default=False
170 If true, initial kernel locations are not locations of all
171 points, but rather the location of the discretized version of
172 points, where points are binned onto a grid whose coarseness
173 corresponds to the bandwidth. Setting this option to True will speed
174 up the algorithm because fewer seeds will be initialized.
175 Ignored if seeds argument is not None.
176
177 min_bin_freq : int, default=1
178 To speed up the algorithm, accept only those bins with at least
179 min_bin_freq points as seeds.
180
181 cluster_all : bool, default=True
182 If true, then all points are clustered, even those orphans that are
183 not within any kernel. Orphans are assigned to the nearest kernel.
184 If false, then orphans are given cluster label -1.
185
186 max_iter : int, default=300
187 Maximum number of iterations, per seed point before the clustering
188 operation terminates (for that seed point), if has not converged yet.
189
190 n_jobs : int, default=None
191 The number of jobs to use for the computation. The following tasks benefit
192 from the parallelization:
193
194 - The search of nearest neighbors for bandwidth estimation and label
195 assignments. See the details in the docstring of the
196 ``NearestNeighbors`` class.
197 - Hill-climbing optimization for all seeds.
198
199 See :term:`Glossary <n_jobs>` for more details.
200
201 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
202 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
203 for more details.
204
205 .. versionadded:: 0.17
206 Parallel Execution using *n_jobs*.
207
208 Returns
209 -------
210
211 cluster_centers : ndarray of shape (n_clusters, n_features)
212 Coordinates of cluster centers.
213
214 labels : ndarray of shape (n_samples,)
215 Cluster labels for each point.
216
217 Notes
218 -----
219 For a usage example, see
220 :ref:`sphx_glr_auto_examples_cluster_plot_mean_shift.py`.
221
222 Examples
223 --------
224 >>> import numpy as np
225 >>> from sklearn.cluster import mean_shift
226 >>> X = np.array([[1, 1], [2, 1], [1, 0],
227 ... [4, 7], [3, 5], [3, 6]])
228 >>> cluster_centers, labels = mean_shift(X, bandwidth=2)
229 >>> cluster_centers
230 array([[3.33, 6. ],
231 [1.33, 0.66]])
232 >>> labels
233 array([1, 1, 1, 0, 0, 0])
234 """
235 model = MeanShift(
236 bandwidth=bandwidth,
237 seeds=seeds,
238 min_bin_freq=min_bin_freq,
239 bin_seeding=bin_seeding,
240 cluster_all=cluster_all,
241 n_jobs=n_jobs,
242 max_iter=max_iter,
243 ).fit(X)
244 return model.cluster_centers_, model.labels_

247def get_bin_seeds(X, bin_size, min_bin_freq=1):
248 """Find seeds for mean_shift.
249
250 Finds seeds by first binning data onto a grid whose lines are
251 spaced bin_size apart, and then choosing those bins with at least
252 min_bin_freq points.
253
254 Parameters
255 ----------
256
257 X : array-like of shape (n_samples, n_features)
258 Input points, the same points that will be used in mean_shift.
259
260 bin_size : float
261 Controls the coarseness of the binning. Smaller values lead
262 to more seeding (which is computationally more expensive). If you're
263 not sure how to set this, set it to the value of the bandwidth used
264 in clustering.mean_shift.
265
266 min_bin_freq : int, default=1
267 Only bins with at least min_bin_freq will be selected as seeds.
268 Raising this value decreases the number of seeds found, which
269 makes mean_shift computationally cheaper.
270
271 Returns
272 -------
273 bin_seeds : array-like of shape (n_samples, n_features)
274 Points used as initial kernel positions in clustering.mean_shift.
275 """
276 if bin_size == 0:
277 return X
278
279 # Bin points
280 bin_sizes = defaultdict(int)
281 for point in X:
282 binned_point = np.round(point / bin_size)
283 bin_sizes[tuple(binned_point)] += 1
284
285 # Select only those bins as seeds which have enough members
286 bin_seeds = np.array(
287 [point for point, freq in bin_sizes.items() if freq >= min_bin_freq],
288 dtype=np.float32,
289 )
290 if len(bin_seeds) == len(X):
291 warnings.warn(
292 "Binning data failed with provided bin_size=%f, using data points as seeds."
293 % bin_size
294 )
295 return X
296 bin_seeds = bin_seeds * bin_size
297 return bin_seeds

300class MeanShift(ClusterMixin, BaseEstimator):
301 """Mean shift clustering using a flat kernel.
302
303 Mean shift clustering aims to discover "blobs" in a smooth density of
304 samples. It is a centroid-based algorithm, which works by updating
305 candidates for centroids to be the mean of the points within a given
306 region. These candidates are then filtered in a post-processing stage to
307 eliminate near-duplicates to form the final set of centroids.
308
309 Seeding is performed using a binning technique for scalability.
310
311 For an example of how to use MeanShift clustering, refer to:
312 :ref:`sphx_glr_auto_examples_cluster_plot_mean_shift.py`.
313
314 Read more in the :ref:`User Guide <mean_shift>`.
315
316 Parameters
317 ----------
318 bandwidth : float, default=None
319 Bandwidth used in the flat kernel.
320
321 If not given, the bandwidth is estimated using
322 sklearn.cluster.estimate_bandwidth; see the documentation for that
323 function for hints on scalability (see also the Notes, below).
324
325 seeds : array-like of shape (n_samples, n_features), default=None
326 Seeds used to initialize kernels. If not set,
327 the seeds are calculated by clustering.get_bin_seeds
328 with bandwidth as the grid size and default values for
329 other parameters.
330
331 bin_seeding : bool, default=False
332 If true, initial kernel locations are not locations of all
333 points, but rather the location of the discretized version of
334 points, where points are binned onto a grid whose coarseness
335 corresponds to the bandwidth. Setting this option to True will speed
336 up the algorithm because fewer seeds will be initialized.
337 The default value is False.
338 Ignored if seeds argument is not None.
339
340 min_bin_freq : int, default=1
341 To speed up the algorithm, accept only those bins with at least
342 min_bin_freq points as seeds.
343
344 cluster_all : bool, default=True
345 If true, then all points are clustered, even those orphans that are
346 not within any kernel. Orphans are assigned to the nearest kernel.
347 If false, then orphans are given cluster label -1.
348
349 n_jobs : int, default=None
350 The number of jobs to use for the computation. The following tasks benefit
351 from the parallelization:
352
353 - The search of nearest neighbors for bandwidth estimation and label
354 assignments. See the details in the docstring of the
355 ``NearestNeighbors`` class.
356 - Hill-climbing optimization for all seeds.
357
358 See :term:`Glossary <n_jobs>` for more details.
359
360 ``None`` means 1 unless in a :obj:`joblib.parallel_backend` context.
361 ``-1`` means using all processors. See :term:`Glossary <n_jobs>`
362 for more details.
363
364 max_iter : int, default=300
365 Maximum number of iterations, per seed point before the clustering
366 operation terminates (for that seed point), if has not converged yet.
367
368 .. versionadded:: 0.22
369
370 Attributes
371 ----------
372 cluster_centers_ : ndarray of shape (n_clusters, n_features)
373 Coordinates of cluster centers.
374
375 labels_ : ndarray of shape (n_samples,)
376 Labels of each point.
377
378 n_iter_ : int
379 Maximum number of iterations performed on each seed.
380
381 .. versionadded:: 0.22
382
383 n_features_in_ : int
384 Number of features seen during :term:`fit`.
385
386 .. versionadded:: 0.24
387
388 feature_names_in_ : ndarray of shape (`n_features_in_`,)
389 Names of features seen during :term:`fit`. Defined only when `X`
390 has feature names that are all strings.
391
392 .. versionadded:: 1.0
393
394 See Also
395 --------
396 KMeans : K-Means clustering.
397
398 Notes
399 -----
400
401 Scalability:
402
403 Because this implementation uses a flat kernel and
404 a Ball Tree to look up members of each kernel, the complexity will tend
405 towards O(T*n*log(n)) in lower dimensions, with n the number of samples
406 and T the number of points. In higher dimensions the complexity will
407 tend towards O(T*n^2).
408
409 Scalability can be boosted by using fewer seeds, for example by using
410 a higher value of min_bin_freq in the get_bin_seeds function.
411
412 Note that the estimate_bandwidth function is much less scalable than the
413 mean shift algorithm and will be the bottleneck if it is used.
414
415 References
416 ----------
417
418 Dorin Comaniciu and Peter Meer, "Mean Shift: A robust approach toward
419 feature space analysis". IEEE Transactions on Pattern Analysis and
420 Machine Intelligence. 2002. pp. 603-619.
421
422 Examples
423 --------
424 >>> from sklearn.cluster import MeanShift
425 >>> import numpy as np
426 >>> X = np.array([[1, 1], [2, 1], [1, 0],
427 ... [4, 7], [3, 5], [3, 6]])
428 >>> clustering = MeanShift(bandwidth=2).fit(X)
429 >>> clustering.labels_
430 array([1, 1, 1, 0, 0, 0])
431 >>> clustering.predict([[0, 0], [5, 5]])
432 array([1, 0])
433 >>> clustering
434 MeanShift(bandwidth=2)
435
436 For a comparison of Mean Shift clustering with other clustering algorithms, see
437 :ref:`sphx_glr_auto_examples_cluster_plot_cluster_comparison.py`

469 @_fit_context(prefer_skip_nested_validation=True)
470 def fit(self, X, y=None):
471 """Perform clustering.
472
473 Parameters
474 ----------
475 X : array-like of shape (n_samples, n_features)
476 Samples to cluster.
477
478 y : Ignored
479 Not used, present for API consistency by convention.
480
481 Returns
482 -------
483 self : object
484 Fitted instance.
485 """
486 X = validate_data(self, X)
487 bandwidth = self.bandwidth
488 if bandwidth is None:
489 bandwidth = estimate_bandwidth(X, n_jobs=self.n_jobs)
490
491 seeds = self.seeds
492 if seeds is None:
493 if self.bin_seeding:
494 seeds = get_bin_seeds(X, bandwidth, self.min_bin_freq)
495 else:
496 seeds = X
497 n_samples, n_features = X.shape
498 center_intensity_dict = {}
499
500 # We use n_jobs=1 because this will be used in nested calls under
501 # parallel calls to _mean_shift_single_seed so there is no need for
502 # for further parallelism.
503 nbrs = NearestNeighbors(radius=bandwidth, n_jobs=1).fit(X)
504
505 # execute iterations on all seeds in parallel
506 all_res = Parallel(n_jobs=self.n_jobs)(
507 delayed(_mean_shift_single_seed)(seed, X, nbrs, self.max_iter)
508 for seed in seeds
509 )
510 # copy results in a dictionary
511 for i in range(len(seeds)):
512 if all_res[i][1]: # i.e. len(points_within) > 0
513 center_intensity_dict[all_res[i][0]] = all_res[i][1]
514
515 self.n_iter_ = max([x[2] for x in all_res])
516
517 if not center_intensity_dict:
518 # nothing near seeds
519 raise ValueError(
520 "No point was within bandwidth=%f of any seed. Try a different seeding"
521 " strategy or increase the bandwidth."
522 % bandwidth
523 )
524
525 # POST PROCESSING: remove near duplicate points
526 # If the distance between two kernels is less than the bandwidth,
527 # then we have to remove one because it is a duplicate. Remove the
528 # one with fewer points.
529
530 sorted_by_intensity = sorted(
531 center_intensity_dict.items(),
532 key=lambda tup: (tup[1], tup[0]),
533 reverse=True,
534 )
535 sorted_centers = np.array([tup[0] for tup in sorted_by_intensity])
536 unique = np.ones(len(sorted_centers), dtype=bool)
537 nbrs = NearestNeighbors(radius=bandwidth, n_jobs=self.n_jobs).fit(
538 sorted_centers
539 )
540 for i, center in enumerate(sorted_centers):
541 if unique[i]:
542 neighbor_idxs = nbrs.radius_neighbors([center], return_distance=False)[
543 0
544 ]
545 unique[neighbor_idxs] = 0
546 unique[i] = 1 # leave the current point as unique
547 cluster_centers = sorted_centers[unique]
548
549 # ASSIGN LABELS: a point belongs to the cluster that it is closest to
550 nbrs = NearestNeighbors(n_neighbors=1, n_jobs=self.n_jobs).fit(cluster_centers)
551 labels = np.zeros(n_samples, dtype=int)
552 distances, idxs = nbrs.kneighbors(X)
553 if self.cluster_all:
554 labels = idxs.flatten()
555 else:
556 labels.fill(-1)
557 bool_selector = distances.flatten() <= bandwidth
558 labels[bool_selector] = idxs.flatten()[bool_selector]
559
560 self.cluster_centers_, self.labels_ = cluster_centers, labels
561 return self
