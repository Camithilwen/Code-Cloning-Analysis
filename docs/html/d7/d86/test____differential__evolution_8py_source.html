655 constraints=(nlc))

658 assert_almost_equal(cv, 0.1)
659
660 nlc2 = NonlinearConstraint(constr_f2, -np.inf, 1.8)
661 solver = DifferentialEvolutionSolver(rosen, [(0, 2), (0, 2)],
662 constraints=(nlc, nlc2))
663
