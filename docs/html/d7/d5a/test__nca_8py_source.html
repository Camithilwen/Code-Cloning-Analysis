36def test_simple_example():
37 """Test on a simple example.

40 next to each other. After transform the samples from the same class
41 should be next to each other.
42
43 """

54def test_toy_example_collapse_points():
55 """Test on a toy example of three points that should collapse
56
57 We build a simple example: two points from the same class and a point from
58 a different class in the middle of them. On this simple example, the new
59 (transformed) points should all collapse into one single point. Indeed, the
60 objective is 2/(1 + exp(d/2)), with d the euclidean distance between the
61 two samples from the same class. This is maximized for d=0 (because d>=0),
62 with an objective equal to 1 (loss=-1.).
63
64 """
65 rng = np.random.RandomState(42)
66 input_dim = 5
67 two_points = rng.randn(2, input_dim)
68 X = np.vstack([two_points, two_points.mean(axis=0)[np.newaxis, :]])
69 y = [0, 0, 1]

71 class LossStorer:
72 def __init__(self, X, y):
73 self.loss = np.inf # initialize the loss to very high
74 # Initialize a fake NCA and variables needed to compute the loss:
75 self.fake_nca = NeighborhoodComponentsAnalysis()
76 self.fake_nca.n_iter_ = np.inf
77 self.X, y = validate_data(self.fake_nca, X, y, ensure_min_samples=2)
78 y = LabelEncoder().fit_transform(y)
79 self.same_class_mask = y[:, np.newaxis] == y[np.newaxis, :]
80
81 def callback(self, transformation, n_iter):
82 """Stores the last value of the loss function"""
83 self.loss, _ = self.fake_nca._loss_grad_lbfgs(
84 transformation, self.X, self.same_class_mask, -1.0
85 )
86
87 loss_storer = LossStorer(X, y)
88 nca = NeighborhoodComponentsAnalysis(random_state=42, callback=loss_storer.callback)
89 X_t = nca.fit_transform(X, y)
90 print(X_t)
91 # test that points are collapsed into one point
92 assert_array_almost_equal(X_t - X_t[0], 0.0)
93 assert abs(loss_storer.loss + 1) < 1e-10

96def test_finite_differences(global_random_seed):
97 """Test gradient of loss function
98
99 Assert that the gradient is almost equal to its finite differences
100 approximation.
101 """
102 # Initialize the transformation `M`, as well as `X` and `y` and `NCA`
103 rng = np.random.RandomState(global_random_seed)
104 X, y = make_classification(random_state=global_random_seed)
105 M = rng.randn(rng.randint(1, X.shape[1] + 1), X.shape[1])
106 nca = NeighborhoodComponentsAnalysis()
107 nca.n_iter_ = 0
108 mask = y[:, np.newaxis] == y[np.newaxis, :]
109
110 def fun(M):
111 return nca._loss_grad_lbfgs(M, X, mask)[0]
112
113 def grad(M):
114 return nca._loss_grad_lbfgs(M, X, mask)[1]
115
116 # compare the gradient to a finite difference approximation
117 diff = check_grad(fun, grad, M.ravel())
118 assert diff == pytest.approx(0.0, abs=1e-4)

289 nca = clone(nca_base)

331def test_warm_start_effectiveness():
332 # A 1-iteration second fit on same data should give almost same result
333 # with warm starting, and quite different result without warm starting.
334
335 nca_warm = NeighborhoodComponentsAnalysis(warm_start=True, random_state=0)
336 nca_warm.fit(iris_data, iris_target)
337 transformation_warm = nca_warm.components_
338 nca_warm.max_iter = 1
339 nca_warm.fit(iris_data, iris_target)
340 transformation_warm_plus_one = nca_warm.components_
341
342 nca_cold = NeighborhoodComponentsAnalysis(warm_start=False, random_state=0)
343 nca_cold.fit(iris_data, iris_target)
344 transformation_cold = nca_cold.components_
345 nca_cold.max_iter = 1
346 nca_cold.fit(iris_data, iris_target)
347 transformation_cold_plus_one = nca_cold.components_
348
349 diff_warm = np.sum(np.abs(transformation_warm_plus_one - transformation_warm))
350 diff_cold = np.sum(np.abs(transformation_cold_plus_one - transformation_cold))
351 assert diff_warm < 3.0, (
352 "Transformer changed significantly after one "
353 "iteration even though it was warm-started."
354 )
355
356 assert diff_cold > diff_warm, (
357 "Cold-started transformer changed less "
358 "significantly than warm-started "
359 "transformer after one iteration."
360 )
