362 def test_large_chebyshev(self):
363 # The weights for Chebyshev points of the second kind have analytically
364 # solvable weights. Naive calculation of barycentric weights will fail
365 # for large N because of numerical underflow and overflow. We test
366 # correctness for large N against analytical Chebyshev weights.
367
368 # Without capacity scaling or permutation, n=800 fails,
369 # With just capacity scaling, n=1097 fails
370 # With both capacity scaling and random permutation, n=30000 succeeds
371 n = 800
372 j = np.arange(n + 1).astype(np.float64)
373 x = np.cos(j * np.pi / n)
374
375 # See page 506 of Berrut and Trefethen 2004 for this formula
376 w = (-1) ** j
377 w[0] *= 0.5
378 w[-1] *= 0.5
379
380 P = BarycentricInterpolator(x)
381
382 # It's okay to have a constant scaling factor in the weights because it
383 # cancels out in the evaluation of the polynomial.
384 factor = P.wi[0]
385 assert_almost_equal(P.wi / (2 * factor), w)

525class TestCubicSpline:
526 @staticmethod
527 def check_correctness(S, bc_start='not-a-knot', bc_end='not-a-knot',
528 tol=1e-14):
529 """Check that spline coefficients satisfy the continuity and boundary
530 conditions."""
531 x = S.x
532 c = S.c
533 dx = np.diff(x)
534 dx = dx.reshape([dx.shape[0]] + [1] * (c.ndim - 2))
535 dxi = dx[:-1]
536
537 # Check C2 continuity.
538 assert_allclose(c[3, 1:], c[0, :-1] * dxi**3 + c[1, :-1] * dxi**2 +
539 c[2, :-1] * dxi + c[3, :-1], rtol=tol, atol=tol)
540 assert_allclose(c[2, 1:], 3 * c[0, :-1] * dxi**2 +
541 2 * c[1, :-1] * dxi + c[2, :-1], rtol=tol, atol=tol)
542 assert_allclose(c[1, 1:], 3 * c[0, :-1] * dxi + c[1, :-1],
543 rtol=tol, atol=tol)
544
545 # Check that we found a parabola, the third derivative is 0.

573 if x.size == 2:
