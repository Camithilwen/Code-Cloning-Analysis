5110def toUnicode(glyph, isZapfDingbats=False):
5111 """Convert glyph names to Unicode, such as ``'longs_t.oldstyle'`` --> ``u'≈øt'``
5112
5113 If ``isZapfDingbats`` is ``True``, the implementation recognizes additional
5114 glyph names (as required by the AGL specification).
5115 """
5116 # https://github.com/adobe-type-tools/agl-specification#2-the-mapping
5117 #
5118 # 1. Drop all the characters from the glyph name starting with
5119 # the first occurrence of a period (U+002E; FULL STOP), if any.
5120 glyph = glyph.split(".", 1)[0]
5121
5122 # 2. Split the remaining string into a sequence of components,
5123 # using underscore (U+005F; LOW LINE) as the delimiter.
5124 components = glyph.split("_")
5125
5126 # 3. Map each component to a character string according to the
5127 # procedure below, and concatenate those strings; the result
5128 # is the character string to which the glyph name is mapped.
5129 result = [_glyphComponentToUnicode(c, isZapfDingbats) for c in components]
5130 return "".join(result)

5142 # to the corresponding character in that list.
5143 uchars = LEGACY_AGL2UV.get(component)

5168 # to the string made of this scalar value.
