64 def test_solver_concordance(self):
65 # Assert that dogleg uses fewer iterations than ncg on the Rosenbrock
66 # test function, although this does not necessarily mean

80 options={'return_all': True})
81 r_ncg = minimize(f, x0, jac=g, hess=h, tol=1e-8,
82 method='newton-cg', options={'return_all': True})
83 r_iterative = minimize(f, x0, jac=g, hess=h, tol=1e-8,
84 method='trust-exact',
