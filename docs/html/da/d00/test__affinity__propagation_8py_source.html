71def test_affinity_propagation_no_copy():
72 """Check behaviour of not copying the input data."""
73 S = -euclidean_distances(X, squared=True)
74 S_original = S.copy()
75 preference = np.median(S) * 10
76 assert not np.allclose(S.diagonal(), preference)
77
78 # with copy=True S should not be modified
79 affinity_propagation(S, preference=preference, copy=True, random_state=0)
80 assert_allclose(S, S_original)
81 assert not np.allclose(S.diagonal(), preference)
82 assert_allclose(S.diagonal(), np.zeros(S.shape[0]))
83
84 # with copy=False S will be modified inplace
85 affinity_propagation(S, preference=preference, copy=False, random_state=0)
86 assert_allclose(S.diagonal(), preference)
87
88 # test that copy=True and copy=False lead to the same result
89 S = S_original.copy()
90 af = AffinityPropagation(preference=preference, verbose=True, random_state=0)
91
92 labels = af.fit(X).labels_
93 _, labels_no_copy = affinity_propagation(
94 S, preference=preference, copy=False, random_state=74
95 )
96 assert_array_equal(labels, labels_no_copy)

153def test_affinity_propagation_equal_mutual_similarities(global_dtype):
154 X = np.array([[-1, 1], [1, -1]], dtype=global_dtype)
155 S = -euclidean_distances(X, squared=True)
156
157 # setting preference > similarity
158 with pytest.warns(UserWarning, match="mutually equal"):
159 cluster_center_indices, labels = affinity_propagation(S, preference=0)
160
161 # expect every sample to become an exemplar
162 assert_array_equal([0, 1], cluster_center_indices)
163 assert_array_equal([0, 1], labels)
164
165 # setting preference < similarity
166 with pytest.warns(UserWarning, match="mutually equal"):
167 cluster_center_indices, labels = affinity_propagation(S, preference=-10)
168
169 # expect one cluster, with arbitrary (first) sample as exemplar
170 assert_array_equal([0], cluster_center_indices)
171 assert_array_equal([0, 0], labels)
172
173 # setting different preferences
174 with warnings.catch_warnings():
175 warnings.simplefilter("error", UserWarning)
176 cluster_center_indices, labels = affinity_propagation(
177 S, preference=[-20, -10], random_state=37
178 )
179
180 # expect one cluster, with highest-preference sample as exemplar
181 assert_array_equal([1], cluster_center_indices)

217def test_equal_similarities_and_preferences(global_dtype):
218 # Unequal distances
219 X = np.array([[0, 0], [1, 1], [-2, -2]], dtype=global_dtype)
220 S = -euclidean_distances(X, squared=True)
