68class CheckingClassifier(ClassifierMixin, BaseEstimator):
69 """Dummy classifier to test pipelining and meta-estimators.
70
71 Checks some property of `X` and `y`in fit / predict.
72 This allows testing whether pipelines / cross-validation or metaestimators
73 changed the input.
74
75 Can also be used to check if `fit_params` are passed correctly, and
76 to force a certain score to be returned.
77
78 Parameters
79 ----------
80 check_y, check_X : callable, default=None
81 The callable used to validate `X` and `y`. These callable should return
82 a bool where `False` will trigger an `AssertionError`. If `None`, the
83 data is not validated. Default is `None`.
84
85 check_y_params, check_X_params : dict, default=None
86 The optional parameters to pass to `check_X` and `check_y`. If `None`,
87 then no parameters are passed in.
88
89 methods_to_check : "all" or list of str, default="all"
90 The methods in which the checks should be applied. By default,
91 all checks will be done on all methods (`fit`, `predict`,
92 `predict_proba`, `decision_function` and `score`).
93
94 foo_param : int, default=0
95 A `foo` param. When `foo > 1`, the output of :meth:`score` will be 1
96 otherwise it is 0.
97
98 expected_sample_weight : bool, default=False
99 Whether to check if a valid `sample_weight` was passed to `fit`.
100
101 expected_fit_params : list of str, default=None
102 A list of the expected parameters given when calling `fit`.
103
104 Attributes
105 ----------
106 classes_ : int
107 The classes seen during `fit`.
108
109 n_features_in_ : int
110 The number of features seen during `fit`.
111
112 Examples
113 --------
114 >>> from sklearn.utils._mocking import CheckingClassifier
115
116 This helper allow to assert to specificities regarding `X` or `y`. In this
117 case we expect `check_X` or `check_y` to return a boolean.
118
119 >>> from sklearn.datasets import load_iris
120 >>> X, y = load_iris(return_X_y=True)
121 >>> clf = CheckingClassifier(check_X=lambda x: x.shape == (150, 4))
122 >>> clf.fit(X, y)
123 CheckingClassifier(...)
124
125 We can also provide a check which might raise an error. In this case, we
126 expect `check_X` to return `X` and `check_y` to return `y`.
127
128 >>> from sklearn.utils import check_array
129 >>> clf = CheckingClassifier(check_X=check_array)
130 >>> clf.fit(X, y)
131 CheckingClassifier(...)
132 """
133
134 def __init__(
135 self,
136 *,
137 check_y=None,
138 check_y_params=None,
139 check_X=None,
140 check_X_params=None,
141 methods_to_check="all",
142 foo_param=0,
143 expected_sample_weight=None,
144 expected_fit_params=None,
145 random_state=None,
146 ):
147 self.check_y = check_y
148 self.check_y_params = check_y_params
149 self.check_X = check_X
150 self.check_X_params = check_X_params
151 self.methods_to_check = methods_to_check
152 self.foo_param = foo_param
153 self.expected_sample_weight = expected_sample_weight
154 self.expected_fit_params = expected_fit_params
155 self.random_state = random_state
156
157 def _check_X_y(self, X, y=None, should_be_fitted=True):
158 """Validate X and y and make extra check.
159
160 Parameters
161 ----------
162 X : array-like of shape (n_samples, n_features)
163 The data set.
164 `X` is checked only if `check_X` is not `None` (default is None).
165 y : array-like of shape (n_samples), default=None
166 The corresponding target, by default `None`.
167 `y` is checked only if `check_y` is not `None` (default is None).
168 should_be_fitted : bool, default=True
169 Whether or not the classifier should be already fitted.
170 By default True.
171
172 Returns
173 -------
174 X, y
175 """
176 if should_be_fitted:
177 check_is_fitted(self)
178 if self.check_X is not None:
179 params = {} if self.check_X_params is None else self.check_X_params
180 checked_X = self.check_X(X, **params)
181 if isinstance(checked_X, (bool, np.bool_)):
182 assert checked_X
183 else:
184 X = checked_X
185 if y is not None and self.check_y is not None:
186 params = {} if self.check_y_params is None else self.check_y_params
187 checked_y = self.check_y(y, **params)
188 if isinstance(checked_y, (bool, np.bool_)):
189 assert checked_y
190 else:
191 y = checked_y
192 return X, y

194 def fit(self, X, y, sample_weight=None, **fit_params):
195 """Fit classifier.
196
197 Parameters
198 ----------
199 X : array-like of shape (n_samples, n_features)
200 Training vector, where `n_samples` is the number of samples and
201 `n_features` is the number of features.
202
203 y : array-like of shape (n_samples, n_outputs) or (n_samples,), \
204 default=None
205 Target relative to X for classification or regression;
206 None for unsupervised learning.
207
208 sample_weight : array-like of shape (n_samples,), default=None
209 Sample weights. If None, then samples are equally weighted.
210
211 **fit_params : dict of string -> object
212 Parameters passed to the ``fit`` method of the estimator
213
214 Returns
215 -------
216 self
217 """
218 assert _num_samples(X) == _num_samples(y)
219 if self.methods_to_check == "all" or "fit" in self.methods_to_check:
220 X, y = self._check_X_y(X, y, should_be_fitted=False)
221 self.n_features_in_ = np.shape(X)[1]
222 self.classes_ = np.unique(check_array(y, ensure_2d=False, allow_nd=True))
223 if self.expected_fit_params:
224 missing = set(self.expected_fit_params) - set(fit_params)
225 if missing:
226 raise AssertionError(
227 f"Expected fit parameter(s) {list(missing)} not seen."
228 )
229 for key, value in fit_params.items():
230 if _num_samples(value) != _num_samples(X):
231 raise AssertionError(
232 f"Fit parameter {key} has length {_num_samples(value)}"
233 f"; expected {_num_samples(X)}."
234 )
235 if self.expected_sample_weight:
236 if sample_weight is None:
237 raise AssertionError("Expected sample_weight to be passed")
238 _check_sample_weight(sample_weight, X)
239
240 return self
241
242 def predict(self, X):
243 """Predict the first class seen in `classes_`.
244
245 Parameters
246 ----------
247 X : array-like of shape (n_samples, n_features)
248 The input data.
249
250 Returns
251 -------
252 preds : ndarray of shape (n_samples,)
253 Predictions of the first class seen in `classes_`.
254 """
255 if self.methods_to_check == "all" or "predict" in self.methods_to_check:
256 X, y = self._check_X_y(X)
257 rng = check_random_state(self.random_state)
258 return rng.choice(self.classes_, size=_num_samples(X))
259
260 def predict_proba(self, X):
261 """Predict probabilities for each class.
262
263 Here, the dummy classifier will provide a probability of 1 for the
264 first class of `classes_` and 0 otherwise.
265
266 Parameters
267 ----------
268 X : array-like of shape (n_samples, n_features)
269 The input data.
270
271 Returns
272 -------
273 proba : ndarray of shape (n_samples, n_classes)
274 The probabilities for each sample and class.
275 """
276 if self.methods_to_check == "all" or "predict_proba" in self.methods_to_check:
277 X, y = self._check_X_y(X)
278 rng = check_random_state(self.random_state)
279 proba = rng.randn(_num_samples(X), len(self.classes_))
280 proba = np.abs(proba, out=proba)
281 proba /= np.sum(proba, axis=1)[:, np.newaxis]
282 return proba

284 def decision_function(self, X):
285 """Confidence score.
286
287 Parameters
288 ----------
289 X : array-like of shape (n_samples, n_features)
290 The input data.
291
292 Returns
293 -------
294 decision : ndarray of shape (n_samples,) if n_classes == 2\
295 else (n_samples, n_classes)
296 Confidence score.
297 """
298 if (
299 self.methods_to_check == "all"
300 or "decision_function" in self.methods_to_check
301 ):
302 X, y = self._check_X_y(X)
303 rng = check_random_state(self.random_state)
304 if len(self.classes_) == 2:
305 # for binary classifier, the confidence score is related to
306 # classes_[1] and therefore should be null.
307 return rng.randn(_num_samples(X))
308 else:
309 return rng.randn(_num_samples(X), len(self.classes_))

311 def score(self, X=None, Y=None):
312 """Fake score.
313
314 Parameters
315 ----------
316 X : array-like of shape (n_samples, n_features)
317 Input data, where `n_samples` is the number of samples and
318 `n_features` is the number of features.
319
320 Y : array-like of shape (n_samples, n_output) or (n_samples,)
321 Target relative to X for classification or regression;
322 None for unsupervised learning.
323
324 Returns
325 -------
326 score : float
327 Either 0 or 1 depending of `foo_param` (i.e. `foo_param > 1 =>
328 score=1` otherwise `score=0`).
329 """
330 if self.methods_to_check == "all" or "score" in self.methods_to_check:
331 self._check_X_y(X, Y)
332 if self.foo_param > 1:
333 score = 1.0
334 else:
335 score = 0.0
336 return score
