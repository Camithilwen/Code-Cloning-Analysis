51def test_classification(estimator, selection_crit):
52 # Check classification for various parameter settings.
53 # Also assert that predictions for strings and numerical labels are equal.
54 # Also test for multioutput classification
55 threshold = 0.75
56 max_iter = 10
57 st = SelfTrainingClassifier(
58 estimator, max_iter=max_iter, threshold=threshold, criterion=selection_crit
59 )
60 st.fit(X_train, y_train_missing_labels)
61 pred = st.predict(X_test)
62 proba = st.predict_proba(X_test)
63
64 st_string = SelfTrainingClassifier(
65 estimator, max_iter=max_iter, criterion=selection_crit, threshold=threshold
66 )
67 st_string.fit(X_train, y_train_missing_strings)
68 pred_string = st_string.predict(X_test)
69 proba_string = st_string.predict_proba(X_test)
70
71 assert_array_equal(np.vectorize(mapping.get)(pred), pred_string)
72 assert_array_equal(proba, proba_string)
73
74 assert st.termination_condition_ == st_string.termination_condition_
75 # Check consistency between labeled_iter, n_iter and max_iter
76 labeled = y_train_missing_labels != -1
77 # assert that labeled samples have labeled_iter = 0
78 assert_array_equal(st.labeled_iter_ == 0, labeled)
79 # assert that labeled samples do not change label during training
80 assert_array_equal(y_train_missing_labels[labeled], st.transduction_[labeled])

265 # Tests that the labels added by st really are the 10 best labels.
266 svc = SVC(gamma="scale", probability=True, random_state=0)
267 st = SelfTrainingClassifier(svc, criterion="k_best", max_iter=1, k_best=10)
268 has_label = y_train_missing_labels != -1
269 st.fit(X_train, y_train_missing_labels)
270
271 got_label = ~has_label & (st.transduction_ != -1)
272
273 svc.fit(X_train[has_label], y_train_missing_labels[has_label])
274 pred = svc.predict_proba(X_train[~has_label])
275 max_proba = np.max(pred, axis=1)
276
277 most_confident_svc = X_train[~has_label][np.argsort(max_proba)[-10:]]

320def test_self_training_estimator_attribute_error():
321 """Check that we raise the proper AttributeErrors when the `estimator`
322 does not implement the `predict_proba` method, which is called from within
323 `fit`, or `decision_function`, which is decorated with `available_if`.
324
325 Non-regression test for:
326 https://github.com/scikit-learn/scikit-learn/issues/28108
327 """
328 # `SVC` with `probability=False` does not implement 'predict_proba' that
329 # is required internally in `fit` of `SelfTrainingClassifier`. We expect
330 # an AttributeError to be raised.
331 estimator = SVC(probability=False, gamma="scale")
332 self_training = SelfTrainingClassifier(estimator)
333
334 with pytest.raises(AttributeError, match="has no attribute 'predict_proba'"):
335 self_training.fit(X_train, y_train_missing_labels)
336
337 # `DecisionTreeClassifier` does not implement 'decision_function' and
338 # should raise an AttributeError
339 self_training = SelfTrainingClassifier(estimator=DecisionTreeClassifier())
340
341 outer_msg = "This 'SelfTrainingClassifier' has no attribute 'decision_function'"
342 inner_msg = "'DecisionTreeClassifier' object has no attribute 'decision_function'"
343 with pytest.raises(AttributeError, match=outer_msg) as exec_info:
344 self_training.fit(X_train, y_train_missing_labels).decision_function(X_train)
345 assert isinstance(exec_info.value.__cause__, AttributeError)
346 assert inner_msg in str(exec_info.value.__cause__)
