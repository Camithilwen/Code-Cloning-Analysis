52def test_lof_performance(global_dtype):
53 # Generate train/test data
54 rng = check_random_state(2)
55 X = 0.3 * rng.randn(120, 2).astype(global_dtype, copy=False)
56 X_train = X[:100]
57
58 # Generate some abnormal novel observations
59 X_outliers = rng.uniform(low=-4, high=4, size=(20, 2)).astype(
60 global_dtype, copy=False
61 )
62 X_test = np.r_[X[100:], X_outliers]
63 y_test = np.array([0] * 20 + [1] * 20)
64
65 # fit the model for novelty detection
66 clf = neighbors.LocalOutlierFactor(novelty=True).fit(X_train)
67
68 # predict scores (the lower, the more normal)
69 y_pred = -clf.decision_function(X_test)
70
71 # check that roc_auc is good
72 assert roc_auc_score(y_test, y_pred) > 0.99

75def test_lof_values(global_dtype):
76 # toy samples:
77 X_train = np.asarray([[1, 1], [1, 2], [2, 1]], dtype=global_dtype)
78 clf1 = neighbors.LocalOutlierFactor(
79 n_neighbors=2, contamination=0.1, novelty=True
80 ).fit(X_train)
81 clf2 = neighbors.LocalOutlierFactor(n_neighbors=2, novelty=True).fit(X_train)
82 s_0 = 2.0 * sqrt(2.0) / (1.0 + sqrt(2.0))
83 s_1 = (1.0 + sqrt(2)) * (1.0 / (4.0 * sqrt(2.0)) + 1.0 / (2.0 + 2.0 * sqrt(2)))
84 # check predict()
85 assert_allclose(-clf1.negative_outlier_factor_, [s_0, s_1, s_1])
86 assert_allclose(-clf2.negative_outlier_factor_, [s_0, s_1, s_1])
87 # check predict(one sample not in train)
88 assert_allclose(-clf1.score_samples([[2.0, 2.0]]), [s_0])
89 assert_allclose(-clf2.score_samples([[2.0, 2.0]]), [s_0])
90 # check predict(one sample already in train)
91 assert_allclose(-clf1.score_samples([[1.0, 1.0]]), [s_1])
92 assert_allclose(-clf2.score_samples([[1.0, 1.0]]), [s_1])

95def test_lof_precomputed(global_dtype, random_state=42):
96 """Tests LOF with a distance matrix."""
97 # Note: smaller samples may result in spurious test success
98 rng = np.random.RandomState(random_state)
99 X = rng.random_sample((10, 4)).astype(global_dtype, copy=False)
100 Y = rng.random_sample((3, 4)).astype(global_dtype, copy=False)
101 DXX = metrics.pairwise_distances(X, metric="euclidean")
102 DYX = metrics.pairwise_distances(Y, X, metric="euclidean")
103 # As a feature matrix (n_samples by n_features)
104 lof_X = neighbors.LocalOutlierFactor(n_neighbors=3, novelty=True)
105 lof_X.fit(X)
106 pred_X_X = lof_X._predict()
107 pred_X_Y = lof_X.predict(Y)
108
109 # As a dense distance matrix (n_samples by n_samples)
110 lof_D = neighbors.LocalOutlierFactor(
111 n_neighbors=3, algorithm="brute", metric="precomputed", novelty=True
112 )
113 lof_D.fit(DXX)
114 pred_D_X = lof_D._predict()
115 pred_D_Y = lof_D.predict(DYX)
116
117 assert_allclose(pred_X_X, pred_D_X)
118 assert_allclose(pred_X_Y, pred_D_Y)

235 # expected outliers unless there are ties in the abnormality scores.
236 X = iris.data
237 n_samples = X.shape[0]

330@pytest.mark.parametrize("algorithm", ["auto", "ball_tree", "kd_tree", "brute"])
331@pytest.mark.parametrize("novelty", [True, False])
332@pytest.mark.parametrize("contamination", [0.5, "auto"])
333def test_lof_dtype_equivalence(algorithm, novelty, contamination):
334 """Check the equivalence of the results with 32 and 64 bits input."""
335
336 inliers = iris.data[:50] # setosa iris are really distinct from others
337 outliers = iris.data[-5:] # virginica will be considered as outliers
338 # lower the precision of the input data to check that we have an equivalence when
339 # making the computation in 32 and 64 bits.
340 X = np.concatenate([inliers, outliers], axis=0).astype(np.float32)
341
342 lof_32 = neighbors.LocalOutlierFactor(
343 algorithm=algorithm, novelty=novelty, contamination=contamination
344 )
345 X_32 = X.astype(np.float32, copy=True)
346 lof_32.fit(X_32)
347
348 lof_64 = neighbors.LocalOutlierFactor(
349 algorithm=algorithm, novelty=novelty, contamination=contamination
350 )
351 X_64 = X.astype(np.float64, copy=True)
352 lof_64.fit(X_64)
353
354 assert_allclose(lof_32.negative_outlier_factor_, lof_64.negative_outlier_factor_)
355
356 for method in ("score_samples", "decision_function", "predict", "fit_predict"):
357 if hasattr(lof_32, method):
358 y_pred_32 = getattr(lof_32, method)(X_32)
359 y_pred_64 = getattr(lof_64, method)(X_64)
360 assert_allclose(y_pred_32, y_pred_64, atol=0.0002)
