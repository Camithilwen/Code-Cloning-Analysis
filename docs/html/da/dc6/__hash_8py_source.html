22class FeatureHasher(TransformerMixin, BaseEstimator):
23 """Implements feature hashing, aka the hashing trick.
24
25 This class turns sequences of symbolic feature names (strings) into
26 scipy.sparse matrices, using a hash function to compute the matrix column
27 corresponding to a name. The hash function employed is the signed 32-bit
28 version of Murmurhash3.
29
30 Feature names of type byte string are used as-is. Unicode strings are
31 converted to UTF-8 first, but no Unicode normalization is done.
32 Feature values must be (finite) numbers.
33
34 This class is a low-memory alternative to DictVectorizer and
35 CountVectorizer, intended for large-scale (online) learning and situations
36 where memory is tight, e.g. when running prediction code on embedded
37 devices.
38
39 For an efficiency comparison of the different feature extractors, see
40 :ref:`sphx_glr_auto_examples_text_plot_hashing_vs_dict_vectorizer.py`.
41
42 Read more in the :ref:`User Guide <feature_hashing>`.
43
44 .. versionadded:: 0.13
45
46 Parameters
47 ----------
48 n_features : int, default=2**20
49 The number of features (columns) in the output matrices. Small numbers
50 of features are likely to cause hash collisions, but large numbers
51 will cause larger coefficient dimensions in linear learners.
52 input_type : str, default='dict'
53 Choose a string from {'dict', 'pair', 'string'}.
54 Either "dict" (the default) to accept dictionaries over
55 (feature_name, value); "pair" to accept pairs of (feature_name, value);
56 or "string" to accept single strings.
57 feature_name should be a string, while value should be a number.
58 In the case of "string", a value of 1 is implied.
59 The feature_name is hashed to find the appropriate column for the
60 feature. The value's sign might be flipped in the output (but see
61 non_negative, below).
62 dtype : numpy dtype, default=np.float64
63 The type of feature values. Passed to scipy.sparse matrix constructors
64 as the dtype argument. Do not set this to bool, np.boolean or any
65 unsigned integer type.
66 alternate_sign : bool, default=True
67 When True, an alternating sign is added to the features as to
68 approximately conserve the inner product in the hashed space even for
69 small n_features. This approach is similar to sparse random projection.
70
71 .. versionchanged:: 0.19
72 ``alternate_sign`` replaces the now deprecated ``non_negative``
73 parameter.
74
75 See Also
76 --------
77 DictVectorizer : Vectorizes string-valued features using a hash table.
78 sklearn.preprocessing.OneHotEncoder : Handles nominal/categorical features.
79
80 Notes
81 -----
82 This estimator is :term:`stateless` and does not need to be fitted.
83 However, we recommend to call :meth:`fit_transform` instead of
84 :meth:`transform`, as parameter validation is only performed in
85 :meth:`fit`.
86
87 Examples
88 --------
89 >>> from sklearn.feature_extraction import FeatureHasher
90 >>> h = FeatureHasher(n_features=10)
91 >>> D = [{'dog': 1, 'cat':2, 'elephant':4},{'dog': 2, 'run': 5}]
92 >>> f = h.transform(D)
93 >>> f.toarray()
94 array([[ 0., 0., -4., -1., 0., 0., 0., 0., 0., 2.],
95 [ 0., 0., 0., -2., -5., 0., 0., 0., 0., 0.]])
96
97 With `input_type="string"`, the input must be an iterable over iterables of
98 strings:
99
100 >>> h = FeatureHasher(n_features=8, input_type="string")
101 >>> raw_X = [["dog", "cat", "snake"], ["snake", "dog"], ["cat", "bird"]]
102 >>> f = h.transform(raw_X)
103 >>> f.toarray()
104 array([[ 0., 0., 0., -1., 0., -1., 0., 1.],
105 [ 0., 0., 0., -1., 0., -1., 0., 0.],
106 [ 0., -1., 0., 0., 0., 0., 0., 1.]])
107 """

154 def transform(self, raw_X):
155 """Transform a sequence of instances to a scipy.sparse matrix.
156
157 Parameters
158 ----------
159 raw_X : iterable over iterable over raw features, length = n_samples
160 Samples. Each sample must be iterable an (e.g., a list or tuple)
161 containing/generating feature names (and optionally values, see
162 the input_type constructor argument) which will be hashed.
163 raw_X need not support the len function, so it can be the result
164 of a generator; n_samples is determined on the fly.
165
166 Returns
167 -------
168 X : sparse matrix of shape (n_samples, n_features)
169 Feature matrix, for use with estimators or further transformers.
170 """
171 raw_X = iter(raw_X)
172 if self.input_type == "dict":
173 raw_X = (_iteritems(d) for d in raw_X)
174 elif self.input_type == "string":
175 first_raw_X = next(raw_X)
176 if isinstance(first_raw_X, str):
177 raise ValueError(
178 "Samples can not be a single string. The input must be an iterable"
179 " over iterables of strings."
180 )
181 raw_X_ = chain([first_raw_X], raw_X)
182 raw_X = (((f, 1) for f in x) for x in raw_X_)
183
184 indices, indptr, values = _hashing_transform(
185 raw_X, self.n_features, self.dtype, self.alternate_sign, seed=0
186 )
187 n_samples = indptr.shape[0] - 1
188
189 if n_samples == 0:
190 raise ValueError("Cannot vectorize empty sequence.")
191
192 X = sp.csr_matrix(
193 (values, indices, indptr),
194 dtype=self.dtype,
195 shape=(n_samples, self.n_features),
196 )
197 X.sum_duplicates() # also sorts the indices
198
199 return X
