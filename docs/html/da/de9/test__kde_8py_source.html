16def compute_kernel_slow(Y, X, kernel, h):
17 if h == "scott":
18 h = X.shape[0] ** (-1 / (X.shape[1] + 4))
19 elif h == "silverman":
20 h = (X.shape[0] * (X.shape[1] + 2) / 4) ** (-1 / (X.shape[1] + 4))
21
22 d = np.sqrt(((Y[:, None, :] - X) ** 2).sum(-1))
23 norm = kernel_norm(h, X.shape[1], kernel) / X.shape[0]
24
25 if kernel == "gaussian":
26 return norm * np.exp(-0.5 * (d * d) / (h * h)).sum(-1)
27 elif kernel == "tophat":
28 return norm * (d < h).sum(-1)
29 elif kernel == "epanechnikov":
30 return norm * ((1.0 - (d * d) / (h * h)) * (d < h)).sum(-1)
31 elif kernel == "exponential":
32 return norm * (np.exp(-d / h)).sum(-1)
33 elif kernel == "linear":
34 return norm * ((1 - d / h) * (d < h)).sum(-1)
35 elif kernel == "cosine":
36 return norm * (np.cos(0.5 * np.pi * d / h) * (d < h)).sum(-1)
37 else:
38 raise ValueError("kernel not recognized")

154def test_kde_sample_weights():
155 n_samples = 400
156 size_test = 20
157 weights_neutral = np.full(n_samples, 3.0)
158 for d in [1, 2, 10]:
159 rng = np.random.RandomState(0)
160 X = rng.rand(n_samples, d)
161 weights = 1 + (10 * X.sum(axis=1)).astype(np.int8)
162 X_repetitions = np.repeat(X, weights, axis=0)
163 n_samples_test = size_test // d
164 test_points = rng.rand(n_samples_test, d)
165 for algorithm in ["auto", "ball_tree", "kd_tree"]:
166 for metric in ["euclidean", "minkowski", "manhattan", "chebyshev"]:
167 if algorithm != "kd_tree" or metric in KDTree.valid_metrics:
168 kde = KernelDensity(algorithm=algorithm, metric=metric)
169
170 # Test that adding a constant sample weight has no effect
171 kde.fit(X, sample_weight=weights_neutral)
172 scores_const_weight = kde.score_samples(test_points)
173 sample_const_weight = kde.sample(random_state=1234)
174 kde.fit(X)
175 scores_no_weight = kde.score_samples(test_points)
176 sample_no_weight = kde.sample(random_state=1234)
177 assert_allclose(scores_const_weight, scores_no_weight)
178 assert_allclose(sample_const_weight, sample_no_weight)
179
180 # Test equivalence between sampling and (integer) weights
181 kde.fit(X, sample_weight=weights)
182 scores_weight = kde.score_samples(test_points)
183 sample_weight = kde.sample(random_state=1234)
184 kde.fit(X_repetitions)
185 scores_ref_sampling = kde.score_samples(test_points)
186 sample_ref_sampling = kde.sample(random_state=1234)
187 assert_allclose(scores_weight, scores_ref_sampling)
188 assert_allclose(sample_weight, sample_ref_sampling)
189
190 # Test that sample weights has a non-trivial effect
191 diff = np.max(np.abs(scores_no_weight - scores_weight))
192 assert diff > 0.001
193
194 # Test invariance with respect to arbitrary scaling
195 scale_factor = rng.rand()
196 kde.fit(X, sample_weight=(scale_factor * weights))
197 scores_scaled_weight = kde.score_samples(test_points)
198 assert_allclose(scores_scaled_weight, scores_weight)
