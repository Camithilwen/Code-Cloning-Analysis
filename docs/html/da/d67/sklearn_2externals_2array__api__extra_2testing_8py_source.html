39def lazy_xp_function( # type: ignore[explicit-any]
40 func: Callable[..., Any],
41 *,
42 allow_dask_compute: int = 0,
43 jax_jit: bool = True,
44 static_argnums: int | Sequence[int] | None = None,
45 static_argnames: str | Iterable[str] | None = None,
46) -> None: # numpydoc ignore=GL07
47 """
48 Tag a function to be tested on lazy backends.
49
50 Tag a function so that when any tests are executed with ``xp=jax.numpy`` the
51 function is replaced with a jitted version of itself, and when it is executed with
52 ``xp=dask.array`` the function will raise if it attempts to materialize the graph.
53 This will be later expanded to provide test coverage for other lazy backends.
54
55 In order for the tag to be effective, the test or a fixture must call
56 :func:`patch_lazy_xp_functions`.
57
58 Parameters
59 ----------
60 func : callable
61 Function to be tested.
62 allow_dask_compute : int, optional
63 Number of times `func` is allowed to internally materialize the Dask graph. This
64 is typically triggered by ``bool()``, ``float()``, or ``np.asarray()``.
65
66 Set to 1 if you are aware that `func` converts the input parameters to NumPy and
67 want to let it do so at least for the time being, knowing that it is going to be
68 extremely detrimental for performance.
69
70 If a test needs values higher than 1 to pass, it is a canary that the conversion
71 to NumPy/bool/float is happening multiple times, which translates to multiple
72 computations of the whole graph. Short of making the function fully lazy, you
73 should at least add explicit calls to ``np.asarray()`` early in the function.
74 *Note:* the counter of `allow_dask_compute` resets after each call to `func`, so
75 a test function that invokes `func` multiple times should still work with this
76 parameter set to 1.
77
78 Default: 0, meaning that `func` must be fully lazy and never materialize the
79 graph.
80 jax_jit : bool, optional
81 Set to True to replace `func` with ``jax.jit(func)`` after calling the
82 :func:`patch_lazy_xp_functions` test helper with ``xp=jax.numpy``. Set to False
83 if `func` is only compatible with eager (non-jitted) JAX. Default: True.
84 static_argnums : int | Sequence[int], optional
85 Passed to jax.jit. Positional arguments to treat as static (compile-time
86 constant). Default: infer from `static_argnames` using
87 `inspect.signature(func)`.
88 static_argnames : str | Iterable[str], optional
89 Passed to jax.jit. Named arguments to treat as static (compile-time constant).
90 Default: infer from `static_argnums` using `inspect.signature(func)`.
91
92 See Also
93 --------
94 patch_lazy_xp_functions : Companion function to call from the test or fixture.
95 jax.jit : JAX function to compile a function for performance.
96
97 Examples
98 --------
99 In ``test_mymodule.py``::
100
101 from array_api_extra.testing import lazy_xp_function from mymodule import myfunc
102
103 lazy_xp_function(myfunc)
104
105 def test_myfunc(xp):
106 a = xp.asarray([1, 2])
107 # When xp=jax.numpy, this is the same as `b = jax.jit(myfunc)(a)`
108 # When xp=dask.array, crash on compute() or persist()
109 b = myfunc(a)
110
111 Notes
112 -----
113 In order for this tag to be effective, the test function must be imported into the
114 test module globals without its namespace; alternatively its namespace must be
115 declared in a ``lazy_xp_modules`` list in the test module globals.
116
117 Example 1::
118
119 from mymodule import myfunc
120
121 lazy_xp_function(myfunc)
122
123 def test_myfunc(xp):
124 x = myfunc(xp.asarray([1, 2]))
125
126 Example 2::
127
128 import mymodule
129
130 lazy_xp_modules = [mymodule]
131 lazy_xp_function(mymodule.myfunc)
132
133 def test_myfunc(xp):
134 x = mymodule.myfunc(xp.asarray([1, 2]))
135
136 A test function can circumvent this monkey-patching system by using a namespace
137 outside of the two above patterns. You need to sanitize your code to make sure this
138 only happens intentionally.
139
140 Example 1::
141
142 import mymodule
143 from mymodule import myfunc
144
145 lazy_xp_function(myfunc)
146
147 def test_myfunc(xp):
148 a = xp.asarray([1, 2])
149 b = myfunc(a) # This is wrapped when xp=jax.numpy or xp=dask.array
150 c = mymodule.myfunc(a) # This is not
151
152 Example 2::
153
154 import mymodule
155
156 class naked:
157 myfunc = mymodule.myfunc
158
159 lazy_xp_modules = [mymodule]
160 lazy_xp_function(mymodule.myfunc)
161
162 def test_myfunc(xp):
163 a = xp.asarray([1, 2])
164 b = mymodule.myfunc(a) # This is wrapped when xp=jax.numpy or xp=dask.array
165 c = naked.myfunc(a) # This is not
166 """
167 tags = {
168 "allow_dask_compute": allow_dask_compute,
169 "jax_jit": jax_jit,
170 "static_argnums": static_argnums,
171 "static_argnames": static_argnames,
172 }

179def patch_lazy_xp_functions(
180 request: pytest.FixtureRequest, monkeypatch: pytest.MonkeyPatch, *, xp: ModuleType
181) -> None:
182 """
183 Test lazy execution of functions tagged with :func:`lazy_xp_function`.
184
185 If ``xp==jax.numpy``, search for all functions which have been tagged with
186 :func:`lazy_xp_function` in the globals of the module that defines the current test,
187 as well as in the ``lazy_xp_modules`` list in the globals of the same module,
188 and wrap them with :func:`jax.jit`. Unwrap them at the end of the test.
189
190 If ``xp==dask.array``, wrap the functions with a decorator that disables
191 ``compute()`` and ``persist()`` and ensures that exceptions and warnings are raised
192 eagerly.
193
194 This function should be typically called by your library's `xp` fixture that runs
195 tests on multiple backends::
196
197 @pytest.fixture(params=[numpy, array_api_strict, jax.numpy, dask.array])
198 def xp(request, monkeypatch):
199 patch_lazy_xp_functions(request, monkeypatch, xp=request.param)
200 return request.param
201
202 but it can be otherwise be called by the test itself too.
203
204 Parameters
205 ----------
206 request : pytest.FixtureRequest
207 Pytest fixture, as acquired by the test itself or by one of its fixtures.
208 monkeypatch : pytest.MonkeyPatch
209 Pytest fixture, as acquired by the test itself or by one of its fixtures.
210 xp : array_namespace
211 Array namespace to be tested.
212
213 See Also
214 --------
215 lazy_xp_function : Tag a function to be tested on lazy backends.
216 pytest.FixtureRequest : `request` test function parameter.
217 """
218 mod = cast(ModuleType, request.module)
219 mods = [mod, *cast(list[ModuleType], getattr(mod, "lazy_xp_modules", []))]
220
221 def iter_tagged() -> ( # type: ignore[explicit-any]
222 Iterator[tuple[ModuleType, str, Callable[..., Any], dict[str, Any]]]
223 ):
224 for mod in mods:
225 for name, func in mod.__dict__.items():
226 tags: dict[str, Any] | None = None # type: ignore[explicit-any]
227 with contextlib.suppress(AttributeError):
228 tags = func._lazy_xp_function # pylint: disable=protected-access
229 if tags is None:
230 with contextlib.suppress(KeyError, TypeError):
231 tags = _ufuncs_tags[func]
232 if tags is not None:
233 yield mod, name, func, tags
234
235 if is_dask_namespace(xp):
236 for mod, name, func, tags in iter_tagged():
237 n = tags["allow_dask_compute"]
238 wrapped = _dask_wrap(func, n)
239 monkeypatch.setattr(mod, name, wrapped)
240
241 elif is_jax_namespace(xp):
242 import jax
243
244 for mod, name, func, tags in iter_tagged():
245 if tags["jax_jit"]:
246 # suppress unused-ignore to run mypy in -e lint as well as -e dev
247 wrapped = cast( # type: ignore[explicit-any]
248 Callable[..., Any],
249 jax.jit(
250 func,
251 static_argnums=tags["static_argnums"],
252 static_argnames=tags["static_argnames"],
253 ),
254 )
255 monkeypatch.setattr(mod, name, wrapped)

260 Dask scheduler that counts how many times `dask.compute` is called.
261
262 If the number of times exceeds 'max_count', it raises an error.

264

294def _dask_wrap(
295 func: Callable[P, T], n: int
296) -> Callable[P, T]: # numpydoc ignore=PR01,RT01
297 """
298 Wrap `func` to raise if it attempts to call `dask.compute` more than `n` times.
299
300 After the function returns, materialize the graph in order to re-raise exceptions.
301 """
302 import dask
303
304 func_name = getattr(func, "__name__", str(func))
305 n_str = f"only up to {n}" if n else "no"
306 msg = (
307 f"Called `dask.compute()` or `dask.persist()` {n + 1} times, "
308 f"but {n_str} calls are allowed. Set "
309 f"`lazy_xp_function({func_name}, allow_dask_compute={n + 1})` "
310 "to allow for more (but note that this will harm performance). "
311 )
312
313 @wraps(func)
314 def wrapper(*args: P.args, **kwargs: P.kwargs) -> T: # numpydoc ignore=GL08
315 scheduler = CountingDaskScheduler(n, msg)
316 with dask.config.set({"scheduler": scheduler}): # pyright: ignore[reportPrivateImportUsage]
317 out = func(*args, **kwargs)
318
319 # Block until the graph materializes and reraise exceptions. This allows
320 # `pytest.raises` and `pytest.warns` to work as expected. Note that this would
321 # not work on scheduler='distributed', as it would not block.
322 return dask.persist(out, scheduler="threads")[0] # type: ignore[attr-defined,no-untyped-call,func-returns-value,index] # pyright: ignore[reportPrivateImportUsage]

324 return wrapper
