131 def __lt__(self, other_node):
132 """Comparison for priority queue.
133
134 Nodes with high gain are higher priority than nodes with low gain.
135
136 heapq.heappush only need the '<' operator.

138 priority).

141 ----------
142 other_node : TreeNode
143 The node to compare with.

145 return self.split_info.gain > other_node.split_info.gain

148class TreeGrower:
149 """Tree grower class used to build a tree.
150
151 The tree is fitted to predict the values of a Newton-Raphson step. The
152 splits are considered in a best-first fashion, and the quality of a
153 split is defined in splitting._split_gain.
154
155 Parameters
156 ----------
157 X_binned : ndarray of shape (n_samples, n_features), dtype=np.uint8
158 The binned input samples. Must be Fortran-aligned.
159 gradients : ndarray of shape (n_samples,)
160 The gradients of each training sample. Those are the gradients of the
161 loss w.r.t the predictions, evaluated at iteration ``i - 1``.
162 hessians : ndarray of shape (n_samples,)
163 The hessians of each training sample. Those are the hessians of the
164 loss w.r.t the predictions, evaluated at iteration ``i - 1``.
165 max_leaf_nodes : int, default=None
166 The maximum number of leaves for each tree. If None, there is no
167 maximum limit.
168 max_depth : int, default=None
169 The maximum depth of each tree. The depth of a tree is the number of
170 edges to go from the root to the deepest leaf.
171 Depth isn't constrained by default.
172 min_samples_leaf : int, default=20
173 The minimum number of samples per leaf.
174 min_gain_to_split : float, default=0.
175 The minimum gain needed to split a node. Splits with lower gain will
176 be ignored.
177 min_hessian_to_split : float, default=1e-3
178 The minimum sum of hessians needed in each node. Splits that result in
179 at least one child having a sum of hessians less than
180 ``min_hessian_to_split`` are discarded.
181 n_bins : int, default=256
182 The total number of bins, including the bin for missing values. Used
183 to define the shape of the histograms.
184 n_bins_non_missing : ndarray, dtype=np.uint32, default=None
185 For each feature, gives the number of bins actually used for
186 non-missing values. For features with a lot of unique values, this
187 is equal to ``n_bins - 1``. If it's an int, all features are
188 considered to have the same number of bins. If None, all features
189 are considered to have ``n_bins - 1`` bins.
190 has_missing_values : bool or ndarray, dtype=bool, default=False
191 Whether each feature contains missing values (in the training data).
192 If it's a bool, the same value is used for all features.
193 is_categorical : ndarray of bool of shape (n_features,), default=None
194 Indicates categorical features.
195 monotonic_cst : array-like of int of shape (n_features,), dtype=int, default=None
196 Indicates the monotonic constraint to enforce on each feature.
197 - 1: monotonic increase
198 - 0: no constraint
199 - -1: monotonic decrease
200
201 Read more in the :ref:`User Guide <monotonic_cst_gbdt>`.
202 interaction_cst : list of sets of integers, default=None
203 List of interaction constraints.
204 l2_regularization : float, default=0.
205 The L2 regularization parameter penalizing leaves with small hessians.
206 Use ``0`` for no regularization (default).
207 feature_fraction_per_split : float, default=1
208 Proportion of randomly chosen features in each and every node split.
209 This is a form of regularization, smaller values make the trees weaker
210 learners and might prevent overfitting.
211 rng : Generator
212 Numpy random Generator used for feature subsampling.
213 shrinkage : float, default=1.
214 The shrinkage parameter to apply to the leaves values, also known as
215 learning rate.
216 n_threads : int, default=None
217 Number of OpenMP threads to use. `_openmp_effective_n_threads` is called
218 to determine the effective number of threads use, which takes cgroups CPU
219 quotes into account. See the docstring of `_openmp_effective_n_threads`
220 for details.
221
222 Attributes
223 ----------
224 histogram_builder : HistogramBuilder
225 splitter : Splitter
226 root : TreeNode
227 finalized_leaves : list of TreeNode
228 splittable_nodes : list of TreeNode
229 missing_values_bin_idx : int
230 Equals n_bins - 1
231 n_categorical_splits : int
232 n_features : int
233 n_nodes : int
234 total_find_split_time : float
235 Time spent finding the best splits
236 total_compute_hist_time : float
237 Time spent computing histograms
238 total_apply_split_time : float
239 Time spent splitting nodes
240 with_monotonic_cst : bool
241 Whether there are monotonic constraints that apply. False iff monotonic_cst is
242 None.
243 """

384 def grow(self):
385 """Grow the tree, from root to leaves."""
386 while self.splittable_nodes:
387 self.split_next()
388
389 self._apply_shrinkage()

391 def _apply_shrinkage(self):
392 """Multiply leaves values by shrinkage parameter.
393
394 This must be done at the very end of the growing process. If this were
395 done during the growing process e.g. in finalize_leaf(), then a leaf
396 would be shrunk but its sibling would potentially not be (if it's a
397 non-leaf), which would lead to a wrong computation of the 'middle'
398 value needed to enforce the monotonic constraints.
399 """
400 for leaf in self.finalized_leaves:
401 leaf.value *= self.shrinkage

403 def _initialize_root(self):
404 """Initialize root node and finalize it if needed."""
405 tic = time()
406 if self.interaction_cst is not None:

418 # arbitrary feature histogram. This way we replace a loop over n_samples by a
419 # loop over n_bins.

421 self.splitter.partition, # =self.root.sample_indices

462 def _compute_best_split_and_push(self, node):
463 """Compute the best possible split (SplitInfo) of a given node.
464
465 Also push it in the heap of splittable nodes if gain isn't zero.
466 The gain of a node is 0 if either all the leaves are pure
467 (best gain = 0), or if no split would satisfy the constraints,
468 (min_hessians_to_split, min_gain_to_split, min_samples_leaf)
469 """
470
471 node.split_info = self.splitter.find_node_split(
472 n_samples=node.n_samples,
473 histograms=node.histograms,
474 sum_gradients=node.sum_gradients,
475 sum_hessians=node.sum_hessians,
476 value=node.value,
477 lower_bound=node.children_lower_bound,
478 upper_bound=node.children_upper_bound,
479 allowed_features=node.allowed_features,
480 )
481
482 if node.split_info.gain <= 0: # no valid split
483 self._finalize_leaf(node)
484 else:

487 def split_next(self):
488 """Split the node with highest potential gain.
489
490 Returns
491 -------
492 left : TreeNode
493 The resulting left child.
494 right : TreeNode
495 The resulting right child.
496 """
497 # Consider the node with the highest loss reduction (a.k.a. gain)
498 node = heappop(self.splittable_nodes)
499
500 tic = time()
501 (
502 sample_indices_left,
503 sample_indices_right,
504 right_child_pos,
505 ) = self.splitter.split_indices(node.split_info, node.sample_indices)
506 self.total_apply_split_time += time() - tic
507
508 depth = node.depth + 1
509 n_leaf_nodes = len(self.finalized_leaves) + len(self.splittable_nodes)
510 n_leaf_nodes += 2
511
512 left_child_node = TreeNode(
513 depth=depth,
514 sample_indices=sample_indices_left,
515 partition_start=node.partition_start,
516 partition_stop=node.partition_start + right_child_pos,
517 sum_gradients=node.split_info.sum_gradient_left,
518 sum_hessians=node.split_info.sum_hessian_left,
519 value=node.split_info.value_left,
520 )
521 right_child_node = TreeNode(
522 depth=depth,
523 sample_indices=sample_indices_right,
524 partition_start=left_child_node.partition_stop,
525 partition_stop=node.partition_stop,
526 sum_gradients=node.split_info.sum_gradient_right,
527 sum_hessians=node.split_info.sum_hessian_right,
528 value=node.split_info.value_right,
529 )
530
531 node.right_child = right_child_node
532 node.left_child = left_child_node
533
534 # set interaction constraints (the indices of the constraints sets)
535 if self.interaction_cst is not None:
536 # Calculate allowed_features and interaction_cst_indices only once. Child
537 # nodes inherit them before they get split.
538 (
539 left_child_node.allowed_features,
540 left_child_node.interaction_cst_indices,
541 ) = self._compute_interactions(node)
542 right_child_node.interaction_cst_indices = (
543 left_child_node.interaction_cst_indices
544 )
545 right_child_node.allowed_features = left_child_node.allowed_features
546
547 if not self.has_missing_values[node.split_info.feature_idx]:
548 # If no missing values are encountered at fit time, then samples
549 # with missing values during predict() will go to whichever child
550 # has the most samples.
551 node.split_info.missing_go_to_left = (
552 left_child_node.n_samples > right_child_node.n_samples
553 )
554
555 self.n_nodes += 2
556 self.n_categorical_splits += node.split_info.is_categorical
557
558 if self.max_leaf_nodes is not None and n_leaf_nodes == self.max_leaf_nodes:
559 self._finalize_leaf(left_child_node)
560 self._finalize_leaf(right_child_node)
561 self._finalize_splittable_nodes()
562 return left_child_node, right_child_node
563
564 if self.max_depth is not None and depth == self.max_depth:
565 self._finalize_leaf(left_child_node)
566 self._finalize_leaf(right_child_node)
567 return left_child_node, right_child_node
568
569 if left_child_node.n_samples < self.min_samples_leaf * 2:
570 self._finalize_leaf(left_child_node)
571 if right_child_node.n_samples < self.min_samples_leaf * 2:
572 self._finalize_leaf(right_child_node)
573
574 if self.with_monotonic_cst:
575 # Set value bounds for respecting monotonic constraints
576 # See test_nodes_values() for details
577 if (
578 self.monotonic_cst[node.split_info.feature_idx]
579 == MonotonicConstraint.NO_CST
580 ):
581 lower_left = lower_right = node.children_lower_bound
582 upper_left = upper_right = node.children_upper_bound
583 else:
584 mid = (left_child_node.value + right_child_node.value) / 2
585 if (
586 self.monotonic_cst[node.split_info.feature_idx]
587 == MonotonicConstraint.POS
588 ):
589 lower_left, upper_left = node.children_lower_bound, mid
590 lower_right, upper_right = mid, node.children_upper_bound
591 else: # NEG
592 lower_left, upper_left = mid, node.children_upper_bound
593 lower_right, upper_right = node.children_lower_bound, mid
594 left_child_node.set_children_bounds(lower_left, upper_left)
595 right_child_node.set_children_bounds(lower_right, upper_right)
596
597 # Compute histograms of children, and compute their best possible split
598 # (if needed)
599 should_split_left = not left_child_node.is_leaf
600 should_split_right = not right_child_node.is_leaf
601 if should_split_left or should_split_right:
602 # We will compute the histograms of both nodes even if one of them
603 # is a leaf, since computing the second histogram is very cheap
604 # (using histogram subtraction).
605 n_samples_left = left_child_node.sample_indices.shape[0]
606 n_samples_right = right_child_node.sample_indices.shape[0]
607 if n_samples_left < n_samples_right:
608 smallest_child = left_child_node
609 largest_child = right_child_node
610 else:
611 smallest_child = right_child_node
612 largest_child = left_child_node
613
614 # We use the brute O(n_samples) method on the child that has the
615 # smallest number of samples, and the subtraction trick O(n_bins)
616 # on the other one.
617 # Note that both left and right child have the same allowed_features.
618 tic = time()
619 smallest_child.histograms = self.histogram_builder.compute_histograms_brute(
620 smallest_child.sample_indices, smallest_child.allowed_features
621 )
622 largest_child.histograms = (
623 self.histogram_builder.compute_histograms_subtraction(
624 node.histograms,
625 smallest_child.histograms,
626 smallest_child.allowed_features,
627 )
628 )
629 # node.histograms is reused in largest_child.histograms. To break cyclic
630 # memory references and help garbage collection, we set it to None.
631 node.histograms = None
632 self.total_compute_hist_time += time() - tic
633
634 tic = time()
635 if should_split_left:
636 self._compute_best_split_and_push(left_child_node)
637 if should_split_right:
638 self._compute_best_split_and_push(right_child_node)
639 self.total_find_split_time += time() - tic
640
641 # Release memory used by histograms as they are no longer needed
642 # for leaf nodes since they won't be split.
643 for child in (left_child_node, right_child_node):
644 if child.is_leaf:
645 del child.histograms
646
647 # Release memory used by histograms as they are no longer needed for
648 # internal nodes once children histograms have been computed.
649 del node.histograms
650
651 return left_child_node, right_child_node

653 def _compute_interactions(self, node):
654 r"""Compute features allowed by interactions to be inherited by child nodes.
655
656 Example: Assume constraints [{0, 1}, {1, 2}].
657 1 <- Both constraint groups could be applied from now on
658 / \
659 1 2 <- Left split still fulfills both constraint groups.
660 / \ / \ Right split at feature 2 has only group {1, 2} from now on.
661
662 LightGBM uses the same logic for overlapping groups. See
663 https://github.com/microsoft/LightGBM/issues/4481 for details.
664
665 Parameters:
666 ----------
667 node : TreeNode
668 A node that might have children. Based on its feature_idx, the interaction
669 constraints for possible child nodes are computed.
670
671 Returns
672 -------
673 allowed_features : ndarray, dtype=uint32
674 Indices of features allowed to split for children.
675 interaction_cst_indices : list of ints
676 Indices of the interaction sets that have to be applied on splits of
677 child nodes. The fewer sets the stronger the constraint as fewer sets
678 contain fewer features.
679 """
680 # Note:
681 # - Case of no interactions is already captured before function call.
682 # - This is for nodes that are already split and have a
683 # node.split_info.feature_idx.
684 allowed_features = set()
685 interaction_cst_indices = []
686 for i in node.interaction_cst_indices:
687 if node.split_info.feature_idx in self.interaction_cst[i]:
688 interaction_cst_indices.append(i)
689 allowed_features.update(self.interaction_cst[i])
690 return (
691 np.fromiter(allowed_features, dtype=np.uint32, count=len(allowed_features)),
692 interaction_cst_indices,
693 )

756 """Helper used in make_predictor to set the TreePredictor fields."""

758 node["count"] = grower_node.n_samples
759 node["depth"] = grower_node.depth
760 if grower_node.split_info is not None:
761 node["gain"] = grower_node.split_info.gain
762 else:
763 node["gain"] = -1
764
765 node["value"] = grower_node.value
766
767 if grower_node.is_leaf:
768 # Leaf node
769 node["is_leaf"] = True
770 return next_free_node_idx + 1, next_free_bitset_idx
771
772 split_info = grower_node.split_info
773 feature_idx, bin_idx = split_info.feature_idx, split_info.bin_idx
774 node["feature_idx"] = feature_idx
775 node["bin_threshold"] = bin_idx
776 node["missing_go_to_left"] = split_info.missing_go_to_left
777 node["is_categorical"] = split_info.is_categorical
778
779 if split_info.bin_idx == n_bins_non_missing[feature_idx] - 1:
780 # Split is on the last non-missing bin: it's a "split on nans".
781 # All nans go to the right, the rest go to the left.
782 # Note: for categorical splits, bin_idx is 0 and we rely on the bitset
783 node["num_threshold"] = np.inf
784 elif split_info.is_categorical:
785 categories = binning_thresholds[feature_idx]
786 node["bitset_idx"] = next_free_bitset_idx

789 raw_left_cat_bitsets[next_free_bitset_idx],
790 split_info.left_cat_bitset,
791 categories,
792 )

794 else:

797 next_free_node_idx += 1

801 predictor_nodes,
802 binned_left_cat_bitsets,

807 next_free_node_idx=next_free_node_idx,
808 next_free_bitset_idx=next_free_bitset_idx,
809 )

813 predictor_nodes,
