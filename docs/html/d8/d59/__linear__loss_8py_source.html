14def sandwich_dot(X, W):
15 """Compute the sandwich product X.T @ diag(W) @ X."""
16 # TODO: This "sandwich product" is the main computational bottleneck for solvers
17 # that use the full hessian matrix. Here, thread parallelism would pay-off the
18 # most.
19 # While a dedicated Cython routine could exploit the symmetry, it is very hard to
20 # beat BLAS GEMM, even thought the latter cannot exploit the symmetry, unless one
21 # pays the price of taking square roots and implements
22 # sqrtWX = sqrt(W)[: None] * X
23 # return sqrtWX.T @ sqrtWX
24 # which (might) detect the symmetry and use BLAS SYRK under the hood.
25 n_samples = X.shape[0]
26 if sparse.issparse(X):
27 return (
28 X.T @ sparse.dia_matrix((W, 0), shape=(n_samples, n_samples)) @ X
29 ).toarray()
30 else:
31 # np.einsum may use less memory but the following, using BLAS matrix
32 # multiplication (gemm), is by far faster.
33 WX = W[:, None] * X
34 return X.T @ WX

37class LinearModelLoss:
38 """General class for loss functions with raw_prediction = X @ coef + intercept.
39
40 Note that raw_prediction is also known as linear predictor.
41
42 The loss is the average of per sample losses and includes a term for L2
43 regularization::
44
45 loss = 1 / s_sum * sum_i s_i loss(y_i, X_i @ coef + intercept)
46 + 1/2 * l2_reg_strength * ||coef||_2^2
47
48 with sample weights s_i=1 if sample_weight=None and s_sum=sum_i s_i.
49
50 Gradient and hessian, for simplicity without intercept, are::
51
52 gradient = 1 / s_sum * X.T @ loss.gradient + l2_reg_strength * coef
53 hessian = 1 / s_sum * X.T @ diag(loss.hessian) @ X
54 + l2_reg_strength * identity
55
56 Conventions:
57 if fit_intercept:
58 n_dof = n_features + 1
59 else:
60 n_dof = n_features
61
62 if base_loss.is_multiclass:
63 coef.shape = (n_classes, n_dof) or ravelled (n_classes * n_dof,)
64 else:
65 coef.shape = (n_dof,)
66
67 The intercept term is at the end of the coef array:
68 if base_loss.is_multiclass:
69 if coef.shape (n_classes, n_dof):
70 intercept = coef[:, -1]
71 if coef.shape (n_classes * n_dof,)
72 intercept = coef[n_features::n_dof] = coef[(n_dof-1)::n_dof]
73 intercept.shape = (n_classes,)
74 else:
75 intercept = coef[-1]
76
77 Shape of gradient follows shape of coef.
78 gradient.shape = coef.shape
79
80 But hessian (to make our lives simpler) are always 2-d:
81 if base_loss.is_multiclass:
82 hessian.shape = (n_classes * n_dof, n_classes * n_dof)
83 else:
84 hessian.shape = (n_dof, n_dof)
85
86 Note: If coef has shape (n_classes * n_dof,), the 2d-array can be reconstructed as
87
88 coef.reshape((n_classes, -1), order="F")
89
90 The option order="F" makes coef[:, i] contiguous. This, in turn, makes the
91 coefficients without intercept, coef[:, :-1], contiguous and speeds up
92 matrix-vector computations.
93
94 Note: If the average loss per sample is wanted instead of the sum of the loss per
95 sample, one can simply use a rescaled sample_weight such that
96 sum(sample_weight) = 1.
97
98 Parameters
99 ----------
100 base_loss : instance of class BaseLoss from sklearn._loss.
101 fit_intercept : bool
102 """
103
104 def __init__(self, base_loss, fit_intercept):
105 self.base_loss = base_loss

108 def init_zero_coef(self, X, dtype=None):
109 """Allocate coef of correct shape with zeros.
110
111 Parameters:
112 -----------
113 X : {array-like, sparse matrix} of shape (n_samples, n_features)
114 Training data.
115 dtype : data-type, default=None
116 Overrides the data type of coef. With dtype=None, coef will have the same
117 dtype as X.
118
119 Returns
120 -------
121 coef : ndarray of shape (n_dof,) or (n_classes, n_dof)
122 Coefficients of a linear model.
123 """
124 n_features = X.shape[1]
125 n_classes = self.base_loss.n_classes
126 if self.fit_intercept:
127 n_dof = n_features + 1
128 else:
129 n_dof = n_features
130 if self.base_loss.is_multiclass:
131 coef = np.zeros_like(X, shape=(n_classes, n_dof), dtype=dtype, order="F")
132 else:
133 coef = np.zeros_like(X, shape=n_dof, dtype=dtype)
134 return coef

136 def weight_intercept(self, coef):
137 """Helper function to get coefficients and intercept.
138
139 Parameters
140 ----------
141 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
142 Coefficients of a linear model.
143 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
144 i.e. one reconstructs the 2d-array via
145 coef.reshape((n_classes, -1), order="F").
146
147 Returns
148 -------
149 weights : ndarray of shape (n_features,) or (n_classes, n_features)
150 Coefficients without intercept term.
151 intercept : float or ndarray of shape (n_classes,)
152 Intercept terms.
153 """
154 if not self.base_loss.is_multiclass:
155 if self.fit_intercept:
156 intercept = coef[-1]
157 weights = coef[:-1]
158 else:
159 intercept = 0.0
160 weights = coef
161 else:
162 # reshape to (n_classes, n_dof)
163 if coef.ndim == 1:
164 weights = coef.reshape((self.base_loss.n_classes, -1), order="F")
165 else:
166 weights = coef
167 if self.fit_intercept:
168 intercept = weights[:, -1]
169 weights = weights[:, :-1]
170 else:
171 intercept = 0.0
172
173 return weights, intercept

175 def weight_intercept_raw(self, coef, X):
176 """Helper function to get coefficients, intercept and raw_prediction.
177
178 Parameters
179 ----------
180 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
181 Coefficients of a linear model.
182 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
183 i.e. one reconstructs the 2d-array via
184 coef.reshape((n_classes, -1), order="F").
185 X : {array-like, sparse matrix} of shape (n_samples, n_features)
186 Training data.
187
188 Returns
189 -------
190 weights : ndarray of shape (n_features,) or (n_classes, n_features)
191 Coefficients without intercept term.
192 intercept : float or ndarray of shape (n_classes,)
193 Intercept terms.
194 raw_prediction : ndarray of shape (n_samples,) or \
195 (n_samples, n_classes)
196 """
197 weights, intercept = self.weight_intercept(coef)
198
199 if not self.base_loss.is_multiclass:
200 raw_prediction = X @ weights + intercept
201 else:
202 # weights has shape (n_classes, n_dof)
203 raw_prediction = X @ weights.T + intercept # ndarray, likely C-contiguous
204
205 return weights, intercept, raw_prediction

207 def l2_penalty(self, weights, l2_reg_strength):
208 """Compute L2 penalty term l2_reg_strength/2 *||w||_2^2."""
209 norm2_w = weights @ weights if weights.ndim == 1 else squared_norm(weights)
210 return 0.5 * l2_reg_strength * norm2_w

212 def loss(
213 self,
214 coef,
215 X,
216 y,
217 sample_weight=None,
218 l2_reg_strength=0.0,
219 n_threads=1,
220 raw_prediction=None,

222 """Compute the loss as weighted average over point-wise losses.
223
224 Parameters
225 ----------
226 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
227 Coefficients of a linear model.
228 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
229 i.e. one reconstructs the 2d-array via
230 coef.reshape((n_classes, -1), order="F").
231 X : {array-like, sparse matrix} of shape (n_samples, n_features)
232 Training data.
233 y : contiguous array of shape (n_samples,)
234 Observed, true target values.
235 sample_weight : None or contiguous array of shape (n_samples,), default=None
236 Sample weights.
237 l2_reg_strength : float, default=0.0
238 L2 regularization strength
239 n_threads : int, default=1
240 Number of OpenMP threads to use.
241 raw_prediction : C-contiguous array of shape (n_samples,) or array of \
242 shape (n_samples, n_classes)
243 Raw prediction values (in link space). If provided, these are used. If
244 None, then raw_prediction = X @ coef + intercept is calculated.
245
246 Returns
247 -------
248 loss : float
249 Weighted average of losses per sample, plus penalty.
250 """
251 if raw_prediction is None:
252 weights, intercept, raw_prediction = self.weight_intercept_raw(coef, X)
253 else:
254 weights, intercept = self.weight_intercept(coef)
255
256 loss = self.base_loss.loss(
257 y_true=y,
258 raw_prediction=raw_prediction,
259 sample_weight=None,
260 n_threads=n_threads,
261 )
262 loss = np.average(loss, weights=sample_weight)
263
264 return loss + self.l2_penalty(weights, l2_reg_strength)

266 def loss_gradient(
267 self,

269 X,
270 y,
271 sample_weight=None,
272 l2_reg_strength=0.0,
273 n_threads=1,
274 raw_prediction=None,
275 ):
276 """Computes the sum of loss and gradient w.r.t. coef.
277
278 Parameters
279 ----------
280 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
281 Coefficients of a linear model.
282 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
283 i.e. one reconstructs the 2d-array via
284 coef.reshape((n_classes, -1), order="F").
285 X : {array-like, sparse matrix} of shape (n_samples, n_features)
286 Training data.
287 y : contiguous array of shape (n_samples,)
288 Observed, true target values.
289 sample_weight : None or contiguous array of shape (n_samples,), default=None
290 Sample weights.
291 l2_reg_strength : float, default=0.0
292 L2 regularization strength
293 n_threads : int, default=1
294 Number of OpenMP threads to use.
295 raw_prediction : C-contiguous array of shape (n_samples,) or array of \
296 shape (n_samples, n_classes)
297 Raw prediction values (in link space). If provided, these are used. If
298 None, then raw_prediction = X @ coef + intercept is calculated.
299
300 Returns
301 -------
302 loss : float
303 Weighted average of losses per sample, plus penalty.
304
305 gradient : ndarray of shape coef.shape
306 The gradient of the loss.
307 """
308 (n_samples, n_features), n_classes = X.shape, self.base_loss.n_classes
309 n_dof = n_features + int(self.fit_intercept)
310
311 if raw_prediction is None:
312 weights, intercept, raw_prediction = self.weight_intercept_raw(coef, X)
313 else:
314 weights, intercept = self.weight_intercept(coef)
315
316 loss, grad_pointwise = self.base_loss.loss_gradient(
317 y_true=y,
318 raw_prediction=raw_prediction,
319 sample_weight=sample_weight,
320 n_threads=n_threads,
321 )
322 sw_sum = n_samples if sample_weight is None else np.sum(sample_weight)
323 loss = loss.sum() / sw_sum
324 loss += self.l2_penalty(weights, l2_reg_strength)
325
326 grad_pointwise /= sw_sum
327
328 if not self.base_loss.is_multiclass:
329 grad = np.empty_like(coef, dtype=weights.dtype)
330 grad[:n_features] = X.T @ grad_pointwise + l2_reg_strength * weights
331 if self.fit_intercept:
332 grad[-1] = grad_pointwise.sum()
333 else:
334 grad = np.empty((n_classes, n_dof), dtype=weights.dtype, order="F")
335 # grad_pointwise.shape = (n_samples, n_classes)
336 grad[:, :n_features] = grad_pointwise.T @ X + l2_reg_strength * weights
337 if self.fit_intercept:
338 grad[:, -1] = grad_pointwise.sum(axis=0)
339 if coef.ndim == 1:
340 grad = grad.ravel(order="F")
341
342 return loss, grad

417 def gradient_hessian(
418 self,
419 coef,
420 X,
421 y,
422 sample_weight=None,
423 l2_reg_strength=0.0,
424 n_threads=1,
425 gradient_out=None,
426 hessian_out=None,
427 raw_prediction=None,
428 ):
429 """Computes gradient and hessian w.r.t. coef.
430
431 Parameters
432 ----------
433 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
434 Coefficients of a linear model.
435 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
436 i.e. one reconstructs the 2d-array via
437 coef.reshape((n_classes, -1), order="F").
438 X : {array-like, sparse matrix} of shape (n_samples, n_features)
439 Training data.
440 y : contiguous array of shape (n_samples,)
441 Observed, true target values.
442 sample_weight : None or contiguous array of shape (n_samples,), default=None
443 Sample weights.
444 l2_reg_strength : float, default=0.0
445 L2 regularization strength
446 n_threads : int, default=1
447 Number of OpenMP threads to use.
448 gradient_out : None or ndarray of shape coef.shape
449 A location into which the gradient is stored. If None, a new array
450 might be created.
451 hessian_out : None or ndarray of shape (n_dof, n_dof) or \
452 (n_classes * n_dof, n_classes * n_dof)
453 A location into which the hessian is stored. If None, a new array
454 might be created.
455 raw_prediction : C-contiguous array of shape (n_samples,) or array of \
456 shape (n_samples, n_classes)
457 Raw prediction values (in link space). If provided, these are used. If
458 None, then raw_prediction = X @ coef + intercept is calculated.
459
460 Returns
461 -------
462 gradient : ndarray of shape coef.shape
463 The gradient of the loss.
464
465 hessian : ndarray of shape (n_dof, n_dof) or \
466 (n_classes, n_dof, n_dof, n_classes)
467 Hessian matrix.
468
469 hessian_warning : bool
470 True if pointwise hessian has more than 25% of its elements non-positive.
471 """
472 (n_samples, n_features), n_classes = X.shape, self.base_loss.n_classes
473 n_dof = n_features + int(self.fit_intercept)
474 if raw_prediction is None:
475 weights, intercept, raw_prediction = self.weight_intercept_raw(coef, X)
476 else:
477 weights, intercept = self.weight_intercept(coef)
478 sw_sum = n_samples if sample_weight is None else np.sum(sample_weight)
479
480 # Allocate gradient.
481 if gradient_out is None:
482 grad = np.empty_like(coef, dtype=weights.dtype, order="F")
483 elif gradient_out.shape != coef.shape:
484 raise ValueError(
485 f"gradient_out is required to have shape coef.shape = {coef.shape}; "
486 f"got {gradient_out.shape}."
487 )
488 elif self.base_loss.is_multiclass and not gradient_out.flags.f_contiguous:
489 raise ValueError("gradient_out must be F-contiguous.")
490 else:
491 grad = gradient_out
492 # Allocate hessian.
493 n = coef.size # for multinomial this equals n_dof * n_classes
494 if hessian_out is None:
495 hess = np.empty((n, n), dtype=weights.dtype)
496 elif hessian_out.shape != (n, n):
497 raise ValueError(
498 f"hessian_out is required to have shape ({n, n}); got "
499 f"{hessian_out.shape=}."
500 )
501 elif self.base_loss.is_multiclass and (
502 not hessian_out.flags.c_contiguous and not hessian_out.flags.f_contiguous
503 ):
504 raise ValueError("hessian_out must be contiguous.")
505 else:
506 hess = hessian_out
507
508 if not self.base_loss.is_multiclass:
509 grad_pointwise, hess_pointwise = self.base_loss.gradient_hessian(
510 y_true=y,
511 raw_prediction=raw_prediction,
512 sample_weight=sample_weight,
513 n_threads=n_threads,
514 )
515 grad_pointwise /= sw_sum
516 hess_pointwise /= sw_sum
517
518 # For non-canonical link functions and far away from the optimum, the
519 # pointwise hessian can be negative. We take care that 75% of the hessian
520 # entries are positive.
521 hessian_warning = (
522 np.average(hess_pointwise <= 0, weights=sample_weight) > 0.25
523 )
524 hess_pointwise = np.abs(hess_pointwise)
525
526 grad[:n_features] = X.T @ grad_pointwise + l2_reg_strength * weights
527 if self.fit_intercept:
528 grad[-1] = grad_pointwise.sum()
529
530 if hessian_warning:
531 # Exit early without computing the hessian.
532 return grad, hess, hessian_warning
533
534 hess[:n_features, :n_features] = sandwich_dot(X, hess_pointwise)
535
536 if l2_reg_strength > 0:
537 # The L2 penalty enters the Hessian on the diagonal only. To add those
538 # terms, we use a flattened view of the array.
539 order = "C" if hess.flags.c_contiguous else "F"
540 hess.reshape(-1, order=order)[: (n_features * n_dof) : (n_dof + 1)] += (
541 l2_reg_strength
542 )
543
544 if self.fit_intercept:
545 # With intercept included as added column to X, the hessian becomes
546 # hess = (X, 1)' @ diag(h) @ (X, 1)
547 # = (X' @ diag(h) @ X, X' @ h)
548 # ( h @ X, sum(h))
549 # The left upper part has already been filled, it remains to compute
550 # the last row and the last column.
551 Xh = X.T @ hess_pointwise
552 hess[:-1, -1] = Xh
553 hess[-1, :-1] = Xh
554 hess[-1, -1] = hess_pointwise.sum()
555 else:
556 # Here we may safely assume HalfMultinomialLoss aka categorical
557 # cross-entropy.
558 # HalfMultinomialLoss computes only the diagonal part of the hessian, i.e.
559 # diagonal in the classes. Here, we want the full hessian. Therefore, we
560 # call gradient_proba.
561 grad_pointwise, proba = self.base_loss.gradient_proba(
562 y_true=y,
563 raw_prediction=raw_prediction,
564 sample_weight=sample_weight,
565 n_threads=n_threads,
566 )
567 grad_pointwise /= sw_sum
568 grad = grad.reshape((n_classes, n_dof), order="F")
569 grad[:, :n_features] = grad_pointwise.T @ X + l2_reg_strength * weights
570 if self.fit_intercept:
571 grad[:, -1] = grad_pointwise.sum(axis=0)
572 if coef.ndim == 1:
573 grad = grad.ravel(order="F")
574
575 # The full hessian matrix, i.e. not only the diagonal part, dropping most
576 # indices, is given by:
577 #
578 # hess = X' @ h @ X
579 #
580 # Here, h is a priori a 4-dimensional matrix of shape
581 # (n_samples, n_samples, n_classes, n_classes). It is diagonal its first
582 # two dimensions (the ones with n_samples), i.e. it is
583 # effectively a 3-dimensional matrix (n_samples, n_classes, n_classes).
584 #
585 # h = diag(p) - p' p
586 #
587 # or with indices k and l for classes
588 #
589 # h_kl = p_k * delta_kl - p_k * p_l
590 #
591 # with p_k the (predicted) probability for class k. Only the dimension in
592 # n_samples multiplies with X.
593 # For 3 classes and n_samples = 1, this looks like ("@" is a bit misused
594 # here):
595 #
596 # hess = X' @ (h00 h10 h20) @ X
597 # (h10 h11 h12)
598 # (h20 h12 h22)
599 # = (X' @ diag(h00) @ X, X' @ diag(h10), X' @ diag(h20))
600 # (X' @ diag(h10) @ X, X' @ diag(h11), X' @ diag(h12))
601 # (X' @ diag(h20) @ X, X' @ diag(h12), X' @ diag(h22))
602 #
603 # Now coef of shape (n_classes * n_dof) is contiguous in n_classes.
604 # Therefore, we want the hessian to follow this convention, too, i.e.
605 # hess[:n_classes, :n_classes] = (x0' @ h00 @ x0, x0' @ h10 @ x0, ..)
606 # (x0' @ h10 @ x0, x0' @ h11 @ x0, ..)
607 # (x0' @ h20 @ x0, x0' @ h12 @ x0, ..)
608 # is the first feature, x0, for all classes. In our implementation, we
609 # still want to take advantage of BLAS "X.T @ X". Therefore, we have some
610 # index/slicing battle to fight.
611 if sample_weight is not None:
612 sw = sample_weight / sw_sum
613 else:
614 sw = 1.0 / sw_sum
615
616 for k in range(n_classes):
617 # Diagonal terms (in classes) hess_kk.
618 # Note that this also writes to some of the lower triangular part.
619 h = proba[:, k] * (1 - proba[:, k]) * sw
620 hess[
621 k : n_classes * n_features : n_classes,
622 k : n_classes * n_features : n_classes,
623 ] = sandwich_dot(X, h)
624 if self.fit_intercept:
625 # See above in the non multiclass case.
626 Xh = X.T @ h
627 hess[
628 k : n_classes * n_features : n_classes,
629 n_classes * n_features + k,
630 ] = Xh
631 hess[
632 n_classes * n_features + k,
633 k : n_classes * n_features : n_classes,
634 ] = Xh
635 hess[n_classes * n_features + k, n_classes * n_features + k] = (
636 h.sum()
637 )
638 # Off diagonal terms (in classes) hess_kl.
639 for l in range(k + 1, n_classes):
640 # Upper triangle (in classes).
641 h = -proba[:, k] * proba[:, l] * sw
642 hess[
643 k : n_classes * n_features : n_classes,
644 l : n_classes * n_features : n_classes,
645 ] = sandwich_dot(X, h)
646 if self.fit_intercept:
647 Xh = X.T @ h
648 hess[
649 k : n_classes * n_features : n_classes,
650 n_classes * n_features + l,
651 ] = Xh
652 hess[
653 n_classes * n_features + k,
654 l : n_classes * n_features : n_classes,
655 ] = Xh
656 hess[n_classes * n_features + k, n_classes * n_features + l] = (
657 h.sum()
658 )
659 # Fill lower triangle (in classes).
660 hess[l::n_classes, k::n_classes] = hess[k::n_classes, l::n_classes]
661
662 if l2_reg_strength > 0:
663 # See above in the non multiclass case.
664 order = "C" if hess.flags.c_contiguous else "F"
665 hess.reshape(-1, order=order)[
666 : (n_classes**2 * n_features * n_dof) : (n_classes * n_dof + 1)
667 ] += l2_reg_strength
668
669 # The pointwise hessian is always non-negative for the multinomial loss.
670 hessian_warning = False
671
672 return grad, hess, hessian_warning

674 def gradient_hessian_product(
675 self, coef, X, y, sample_weight=None, l2_reg_strength=0.0, n_threads=1
676 ):
677 """Computes gradient and hessp (hessian product function) w.r.t. coef.
678
679 Parameters
680 ----------
681 coef : ndarray of shape (n_dof,), (n_classes, n_dof) or (n_classes * n_dof,)
682 Coefficients of a linear model.
683 If shape (n_classes * n_dof,), the classes of one feature are contiguous,
684 i.e. one reconstructs the 2d-array via
685 coef.reshape((n_classes, -1), order="F").
686 X : {array-like, sparse matrix} of shape (n_samples, n_features)
687 Training data.
688 y : contiguous array of shape (n_samples,)
689 Observed, true target values.
690 sample_weight : None or contiguous array of shape (n_samples,), default=None
691 Sample weights.
692 l2_reg_strength : float, default=0.0
693 L2 regularization strength
694 n_threads : int, default=1
695 Number of OpenMP threads to use.
696
697 Returns
698 -------
699 gradient : ndarray of shape coef.shape
700 The gradient of the loss.
701
702 hessp : callable
703 Function that takes in a vector input of shape of gradient and
704 and returns matrix-vector product with hessian.
705 """
706 (n_samples, n_features), n_classes = X.shape, self.base_loss.n_classes
707 n_dof = n_features + int(self.fit_intercept)
708 weights, intercept, raw_prediction = self.weight_intercept_raw(coef, X)
709 sw_sum = n_samples if sample_weight is None else np.sum(sample_weight)
710
711 if not self.base_loss.is_multiclass:
712 grad_pointwise, hess_pointwise = self.base_loss.gradient_hessian(
713 y_true=y,
714 raw_prediction=raw_prediction,
715 sample_weight=sample_weight,
716 n_threads=n_threads,
717 )
718 grad_pointwise /= sw_sum
719 hess_pointwise /= sw_sum
720 grad = np.empty_like(coef, dtype=weights.dtype)
721 grad[:n_features] = X.T @ grad_pointwise + l2_reg_strength * weights
722 if self.fit_intercept:
723 grad[-1] = grad_pointwise.sum()
724
725 # Precompute as much as possible: hX, hX_sum and hessian_sum
726 hessian_sum = hess_pointwise.sum()
727 if sparse.issparse(X):
728 hX = (
729 sparse.dia_matrix((hess_pointwise, 0), shape=(n_samples, n_samples))
730 @ X
731 )
732 else:
733 hX = hess_pointwise[:, np.newaxis] * X
734
735 if self.fit_intercept:
736 # Calculate the double derivative with respect to intercept.
737 # Note: In case hX is sparse, hX.sum is a matrix object.
738 hX_sum = np.squeeze(np.asarray(hX.sum(axis=0)))
739 # prevent squeezing to zero-dim array if n_features == 1
740 hX_sum = np.atleast_1d(hX_sum)
741
742 # With intercept included and l2_reg_strength = 0, hessp returns
743 # res = (X, 1)' @ diag(h) @ (X, 1) @ s
744 # = (X, 1)' @ (hX @ s[:n_features], sum(h) * s[-1])
745 # res[:n_features] = X' @ hX @ s[:n_features] + sum(h) * s[-1]
746 # res[-1] = 1' @ hX @ s[:n_features] + sum(h) * s[-1]
747 def hessp(s):
748 ret = np.empty_like(s)
749 if sparse.issparse(X):
750 ret[:n_features] = X.T @ (hX @ s[:n_features])
751 else:
752 ret[:n_features] = np.linalg.multi_dot([X.T, hX, s[:n_features]])
753 ret[:n_features] += l2_reg_strength * s[:n_features]
754
755 if self.fit_intercept:
756 ret[:n_features] += s[-1] * hX_sum
757 ret[-1] = hX_sum @ s[:n_features] + hessian_sum * s[-1]
758 return ret
759
760 else:
761 # Here we may safely assume HalfMultinomialLoss aka categorical
762 # cross-entropy.
763 # HalfMultinomialLoss computes only the diagonal part of the hessian, i.e.
764 # diagonal in the classes. Here, we want the matrix-vector product of the
765 # full hessian. Therefore, we call gradient_proba.
766 grad_pointwise, proba = self.base_loss.gradient_proba(
767 y_true=y,
768 raw_prediction=raw_prediction,
769 sample_weight=sample_weight,
770 n_threads=n_threads,
771 )
772 grad_pointwise /= sw_sum
773 grad = np.empty((n_classes, n_dof), dtype=weights.dtype, order="F")
774 grad[:, :n_features] = grad_pointwise.T @ X + l2_reg_strength * weights
775 if self.fit_intercept:
776 grad[:, -1] = grad_pointwise.sum(axis=0)
777
778 # Full hessian-vector product, i.e. not only the diagonal part of the
779 # hessian. Derivation with some index battle for input vector s:
780 # - sample index i
781 # - feature indices j, m
782 # - class indices k, l
783 # - 1_{k=l} is one if k=l else 0
784 # - p_i_k is the (predicted) probability that sample i belongs to class k
785 # for all i: sum_k p_i_k = 1
786 # - s_l_m is input vector for class l and feature m
787 # - X' = X transposed
788 #
789 # Note: Hessian with dropping most indices is just:
790 # X' @ p_k (1(k=l) - p_l) @ X
791 #
792 # result_{k j} = sum_{i, l, m} Hessian_{i, k j, m l} * s_l_m
793 # = sum_{i, l, m} (X')_{ji} * p_i_k * (1_{k=l} - p_i_l)
794 # * X_{im} s_l_m
795 # = sum_{i, m} (X')_{ji} * p_i_k
796 # * (X_{im} * s_k_m - sum_l p_i_l * X_{im} * s_l_m)
797 #
798 # See also https://github.com/scikit-learn/scikit-learn/pull/3646#discussion_r17461411
799 def hessp(s):
800 s = s.reshape((n_classes, -1), order="F") # shape = (n_classes, n_dof)
801 if self.fit_intercept:
802 s_intercept = s[:, -1]
803 s = s[:, :-1] # shape = (n_classes, n_features)
804 else:
805 s_intercept = 0
806 tmp = X @ s.T + s_intercept # X_{im} * s_k_m
807 tmp += (-proba * tmp).sum(axis=1)[:, np.newaxis] # - sum_l ..
808 tmp *= proba # * p_i_k
809 if sample_weight is not None:
810 tmp *= sample_weight[:, np.newaxis]
811 # hess_prod = empty_like(grad), but we ravel grad below and this
812 # function is run after that.
813 hess_prod = np.empty((n_classes, n_dof), dtype=weights.dtype, order="F")
814 hess_prod[:, :n_features] = (tmp.T @ X) / sw_sum + l2_reg_strength * s
815 if self.fit_intercept:
816 hess_prod[:, -1] = tmp.sum(axis=0) / sw_sum
817 if coef.ndim == 1:
818 return hess_prod.ravel(order="F")
819 else:
820 return hess_prod
821
822 if coef.ndim == 1:
823 return grad.ravel(order="F"), hessp
824
825 return grad, hessp
