30def _calculate_threshold(estimator, importances, threshold):
31 """Interpret the threshold value"""
32
33 if threshold is None:
34 # determine default from estimator
35 est_name = estimator.__class__.__name__
36 is_l1_penalized = hasattr(estimator, "penalty") and estimator.penalty == "l1"
37 is_lasso = "Lasso" in est_name
38 is_elasticnet_l1_penalized = est_name == "ElasticNet" and (
39 hasattr(estimator, "l1_ratio") and np.isclose(estimator.l1_ratio, 1.0)
40 )
41 is_elasticnetcv_l1_penalized = est_name == "ElasticNetCV" and (
42 hasattr(estimator, "l1_ratio_") and np.isclose(estimator.l1_ratio_, 1.0)
43 )
44 if (
45 is_l1_penalized
46 or is_lasso
47 or is_elasticnet_l1_penalized
48 or is_elasticnetcv_l1_penalized
49 ):
50 # the natural default threshold is 0 when l1 penalty was used
51 threshold = 1e-5
52 else:
53 threshold = "mean"
54
55 if isinstance(threshold, str):
56 if "*" in threshold:
57 scale, reference = threshold.split("*")
58 scale = float(scale.strip())
59 reference = reference.strip()
60
61 if reference == "median":
62 reference = np.median(importances)
63 elif reference == "mean":
64 reference = np.mean(importances)
65 else:
66 raise ValueError("Unknown reference: " + reference)
67
68 threshold = scale * reference
69
70 elif threshold == "median":
71 threshold = np.median(importances)
72
73 elif threshold == "mean":
74 threshold = np.mean(importances)
75
76 else:
77 raise ValueError(
78 "Expected threshold='mean' or threshold='median' got %s" % threshold
79 )
80
81 else:
82 threshold = float(threshold)
83
84 return threshold

87class SelectFromModel(MetaEstimatorMixin, SelectorMixin, BaseEstimator):
88 """Meta-transformer for selecting features based on importance weights.
89
90 .. versionadded:: 0.17
91
92 Read more in the :ref:`User Guide <select_from_model>`.
93
94 Parameters
95 ----------
96 estimator : object
97 The base estimator from which the transformer is built.
98 This can be both a fitted (if ``prefit`` is set to True)
99 or a non-fitted estimator. The estimator should have a
100 ``feature_importances_`` or ``coef_`` attribute after fitting.
101 Otherwise, the ``importance_getter`` parameter should be used.
102
103 threshold : str or float, default=None
104 The threshold value to use for feature selection. Features whose
105 absolute importance value is greater or equal are kept while the others
106 are discarded. If "median" (resp. "mean"), then the ``threshold`` value
107 is the median (resp. the mean) of the feature importances. A scaling
108 factor (e.g., "1.25*mean") may also be used. If None and if the
109 estimator has a parameter penalty set to l1, either explicitly
110 or implicitly (e.g, Lasso), the threshold used is 1e-5.
111 Otherwise, "mean" is used by default.
112
113 prefit : bool, default=False
114 Whether a prefit model is expected to be passed into the constructor
115 directly or not.
116 If `True`, `estimator` must be a fitted estimator.
117 If `False`, `estimator` is fitted and updated by calling
118 `fit` and `partial_fit`, respectively.
119
120 norm_order : non-zero int, inf, -inf, default=1
121 Order of the norm used to filter the vectors of coefficients below
122 ``threshold`` in the case where the ``coef_`` attribute of the
123 estimator is of dimension 2.
124
125 max_features : int, callable, default=None
126 The maximum number of features to select.
127
128 - If an integer, then it specifies the maximum number of features to
129 allow.
130 - If a callable, then it specifies how to calculate the maximum number of
131 features allowed by using the output of `max_features(X)`.
132 - If `None`, then all features are kept.
133
134 To only select based on ``max_features``, set ``threshold=-np.inf``.
135
136 .. versionadded:: 0.20
137 .. versionchanged:: 1.1
138 `max_features` accepts a callable.
139
140 importance_getter : str or callable, default='auto'
141 If 'auto', uses the feature importance either through a ``coef_``
142 attribute or ``feature_importances_`` attribute of estimator.
143
144 Also accepts a string that specifies an attribute name/path
145 for extracting feature importance (implemented with `attrgetter`).
146 For example, give `regressor_.coef_` in case of
147 :class:`~sklearn.compose.TransformedTargetRegressor` or
148 `named_steps.clf.feature_importances_` in case of
149 :class:`~sklearn.pipeline.Pipeline` with its last step named `clf`.
150
151 If `callable`, overrides the default feature importance getter.
152 The callable is passed with the fitted estimator and it should
153 return importance for each feature.
154
155 .. versionadded:: 0.24
156
157 Attributes
158 ----------
159 estimator_ : estimator
160 The base estimator from which the transformer is built. This attribute
161 exist only when `fit` has been called.
162
163 - If `prefit=True`, it is a deep copy of `estimator`.
164 - If `prefit=False`, it is a clone of `estimator` and fit on the data
165 passed to `fit` or `partial_fit`.
166
167 n_features_in_ : int
168 Number of features seen during :term:`fit`. Only defined if the
169 underlying estimator exposes such an attribute when fit.
170
171 .. versionadded:: 0.24
172
173 max_features_ : int
174 Maximum number of features calculated during :term:`fit`. Only defined
175 if the ``max_features`` is not `None`.
176
177 - If `max_features` is an `int`, then `max_features_ = max_features`.
178 - If `max_features` is a callable, then `max_features_ = max_features(X)`.
179
180 .. versionadded:: 1.1
181
182 feature_names_in_ : ndarray of shape (`n_features_in_`,)
183 Names of features seen during :term:`fit`. Defined only when `X`
184 has feature names that are all strings.
185
186 .. versionadded:: 1.0
187
188 threshold_ : float
189 The threshold value used for feature selection.
190
191 See Also
192 --------
193 RFE : Recursive feature elimination based on importance weights.
194 RFECV : Recursive feature elimination with built-in cross-validated
195 selection of the best number of features.
196 SequentialFeatureSelector : Sequential cross-validation based feature
197 selection. Does not rely on importance weights.
198
199 Notes
200 -----
201 Allows NaN/Inf in the input if the underlying estimator does as well.
202
203 Examples
204 --------
205 >>> from sklearn.feature_selection import SelectFromModel
206 >>> from sklearn.linear_model import LogisticRegression
207 >>> X = [[ 0.87, -1.34, 0.31 ],
208 ... [-2.79, -0.02, -0.85 ],
209 ... [-1.34, -0.48, -2.55 ],
210 ... [ 1.92, 1.48, 0.65 ]]
211 >>> y = [0, 1, 0, 1]
212 >>> selector = SelectFromModel(estimator=LogisticRegression()).fit(X, y)
213 >>> selector.estimator_.coef_
214 array([[-0.3252, 0.8345, 0.4976]])
215 >>> selector.threshold_
216 np.float64(0.55249)
217 >>> selector.get_support()
218 array([False, True, False])
219 >>> selector.transform(X)
220 array([[-1.34],
221 [-0.02],
222 [-0.48],
223 [ 1.48]])
224
225 Using a callable to create a selector that can use no more than half
226 of the input features.
227
228 >>> def half_callable(X):
229 ... return round(len(X[0]) / 2)
230 >>> half_selector = SelectFromModel(estimator=LogisticRegression(),
231 ... max_features=half_callable)
232 >>> _ = half_selector.fit(X, y)
233 >>> half_selector.max_features_
234 2
235 """

267 def _get_support_mask(self):
268 estimator = getattr(self, "estimator_", self.estimator)
269 max_features = getattr(self, "max_features_", self.max_features)
270
271 if self.prefit:
272 try:
273 check_is_fitted(self.estimator)
274 except NotFittedError as exc:
275 raise NotFittedError(
276 "When `prefit=True`, `estimator` is expected to be a fitted "
277 "estimator."
278 ) from exc
279 if callable(max_features):
280 # This branch is executed when `transform` is called directly and thus
281 # `max_features_` is not set and we fallback using `self.max_features`
282 # that is not validated
283 raise NotFittedError(
284 "When `prefit=True` and `max_features` is a callable, call `fit` "
285 "before calling `transform`."
286 )
287 elif max_features is not None and not isinstance(max_features, Integral):
288 raise ValueError(
289 f"`max_features` must be an integer. Got `max_features={max_features}` "
290 "instead."
291 )
292
293 scores = _get_feature_importances(
294 estimator=estimator,
295 getter=self.importance_getter,
296 transform_func="norm",
297 norm_order=self.norm_order,
298 )
299 threshold = _calculate_threshold(estimator, scores, self.threshold)
300 if self.max_features is not None:
301 mask = np.zeros_like(scores, dtype=bool)
302 candidate_indices = np.argsort(-scores, kind="mergesort")[:max_features]
303 mask[candidate_indices] = True
304 else:
305 mask = np.ones_like(scores, dtype=bool)
306 mask[scores < threshold] = False
307 return mask

332 """Fit the SelectFromModel meta-transformer.
333
334 Parameters
335 ----------
336 X : array-like of shape (n_samples, n_features)
337 The training input samples.
338
339 y : array-like of shape (n_samples,), default=None
340 The target values (integers that correspond to classes in
341 classification, real numbers in regression).
342
343 **fit_params : dict
344 - If `enable_metadata_routing=False` (default): Parameters directly passed
345 to the `fit` method of the sub-estimator. They are ignored if
346 `prefit=True`.
347
348 - If `enable_metadata_routing=True`: Parameters safely routed to the `fit`
349 method of the sub-estimator. They are ignored if `prefit=True`.
350
351 .. versionchanged:: 1.4
352 See :ref:`Metadata Routing User Guide <metadata_routing>` for
353 more details.
354
355 Returns
356 -------
357 self : object
358 Fitted estimator.
359 """
360 self._check_max_features(X)
361
362 if self.prefit:
363 try:
364 check_is_fitted(self.estimator)
365 except NotFittedError as exc:
366 raise NotFittedError(
367 "When `prefit=True`, `estimator` is expected to be a fitted "
368 "estimator."
369 ) from exc
370 self.estimator_ = deepcopy(self.estimator)
371 else:
372 if _routing_enabled():
373 routed_params = process_routing(self, "fit", **fit_params)

382 self.feature_names_in_ = self.estimator_.feature_names_in_
