15def assert_identical(a, b):
16 """Assert whether value AND type are the same"""
17 assert_equal(a, b)
18 if type(b) is str:
19 assert_equal(type(a), type(b))
20 else:
21 assert_equal(np.asarray(a).dtype.type, np.asarray(b).dtype.type)
