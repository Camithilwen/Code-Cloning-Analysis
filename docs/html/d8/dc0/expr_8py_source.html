51def _rewrite_assign(tok: tuple[int, str]) -> tuple[int, str]:
52 """
53 Rewrite the assignment operator for PyTables expressions that use ``=``
54 as a substitute for ``==``.
55
56 Parameters
57 ----------
58 tok : tuple of int, str
59 ints correspond to the all caps constants in the tokenize module
60
61 Returns
62 -------
63 tuple of int, str
64 Either the input or token or the replacement values

66 toknum, tokval = tok

70def _replace_booleans(tok: tuple[int, str]) -> tuple[int, str]:
71 """
72 Replace ``&`` with ``and`` and ``|`` with ``or`` so that bitwise
73 precedence is changed to boolean precedence.
74
75 Parameters
76 ----------
77 tok : tuple of int, str
78 ints correspond to the all caps constants in the tokenize module
79
80 Returns
81 -------
82 tuple of int, str
83 Either the input or token or the replacement values
84 """
85 toknum, tokval = tok
86 if toknum == tokenize.OP:
87 if tokval == "&":
88 return tokenize.NAME, "and"
89 elif tokval == "|":
90 return tokenize.NAME, "or"
91 return toknum, tokval
92 return toknum, tokval

95def _replace_locals(tok: tuple[int, str]) -> tuple[int, str]:
96 """
97 Replace local variables with a syntactically valid name.
98
99 Parameters
100 ----------
101 tok : tuple of int, str
102 ints correspond to the all caps constants in the tokenize module
103
104 Returns
105 -------
106 tuple of int, str
107 Either the input or token or the replacement values
108
109 Notes
110 -----
111 This is somewhat of a hack in that we rewrite a string such as ``'@a'`` as
112 ``'__pd_eval_local_a'`` by telling the tokenizer that ``__pd_eval_local_``
113 is a ``tokenize.OP`` and to replace the ``'@'`` symbol with it.
114 """
115 toknum, tokval = tok
116 if toknum == tokenize.OP and tokval == "@":
117 return tokenize.OP, LOCAL_TAG
118 return toknum, tokval

136def _preparse(
137 source: str,
138 f=_compose(
139 _replace_locals, _replace_booleans, _rewrite_assign, clean_backtick_quoted_toks
140 ),
141) -> str:
142 """
143 Compose a collection of tokenization functions.
144
145 Parameters
146 ----------
147 source : str
148 A Python source code string
149 f : callable
150 This takes a tuple of (toknum, tokval) as its argument and returns a
151 tuple with the same structure but possibly different elements. Defaults
152 to the composition of ``_rewrite_assign``, ``_replace_booleans``, and
153 ``_replace_locals``.
154
155 Returns
156 -------
157 str
158 Valid Python source code
159
160 Notes
161 -----
162 The `f` parameter can be any callable that takes *and* returns input of the
163 form ``(toknum, tokval)``, where ``toknum`` is one of the constants from
164 the ``tokenize`` module and ``tokval`` is a string.
165 """
166 assert callable(f), "f must be callable"
167 return tokenize.untokenize(f(x) for x in tokenize_string(source))

271def disallow(nodes: set[str]) -> Callable[[type[_T]], type[_T]]:
272 """
273 Decorator to disallow certain nodes from parsing. Raises a
274 NotImplementedError instead.
275
276 Returns
277 -------
278 callable
279 """
280
281 def disallowed(cls: type[_T]) -> type[_T]:
282 # error: "Type[_T]" has no attribute "unsupported_nodes"
283 cls.unsupported_nodes = () # type: ignore[attr-defined]
284 for node in nodes:
285 new_method = _node_not_implemented(node)
286 name = f"visit_{node}"
287 # error: "Type[_T]" has no attribute "unsupported_nodes"
288 cls.unsupported_nodes += (name,) # type: ignore[attr-defined]
289 setattr(cls, name, new_method)
290 return cls
291
292 return disallowed

320def add_ops(op_classes):
321 """
322 Decorator to add default implementation of ops.
323 """
324
325 def f(cls):
326 for op_attr_name, op_class in op_classes.items():
327 ops = getattr(cls, f"{op_attr_name}_ops")
328 ops_map = getattr(cls, f"{op_attr_name}_op_nodes_map")

343 Custom ast walker. Parsers of other engines should subclass this class

394 def __init__(self, env, engine, parser, preparser=_preparse) -> None:
395 self.env = env
396 self.engine = engine
397 self.parser = parser

401 def visit(self, node, **kwargs):
402 if isinstance(node, str):
403 clean = self.preparser(node)
404 try:
405 node = ast.fix_missing_locations(ast.parse(clean))
406 except SyntaxError as e:
407 if any(iskeyword(x) for x in clean.split()):
408 e.msg = "Python keyword not valid identifier in numexpr query"
409 raise e

412 visitor = getattr(self, method)
413 return visitor(node, **kwargs)

424 def _rewrite_membership_op(self, node, left, right):
425 # the kind of the operator (is actually an instance)
426 op_instance = node.op
427 op_type = type(op_instance)
428
429 # must be two terms and the comparison operator must be ==/!=/in/not in
430 if is_term(left) and is_term(right) and op_type in self.rewrite_map:
431 left_list, right_list = map(_is_list, (left, right))
432 left_str, right_str = map(_is_str, (left, right))
433
434 # if there are any strings or lists in the expression
435 if left_list or right_list or left_str or right_str:
436 op_instance = self.rewrite_map[op_type]()
437
438 # pop the string variable out of locals and replace it with a list
439 # of one string, kind of a hack
440 if right_str:
441 name = self.env.add_tmp([right.value])
442 right = self.term_type(name, self.env)
443
444 if left_str:
445 name = self.env.add_tmp([left.value])
446 left = self.term_type(name, self.env)
447
448 op = self.visit(op_instance)
449 return op, op_instance, left, right

451 def _maybe_transform_eq_ne(self, node, left=None, right=None):
452 if left is None:
453 left = self.visit(node.left, side="left")
454 if right is None:
455 right = self.visit(node.right, side="right")
456 op, op_class, left, right = self._rewrite_membership_op(node, left, right)
457 return op, op_class, left, right

459 def _maybe_downcast_constants(self, left, right):
460 f32 = np.dtype(np.float32)
461 if (
462 left.is_scalar
463 and hasattr(left, "value")
464 and not right.is_scalar
465 and right.return_type == f32
466 ):
467 # right is a float32 array, left is a scalar
468 name = self.env.add_tmp(np.float32(left.value))
469 left = self.term_type(name, self.env)
470 if (
471 right.is_scalar
472 and hasattr(right, "value")
473 and not left.is_scalar
474 and left.return_type == f32
475 ):
476 # left is a float32 array, right is a scalar
477 name = self.env.add_tmp(np.float32(right.value))
478 right = self.term_type(name, self.env)
479
480 return left, right

482 def _maybe_eval(self, binop, eval_in_python):
483 # eval `in` and `not in` (for now) in "partial" python space
484 # things that can be evaluated in "eval" space will be turned into
485 # temporary variables. for example,
486 # [1,2] in a + 2 * b
487 # in that case a + 2 * b will be evaluated using numexpr, and the "in"
488 # call will be evaluated using isin (in python space)
489 return binop.evaluate(
490 self.env, self.engine, self.parser, self.term_type, eval_in_python
491 )

493 def _maybe_evaluate_binop(
494 self,
495 op,
496 op_class,
497 lhs,
498 rhs,
499 eval_in_python=("in", "not in"),
500 maybe_eval_in_python=("==", "!=", "<", ">", "<=", ">="),
501 ):
502 res = op(lhs, rhs)
503
504 if res.has_invalid_return_type:
505 raise TypeError(
506 f"unsupported operand type(s) for {res.op}: "
507 f"'{lhs.type}' and '{rhs.type}'"
508 )
509
510 if self.engine != "pytables" and (
511 res.op in CMP_OPS_SYMS
512 and getattr(lhs, "is_datetime", False)
513 or getattr(rhs, "is_datetime", False)
514 ):
515 # all date ops must be done in python bc numexpr doesn't work
516 # well with NaT
517 return self._maybe_eval(res, self.binary_ops)
518
519 if res.op in eval_in_python:
520 # "in"/"not in" ops are always evaluated in python
521 return self._maybe_eval(res, eval_in_python)
522 elif self.engine != "pytables":
523 if (
524 getattr(lhs, "return_type", None) == object
525 or is_string_dtype(getattr(lhs, "return_type", None))
526 or getattr(rhs, "return_type", None) == object
527 or is_string_dtype(getattr(rhs, "return_type", None))
528 ):
529 # evaluate "==" and "!=" in python if either of our operands
530 # has an object or string return type
531 return self._maybe_eval(res, eval_in_python + maybe_eval_in_python)
532 return res

607 def visit_Assign(self, node, **kwargs):
608 """
609 support a single assignment node, like
610
611 c = a + b
612
613 set the assigner at the top level, must be a Name node which
614 might or might not exist in the resolvers
615
616 """
617 if len(node.targets) != 1:
618 raise SyntaxError("can only assign a single expression")
619 if not isinstance(node.targets[0], ast.Name):
620 raise SyntaxError("left hand side of an assignment must be a single name")
621 if self.env.target is None:
622 raise ValueError("cannot assign without a target object")
623
624 try:
625 assigner = self.visit(node.targets[0], **kwargs)
626 except UndefinedVariableError:
627 assigner = node.targets[0].id
628
629 self.assigner = getattr(assigner, "name", assigner)
630 if self.assigner is None:
631 raise SyntaxError(
632 "left hand side of an assignment must be a single resolvable name"
633 )
634
635 return self.visit(node.value, **kwargs)

637 def visit_Attribute(self, node, **kwargs):
638 attr = node.attr
639 value = node.value
640
641 ctx = node.ctx
642 if isinstance(ctx, ast.Load):
643 # resolve the value
644 resolved = self.visit(value).value
645 try:
646 v = getattr(resolved, attr)
647 name = self.env.add_tmp(v)
648 return self.term_type(name, self.env)
649 except AttributeError:
650 # something like datetime.datetime where scope is overridden
651 if isinstance(value, ast.Name) and value.id == attr:
652 return resolved
653 raise
654
655 raise ValueError(f"Invalid Attribute context {type(ctx).__name__}")

657 def visit_Call(self, node, side=None, **kwargs):
658 if isinstance(node.func, ast.Attribute) and node.func.attr != "__call__":
659 res = self.visit_Attribute(node.func)
660 elif not isinstance(node.func, ast.Name):
661 raise TypeError("Only named functions are supported")
662 else:
663 try:
664 res = self.visit(node.func)
665 except UndefinedVariableError:
666 # Check if this is a supported function name
667 try:
668 res = FuncNode(node.func.id)
669 except ValueError:
670 # Raise original error
671 raise
672
673 if res is None:
674 # error: "expr" has no attribute "id"
675 raise ValueError(
676 f"Invalid function call {node.func.id}" # type: ignore[attr-defined]
677 )
678 if hasattr(res, "value"):
679 res = res.value
680
681 if isinstance(res, FuncNode):
682 new_args = [self.visit(arg) for arg in node.args]
683
684 if node.keywords:
685 raise TypeError(
686 f'Function "{res.name}" does not support keyword arguments'
687 )
688
689 return res(*new_args)
690
691 else:
692 new_args = [self.visit(arg)(self.env) for arg in node.args]
693
694 for key in node.keywords:
695 if not isinstance(key, ast.keyword):
696 # error: "expr" has no attribute "id"
697 raise ValueError(
698 "keyword error in function call "
699 f"'{node.func.id}'" # type: ignore[attr-defined]
700 )
701
702 if key.arg:
703 kwargs[key.arg] = self.visit(key.value)(self.env)
704
705 name = self.env.add_tmp(res(*new_args, **kwargs))
706 return self.term_type(name=name, env=self.env)

708 def translate_In(self, op):
709 return op

711 def visit_Compare(self, node, **kwargs):
712 ops = node.ops
713 comps = node.comparators
714
715 # base case: we have something like a CMP b
716 if len(comps) == 1:
717 op = self.translate_In(ops[0])
718 binop = ast.BinOp(op=op, left=node.left, right=comps[0])
719 return self.visit(binop)
720
721 # recursive case: we have a chained comparison, a CMP b CMP c, etc.
722 left = node.left
723 values = []
724 for op, comp in zip(ops, comps):
725 new_node = self.visit(
726 ast.Compare(comparators=[comp], left=left, ops=[self.translate_In(op)])
727 )
728 left = comp
729 values.append(new_node)
730 return self.visit(ast.BoolOp(op=ast.And(), values=values))

732 def _try_visit_binop(self, bop):
733 if isinstance(bop, (Op, Term)):
734 return bop
735 return self.visit(bop)

737 def visit_BoolOp(self, node, **kwargs):
738 def visitor(x, y):
739 lhs = self._try_visit_binop(x)
740 rhs = self._try_visit_binop(y)
741
742 op, op_class, lhs, rhs = self._maybe_transform_eq_ne(node, lhs, rhs)
743 return self._maybe_evaluate_binop(op, node.op, lhs, rhs)
744
745 operands = node.values
746 return reduce(visitor, operands)
