401 from xml.etree.ElementTree import register_namespace
402
403 uri = ""

405 for p, n in self.namespaces.items():
406 if isinstance(p, str) and isinstance(n, str):

410 uri = f"{{{self.namespaces[self.prefix]}}}"

414 uri = f'{{{self.namespaces[""]}}}'
