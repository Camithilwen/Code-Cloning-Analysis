251 def test_numeric_only(self, kernel, has_numeric_only, dtype):
252 # GH#47500
253 ser = Series([0, 1, 1], dtype=dtype)
254 if kernel == "corrwith":
255 args = (ser,)
256 elif kernel == "corr":
257 args = (ser,)
258 elif kernel == "cov":
259 args = (ser,)
260 elif kernel == "nth":
261 args = (0,)
262 elif kernel == "fillna":
263 args = (True,)
264 elif kernel == "fillna":
265 args = ("ffill",)
266 elif kernel == "take":
267 args = ([0],)
268 elif kernel == "quantile":
269 args = (0.5,)
270 else:
271 args = ()
272 method = getattr(ser, kernel)
273 if not has_numeric_only:
274 msg = (
275 "(got an unexpected keyword argument 'numeric_only'"
276 "|too many arguments passed in)"
277 )
278 with pytest.raises(TypeError, match=msg):
279 method(*args, numeric_only=True)
280 elif dtype is object:
281 msg = f"Series.{kernel} does not allow numeric_only=True with non-numeric"
282 with pytest.raises(TypeError, match=msg):
283 method(*args, numeric_only=True)
284 else:
285 result = method(*args, numeric_only=True)
286 expected = method(*args, numeric_only=False)
287 if isinstance(expected, Series):
288 # transformer
289 tm.assert_series_equal(result, expected)
290 else:
291 # reducer
292 assert result == expected
