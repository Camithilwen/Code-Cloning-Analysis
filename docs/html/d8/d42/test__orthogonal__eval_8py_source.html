39 def check_poly(self, func, cls, param_ranges=[], x_range=[], nn=10,
40 nparam=10, nx=10, rtol=1e-8):
41 np.random.seed(1234)
42
43 dataset = []
44 for n in np.arange(nn):
45 params = [a + (b-a)*np.random.rand(nparam) for a,b in param_ranges]
46 params = np.asarray(params).T
47 if not param_ranges:
48 params = [0]
49 for p in params:
50 if param_ranges:
51 p = (n,) + tuple(p)
52 else:
53 p = (n,)
54 x = x_range[0] + (x_range[1] - x_range[0])*np.random.rand(nx)
55 x[0] = x_range[0] # always include domain start point
56 x[1] = x_range[1] # always include domain end point
57 poly = np.poly1d(cls(*p).coef)
58 z = np.c_[np.tile(p, (nx,1)), x, poly(x)]
59 dataset.append(z)
60
61 dataset = np.concatenate(dataset, axis=0)
62
63 def polyfunc(*p):
64 p = (p[0].astype(int),) + p[1:]
65 return func(*p)
66
67 with np.errstate(all='raise'):
68 ds = FuncData(polyfunc, dataset, list(range(len(param_ranges)+2)), -1,
69 rtol=rtol)
70 ds.check()
