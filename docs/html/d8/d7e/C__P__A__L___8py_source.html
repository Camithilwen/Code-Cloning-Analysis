154 def _compilePalette(self, palette):
155 assert len(palette) == self.numPaletteEntries
156 pack = lambda c: struct.pack(">BBBB", c.blue, c.green, c.red, c.alpha)
157 return bytesjoin([pack(color) for color in palette])

159 def _compileColorRecords(self):
160 colorRecords, colorRecordIndices, pool = [], [], {}
161 for palette in self.palettes:
162 packedPalette = self._compilePalette(palette)
163 if packedPalette in pool:
164 index = pool[packedPalette]
165 else:
166 index = len(colorRecords)
167 colorRecords.append(packedPalette)
168 pool[packedPalette] = index
169 colorRecordIndices.append(struct.pack(">H", index * self.numPaletteEntries))
170 return bytesjoin(colorRecordIndices), bytesjoin(colorRecords)
