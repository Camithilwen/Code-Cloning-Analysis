277@pytest.mark.parametrize("n_samples", [10, 100, 10_000])
278@pytest.mark.parametrize("average", [1e-10, 1, 1e10])
279@pytest.mark.parametrize("dtype", [np.float32, np.float64])
280@pytest.mark.parametrize("sparse_container", [None] + CSC_CONTAINERS + CSR_CONTAINERS)
281def test_standard_scaler_near_constant_features(
282 n_samples, sparse_container, average, dtype
283):
284 # Check that when the variance is too small (var << mean**2) the feature
285 # is considered constant and not scaled.
286
287 scale_min, scale_max = -30, 19
288 scales = np.array([10**i for i in range(scale_min, scale_max + 1)], dtype=dtype)
289
290 n_features = scales.shape[0]
291 X = np.empty((n_samples, n_features), dtype=dtype)
292 # Make a dataset of known var = scales**2 and mean = average
293 X[: n_samples // 2, :] = average + scales
294 X[n_samples // 2 :, :] = average - scales
295 X_array = X if sparse_container is None else sparse_container(X)
296
297 scaler = StandardScaler(with_mean=False).fit(X_array)
298
299 # StandardScaler uses float64 accumulators even if the data has a float32
300 # dtype.
301 eps = np.finfo(np.float64).eps
302
303 # if var < bound = N.eps.var + N.eps.mean, the feature is considered
304 # constant and the scale_ attribute is set to 1.
305 bounds = n_samples * eps * scales**2 + n_samples**2 * eps**2 * average**2
306 within_bounds = scales**2 <= bounds
307
308 # Check that scale_min is small enough to have some scales below the
309 # bound and therefore detected as constant:
310 assert np.any(within_bounds)
311
312 # Check that such features are actually treated as constant by the scaler:
313 assert all(scaler.var_[within_bounds] <= bounds[within_bounds])
314 assert_allclose(scaler.scale_[within_bounds], 1.0)
315
316 # Depending the on the dtype of X, some features might not actually be
317 # representable as non constant for small scales (even if above the
318 # precision bound of the float64 variance estimate). Such feature should
319 # be correctly detected as constants with 0 variance by StandardScaler.
320 representable_diff = X[0, :] - X[-1, :] != 0
321 assert_allclose(scaler.var_[np.logical_not(representable_diff)], 0)
322 assert_allclose(scaler.scale_[np.logical_not(representable_diff)], 1)
323
324 # The other features are scaled and scale_ is equal to sqrt(var_) assuming
325 # that scales are large enough for average + scale and average - scale to
326 # be distinct in X (depending on X's dtype).
327 common_mask = np.logical_and(scales**2 > bounds, representable_diff)
328 assert_allclose(scaler.scale_[common_mask], np.sqrt(scaler.var_)[common_mask])
