13def test_feature_agglomeration():
14 n_clusters = 1

17 agglo_mean = FeatureAgglomeration(n_clusters=n_clusters, pooling_func=np.mean)
18 agglo_median = FeatureAgglomeration(n_clusters=n_clusters, pooling_func=np.median)
19 agglo_mean.fit(X)
20 agglo_median.fit(X)
21
22 assert np.size(np.unique(agglo_mean.labels_)) == n_clusters
23 assert np.size(np.unique(agglo_median.labels_)) == n_clusters
24 assert np.size(agglo_mean.labels_) == X.shape[1]
25 assert np.size(agglo_median.labels_) == X.shape[1]
26
27 # Test transform
28 Xt_mean = agglo_mean.transform(X)
29 Xt_median = agglo_median.transform(X)
30 assert Xt_mean.shape[1] == n_clusters
31 assert Xt_median.shape[1] == n_clusters
32 assert Xt_mean == np.array([1 / 3.0])
33 assert Xt_median == np.array([0.0])
34
35 # Test inverse transform
36 X_full_mean = agglo_mean.inverse_transform(Xt_mean)
37 X_full_median = agglo_median.inverse_transform(Xt_median)
38 assert np.unique(X_full_mean[0]).size == n_clusters
39 assert np.unique(X_full_median[0]).size == n_clusters
40
41 assert_array_almost_equal(agglo_mean.transform(X_full_mean), Xt_mean)
