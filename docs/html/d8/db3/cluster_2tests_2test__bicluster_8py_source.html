63@pytest.mark.parametrize("csr_container", CSR_CONTAINERS)
64def test_spectral_coclustering(global_random_seed, csr_container):
65 # Test Dhillon's Spectral CoClustering on a simple problem.
66 param_grid = {
67 "svd_method": ["randomized", "arpack"],
68 "n_svd_vecs": [None, 20],
69 "mini_batch": [False, True],

87 assert_array_equal(model.columns_.sum(axis=0), np.ones(30))

135def _do_scale_test(scaled):
136 """Check that rows sum to one constant, and columns to another."""
137 row_sum = scaled.sum(axis=1)
138 col_sum = scaled.sum(axis=0)
139 if issparse(scaled):
140 row_sum = np.asarray(row_sum).squeeze()
141 col_sum = np.asarray(col_sum).squeeze()
142 assert_array_almost_equal(row_sum, np.tile(row_sum.mean(), 100), decimal=1)
143 assert_array_almost_equal(col_sum, np.tile(col_sum.mean(), 100), decimal=1)
144

146def _do_bistochastic_test(scaled):
147 """Check that rows and columns sum to the same constant."""
148 _do_scale_test(scaled)
149 assert_almost_equal(scaled.sum(axis=0).mean(), scaled.sum(axis=1).mean(), decimal=1)
