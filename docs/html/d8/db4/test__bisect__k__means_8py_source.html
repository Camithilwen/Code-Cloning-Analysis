10@pytest.mark.parametrize("bisecting_strategy", ["biggest_inertia", "largest_cluster"])
11@pytest.mark.parametrize("init", ["k-means++", "random"])
12def test_three_clusters(bisecting_strategy, init):
13 """Tries to perform bisect k-means for three clusters to check
14 if splitting data is performed correctly.
15 """
16 X = np.array(
17 [[1, 1], [10, 1], [3, 1], [10, 0], [2, 1], [10, 2], [10, 8], [10, 9], [10, 10]]
18 )
19 bisect_means = BisectingKMeans(
20 n_clusters=3,
21 random_state=0,
22 bisecting_strategy=bisecting_strategy,
23 init=init,
24 )
25 bisect_means.fit(X)
26
27 expected_centers = [[2, 1], [10, 1], [10, 9]]
28 expected_labels = [0, 1, 0, 1, 0, 1, 2, 2, 2]
29
30 assert_allclose(
31 sorted(expected_centers), sorted(bisect_means.cluster_centers_.tolist())
32 )
33 assert_allclose(v_measure_score(expected_labels, bisect_means.labels_), 1.0)
