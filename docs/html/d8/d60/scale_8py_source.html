46class ScaleBase:
47 """
48 The base class for all scales.
49
50 Scales are separable transformations, working on a single dimension.
51
52 Subclasses should override
53
54 :attr:`name`
55 The scale's name.
56 :meth:`get_transform`
57 A method returning a `.Transform`, which converts data coordinates to
58 scaled coordinates. This transform should be invertible, so that e.g.
59 mouse positions can be converted back to data coordinates.
60 :meth:`set_default_locators_and_formatters`
61 A method that sets default locators and formatters for an `~.axis.Axis`
62 that uses this scale.
63 :meth:`limit_range_for_scale`
64 An optional method that "fixes" the axis range to acceptable values,
65 e.g. restricting log-scaled axes to positive values.
66 """
67
68 def __init__(self, axis):
69 r"""
70 Construct a new scale.
71
72 Notes
73 -----
74 The following note is for scale implementers.
75
76 For back-compatibility reasons, scales take an `~matplotlib.axis.Axis`
77 object as first argument. However, this argument should not
78 be used: a single scale object should be usable by multiple
79 `~matplotlib.axis.Axis`\es at the same time.
80 """
81

85 """

93 raise NotImplementedError()

97 Return the range *vmin*, *vmax*, restricted to the

99
100 *minpos* should be the minimum positive value in the data.
101 This is used by log scales to determine a minimum value.
102 """
103 return vmin, vmax

140class FuncTransform(Transform):
141 """
142 A simple transform that takes and arbitrary function for the
143 forward and inverse transform.
144 """
145
146 input_dims = output_dims = 1
147
148 def __init__(self, forward, inverse):
149 """
150 Parameters
151 ----------
152 forward : callable
153 The forward function for the transform. This function must have
154 an inverse and, for best behavior, be monotonic.
155 It must have the signature::
156
157 def forward(values: array-like) -> array-like
158
159 inverse : callable
160 The inverse of the forward function. Signature as ``forward``.
161 """
162 super().__init__()
163 if callable(forward) and callable(inverse):
164 self._forward = forward
165 self._inverse = inverse
166 else:
167 raise ValueError('arguments to FuncTransform must be functions')
168
169 def transform_non_affine(self, values):
170 return self._forward(values)
171
172 def inverted(self):
173 return FuncTransform(self._inverse, self._forward)

218class LogTransform(Transform):
219 input_dims = output_dims = 1
220
221 def __init__(self, base, nonpositive='clip'):
222 super().__init__()
223 if base <= 0 or base == 1:
224 raise ValueError('The log base cannot be <= 0 or == 1')
225 self.base = base
226 self._clip = _api.check_getitem(
227 {"clip": True, "mask": False}, nonpositive=nonpositive)
228
229 def __str__(self):
230 return "{}(base={}, nonpositive={!r})".format(
231 type(self).__name__, self.base, "clip" if self._clip else "mask")
232
233 def transform_non_affine(self, values):
234 # Ignore invalid values due to nans being passed to the transform.
235 with np.errstate(divide="ignore", invalid="ignore"):
236 log = {np.e: np.log, 2: np.log2, 10: np.log10}.get(self.base)
237 if log: # If possible, do everything in a single call to NumPy.
238 out = log(values)
239 else:
240 out = np.log(values)
241 out /= np.log(self.base)
242 if self._clip:
243 # SVG spec says that conforming viewers must support values up
244 # to 3.4e38 (C float); however experiments suggest that
245 # Inkscape (which uses cairo for rendering) runs into cairo's
246 # 24-bit limit (which is apparently shared by Agg).
247 # Ghostscript (used for pdf rendering appears to overflow even
248 # earlier, with the max value around 2 ** 15 for the tests to
249 # pass. On the other hand, in practice, we want to clip beyond
250 # np.log10(np.nextafter(0, 1)) ~ -323
251 # so 1000 seems safe.
252 out[values <= 0] = -1000
253 return out
254
255 def inverted(self):
256 return InvertedLogTransform(self.base)

363class SymmetricalLogTransform(Transform):
364 input_dims = output_dims = 1
365
366 def __init__(self, base, linthresh, linscale):
367 super().__init__()
368 if base <= 1.0:
369 raise ValueError("'base' must be larger than 1")
370 if linthresh <= 0.0:
371 raise ValueError("'linthresh' must be positive")
372 if linscale <= 0.0:
373 raise ValueError("'linscale' must be positive")
374 self.base = base
375 self.linthresh = linthresh
376 self.linscale = linscale
377 self._linscale_adj = (linscale / (1.0 - self.base ** -1))
378 self._log_base = np.log(base)
379
380 def transform_non_affine(self, values):
381 abs_a = np.abs(values)
382 with np.errstate(divide="ignore", invalid="ignore"):
383 out = np.sign(values) * self.linthresh * (
384 self._linscale_adj +
385 np.log(abs_a / self.linthresh) / self._log_base)
386 inside = abs_a <= self.linthresh
387 out[inside] = values[inside] * self._linscale_adj
388 return out
389
390 def inverted(self):
391 return InvertedSymmetricalLogTransform(self.base, self.linthresh,
392 self.linscale)

395class InvertedSymmetricalLogTransform(Transform):
396 input_dims = output_dims = 1
397
398 def __init__(self, base, linthresh, linscale):
399 super().__init__()
400 symlog = SymmetricalLogTransform(base, linthresh, linscale)
401 self.base = base
402 self.linthresh = linthresh
403 self.invlinthresh = symlog.transform(linthresh)
404 self.linscale = linscale
405 self._linscale_adj = (linscale / (1.0 - self.base ** -1))
406
407 def transform_non_affine(self, values):
408 abs_a = np.abs(values)
409 with np.errstate(divide="ignore", invalid="ignore"):
410 out = np.sign(values) * self.linthresh * (
411 np.power(self.base,
412 abs_a / self.linthresh - self._linscale_adj))
413 inside = abs_a <= self.invlinthresh
414 out[inside] = values[inside] / self._linscale_adj
415 return out
416
417 def inverted(self):
418 return SymmetricalLogTransform(self.base,
419 self.linthresh, self.linscale)

422class SymmetricalLogScale(ScaleBase):
423 """
424 The symmetrical logarithmic scale is logarithmic in both the
425 positive and negative directions from the origin.
426
427 Since the values close to zero tend toward infinity, there is a
428 need to have a range around zero that is linear. The parameter
429 *linthresh* allows the user to specify the size of this range
430 (-*linthresh*, *linthresh*).
431
432 See :doc:`/gallery/scales/symlog_demo` for a detailed description.
433
434 Parameters
435 ----------
436 base : float, default: 10
437 The base of the logarithm.
438
439 linthresh : float, default: 2
440 Defines the range ``(-x, x)``, within which the plot is linear.
441 This avoids having the plot go to infinity around zero.
442
443 subs : sequence of int
444 Where to place the subticks between each major tick.
445 For example, in a log10 scale: ``[2, 3, 4, 5, 6, 7, 8, 9]`` will place
446 8 logarithmically spaced minor ticks between each major tick.
447
448 linscale : float, optional
449 This allows the linear range ``(-linthresh, linthresh)`` to be
450 stretched relative to the logarithmic range. Its value is the number of
451 decades to use for each half of the linear range. For example, when
452 *linscale* == 1.0 (the default), the space used for the positive and
453 negative halves of the linear range will be equal to one decade in
454 the logarithmic range.
455 """
456 name = 'symlog'
457
458 def __init__(self, axis, *, base=10, linthresh=2, subs=None, linscale=1):
459 self._transform = SymmetricalLogTransform(base, linthresh, linscale)
460 self.subs = subs
461
462 base = property(lambda self: self._transform.base)
463 linthresh = property(lambda self: self._transform.linthresh)
464 linscale = property(lambda self: self._transform.linscale)
465
466 def set_default_locators_and_formatters(self, axis):
467 # docstring inherited
468 axis.set_major_locator(SymmetricalLogLocator(self.get_transform()))
469 axis.set_major_formatter(LogFormatterSciNotation(self.base))
470 axis.set_minor_locator(SymmetricalLogLocator(self.get_transform(),
471 self.subs))

475 """Return the `.SymmetricalLogTransform` associated with this scale."""

512class AsinhScale(ScaleBase):
513 """
514 A quasi-logarithmic scale based on the inverse hyperbolic sine (asinh)
515
516 For values close to zero, this is essentially a linear scale,
517 but for large magnitude values (either positive or negative)
518 it is asymptotically logarithmic. The transition between these
519 linear and logarithmic regimes is smooth, and has no discontinuities
520 in the function gradient in contrast to
521 the `.SymmetricalLogScale` ("symlog") scale.
522
523 Specifically, the transformation of an axis coordinate :math:`a` is
524 :math:`a \\rightarrow a_0 \\sinh^{-1} (a / a_0)` where :math:`a_0`
525 is the effective width of the linear region of the transformation.
526 In that region, the transformation is
527 :math:`a \\rightarrow a + \\mathcal{O}(a^3)`.
528 For large values of :math:`a` the transformation behaves as
529 :math:`a \\rightarrow a_0 \\, \\mathrm{sgn}(a) \\ln |a| + \\mathcal{O}(1)`.
530
531 .. note::
532
533 This API is provisional and may be revised in the future
534 based on early user feedback.
535 """
536
537 name = 'asinh'
538
539 auto_tick_multipliers = {
540 3: (2, ),
541 4: (2, ),
542 5: (2, ),
543 8: (2, 4),
544 10: (2, 5),
545 16: (2, 4, 8),
546 64: (4, 16),
547 1024: (256, 512)
548 }
549
550 def __init__(self, axis, *, linear_width=1.0,
551 base=10, subs='auto', **kwargs):
552 """
553 Parameters
554 ----------
555 linear_width : float, default: 1
556 The scale parameter (elsewhere referred to as :math:`a_0`)
557 defining the extent of the quasi-linear region,
558 and the coordinate values beyond which the transformation
559 becomes asymptotically logarithmic.
560 base : int, default: 10
561 The number base used for rounding tick locations
562 on a logarithmic scale. If this is less than one,
563 then rounding is to the nearest integer multiple
564 of powers of ten.
565 subs : sequence of int
566 Multiples of the number base used for minor ticks.
567 If set to 'auto', this will use built-in defaults,
568 e.g. (2, 5) for base=10.
569 """
570 super().__init__(axis)
571 self._transform = AsinhTransform(linear_width)
572 self._base = int(base)
573 if subs == 'auto':
574 self._subs = self.auto_tick_multipliers.get(self._base)
575 else:
576 self._subs = subs
577
578 linear_width = property(lambda self: self._transform.linear_width)
579
580 def get_transform(self):
581 return self._transform
582
583 def set_default_locators_and_formatters(self, axis):
584 axis.set(major_locator=AsinhLocator(self.linear_width,
585 base=self._base),
586 minor_locator=AsinhLocator(self.linear_width,
587 base=self._base,
588 subs=self._subs),
589 minor_formatter=NullFormatter())
590 if self._base > 1:
591 axis.set_major_formatter(LogFormatterSciNotation(self._base))
592 else:
593 axis.set_major_formatter('{x:.3g}')
