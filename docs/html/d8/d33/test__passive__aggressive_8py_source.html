23class MyPassiveAggressive(ClassifierMixin):
24 def __init__(
25 self,
26 C=1.0,
27 epsilon=0.01,
28 loss="hinge",
29 fit_intercept=True,
30 n_iter=1,
31 random_state=None,
32 ):
33 self.C = C
34 self.epsilon = epsilon
35 self.loss = loss
36 self.fit_intercept = fit_intercept
37 self.n_iter = n_iter
38
39 def fit(self, X, y):
40 n_samples, n_features = X.shape
41 self.w = np.zeros(n_features, dtype=np.float64)
42 self.b = 0.0
43
44 for t in range(self.n_iter):
45 for i in range(n_samples):
46 p = self.project(X[i])
47 if self.loss in ("hinge", "squared_hinge"):
48 loss = max(1 - y[i] * p, 0)
49 else:
50 loss = max(np.abs(p - y[i]) - self.epsilon, 0)
51
52 sqnorm = np.dot(X[i], X[i])
53
54 if self.loss in ("hinge", "epsilon_insensitive"):
55 step = min(self.C, loss / sqnorm)
56 elif self.loss in ("squared_hinge", "squared_epsilon_insensitive"):
57 step = loss / (sqnorm + 1.0 / (2 * self.C))
58
59 if self.loss in ("hinge", "squared_hinge"):
60 step *= y[i]
61 else:
62 step *= np.sign(y[i] - p)
63
64 self.w += step * X[i]
65 if self.fit_intercept:
66 self.b += step
67
68 def project(self, X):
69 return np.dot(X, self.w) + self.b
