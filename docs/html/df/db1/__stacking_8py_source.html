87 def _concatenate_predictions(self, X, predictions):
88 """Concatenate the predictions of each first layer learner and
89 possibly the input dataset `X`.
90
91 If `X` is sparse and `self.passthrough` is False, the output of
92 `transform` will be dense (the predictions). If `X` is sparse
93 and `self.passthrough` is True, the output of `transform` will
94 be sparse.
95
96 This helper is in charge of ensuring the predictions are 2D arrays and
97 it will drop one of the probability column when using probabilities
98 in the binary case. Indeed, the p(y|c=0) = 1 - p(y|c=1)
99
100 When `y` type is `"multilabel-indicator"`` and the method used is
101 `predict_proba`, `preds` can be either a `ndarray` of shape
102 `(n_samples, n_class)` or for some estimators a list of `ndarray`.
103 This function will drop one of the probability column in this situation as well.
104 """
105 X_meta = []
106 for est_idx, preds in enumerate(predictions):
107 if isinstance(preds, list):
108 # `preds` is here a list of `n_targets` 2D ndarrays of
109 # `n_classes` columns. The k-th column contains the
110 # probabilities of the samples belonging the k-th class.
111 #
112 # Since those probabilities must sum to one for each sample,
113 # we can work with probabilities of `n_classes - 1` classes.
114 # Hence we drop the first column.
115 for pred in preds:
116 X_meta.append(pred[:, 1:])
117 elif preds.ndim == 1:
118 # Some estimator return a 1D array for predictions
119 # which must be 2-dimensional arrays.
120 X_meta.append(preds.reshape(-1, 1))
121 elif (
122 self.stack_method_[est_idx] == "predict_proba"
123 and len(self.classes_) == 2
124 ):
125 # Remove the first column when using probabilities in
126 # binary classification because both features `preds` are perfectly
127 # collinear.
128 X_meta.append(preds[:, 1:])
129 else:
130 X_meta.append(preds)
131
132 self._n_feature_outs = [pred.shape[1] for pred in X_meta]
133 if self.passthrough:
134 X_meta.append(X)
135 if sparse.issparse(X):
136 return sparse.hstack(X_meta, format=X.format)
137
138 return np.hstack(X_meta)

159 def fit(self, X, y, **fit_params):
160 """Fit the estimators.
161
162 Parameters
163 ----------
164 X : {array-like, sparse matrix} of shape (n_samples, n_features)
165 Training vectors, where `n_samples` is the number of samples and
166 `n_features` is the number of features.
167
168 y : array-like of shape (n_samples,)
169 Target values.
170
171 **fit_params : dict
172 Dict of metadata, potentially containing sample_weight as a
173 key-value pair. If sample_weight is not present, then samples are
174 equally weighted. Note that sample_weight is supported only if all
175 underlying estimators support sample weights.
176
177 .. versionadded:: 1.6
178
179 Returns
180 -------
181 self : object
182 """
183 # all_estimators contains all estimators, the one to be fitted and the
184 # 'drop' string.
185 names, all_estimators = self._validate_estimators()
186 self._validate_final_estimator()
187
188 stack_method = [self.stack_method] * len(all_estimators)
189
190 if _routing_enabled():
191 routed_params = process_routing(self, "fit", **fit_params)
192 else:
193 routed_params = Bunch()
194 for name in names:
195 routed_params[name] = Bunch(fit={})
196 if "sample_weight" in fit_params:
197 routed_params[name].fit["sample_weight"] = fit_params[
198 "sample_weight"
199 ]
200
201 if self.cv == "prefit":
202 self.estimators_ = []
203 for estimator in all_estimators:
204 if estimator != "drop":
205 check_is_fitted(estimator)
206 self.estimators_.append(estimator)
207 else:
208 # Fit the base estimators on the whole training data. Those
209 # base estimators will be used in transform, predict, and
210 # predict_proba. They are exposed publicly.
211 self.estimators_ = Parallel(n_jobs=self.n_jobs)(
212 delayed(_fit_single_estimator)(
213 clone(est), X, y, routed_params[name]["fit"]
214 )
215 for name, est in zip(names, all_estimators)
216 if est != "drop"
217 )
218
219 self.named_estimators_ = Bunch()
220 est_fitted_idx = 0
221 for name_est, org_est in zip(names, all_estimators):
222 if org_est != "drop":
223 current_estimator = self.estimators_[est_fitted_idx]
224 self.named_estimators_[name_est] = current_estimator
225 est_fitted_idx += 1
226 if hasattr(current_estimator, "feature_names_in_"):
227 self.feature_names_in_ = current_estimator.feature_names_in_
228 else:
229 self.named_estimators_[name_est] = "drop"
230
231 self.stack_method_ = [
232 self._method_name(name, est, meth)
233 for name, est, meth in zip(names, all_estimators, stack_method)
234 ]
235
236 if self.cv == "prefit":
237 # Generate predictions from prefit models
238 predictions = [
239 getattr(estimator, predict_method)(X)
240 for estimator, predict_method in zip(all_estimators, self.stack_method_)
241 if estimator != "drop"
242 ]
243 else:
244 # To train the meta-classifier using the most data as possible, we use
245 # a cross-validation to obtain the output of the stacked estimators.
246 # To ensure that the data provided to each estimator are the same,
247 # we need to set the random state of the cv if there is one and we
248 # need to take a copy.
249 cv = check_cv(self.cv, y=y, classifier=is_classifier(self))
250 if hasattr(cv, "random_state") and cv.random_state is None:
251 cv.random_state = np.random.RandomState()
252
253 predictions = Parallel(n_jobs=self.n_jobs)(
254 delayed(cross_val_predict)(
255 clone(est),
256 X,
257 y,
258 cv=deepcopy(cv),
259 method=meth,
260 n_jobs=self.n_jobs,
261 params=routed_params[name]["fit"],
262 verbose=self.verbose,
263 )
264 for name, est, meth in zip(names, all_estimators, self.stack_method_)
265 if est != "drop"
266 )
267
268 # Only not None or not 'drop' estimators will be used in transform.
269 # Remove the None from the method as well.
270 self.stack_method_ = [
271 meth
272 for (meth, est) in zip(self.stack_method_, all_estimators)
273 if est != "drop"
274 ]
275
276 X_meta = self._concatenate_predictions(X, predictions)
277 _fit_single_estimator(self.final_estimator_, X_meta, y, fit_params=fit_params)
278
279 return self

302 def get_feature_names_out(self, input_features=None):
303 """Get output feature names for transformation.
304
305 Parameters
306 ----------
307 input_features : array-like of str or None, default=None
308 Input features. The input feature names are only used when `passthrough` is
309 `True`.
310
311 - If `input_features` is `None`, then `feature_names_in_` is
312 used as feature names in. If `feature_names_in_` is not defined,
313 then names are generated: `[x0, x1, ..., x(n_features_in_ - 1)]`.
314 - If `input_features` is an array-like, then `input_features` must
315 match `feature_names_in_` if `feature_names_in_` is defined.
316
317 If `passthrough` is `False`, then only the names of `estimators` are used
318 to generate the output feature names.
319
320 Returns
321 -------
322 feature_names_out : ndarray of str objects
323 Transformed feature names.
324 """
325 check_is_fitted(self, "n_features_in_")
326 input_features = _check_feature_names_in(
327 self, input_features, generate_names=self.passthrough
328 )
329
330 class_name = self.__class__.__name__.lower()
331 non_dropped_estimators = (
332 name for name, est in self.estimators if est != "drop"
333 )
334 meta_names = []
335 for est, n_features_out in zip(non_dropped_estimators, self._n_feature_outs):
336 if n_features_out == 1:
337 meta_names.append(f"{class_name}_{est}")
338 else:
339 meta_names.extend(
340 f"{class_name}_{est}{i}" for i in range(n_features_out)
341 )
342
343 if self.passthrough:
344 return np.concatenate((meta_names, input_features))
345
346 return np.asarray(meta_names, dtype=object)

422class StackingClassifier(ClassifierMixin, _BaseStacking):
423 """Stack of estimators with a final classifier.
424
425 Stacked generalization consists in stacking the output of individual
426 estimator and use a classifier to compute the final prediction. Stacking
427 allows to use the strength of each individual estimator by using their
428 output as input of a final estimator.
429
430 Note that `estimators_` are fitted on the full `X` while `final_estimator_`
431 is trained using cross-validated predictions of the base estimators using
432 `cross_val_predict`.
433
434 Read more in the :ref:`User Guide <stacking>`.
435
436 .. versionadded:: 0.22
437
438 Parameters
439 ----------
440 estimators : list of (str, estimator)
441 Base estimators which will be stacked together. Each element of the
442 list is defined as a tuple of string (i.e. name) and an estimator
443 instance. An estimator can be set to 'drop' using `set_params`.
444
445 The type of estimator is generally expected to be a classifier.
446 However, one can pass a regressor for some use case (e.g. ordinal
447 regression).
448
449 final_estimator : estimator, default=None
450 A classifier which will be used to combine the base estimators.
451 The default classifier is a
452 :class:`~sklearn.linear_model.LogisticRegression`.
453
454 cv : int, cross-validation generator, iterable, or "prefit", default=None
455 Determines the cross-validation splitting strategy used in
456 `cross_val_predict` to train `final_estimator`. Possible inputs for
457 cv are:
458
459 * None, to use the default 5-fold cross validation,
460 * integer, to specify the number of folds in a (Stratified) KFold,
461 * An object to be used as a cross-validation generator,
462 * An iterable yielding train, test splits,
463 * `"prefit"`, to assume the `estimators` are prefit. In this case, the
464 estimators will not be refitted.
465
466 For integer/None inputs, if the estimator is a classifier and y is
467 either binary or multiclass,
468 :class:`~sklearn.model_selection.StratifiedKFold` is used.
469 In all other cases, :class:`~sklearn.model_selection.KFold` is used.
470 These splitters are instantiated with `shuffle=False` so the splits
471 will be the same across calls.
472
473 Refer :ref:`User Guide <cross_validation>` for the various
474 cross-validation strategies that can be used here.
475
476 If "prefit" is passed, it is assumed that all `estimators` have
477 been fitted already. The `final_estimator_` is trained on the `estimators`
478 predictions on the full training set and are **not** cross validated
479 predictions. Please note that if the models have been trained on the same
480 data to train the stacking model, there is a very high risk of overfitting.
481
482 .. versionadded:: 1.1
483 The 'prefit' option was added in 1.1
484
485 .. note::
486 A larger number of split will provide no benefits if the number
487 of training samples is large enough. Indeed, the training time
488 will increase. ``cv`` is not used for model evaluation but for
489 prediction.
490
491 stack_method : {'auto', 'predict_proba', 'decision_function', 'predict'}, \
492 default='auto'
493 Methods called for each base estimator. It can be:
494
495 * if 'auto', it will try to invoke, for each estimator,
496 `'predict_proba'`, `'decision_function'` or `'predict'` in that
497 order.
498 * otherwise, one of `'predict_proba'`, `'decision_function'` or
499 `'predict'`. If the method is not implemented by the estimator, it
500 will raise an error.
501
502 n_jobs : int, default=None
503 The number of jobs to run in parallel for `fit` of all `estimators`.
504 `None` means 1 unless in a `joblib.parallel_backend` context. -1 means
505 using all processors. See :term:`Glossary <n_jobs>` for more details.
506
507 passthrough : bool, default=False
508 When False, only the predictions of estimators will be used as
509 training data for `final_estimator`. When True, the
510 `final_estimator` is trained on the predictions as well as the
511 original training data.
512
513 verbose : int, default=0
514 Verbosity level.
515
516 Attributes
517 ----------
518 classes_ : ndarray of shape (n_classes,) or list of ndarray if `y` \
519 is of type `"multilabel-indicator"`.
520 Class labels.
521
522 estimators_ : list of estimators
523 The elements of the `estimators` parameter, having been fitted on the
524 training data. If an estimator has been set to `'drop'`, it
525 will not appear in `estimators_`. When `cv="prefit"`, `estimators_`
526 is set to `estimators` and is not fitted again.
527
528 named_estimators_ : :class:`~sklearn.utils.Bunch`
529 Attribute to access any fitted sub-estimators by name.
530
531 n_features_in_ : int
532 Number of features seen during :term:`fit`. Only defined if the
533 underlying estimator exposes such an attribute when fit.
534
535 .. versionadded:: 0.24
536
537 feature_names_in_ : ndarray of shape (`n_features_in_`,)
538 Names of features seen during :term:`fit`. Only defined if the
539 underlying estimators expose such an attribute when fit.
540
541 .. versionadded:: 1.0
542
543 final_estimator_ : estimator
544 The classifier fit on the output of `estimators_` and responsible for
545 final predictions.
546
547 stack_method_ : list of str
548 The method used by each base estimator.
549
550 See Also
551 --------
552 StackingRegressor : Stack of estimators with a final regressor.
553
554 Notes
555 -----
556 When `predict_proba` is used by each estimator (i.e. most of the time for
557 `stack_method='auto'` or specifically for `stack_method='predict_proba'`),
558 the first column predicted by each estimator will be dropped in the case
559 of a binary classification problem. Indeed, both feature will be perfectly
560 collinear.
561
562 In some cases (e.g. ordinal regression), one can pass regressors as the
563 first layer of the :class:`StackingClassifier`. However, note that `y` will
564 be internally encoded in a numerically increasing order or lexicographic
565 order. If this ordering is not adequate, one should manually numerically
566 encode the classes in the desired order.
567
568 References
569 ----------
570 .. [1] Wolpert, David H. "Stacked generalization." Neural networks 5.2
571 (1992): 241-259.
572
573 Examples
574 --------
575 >>> from sklearn.datasets import load_iris
576 >>> from sklearn.ensemble import RandomForestClassifier
577 >>> from sklearn.svm import LinearSVC
578 >>> from sklearn.linear_model import LogisticRegression
579 >>> from sklearn.preprocessing import StandardScaler
580 >>> from sklearn.pipeline import make_pipeline
581 >>> from sklearn.ensemble import StackingClassifier
582 >>> X, y = load_iris(return_X_y=True)
583 >>> estimators = [
584 ... ('rf', RandomForestClassifier(n_estimators=10, random_state=42)),
585 ... ('svr', make_pipeline(StandardScaler(),
586 ... LinearSVC(random_state=42)))
587 ... ]
588 >>> clf = StackingClassifier(
589 ... estimators=estimators, final_estimator=LogisticRegression()
590 ... )
591 >>> from sklearn.model_selection import train_test_split
592 >>> X_train, X_test, y_train, y_test = train_test_split(
593 ... X, y, stratify=y, random_state=42
594 ... )
595 >>> clf.fit(X_train, y_train).score(X_test, y_test)
596 0.9...
597 """

636 def _validate_estimators(self):
637 """Overload the method of `_BaseHeterogeneousEnsemble` to be more
638 lenient towards the type of `estimators`.
639
640 Regressors can be accepted for some cases such as ordinal regression.
641 """
642 if len(self.estimators) == 0:
643 raise ValueError(
644 "Invalid 'estimators' attribute, 'estimators' should be a "
645 "non-empty list of (string, estimator) tuples."
646 )
647 names, estimators = zip(*self.estimators)
648 self._validate_names(names)
649
650 has_estimator = any(est != "drop" for est in estimators)
651 if not has_estimator:
652 raise ValueError(
653 "All estimators are dropped. At least one is required "
654 "to be an estimator."
655 )
656
657 return names, estimators

659 def fit(self, X, y, **fit_params):
660 """Fit the estimators.
661
662 Parameters
663 ----------
664 X : {array-like, sparse matrix} of shape (n_samples, n_features)
665 Training vectors, where `n_samples` is the number of samples and
666 `n_features` is the number of features.
667
668 y : array-like of shape (n_samples,)
669 Target values. Note that `y` will be internally encoded in
670 numerically increasing order or lexicographic order. If the order
671 matter (e.g. for ordinal regression), one should numerically encode
672 the target `y` before calling :term:`fit`.
673
674 **fit_params : dict
675 Parameters to pass to the underlying estimators.
676
677 .. versionadded:: 1.6
678
679 Only available if `enable_metadata_routing=True`, which can be
680 set by using ``sklearn.set_config(enable_metadata_routing=True)``.
681 See :ref:`Metadata Routing User Guide <metadata_routing>` for
682 more details.
683
684 Returns
685 -------
686 self : object
687 Returns a fitted instance of estimator.
688 """
689 _raise_for_params(fit_params, self, "fit", allow=["sample_weight"])
690
691 check_classification_targets(y)
692 if type_of_target(y) == "multilabel-indicator":
693 self._label_encoder = [LabelEncoder().fit(yk) for yk in y.T]
694 self.classes_ = [le.classes_ for le in self._label_encoder]
695 y_encoded = np.array(
696 [
697 self._label_encoder[target_idx].transform(target)
698 for target_idx, target in enumerate(y.T)
699 ]
700 ).T
701 else:
702 self._label_encoder = LabelEncoder().fit(y)
703 self.classes_ = self._label_encoder.classes_
704 y_encoded = self._label_encoder.transform(y)
705
706 return super().fit(X, y_encoded, **fit_params)

841class StackingRegressor(RegressorMixin, _BaseStacking):
842 """Stack of estimators with a final regressor.
843
844 Stacked generalization consists in stacking the output of individual
845 estimator and use a regressor to compute the final prediction. Stacking
846 allows to use the strength of each individual estimator by using their
847 output as input of a final estimator.
848
849 Note that `estimators_` are fitted on the full `X` while `final_estimator_`
850 is trained using cross-validated predictions of the base estimators using
851 `cross_val_predict`.
852
853 Read more in the :ref:`User Guide <stacking>`.
854
855 .. versionadded:: 0.22
856
857 Parameters
858 ----------
859 estimators : list of (str, estimator)
860 Base estimators which will be stacked together. Each element of the
861 list is defined as a tuple of string (i.e. name) and an estimator
862 instance. An estimator can be set to 'drop' using `set_params`.
863
864 final_estimator : estimator, default=None
865 A regressor which will be used to combine the base estimators.
866 The default regressor is a :class:`~sklearn.linear_model.RidgeCV`.
867
868 cv : int, cross-validation generator, iterable, or "prefit", default=None
869 Determines the cross-validation splitting strategy used in
870 `cross_val_predict` to train `final_estimator`. Possible inputs for
871 cv are:
872
873 * None, to use the default 5-fold cross validation,
874 * integer, to specify the number of folds in a (Stratified) KFold,
875 * An object to be used as a cross-validation generator,
876 * An iterable yielding train, test splits,
877 * `"prefit"`, to assume the `estimators` are prefit. In this case, the
878 estimators will not be refitted.
879
880 For integer/None inputs, if the estimator is a classifier and y is
881 either binary or multiclass,
882 :class:`~sklearn.model_selection.StratifiedKFold` is used.
883 In all other cases, :class:`~sklearn.model_selection.KFold` is used.
884 These splitters are instantiated with `shuffle=False` so the splits
885 will be the same across calls.
886
887 Refer :ref:`User Guide <cross_validation>` for the various
888 cross-validation strategies that can be used here.
889
890 If "prefit" is passed, it is assumed that all `estimators` have
891 been fitted already. The `final_estimator_` is trained on the `estimators`
892 predictions on the full training set and are **not** cross validated
893 predictions. Please note that if the models have been trained on the same
894 data to train the stacking model, there is a very high risk of overfitting.
895
896 .. versionadded:: 1.1
897 The 'prefit' option was added in 1.1
898
899 .. note::
900 A larger number of split will provide no benefits if the number
901 of training samples is large enough. Indeed, the training time
902 will increase. ``cv`` is not used for model evaluation but for
903 prediction.
904
905 n_jobs : int, default=None
906 The number of jobs to run in parallel for `fit` of all `estimators`.
907 `None` means 1 unless in a `joblib.parallel_backend` context. -1 means
908 using all processors. See :term:`Glossary <n_jobs>` for more details.
909
910 passthrough : bool, default=False
911 When False, only the predictions of estimators will be used as
912 training data for `final_estimator`. When True, the
913 `final_estimator` is trained on the predictions as well as the
914 original training data.
915
916 verbose : int, default=0
917 Verbosity level.
918
919 Attributes
920 ----------
921 estimators_ : list of estimators
922 The elements of the `estimators` parameter, having been fitted on the
923 training data. If an estimator has been set to `'drop'`, it
924 will not appear in `estimators_`. When `cv="prefit"`, `estimators_`
925 is set to `estimators` and is not fitted again.
926
927 named_estimators_ : :class:`~sklearn.utils.Bunch`
928 Attribute to access any fitted sub-estimators by name.
929
930 n_features_in_ : int
931 Number of features seen during :term:`fit`. Only defined if the
932 underlying estimator exposes such an attribute when fit.
933
934 .. versionadded:: 0.24
935
936 feature_names_in_ : ndarray of shape (`n_features_in_`,)
937 Names of features seen during :term:`fit`. Only defined if the
938 underlying estimators expose such an attribute when fit.
939
940 .. versionadded:: 1.0
941
942 final_estimator_ : estimator
943 The regressor fit on the output of `estimators_` and responsible for
944 final predictions.
945
946 stack_method_ : list of str
947 The method used by each base estimator.
948
949 See Also
950 --------
951 StackingClassifier : Stack of estimators with a final classifier.
952
953 References
954 ----------
955 .. [1] Wolpert, David H. "Stacked generalization." Neural networks 5.2
956 (1992): 241-259.
957
958 Examples
959 --------
960 >>> from sklearn.datasets import load_diabetes
961 >>> from sklearn.linear_model import RidgeCV
962 >>> from sklearn.svm import LinearSVR
963 >>> from sklearn.ensemble import RandomForestRegressor
964 >>> from sklearn.ensemble import StackingRegressor
965 >>> X, y = load_diabetes(return_X_y=True)
966 >>> estimators = [
967 ... ('lr', RidgeCV()),
968 ... ('svr', LinearSVR(random_state=42))
969 ... ]
970 >>> reg = StackingRegressor(
971 ... estimators=estimators,
972 ... final_estimator=RandomForestRegressor(n_estimators=10,
973 ... random_state=42)
974 ... )
975 >>> from sklearn.model_selection import train_test_split
976 >>> X_train, X_test, y_train, y_test = train_test_split(
977 ... X, y, random_state=42
978 ... )
979 >>> reg.fit(X_train, y_train).score(X_test, y_test)
980 0.3...
981 """
