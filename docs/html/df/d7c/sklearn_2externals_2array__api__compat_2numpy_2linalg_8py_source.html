52# Note: unlike np.linalg.solve, the array API solve() only accepts x2 as a
53# vector when it is exactly 1-dimensional. All other cases treat x2 as a stack
54# of matrices. The np.linalg.solve behavior of allowing stacks of both
55# matrices and vectors is ambiguous c.f.
56# https://github.com/numpy/numpy/issues/15349 and
57# https://github.com/data-apis/array-api/issues/285.
58
59# To workaround this, the below is the code from np.linalg.solve except
60# only calling solve1 in the exactly 1D case.
61

63# This code is here instead of in common because it is numpy specific. Also
64# note that CuPy's solve() does not currently support broadcasting (see
65# https://github.com/cupy/cupy/blob/main/cupy/cublas.py#L43).
66def solve(x1: Array, x2: Array, /) -> Array:
67 try:
68 from numpy.linalg._linalg import (
69 _assert_stacked_2d,
70 _assert_stacked_square,
71 _commonType,
72 _makearray,
73 _raise_linalgerror_singular,
74 isComplexType,
75 )
76 except ImportError:
77 from numpy.linalg.linalg import (
78 _assert_stacked_2d,
79 _assert_stacked_square,
80 _commonType,
81 _makearray,
82 _raise_linalgerror_singular,
83 isComplexType,
84 )
85 from numpy.linalg import _umath_linalg
86
87 x1, _ = _makearray(x1)
88 _assert_stacked_2d(x1)
89 _assert_stacked_square(x1)
90 x2, wrap = _makearray(x2)
91 t, result_t = _commonType(x1, x2)
92
93 # This part is different from np.linalg.solve
94 gufunc: np.ufunc
95 if x2.ndim == 1:
96 gufunc = _umath_linalg.solve1
97 else:
98 gufunc = _umath_linalg.solve
99
100 # This does nothing currently but is left in because it will be relevant
101 # when complex dtype support is added to the spec in 2022.
102 signature = "DD->D" if isComplexType(t) else "dd->d"
103 with np.errstate(
104 call=_raise_linalgerror_singular,
105 invalid="call",
106 over="ignore",
107 divide="ignore",
108 under="ignore",
109 ):
110 r: Array = gufunc(x1, x2, signature=signature)
111
112 return wrap(r.astype(result_t, copy=False))
