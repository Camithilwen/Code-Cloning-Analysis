424def test_prefit():
425 # Test all possible combinations of the prefit parameter.
426
427 # Passing a prefit parameter with the selected model
428 # and fitting a unfit model with prefit=False should give same results.
429 clf = SGDClassifier(alpha=0.1, max_iter=10, shuffle=True, random_state=0, tol=None)
430 model = SelectFromModel(clf)
431 model.fit(data, y)
432 X_transform = model.transform(data)
433 clf.fit(data, y)
434 model = SelectFromModel(clf, prefit=True)
435 assert_array_almost_equal(model.transform(data), X_transform)
436 model.fit(data, y)
437 assert model.estimator_ is not clf
438
439 # Check that the model is rewritten if prefit=False and a fitted model is
440 # passed
441 model = SelectFromModel(clf, prefit=False)
442 model.fit(data, y)
443 assert_array_almost_equal(model.transform(data), X_transform)
444
445 # Check that passing an unfitted estimator with `prefit=True` raises a
446 # `ValueError`
447 clf = SGDClassifier(alpha=0.1, max_iter=10, shuffle=True, random_state=0, tol=None)
448 model = SelectFromModel(clf, prefit=True)
449 err_msg = "When `prefit=True`, `estimator` is expected to be a fitted estimator."
450 with pytest.raises(NotFittedError, match=err_msg):
451 model.fit(data, y)
452 with pytest.raises(NotFittedError, match=err_msg):
453 model.partial_fit(data, y)
454 with pytest.raises(NotFittedError, match=err_msg):
455 model.transform(data)
456
457 # Check that the internal parameters of prefitted model are not changed
458 # when calling `fit` or `partial_fit` with `prefit=True`
459 clf = SGDClassifier(alpha=0.1, max_iter=10, shuffle=True, tol=None).fit(data, y)
460 model = SelectFromModel(clf, prefit=True)
461 model.fit(data, y)
462 assert_allclose(model.estimator_.coef_, clf.coef_)
463 model.partial_fit(data, y)
464 assert_allclose(model.estimator_.coef_, clf.coef_)
