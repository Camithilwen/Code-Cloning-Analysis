24def _find_binning_thresholds(col_data, max_bins):
25 """Extract quantiles from a continuous feature.
26
27 Missing values are ignored for finding the thresholds.
28
29 Parameters
30 ----------
31 col_data : array-like, shape (n_samples,)
32 The continuous feature to bin.
33 max_bins: int
34 The maximum number of bins to use for non-missing values. If for a
35 given feature the number of unique values is less than ``max_bins``,
36 then those unique values will be used to compute the bin thresholds,
37 instead of the quantiles
38
39 Return
40 ------
41 binning_thresholds : ndarray of shape(min(max_bins, n_unique_values) - 1,)
42 The increasing numeric values that can be used to separate the bins.
43 A given value x will be mapped into bin value i iff
44 bining_thresholds[i - 1] < x <= binning_thresholds[i]
45 """
46 # ignore missing values when computing bin thresholds
47 missing_mask = np.isnan(col_data)
48 if missing_mask.any():
49 col_data = col_data[~missing_mask]
50 # The data will be sorted anyway in np.unique and again in percentile, so we do it
51 # here. Sorting also returns a contiguous array.
52 col_data = np.sort(col_data)
53 distinct_values = np.unique(col_data).astype(X_DTYPE)
54 if len(distinct_values) <= max_bins:
55 midpoints = distinct_values[:-1] + distinct_values[1:]
56 midpoints *= 0.5
57 else:
58 # We could compute approximate midpoint percentiles using the output of
59 # np.unique(col_data, return_counts) instead but this is more
60 # work and the performance benefit will be limited because we
61 # work on a fixed-size subsample of the full data.
62 percentiles = np.linspace(0, 100, num=max_bins + 1)
63 percentiles = percentiles[1:-1]
64 midpoints = np.percentile(col_data, percentiles, method="midpoint").astype(
65 X_DTYPE
66 )
67 assert midpoints.shape[0] == max_bins - 1
68
69 # We avoid having +inf thresholds: +inf thresholds are only allowed in
70 # a "split on nan" situation.
71 np.clip(midpoints, a_min=None, a_max=ALMOST_INF, out=midpoints)
72 return midpoints

75class _BinMapper(TransformerMixin, BaseEstimator):
76 """Transformer that maps a dataset into integer-valued bins.
77
78 For continuous features, the bins are created in a feature-wise fashion,
79 using quantiles so that each bins contains approximately the same number
80 of samples. For large datasets, quantiles are computed on a subset of the
81 data to speed-up the binning, but the quantiles should remain stable.
82
83 For categorical features, the raw categorical values are expected to be
84 in [0, 254] (this is not validated here though) and each category
85 corresponds to a bin. All categorical values must be known at
86 initialization: transform() doesn't know how to bin unknown categorical
87 values. Note that transform() is only used on non-training data in the
88 case of early stopping.
89
90 Features with a small number of values may be binned into less than
91 ``n_bins`` bins. The last bin (at index ``n_bins - 1``) is always reserved
92 for missing values.
93
94 Parameters
95 ----------
96 n_bins : int, default=256
97 The maximum number of bins to use (including the bin for missing
98 values). Should be in [3, 256]. Non-missing values are binned on
99 ``max_bins = n_bins - 1`` bins. The last bin is always reserved for
100 missing values. If for a given feature the number of unique values is
101 less than ``max_bins``, then those unique values will be used to
102 compute the bin thresholds, instead of the quantiles. For categorical
103 features indicated by ``is_categorical``, the docstring for
104 ``is_categorical`` details on this procedure.
105 subsample : int or None, default=2e5
106 If ``n_samples > subsample``, then ``sub_samples`` samples will be
107 randomly chosen to compute the quantiles. If ``None``, the whole data
108 is used.
109 is_categorical : ndarray of bool of shape (n_features,), default=None
110 Indicates categorical features. By default, all features are
111 considered continuous.
112 known_categories : list of {ndarray, None} of shape (n_features,), \
113 default=none
114 For each categorical feature, the array indicates the set of unique
115 categorical values. These should be the possible values over all the
116 data, not just the training data. For continuous features, the
117 corresponding entry should be None.
118 random_state: int, RandomState instance or None, default=None
119 Pseudo-random number generator to control the random sub-sampling.
120 Pass an int for reproducible output across multiple
121 function calls.
122 See :term:`Glossary <random_state>`.
123 n_threads : int, default=None
124 Number of OpenMP threads to use. `_openmp_effective_n_threads` is called
125 to determine the effective number of threads use, which takes cgroups CPU
126 quotes into account. See the docstring of `_openmp_effective_n_threads`
127 for details.
128
129 Attributes
130 ----------
131 bin_thresholds_ : list of ndarray
132 For each feature, each array indicates how to map a feature into a
133 binned feature. The semantic and size depends on the nature of the
134 feature:
135 - for real-valued features, the array corresponds to the real-valued
136 bin thresholds (the upper bound of each bin). There are ``max_bins
137 - 1`` thresholds, where ``max_bins = n_bins - 1`` is the number of
138 bins used for non-missing values.
139 - for categorical features, the array is a map from a binned category
140 value to the raw category value. The size of the array is equal to
141 ``min(max_bins, category_cardinality)`` where we ignore missing
142 values in the cardinality.
143 n_bins_non_missing_ : ndarray, dtype=np.uint32
144 For each feature, gives the number of bins actually used for
145 non-missing values. For features with a lot of unique values, this is
146 equal to ``n_bins - 1``.
147 is_categorical_ : ndarray of shape (n_features,), dtype=np.uint8
148 Indicator for categorical features.
149 missing_values_bin_idx_ : np.uint8
150 The index of the bin where missing values are mapped. This is a
151 constant across all features. This corresponds to the last bin, and
152 it is always equal to ``n_bins - 1``. Note that if ``n_bins_non_missing_``
153 is less than ``n_bins - 1`` for a given feature, then there are
154 empty (and unused) bins.
155 """

173 def fit(self, X, y=None):
174 """Fit data X by computing the binning thresholds.
175
176 The last bin is reserved for missing values, whether missing values
177 are present in the data or not.
178
179 Parameters
180 ----------
181 X : array-like of shape (n_samples, n_features)
182 The data to bin.
183 y: None
184 Ignored.
185
186 Returns
187 -------
188 self : object
189 """
190 if not (3 <= self.n_bins <= 256):
191 # min is 3: at least 2 distinct bins and a missing values bin
192 raise ValueError(
193 "n_bins={} should be no smaller than 3 and no larger than 256.".format(
194 self.n_bins
195 )
196 )
197
198 X = check_array(X, dtype=[X_DTYPE], ensure_all_finite=False)
199 max_bins = self.n_bins - 1
200
201 rng = check_random_state(self.random_state)
202 if self.subsample is not None and X.shape[0] > self.subsample:
203 subset = rng.choice(X.shape[0], self.subsample, replace=False)
204 X = X.take(subset, axis=0)
205
206 if self.is_categorical is None:
207 self.is_categorical_ = np.zeros(X.shape[1], dtype=np.uint8)
208 else:
209 self.is_categorical_ = np.asarray(self.is_categorical, dtype=np.uint8)
210
211 n_features = X.shape[1]
212 known_categories = self.known_categories
213 if known_categories is None:
214 known_categories = [None] * n_features
215
216 # validate is_categorical and known_categories parameters
217 for f_idx in range(n_features):
218 is_categorical = self.is_categorical_[f_idx]
219 known_cats = known_categories[f_idx]
220 if is_categorical and known_cats is None:
221 raise ValueError(
222 f"Known categories for feature {f_idx} must be provided."
223 )
224 if not is_categorical and known_cats is not None:
225 raise ValueError(
226 f"Feature {f_idx} isn't marked as a categorical feature, "
227 "but categories were passed."
228 )
229
230 self.missing_values_bin_idx_ = self.n_bins - 1
231
232 self.bin_thresholds_ = [None] * n_features
233 n_bins_non_missing = [None] * n_features
234
235 non_cat_thresholds = Parallel(n_jobs=self.n_threads, backend="threading")(
236 delayed(_find_binning_thresholds)(X[:, f_idx], max_bins)
237 for f_idx in range(n_features)
238 if not self.is_categorical_[f_idx]
239 )
240
241 non_cat_idx = 0
242 for f_idx in range(n_features):
243 if self.is_categorical_[f_idx]:
244 # Since categories are assumed to be encoded in
245 # [0, n_cats] and since n_cats <= max_bins,
246 # the thresholds *are* the unique categorical values. This will
247 # lead to the correct mapping in transform()
248 thresholds = known_categories[f_idx]
249 n_bins_non_missing[f_idx] = thresholds.shape[0]
250 self.bin_thresholds_[f_idx] = thresholds
251 else:
252 self.bin_thresholds_[f_idx] = non_cat_thresholds[non_cat_idx]
253 n_bins_non_missing[f_idx] = self.bin_thresholds_[f_idx].shape[0] + 1
254 non_cat_idx += 1
255
256 self.n_bins_non_missing_ = np.array(n_bins_non_missing, dtype=np.uint32)
257 return self

261
262 Missing values will be mapped to the last bin.
263
264 For categorical features, the mapping will be incorrect for unknown
265 categories. Since the BinMapper is given known_categories of the
266 entire training data (i.e. before the call to train_test_split() in
267 case of early-stopping), this never happens.
268
269 Parameters
270 ----------
271 X : array-like of shape (n_samples, n_features)
272 The data to bin.
273
274 Returns
275 -------
276 X_binned : array-like of shape (n_samples, n_features)
277 The binned data (fortran-aligned).
278 """
279 X = check_array(X, dtype=[X_DTYPE], ensure_all_finite=False)
280 check_is_fitted(self)
281 if X.shape[1] != self.n_bins_non_missing_.shape[0]:
282 raise ValueError(
283 "This estimator was fitted with {} features but {} got passed "
284 "to transform()".format(self.n_bins_non_missing_.shape[0], X.shape[1])
285 )
286
287 n_threads = _openmp_effective_n_threads(self.n_threads)

290 X,
291 self.bin_thresholds_,
292 self.is_categorical_,
293 self.missing_values_bin_idx_,
294 n_threads,

296 )
