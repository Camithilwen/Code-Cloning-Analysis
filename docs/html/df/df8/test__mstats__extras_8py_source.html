122 def test_hdquantiles_sd(self):
123 # Standard deviation is a jackknife estimator, so we can check if
124 # the efficient version (hdquantiles_sd) matches a rudimentary,
125 # but clear version here.
126
127 hd_std_errs = ms.hdquantiles_sd(self.data)
128
129 # jacknnife standard error, Introduction to the Bootstrap Eq. 11.5
130 n = len(self.data)
131 jdata = np.broadcast_to(self.data, (n, n))
132 jselector = np.logical_not(np.eye(n)) # leave out one sample each row
133 jdata = jdata[jselector].reshape(n, n-1)
134 jdist = ms.hdquantiles(jdata, axis=1)
135 jdist_mean = np.mean(jdist, axis=0)
136 jstd = ((n-1)/n * np.sum((jdist - jdist_mean)**2, axis=0))**.5
137
138 assert_almost_equal(hd_std_errs, jstd)
139 # Test actual values for good measure
140 assert_almost_equal(hd_std_errs, [0.0379258, 0.0380656, 0.0380013])
141
142 two_data_points = ms.hdquantiles_sd([1, 2])
143 assert_almost_equal(two_data_points, [0.5, 0.5, 0.5])
