52@pytest.mark.parametrize('a', [0, 1e-6, 0.1, 0.5, 1, 10])
53@pytest.mark.parametrize('b', [1, 1 + 1e-3, 2, 5, 10])
54@pytest.mark.parametrize('x', [0, 1e-6, 0.1, 0.5, 1, 5, 10, 100])
55def test_wright_functional(a, b, x):
56 """Test functional relation of wright_bessel.
57
58 Phi(a, b-1, z) = a*z*Phi(a, b+a, z) + (b-1)*Phi(a, b, z)
59
60 Note that d/dx Phi(a, b, x) = Phi(a, b-1, x)
61 See Eq. (22) of
62 B. Stankovic, On the Function of E. M. Wright,
63 Publ. de l' Institut Mathematique, Beograd,
64 Nouvelle S`er. 10 (1970), 113-124.
65 """
66 assert_allclose(wright_bessel(a, b - 1, x),
67 a * x * wright_bessel(a, b + a, x)
68 + (b - 1) * wright_bessel(a, b, x),
69 rtol=1e-8, atol=1e-8)
