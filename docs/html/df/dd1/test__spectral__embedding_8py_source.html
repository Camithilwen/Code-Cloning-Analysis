56def _assert_equal_with_sign_flipping(A, B, tol=0.0):
57 """Check array A and B are equal with possible sign flipping on
58 each column"""
59 tol_squared = tol**2
60 for A_col, B_col in zip(A.T, B.T):
61 assert (
62 np.max((A_col - B_col) ** 2) <= tol_squared
63 or np.max((A_col + B_col) ** 2) <= tol_squared
64 )

67@pytest.mark.parametrize("coo_container", COO_CONTAINERS)
68def test_sparse_graph_connected_component(coo_container):
69 rng = np.random.RandomState(42)
70 n_samples = 300
71 boundaries = [0, 42, 121, 200, n_samples]
72 p = rng.permutation(n_samples)
73 connections = []
74
75 for start, stop in itertools.pairwise(boundaries):
76 group = p[start:stop]
77 # Connect all elements within the group at least once via an
78 # arbitrary path that spans the group.
79 for i in range(len(group) - 1):
80 connections.append((group[i], group[i + 1]))
81
82 # Add some more random connections within the group
83 min_idx, max_idx = 0, len(group) - 1
84 n_random_connections = 1000
85 source = rng.randint(min_idx, max_idx, size=n_random_connections)
86 target = rng.randint(min_idx, max_idx, size=n_random_connections)
87 connections.extend(zip(group[source], group[target]))
88
89 # Build a symmetric affinity matrix
90 row_idx, column_idx = tuple(np.array(connections).T)
91 data = rng.uniform(0.1, 42, size=len(connections))
92 affinity = coo_container((data, (row_idx, column_idx)))
93 affinity = 0.5 * (affinity + affinity.T)
94
95 for start, stop in itertools.pairwise(boundaries):
96 component_1 = _graph_connected_component(affinity, p[start])
97 component_size = stop - start
98 assert component_1.sum() == component_size
99
100 # We should retrieve the same component mask by starting by both ends
101 # of the group
102 component_2 = _graph_connected_component(affinity, p[stop - 1])
103 assert component_2.sum() == component_size
104 assert_array_equal(component_1, component_2)

119@pytest.mark.parametrize("dtype", [np.float32, np.float64])
120def test_spectral_embedding_two_components(eigen_solver, dtype, seed=0):
121 # Test spectral embedding with two components
122 random_state = np.random.RandomState(seed)
123 n_sample = 100
124 affinity = np.zeros(shape=[n_sample * 2, n_sample * 2])
125 # first component
126 affinity[0:n_sample, 0:n_sample] = (
127 np.abs(random_state.randn(n_sample, n_sample)) + 2
128 )
129 # second component
130 affinity[n_sample::, n_sample::] = (
131 np.abs(random_state.randn(n_sample, n_sample)) + 2
132 )
133
134 # Test of internal _graph_connected_component before connection
135 component = _graph_connected_component(affinity, 0)
136 assert component[:n_sample].all()
137 assert not component[n_sample:].any()
138 component = _graph_connected_component(affinity, -1)
139 assert not component[:n_sample].any()
140 assert component[n_sample:].all()
141
142 # connection
143 affinity[0, n_sample + 1] = 1
144 affinity[n_sample + 1, 0] = 1
145 affinity.flat[:: 2 * n_sample + 1] = 0
146 affinity = 0.5 * (affinity + affinity.T)
147
148 true_label = np.zeros(shape=2 * n_sample)
149 true_label[0:n_sample] = 1
150
151 se_precomp = SpectralEmbedding(
152 n_components=1,
153 affinity="precomputed",
154 random_state=np.random.RandomState(seed),
155 eigen_solver=eigen_solver,
156 )
157
158 embedded_coordinate = se_precomp.fit_transform(affinity.astype(dtype))
159 # thresholding on the first components using 0.
160 label_ = np.array(embedded_coordinate.ravel() < 0, dtype=np.int64)
161 assert normalized_mutual_info_score(true_label, label_) == pytest.approx(1.0)
