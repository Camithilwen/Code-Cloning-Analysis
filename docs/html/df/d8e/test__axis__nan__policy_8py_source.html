95def _mixed_data_generator(n_samples, n_repetitions, axis, rng,
96 paired=False):
97 # generate random samples to check the response of hypothesis tests to
98 # samples with different (but broadcastable) shapes and various
99 # nan patterns (e.g. all nans, some nans, no nans) along axis-slices
100
101 data = []
102 for i in range(n_samples):
103 n_patterns = 6 # number of distinct nan patterns
104 n_obs = 20 if paired else 20 + i # observations per axis-slice
105 x = np.ones((n_repetitions, n_patterns, n_obs)) * np.nan
106
107 for j in range(n_repetitions):
108 samples = x[j, :, :]
109
110 # case 0: axis-slice with all nans (0 reals)
111 # cases 1-3: axis-slice with 1-3 reals (the rest nans)
112 # case 4: axis-slice with mostly (all but two) reals
113 # case 5: axis slice with all reals
114 for k, n_reals in enumerate([0, 1, 2, 3, n_obs-2, n_obs]):
115 # for cases 1-3, need paired nansw to be in the same place
116 indices = rng.permutation(n_obs)[:n_reals]
117 samples[k, indices] = rng.random(size=n_reals)
118
119 # permute the axis-slices just to show that order doesn't matter
120 samples[:] = rng.permutation(samples, axis=0)
121
122 # For multi-sample tests, we want to test broadcasting and check
123 # that nan policy works correctly for each nan pattern for each input.
124 # This takes care of both simultaneosly.
125 new_shape = [n_repetitions] + [1]*n_samples + [n_obs]
126 new_shape[1 + i] = 6
127 x = x.reshape(new_shape)
128
129 x = np.moveaxis(x, -1, axis)
130 data.append(x)
131 return data

134def _homogeneous_data_generator(n_samples, n_repetitions, axis, rng,
135 paired=False, all_nans=True):
136 # generate random samples to check the response of hypothesis tests to
137 # samples with different (but broadcastable) shapes and homogeneous
138 # data (all nans or all finite)
139 data = []
140 for i in range(n_samples):
141 n_obs = 20 if paired else 20 + i # observations per axis-slice
142 shape = [n_repetitions] + [1]*n_samples + [n_obs]
143 shape[1 + i] = 2
144 x = np.ones(shape) * np.nan if all_nans else rng.random(shape)
145 x = np.moveaxis(x, -1, axis)
146 data.append(x)
147 return data

150def nan_policy_1d(hypotest, data1d, unpacker, *args, n_outputs=2,
151 nan_policy='raise', paired=False, _no_deco=True, **kwds):
152 # Reference implementation for how `nan_policy` should work for 1d samples
153
154 if nan_policy == 'raise':
155 for sample in data1d:
156 if np.any(np.isnan(sample)):
157 raise ValueError("The input contains nan values")
158
159 elif (nan_policy == 'propagate'
160 and hypotest not in override_propagate_funcs):
161 # For all hypothesis tests tested, returning nans is the right thing.
162 # But many hypothesis tests don't propagate correctly (e.g. they treat
163 # np.nan the same as np.inf, which doesn't make sense when ranks are
164 # involved) so override that behavior here.
165 for sample in data1d:
166 if np.any(np.isnan(sample)):
167 return np.full(n_outputs, np.nan)
168
169 elif nan_policy == 'omit':
170 # manually omit nans (or pairs in which at least one element is nan)
171 if not paired:
172 data1d = [sample[~np.isnan(sample)] for sample in data1d]
173 else:
174 nan_mask = np.isnan(data1d[0])
175 for sample in data1d[1:]:
176 nan_mask = np.logical_or(nan_mask, np.isnan(sample))
177 data1d = [sample[~nan_mask] for sample in data1d]
178
179 return unpacker(hypotest(*data1d, *args, _no_deco=_no_deco, **kwds))

212def _axis_nan_policy_test(hypotest, args, kwds, n_samples, n_outputs, paired,
213 unpacker, nan_policy, axis, data_generator):
214 # Tests the 1D and vectorized behavior of hypothesis tests against a
215 # reference implementation (nan_policy_1d with np.ndenumerate)
216
217 # Some hypothesis tests return a non-iterable that needs an `unpacker` to
218 # extract the statistic and p-value. For those that don't:
219 if not unpacker:
220 def unpacker(res):
221 return res
222
223 rng = np.random.default_rng(0)
224
225 # Generate multi-dimensional test data with all important combinations
226 # of patterns of nans along `axis`
227 n_repetitions = 3 # number of repetitions of each pattern
228 data_gen_kwds = {'n_samples': n_samples, 'n_repetitions': n_repetitions,
229 'axis': axis, 'rng': rng, 'paired': paired}
230 if data_generator == 'mixed':
231 inherent_size = 6 # number of distinct types of patterns
232 data = _mixed_data_generator(**data_gen_kwds)
233 elif data_generator == 'all_nans':
234 inherent_size = 2 # hard-coded in _homogeneous_data_generator
235 data_gen_kwds['all_nans'] = True
236 data = _homogeneous_data_generator(**data_gen_kwds)
237 elif data_generator == 'all_finite':
238 inherent_size = 2 # hard-coded in _homogeneous_data_generator
239 data_gen_kwds['all_nans'] = False
240 data = _homogeneous_data_generator(**data_gen_kwds)
241
242 output_shape = [n_repetitions] + [inherent_size]*n_samples
243
244 # To generate reference behavior to compare against, loop over the axis-
245 # slices in data. Make indexing easier by moving `axis` to the end and
246 # broadcasting all samples to the same shape.
247 data_b = [np.moveaxis(sample, axis, -1) for sample in data]
248 data_b = [np.broadcast_to(sample, output_shape + [sample.shape[-1]])
249 for sample in data_b]
250 statistics = np.zeros(output_shape)
251 pvalues = np.zeros(output_shape)
252
253 for i, _ in np.ndenumerate(statistics):
254 data1d = [sample[i] for sample in data_b]
255 with np.errstate(divide='ignore', invalid='ignore'):
256 try:
257 res1d = nan_policy_1d(hypotest, data1d, unpacker, *args,
258 n_outputs=n_outputs,
259 nan_policy=nan_policy,
260 paired=paired, _no_deco=True, **kwds)
261
262 # Eventually we'll check the results of a single, vectorized
263 # call of `hypotest` against the arrays `statistics` and
264 # `pvalues` populated using the reference `nan_policy_1d`.
265 # But while we're at it, check the results of a 1D call to
266 # `hypotest` against the reference `nan_policy_1d`.
267 res1db = unpacker(hypotest(*data1d, *args,
268 nan_policy=nan_policy, **kwds))
269 assert_equal(res1db[0], res1d[0])
270 if len(res1db) == 2:
271 assert_equal(res1db[1], res1d[1])
272
273 # When there is not enough data in 1D samples, many existing
274 # hypothesis tests raise errors instead of returning nans .
275 # For vectorized calls, we put nans in the corresponding elements
276 # of the output.
277 except (RuntimeWarning, UserWarning, ValueError,
278 ZeroDivisionError) as e:
279
280 # whatever it is, make sure same error is raised by both
281 # `nan_policy_1d` and `hypotest`
282 with pytest.raises(type(e), match=re.escape(str(e))):
283 nan_policy_1d(hypotest, data1d, unpacker, *args,
284 n_outputs=n_outputs, nan_policy=nan_policy,
285 paired=paired, _no_deco=True, **kwds)
286 with pytest.raises(type(e), match=re.escape(str(e))):
287 hypotest(*data1d, *args, nan_policy=nan_policy, **kwds)
288
289 if any([str(e).startswith(message)
290 for message in too_small_messages]):
291 res1d = np.full(n_outputs, np.nan)
292 elif any([str(e).startswith(message)
293 for message in inaccuracy_messages]):
294 with suppress_warnings() as sup:
295 sup.filter(RuntimeWarning)
296 sup.filter(UserWarning)
297 res1d = nan_policy_1d(hypotest, data1d, unpacker,
298 *args, n_outputs=n_outputs,
299 nan_policy=nan_policy,
300 paired=paired, _no_deco=True,
301 **kwds)
302 else:
303 raise e
304 statistics[i] = res1d[0]
305 if len(res1d) == 2:
306 pvalues[i] = res1d[1]
307
308 # Perform a vectorized call to the hypothesis test.
309 # If `nan_policy == 'raise'`, check that it raises the appropriate error.
310 # If not, compare against the output against `statistics` and `pvalues`
311 if nan_policy == 'raise' and not data_generator == "all_finite":
312 message = 'The input contains nan values'
313 with pytest.raises(ValueError, match=message):
314 hypotest(*data, axis=axis, nan_policy=nan_policy, *args, **kwds)
315
316 else:
317 with suppress_warnings() as sup, \
318 np.errstate(divide='ignore', invalid='ignore'):
319 sup.filter(RuntimeWarning, "Precision loss occurred in moment")
320 sup.filter(UserWarning, "Sample size too small for normal "
321 "approximation.")
322 res = unpacker(hypotest(*data, axis=axis, nan_policy=nan_policy,
323 *args, **kwds))
324 assert_allclose(res[0], statistics, rtol=1e-15)
325 assert_equal(res[0].dtype, statistics.dtype)
326
327 if len(res) == 2:
328 assert_allclose(res[1], pvalues, rtol=1e-15)
329 assert_equal(res[1].dtype, pvalues.dtype)

590def _check_arrays_broadcastable(arrays, axis):
591 # https://numpy.org/doc/stable/user/basics.broadcasting.html
592 # "When operating on two arrays, NumPy compares their shapes element-wise.
593 # It starts with the trailing (i.e. rightmost) dimensions and works its
594 # way left.
595 # Two dimensions are compatible when
596 # 1. they are equal, or
597 # 2. one of them is 1
598 # ...
599 # Arrays do not need to have the same number of dimensions."
600 # (Clarification: if the arrays are compatible according to the criteria
601 # above and an array runs out of dimensions, it is still compatible.)
602 # Below, we follow the rules above except ignoring `axis`
603
604 n_dims = max([arr.ndim for arr in arrays])
605 if axis is not None:
606 # convert to negative axis
607 axis = (-n_dims + axis) if axis >= 0 else axis
608
609 for dim in range(1, n_dims+1): # we'll index from -1 to -n_dims, inclusive
610 if -dim == axis:
611 continue # ignore lengths along `axis`
612
613 dim_lengths = set()
614 for arr in arrays:
615 if dim <= arr.ndim and arr.shape[-dim] != 1:
616 dim_lengths.add(arr.shape[-dim])
617
618 if len(dim_lengths) > 1:
619 return False
620 return True
