91 # Initialization
92 sn = max(1, math.sqrt(nu)); ch, az, bz = _chlrps(sigma, a/sn, b/sn)
93 n = len(sigma); N = 10; P = math.ceil(m/N); on = np.ones(P); p = 0; e = 0
94 ps = np.sqrt(_primes(5*n*math.log(n+4)/4)); q = ps[:, np.newaxis] # Richtmyer gens.
95
96 # Randomization loop for ns samples
97 c = None; dc = None
98 for S in range(N):
99 vp = on.copy(); s = np.zeros((n, P))
100 for i in range(n):
101 x = np.abs(2*np.mod(q[i]*np.arange(1, P+1) + rng.random(), 1)-1) # periodizing transform
102 if i == 0:
103 r = on
104 if nu > 0:
105 r = np.sqrt(2*_gaminv(x, nu/2))
106 else:

108 s[i:] += ch[i:, i-1:i] * y
109 si = s[i, :]; c = on.copy(); ai = az[i]*r - si; d = on.copy(); bi = bz[i]*r - si

127def _chlrps(R, a, b):
128 """
129 Computes permuted and scaled lower Cholesky factor c for R which may be
130 singular, also permuting and scaling integration limit vectors a and b.
131 """
132 ep = 1e-10 # singularity tolerance
133 eps = np.finfo(R.dtype).eps
134
135 n = len(R); c = R.copy(); ap = a.copy(); bp = b.copy(); d = np.sqrt(np.maximum(np.diag(c), 0))
136 for i in range(n):
137 if d[i] > 0:
138 c[:, i] /= d[i]; c[i, :] /= d[i]
139 ap[i] /= d[i]; bp[i] /= d[i]
140 y = np.zeros((n, 1)); sqtp = math.sqrt(2*math.pi)
141
142 for k in range(n):
143 im = k; ckk = 0; dem = 1; s = 0
144 for i in range(k, n):
145 if c[i, i] > eps:
146 cii = math.sqrt(max(c[i, i], 0))
147 if i > 0: s = c[i, :k] @ y[:k]
148 ai = (ap[i]-s)/cii; bi = (bp[i]-s)/cii; de = _Phi(bi)-_Phi(ai)
149 if de <= dem:
150 ckk = cii; dem = de; am = ai; bm = bi; im = i
151 if im > k:
152 ap[[im, k]] = ap[[k, im]]; bp[[im, k]] = bp[[k, im]]; c[im, im] = c[k, k]
153 t = c[im, :k].copy(); c[im, :k] = c[k, :k]; c[k, :k] = t
154 t = c[im+1:, im].copy(); c[im+1:, im] = c[im+1:, k]; c[im+1:, k] = t
155 t = c[k+1:im, k].copy(); c[k+1:im, k] = c[im, k+1:im].T; c[im, k+1:im] = t.T
156 if ckk > ep*(k+1):
157 c[k, k] = ckk; c[k, k+1:] = 0
158 for i in range(k+1, n):
159 c[i, k] = c[i, k]/ckk; c[i, k+1:i+1] = c[i, k+1:i+1] - c[i, k]*c[k+1:i+1, k].T
160 if abs(dem) > ep:
161 y[k] = (np.exp(-am**2/2) - np.exp(-bm**2/2)) / (sqtp*dem)
162 else:
163 y[k] = (am + bm) / 2
164 if am < -10:
165 y[k] = bm
166 elif bm > 10:
167 y[k] = am
168 c[k, :k+1] /= ckk; ap[k] /= ckk; bp[k] /= ckk
169 else:
170 c[k:, k] = 0; y[k] = (ap[k] + bp[k])/2
171 pass
172 return c, ap, bp
