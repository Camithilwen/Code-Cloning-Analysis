39def alt_sg_coeffs(window_length, polyorder, pos):
40 """This is an alternative implementation of the SG coefficients.
41
42 It uses numpy.polyfit and numpy.polyval. The results should be
43 equivalent to those of savgol_coeffs(), but this implementation
44 is slower.
45
46 window_length should be odd.
47
48 """
49 if pos is None:
50 pos = window_length // 2
51 t = np.arange(window_length)
52 unit = (t == pos).astype(int)
53 h = np.polyval(np.polyfit(t, unit, polyorder), t)
54 return h

94def test_sg_coeffs_exact():
95 polyorder = 4
96 window_length = 9
97 halflen = window_length // 2
98
99 x = np.linspace(0, 21, 43)
100 delta = x[1] - x[0]
101
102 # The data is a cubic polynomial. We'll use an order 4
103 # SG filter, so the filtered values should equal the input data
104 # (except within half window_length of the edges).
105 y = 0.5 * x ** 3 - x
106 h = savgol_coeffs(window_length, polyorder)
107 y0 = convolve1d(y, h)
108 assert_allclose(y0[halflen:-halflen], y[halflen:-halflen])
109
110 # Check the same input, but use deriv=1. dy is the exact result.
111 dy = 1.5 * x ** 2 - 1
112 h = savgol_coeffs(window_length, polyorder, deriv=1, delta=delta)
113 y1 = convolve1d(y, h)
114 assert_allclose(y1[halflen:-halflen], dy[halflen:-halflen])
115
116 # Check the same input, but use deriv=2. d2y is the exact result.
117 d2y = 3.0 * x
118 h = savgol_coeffs(window_length, polyorder, deriv=2, delta=delta)
119 y2 = convolve1d(y, h)
120 assert_allclose(y2[halflen:-halflen], d2y[halflen:-halflen])

166def test_sg_coeffs_even_window_length():
167 # Simple case - deriv=0, polyorder=0, 1
168 window_lengths = [4, 6, 8, 10, 12, 14, 16]
169 for length in window_lengths:
170 h_p_d = savgol_coeffs(length, 0, 0)
171 assert_allclose(h_p_d, 1/length)
172
173 # Verify with closed forms
174 # deriv=1, polyorder=1, 2
175 def h_p_d_closed_form_1(k, m):
176 return 6*(k - 0.5)/((2*m + 1)*m*(2*m - 1))
177
178 # deriv=2, polyorder=2
179 def h_p_d_closed_form_2(k, m):
180 numer = 15*(-4*m**2 + 1 + 12*(k - 0.5)**2)
181 denom = 4*(2*m + 1)*(m + 1)*m*(m - 1)*(2*m - 1)
182 return numer/denom
183
184 for length in window_lengths:
185 m = length//2
186 expected_output = [h_p_d_closed_form_1(k, m)
187 for k in range(-m + 1, m + 1)][::-1]
188 actual_output = savgol_coeffs(length, 1, 1)
189 assert_allclose(expected_output, actual_output)
190 actual_output = savgol_coeffs(length, 2, 1)
191 assert_allclose(expected_output, actual_output)
192
193 expected_output = [h_p_d_closed_form_2(k, m)
194 for k in range(-m + 1, m + 1)][::-1]
195 actual_output = savgol_coeffs(length, 2, 2)
196 assert_allclose(expected_output, actual_output)
197 actual_output = savgol_coeffs(length, 3, 2)
198 assert_allclose(expected_output, actual_output)
