95 def check_continuous_case(self, a, q):
96 x = solve_continuous_lyapunov(a, q)
97 assert_array_almost_equal(
98 np.dot(a, x) + np.dot(x, a.conj().transpose()), q)
99
100 def check_discrete_case(self, a, q, method=None):
101 x = solve_discrete_lyapunov(a, q, method=method)
102 assert_array_almost_equal(
103 np.dot(np.dot(a, x), a.conj().transpose()) - x, -1.0*q)

283 # Makes the minimum precision requirements customized to the test.
284 # Here numbers represent the number of decimals that agrees with zero
285 # matrix when the solution x is plugged in to the equation.
286 #
287 # res = array([[8e-3,1e-16],[1e-16,1e-20]]) --> min_decimal[k] = 2
288 #
289 # If the test is failing use "None" for that entry.

291 min_decimal = (14, 12, 13, 14, 11, 6, None, 5, 7, 14, 14,
292 None, 9, 14, 13, 14, None, 12, None, None)

294 def _test_factory(case, dec):
295 """Checks if 0 = XA + A'X - XB(R)^{-1} B'X + Q is true"""
296 a, b, q, r, knownfailure = case
297 if knownfailure:
298 pytest.xfail(reason=knownfailure)
299
300 x = solve_continuous_are(a, b, q, r)
301 res = x.dot(a) + a.conj().T.dot(x) + q
302 out_fact = x.dot(b)
303 res -= out_fact.dot(solve(np.atleast_2d(r), out_fact.conj().T))
304 assert_array_almost_equal(res, np.zeros_like(res), decimal=dec)
305
306 for ind, case in enumerate(cases):
307 _test_factory(case, min_decimal[ind])
