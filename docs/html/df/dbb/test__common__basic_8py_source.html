43def test_override_set_noconvert_columns():
44 # see gh-17351
45 #
46 # Usecols needs to be sorted in _set_noconvert_columns based
47 # on the test_usecols_with_parse_dates test from test_usecols.py
48 class MyTextFileReader(TextFileReader):
49 def __init__(self) -> None:
50 self._currow = 0
51 self.squeeze = False
52
53 class MyCParserWrapper(CParserWrapper):
54 def _set_noconvert_columns(self):
55 if self.usecols_dtype == "integer":
56 # self.usecols is a set, which is documented as unordered
57 # but in practice, a CPython set of integers is sorted.
58 # In other implementations this assumption does not hold.
59 # The following code simulates a different order, which
60 # before GH 17351 would cause the wrong columns to be
61 # converted via the parse_dates parameter
62 self.usecols = list(self.usecols)
63 self.usecols.reverse()
64 return CParserWrapper._set_noconvert_columns(self)
65
66 data = """a,b,c,d,e
670,1,2014-01-01,09:00,4
680,1,2014-01-02,10:00,4"""
69
70 parse_dates = [[1, 2]]
71 cols = {
72 "a": [0, 0],
73 "c_d": [Timestamp("2014-01-01 09:00:00"), Timestamp("2014-01-02 10:00:00")],
74 }
75 expected = DataFrame(cols, columns=["c_d", "a"])
76
77 parser = MyTextFileReader()
78 parser.options = {
79 "usecols": [0, 2, 3],
80 "parse_dates": parse_dates,
81 "delimiter": ",",
82 }
83 parser.engine = "c"
84 parser._engine = MyCParserWrapper(StringIO(data), **parser.options)
85
86 result = parser.read()
87 tm.assert_frame_equal(result, expected)
