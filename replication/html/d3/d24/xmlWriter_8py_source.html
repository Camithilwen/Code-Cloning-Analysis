184def escape(data):
185 """Escape characters not allowed in `XML 1.0 <https://www.w3.org/TR/xml/#NT-Char>`_."""
186 data = tostr(data, "utf_8")

199 "Illegal XML character(s) found; replacing offending " "string %r with %r",
200 preview,
201 REPLACEMENT,
