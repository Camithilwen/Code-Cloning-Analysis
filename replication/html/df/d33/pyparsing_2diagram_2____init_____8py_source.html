63def _make_bookmark(s: str) -> str:
64 """
65 Converts a string into a valid HTML bookmark (ID or anchor name).
66 """
67 if s in _bookmark_lookup:
68 return _bookmark_lookup[s]
69
70 # Replace invalid characters with hyphens and ensure only valid characters
71 bookmark = re.sub(r'[^a-zA-Z0-9-]+', '-', s)
72
73 # Ensure it starts with a letter by adding 'z' if necessary
74 if not bookmark[:1].isalpha():
75 bookmark = f"z{bookmark}"
76
77 # Convert to lowercase and strip hyphens
78 bookmark = bookmark.lower().strip('-')
79
80 _bookmark_lookup[s] = bookmark = f"{bookmark}-{next(_bookmark_ids):04d}"
81
82 return bookmark

85def _collapse_verbose_regex(regex_str: str) -> str:
86 if "\n" not in regex_str:
87 return regex_str

138class EditablePartial(Generic[T]):
139 """
140 Acts like a functools.partial, but can be edited. In other words, it represents a type that hasn't yet been
141 constructed.
142 """
143
144 # We need this here because the railroad constructors actually transform the data, so can't be called until the
145 # entire tree is assembled
146
147 def __init__(self, func: Callable[..., T], args: list, kwargs: dict) -> None:
148 self.func = func
149 self.args = args
150 self.kwargs = kwargs
151
152 @classmethod
153 def from_call(cls, func: Callable[..., T], *args, **kwargs) -> EditablePartial[T]:
154 """
155 If you call this function in the same way that you would call the constructor, it will store the arguments
156 as you expect. For example EditablePartial.from_call(Fraction, 1, 3)() == Fraction(1, 3)
157 """
158 return EditablePartial(func=func, args=list(args), kwargs=kwargs)
159
160 @property
161 def name(self):
162 return self.kwargs["name"]
163
164 def __call__(self) -> T:
165 """
166 Evaluate the partial and return the result
167 """
168 args = self.args.copy()
169 kwargs = self.kwargs.copy()
170
171 # This is a helpful hack to allow you to specify varargs parameters (e.g. *args) as keyword args (e.g.
172 # args=['list', 'of', 'things'])
173 arg_spec = inspect.getfullargspec(self.func)
174 if arg_spec.varargs in self.kwargs:
175 args += kwargs.pop(arg_spec.varargs)
176
177 return self.func(*args, **kwargs)

180def railroad_to_html(diagrams: list[NamedDiagram], embed=False, **kwargs) -> str:
181 """
182 Given a list of NamedDiagram, produce a single HTML string that visualises those diagrams
183 :params kwargs: kwargs to be passed in to the template
184 """
185 data = []
186 for diagram in diagrams:
187 if diagram.diagram is None:
188 continue
189 io = StringIO()
190 try:
191 css = kwargs.get("css")
192 diagram.diagram.writeStandalone(io.write, css=css)
193 except AttributeError:
194 diagram.diagram.writeSvg(io.write)
195 title = diagram.name
196 if diagram.index == 0:
197 title += " (root)"
198 data.append(
199 {
200 "title": title, "text": "", "svg": io.getvalue(), "bookmark": diagram.bookmark
201 }
202 )
203
204 return template.render(diagrams=data, embed=embed, **kwargs)

207def resolve_partial(partial: EditablePartial[T]) -> T:
208 """
209 Recursively resolves a collection of Partials into whatever type they are
210 """
211 if isinstance(partial, EditablePartial):
212 partial.args = resolve_partial(partial.args)
213 partial.kwargs = resolve_partial(partial.kwargs)
214 return partial()
215 elif isinstance(partial, list):
216 return [resolve_partial(x) for x in partial]
217 elif isinstance(partial, dict):
218 return {key: resolve_partial(x) for key, x in partial.items()}
219 else:
220 return partial

223def to_railroad(
224 element: pyparsing.ParserElement,
225 diagram_kwargs: typing.Optional[dict] = None,
226 vertical: int = 3,
227 show_results_names: bool = False,
228 show_groups: bool = False,
229 show_hidden: bool = False,
230) -> list[NamedDiagram]:
231 """
232 Convert a pyparsing element tree into a list of diagrams. This is the recommended entrypoint to diagram
233 creation if you want to access the Railroad tree before it is converted to HTML
234 :param element: base element of the parser being diagrammed
235 :param diagram_kwargs: kwargs to pass to the Diagram() constructor
236 :param vertical: (optional) - int - limit at which number of alternatives should be
237 shown vertically instead of horizontally
238 :param show_results_names - bool to indicate whether results name annotations should be
239 included in the diagram
240 :param show_groups - bool to indicate whether groups should be highlighted with an unlabeled
241 surrounding box
242 :param show_hidden - bool to indicate whether internal elements that are typically hidden
243 should be shown
244 """
245 # Convert the whole tree underneath the root
246 lookup = ConverterState(diagram_kwargs=diagram_kwargs or {})
247 _to_diagram_element(
248 element,
249 lookup=lookup,
250 parent=None,
251 vertical=vertical,
252 show_results_names=show_results_names,
253 show_groups=show_groups,
254 show_hidden=show_hidden,
255 )
256
257 root_id = id(element)
258 # Convert the root if it hasn't been already
259 if root_id in lookup:
260 if not element.customName:
261 lookup[root_id].name = ""
262 lookup[root_id].mark_for_extraction(root_id, lookup, force=True)
263
264 # Now that we're finished, we can convert from intermediate structures into Railroad elements
265 diags = list(lookup.diagrams.values())
266 if len(diags) > 1:
267 # collapse out duplicate diags with the same name
268 seen = set()
269 deduped_diags = []
270 for d in diags:
271 # don't extract SkipTo elements, they are uninformative as subdiagrams
272 if d.name == "...":
273 continue
274 if d.name is not None and d.name not in seen:
275 seen.add(d.name)
276 deduped_diags.append(d)
277 resolved = [resolve_partial(partial) for partial in deduped_diags]
278 else:
279 # special case - if just one diagram, always display it, even if
280 # it has no name
281 resolved = [resolve_partial(partial) for partial in diags]
282 return sorted(resolved, key=lambda diag: diag.index)

325 :param el_id: id of the element

400 def extract_into_diagram(self, el_id: int):
401 """
402 Used when we encounter the same token twice in the same tree. When this
403 happens, we replace all instances of that token with a terminal, and
404 create a new subdiagram for the token
405 """
406 position = self[el_id]
407
408 # Replace the original definition of this element with a regular block
409 if position.parent:
410 href = f"#{_make_bookmark(position.name)}"
411 ret = EditablePartial.from_call(railroad.NonTerminal, text=position.name, href=href)
412 if "item" in position.parent.kwargs:
413 position.parent.kwargs["item"] = ret
414 elif "items" in position.parent.kwargs:
415 position.parent.kwargs["items"][position.parent_index] = ret
416
417 # If the element we're extracting is a group, skip to its content but keep the title
418 if position.converted.func == railroad.Group:
419 content = position.converted.kwargs["item"]
420 else:
421 content = position.converted
422
423 self.diagrams[el_id] = EditablePartial.from_call(
424 NamedDiagram,
425 name=position.name,
426 diagram=EditablePartial.from_call(
427 railroad.Diagram, content, **self.diagram_kwargs
428 ),
429 index=position.number,
430 )
431
432 del self[el_id]

435def _worth_extracting(element: pyparsing.ParserElement) -> bool:
436 """
437 Returns true if this element is worth having its own sub-diagram. Simply, if any of its children
438 themselves have children, then its complex enough to extract
439 """
