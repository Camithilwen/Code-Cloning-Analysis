84# --- ENSEMBLE ASSESSMENT ---
85def ensemble_assessment(code1, code2, model_name, prompt, n=3):
86 predictions = []
87 for _ in range(n):
88 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name, prompt)
89 predictions.append((results, predicted_type, predicted_sim))
90 # Majority voting on predicted_sim
91 sim_votes = [pred[2] for pred in predictions]
92 final_sim = max(set(sim_votes), key=sim_votes.count)
93 # Most frequent predicted_type
94 type_votes = [pred[1] for pred in predictions]
95 final_type = max(set(type_votes), key=type_votes.count)
96 # Average results for reporting
97 average_res = dict()
98 for t in ["Type-1", "Type-2", "Type-3", "Type-4"]:
99 sum_yes = 0
100 for pred in predictions:
101 n = pred[0][t]
102 print(f'n: {n}')
103 if n == 'yes' or n=='Yes' or 'yes' in n or 'Yes' in n:
104 n = 1
105 else:
106 n = 0
107 sum_yes += n
108 try:
109 average_res[t] = sum_yes / n
110 except ZeroDivisionError:
111 average_res[t] = 0
112 return average_res, final_type, final_sim

115def determine_similarity(results):
116 if 'yes' in list(results.values()):

118 return 'Type-4', 1

124# --- PROMPT-AWARE ASSESSMENT ---
125def rag_similarity_assessment(code1, code2, model_name, prompt):
126 model = lms.llm(model_name)
127 # Format the prompt with the code snippets
128 prompt_filled = prompt.format(code1=code1, code2=code2)
129 schema = {
130 "type": "object",
131 "properties": {
132 "Type-1": {"type": "string", "enum": ["yes", "no"]},
133 "Type-2": {"type": "string", "enum": ["yes", "no"]},
134 "Type-3": {"type": "string", "enum": ["yes", "no"]},
135 "Type-4": {"type": "string", "enum": ["yes", "no"]}
136 },
137 "required": ["Type-1", "Type-2", "Type-3", "Type-4"]
138}
139
140 response = model.respond(prompt_filled, response_format=schema)
141 results = response.parsed
142 predicted_type, predicted_sim = determine_similarity(results)
143 return results, predicted_type, predicted_sim
