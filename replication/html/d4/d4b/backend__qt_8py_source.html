150def _allow_interrupt_qt(qapp_or_eventloop):
151 """A context manager that allows terminating a plot by sending a SIGINT."""

153 # Use QSocketNotifier to read the socketpair while the Qt event loop runs.
154
155 def prepare_notifier(rsock):
156 sn = QtCore.QSocketNotifier(rsock.fileno(), QtCore.QSocketNotifier.Type.Read)
157
158 @sn.activated.connect
159 def _may_clear_sock():
160 # Running a Python function on socket activation gives the interpreter a
161 # chance to handle the signal in Python land. We also need to drain the
162 # socket with recv() to re-arm it, because it will be written to as part of
163 # the wakeup. (We need this in case set_wakeup_fd catches a signal other
164 # than SIGINT and we shall continue waiting.)
165 try:
166 rsock.recv(1)
167 except BlockingIOError:
168 # This may occasionally fire too soon or more than once on Windows, so
169 # be forgiving about reading an empty socket.
170 pass
171
172 return sn # Actually keep the notifier alive.

176 qapp_or_eventloop.closeAllWindows()

426 try:
427 # for certain keys (enter, left, backspace, etc) use a word for the

723 def _icon(self, name):
724 """
725 Construct a `.QIcon` from an image file *name*, including the extension
726 and relative to Matplotlib's "images" data directory.
727 """
728 # use a high-resolution icon with suffix '_large' if available
729 # note: user-provided icons may not have '_large' versions
730 path_regular = cbook._get_data_path('images', name)
731 path_large = path_regular.with_name(
732 path_regular.name.replace('.png', '_large.png'))
733 filename = str(path_large if path_large.exists() else path_regular)
734
735 pm = QtGui.QPixmap(filename)
736 pm.setDevicePixelRatio(
737 self.devicePixelRatioF() or 1) # rarely, devicePixelRatioF=0
738 if self.palette().color(self.backgroundRole()).value() < 128:
739 icon_color = self.palette().color(self.foregroundRole())
740 mask = pm.createMaskFromColor(
741 QtGui.QColor('black'),
742 QtCore.Qt.MaskMode.MaskOutColor)
743 pm.fill(icon_color)
744 pm.setMask(mask)
745 return QtGui.QIcon(pm)

904 def update_from_current_subplotpars(self):

909 def _export_values(self):

912 self._export_values_dialog = QtWidgets.QDialog()

914 self._export_values_dialog.setLayout(layout)

930 def _on_value_changed(self):
931 spinboxes = self._spinboxes
932 # Set all mins and maxes, so that this can also be used in _reset().
933 for lower, higher in [("bottom", "top"), ("left", "right")]:
934 spinboxes[higher].setMinimum(spinboxes[lower].value() + .001)
935 spinboxes[lower].setMaximum(spinboxes[higher].value() - .001)
936 self._figure.subplots_adjust(
937 **{attr: spinbox.value() for attr, spinbox in spinboxes.items()})
938 self._figure.canvas.draw_idle()

940 def _tight_layout(self):
941 self._figure.tight_layout()
942 for attr, spinbox in self._spinboxes.items():
943 spinbox.blockSignals(True)
944 spinbox.setValue(getattr(self._figure.subplotpars, attr))
945 spinbox.blockSignals(False)
946 self._figure.canvas.draw_idle()
947
948 def _reset(self):
949 for spinbox, value in self._defaults.items():
950 spinbox.setRange(0, 1)
951 spinbox.blockSignals(True)
952 spinbox.setValue(value)
953 spinbox.blockSignals(False)
