54def check_identity_lazy(func, accumulator, location):
55 """Given a function and an accumulator (a list that grows every
56 time the function is called), check that the function can be
57 decorated by memory to be a lazy identity.
58 """
59 # Call each function with several arguments, and check that it is
60 # evaluated only once per argument.
61 memory = Memory(location=location, verbose=0)
62 func = memory.cache(func)
63 for i in range(3):
64 for _ in range(2):
65 assert func(i) == i
66 assert len(accumulator) == i + 1

138@parametrize("call_before_reducing", [True, False])
139def test_parallel_call_cached_function_defined_in_jupyter(tmpdir, call_before_reducing):
140 # Calling an interactively defined memory.cache()'d function inside a
141 # Parallel call used to clear the existing cache related to the said
142 # function (https://github.com/joblib/joblib/issues/1035)
143
144 # This tests checks that this is no longer the case.
145
146 # TODO: test that the cache related to the function cache persists across
147 # ipython sessions (provided that no code change were made to the
148 # function's source)?
149
150 # The first part of the test makes the necessary low-level calls to emulate
151 # the definition of a function in an jupyter notebook cell. Joblib has
152 # some custom code to treat functions defined specifically in jupyter
153 # notebooks/ipython session -- we want to test this code, which requires
154 # the emulation to be rigorous.
155 for session_no in [0, 1]:
156 ipython_cell_source = """
157 def f(x):
158 return x
159 """
160
161 ipython_cell_id = "<ipython-input-{}-000000000000>".format(session_no)
162
163 my_locals = {}
164 exec(
165 compile(
166 textwrap.dedent(ipython_cell_source),
167 filename=ipython_cell_id,
168 mode="exec",
169 ),
170 # TODO when Python 3.11 is the minimum supported version, use
171 # locals=my_locals instead of passing globals and locals in the
172 # next two lines as positional arguments
173 None,
174 my_locals,
175 )
176 f = my_locals["f"]
177 f.__module__ = "__main__"
178
179 # Preliminary sanity checks, and tests checking that joblib properly
180 # identified f as an interactive function defined in a jupyter notebook
181 assert f(1) == 1
182 assert f.__code__.co_filename == ipython_cell_id
183
184 memory = Memory(location=tmpdir.strpath, verbose=0)
185 cached_f = memory.cache(f)
186
187 assert len(os.listdir(tmpdir / "joblib")) == 1
188 f_cache_relative_directory = os.listdir(tmpdir / "joblib")[0]
189 assert "ipython-input" in f_cache_relative_directory
190
191 f_cache_directory = tmpdir / "joblib" / f_cache_relative_directory
192
193 if session_no == 0:
194 # The cache should be empty as cached_f has not been called yet.
195 assert os.listdir(f_cache_directory) == ["f"]
196 assert os.listdir(f_cache_directory / "f") == []
197
198 if call_before_reducing:
199 cached_f(3)
200 # Two files were just created, func_code.py, and a folder
201 # containing the information (inputs hash/ouptput) of
202 # cached_f(3)
203 assert len(os.listdir(f_cache_directory / "f")) == 2
204
205 # Now, testing #1035: when calling a cached function, joblib
206 # used to dynamically inspect the underlying function to
207 # extract its source code (to verify it matches the source code
208 # of the function as last inspected by joblib) -- however,
209 # source code introspection fails for dynamic functions sent to
210 # child processes - which would eventually make joblib clear
211 # the cache associated to f
212 Parallel(n_jobs=2)(delayed(cached_f)(i) for i in [1, 2])
213 else:
214 # Submit the function to the joblib child processes, although
215 # the function has never been called in the parent yet. This
216 # triggers a specific code branch inside
217 # MemorizedFunc.__reduce__.
218 Parallel(n_jobs=2)(delayed(cached_f)(i) for i in [1, 2])
219 # Ensure the child process has time to close the file.
220 # Wait up to 5 seconds for slow CI runs
221 for _ in range(25):
222 if len(os.listdir(f_cache_directory / "f")) == 3:
223 break
224 time.sleep(0.2) # pragma: no cover
225 assert len(os.listdir(f_cache_directory / "f")) == 3
226
227 cached_f(3)
228
229 # Making sure f's cache does not get cleared after the parallel
230 # calls, and contains ALL cached functions calls (f(1), f(2), f(3))
231 # and 'func_code.py'
232 assert len(os.listdir(f_cache_directory / "f")) == 4
233 else:
234 # For the second session, there should be an already existing cache
235 assert len(os.listdir(f_cache_directory / "f")) == 4
236
237 cached_f(3)
238
239 # The previous cache should not be invalidated after calling the
240 # function in a new session
241 assert len(os.listdir(f_cache_directory / "f")) == 4
