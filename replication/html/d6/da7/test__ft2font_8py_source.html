16def test_ft2image_draw_rect_filled():
17 width = 23
18 height = 42
19 for x0, y0, x1, y1 in itertools.product([1, 100], [2, 200], [4, 400], [8, 800]):
20 im = ft2font.FT2Image(width, height)
21 im.draw_rect_filled(x0, y0, x1, y1)
22 a = np.asarray(im)
23 assert a.dtype == np.uint8
24 assert a.shape == (height, width)
25 if x0 == 100 or y0 == 200:
26 # All the out-of-bounds starts should get automatically clipped.
27 assert np.sum(a) == 0
28 else:
29 # Otherwise, ends are clipped to the dimension, but are also _inclusive_.
30 filled = (min(x1 + 1, width) - x0) * (min(y1 + 1, height) - y0)
31 assert np.sum(a) == 255 * filled

134def test_ft2font_invalid_args(tmp_path):
135 # filename argument.
136 with pytest.raises(TypeError, match='to a font file or a binary-mode file object'):
137 ft2font.FT2Font(None)
138 with pytest.raises(TypeError, match='to a font file or a binary-mode file object'):
139 ft2font.FT2Font(object()) # Not bytes or string, and has no read() method.
140 file = tmp_path / 'invalid-font.ttf'
141 file.write_text('This is not a valid font file.')
142 with (pytest.raises(TypeError, match='to a font file or a binary-mode file object'),
143 file.open('rt') as fd):
144 ft2font.FT2Font(fd)
145 with (pytest.raises(TypeError, match='to a font file or a binary-mode file object'),
146 file.open('wt') as fd):
147 ft2font.FT2Font(fd)
148 with (pytest.raises(TypeError, match='to a font file or a binary-mode file object'),
149 file.open('wb') as fd):
150 ft2font.FT2Font(fd)
151
152 file = fm.findfont('DejaVu Sans')
153
154 # hinting_factor argument.
155 with pytest.raises(TypeError, match='incompatible constructor arguments'):
156 ft2font.FT2Font(file, 1.3)
157 with pytest.raises(ValueError, match='hinting_factor must be greater than 0'):
158 ft2font.FT2Font(file, 0)
159
160 with pytest.raises(TypeError, match='incompatible constructor arguments'):
161 # failing to be a list will fail before the 0
162 ft2font.FT2Font(file, _fallback_list=(0,)) # type: ignore[arg-type]
163 with pytest.raises(TypeError, match='incompatible constructor arguments'):
164 ft2font.FT2Font(file, _fallback_list=[0]) # type: ignore[list-item]
165
166 # kerning_factor argument.
167 with pytest.raises(TypeError, match='incompatible constructor arguments'):
168 ft2font.FT2Font(file, _kerning_factor=1.3)

201def test_ft2font_charmaps():
202 def enc(name):
203 # We don't expose the encoding enum from FreeType, but can generate it here.
204 # For DejaVu, there are 5 charmaps, but only 2 have enum entries in FreeType.
205 e = 0
206 for x in name:
207 e <<= 8
208 e += ord(x)
209 return e
210
211 file = fm.findfont('DejaVu Sans')
212 font = ft2font.FT2Font(file)
213 assert font.num_charmaps == 5
214
215 # Unicode.
216 font.select_charmap(enc('unic'))
217 unic = font.get_charmap()
218 font.set_charmap(0) # Unicode platform, Unicode BMP only.
219 after = font.get_charmap()
220 assert len(after) <= len(unic)
221 for chr, glyph in after.items():
222 assert unic[chr] == glyph == font.get_char_index(chr)
223 font.set_charmap(1) # Unicode platform, modern subtable.
224 after = font.get_charmap()
225 assert unic == after
226 font.set_charmap(3) # Windows platform, Unicode BMP only.
227 after = font.get_charmap()
228 assert len(after) <= len(unic)
229 for chr, glyph in after.items():
230 assert unic[chr] == glyph == font.get_char_index(chr)
231 font.set_charmap(4) # Windows platform, Unicode full repertoire, modern subtable.
232 after = font.get_charmap()
233 assert unic == after
234
235 # This is just a random sample from FontForge.
236 glyph_names = {
237 'non-existent-glyph-name': 0,
238 'plusminus': 115,
239 'Racute': 278,
240 'perthousand': 2834,
241 'seveneighths': 3057,
242 'triagup': 3721,
243 'uni01D3': 405,
244 'uni0417': 939,
245 'uni2A02': 4464,
246 'u1D305': 5410,
247 'u1F0A1': 5784,
248 }
249 for name, index in glyph_names.items():
250 assert font.get_name_index(name) == index
251 if name == 'non-existent-glyph-name':
252 name = '.notdef'
253 # This doesn't always apply, but it does for DejaVu Sans.
254 assert font.get_glyph_name(index) == name
255
256 # Apple Roman.
257 font.select_charmap(enc('armn'))
258 armn = font.get_charmap()
259 font.set_charmap(2) # Macintosh platform, Roman.
260 after = font.get_charmap()
261 assert armn == after
262 assert len(armn) <= 256 # 8-bit encoding.
263 # The first 128 characters of Apple Roman match ASCII, which also matches Unicode.
264 for o in range(1, 128):
265 if o not in armn or o not in unic:
266 continue
267 assert unic[o] == armn[o]
268 # Check a couple things outside the ASCII set that are different in each charset.
269 examples = [
270 # (Unicode, Macintosh)
271 (0x2020, 0xA0), # Dagger.
272 (0x00B0, 0xA1), # Degree symbol.
273 (0x00A3, 0xA3), # Pound sign.
274 (0x00A7, 0xA4), # Section sign.
275 (0x00B6, 0xA6), # Pilcrow.
276 (0x221E, 0xB0), # Infinity symbol.
277 ]
278 for u, m in examples:
279 # Though the encoding is different, the glyph should be the same.
280 assert unic[u] == armn[m]
