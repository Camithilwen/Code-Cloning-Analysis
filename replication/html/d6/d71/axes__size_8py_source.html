1"""
2Provides classes of simple units that will be used with `.AxesDivider`
3class (or others) to determine the size of each Axes. The unit
4classes define `get_size` method that returns a tuple of two floats,
5meaning relative and absolute sizes, respectively.
6
7Note that this class is nothing more than a simple tuple of two
8floats. Take a look at the Divider class to see how these two
9values are used.
10
11Once created, the unit classes can be modified by simple arithmetic

112class AxesX(_Base):
113 """
114 Scaled size whose relative part corresponds to the data width
115 of the *axes* multiplied by the *aspect*.
116 """
117
118 def __init__(self, axes, aspect=1., ref_ax=None):
119 self._axes = axes
120 self._aspect = aspect
121 if aspect == "axes" and ref_ax is None:
122 raise ValueError("ref_ax must be set when aspect='axes'")
123 self._ref_ax = ref_ax
124
125 def get_size(self, renderer):
126 l1, l2 = self._axes.get_xlim()
127 if self._aspect == "axes":
128 ref_aspect = _get_axes_aspect(self._ref_ax)
129 aspect = ref_aspect / _get_axes_aspect(self._axes)
130 else:
131 aspect = self._aspect
132
133 rel_size = abs(l2-l1)*aspect
134 abs_size = 0.
135 return rel_size, abs_size

206class Fraction(_Base):
207 """
208 An instance whose size is a *fraction* of the *ref_size*.
209
210 >>> s = Fraction(0.3, AxesX(ax))
211 """
212
213 def __init__(self, fraction, ref_size):
214 _api.check_isinstance(Real, fraction=fraction)
215 self._fraction_ref = ref_size
216 self._fraction = fraction
217
218 def get_size(self, renderer):
219 if self._fraction_ref is None:
220 return self._fraction, 0.
221 else:
222 r, a = self._fraction_ref.get_size(renderer)
223 rel_size = r*self._fraction

228def from_any(size, fraction_ref=None):
229 """
230 Create a Fixed unit when the first argument is a float, or a
231 Fraction unit if that is a string that ends with %. The second
232 argument is only meaningful when Fraction unit is created.
233
234 >>> from mpl_toolkits.axes_grid1.axes_size import from_any
235 >>> a = from_any(1.2) # => Fixed(1.2)
236 >>> from_any("50%", a) # => Fraction(0.5, a)
237 """
238 if isinstance(size, Real):
239 return Fixed(size)
240 elif isinstance(size, str):
241 if size[-1] == "%":
242 return Fraction(float(size[:-1]) / 100, fraction_ref)
243 raise ValueError("Unknown format")

246class _AxesDecorationsSize(_Base):
247 """
248 Fixed size, corresponding to the size of decorations on a given Axes side.
249 """
250
251 _get_size_map = {
252 "left": lambda tight_bb, axes_bb: axes_bb.xmin - tight_bb.xmin,
253 "right": lambda tight_bb, axes_bb: tight_bb.xmax - axes_bb.xmax,
254 "bottom": lambda tight_bb, axes_bb: axes_bb.ymin - tight_bb.ymin,
255 "top": lambda tight_bb, axes_bb: tight_bb.ymax - axes_bb.ymax,
256 }
257
258 def __init__(self, ax, direction):
259 _api.check_in_list(self._get_size_map, direction=direction)
260 self._direction = direction
261 self._ax_list = [ax] if isinstance(ax, Axes) else ax
262
263 def get_size(self, renderer):
264 sz = max([
265 self._get_size_map[self._direction](
266 ax.get_tightbbox(renderer, call_axes_locator=False), ax.bbox)
267 for ax in self._ax_list])
268 dpi = renderer.points_to_pixels(72)
269 abs_size = sz / dpi
270 rel_size = 0
271 return rel_size, abs_size
