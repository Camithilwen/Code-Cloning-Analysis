389
390 # This is a regression test for matplotlib/matplotlib#3196. If one of the
391 # arguments returned by _as_mpl_axes defines __getitem__ but is not
392 # iterable, this would raise an exception. This is because we check
393 # whether the arguments are iterable, and if so we try and convert them
394 # to a tuple. However, the ``iterable`` function returns True if
395 # __getitem__ is present, but some classes can define __getitem__ without
396 # being iterable. The tuple conversion is now done in a try...except in
397 # case it fails.
398
399 class MyAxes(Axes):
400 def __init__(self, *args, myclass=None, **kwargs):
401 Axes.__init__(self, *args, **kwargs)
402
403 class MyClass:
404
405 def __getitem__(self, item):
406 if item != 'a':
407 raise ValueError("item should be a")
408
409 def _as_mpl_axes(self):
410 return MyAxes, {'myclass': self}
411
412 fig = plt.figure()
413 fig.add_subplot(1, 1, 1, projection=MyClass())
414 plt.close(fig)
