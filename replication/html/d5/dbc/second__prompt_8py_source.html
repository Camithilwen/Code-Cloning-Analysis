115def ensemble_assessment(code1, code2, model_name, prompt, n=3):
116 predictions = []
117 for _ in range(n):
118 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name, prompt)
119 predictions.append((results, predicted_type, predicted_sim))
120 # Majority voting on predicted_sim
121 sim_votes = [pred[2] for pred in predictions]
122 final_sim = max(set(sim_votes), key=sim_votes.count)
123 print("sim votes: ", sim_votes)
124 print("final_sim: ", final_sim)
125 # Most frequent predicted_type
126 type_votes = [pred[1] for pred in predictions]
127 final_type = max(set(type_votes), key=type_votes.count)
128 print("type_votes: ", type_votes)
129 print("final_type: ", final_type)
130 # Average results for reporting
131 average_res = dict()
132 for t in ["Type-1", "Type-2", "Type-3", "Type-4"]:
133 sum_yes = 0
134 for pred in predictions:
135 n = pred[0][t]
136 # print(f'n: {n}')
137 if n == 'yes' or n=='Yes' or 'yes' in n or 'Yes' in n:
138 n = 1
139 else:
140 n = 0
141 sum_yes += n
142 try:
143 average_res[t] = sum_yes / n
144 except ZeroDivisionError:
145 average_res[t] = 0
146 print("average res", average_res)
147 for key, value in average_res.items():
148 if value == 0:
149 average_res[key] = "no"
150 else:
151 average_res[key] = "yes"
152 print("average res", average_res)
153 return average_res, final_type, final_sim
