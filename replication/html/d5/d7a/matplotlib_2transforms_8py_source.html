1"""
2Matplotlib includes a framework for arbitrary geometric transformations that is used to
3determine the final position of all elements drawn on the canvas.
4
5Transforms are composed into trees of `TransformNode` objects
6whose actual value depends on their children. When the contents of
7children change, their parents are automatically invalidated. The
8next time an invalidated transform is accessed, it is recomputed to
9reflect those changes. This invalidation/caching approach prevents
10unnecessary recomputations of transforms, and contributes to better
11interactive performance.
12
13For example, here is a graph of the transform tree used to plot data to the figure:
14
15.. graphviz:: /api/transforms.dot
16 :alt: Diagram of transform tree from data to figure coordinates.
17
18The framework can be used for both affine and non-affine
19transformations. However, for speed, we want to use the backend
20renderers to perform affine transformations whenever possible.
21Therefore, it is possible to perform just the affine or non-affine
22part of a transformation on a set of data. The affine is always
23assumed to occur after the non-affine. For any transform::
24
25 full transform == non-affine part + affine part
26
27The backends are not expected to handle non-affine transformations
28themselves.
29
30See the tutorial :ref:`transforms_tutorial` for examples
31of how to use transforms.
32"""

84class TransformNode:
85 """
86 The base class for anything that participates in the transform tree
87 and needs to invalidate its parents or be invalidated. This includes
88 classes that are not really transforms, such as bounding boxes, since some
89 transforms depend on bounding boxes to compute their values.

110 def __init__(self, shorthand_name=None):
111 """
112 Parameters
113 ----------
114 shorthand_name : str
115 A string representing the "name" of the transform. The name carries
116 no significance other than to improve the readability of
117 ``str(transform)`` when DEBUG=True.
118 """

120 # Initially invalid, until first computation.
121 self._invalid = self._INVALID_FULL
122 self._shorthand_name = shorthand_name or ''

129 def __getstate__(self):
130 # turn the dictionary with weak values into a normal dictionary
131 return {**self.__dict__,
132 '_parents': {k: v() for k, v in self._parents.items()}}
133
134 def __setstate__(self, data_dict):
135 self.__dict__ = data_dict
136 # turn the normal dictionary back into a dictionary with weak values
137 # The extra lambda is to provide a callback to remove dead
138 # weakrefs from the dictionary when garbage collection is done.
139 self._parents = {
140 k: weakref.ref(v, lambda _, pop=self._parents.pop, k=k: pop(k))
141 for k, v in self._parents.items() if v is not None}
142
143 def __copy__(self):
144 other = copy.copy(super())
145 # If `c = a + b; a1 = copy(a)`, then modifications to `a1` do not
146 # propagate back to `c`, i.e. we need to clear the parents of `a1`.
147 other._parents = {}
148 # If `c = a + b; c1 = copy(c)`, then modifications to `a` also need to
149 # be propagated to `c1`.
150 for key, val in vars(self).items():
151 if isinstance(val, TransformNode) and id(self) in val._parents:
152 other.set_children(val) # val == getattr(other, key)
153 return other

155 def invalidate(self):
156 """
157 Invalidate this `TransformNode` and triggers an invalidation of its
158 ancestors. Should be called any time the transform changes.
159 """
160 return self._invalidate_internal(
161 level=self._INVALID_AFFINE_ONLY if self.is_affine else self._INVALID_FULL,
162 invalidating_node=self)
163
164 def _invalidate_internal(self, level, invalidating_node):
165 """
166 Called by :meth:`invalidate` and subsequently ascends the transform
167 stack calling each TransformNode's _invalidate_internal method.
168 """
169 # If we are already more invalid than the currently propagated invalidation,
170 # then we don't need to do anything.
171 if level <= self._invalid and not self.pass_through:
172 return
173 self._invalid = level
174 for parent in list(self._parents.values()):
175 parent = parent() # Dereference the weak reference.
176 if parent is not None:
177 parent._invalidate_internal(level=level, invalidating_node=self)

179 def set_children(self, *children):
180 """
181 Set the children of the transform, to let the invalidation
182 system know which transforms can invalidate this transform.
183 Should be called from the constructor of any transforms that
184 depend on other transforms.
185 """
186 # Parents are stored as weak references, so that if the
187 # parents are destroyed, references from the children won't
188 # keep them alive.
189 id_self = id(self)
190 for child in children:
191 # Use weak references so this dictionary won't keep obsolete nodes
192 # alive; the callback deletes the dictionary entry. This is a
193 # performance improvement over using WeakValueDictionary.
194 ref = weakref.ref(
195 self, lambda _, pop=child._parents.pop, k=id_self: pop(k))
196 child._parents[id_self] = ref

668class Bbox(BboxBase):
669 """
670 A mutable bounding box.
671
672 Examples
673 --------
674 **Create from known bounds**
675
676 The default constructor takes the boundary "points" ``[[xmin, ymin],
677 [xmax, ymax]]``.
678
679 >>> Bbox([[1, 1], [3, 7]])
680 Bbox([[1.0, 1.0], [3.0, 7.0]])
681
682 Alternatively, a Bbox can be created from the flattened points array, the
683 so-called "extents" ``(xmin, ymin, xmax, ymax)``
684
685 >>> Bbox.from_extents(1, 1, 3, 7)
686 Bbox([[1.0, 1.0], [3.0, 7.0]])
687
688 or from the "bounds" ``(xmin, ymin, width, height)``.
689
690 >>> Bbox.from_bounds(1, 1, 2, 6)
691 Bbox([[1.0, 1.0], [3.0, 7.0]])
692
693 **Create from collections of points**
694
695 The "empty" object for accumulating Bboxs is the null bbox, which is a
696 stand-in for the empty set.
697
698 >>> Bbox.null()
699 Bbox([[inf, inf], [-inf, -inf]])
700
701 Adding points to the null bbox will give you the bbox of those points.
702
703 >>> box = Bbox.null()
704 >>> box.update_from_data_xy([[1, 1]])
705 >>> box
706 Bbox([[1.0, 1.0], [1.0, 1.0]])
707 >>> box.update_from_data_xy([[2, 3], [3, 2]], ignore=False)
708 >>> box
709 Bbox([[1.0, 1.0], [3.0, 3.0]])
710
711 Setting ``ignore=True`` is equivalent to starting over from a null bbox.
712
713 >>> box.update_from_data_xy([[1, 1]], ignore=True)
714 >>> box
715 Bbox([[1.0, 1.0], [1.0, 1.0]])
716
717 .. warning::
718
719 It is recommended to always specify ``ignore`` explicitly. If not, the
720 default value of ``ignore`` can be changed at any time by code with
721 access to your Bbox, for example using the method `~.Bbox.ignore`.

1435 important for multiple reasons:
1436
1437 - It avoids floating-point inaccuracies when computing the inverse of
1438 B: ``B - B`` is guaranteed to cancel out exactly (resulting in the
1439 identity transform), whereas ``B + B.inverted()`` may differ by a
1440 small epsilon.
1441 - ``B.inverted()`` always returns a frozen transform: if one computes
1442 ``A + B + B.inverted()`` and later mutates ``B``, then
1443 ``B.inverted()`` won't be updated and the last two terms won't cancel
1444 out anymore; on the other hand, ``A + B - B`` will always be equal to
1445 ``A`` even if ``B`` is mutated.
1446 """
1447 # we only know how to do this operation if other is a Transform.
1448 if not isinstance(other, Transform):
1449 return NotImplemented
1450 for remainder, sub_tree in self._iter_break_from_left_to_right():
1451 if sub_tree == other:
1452 return remainder
1453 for remainder, sub_tree in other._iter_break_from_left_to_right():
1454 if sub_tree == self:
1455 if not remainder.has_inverse:
1456 raise ValueError(
1457 "The shortcut cannot be computed since 'other' "
1458 "includes a non-invertible component")
1459 return remainder.inverted()
1460 # if we have got this far, then there was no shortcut possible
1461 if other.has_inverse:
1462 return self + other.inverted()
1463 else:
1464 raise ValueError('It is not possible to compute transA - transB '
1465 'since transB cannot be inverted and there is no '
1466 'shortcut possible.')

1623 def transform_angles(self, angles, pts, radians=False, pushoff=1e-5):
1624 """
1625 Transform a set of angles anchored at specific locations.
1626
1627 Parameters
1628 ----------
1629 angles : (N,) array-like
1630 The angles to transform.
1631 pts : (N, 2) array-like
1632 The points where the angles are anchored.
1633 radians : bool, default: False
1634 Whether *angles* are radians or degrees.
1635 pushoff : float
1636 For each point in *pts* and angle in *angles*, the transformed
1637 angle is computed by transforming a segment of length *pushoff*
1638 starting at that point and making that angle relative to the
1639 horizontal axis, and measuring the angle between the horizontal
1640 axis and the transformed segment.

1643 -------

1646 # Must be 2D

1648 raise NotImplementedError('Only defined in 2D')
1649 angles = np.asarray(angles)
1650 pts = np.asarray(pts)
1651 _api.check_shape((None, 2), pts=pts)
1652 _api.check_shape((None,), angles=angles)
1653 if len(angles) != len(pts):
1654 raise ValueError("There must be as many 'angles' as 'pts'")
1655 # Convert to radians if desired
1656 if not radians:
1657 angles = np.deg2rad(angles)
1658 # Move a short distance away
1659 pts2 = pts + pushoff * np.column_stack([np.cos(angles),
1660 np.sin(angles)])
1661 # Transform both sets of points
1662 tpts = self.transform(pts)
1663 tpts2 = self.transform(pts2)
1664 # Calculate transformed angles
1665 d = tpts2 - tpts
1666 a = np.arctan2(d[:, 1], d[:, 0])
1667 # Convert back to degrees if desired
1668 if not radians:
1669 a = np.rad2deg(a)
1670 return a

2837def nonsingular(vmin, vmax, expander=0.001, tiny=1e-15, increasing=True):
2838 """
2839 Modify the endpoints of a range as needed to avoid singularities.
2840
2841 Parameters
2842 ----------
2843 vmin, vmax : float
2844 The initial endpoints.
2845 expander : float, default: 0.001
2846 Fractional amount by which *vmin* and *vmax* are expanded if
2847 the original interval is too small, based on *tiny*.
2848 tiny : float, default: 1e-15
2849 Threshold for the ratio of the interval to the maximum absolute
2850 value of its endpoints. If the interval is smaller than
2851 this, it will be expanded. This value should be around
2852 1e-15 or larger; otherwise the interval will be approaching
2853 the double precision resolution limit.
2854 increasing : bool, default: True
2855 If True, swap *vmin*, *vmax* if *vmin* > *vmax*.
2856
2857 Returns
2858 -------
2859 vmin, vmax : float
2860 Endpoints, expanded and/or swapped if necessary.
2861 If either input is inf or NaN, or if both inputs are 0 or very
2862 close to zero, it returns -*expander*, *expander*.
2863 """
2864
2865 if (not np.isfinite(vmin)) or (not np.isfinite(vmax)):
2866 return -expander, expander
2867
2868 swapped = False
2869 if vmax < vmin:
2870 vmin, vmax = vmax, vmin
2871 swapped = True
2872
2873 # Expand vmin, vmax to float: if they were integer types, they can wrap
2874 # around in abs (abs(np.int8(-128)) == -128) and vmax - vmin can overflow.
2875 vmin, vmax = map(float, [vmin, vmax])
2876
2877 maxabsvalue = max(abs(vmin), abs(vmax))
2878 if maxabsvalue < (1e6 / tiny) * np.finfo(float).tiny:
2879 vmin = -expander
2880 vmax = expander
2881
2882 elif vmax - vmin <= maxabsvalue * tiny:
2883 if vmax == 0 and vmin == 0:
2884 vmin = -expander
2885 vmax = expander
2886 else:
2887 vmin -= expander*abs(vmin)
2888 vmax += expander*abs(vmax)
2889
2890 if swapped and not increasing:
2891 vmin, vmax = vmax, vmin
2892 return vmin, vmax
