53def script(char):
54 """Return the four-letter script code assigned to the Unicode character
55 'char' as string.
56
57 >>> script("a")
58 'Latn'
59 >>> script(",")
60 'Zyyy'
61 >>> script(chr(0x10FFFF))
62 'Zzzz'
63 """
64 code = byteord(char)
65 # 'bisect_right(a, x, lo=0, hi=len(a))' returns an insertion point which
66 # comes after (to the right of) any existing entries of x in a, and it
67 # partitions array a into two halves so that, for the left side
68 # all(val <= x for val in a[lo:i]), and for the right side
69 # all(val > x for val in a[i:hi]).
70 # Our 'SCRIPT_RANGES' is a sorted list of ranges (only their starting
71 # breakpoints); we want to use `bisect_right` to look up the range that
72 # contains the given codepoint: i.e. whose start is less than or equal
73 # to the codepoint. Thus, we subtract -1 from the index returned.
74 i = bisect_right(Scripts.RANGES, code)
75 return Scripts.VALUES[i - 1]

78def script_extension(char):
79 """Return the script extension property assigned to the Unicode character
80 'char' as a set of string.
81
82 >>> script_extension("a") == {'Latn'}
83 True
84 >>> script_extension(chr(0x060C)) == {'Nkoo', 'Arab', 'Rohg', 'Thaa', 'Syrc', 'Gara', 'Yezi'}
85 True
86 >>> script_extension(chr(0x10FFFF)) == {'Zzzz'}
87 True
88 """
89 code = byteord(char)
90 i = bisect_right(ScriptExtensions.RANGES, code)
91 value = ScriptExtensions.VALUES[i - 1]
92 if value is None:
93 # code points not explicitly listed for Script Extensions
94 # have as their value the corresponding Script property value
95 return {script(char)}
96 return value

248def ot_tags_from_script(script_code):
249 """Return a list of OpenType script tags associated with a given
250 Unicode script code.
251 Return ['DFLT'] script tag for invalid/unknown script codes.
252 """
253 if script_code in OTTags.SCRIPT_EXCEPTIONS:
254 return [OTTags.SCRIPT_EXCEPTIONS[script_code]]
255
256 if script_code not in Scripts.NAMES:
257 return [OTTags.DEFAULT_SCRIPT]
258
259 script_tags = [script_code[0].lower() + script_code[1:]]
260 if script_code in OTTags.NEW_SCRIPT_TAGS:
261 script_tags.extend(OTTags.NEW_SCRIPT_TAGS[script_code])
262 script_tags.reverse() # last in, first out
263
264 return script_tags

267def ot_tag_to_script(tag):
268 """Return the Unicode script code for the given OpenType script tag, or
269 None for "DFLT" tag or if there is no Unicode script associated with it.
270 Raises ValueError if the tag is invalid.
271 """
272 tag = tostr(tag).strip()
273 if not tag or " " in tag or len(tag) > 4:
274 raise ValueError("invalid OpenType tag: %r" % tag)
275
276 if tag in OTTags.SCRIPT_ALIASES:
277 tag = OTTags.SCRIPT_ALIASES[tag]
278
279 while len(tag) != 4:
280 tag += str(" ") # pad with spaces
281
282 if tag == OTTags.DEFAULT_SCRIPT:
283 # it's unclear which Unicode script the "DFLT" OpenType tag maps to,
284 # so here we return None
285 return None
286
287 if tag in OTTags.NEW_SCRIPT_TAGS_REVERSED:
288 return OTTags.NEW_SCRIPT_TAGS_REVERSED[tag]
289
290 if tag in OTTags.SCRIPT_EXCEPTIONS_REVERSED:
291 return OTTags.SCRIPT_EXCEPTIONS_REVERSED[tag]
292
293 # This side of the conversion is fully algorithmic
294
295 # Any spaces at the end of the tag are replaced by repeating the last
296 # letter. Eg 'nko ' -> 'Nkoo'.
297 # Change first char to uppercase
298 script_code = tag[0].upper() + tag[1]
299 for i in range(2, 4):
300 script_code += script_code[i - 1] if tag[i] == " " else tag[i]
301
302 if script_code not in Scripts.NAMES:
303 return None
304 return script_code
