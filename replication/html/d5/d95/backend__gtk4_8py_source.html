94 def _mpl_coords(self, xy=None):
95 """
96 Convert the *xy* position of a GTK event, or of the current cursor
97 position if *xy* is None, to Matplotlib coordinates.
98
99 GTK use logical pixels, but the figure is scaled to physical pixels for
100 rendering. Transform to physical pixels so that all of the down-stream
101 transforms work as expected.
102
103 Also, the origin is different and needs to be corrected.
104 """
105 if xy is None:
106 surface = self.get_native().get_surface()
107 is_over, x, y, mask = surface.get_device_position(
108 self.get_display().get_default_seat().get_pointer())
109 else:
110 x, y = xy
111 x = x * self.device_pixel_ratio
112 # flip y so y=0 is bottom of canvas
113 y = self.figure.bbox.height - y * self.device_pixel_ratio
114 return x, y

188 def _mpl_buttons(self, controller):
189 # NOTE: This spews "Broken accounting of active state" warnings on
190 # right click on macOS.
191 surface = self.get_native().get_surface()
192 is_over, x, y, event_state = surface.get_device_position(
193 self.get_display().get_default_seat().get_pointer())
194 # NOTE: alternatively we could use
195 # event_state = controller.get_current_event_state()
196 # but for button_press/button_release this would report the state
197 # *prior* to the event rather than after it; the above reports the
198 # state *after* it.

200 (MouseButton.LEFT, Gdk.ModifierType.BUTTON1_MASK),
201 (MouseButton.MIDDLE, Gdk.ModifierType.BUTTON2_MASK),
202 (MouseButton.RIGHT, Gdk.ModifierType.BUTTON3_MASK),
203 (MouseButton.BACK, Gdk.ModifierType.BUTTON4_MASK),

223 def _get_key(self, keyval, keycode, state):
224 unikey = chr(Gdk.keyval_to_unicode(keyval))
225 key = cbook._unikey_or_keysym_to_mplkey(
226 unikey,
227 Gdk.keyval_name(keyval))
228 modifiers = [
229 ("ctrl", Gdk.ModifierType.CONTROL_MASK, "control"),
230 ("alt", Gdk.ModifierType.ALT_MASK, "alt"),
231 ("shift", Gdk.ModifierType.SHIFT_MASK, "shift"),
232 ("super", Gdk.ModifierType.SUPER_MASK, "super"),
233 ]
234 mods = [
235 mod for mod, mask, mod_key in modifiers
236 if (mod_key != key and state & mask
237 and not (mod == "shift" and unikey.isprintable()))]
238 return "+".join([*mods, key])

527 def _normalize_shortcut(self, key):
528 """
529 Convert Matplotlib key presses to GTK+ accelerator identifiers.
530
531 Related to `FigureCanvasGTK4._get_key`.
532 """
533 special = {
534 'backspace': 'BackSpace',
535 'pagedown': 'Page_Down',
536 'pageup': 'Page_Up',
537 'scroll_lock': 'Scroll_Lock',
538 }
539
540 parts = key.split('+')
541 mods = ['<' + mod + '>' for mod in parts[:-1]]
542 key = parts[-1]
543
544 if key in special:
545 key = special[key]
546 elif len(key) > 1:
547 key = key.capitalize()
548 elif key.isupper():
549 mods += ['<shift>']
550
551 return ''.join(mods) + key
552
553 def _is_valid_shortcut(self, key):
554 """
555 Check for a valid shortcut to be displayed.
556
557 - GTK will never send 'cmd+' (see `FigureCanvasGTK4._get_key`).
558 - The shortcut window only shows keyboard shortcuts, not mouse buttons.
559 """
560 return 'cmd+' not in key and not key.startswith('MouseButton.')

562 def trigger(self, *args):
563 section = Gtk.ShortcutsSection()
564
565 for name, tool in sorted(self.toolmanager.tools.items()):
566 if not tool.description:
567 continue
568
569 # Putting everything in a separate group allows GTK to
570 # automatically split them into separate columns/pages, which is
571 # useful because we have lots of shortcuts, some with many keys
572 # that are very wide.
573 group = Gtk.ShortcutsGroup()
574 section.append(group)
575 # A hack to remove the title since we have no group naming.
576 child = group.get_first_child()
577 while child is not None:
578 child.set_visible(False)
579 child = child.get_next_sibling()
580
581 shortcut = Gtk.ShortcutsShortcut(
582 accelerator=' '.join(
583 self._normalize_shortcut(key)
584 for key in self.toolmanager.get_tool_keymap(name)
585 if self._is_valid_shortcut(key)),
586 title=tool.name,
587 subtitle=tool.description)
588 group.append(shortcut)
589
590 window = Gtk.ShortcutsWindow(
591 title='Help',
592 modal=True,
593 transient_for=self._figure.canvas.get_root())
594 window.set_child(section)
