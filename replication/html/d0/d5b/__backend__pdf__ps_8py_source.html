23def get_glyphs_subset(fontfile, characters):
24 """
25 Subset a TTF font
26
27 Reads the named fontfile and restricts the font to the characters.
28
29 Parameters
30 ----------
31 fontfile : str
32 Path to the font file
33 characters : str
34 Continuous set of characters to include in subset
35
36 Returns
37 -------
38 fontTools.ttLib.ttFont.TTFont
39 An open font object representing the subset, which needs to
40 be closed by the caller.
41 """
42
43 options = subset.Options(glyph_names=True, recommended_glyphs=True)
44
45 # Prevent subsetting extra tables.
46 options.drop_tables += [
47 'FFTM', # FontForge Timestamp.
48 'PfEd', # FontForge personal table.
49 'BDF', # X11 BDF header.
50 'meta', # Metadata stores design/supported languages (meaningless for subsets).
51 'MERG', # Merge Table.
52 'TSIV', # Microsoft Visual TrueType extension.
53 'Zapf', # Information about the individual glyphs in the font.
54 'bdat', # The bitmap data table.
55 'bloc', # The bitmap location table.
56 'cidg', # CID to Glyph ID table (Apple Advanced Typography).
57 'fdsc', # The font descriptors table.
58 'feat', # Feature name table (Apple Advanced Typography).
59 'fmtx', # The Font Metrics Table.
60 'fond', # Data-fork font information (Apple Advanced Typography).
61 'just', # The justification table (Apple Advanced Typography).
62 'kerx', # An extended kerning table (Apple Advanced Typography).
63 'ltag', # Language Tag.
64 'morx', # Extended Glyph Metamorphosis Table.
65 'trak', # Tracking table.
66 'xref', # The cross-reference table (some Apple font tooling information).
67 ]
68 # if fontfile is a ttc, specify font number
69 if fontfile.endswith(".ttc"):
70 options.font_number = 0
71
72 font = subset.load_font(fontfile, options)
73 subsetter = subset.Subsetter(options=options)
74 subsetter.populate(text=characters)
75 subsetter.subset(font)
76 return font

109 def track(self, font, s):
110 """Record that string *s* is being typeset using font *font*."""
111 char_to_font = font._get_fontmap(s)
112 for _c, _f in char_to_font.items():
113 self.used.setdefault(_f.fname, set()).add(ord(_c))
