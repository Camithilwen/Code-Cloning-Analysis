110 # notice that if we just looked at the control points to get the bounding

384@pytest.mark.parametrize('phi', np.concatenate([
385 np.array([0, 15, 30, 45, 60, 75, 90, 105, 120, 135]) + delta
386 for delta in [-1, 0, 1]]))
387def test_path_intersect_path(phi):
388 # test for the range of intersection angles
389 eps_array = [1e-5, 1e-8, 1e-10, 1e-12]
390
391 transform = transforms.Affine2D().rotate(np.deg2rad(phi))
392
393 # a and b intersect at angle phi
394 a = Path([(-2, 0), (2, 0)])
395 b = transform.transform_path(a)
396 assert a.intersects_path(b) and b.intersects_path(a)
397
398 # a and b touch at angle phi at (0, 0)
399 a = Path([(0, 0), (2, 0)])
400 b = transform.transform_path(a)
401 assert a.intersects_path(b) and b.intersects_path(a)
402
403 # a and b are orthogonal and intersect at (0, 3)
404 a = transform.transform_path(Path([(0, 1), (0, 3)]))
405 b = transform.transform_path(Path([(1, 3), (0, 3)]))
406 assert a.intersects_path(b) and b.intersects_path(a)
407
408 # a and b are collinear and intersect at (0, 3)
409 a = transform.transform_path(Path([(0, 1), (0, 3)]))
410 b = transform.transform_path(Path([(0, 5), (0, 3)]))
411 assert a.intersects_path(b) and b.intersects_path(a)
412
413 # self-intersect
414 assert a.intersects_path(a)
415
416 # a contains b
417 a = transform.transform_path(Path([(0, 0), (5, 5)]))
418 b = transform.transform_path(Path([(1, 1), (3, 3)]))
419 assert a.intersects_path(b) and b.intersects_path(a)
420
421 # a and b are collinear but do not intersect
422 a = transform.transform_path(Path([(0, 1), (0, 5)]))
423 b = transform.transform_path(Path([(3, 0), (3, 3)]))
424 assert not a.intersects_path(b) and not b.intersects_path(a)
425
426 # a and b are on the same line but do not intersect
427 a = transform.transform_path(Path([(0, 1), (0, 5)]))
428 b = transform.transform_path(Path([(0, 6), (0, 7)]))
429 assert not a.intersects_path(b) and not b.intersects_path(a)
430
431 # Note: 1e-13 is the absolute tolerance error used for
432 # `isclose` function from src/_path.h
433
434 # a and b are parallel but do not touch
435 for eps in eps_array:
436 a = transform.transform_path(Path([(0, 1), (0, 5)]))
437 b = transform.transform_path(Path([(0 + eps, 1), (0 + eps, 5)]))
438 assert not a.intersects_path(b) and not b.intersects_path(a)
439
440 # a and b are on the same line but do not intersect (really close)
441 for eps in eps_array:
442 a = transform.transform_path(Path([(0, 1), (0, 5)]))
443 b = transform.transform_path(Path([(0, 5 + eps), (0, 7)]))
444 assert not a.intersects_path(b) and not b.intersects_path(a)
445
446 # a and b are on the same line and intersect (really close)
447 for eps in eps_array:
448 a = transform.transform_path(Path([(0, 1), (0, 5)]))
449 b = transform.transform_path(Path([(0, 5 - eps), (0, 7)]))
450 assert a.intersects_path(b) and b.intersects_path(a)
451
452 # b is the same as a but with an extra point
453 a = transform.transform_path(Path([(0, 1), (0, 5)]))
454 b = transform.transform_path(Path([(0, 1), (0, 2), (0, 5)]))
455 assert a.intersects_path(b) and b.intersects_path(a)
456
457 # a and b are collinear but do not intersect
458 a = transform.transform_path(Path([(1, -1), (0, -1)]))
459 b = transform.transform_path(Path([(0, 1), (0.9, 1)]))
460 assert not a.intersects_path(b) and not b.intersects_path(a)
461
462 # a and b are collinear but do not intersect
463 a = transform.transform_path(Path([(0., -5.), (1., -5.)]))
464 b = transform.transform_path(Path([(1., 5.), (0., 5.)]))
465 assert not a.intersects_path(b) and not b.intersects_path(a)
