42def get_dir_vector(zdir):
43 """
44 Return a direction vector.
45
46 Parameters
47 ----------
48 zdir : {'x', 'y', 'z', None, 3-tuple}
49 The direction. Possible values are:
50
51 - 'x': equivalent to (1, 0, 0)
52 - 'y': equivalent to (0, 1, 0)
53 - 'z': equivalent to (0, 0, 1)
54 - *None*: equivalent to (0, 0, 0)
55 - an iterable (x, y, z) is converted to an array
56
57 Returns
58 -------
59 x, y, z : array
60 The direction vector.
61 """
62 if zdir == 'x':
63 return np.array((1, 0, 0))
64 elif zdir == 'y':
65 return np.array((0, 1, 0))
66 elif zdir == 'z':
67 return np.array((0, 0, 1))
68 elif zdir is None:
69 return np.array((0, 0, 0))
70 elif np.iterable(zdir) and len(zdir) == 3:
71 return np.array(zdir)
72 else:
73 raise ValueError("'x', 'y', 'z', None or vector of length 3 expected")

76def _viewlim_mask(xs, ys, zs, axes):
77 """
78 Return original points with points outside the axes view limits masked.
79
80 Parameters
81 ----------
82 xs, ys, zs : array-like
83 The points to mask.
84 axes : Axes3D
85 The axes to use for the view limits.
86
87 Returns
88 -------
89 xs_masked, ys_masked, zs_masked : np.ma.array
90 The masked points.
91 """
92 mask = np.logical_or.reduce((xs < axes.xy_viewLim.xmin,
93 xs > axes.xy_viewLim.xmax,
94 ys < axes.xy_viewLim.ymin,
95 ys > axes.xy_viewLim.ymax,
96 zs < axes.zz_viewLim.xmin,
97 zs > axes.zz_viewLim.xmax))
98 xs_masked = np.ma.array(xs, mask=mask)
99 ys_masked = np.ma.array(ys, mask=mask)
100 zs_masked = np.ma.array(zs, mask=mask)
101 return xs_masked, ys_masked, zs_masked

1124 # force update of color mapping because we re-order them

1138 if self._vec.shape[0] == 4: # Will be 3 (xyz) or 4 (xyzw)

1140 vec = np.ma.array([xs, ys, zs, w_masked])
1141 else:

1294 """Modify the alphas of the color list according to depth."""
1295 # FIXME: This only works well if the points for *zs* are well-spaced
1296 # in all three dimensions. Otherwise, at certain orientations,
1297 # the min and max zs are very close together.
1298 # Should really normalize against the viewing depth.
1299 if len(colors) == 0 or len(zs) == 0:

1310 ignored.
1311
1312 Parameters
1313 ----------
1314 xs, ys, zs : array-like
1315 The x, y, and z coordinates of the points.
1316 atol : float, default: 1e-8
1317 The tolerance for the equality check.
1318 """
1319 xs, ys, zs = np.asarray(xs), np.asarray(ys), np.asarray(zs)
1320 points = np.column_stack([xs, ys, zs])
1321 points = points[~np.isnan(points).any(axis=1)]
1322 # Check for the case where we have less than 3 unique points
1323 points = np.unique(points, axis=0)
1324 if len(points) <= 3:
1325 return True
1326 # Calculate the vectors from the first point to all other points
1327 vs = (points - points[0])[1:]
1328 vs = vs / np.linalg.norm(vs, axis=1)[:, np.newaxis]
1329 # Filter out parallel vectors
1330 vs = np.unique(vs, axis=0)
1331 if len(vs) <= 2:
1332 return True
1333 # Filter out parallel and antiparallel vectors to the first vector
1334 cross_norms = np.linalg.norm(np.cross(vs[0], vs[1:]), axis=1)
1335 zero_cross_norms = np.where(np.isclose(cross_norms, 0, atol=atol))[0] + 1
1336 vs = np.delete(vs, zero_cross_norms, axis=0)
1337 if len(vs) <= 2:
1338 return True
1339 # Calculate the normal vector from the first three points
1340 n = np.cross(vs[0], vs[1])
1341 n = n / np.linalg.norm(n)
1342 # If the dot product of the normal vector and all other vectors is zero,
1343 # all points are on the same plane
1344 dots = np.dot(n, vs.transpose())
1345 return np.allclose(dots, 0, atol=atol)

1348def _generate_normals(polygons):
1349 """
1350 Compute the normals of a list of polygons, one normal per polygon.
1351
1352 Normals point towards the viewer for a face with its vertices in
1353 counterclockwise order, following the right hand rule.
1354
1355 Uses three points equally spaced around the polygon. This method assumes
1356 that the points are in a plane. Otherwise, more than one shade is required,
1357 which is not supported.
1358
1359 Parameters
1360 ----------
1361 polygons : list of (M_i, 3) array-like, or (..., M, 3) array-like
1362 A sequence of polygons to compute normals for, which can have
1363 varying numbers of vertices. If the polygons all have the same
1364 number of vertices and array is passed, then the operation will
1365 be vectorized.
1366
1367 Returns
1368 -------
1369 normals : (..., 3) array
1370 A normal vector estimated for the polygon.
1371 """
1372 if isinstance(polygons, np.ndarray):
1373 # optimization: polygons all have the same number of points, so can
1374 # vectorize
1375 n = polygons.shape[-2]
1376 i1, i2, i3 = 0, n//3, 2*n//3
1377 v1 = polygons[..., i1, :] - polygons[..., i2, :]
1378 v2 = polygons[..., i2, :] - polygons[..., i3, :]
1379 else:
1380 # The subtraction doesn't vectorize because polygons is jagged.
1381 v1 = np.empty((len(polygons), 3))
1382 v2 = np.empty((len(polygons), 3))
1383 for poly_i, ps in enumerate(polygons):
1384 n = len(ps)
1385 ps = np.asarray(ps)
1386 i1, i2, i3 = 0, n//3, 2*n//3
1387 v1[poly_i, :] = ps[i1, :] - ps[i2, :]
1388 v2[poly_i, :] = ps[i2, :] - ps[i3, :]
1389 return np.cross(v1, v2)

1392def _shade_colors(color, normals, lightsource=None):
1393 """
1394 Shade *color* using normal vectors given by *normals*,
1395 assuming a *lightsource* (using default position if not given).
1396 *color* can also be an array of the same length as *normals*.
1397 """
1398 if lightsource is None:
1399 # chosen for backwards-compatibility
1400 lightsource = mcolors.LightSource(azdeg=225, altdeg=19.4712)
1401
1402 with np.errstate(invalid="ignore"):
1403 shade = ((normals / np.linalg.norm(normals, axis=1, keepdims=True))
1404 @ lightsource.direction)
1405 mask = ~np.isnan(shade)
1406
1407 if mask.any():
1408 # convert dot product to allowed shading fractions
1409 in_norm = mcolors.Normalize(-1, 1)
1410 out_norm = mcolors.Normalize(0.3, 1).inverse
1411
1412 def norm(x):
1413 return out_norm(in_norm(x))
1414
1415 shade[~mask] = 0
1416
1417 color = mcolors.to_rgba_array(color)
1418 # shape of color should be (M, 4) (where M is number of faces)
1419 # shape of shade should be (M,)
1420 # colors should have final shape of (M, 4)
1421 alpha = color[:, 3]
1422 colors = norm(shade)[:, np.newaxis] * color
1423 colors[:, 3] = alpha
1424 else:
1425 colors = np.asanyarray(color).copy()
1426
