34def parse_json_response(response_text):
35 """
36 Enhanced JSON parsing with multiple fallback strategies
37 """
38 # First, try direct JSON parsing
39 try:
40 return json.loads(response_text.strip())
41 except json.JSONDecodeError:
42 pass
43
44 # Try to extract JSON from text using regex
45 json_patterns = [
46 r'\{[^{}]*"Type-1"[^{}]*\}', # Simple JSON object
47 r'\{.*?"Type-1".*?\}', # More flexible JSON
48 r'\{.*\}', # Any JSON-like structure
49 ]
50
51 for pattern in json_patterns:
52 matches = re.findall(pattern, response_text, re.DOTALL)
53 for match in matches:
54 try:
55 result = json.loads(match)
56 if all(key in result for key in ["Type-1", "Type-2", "Type-3", "Type-4"]):
57 return result
58 except json.JSONDecodeError:
59 continue
60
61 # Try to parse line by line for structured output
62 lines = response_text.split('\n')
63 result = {}
64 for line in lines:
65 for type_name in ["Type-1", "Type-2", "Type-3", "Type-4"]:
66 if type_name in line:
67 # Extract number from line
68 numbers = re.findall(r'(\d*\.?\d+)', line)
69 if numbers:
70 try:
71 score = float(numbers[0])
72 if 0 <= score <= 1:
73 result[type_name] = score
74 except ValueError:
75 pass
76
77 # If we found all required keys, return the result
78 if len(result) == 4:
79 return result
80
81 # Final fallback: return default values
82 print(f"Failed to parse JSON response: {response_text[:200]}...")
83 return {"Type-1": 0.0, "Type-2": 0.0, "Type-3": 0.0, "Type-4": 0.0}

86def ensemble_assessment(code1, code2, model_name, n=3):
87 predictions = []

92 print(f"Ensemble {i+1}: {predicted_type}, sim={predicted_sim}")
93 except Exception as e:
94 print(f"Error in ensemble assessment {i+1}: {e}")
95 # Add default prediction
96 default_results = {"Type-1": 0.0, "Type-2": 0.0, "Type-3": 0.0, "Type-4": 0.0}
97 predictions.append((default_results, "Non-clone", 0))
98
99 if not predictions:
100 # Complete fallback
101 default_results = {"Type-1": 0.0, "Type-2": 0.0, "Type-3": 0.0, "Type-4": 0.0}
102 return default_results, "Non-clone", 0
103
104 # Majority voting on predicted_sim
105 sim_votes = [pred[2] for pred in predictions]
106 final_sim = max(set(sim_votes), key=sim_votes.count)
107
108 # Most frequent predicted_type
109 type_votes = [pred[1] for pred in predictions]
110 final_type = max(set(type_votes), key=type_votes.count)
111
112 # Average results for reporting
113 avg_results = {}
114 for t in ["Type-1", "Type-2", "Type-3", "Type-4"]:
115 scores = [pred[0][t] for pred in predictions if t in pred[0]]
116 avg_results[t] = sum(scores) / len(scores) if scores else 0.0
