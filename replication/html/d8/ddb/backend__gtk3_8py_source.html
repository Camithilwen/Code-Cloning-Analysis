101 def _mpl_coords(self, event=None):
102 """
103 Convert the position of a GTK event, or of the current cursor position
104 if *event* is None, to Matplotlib coordinates.
105
106 GTK use logical pixels, but the figure is scaled to physical pixels for
107 rendering. Transform to physical pixels so that all of the down-stream
108 transforms work as expected.
109
110 Also, the origin is different and needs to be corrected.
111 """
112 if event is None:
113 window = self.get_window()
114 t, x, y, state = window.get_device_position(
115 window.get_display().get_device_manager().get_client_pointer())
116 else:
117 x, y = event.x, event.y
118 x = x * self.device_pixel_ratio
119 # flip y so y=0 is bottom of canvas
120 y = self.figure.bbox.height - y * self.device_pixel_ratio
121 return x, y

209 def _get_key(self, event):
210 unikey = chr(Gdk.keyval_to_unicode(event.keyval))
211 key = cbook._unikey_or_keysym_to_mplkey(
212 unikey, Gdk.keyval_name(event.keyval))
213 mods = self._mpl_modifiers(event.state, exclude=key)
214 if "shift" in mods and unikey.isprintable():
215 mods.remove("shift")
216 return "+".join([*mods, key])

487 def _normalize_shortcut(self, key):
488 """
489 Convert Matplotlib key presses to GTK+ accelerator identifiers.
490
491 Related to `FigureCanvasGTK3._get_key`.
492 """
493 special = {
494 'backspace': 'BackSpace',
495 'pagedown': 'Page_Down',
496 'pageup': 'Page_Up',
497 'scroll_lock': 'Scroll_Lock',
498 }
499
500 parts = key.split('+')
501 mods = ['<' + mod + '>' for mod in parts[:-1]]
502 key = parts[-1]
503
504 if key in special:
505 key = special[key]
506 elif len(key) > 1:
507 key = key.capitalize()
508 elif key.isupper():
509 mods += ['<shift>']
510
511 return ''.join(mods) + key
512
513 def _is_valid_shortcut(self, key):
514 """
515 Check for a valid shortcut to be displayed.
516
517 - GTK will never send 'cmd+' (see `FigureCanvasGTK3._get_key`).
518 - The shortcut window only shows keyboard shortcuts, not mouse buttons.
519 """
520 return 'cmd+' not in key and not key.startswith('MouseButton.')

522 def _show_shortcuts_window(self):
523 section = Gtk.ShortcutsSection()
524
525 for name, tool in sorted(self.toolmanager.tools.items()):
526 if not tool.description:
527 continue
528
529 # Putting everything in a separate group allows GTK to
530 # automatically split them into separate columns/pages, which is
531 # useful because we have lots of shortcuts, some with many keys
532 # that are very wide.
533 group = Gtk.ShortcutsGroup()
534 section.add(group)
535 # A hack to remove the title since we have no group naming.
536 group.forall(lambda widget, data: widget.set_visible(False), None)
537
538 shortcut = Gtk.ShortcutsShortcut(
539 accelerator=' '.join(
540 self._normalize_shortcut(key)
541 for key in self.toolmanager.get_tool_keymap(name)
542 if self._is_valid_shortcut(key)),
543 title=tool.name,
544 subtitle=tool.description)
545 group.add(shortcut)
546
547 window = Gtk.ShortcutsWindow(
548 title='Help',
549 modal=True,
550 transient_for=self._figure.canvas.get_toplevel())
551 section.show() # Must be done explicitly before add!
552 window.add(section)
553
554 window.show_all()
