1"""
2Add a ``figure-mpl`` directive that is a responsive version of ``figure``.
3
4This implementation is very similar to ``.. figure::``, except it also allows a

32class FigureMpl(Figure):
33 """
34 Implements a directive to allow an optional hidpi image.
35
36 Meant to be used with the *plot_srcset* configuration option in conf.py,
37 and gets set in the TEMPLATE of plot_directive.py
38
39 e.g.::
40
41 .. figure-mpl:: plot_directive/some_plots-1.png
42 :alt: bar
43 :srcset: plot_directive/some_plots-1.png,
44 plot_directive/some_plots-1.2x.png 2.00x
45 :class: plot-directive
46
47 The resulting html (at ``some_plots.html``) is::
48
49 <img src="sphx_glr_bar_001_hidpi.png"
50 srcset="_images/some_plot-1.png,
51 _images/some_plots-1.2x.png 2.00x",
52 alt="bar"
53 class="plot_directive" />
54
55 Note that the handling of subdirectories is different than that used by the sphinx
56 figure directive::
57
58 .. figure-mpl:: plot_directive/nestedpage/index-1.png
59 :alt: bar
60 :srcset: plot_directive/nestedpage/index-1.png
61 plot_directive/nestedpage/index-1.2x.png 2.00x
62 :class: plot_directive
63
64 The resulting html (at ``nestedpage/index.html``)::
65
66 <img src="../_images/nestedpage-index-1.png"
67 srcset="../_images/nestedpage-index-1.png,
68 ../_images/_images/nestedpage-index-1.2x.png 2.00x",
69 alt="bar"
70 class="sphx-glr-single-img" />
71
72 where the subdirectory is included in the image name for uniqueness.
73 """
74
75 has_content = False
76 required_arguments = 1
77 optional_arguments = 2
78 final_argument_whitespace = False
79 option_spec = {
80 'alt': directives.unchanged,
81 'height': directives.length_or_unitless,
82 'width': directives.length_or_percentage_or_unitless,
83 'scale': directives.nonnegative_int,
84 'align': Image.align,
85 'class': directives.class_option,
86 'caption': directives.unchanged,
87 'srcset': directives.unchanged,
88 }

108 image_node['srcset'] = self.options.get('srcset', None)
109

113def _parse_srcsetNodes(st):
114 """
115 parse srcset...
116 """
117 entries = st.split(',')
118 srcset = {}
119 for entry in entries:
120 spl = entry.strip().split(' ')
121 if len(spl) == 1:
122 srcset[0] = spl[0]
123 elif len(spl) == 2:
124 mult = spl[1][:-1]
125 srcset[float(mult)] = spl[0]
126 else:
127 raise ExtensionError(f'srcset argument "{entry}" is invalid.')
128 return srcset

131def _copy_images_figmpl(self, node):
132
133 # these will be the temporary place the plot-directive put the images eg:
134 # ../../../build/html/plot_directive/users/explain/artists/index-1.png
135 if node['srcset']:
136 srcset = _parse_srcsetNodes(node['srcset'])
137 else:
138 srcset = None
139
140 # the rst file's location: eg /Users/username/matplotlib/doc/users/explain/artists
141 docsource = PurePath(self.document['source']).parent
142
143 # get the relpath relative to root:
144 srctop = self.builder.srcdir
145 rel = relpath(docsource, srctop).replace('.', '').replace(os.sep, '-')
146 if len(rel):
147 rel += '-'
148 # eg: users/explain/artists
149
150 imagedir = PurePath(self.builder.outdir, self.builder.imagedir)
151 # eg: /Users/username/matplotlib/doc/build/html/_images/users/explain/artists
152
153 Path(imagedir).mkdir(parents=True, exist_ok=True)
154
155 # copy all the sources to the imagedir:
156 if srcset:
157 for src in srcset.values():
158 # the entries in srcset are relative to docsource's directory
159 abspath = PurePath(docsource, src)
160 name = rel + abspath.name
161 shutil.copyfile(abspath, imagedir / name)
162 else:
163 abspath = PurePath(docsource, node['uri'])
164 name = rel + abspath.name
165 shutil.copyfile(abspath, imagedir / name)
166
167 return imagedir, srcset, rel

170def visit_figmpl_html(self, node):
171
172 imagedir, srcset, rel = _copy_images_figmpl(self, node)
173
174 # /doc/examples/subd/plot_1.rst
175 docsource = PurePath(self.document['source'])
176 # /doc/
177 # make sure to add the trailing slash:
178 srctop = PurePath(self.builder.srcdir, '')
179 # examples/subd/plot_1.rst
180 relsource = relpath(docsource, srctop)
181 # /doc/build/html
182 desttop = PurePath(self.builder.outdir, '')
183 # /doc/build/html/examples/subd
184 dest = desttop / relsource
185
186 # ../../_images/ for dirhtml and ../_images/ for html
187 imagerel = PurePath(relpath(imagedir, dest.parent)).as_posix()
188 if self.builder.name == "dirhtml":
189 imagerel = f'..{imagerel}'
190
191 # make uri also be relative...
192 nm = PurePath(node['uri'][1:]).name
193 uri = f'{imagerel}/{rel}{nm}'
194 img_attrs = {'src': uri, 'alt': node['alt']}
195
196 # make srcset str. Need to change all the prefixes!
197 maxsrc = uri
198 if srcset:
199 maxmult = -1
200 srcsetst = ''
201 for mult, src in srcset.items():
202 nm = PurePath(src[1:]).name
203 # ../../_images/plot_1_2_0x.png
204 path = f'{imagerel}/{rel}{nm}'
205 srcsetst += path
206 if mult == 0:
207 srcsetst += ', '
208 else:
209 srcsetst += f' {mult:1.2f}x, '
210
211 if mult > maxmult:
212 maxmult = mult
213 maxsrc = path
214
215 # trim trailing comma and space...
216 img_attrs['srcset'] = srcsetst[:-2]
217
218 if node['class'] is not None:
219 img_attrs['class'] = ' '.join(node['class'])
220 for style in ['width', 'height', 'scale']:
221 if node[style]:
222 if 'style' not in img_attrs:
223 img_attrs['style'] = f'{style}: {node[style]};'
224 else:
225 img_attrs['style'] += f'{style}: {node[style]};'
226
227 # <figure class="align-default" id="id1">
228 # <a class="reference internal image-reference" href="_images/index-1.2x.png">
229 # <img alt="_images/index-1.2x.png"
230 # src="_images/index-1.2x.png" style="width: 53%;" />
231 # </a>
232 # <figcaption>
233 # <p><span class="caption-text">Figure caption is here....</span>
234 # <a class="headerlink" href="#id1" title="Permalink to this image">#</a></p>
235 # </figcaption>
236 # </figure>
237 self.body.append(
238 self.starttag(
239 node, 'figure',
240 CLASS=f'align-{node["align"]}' if node['align'] else 'align-center'))
241 self.body.append(
242 self.starttag(node, 'a', CLASS='reference internal image-reference',
243 href=maxsrc) +
244 self.emptytag(node, 'img', **img_attrs) +
245 '</a>\n')
246 if node['caption']:
247 self.body.append(self.starttag(node, 'figcaption'))
248 self.body.append(self.starttag(node, 'p'))
249 self.body.append(self.starttag(node, 'span', CLASS='caption-text'))
250 self.body.append(node['caption'])
251 self.body.append('</span></p></figcaption>\n')
252 self.body.append('</figure>\n')

255def visit_figmpl_latex(self, node):
256
257 if node['srcset'] is not None:
258 imagedir, srcset = _copy_images_figmpl(self, node)
259 maxmult = -1
260 # choose the highest res version for latex:
261 maxmult = max(srcset, default=-1)
262 node['uri'] = PurePath(srcset[maxmult]).name
