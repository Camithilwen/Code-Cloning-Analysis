26def compare_figure(fname, savefig_kwargs={}, tol=0):
27 actual = os.path.join(result_dir, fname)
28 plt.savefig(actual, **savefig_kwargs)
29
30 expected = os.path.join(result_dir, "expected_%s" % fname)
31 shutil.copyfile(os.path.join(baseline_dir, fname), expected)
32 err = compare_images(expected, actual, tol=tol)
33 if err:
34 raise ImageComparisonFailure(err)
