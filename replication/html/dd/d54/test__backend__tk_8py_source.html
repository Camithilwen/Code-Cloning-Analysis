17def _isolated_tk_test(success_count, func=None):
18 """
19 A decorator to run *func* in a subprocess and assert that it prints
20 "success" *success_count* times and nothing on stderr.
21
22 TkAgg tests seem to have interactions between tests, so isolate each test
23 in a subprocess. See GH#18261
24 """
25
26 if func is None:
27 return functools.partial(_isolated_tk_test, success_count)
28
29 if "MPL_TEST_ESCAPE_HATCH" in os.environ:
30 # set in subprocess_run_helper() below
31 return func
32
33 @pytest.mark.skipif(
34 not importlib.util.find_spec('tkinter'),
35 reason="missing tkinter"
36 )
37 @pytest.mark.skipif(
38 sys.platform == "linux" and not _c_internal_utils.xdisplay_is_valid(),
39 reason="$DISPLAY is unset"
40 )
41 @pytest.mark.xfail( # https://github.com/actions/setup-python/issues/649
42 ('TF_BUILD' in os.environ or 'GITHUB_ACTION' in os.environ) and
43 sys.platform == 'darwin' and sys.version_info[:2] < (3, 11),
44 reason='Tk version mismatch on Azure macOS CI'
45 )
46 @functools.wraps(func)
47 def test_func():
48 # even if the package exists, may not actually be importable this can
49 # be the case on some CI systems.
50 pytest.importorskip('tkinter')
51 try:
52 proc = subprocess_run_helper(
53 func, timeout=_test_timeout, extra_env=dict(
54 MPLBACKEND="TkAgg", MPL_TEST_ESCAPE_HATCH="1"))
55 except subprocess.TimeoutExpired:
56 pytest.fail("Subprocess timed out")
57 except subprocess.CalledProcessError as e:
58 pytest.fail("Subprocess failed to test intended behavior\n"
59 + str(e.stderr))
60 else:
61 # macOS may actually emit irrelevant errors about Accelerated
62 # OpenGL vs. software OpenGL, or some permission error on Azure, so
63 # suppress them.
64 # Asserting stderr first (and printing it on failure) should be
65 # more helpful for debugging that printing a failed success count.
66 ignored_lines = ["OpenGL", "CFMessagePort: bootstrap_register",
67 "/usr/include/servers/bootstrap_defs.h"]
68 assert not [line for line in proc.stderr.splitlines()
69 if all(msg not in line for msg in ignored_lines)]
70 assert proc.stdout.count("success") == success_count
71
72 return test_func
