55def truncate_code(code, max_lines=50):

68def ensemble_assessment(code1, code2, model_name, n=3):
69 predictions = []
70 for _ in range(n):
71 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name)
72 predictions.append((results, predicted_type, predicted_sim))
73 # Majority voting on predicted_sim
74 sim_votes = [pred[2] for pred in predictions]
75 final_sim = max(set(sim_votes), key=sim_votes.count)
76 # Most frequent predicted_type
77 type_votes = [pred[1] for pred in predictions]
78 final_type = max(set(type_votes), key=type_votes.count)
79 # Average results for reporting
80 avg_results = {t: sum(pred[0][t] for pred in predictions) / n for t in ["Type-1", "Type-2", "Type-3", "Type-4"]}
81 return avg_results, final_type, final_sim

102def rag_similarity_assessment(code1, code2, model_name):
103 prompt = f"""You are a code similarity expert. Analyze the following code pair step by step for clone detection.
104
105Definitions:
106- Type-1: Identical except for whitespace, comments, layout.
107- Type-2: Identical except for variable/function names (plus Type-1 differences).
108- Type-3: Similar, but with some statements added/removed/modified (plus Type-2 differences).
109- Type-4: Syntactically different but functionally identical (same outputs for same inputs).
110
111Examples:
112Type-1: 'int a=5;' vs 'int a = 5; // set a'
113Type-2: 'int a=5;' vs 'int b=5;'
114Type-3: 'int a=5; print(a);' vs 'int a=5;'
115Type-4: 'for(int i=0;i<5;i++)sum+=i;' vs 'sum = sum_of_first_n(5);'
116
117Step-by-step:
1181. Are the outputs always identical for all inputs? (Type-4)
1192. Are they identical except for whitespace/comments? (Type-1)
1203. Are names changed but structure identical? (Type-2)
1214. Are there statement-level edits? (Type-3)
122Explain your reasoning for each type and provide a confidence score (0-1).
123
124Target Code:
125{code1}
126
127Similar Code:
128{code2}
129
130Respond ONLY with a JSON object with the following keys: Type-1, Type-2, Type-3, Type-4, each with a confidence score between 0 and 1.
131"""
132
133 # Call Ollama to get the response
134 result = ollama.generate(
135 model=model_name,
136 prompt=prompt
137 )
138 response_text = result['response']
139
140 # Parse the response as JSON
141 try:
142 results = json.loads(response_text)
143 except Exception:
144 # Fallback: handle parsing error
145 results = {"Type-1": 0, "Type-2": 0, "Type-3": 0, "Type-4": 0}
146
147 predicted_type, predicted_sim = determine_similarity(results)
148 return results, predicted_type, predicted_sim
