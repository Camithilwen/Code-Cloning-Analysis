29def _contour_labeler_event_handler(cs, inline, inline_spacing, event):
30 canvas = cs.axes.get_figure(root=True).canvas
31 is_button = event.name == "button_press_event"
32 is_key = event.name == "key_press_event"
33 # Quit (even if not in infinite mode; this is consistent with
34 # MATLAB and sometimes quite useful, but will require the user to
35 # test how many points were actually returned before using data).
36 if (is_button and event.button == MouseButton.MIDDLE

52 if cs.axes.contains(event)[0]:

246 def locate_label(self, linecontour, labelwidth):
247 """
248 Find good place to draw a label (relatively flat part of the contour).
249 """
250 ctr_size = len(linecontour)
251 n_blocks = int(np.ceil(ctr_size / labelwidth)) if labelwidth > 1 else 1
252 block_size = ctr_size if n_blocks == 1 else int(labelwidth)
253 # Split contour into blocks of length ``block_size``, filling the last
254 # block by cycling the contour start (per `np.resize` semantics). (Due
255 # to cycling, the index returned is taken modulo ctr_size.)
256 xx = np.resize(linecontour[:, 0], (n_blocks, block_size))
257 yy = np.resize(linecontour[:, 1], (n_blocks, block_size))
258 yfirst = yy[:, :1]
259 ylast = yy[:, -1:]
260 xfirst = xx[:, :1]
261 xlast = xx[:, -1:]
262 s = (yfirst - yy) * (xlast - xfirst) - (xfirst - xx) * (ylast - yfirst)
263 l = np.hypot(xlast - xfirst, ylast - yfirst)
264 # Ignore warning that divide by zero throws, as this is a valid option
265 with np.errstate(divide='ignore', invalid='ignore'):
266 distances = (abs(s) / l).sum(axis=-1)
267 # Labels are drawn in the middle of the block (``hbsize``) where the
268 # contour is the closest (per ``distances``) to a straight line, but
269 # not `too_close()` to a preexisting label.
270 hbsize = block_size // 2
271 adist = np.argsort(distances)
272 # If all candidates are `too_close()`, go back to the straightest part
273 # (``adist[0]``).
274 for idx in np.append(adist, adist[0]):
275 x, y = xx[idx, hbsize], yy[idx, hbsize]
276 if not self.too_close(x, y, labelwidth):
277 break
278 return x, y, (idx * block_size + hbsize) % ctr_size

280 def _split_path_and_get_label_rotation(self, path, idx, screen_pos, lw, spacing=5):
281 """
282 Prepare for insertion of a label at index *idx* of *path*.
283
284 Parameters
285 ----------
286 path : Path
287 The path where the label will be inserted, in data space.
288 idx : int
289 The vertex index after which the label will be inserted.
290 screen_pos : (float, float)
291 The position where the label will be inserted, in screen space.
292 lw : float
293 The label width, in screen space.
294 spacing : float
295 Extra spacing around the label, in screen space.
296
297 Returns
298 -------
299 path : Path
300 The path, broken so that the label can be drawn over it.
301 angle : float
302 The rotation of the label.
303
304 Notes
305 -----
306 Both tasks are done together to avoid calculating path lengths multiple times,
307 which is relatively costly.
308
309 The method used here involves computing the path length along the contour in
310 pixel coordinates and then looking (label width / 2) away from central point to
311 determine rotation and then to break contour if desired. The extra spacing is
312 taken into account when breaking the path, but not when computing the angle.
313 """
314 xys = path.vertices
315 codes = path.codes
316
317 # Insert a vertex at idx/pos (converting back to data space), if there isn't yet
318 # a vertex there. With infinite precision one could also always insert the
319 # extra vertex (it will get masked out by the label below anyways), but floating
320 # point inaccuracies (the point can have undergone a data->screen->data
321 # transform loop) can slightly shift the point and e.g. shift the angle computed
322 # below from exactly zero to nonzero.
323 pos = self.get_transform().inverted().transform(screen_pos)
324 if not np.allclose(pos, xys[idx]):
325 xys = np.insert(xys, idx, pos, axis=0)
326 codes = np.insert(codes, idx, Path.LINETO)
327
328 # Find the connected component where the label will be inserted. Note that a
329 # path always starts with a MOVETO, and we consider there's an implicit
330 # MOVETO (closing the last path) at the end.
331 movetos = (codes == Path.MOVETO).nonzero()[0]
332 start = movetos[movetos <= idx][-1]
333 try:
334 stop = movetos[movetos > idx][0]
335 except IndexError:
336 stop = len(codes)
337
338 # Restrict ourselves to the connected component.
339 cc_xys = xys[start:stop]
340 idx -= start
341
342 # If the path is closed, rotate it s.t. it starts at the label.
343 is_closed_path = codes[stop - 1] == Path.CLOSEPOLY
344 if is_closed_path:
345 cc_xys = np.concatenate([cc_xys[idx:-1], cc_xys[:idx+1]])
346 idx = 0
347
348 # Like np.interp, but additionally vectorized over fp.
349 def interp_vec(x, xp, fp): return [np.interp(x, xp, col) for col in fp.T]
350
351 # Use cumulative path lengths ("cpl") as curvilinear coordinate along contour.
352 screen_xys = self.get_transform().transform(cc_xys)
353 path_cpls = np.insert(
354 np.cumsum(np.hypot(*np.diff(screen_xys, axis=0).T)), 0, 0)
355 path_cpls -= path_cpls[idx]
356
357 # Use linear interpolation to get end coordinates of label.
358 target_cpls = np.array([-lw/2, lw/2])
359 if is_closed_path: # For closed paths, target from the other end.
360 target_cpls[0] += (path_cpls[-1] - path_cpls[0])
361 (sx0, sx1), (sy0, sy1) = interp_vec(target_cpls, path_cpls, screen_xys)
362 angle = np.rad2deg(np.arctan2(sy1 - sy0, sx1 - sx0)) # Screen space.
363 if self.rightside_up: # Fix angle so text is never upside-down
364 angle = (angle + 90) % 180 - 90
365
366 target_cpls += [-spacing, +spacing] # Expand range by spacing.
367
368 # Get indices near points of interest; use -1 as out of bounds marker.
369 i0, i1 = np.interp(target_cpls, path_cpls, range(len(path_cpls)),
370 left=-1, right=-1)
371 i0 = math.floor(i0)
372 i1 = math.ceil(i1)
373 (x0, x1), (y0, y1) = interp_vec(target_cpls, path_cpls, cc_xys)
374
375 # Actually break contours (dropping zero-len parts).
376 new_xy_blocks = []
377 new_code_blocks = []
378 if is_closed_path:
379 if i0 != -1 and i1 != -1:
380 # This is probably wrong in the case that the entire contour would
381 # be discarded, but ensures that a valid path is returned and is
382 # consistent with behavior of mpl <3.8
383 points = cc_xys[i1:i0+1]
384 new_xy_blocks.extend([[(x1, y1)], points, [(x0, y0)]])
385 nlines = len(points) + 1
386 new_code_blocks.extend([[Path.MOVETO], [Path.LINETO] * nlines])
387 else:
388 if i0 != -1:
389 new_xy_blocks.extend([cc_xys[:i0 + 1], [(x0, y0)]])
390 new_code_blocks.extend([[Path.MOVETO], [Path.LINETO] * (i0 + 1)])
391 if i1 != -1:
392 new_xy_blocks.extend([[(x1, y1)], cc_xys[i1:]])
393 new_code_blocks.extend([
394 [Path.MOVETO], [Path.LINETO] * (len(cc_xys) - i1)])
395
396 # Back to the full path.
397 xys = np.concatenate([xys[:start], *new_xy_blocks, xys[stop:]])
398 codes = np.concatenate([codes[:start], *new_code_blocks, codes[stop:]])
399
400 return angle, Path(xys, codes)

503def _find_closest_point_on_path(xys, p):
504 """
505 Parameters
506 ----------
507 xys : (N, 2) array-like
508 Coordinates of vertices.
509 p : (float, float)
510 Coordinates of point.
511
512 Returns
513 -------
514 d2min : float
515 Minimum square distance of *p* to *xys*.
516 proj : (float, float)
517 Projection of *p* onto *xys*.
518 imin : (int, int)
519 Consecutive indices of vertices of segment in *xys* where *proj* is.
520 Segments are considered as including their end-points; i.e. if the
521 closest point on the path is a node in *xys* with index *i*, this
522 returns ``(i-1, i)``. For the special case where *xys* is a single
523 point, this returns ``(0, 0)``.
524 """
525 if len(xys) == 1:
526 return (((p - xys[0]) ** 2).sum(), xys[0], (0, 0))
527 dxys = xys[1:] - xys[:-1] # Individual segment vectors.
528 norms = (dxys ** 2).sum(axis=1)
529 norms[norms == 0] = 1 # For zero-length segment, replace 0/0 by 0/1.
530 rel_projs = np.clip( # Project onto each segment in relative 0-1 coords.
531 ((p - xys[:-1]) * dxys).sum(axis=1) / norms,
532 0, 1)[:, None]
533 projs = xys[:-1] + rel_projs * dxys # Projs. onto each segment, in (x, y).
534 d2s = ((projs - p) ** 2).sum(axis=1) # Squared distances.
535 imin = np.argmin(d2s)
536 return (d2s[imin], projs[imin], (imin, imin+1))

876 def _process_args(self, *args, **kwargs):
877 """
878 Process *args* and *kwargs*; override in derived classes.
879
880 Must set self.levels, self.zmin and self.zmax, and update Axes limits.
881 """
882 self.levels = args[0]
883 allsegs = args[1]
884 allkinds = args[2] if len(args) > 2 else None
885 self.zmax = np.max(self.levels)
886 self.zmin = np.min(self.levels)
887
888 if allkinds is None:
889 allkinds = [[None] * len(segs) for segs in allsegs]
890
891 # Check lengths of levels and allsegs.
892 if self.filled:
893 if len(allsegs) != len(self.levels) - 1:
894 raise ValueError('must be one less number of segments as '
895 'levels')
896 else:
897 if len(allsegs) != len(self.levels):
898 raise ValueError('must be same number of segments as levels')
899
900 # Check length of allkinds.
901 if len(allkinds) != len(allsegs):
902 raise ValueError('allkinds has different length to allsegs')
903
904 # Determine x, y bounds and update axes data limits.
905 flatseglist = [s for seg in allsegs for s in seg]
906 points = np.concatenate(flatseglist, axis=0)
907 self._mins = points.min(axis=0)
908 self._maxs = points.max(axis=0)

910 # Each entry in (allsegs, allkinds) is a list of (segs, kinds): segs is a list
911 # of (N, 2) arrays of xy coordinates, kinds is a list of arrays of corresponding
912 # pathcodes. However, kinds can also be None; in which case all paths in that
913 # list are codeless (this case is normalized above). These lists are used to
914 # construct paths, which then get concatenated.
915 self._paths = [Path.make_compound_path(*map(Path, segs, kinds))
916 for segs, kinds in zip(allsegs, allkinds)]
917
918 return kwargs

964 def _autolev(self, N):
965 """
966 Select contour levels to span the data.
967
968 The target number of levels, *N*, is used only when the
969 scale is not log and default locator is used.
970
971 We need two more levels for filled contours than for
972 line contours, because for the latter we need to specify
973 the lower and upper boundary of each range. For example,
974 a single contour boundary, say at z = 0, requires only
975 one contour line, but two filled regions, and therefore
976 three levels to provide boundaries for both regions.
977 """
978 if self.locator is None:
979 if self.logscale:
980 self.locator = ticker.LogLocator()
981 else:
982 self.locator = ticker.MaxNLocator(N + 1, min_n_ticks=1)
983
984 lev = self.locator.tick_values(self.zmin, self.zmax)
985
986 try:
987 if self.locator._symmetric:
988 return lev
989 except AttributeError:
990 pass
991
992 # Trim excess levels the locator may have supplied.
993 under = np.nonzero(lev < self.zmin)[0]
994 i0 = under[-1] if len(under) else 0
995 over = np.nonzero(lev > self.zmax)[0]
996 i1 = over[0] + 1 if len(over) else len(lev)
997 if self.extend in ('min', 'both'):
998 i0 += 1
999 if self.extend in ('max', 'both'):
1000 i1 -= 1
1001
1002 if i1 - i0 < 3:
1003 i0, i1 = 0, len(lev)
1004
1005 return lev[i0:i1]

1032 def _process_levels(self):
1033 """
1034 Assign values to :attr:`layers` based on :attr:`levels`,
1035 adding extended layers as needed if contours are filled.
1036
1037 For line contours, layers simply coincide with levels;
1038 a line is a thin layer. No extended levels are needed
1039 with line contours.
1040 """
1041 # Make a private _levels to include extended regions; we
1042 # want to leave the original levels attribute unchanged.
1043 # (Colorbar needs this even for line contours.)
1044 self._levels = list(self.levels)
1045
1046 if self.logscale:
1047 lower, upper = 1e-250, 1e250
1048 else:
1049 lower, upper = -1e250, 1e250
1050
1051 if self.extend in ('both', 'min'):
1052 self._levels.insert(0, lower)
1053 if self.extend in ('both', 'max'):
1054 self._levels.append(upper)
1055 self._levels = np.asarray(self._levels)
1056
1057 if not self.filled:
1058 self.layers = self.levels
1059 return
1060
1061 # Layer values are mid-way between levels in screen space.
1062 if self.logscale:
1063 # Avoid overflow by taking sqrt before multiplying.
1064 self.layers = (np.sqrt(self._levels[:-1])
1065 * np.sqrt(self._levels[1:]))
1066 else:
1067 self.layers = 0.5 * (self._levels[:-1] + self._levels[1:])

1069 def _process_colors(self):
1070 """
1071 Color argument processing for contouring.
1072
1073 Note that we base the colormapping on the contour levels
1074 and layers, not on the actual range of the Z values. This
1075 means we don't have to worry about bad values in Z, and we
1076 always have the full dynamic range available for the selected
1077 levels.
1078
1079 The color is based on the midpoint of the layer, except for
1080 extended end layers. By default, the norm vmin and vmax
1081 are the extreme values of the non-extended levels. Hence,
1082 the layer color extremes are not the extreme values of
1083 the colormap itself, but approach those values as the number
1084 of levels increases. An advantage of this scheme is that
1085 line contours, when added to filled contours, take on
1086 colors that are consistent with those of the filled regions;
1087 for example, a contour line on the boundary between two
1088 regions will have a color intermediate between those
1089 of the regions.
1090
1091 """
1092 self.monochrome = self.cmap.monochrome
1093 if self.colors is not None:
1094 # Generate integers for direct indexing.
1095 i0, i1 = 0, len(self.levels)
1096 if self.filled:
1097 i1 -= 1
1098 # Out of range indices for over and under:
1099 if self.extend in ('both', 'min'):
1100 i0 -= 1
1101 if self.extend in ('both', 'max'):
1102 i1 += 1
1103 self.cvalues = list(range(i0, i1))
1104 self.set_norm(mcolors.NoNorm())
1105 else:
1106 self.cvalues = self.layers
1107 self.norm.autoscale_None(self.levels)
1108 self.set_array(self.cvalues)
1109 self.update_scalarmappable()
1110 if self.extend in ('both', 'max', 'min'):
1111 self.norm.clip = False

1149 def _find_nearest_contour(self, xy, indices=None):
1150 """
1151 Find the point in the unfilled contour plot that is closest (in screen
1152 space) to point *xy*.
1153
1154 Parameters
1155 ----------
1156 xy : tuple[float, float]
1157 The reference point (in screen space).
1158 indices : list of int or None, default: None
1159 Indices of contour levels to consider. If None (the default), all levels
1160 are considered.
1161
1162 Returns
1163 -------
1164 idx_level_min : int
1165 The index of the contour level closest to *xy*.
1166 idx_vtx_min : int
1167 The index of the `.Path` segment closest to *xy* (at that level).
1168 proj : (float, float)
1169 The point in the contour plot closest to *xy*.
1170 """
1171
1172 # Convert each contour segment to pixel coordinates and then compare the given
1173 # point to those coordinates for each contour. This is fast enough in normal
1174 # cases, but speedups may be possible.
1175
1176 if self.filled:
1177 raise ValueError("Method does not support filled contours")
1178
1179 if indices is None:
1180 indices = range(len(self._paths))
1181
1182 d2min = np.inf
1183 idx_level_min = idx_vtx_min = proj_min = None
1184
1185 for idx_level in indices:
1186 path = self._paths[idx_level]
1187 idx_vtx_start = 0
1188 for subpath in path._iter_connected_components():
1189 if not len(subpath.vertices):
1190 continue
1191 lc = self.get_transform().transform(subpath.vertices)
1192 d2, proj, leg = _find_closest_point_on_path(lc, xy)
1193 if d2 < d2min:
1194 d2min = d2
1195 idx_level_min = idx_level
1196 idx_vtx_min = leg[1] + idx_vtx_start
1197 proj_min = proj
1198 idx_vtx_start += len(subpath)
1199
1200 return idx_level_min, idx_vtx_min, proj_min

1202 def find_nearest_contour(self, x, y, indices=None, pixel=True):
1203 """
1204 Find the point in the contour plot that is closest to ``(x, y)``.
1205
1206 This method does not support filled contours.
1207
1208 Parameters
1209 ----------
1210 x, y : float
1211 The reference point.
1212 indices : list of int or None, default: None
1213 Indices of contour levels to consider. If None (the default), all
1214 levels are considered.
1215 pixel : bool, default: True
1216 If *True*, measure distance in pixel (screen) space, which is
1217 useful for manual contour labeling; else, measure distance in axes
1218 space.
1219
1220 Returns
1221 -------
1222 path : int
1223 The index of the path that is closest to ``(x, y)``. Each path corresponds
1224 to one contour level.
1225 subpath : int
1226 The index within that closest path of the subpath that is closest to
1227 ``(x, y)``. Each subpath corresponds to one unbroken contour line.
1228 index : int
1229 The index of the vertices within that subpath that are closest to
1230 ``(x, y)``.
1231 xmin, ymin : float
1232 The point in the contour plot that is closest to ``(x, y)``.
1233 d2 : float
1234 The squared distance from ``(xmin, ymin)`` to ``(x, y)``.
1235 """
1236 segment = index = d2 = None
1237
1238 with ExitStack() as stack:
1239 if not pixel:
1240 # _find_nearest_contour works in pixel space. We want axes space, so
1241 # effectively disable the transformation here by setting to identity.
1242 stack.enter_context(self._cm_set(
1243 transform=mtransforms.IdentityTransform()))
1244
1245 i_level, i_vtx, (xmin, ymin) = self._find_nearest_contour((x, y), indices)
1246
1247 if i_level is not None:
1248 cc_cumlens = np.cumsum(
1249 [*map(len, self._paths[i_level]._iter_connected_components())])
1250 segment = cc_cumlens.searchsorted(i_vtx, "right")
1251 index = i_vtx if segment == 0 else i_vtx - cc_cumlens[segment - 1]
1252 d2 = (xmin-x)**2 + (ymin-y)**2
1253
1254 return (i_level, segment, index, xmin, ymin, d2)

1416 def _initialize_x_y(self, z):
1417 """
1418 Return X, Y arrays such that contour(Z) will match imshow(Z)
1419 if origin is not None.
1420 The center of pixel Z[i, j] depends on origin:
1421 if origin is None, x = j, y = i;
1422 if origin is 'lower', x = j + 0.5, y = i + 0.5;
1423 if origin is 'upper', x = j + 0.5, y = Nrows - i - 0.5
1424 If extent is not None, x and y will be scaled to match,
1425 as in imshow.
1426 If origin is None and extent is not None, then extent
1427 will give the minimum and maximum values of x and y.
1428 """
1429 if z.ndim != 2:
1430 raise TypeError(f"Input z must be 2D, not {z.ndim}D")
1431 elif z.shape[0] < 2 or z.shape[1] < 2:
1432 raise TypeError(f"Input z must be at least a (2, 2) shaped array, "
1433 f"but has shape {z.shape}")
1434 else:
1435 Ny, Nx = z.shape
1436 if self.origin is None: # Not for image-matching.
1437 if self.extent is None:
1438 return np.meshgrid(np.arange(Nx), np.arange(Ny))
1439 else:
1440 x0, x1, y0, y1 = self.extent
1441 x = np.linspace(x0, x1, Nx)
1442 y = np.linspace(y0, y1, Ny)
1443 return np.meshgrid(x, y)
1444 # Match image behavior:
1445 if self.extent is None:
1446 x0, x1, y0, y1 = (0, Nx, 0, Ny)
1447 else:
1448 x0, x1, y0, y1 = self.extent
1449 dx = (x1 - x0) / Nx
1450 dy = (y1 - y0) / Ny
1451 x = x0 + (np.arange(Nx) + 0.5) * dx
1452 y = y0 + (np.arange(Ny) + 0.5) * dy
1453 if self.origin == 'upper':
1454 y = y[::-1]
1455 return np.meshgrid(x, y)
