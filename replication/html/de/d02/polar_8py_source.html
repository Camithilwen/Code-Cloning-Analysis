35
36 This transform maps polar coordinates :math:`\theta, r` into Cartesian
37 coordinates :math:`x, y = r \cos(\theta), r \sin(\theta)`
38 (but does not fully transform into Axes coordinates or

42 along the radial axis (e.g. log-scaling) has been applied to the input
43 data.

157class PolarAffine(mtransforms.Affine2DBase):
158 r"""
159 The affine part of the polar projection.
160
161 Scales the output so that maximum radius rests on the edge of the Axes
162 circle and the origin is mapped to (0.5, 0.5). The transform applied is
163 the same to x and y components and given by:
164
165 .. math::
166
167 x_{1} = 0.5 \left [ \frac{x_{0}}{(r_{\max} - r_{\min})} + 1 \right ]
168
169 :math:`r_{\min}, r_{\max}` are the minimum and maximum radial limits after
170 any scaling (e.g. log scaling) has been removed.
171 """
172 def __init__(self, scale_transform, limits):
173 """
174 Parameters
175 ----------
176 scale_transform : `~matplotlib.transforms.Transform`
177 Scaling transform for the data. This is used to remove any scaling
178 from the radial view limits.
179 limits : `~matplotlib.transforms.BboxBase`
180 View limits of the data. The only part of its bounds that is used
181 is the y limits (for the radius limits).
182 """
183 super().__init__()
184 self._scale_transform = scale_transform
185 self._limits = limits
186 self.set_children(scale_transform, limits)
187 self._mtx = None
188
189 __str__ = mtransforms._make_str_method("_scale_transform", "_limits")
190
191 def get_matrix(self):
192 # docstring inherited
193 if self._invalid:
194 limits_scaled = self._limits.transformed(self._scale_transform)
195 yscale = limits_scaled.ymax - limits_scaled.ymin
196 affine = mtransforms.Affine2D() \
197 .scale(0.5 / yscale) \
198 .translate(0.5, 0.5)

756class _WedgeBbox(mtransforms.Bbox):
757 """
758 Transform (theta, r) wedge Bbox into Axes bounding box.
759
760 Parameters
761 ----------
762 center : (float, float)
763 Center of the wedge
764 viewLim : `~matplotlib.transforms.Bbox`
765 Bbox determining the boundaries of the wedge
766 originLim : `~matplotlib.transforms.Bbox`
767 Bbox determining the origin for the wedge, if different from *viewLim*
768 """
769 def __init__(self, center, viewLim, originLim, **kwargs):
770 super().__init__([[0, 0], [1, 1]], **kwargs)
771 self._center = center
772 self._viewLim = viewLim
773 self._originLim = originLim
774 self.set_children(viewLim, originLim)

776 __str__ = mtransforms._make_str_method("_center", "_viewLim", "_originLim")

778 def get_points(self):
779 # docstring inherited
780 if self._invalid:
781 points = self._viewLim.get_points().copy()
782 # Scale angular limits to work with Wedge.
783 points[:, 0] *= 180 / np.pi
784 if points[0, 0] > points[1, 0]:
785 points[:, 0] = points[::-1, 0]
786
787 # Scale radial limits based on origin radius.
788 points[:, 1] -= self._originLim.y0
789
790 # Scale radial limits to match axes limits.
791 rscale = 0.5 / points[1, 1]
792 points[:, 1] *= rscale
793 width = min(points[1, 1] - points[0, 1], 0.5)
794
795 # Generate bounding box for wedge.
796 wedge = mpatches.Wedge(self._center, points[1, 1],
797 points[0, 0], points[1, 0],
798 width=width)
799 self.update_from_path(wedge.get_path())
800
801 # Ensure equal aspect ratio.
802 w, h = self._points[1] - self._points[0]
803 deltah = max(w - h, 0) / 2
804 deltaw = max(h - w, 0) / 2
805 self._points += np.array([[-deltaw, -deltah], [deltaw, deltah]])
806
807 self._invalid = 0
808
809 return self._points

861 def _set_lim_and_transforms(self):
862 # A view limit where the minimum radius can be locked if the user
863 # specifies an alternate origin.
864 self._originViewLim = mtransforms.LockableBbox(self.viewLim)
865
866 # Handle angular offset and direction.
867 self._direction = mtransforms.Affine2D() \
868 .scale(self._default_theta_direction, 1.0)
869 self._theta_offset = mtransforms.Affine2D() \
870 .translate(self._default_theta_offset, 0.0)
871 self.transShift = self._direction + self._theta_offset
872 # A view limit shifted to the correct location after accounting for
873 # orientation and offset.
874 self._realViewLim = mtransforms.TransformedBbox(self.viewLim,
875 self.transShift)
876
877 # Transforms the x and y axis separately by a scale factor
878 # It is assumed that this part will have non-linear components
879 self.transScale = mtransforms.TransformWrapper(
880 mtransforms.IdentityTransform())
881
882 # Scale view limit into a bbox around the selected wedge. This may be
883 # smaller than the usual unit axes rectangle if not plotting the full
884 # circle.
885 self.axesLim = _WedgeBbox((0.5, 0.5),
886 self._realViewLim, self._originViewLim)
887
888 # Scale the wedge to fill the axes.
889 self.transWedge = mtransforms.BboxTransformFrom(self.axesLim)
890
891 # Scale the axes to fill the figure.
892 self.transAxes = mtransforms.BboxTransformTo(self.bbox)
893
894 # A (possibly non-linear) projection on the (already scaled)
895 # data. This one is aware of rmin
896 self.transProjection = self.PolarTransform(
897 self,
898 apply_theta_transforms=False,
899 scale_transform=self.transScale
900 )
901 # Add dependency on rorigin.
902 self.transProjection.set_children(self._originViewLim)
903
904 # An affine transformation on the data, generally to limit the
905 # range of the axes
906 self.transProjectionAffine = self.PolarAffine(self.transScale,
907 self._originViewLim)
908
909 # The complete data transformation stack -- from data all the
910 # way to display coordinates
911 #
912 # 1. Remove any radial axis scaling (e.g. log scaling)
913 # 2. Shift data in the theta direction
914 # 3. Project the data from polar to cartesian values
915 # (with the origin in the same place)
916 # 4. Scale and translate the cartesian values to Axes coordinates
917 # (here the origin is moved to the lower left of the Axes)
918 # 5. Move and scale to fill the Axes
919 # 6. Convert from Axes coordinates to Figure coordinates
920 self.transData = (
921 self.transScale +
922 self.transShift +
923 self.transProjection +
924 (
925 self.transProjectionAffine +
926 self.transWedge +
927 self.transAxes
928 )
929 )
930
931 # This is the transform for theta-axis ticks. It is
932 # equivalent to transData, except it always puts r == 0.0 and r == 1.0
933 # at the edge of the axis circles.
934 self._xaxis_transform = (
935 mtransforms.blended_transform_factory(
936 mtransforms.IdentityTransform(),
937 mtransforms.BboxTransformTo(self.viewLim)) +
938 self.transData)

940 # the outside by default. This should work the same as before.

944 .translate(0.0, 0.5)

949 # thetamax.

951 mtransforms.blended_transform_factory(
952 mtransforms.BboxTransformTo(self.viewLim),
953 mtransforms.IdentityTransform()) +
954 self.transData)

956 self._r_label_position = mtransforms.Affine2D() \
957 .translate(self._default_rlabel_position, 0.0)
958 self._yaxis_text_transform = mtransforms.TransformWrapper(
959 self._r_label_position + self.transData)

1078 def set_thetalim(self, *args, **kwargs):
1079 r"""
1080 Set the minimum and maximum theta values.
1081
1082 Can take the following signatures:
1083
1084 - ``set_thetalim(minval, maxval)``: Set the limits in radians.
1085 - ``set_thetalim(thetamin=minval, thetamax=maxval)``: Set the limits
1086 in degrees.
1087
1088 where minval and maxval are the minimum and maximum limits. Values are
1089 wrapped in to the range :math:`[0, 2\pi]` (in radians), so for example
1090 it is possible to do ``set_thetalim(-np.pi / 2, np.pi / 2)`` to have
1091 an axis symmetric around 0. A ValueError is raised if the absolute
1092 angle difference is larger than a full circle.
1093 """
1094 orig_lim = self.get_xlim() # in radians
1095 if 'thetamin' in kwargs:
1096 kwargs['xmin'] = np.deg2rad(kwargs.pop('thetamin'))

1098 kwargs['xmax'] = np.deg2rad(kwargs.pop('thetamax'))

1100 # Parsing all permutations of *args, **kwargs is tricky; it is simpler
1101 # to let set_xlim() do it and then validate the limits.
1102 if abs(new_max - new_min) > 2 * np.pi:
1103 self.set_xlim(orig_lim) # un-accept the change
1104 raise ValueError("The angle range must be less than a full circle")

1424 def format_coord(self, theta, r):
1425 # docstring inherited
1426 screen_xy = self.transData.transform((theta, r))
1427 screen_xys = screen_xy + np.stack(
1428 np.meshgrid([-1, 0, 1], [-1, 0, 1])).reshape((2, -1)).T
1429 ts, rs = self.transData.inverted().transform(screen_xys).T
1430 delta_t = abs((ts - theta + np.pi) % (2 * np.pi) - np.pi).max()
1431 delta_t_halfturns = delta_t / np.pi
1432 delta_t_degrees = delta_t_halfturns * 180
1433 delta_r = abs(rs - r).max()

1435 theta += 2 * np.pi

1441 # notation doesn't make sense and the trailing dot is ugly.
