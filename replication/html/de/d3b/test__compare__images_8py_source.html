14 [

18 ('basn3p02.png', 'basn3p02-minorchange.png', 10, None),
19 # Now test with no tolerance.
20 ('basn3p02.png', 'basn3p02-minorchange.png', 0, 6.50646),
21 # Comparison with an image that is shifted by 1px in the X axis.
22 ('basn3p02.png', 'basn3p02-1px-offset.png', 0, 90.15611),

25 ('basn3p02.png', 'basn3p02-half-1px-offset.png', 0, 63.75),
26 # Comparison of an image and the same image scrambled.
27 # This expects the images to compare completely different, with a very
28 # large RMS.
29 # Note: The image has been scrambled in a specific way, by having
30 # each color component of each pixel randomly placed somewhere in the

32 # value of R, G and B, but in a totally different position.
33 # Test with no tolerance to make sure that we pick up even a very small
34 # RMS error.
35 ('basn3p02.png', 'basn3p02-scrambled.png', 0, 172.63582),

45def test_image_comparison_expect_rms(im1, im2, tol, expect_rms, tmp_path,
46 monkeypatch):
47 """
48 Compare two images, expecting a particular RMS error.
49
50 im1 and im2 are filenames relative to the baseline_dir directory.
51
52 tol is the tolerance to pass to compare_images.
53
54 expect_rms is the expected RMS value, or None. If None, the test will
55 succeed if compare_images succeeds. Otherwise, the test will succeed if
56 compare_images fails and returns an RMS error almost equal to this value.
57 """
58 # Change the working directory using monkeypatch to use a temporary
59 # test specific directory
60 monkeypatch.chdir(tmp_path)
61 baseline_dir, result_dir = map(Path, _image_directories(lambda: "dummy"))
62 # Copy "test" image to result_dir, so that compare_images writes
63 # the diff to result_dir, rather than to the source tree
64 result_im2 = result_dir / im1
65 shutil.copyfile(baseline_dir / im2, result_im2)
66 results = compare_images(
67 baseline_dir / im1, result_im2, tol=tol, in_decorator=True)
68
69 if expect_rms is None:
70 assert results is None
71 else:
72 assert results is not None
73 assert results['rms'] == approx(expect_rms, abs=1e-4)
