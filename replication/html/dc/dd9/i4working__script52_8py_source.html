65# --- ENSEMBLE ASSESSMENT ---
66def ensemble_assessment(code1, code2, model_name, n=3):
67 predictions = []
68 for _ in range(n):
69 results, predicted_type, predicted_sim = rag_similarity_assessment(code1, code2, model_name)
70 predictions.append((results, predicted_type, predicted_sim))
71 # Majority voting on predicted_sim
72 sim_votes = [pred[2] for pred in predictions]
73 final_sim = max(set(sim_votes), key=sim_votes.count)
74 # Most frequent predicted_type
75 type_votes = [pred[1] for pred in predictions]
76 final_type = max(set(type_votes), key=type_votes.count)
77 # Average results for reporting
78 avg_results = {t: sum(pred[0][t] for pred in predictions) / n for t in ["Type-1", "Type-2", "Type-3", "Type-4"]}
79 return avg_results, final_type, final_sim
