70class _AxesStack:
71 """
72 Helper class to track Axes in a figure.
73
74 Axes are tracked both in the order in which they have been added
75 (``self._axes`` insertion/iteration order) and in the separate "gca" stack
76 (which is the index to which they map in the ``self._axes`` dict).
77 """
78
79 def __init__(self):
80 self._axes = {} # Mapping of Axes to "gca" order.
81 self._counter = itertools.count()
82
83 def as_list(self):
84 """List the Axes that have been added to the figure."""
85 return [*self._axes] # This relies on dict preserving order.
86
87 def remove(self, a):
88 """Remove the Axes from the stack."""
89 self._axes.pop(a)
90
91 def bubble(self, a):
92 """Move an Axes, which must already exist in the stack, to the top."""
93 if a not in self._axes:
94 raise ValueError("Axes has not been added yet")
95 self._axes[a] = next(self._counter)
96
97 def add(self, a):
98 """Add an Axes to the stack, ignoring it if already present."""
99 if a not in self._axes:
100 self._axes[a] = next(self._counter)
101
102 def current(self):
103 """Return the active Axes, or None if the stack is empty."""
104 return max(self._axes, key=self._axes.__getitem__, default=None)

106 def __getstate__(self):
107 return {
108 **vars(self),
109 "_counter": max(self._axes.values(), default=0)
110 }
111
112 def __setstate__(self, state):
113 next_counter = state.pop('_counter')
114 vars(self).update(state)
115 self._counter = itertools.count(next_counter)

785 def subplots(self, nrows=1, ncols=1, *, sharex=False, sharey=False,
786 squeeze=True, width_ratios=None, height_ratios=None,
787 subplot_kw=None, gridspec_kw=None):
788 """
789 Add a set of subplots to this figure.
790
791 This utility wrapper makes it convenient to create common layouts of
792 subplots in a single call.
793
794 Parameters
795 ----------
796 nrows, ncols : int, default: 1
797 Number of rows/columns of the subplot grid.
798
799 sharex, sharey : bool or {'none', 'all', 'row', 'col'}, default: False
800 Controls sharing of x-axis (*sharex*) or y-axis (*sharey*):
801
802 - True or 'all': x- or y-axis will be shared among all subplots.
803 - False or 'none': each subplot x- or y-axis will be independent.
804 - 'row': each subplot row will share an x- or y-axis.
805 - 'col': each subplot column will share an x- or y-axis.
806
807 When subplots have a shared x-axis along a column, only the x tick
808 labels of the bottom subplot are created. Similarly, when subplots
809 have a shared y-axis along a row, only the y tick labels of the
810 first column subplot are created. To later turn other subplots'
811 ticklabels on, use `~matplotlib.axes.Axes.tick_params`.
812
813 When subplots have a shared axis that has units, calling
814 `.Axis.set_units` will update each axis with the new units.
815
816 Note that it is not possible to unshare axes.
817
818 squeeze : bool, default: True
819 - If True, extra dimensions are squeezed out from the returned
820 array of Axes:
821
822 - if only one subplot is constructed (nrows=ncols=1), the
823 resulting single Axes object is returned as a scalar.
824 - for Nx1 or 1xM subplots, the returned object is a 1D numpy
825 object array of Axes objects.
826 - for NxM, subplots with N>1 and M>1 are returned as a 2D array.
827
828 - If False, no squeezing at all is done: the returned Axes object
829 is always a 2D array containing Axes instances, even if it ends
830 up being 1x1.
831
832 width_ratios : array-like of length *ncols*, optional
833 Defines the relative widths of the columns. Each column gets a
834 relative width of ``width_ratios[i] / sum(width_ratios)``.
835 If not given, all columns will have the same width. Equivalent
836 to ``gridspec_kw={'width_ratios': [...]}``.
837
838 height_ratios : array-like of length *nrows*, optional
839 Defines the relative heights of the rows. Each row gets a
840 relative height of ``height_ratios[i] / sum(height_ratios)``.
841 If not given, all rows will have the same height. Equivalent
842 to ``gridspec_kw={'height_ratios': [...]}``.
843
844 subplot_kw : dict, optional
845 Dict with keywords passed to the `.Figure.add_subplot` call used to
846 create each subplot.
847
848 gridspec_kw : dict, optional
849 Dict with keywords passed to the
850 `~matplotlib.gridspec.GridSpec` constructor used to create
851 the grid the subplots are placed on.
852
853 Returns
854 -------
855 `~.axes.Axes` or array of Axes
856 Either a single `~matplotlib.axes.Axes` object or an array of Axes
857 objects if more than one subplot was created. The dimensions of the
858 resulting array can be controlled with the *squeeze* keyword, see
859 above.
860
861 See Also
862 --------
863 .pyplot.subplots
864 .Figure.add_subplot
865 .pyplot.subplot
866
867 Examples
868 --------
869 ::
870
871 # First create some toy data:
872 x = np.linspace(0, 2*np.pi, 400)
873 y = np.sin(x**2)
874
875 # Create a figure
876 fig = plt.figure()
877
878 # Create a subplot
879 ax = fig.subplots()
880 ax.plot(x, y)
881 ax.set_title('Simple plot')
882
883 # Create two subplots and unpack the output array immediately
884 ax1, ax2 = fig.subplots(1, 2, sharey=True)
885 ax1.plot(x, y)
886 ax1.set_title('Sharing Y axis')
887 ax2.scatter(x, y)
888
889 # Create four polar Axes and access them through the returned array
890 axes = fig.subplots(2, 2, subplot_kw=dict(projection='polar'))
891 axes[0, 0].plot(x, y)
892 axes[1, 1].scatter(x, y)
893
894 # Share an X-axis with each column of subplots
895 fig.subplots(2, 2, sharex='col')
896
897 # Share a Y-axis with each row of subplots
898 fig.subplots(2, 2, sharey='row')
899
900 # Share both X- and Y-axes with all subplots
901 fig.subplots(2, 2, sharex='all', sharey='all')
902
903 # Note that this is the same as
904 fig.subplots(2, 2, sharex=True, sharey=True)
905 """
906 gridspec_kw = dict(gridspec_kw or {})
907 if height_ratios is not None:
908 if 'height_ratios' in gridspec_kw:
909 raise ValueError("'height_ratios' must not be defined both as "
910 "parameter and as key in 'gridspec_kw'")
911 gridspec_kw['height_ratios'] = height_ratios
912 if width_ratios is not None:
913 if 'width_ratios' in gridspec_kw:
914 raise ValueError("'width_ratios' must not be defined both as "
915 "parameter and as key in 'gridspec_kw'")
916 gridspec_kw['width_ratios'] = width_ratios
917
918 gs = self.add_gridspec(nrows, ncols, figure=self, **gridspec_kw)
919 axs = gs.subplots(sharex=sharex, sharey=sharey, squeeze=squeeze,
920 subplot_kw=subplot_kw)
921 return axs

1603 def subfigures(self, nrows=1, ncols=1, squeeze=True,
1604 wspace=None, hspace=None,
1605 width_ratios=None, height_ratios=None,
1606 **kwargs):
1607 """
1608 Add a set of subfigures to this figure or subfigure.
1609
1610 A subfigure has the same artist methods as a figure, and is logically
1611 the same as a figure, but cannot print itself.
1612 See :doc:`/gallery/subplots_axes_and_figures/subfigures`.
1613
1614 .. versionchanged:: 3.10
1615 subfigures are now added in row-major order.
1616
1617 Parameters
1618 ----------
1619 nrows, ncols : int, default: 1
1620 Number of rows/columns of the subfigure grid.
1621
1622 squeeze : bool, default: True
1623 If True, extra dimensions are squeezed out from the returned
1624 array of subfigures.
1625
1626 wspace, hspace : float, default: None
1627 The amount of width/height reserved for space between subfigures,
1628 expressed as a fraction of the average subfigure width/height.
1629 If not given, the values will be inferred from rcParams if using
1630 constrained layout (see `~.ConstrainedLayoutEngine`), or zero if
1631 not using a layout engine.
1632
1633 width_ratios : array-like of length *ncols*, optional
1634 Defines the relative widths of the columns. Each column gets a
1635 relative width of ``width_ratios[i] / sum(width_ratios)``.
1636 If not given, all columns will have the same width.
1637
1638 height_ratios : array-like of length *nrows*, optional
1639 Defines the relative heights of the rows. Each row gets a
1640 relative height of ``height_ratios[i] / sum(height_ratios)``.
1641 If not given, all rows will have the same height.
1642 """
1643 gs = GridSpec(nrows=nrows, ncols=ncols, figure=self,
1644 wspace=wspace, hspace=hspace,
1645 width_ratios=width_ratios,
1646 height_ratios=height_ratios,
1647 left=0, right=1, bottom=0, top=1)
1648
1649 sfarr = np.empty((nrows, ncols), dtype=object)
1650 for i in range(nrows):
1651 for j in range(ncols):
1652 sfarr[i, j] = self.add_subfigure(gs[i, j], **kwargs)
1653
1654 if self.get_layout_engine() is None and (wspace is not None or
1655 hspace is not None):
1656 # Gridspec wspace and hspace is ignored on subfigure instantiation,
1657 # and no space is left. So need to account for it here if required.
1658 bottoms, tops, lefts, rights = gs.get_grid_positions(self)
1659 for sfrow, bottom, top in zip(sfarr, bottoms, tops):
1660 for sf, left, right in zip(sfrow, lefts, rights):
1661 bbox = Bbox.from_extents(left, bottom, right, top)
1662 sf._redo_transform_rel_fig(bbox=bbox)
1663
1664 if squeeze:
1665 # Discarding unneeded dimensions that equal 1. If we only have one
1666 # subfigure, just return it instead of a 1-element array.
1667 return sfarr.item() if sfarr.size == 1 else sfarr.squeeze()
1668 else:
1669 # Returned axis array will be always 2-d, even if nrows=ncols=1.
1670 return sfarr

1800 def get_tightbbox(self, renderer=None, *, bbox_extra_artists=None):
1801 """
1802 Return a (tight) bounding box of the figure *in inches*.
1803
1804 Note that `.FigureBase` differs from all other artists, which return
1805 their `.Bbox` in pixels.
1806
1807 Artists that have ``artist.set_in_layout(False)`` are not included
1808 in the bbox.
1809
1810 Parameters
1811 ----------
1812 renderer : `.RendererBase` subclass
1813 Renderer that will be used to draw the figures (i.e.
1814 ``fig.canvas.get_renderer()``)
1815
1816 bbox_extra_artists : list of `.Artist` or ``None``
1817 List of artists to include in the tight bounding box. If
1818 ``None`` (default), then all artist children of each Axes are
1819 included in the tight bounding box.
1820
1821 Returns
1822 -------
1823 `.BboxBase`
1824 containing the bounding box (in figure inches).
1825 """
1826
1827 if renderer is None:
1828 renderer = self.get_figure(root=True)._get_renderer()
1829
1830 bb = []
1831 if bbox_extra_artists is None:
1832 artists = [artist for artist in self.get_children()
1833 if (artist not in self.axes and artist.get_visible()
1834 and artist.get_in_layout())]
1835 else:
1836 artists = bbox_extra_artists
1837
1838 for a in artists:
1839 bbox = a.get_tightbbox(renderer)
1840 if bbox is not None:
1841 bb.append(bbox)
1842
1843 for ax in self.axes:
1844 if ax.get_visible():
1845 # some Axes don't take the bbox_extra_artists kwarg so we
1846 # need this conditional....
1847 try:
1848 bbox = ax.get_tightbbox(
1849 renderer, bbox_extra_artists=bbox_extra_artists)
1850 except TypeError:
1851 bbox = ax.get_tightbbox(renderer)
1852 bb.append(bbox)
1853 bb = [b for b in bb
1854 if (np.isfinite(b.width) and np.isfinite(b.height)
1855 and (b.width != 0 or b.height != 0))]
1856
1857 isfigure = hasattr(self, 'bbox_inches')
1858 if len(bb) == 0:
1859 if isfigure:
1860 return self.bbox_inches
1861 else:
1862 # subfigures do not have bbox_inches, but do have a bbox
1863 bb = [self.bbox]
1864
1865 _bbox = Bbox.union(bb)
1866
1867 if isfigure:
1868 # transform from pixels to inches...
1869 _bbox = TransformedBbox(_bbox, self.dpi_scale_trans.inverted())
1870
1871 return _bbox

1902 """
1903 Build a layout of Axes based on ASCII art or nested lists.
1904
1905 This is a helper function to build complex GridSpec layouts visually.
1906
1907 See :ref:`mosaic`
1908 for an example and full API documentation
1909
1910 Parameters
1911 ----------
1912 mosaic : list of list of {hashable or nested} or str
1913
1914 A visual layout of how you want your Axes to be arranged
1915 labeled as strings. For example ::
1916
1917 x = [['A panel', 'A panel', 'edge'],
1918 ['C panel', '.', 'edge']]
1919
1920 produces 4 Axes:
1921
1922 - 'A panel' which is 1 row high and spans the first two columns
1923 - 'edge' which is 2 rows high and is on the right edge
1924 - 'C panel' which in 1 row and 1 column wide in the bottom left
1925 - a blank space 1 row and 1 column wide in the bottom center
1926
1927 Any of the entries in the layout can be a list of lists
1928 of the same form to create nested layouts.
1929
1930 If input is a str, then it can either be a multi-line string of
1931 the form ::
1932
1933 '''
1934 AAE
1935 C.E
1936 '''
1937
1938 where each character is a column and each line is a row. Or it
1939 can be a single-line string where rows are separated by ``;``::
1940
1941 'AB;CC'
1942
1943 The string notation allows only single character Axes labels and
1944 does not support nesting but is very terse.
1945
1946 The Axes identifiers may be `str` or a non-iterable hashable
1947 object (e.g. `tuple` s may not be used).
1948
1949 sharex, sharey : bool, default: False
1950 If True, the x-axis (*sharex*) or y-axis (*sharey*) will be shared
1951 among all subplots. In that case, tick label visibility and axis
1952 units behave as for `subplots`. If False, each subplot's x- or
1953 y-axis will be independent.
1954

1956 Defines the relative widths of the columns. Each column gets a
1957 relative width of ``width_ratios[i] / sum(width_ratios)``.
1958 If not given, all columns will have the same width. Equivalent
1959 to ``gridspec_kw={'width_ratios': [...]}``. In the case of nested
1960 layouts, this argument applies only to the outer layout.
1961

1963 Defines the relative heights of the rows. Each row gets a
1964 relative height of ``height_ratios[i] / sum(height_ratios)``.
1965 If not given, all rows will have the same height. Equivalent
1966 to ``gridspec_kw={'height_ratios': [...]}``. In the case of nested
1967 layouts, this argument applies only to the outer layout.

1972 values in *per_subplot_kw*.
1973

1976 to a dictionary of keyword arguments to be passed to the
1977 `.Figure.add_subplot` call used to create each subplot. The values
1978 in these dictionaries have precedence over the values in
1979 *subplot_kw*.
1980
1981 If *mosaic* is a string, and thus all keys are single characters,
1982 it is possible to use a single string instead of a tuple as keys;
1983 i.e. ``"AB"`` is equivalent to ``("A", "B")``.
1984
1985 .. versionadded:: 3.7
1986
1987 gridspec_kw : dict, optional
1988 Dictionary with keywords passed to the `.GridSpec` constructor used

1990 nested layouts, this argument applies only to the outer layout.

1992 to create the nesting.

1995 Entry in the layout to mean "leave this space empty". Defaults
1996 to ``'.'``. Note, if *layout* is a string, it is processed via

1998 interfere with using white-space as the empty sentinel.
1999

2001 -------
2002 dict[label, Axes]
2003 A dictionary mapping the labels to the Axes objects. The order of
2004 the Axes is left-to-right and top-to-bottom of their position in the
2005 total layout.

2007 """

2009 gridspec_kw = dict(gridspec_kw or {})
2010 per_subplot_kw = per_subplot_kw or {}

2012 if height_ratios is not None:
2013 if 'height_ratios' in gridspec_kw:
2014 raise ValueError("'height_ratios' must not be defined both as "
2015 "parameter and as key in 'gridspec_kw'")

2017 if width_ratios is not None:
2018 if 'width_ratios' in gridspec_kw:

2024 if isinstance(mosaic, str):

2093 def _do_layout(gs, mosaic, unique_ids, nested):
2094 """
2095 Recursively do the mosaic.
2096
2097 Parameters
2098 ----------
2099 gs : GridSpec
2100 mosaic : 2D object array

2102 unique_ids : tuple
2103 The identified scalar labels at this level of nesting.
2104 nested : dict[tuple[int, int]], 2D object array
2105 The identified nested mosaics, if any.
2106
2107 Returns
2108 -------
2109 dict[label, Axes]
2110 A flat dict of all of the Axes created.
2111 """
2112 output = dict()
2113
2114 # we need to merge together the Axes at this level and the Axes
2115 # in the (recursively) nested sub-mosaics so that we can add
2116 # them to the figure in the "natural" order if you were to
2117 # ravel in c-order all of the Axes that will be created
2118 #
2119 # This will stash the upper left index of each object (axes or
2120 # nested mosaic) at this level
2121 this_level = dict()
2122
2123 # go through the unique keys,
2124 for name in unique_ids:
2125 # sort out where each axes starts/ends
2126 indx = np.argwhere(mosaic == name)
2127 start_row, start_col = np.min(indx, axis=0)
2128 end_row, end_col = np.max(indx, axis=0) + 1
2129 # and construct the slice object
2130 slc = (slice(start_row, end_row), slice(start_col, end_col))
2131 # some light error checking
2132 if (mosaic[slc] != name).any():
2133 raise ValueError(
2134 f"While trying to layout\n{mosaic!r}\n"
2135 f"we found that the label {name!r} specifies a "
2136 "non-rectangular or non-contiguous area.")
2137 # and stash this slice for later
2138 this_level[(start_row, start_col)] = (name, slc, 'axes')
2139
2140 # do the same thing for the nested mosaics (simpler because these
2141 # cannot be spans yet!)
2142 for (j, k), nested_mosaic in nested.items():
2143 this_level[(j, k)] = (None, nested_mosaic, 'nested')
2144
2145 # now go through the things in this level and add them
2146 # in order left-to-right top-to-bottom
2147 for key in sorted(this_level):
2148 name, arg, method = this_level[key]
2149 # we are doing some hokey function dispatch here based
2150 # on the 'method' string stashed above to sort out if this
2151 # element is an Axes or a nested mosaic.
2152 if method == 'axes':
2153 slc = arg
2154 # add a single Axes
2155 if name in output:
2156 raise ValueError(f"There are duplicate keys {name} "
2157 f"in the layout\n{mosaic!r}")
2158 ax = self.add_subplot(
2159 gs[slc], **{
2160 'label': str(name),
2161 **subplot_kw,
2162 **per_subplot_kw.get(name, {})
2163 }
2164 )
2165 output[name] = ax
2166 elif method == 'nested':
2167 nested_mosaic = arg
2168 j, k = key
2169 # recursively add the nested mosaic
2170 rows, cols = nested_mosaic.shape
2171 nested_output = _do_layout(
2172 gs[j, k].subgridspec(rows, cols),
2173 nested_mosaic,
2174 *_identify_keys_and_nested(nested_mosaic)
2175 )
2176 overlap = set(output) & set(nested_output)
2177 if overlap:
2178 raise ValueError(
2179 f"There are duplicate keys {overlap} "
2180 f"between the outer layout\n{mosaic!r}\n"
2181 f"and the nested layout\n{nested_mosaic}"
2182 )
2183 output.update(nested_output)
2184 else:
2185 raise RuntimeError("This should never happen")
2186 return output
2187
2188 mosaic = _make_array(mosaic)
2189 rows, cols = mosaic.shape
2190 gs = self.add_gridspec(rows, cols, **gridspec_kw)
2191 ret = _do_layout(gs, mosaic, *_identify_keys_and_nested(mosaic))
2192 ax0 = next(iter(ret.values()))

2195 ax.sharex(ax0)

2197 if sharey:
2198 ax.sharey(ax0)
2199 ax._label_outer_yaxis(skip_non_rectangular_axes=True)

3649def figaspect(arg):
3650 """
3651 Calculate the width and height for a figure with a specified aspect ratio.
3652
3653 While the height is taken from :rc:`figure.figsize`, the width is
3654 adjusted to match the desired aspect ratio. Additionally, it is ensured
3655 that the width is in the range [4., 16.] and the height is in the range
3656 [2., 16.]. If necessary, the default height is adjusted to ensure this.
3657
3658 Parameters
3659 ----------
3660 arg : float or 2D array
3661 If a float, this defines the aspect ratio (i.e. the ratio height /
3662 width).
3663 In case of an array the aspect ratio is number of rows / number of
3664 columns, so that the array could be fitted in the figure undistorted.
3665
3666 Returns
3667 -------
3668 size : (2,) array
3669 The width and height of the figure in inches.
3670

3672 -----
3673 If you want to create an Axes within the figure, that still preserves the
3674 aspect ratio, be sure to create it with equal width and height. See
3675 examples below.
3676
3677 Thanks to Fernando Perez for this function.
3678
3679 Examples
3680 --------
3681 Make a figure twice as tall as it is wide::
3682

3686 ax.imshow(A, **kwargs)
3687
3688 Make a figure with the proper aspect for an array::
3689
3690 A = rand(5, 3)
3691 w, h = figaspect(A)
3692 fig = Figure(figsize=(w, h))
3693 ax = fig.add_axes([0.1, 0.1, 0.8, 0.8])
3694 ax.imshow(A, **kwargs)

3696
3697 isarray = hasattr(arg, 'shape') and not np.isscalar(arg)
3698
3699 # min/max sizes to respect when autoscaling. If John likes the idea, they
3700 # could become rc parameters, for now they're hardwired.
3701 figsize_min = np.array((4.0, 2.0)) # min length for width/height
3702 figsize_max = np.array((16.0, 16.0)) # max length for width/height
3703
3704 # Extract the aspect ratio of the array
3705 if isarray:
3706 nr, nc = arg.shape[:2]
3707 arr_ratio = nr / nc
3708 else:
3709 arr_ratio = arg
3710
3711 # Height of user figure defaults
3712 fig_height = mpl.rcParams['figure.figsize'][1]

3715 newsize = np.array((fig_height / arr_ratio, fig_height))

3718 newsize /= min(1.0, *(newsize / figsize_min))
