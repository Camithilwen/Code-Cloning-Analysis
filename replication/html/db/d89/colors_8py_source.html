1"""
2A module for converting numbers or color arguments to *RGB* or *RGBA*.
3
4*RGB* and *RGBA* are sequences of, respectively, 3 or 4 floats in the
5range 0-1.
6
7This module includes functions and classes for color specification conversions,
8and for mapping numbers to colors in a 1-D array of colors called a colormap.
9
10Mapping data onto colors using a colormap typically involves two steps: a data
11array is first mapped onto the range 0-1 using a subclass of `Normalize`,
12then this number is mapped to a color using a subclass of `Colormap`. Two
13subclasses of `Colormap` provided here: `LinearSegmentedColormap`, which uses
14piecewise-linear interpolation to define colormaps, and `ListedColormap`, which
15makes a colormap from a list of colors.
16
17.. seealso::
18
19 :ref:`colormap-manipulation` for examples of how to
20 make colormaps and
21
22 :ref:`colormaps` for a list of built-in colormaps.
23
24 :ref:`colormapnorms` for more details about data
25 normalization
26
27 More colormaps are available at palettable_.
28
29The module also provides functions for checking whether an object can be
30interpreted as a color (`is_color_like`), for converting such an object
31to an RGBA tuple (`to_rgba`) or to an HTML-like hex string in the
32"#rrggbb" format (`to_hex`), and a sequence of colors to an (n, 4)
33RGBA array (`to_rgba_array`). Caching is used for efficiency.
34

36:ref:`colors_def`.

156 def register(self, name, color_list):
157 """
158 Register a new color sequence.
159
160 The color sequence registry stores a copy of the given *color_list*, so
161 that future changes to the original list do not affect the registered
162 color sequence. Think of this as the registry taking a snapshot
163 of *color_list* at registration.
164
165 Parameters
166 ----------
167 name : str
168 The name for the color sequence.
169
170 color_list : list of :mpltype:`color`
171 An iterable returning valid Matplotlib colors when iterating over.
172 Note however that the returned color sequence will always be a
173 list regardless of the input type.
174
175 """

177 raise ValueError(f"{name!r} is a reserved name for a builtin "
178 "color sequence")
179
180 color_list = list(color_list) # force copy and coerce type to list
181 for color in color_list:
182 try:
183 to_rgba(color)
184 except ValueError:
185 raise ValueError(
186 f"{color!r} is not a valid color specification")
187
188 self._color_sequences[name] = color_list

279 """
280 Convert *c* to an RGBA color.
281
282 Parameters
283 ----------
284 c : Matplotlib color or ``np.ma.masked``
285
286 alpha : float, optional
287 If *alpha* is given, force the alpha value of the returned RGBA tuple
288 to *alpha*.
289
290 If None, the alpha value from *c* is used. If *c* does not have an
291 alpha channel, then alpha defaults to 1.
292
293 *alpha* is ignored for the color value ``"none"`` (case-insensitive),
294 which always maps to ``(0, 0, 0, 0)``.
295
296 Returns
297 -------
298 tuple
299 Tuple of floats ``(r, g, b, a)``, where each channel (red, green, blue,
300 alpha) can assume values between 0 and 1.
301 """
302 if isinstance(c, tuple) and len(c) == 2:
303 if alpha is None:
304 c, alpha = c
305 else:
306 c = c[0]
307 # Special-case nth color syntax because it should not be cached.
308 if _is_nth_color(c):
309 prop_cycler = mpl.rcParams['axes.prop_cycle']
310 colors = prop_cycler.by_key().get('color', ['k'])
311 c = colors[int(c[1:]) % len(colors)]
312 try:
313 rgba = _colors_full_map.cache[c, alpha]
314 except (KeyError, TypeError): # Not in cache, or unhashable.
315 rgba = None
316 if rgba is None: # Suppress exception chaining of cache lookup failure.
317 rgba = _to_rgba_no_colorcycle(c, alpha)
318 try:
319 _colors_full_map.cache[c, alpha] = rgba
320 except TypeError:
321 pass

325def _to_rgba_no_colorcycle(c, alpha=None):

328

346 # This may turn c into a non-string, so we check again below.
347 c = _colors_full_map[c]

376 match = re.match(r"\A#[a-fA-F0-9]{4}\Z", c)

419def to_rgba_array(c, alpha=None):
420 """
421 Convert *c* to a (n, 4) array of RGBA colors.
422
423 Parameters
424 ----------
425 c : Matplotlib color or array of colors
426 If *c* is a masked array, an `~numpy.ndarray` is returned with a
427 (0, 0, 0, 0) row for each masked value or row in *c*.
428
429 alpha : float or sequence of floats, optional
430 If *alpha* is given, force the alpha value of the returned RGBA tuple
431 to *alpha*.
432
433 If None, the alpha value from *c* is used. If *c* does not have an
434 alpha channel, then alpha defaults to 1.
435
436 *alpha* is ignored for the color value ``"none"`` (case-insensitive),
437 which always maps to ``(0, 0, 0, 0)``.
438
439 If *alpha* is a sequence and *c* is a single color, *c* will be
440 repeated to match the length of *alpha*.
441
442 Returns
443 -------
444 array
445 (n, 4) array of RGBA colors, where each channel (red, green, blue,
446 alpha) can assume values between 0 and 1.
447 """
448 if isinstance(c, tuple) and len(c) == 2 and isinstance(c[1], Real):
449 if alpha is None:
450 c, alpha = c
451 else:
452 c = c[0]
453 # Special-case inputs that are already arrays, for performance. (If the
454 # array has the wrong kind or shape, raise the error during one-at-a-time
455 # conversion.)
456 if np.iterable(alpha):
457 alpha = np.asarray(alpha).ravel()
458 if (isinstance(c, np.ndarray) and c.dtype.kind in "if"
459 and c.ndim == 2 and c.shape[1] in [3, 4]):
460 mask = c.mask.any(axis=1) if np.ma.is_masked(c) else None
461 c = np.ma.getdata(c)
462 if np.iterable(alpha):
463 if c.shape[0] == 1 and alpha.shape[0] > 1:
464 c = np.tile(c, (alpha.shape[0], 1))
465 elif c.shape[0] != alpha.shape[0]:
466 raise ValueError("The number of colors must match the number"
467 " of alpha values if there are more than one"
468 " of each.")
469 if c.shape[1] == 3:
470 result = np.column_stack([c, np.zeros(len(c))])
471 result[:, -1] = alpha if alpha is not None else 1.
472 elif c.shape[1] == 4:
473 result = c.copy()
474 if alpha is not None:
475 result[:, -1] = alpha
476 if mask is not None:
477 result[mask] = 0
478 if np.any((result < 0) | (result > 1)):
479 raise ValueError("RGBA values should be within 0-1 range")
480 return result
481 # Handle single values.
482 # Note that this occurs *after* handling inputs that are already arrays, as
483 # `to_rgba(c, alpha)` (below) is expensive for such inputs, due to the need
484 # to format the array in the ValueError message(!).
485 if cbook._str_lower_equal(c, "none"):
486 return np.zeros((0, 4), float)
487 try:
488 if np.iterable(alpha):
489 return np.array([to_rgba(c, a) for a in alpha], float)
490 else:
491 return np.array([to_rgba(c, alpha)], float)
492 except TypeError:
493 pass
494 except ValueError as e:
495 if e.args == ("'alpha' must be between 0 and 1, inclusive", ):
496 # ValueError is from _to_rgba_no_colorcycle().
497 raise e
498 if isinstance(c, str):
499 raise ValueError(f"{c!r} is not a valid color value.")
500
501 if len(c) == 0:
502 return np.zeros((0, 4), float)
503
504 # Quick path if the whole sequence can be directly converted to a numpy
505 # array in one shot.
506 if isinstance(c, Sequence):
507 lens = {len(cc) if isinstance(cc, (list, tuple)) else -1 for cc in c}
508 if lens == {3}:
509 rgba = np.column_stack([c, np.ones(len(c))])
510 elif lens == {4}:
511 rgba = np.array(c)
512 else:
513 rgba = np.array([to_rgba(cc) for cc in c])
514 else:
515 rgba = np.array([to_rgba(cc) for cc in c])
516
517 if alpha is not None:
518 rgba[:, 3] = alpha
519 if isinstance(c, Sequence):
520 # ensure that an explicit alpha does not overwrite full transparency
521 # for "none"
522 none_mask = [cbook._str_equal(cc, "none") for cc in c]
523 rgba[:, 3][none_mask] = 0
524 return rgba

582def _create_lookup_table(N, data, gamma=1.0):
583 r"""
584 Create an *N* -element 1D lookup table.
585
586 This assumes a mapping :math:`f : [0, 1] \rightarrow [0, 1]`. The returned
587 data is an array of N values :math:`y = f(x)` where x is sampled from
588 [0, 1].
589
590 By default (*gamma* = 1) x is equidistantly sampled from [0, 1]. The
591 *gamma* correction factor :math:`\gamma` distorts this equidistant
592 sampling by :math:`x \rightarrow x^\gamma`.
593
594 Parameters
595 ----------
596 N : int
597 The number of elements of the created lookup table; at least 1.
598
599 data : (M, 3) array-like or callable
600 Defines the mapping :math:`f`.
601
602 If a (M, 3) array-like, the rows define values (x, y0, y1). The x
603 values must start with x=0, end with x=1, and all x values be in
604 increasing order.
605
606 A value between :math:`x_i` and :math:`x_{i+1}` is mapped to the range
607 :math:`y^1_{i-1} \ldots y^0_i` by linear interpolation.
608
609 For the simple case of a y-continuous mapping, y0 and y1 are identical.
610
611 The two values of y are to allow for discontinuous mapping functions.
612 E.g. a sawtooth with a period of 0.2 and an amplitude of 1 would be::
613
614 [(0, 1, 0), (0.2, 1, 0), (0.4, 1, 0), ..., [(1, 1, 0)]
615
616 In the special case of ``N == 1``, by convention the returned value
617 is y0 for x == 1.
618
619 If *data* is a callable, it must accept and return numpy arrays::
620
621 data(x : ndarray) -> ndarray
622
623 and map values between 0 - 1 to 0 - 1.
624
625 gamma : float
626 Gamma correction factor for input distribution x of the mapping.
627
628 See also https://en.wikipedia.org/wiki/Gamma_correction.
629
630 Returns
631 -------
632 array
633 The lookup table where ``lut[x * (N-1)]`` gives the closest value
634 for values of x between 0 and 1.
635
636 Notes
637 -----
638 This function is internally used for `.LinearSegmentedColormap`.
639 """
640
641 if callable(data):
642 xind = np.linspace(0, 1, N) ** gamma
643 lut = np.clip(np.array(data(xind), dtype=float), 0, 1)
644 return lut
645
646 try:
647 adata = np.array(data)
648 except Exception as err:
649 raise TypeError("data must be convertible to an array") from err
650 _api.check_shape((None, 3), data=adata)
651
652 x = adata[:, 0]
653 y0 = adata[:, 1]
654 y1 = adata[:, 2]
655
656 if x[0] != 0. or x[-1] != 1.0:
657 raise ValueError(
658 "data mapping points must start with x=0 and end with x=1")
659 if (np.diff(x) < 0).any():
660 raise ValueError("data mapping points must have x in increasing order")
661 # begin generation of lookup table
662 if N == 1:
663 # convention: use the y = f(x=1) value for a 1-element lookup table
664 lut = np.array(y0[-1])
665 else:
666 x = x * (N - 1)
667 xind = (N - 1) * np.linspace(0, 1, N) ** gamma
668 ind = np.searchsorted(x, xind)[1:-1]
669
670 distance = (xind[1:-1] - x[ind - 1]) / (x[ind] - x[ind - 1])
671 lut = np.concatenate([
672 [y1[0]],
673 distance * (y0[ind] - y1[ind - 1]) + y1[ind - 1],
674 [y0[-1]],
675 ])
676 # ensure that the lut is confined to values between 0 and 1 by clipping it
677 return np.clip(lut, 0.0, 1.0)

681 """
682 Baseclass for all scalar to RGBA mappings.
683
684 Typically, Colormap instances are used to convert data values (floats)
685 from the interval ``[0, 1]`` to the RGBA color that the respective
686 Colormap represents. For scaling of data into the ``[0, 1]`` interval see
687 `matplotlib.colors.Normalize`. Subclasses of `matplotlib.cm.ScalarMappable`
688 make heavy use of this ``data -> normalize -> map-to-color`` processing
689 chain.

745 def _get_rgba_and_mask(self, X, alpha=None, bytes=False):
746 r"""
747 Parameters
748 ----------
749 X : float or int or array-like
750 The data value(s) to convert to RGBA.
751 For floats, *X* should be in the interval ``[0.0, 1.0]`` to
752 return the RGBA values ``X*100`` percent along the Colormap line.
753 For integers, *X* should be in the interval ``[0, Colormap.N)`` to
754 return RGBA values *indexed* from the Colormap with index ``X``.
755 alpha : float or array-like or None
756 Alpha must be a scalar between 0 and 1, a sequence of such
757 floats with shape matching X, or None.
758 bytes : bool, default: False
759 If False (default), the returned RGBA values will be floats in the
760 interval ``[0, 1]`` otherwise they will be `numpy.uint8`\s in the
761 interval ``[0, 255]``.
762
763 Returns
764 -------
765 colors : np.ndarray
766 Array of RGBA values with a shape of ``X.shape + (4, )``.
767 mask : np.ndarray
768 Boolean array with True where the input is ``np.nan`` or masked.
769 """
770 if not self._isinit:
771 self._init()
772
773 xa = np.array(X, copy=True)
774 if not xa.dtype.isnative:
775 # Native byteorder is faster.
776 xa = xa.byteswap().view(xa.dtype.newbyteorder())
777 if xa.dtype.kind == "f":
778 xa *= self.N
779 # xa == 1 (== N after multiplication) is not out of range.
780 xa[xa == self.N] = self.N - 1
781 # Pre-compute the masks before casting to int (which can truncate
782 # negative values to zero or wrap large floats to negative ints).
783 mask_under = xa < 0
784 mask_over = xa >= self.N
785 # If input was masked, get the bad mask from it; else mask out nans.
786 mask_bad = X.mask if np.ma.is_masked(X) else np.isnan(xa)
787 with np.errstate(invalid="ignore"):
788 # We need this cast for unsigned ints as well as floats
789 xa = xa.astype(int)
790 xa[mask_under] = self._i_under
791 xa[mask_over] = self._i_over
792 xa[mask_bad] = self._i_bad
793
794 lut = self._lut
795 if bytes:
796 lut = (lut * 255).astype(np.uint8)
797
798 rgba = lut.take(xa, axis=0, mode='clip')
799
800 if alpha is not None:
801 alpha = np.clip(alpha, 0, 1)
802 if bytes:
803 alpha *= 255 # Will be cast to uint8 upon assignment.
804 if alpha.shape not in [(), xa.shape]:
805 raise ValueError(
806 f"alpha is array-like but its shape {alpha.shape} does "
807 f"not match that of X {xa.shape}")
808 rgba[..., -1] = alpha
809 # If the "bad" color is all zeros, then ignore alpha input.
810 if (lut[-1] == 0).all():
811 rgba[mask_bad] = (0, 0, 0, 0)
812
813 return rgba, mask_bad

1003class LinearSegmentedColormap(Colormap):
1004 """
1005 Colormap objects based on lookup tables using linear segments.
1006
1007 The lookup table is generated using linear interpolation for each
1008 primary color, with the 0-1 domain divided into any number of
1009 segments.
1010 """
1011
1012 def __init__(self, name, segmentdata, N=256, gamma=1.0):
1013 """
1014 Create colormap from linear mapping segments
1015
1016 segmentdata argument is a dictionary with a red, green and blue
1017 entries. Each entry should be a list of *x*, *y0*, *y1* tuples,
1018 forming rows in a table. Entries for alpha are optional.
1019
1020 Example: suppose you want red to increase from 0 to 1 over
1021 the bottom half, green to do the same over the middle half,
1022 and blue over the top half. Then you would use::
1023
1024 cdict = {'red': [(0.0, 0.0, 0.0),
1025 (0.5, 1.0, 1.0),
1026 (1.0, 1.0, 1.0)],
1027
1028 'green': [(0.0, 0.0, 0.0),
1029 (0.25, 0.0, 0.0),
1030 (0.75, 1.0, 1.0),
1031 (1.0, 1.0, 1.0)],
1032
1033 'blue': [(0.0, 0.0, 0.0),
1034 (0.5, 0.0, 0.0),
1035 (1.0, 1.0, 1.0)]}
1036
1037 Each row in the table for a given color is a sequence of
1038 *x*, *y0*, *y1* tuples. In each sequence, *x* must increase
1039 monotonically from 0 to 1. For any input value *z* falling
1040 between *x[i]* and *x[i+1]*, the output value of a given color
1041 will be linearly interpolated between *y1[i]* and *y0[i+1]*::
1042
1043 row i: x y0 y1
1044 /
1045 /
1046 row i+1: x y0 y1
1047
1048 Hence y0 in the first row and y1 in the last row are never used.
1049
1050 See Also
1051 --------
1052 LinearSegmentedColormap.from_list
1053 Static method; factory function for generating a smoothly-varying
1054 LinearSegmentedColormap.
1055 """
1056 # True only if all colors in map are identical; needed for contouring.
1057 self.monochrome = False
1058 super().__init__(name, N)
1059 self._segmentdata = segmentdata
1060 self._gamma = gamma
1061
1062 def _init(self):
1063 self._lut = np.ones((self.N + 3, 4), float)
1064 self._lut[:-3, 0] = _create_lookup_table(
1065 self.N, self._segmentdata['red'], self._gamma)
1066 self._lut[:-3, 1] = _create_lookup_table(
1067 self.N, self._segmentdata['green'], self._gamma)
1068 self._lut[:-3, 2] = _create_lookup_table(
1069 self.N, self._segmentdata['blue'], self._gamma)
1070 if 'alpha' in self._segmentdata:
1071 self._lut[:-3, 3] = _create_lookup_table(
1072 self.N, self._segmentdata['alpha'], 1)
1073 self._isinit = True
1074 self._set_extremes()
1075
1076 def set_gamma(self, gamma):
1077 """Set a new gamma value and regenerate colormap."""
1078 self._gamma = gamma
1079 self._init()

1566 def __call__(self, X, alpha=None, bytes=False):
1567 r"""
1568 Parameters
1569 ----------
1570 X : tuple (X0, X1), X0 and X1: float or int or array-like
1571 The data value(s) to convert to RGBA.
1572
1573 - For floats, *X* should be in the interval ``[0.0, 1.0]`` to
1574 return the RGBA values ``X*100`` percent along the Colormap.
1575 - For integers, *X* should be in the interval ``[0, Colormap.N)`` to
1576 return RGBA values *indexed* from the Colormap with index ``X``.
1577
1578 alpha : float or array-like or None, default: None
1579 Alpha must be a scalar between 0 and 1, a sequence of such
1580 floats with shape matching X0, or None.
1581 bytes : bool, default: False
1582 If False (default), the returned RGBA values will be floats in the
1583 interval ``[0, 1]`` otherwise they will be `numpy.uint8`\s in the
1584 interval ``[0, 255]``.
1585
1586 Returns
1587 -------
1588 Tuple of RGBA values if X is scalar, otherwise an array of
1589 RGBA values with a shape of ``X.shape + (4, )``.
1590 """
1591
1592 if len(X) != 2:
1593 raise ValueError(
1594 f'For a `BivarColormap` the data must have a first dimension '
1595 f'2, not {len(X)}')
1596
1597 if not self._isinit:
1598 self._init()
1599
1600 X0 = np.ma.array(X[0], copy=True)
1601 X1 = np.ma.array(X[1], copy=True)
1602 # clip to shape of colormap, circle square, etc.
1603 self._clip((X0, X1))
1604
1605 # Native byteorder is faster.
1606 if not X0.dtype.isnative:
1607 X0 = X0.byteswap().view(X0.dtype.newbyteorder())
1608 if not X1.dtype.isnative:
1609 X1 = X1.byteswap().view(X1.dtype.newbyteorder())
1610
1611 if X0.dtype.kind == "f":
1612 X0 *= self.N
1613 # xa == 1 (== N after multiplication) is not out of range.
1614 X0[X0 == self.N] = self.N - 1
1615
1616 if X1.dtype.kind == "f":
1617 X1 *= self.M
1618 # xa == 1 (== N after multiplication) is not out of range.
1619 X1[X1 == self.M] = self.M - 1
1620
1621 # Pre-compute the masks before casting to int (which can truncate)
1622 mask_outside = (X0 < 0) | (X1 < 0) | (X0 >= self.N) | (X1 >= self.M)
1623 # If input was masked, get the bad mask from it; else mask out nans.
1624 mask_bad_0 = X0.mask if np.ma.is_masked(X0) else np.isnan(X0)
1625 mask_bad_1 = X1.mask if np.ma.is_masked(X1) else np.isnan(X1)
1626 mask_bad = mask_bad_0 | mask_bad_1
1627
1628 with np.errstate(invalid="ignore"):
1629 # We need this cast for unsigned ints as well as floats
1630 X0 = X0.astype(int)
1631 X1 = X1.astype(int)
1632
1633 # Set masked values to zero
1634 # The corresponding rgb values will be replaced later
1635 for X_part in [X0, X1]:
1636 X_part[mask_outside] = 0
1637 X_part[mask_bad] = 0
1638
1639 rgba = self._lut[X0, X1]
1640 if np.isscalar(X[0]):
1641 rgba = np.copy(rgba)
1642 rgba[mask_outside] = self._rgba_outside
1643 rgba[mask_bad] = self._rgba_bad
1644 if bytes:
1645 rgba = (rgba * 255).astype(np.uint8)
1646 if alpha is not None:
1647 alpha = np.clip(alpha, 0, 1)
1648 if bytes:
1649 alpha *= 255 # Will be cast to uint8 upon assignment.
1650 if np.shape(alpha) not in [(), np.shape(X0)]:
1651 raise ValueError(
1652 f"alpha is array-like but its shape {np.shape(alpha)} does "
1653 f"not match that of X[0] {np.shape(X0)}")
1654 rgba[..., -1] = alpha
1655 # If the "bad" color is all zeros, then ignore alpha input.
1656 if (np.array(self._rgba_bad) == 0).all():
1657 rgba[mask_bad] = (0, 0, 0, 0)
1658
1659 if not np.iterable(X[0]):
1660 rgba = tuple(rgba)
1661 return rgba

1728
1729 Note that this function does not move the origin.
1730
1731 Parameters
1732 ----------
1733 lutshape : tuple of ints or None
1734 The tuple must be of length 2, and each entry is either an int or None.
1735

1746 BivarColormap
1747 """
1748

1750 raise ValueError("lutshape must be of length 2")
1751 lutshape = [lutshape[0], lutshape[1]]
1752 if lutshape[0] is None or lutshape[0] == 1:
1753 lutshape[0] = self.N
1754 if lutshape[1] is None or lutshape[1] == 1:
1755 lutshape[1] = self.M
1756
1757 inverted = [False, False]
1758 if lutshape[0] < 0:
1759 inverted[0] = True

1761 if lutshape[0] == 1:

1872 def _clip(self, X):
1873 """
1874 For internal use when applying a BivarColormap to data.
1875 i.e. cm.ScalarMappable().to_rgba()
1876 Clips X[0] and X[1] according to 'self.shape'.
1877 X is modified in-place.
1878
1879 Parameters
1880 ----------
1881 X: np.array
1882 array of floats or ints to be clipped
1883 shape : {'square', 'circle', 'ignore', 'circleignore'}
1884
1885 - If 'square' each variate is clipped to [0,1] independently
1886 - If 'circle' the variates are clipped radially to the center
1887 of the colormap.
1888 It is assumed that a circular mask is applied when the colormap
1889 is displayed
1890 - If 'ignore' the variates are not clipped, but instead assigned the
1891 'outside' color
1892 - If 'circleignore' a circular mask is applied, but the data is not clipped
1893 and instead assigned the 'outside' color
1894
1895 """
1896 if self.shape == 'square':
1897 for X_part, mx in zip(X, (self.N, self.M)):
1898 X_part[X_part < 0] = 0
1899 if X_part.dtype.kind == "f":
1900 X_part[X_part > 1] = 1
1901 else:
1902 X_part[X_part >= mx] = mx - 1
1903
1904 elif self.shape == 'ignore':
1905 for X_part, mx in zip(X, (self.N, self.M)):
1906 X_part[X_part < 0] = -1
1907 if X_part.dtype.kind == "f":
1908 X_part[X_part > 1] = -1
1909 else:
1910 X_part[X_part >= mx] = -1
1911
1912 elif self.shape == 'circle' or self.shape == 'circleignore':
1913 for X_part in X:
1914 if X_part.dtype.kind != "f":
1915 raise NotImplementedError(
1916 "Circular bivariate colormaps are only"
1917 " implemented for use with with floats")
1918 radii_sqr = (X[0] - 0.5)**2 + (X[1] - 0.5)**2
1919 mask_outside = radii_sqr > 0.25
1920 if self.shape == 'circle':
1921 overextend = 2 * np.sqrt(radii_sqr[mask_outside])
1922 X[0][mask_outside] = (X[0][mask_outside] - 0.5) / overextend + 0.5
1923 X[1][mask_outside] = (X[1][mask_outside] - 0.5) / overextend + 0.5
1924 else:
1925 X[0][mask_outside] = -1
1926 X[1][mask_outside] = -1

3298 def hillshade(self, elevation, vert_exag=1, dx=1, dy=1, fraction=1.):
3299 """
3300 Calculate the illumination intensity for a surface using the defined
3301 azimuth and elevation for the light source.
3302
3303 This computes the normal vectors for the surface, and then passes them
3304 on to `shade_normals`
3305
3306 Parameters
3307 ----------
3308 elevation : 2D array-like
3309 The height values used to generate an illumination map
3310 vert_exag : number, optional
3311 The amount to exaggerate the elevation values by when calculating
3312 illumination. This can be used either to correct for differences in
3313 units between the x-y coordinate system and the elevation
3314 coordinate system (e.g. decimal degrees vs. meters) or to
3315 exaggerate or de-emphasize topographic effects.
3316 dx : number, optional
3317 The x-spacing (columns) of the input *elevation* grid.
3318 dy : number, optional
3319 The y-spacing (rows) of the input *elevation* grid.
3320 fraction : number, optional
3321 Increases or decreases the contrast of the hillshade. Values
3322 greater than one will cause intermediate values to move closer to
3323 full illumination or shadow (and clipping any values that move
3324 beyond 0 or 1). Note that this is not visually or mathematically
3325 the same as vertical exaggeration.
3326
3327 Returns
3328 -------
3329 `~numpy.ndarray`
3330 A 2D array of illumination values between 0-1, where 0 is
3331 completely in shadow and 1 is completely illuminated.
3332 """
3333
3334 # Because most image and raster GIS data has the first row in the array
3335 # as the "top" of the image, dy is implicitly negative. This is
3336 # consistent to what `imshow` assumes, as well.
3337 dy = -dy
3338
3339 # compute the normal vectors from the partial derivatives
3340 e_dy, e_dx = np.gradient(vert_exag * elevation, dy, dx)
3341
3342 # .view is to keep subclasses
3343 normal = np.empty(elevation.shape + (3,)).view(type(elevation))
3344 normal[..., 0] = -e_dx
3345 normal[..., 1] = -e_dy
3346 normal[..., 2] = 1
3347 normal /= _vector_magnitude(normal)
3348
3349 return self.shade_normals(normal, fraction)

3351 def shade_normals(self, normals, fraction=1.):
3352 """
3353 Calculate the illumination intensity for the normal vectors of a
3354 surface using the defined azimuth and elevation for the light source.
3355
3356 Imagine an artificial sun placed at infinity in some azimuth and
3357 elevation position illuminating our surface. The parts of the surface
3358 that slope toward the sun should brighten while those sides facing away
3359 should become darker.
3360
3361 Parameters
3362 ----------
3363 fraction : number, optional
3364 Increases or decreases the contrast of the hillshade. Values
3365 greater than one will cause intermediate values to move closer to
3366 full illumination or shadow (and clipping any values that move
3367 beyond 0 or 1). Note that this is not visually or mathematically
3368 the same as vertical exaggeration.
3369
3370 Returns
3371 -------
3372 `~numpy.ndarray`
3373 A 2D array of illumination values between 0-1, where 0 is
3374 completely in shadow and 1 is completely illuminated.
3375 """
3376
3377 intensity = normals.dot(self.direction)
3378
3379 # Apply contrast stretch
3380 imin, imax = intensity.min(), intensity.max()
3381 intensity *= fraction
3382
3383 # Rescale to 0-1, keeping range before contrast stretch
3384 # If constant slope, keep relative scaling (i.e. flat should be 0.5,
3385 # fully occluded 0, etc.)
3386 if (imax - imin) > 1e-6:
3387 # Strictly speaking, this is incorrect. Negative values should be
3388 # clipped to 0 because they're fully occluded. However, rescaling
3389 # in this manner is consistent with the previous implementation and
3390 # visually appears better than a "hard" clip.
3391 intensity -= imin
3392 intensity /= (imax - imin)
3393 intensity = np.clip(intensity, 0, 1)
3394
3395 return intensity

3547 def blend_hsv(self, rgb, intensity, hsv_max_sat=None, hsv_max_val=None,
3548 hsv_min_val=None, hsv_min_sat=None):
3549 """
3550 Take the input data array, convert to HSV values in the given colormap,
3551 then adjust those color values to give the impression of a shaded
3552 relief map with a specified light source. RGBA values are returned,
3553 which can then be used to plot the shaded image with imshow.

3556 values (in HSV colorspace) toward (hsv_min_sat, hsv_min_val) in the
3557 shaded regions, or lightened by sliding (s, v) toward (hsv_max_sat,
3558 hsv_max_val) in regions that are illuminated. The default extremes are
3559 chose so that completely shaded points are nearly black (s = 1, v = 0)
3560 and completely illuminated points are nearly white (s = 0, v = 1).

3563 ----------
3564 rgb : `~numpy.ndarray`
3565 An (M, N, 3) RGB array of floats ranging from 0 to 1 (color image).
3566 intensity : `~numpy.ndarray`
3567 An (M, N, 1) array of floats ranging from 0 to 1 (grayscale image).
3568 hsv_max_sat : number, optional
3569 The maximum saturation value that the *intensity* map can shift the output
3570 image to. If not provided, use the value provided upon initialization.
3571 hsv_min_sat : number, optional
3572 The minimum saturation value that the *intensity* map can shift the output
3573 image to. If not provided, use the value provided upon initialization.
3574 hsv_max_val : number, optional
3575 The maximum value ("v" in "hsv") that the *intensity* map can shift the
3576 output image to. If not provided, use the value provided upon
3577 initialization.
3578 hsv_min_val : number, optional
3579 The minimum value ("v" in "hsv") that the *intensity* map can shift the
3580 output image to. If not provided, use the value provided upon
3581 initialization.
3582
3583 Returns
3584 -------
3585 `~numpy.ndarray`
3586 An (M, N, 3) RGB array representing the combined images.
3587 """
3588 # Backward compatibility...
3589 if hsv_max_sat is None:
3590 hsv_max_sat = self.hsv_max_sat
3591 if hsv_max_val is None:
3592 hsv_max_val = self.hsv_max_val
3593 if hsv_min_sat is None:
3594 hsv_min_sat = self.hsv_min_sat
3595 if hsv_min_val is None:
3596 hsv_min_val = self.hsv_min_val
3597
3598 # Expects a 2D intensity array scaled between -1 to 1...
3599 intensity = intensity[..., 0]
3600 intensity = 2 * intensity - 1
3601
3602 # Convert to rgb, then rgb to hsv
3603 hsv = rgb_to_hsv(rgb[:, :, 0:3])
3604 hue, sat, val = np.moveaxis(hsv, -1, 0)
3605
3606 # Modify hsv values (in place) to simulate illumination.
3607 # putmask(A, mask, B) <=> A[mask] = B[mask]
3608 np.putmask(sat, (np.abs(sat) > 1.e-10) & (intensity > 0),
3609 (1 - intensity) * sat + intensity * hsv_max_sat)

3611 (1 + intensity) * sat - intensity * hsv_min_sat)
3612 np.putmask(val, intensity > 0,
3613 (1 - intensity) * val + intensity * hsv_max_val)
3614 np.putmask(val, intensity < 0,
3615 (1 + intensity) * val - intensity * hsv_min_val)
3616 np.clip(hsv[:, :, 1:], 0, 1, out=hsv[:, :, 1:])
3617
3618 # Convert modified hsv back to rgb.
3619 return hsv_to_rgb(hsv)
