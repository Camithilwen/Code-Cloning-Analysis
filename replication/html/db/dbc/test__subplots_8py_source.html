12def check_shared(axs, x_shared, y_shared):
13 """
14 x_shared and y_shared are n x n boolean matrices; entry (i, j) indicates
15 whether the x (or y) axes of subplots i and j should be shared.
16 """
17 for (i1, ax1), (i2, ax2), (i3, (name, shared)) in itertools.product(
18 enumerate(axs),
19 enumerate(axs),
20 enumerate(zip("xy", [x_shared, y_shared]))):
21 if i2 <= i1:
22 continue
23 assert axs[0]._shared_axes[name].joined(ax1, ax2) == shared[i1, i2], \
24 "axes %i and %i incorrectly %ssharing %s axis" % (
25 i1, i2, "not " if shared[i1, i2] else "", name)

28def check_ticklabel_visible(axs, x_visible, y_visible):
29 """Check that the x and y ticklabel visibility is as specified."""
30 for i, (ax, vx, vy) in enumerate(zip(axs, x_visible, y_visible)):
31 for l in ax.get_xticklabels() + [ax.xaxis.offsetText]:
32 assert l.get_visible() == vx, \
33 f"Visibility of x axis #{i} is incorrectly {vx}"
34 for l in ax.get_yticklabels() + [ax.yaxis.offsetText]:
35 assert l.get_visible() == vy, \
36 f"Visibility of y axis #{i} is incorrectly {vy}"
37 # axis label "visibility" is toggled by label_outer by resetting the
38 # label to empty, but it can also be empty to start with.
39 if not vx:
40 assert ax.get_xlabel() == ""
41 if not vy:
42 assert ax.get_ylabel() == ""

45def check_tick1_visible(axs, x_visible, y_visible):
46 """
47 Check that the x and y tick visibility is as specified.
48
49 Note: This only checks the tick1line, i.e. bottom / left ticks.
50 """
51 for ax, visible, in zip(axs, x_visible):
52 for tick in ax.xaxis.get_major_ticks():
53 assert tick.tick1line.get_visible() == visible
54 for ax, y_visible, in zip(axs, y_visible):
55 for tick in ax.yaxis.get_major_ticks():
56 assert tick.tick1line.get_visible() == visible
