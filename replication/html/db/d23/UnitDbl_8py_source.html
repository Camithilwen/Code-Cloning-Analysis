35 def __init__(self, value, units):
36 """
37 Create a new UnitDbl object.
38
39 Units are internally converted to km, rad, and sec. The only
40 valid inputs for units are [m, km, mile, rad, deg, sec, min, hour].
41
42 The field UnitDbl.value will contain the converted value. Use
43 the convert() method to get a specific type of units back.
44
45 = ERROR CONDITIONS
46 - If the input units are not in the allowed list, an error is thrown.
47
48 = INPUT VARIABLES
49 - value The numeric value of the UnitDbl.
50 - units The string name of the units the value is in.
51 """
52 data = _api.check_getitem(self.allowed, units=units)
53 self._value = float(value * data[0])
