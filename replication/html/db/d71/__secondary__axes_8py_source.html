79 def set_location(self, location, transform=None):
80 """
81 Set the vertical or horizontal location of the axes in
82 parent-normalized coordinates.
83
84 Parameters
85 ----------
86 location : {'top', 'bottom', 'left', 'right'} or float
87 The position to put the secondary axis. Strings can be 'top' or
88 'bottom' for orientation='x' and 'right' or 'left' for
89 orientation='y'. A float indicates the relative position on the
90 parent Axes to put the new Axes, 0.0 being the bottom (or left)
91 and 1.0 being the top (or right).
92
93 transform : `.Transform`, optional
94 Transform for the location to use. Defaults to
95 the parent's ``transAxes``, so locations are normally relative to
96 the parent axes.
97
98 .. versionadded:: 3.9
99 """
100
101 _api.check_isinstance((transforms.Transform, None), transform=transform)
102
103 # This puts the rectangle into figure-relative coordinates.
104 if isinstance(location, str):
105 _api.check_in_list(self._locstrings, location=location)
106 self._pos = 1. if location in ('top', 'right') else 0.
107 elif isinstance(location, numbers.Real):
108 self._pos = location
109 else:
110 raise ValueError(
111 f"location must be {self._locstrings[0]!r}, "
112 f"{self._locstrings[1]!r}, or a float, not {location!r}")
113
114 self._loc = location
115
116 if self._orientation == 'x':
117 # An x-secondary axes is like an inset axes from x = 0 to x = 1 and
118 # from y = pos to y = pos + eps, in the parent's transAxes coords.
119 bounds = [0, self._pos, 1., 1e-10]
120
121 # If a transformation is provided, use its y component rather than
122 # the parent's transAxes. This can be used to place axes in the data
123 # coords, for instance.
124 if transform is not None:
125 transform = transforms.blended_transform_factory(
126 self._parent.transAxes, transform)
127 else: # 'y'
128 bounds = [self._pos, 0, 1e-10, 1]
129 if transform is not None:
130 transform = transforms.blended_transform_factory(
131 transform, self._parent.transAxes) # Use provided x axis
132
133 # If no transform is provided, use the parent's transAxes
134 if transform is None:
135 transform = self._parent.transAxes
136
137 # this locator lets the axes move in the parent axes coordinates.
138 # so it never needs to know where the parent is explicitly in
139 # figure coordinates.
140 # it gets called in ax.apply_aspect() (of all places)
141 self.set_axes_locator(_TransformedBoundsLocator(bounds, transform))

155 def set_functions(self, functions):
156 """
157 Set how the secondary axis converts limits from the parent Axes.
158
159 Parameters
160 ----------
161 functions : 2-tuple of func, or `Transform` with an inverse.
162 Transform between the parent axis values and the secondary axis
163 values.
164
165 If supplied as a 2-tuple of functions, the first function is
166 the forward transform function and the second is the inverse
167 transform.
168
169 If a transform is supplied, then the transform must have an
170 inverse.
171 """
172
173 if (isinstance(functions, tuple) and len(functions) == 2 and
174 callable(functions[0]) and callable(functions[1])):
175 # make an arbitrary convert from a two-tuple of functions
176 # forward and inverse.
177 self._functions = functions
178 elif isinstance(functions, Transform):
179 self._functions = (
180 functions.transform,
181 lambda x: functions.inverted().transform(x)
182 )
183 elif functions is None:
184 self._functions = (lambda x: x, lambda x: x)
185 else:
186 raise ValueError('functions argument of secondary Axes '
187 'must be a two-tuple of callable functions '
188 'with the first function being the transform '
189 'and the second being the inverse')
190 self._set_scale()

206 def _set_scale(self):
207 """
208 Check if parent has set its scale
209 """
210
211 if self._orientation == 'x':
212 pscale = self._parent.xaxis.get_scale()
213 set_scale = self.set_xscale
214 else: # 'y'
215 pscale = self._parent.yaxis.get_scale()
216 set_scale = self.set_yscale
217 if pscale == self._parentscale:
218 return
219
220 if self._ticks_set:
221 ticks = self._axis.get_ticklocs()
222
223 # need to invert the roles here for the ticks to line up.
224 set_scale('functionlog' if pscale == 'log' else 'function',
225 functions=self._functions[::-1])
226
227 # OK, set_scale sets the locators, but if we've called
228 # axsecond.set_ticks, we want to keep those.
229 if self._ticks_set:
230 self._axis.set_major_locator(mticker.FixedLocator(ticks))
231
232 # If the parent scale doesn't change, we can skip this next time.
233 self._parentscale = pscale

235 def _set_lims(self):
236 """
237 Set the limits based on parent limits and the convert method
238 between the parent and this secondary Axes.
239 """
240 if self._orientation == 'x':
241 lims = self._parent.get_xlim()
242 set_lim = self.set_xlim
243 else: # 'y'
244 lims = self._parent.get_ylim()
245 set_lim = self.set_ylim
246 order = lims[0] < lims[1]
247 lims = self._functions[0](np.array(lims))
248 neworder = lims[0] < lims[1]
249 if neworder != order:
250 # Flip because the transform will take care of the flipping.
251 lims = lims[::-1]
252 set_lim(lims)
